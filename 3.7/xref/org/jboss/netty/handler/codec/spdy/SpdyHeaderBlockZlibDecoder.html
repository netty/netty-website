16  package org.jboss.netty.handler.codec.spdy;
17  
18  import static org.jboss.netty.handler.codec.spdy.SpdyCodecUtil.*;
19  
20  import java.util.zip.DataFormatException;
21  import java.util.zip.Inflater;
22  
23  import org.jboss.netty.buffer.ChannelBuffer;
24  import org.jboss.netty.buffer.ChannelBuffers;
25  
26  class SpdyHeaderBlockZlibDecoder extends SpdyHeaderBlockRawDecoder {
27  
28      private final int version;
29      private final byte[] out = new byte[8192];
30      private final Inflater decompressor = new Inflater();
31  
32      private ChannelBuffer decompressed;
33  
34      public SpdyHeaderBlockZlibDecoder(int version, int maxHeaderSize) {
35          super(version, maxHeaderSize);
36          this.version = version;
37      }
38  
39      @Override
40      void decode(ChannelBuffer encoded, SpdyHeadersFrame frame) throws Exception {
41          setInput(encoded);
42  
43          int numBytes;
44          do {
45              numBytes = decompress(frame);
46          } while (!decompressed.readable() && numBytes > 0);
47      }
48  
49      private void setInput(ChannelBuffer compressed) {
50          byte[] in = new byte[compressed.readableBytes()];
51          compressed.readBytes(in);
52          decompressor.setInput(in);
53      }
54  
55      private int decompress(SpdyHeadersFrame frame) throws Exception {
56          if (decompressed == null) {
57              decompressed = ChannelBuffers.dynamicBuffer(8192);
58          }
59  
60          try {
61              int numBytes = decompressor.inflate(out);
62              if (numBytes == 0 && decompressor.needsDictionary()) {
63                  if (version < 3) {
64                      decompressor.setDictionary(SPDY2_DICT);
65                  } else {
66                      decompressor.setDictionary(SPDY_DICT);
67                  }
68                  numBytes = decompressor.inflate(out);
69              }
70              if (frame != null) {
71                  decompressed.writeBytes(out, 0, numBytes);
72                  super.decode(decompressed, frame);
73                  decompressed.discardReadBytes();
74              }
75              return numBytes;
76          } catch (DataFormatException e) {
77              throw new SpdyProtocolException(
78                      "Received invalid header block", e);
79          }
80      }
81  
82      @Override
83      void reset() {
84          decompressed = null;
85          super.reset();
86      }
87  
88      @Override
89      public void end() {
90          decompressed = null;
91          decompressor.end();
92          super.end();
93      }
94  }
