50      private enum State {
51          READ_COMMON_HEADER,
52          READ_CONTROL_FRAME,
53          READ_SETTINGS_FRAME,
54          READ_HEADER_BLOCK_FRAME,
55          READ_HEADER_BLOCK,
56          READ_DATA_FRAME,
57          DISCARD_FRAME,
58          FRAME_ERROR
59      }

76      protected SpdyFrameDecoder(
77              int version, int maxChunkSize, SpdyHeaderBlockDecoder headerBlockDecoder) {
78          super(false);
79          if (version < SPDY_MIN_VERSION || version > SPDY_MAX_VERSION) {
80              throw new IllegalArgumentException(
81                      "unsupported version: " + version);
82          }
83          if (maxChunkSize <= 0) {
84              throw new IllegalArgumentException(
85                      "maxChunkSize must be a positive integer: " + maxChunkSize);
86          }
87          spdyVersion = version;
88          this.maxChunkSize = maxChunkSize;
89          this.headerBlockDecoder = headerBlockDecoder;
90          state = State.READ_COMMON_HEADER;
91      }

93      @Override
94      protected Object decodeLast(
95              ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer)
96              throws Exception {
97          try {
98              return decode(ctx, channel, buffer);
99          } finally {
100             headerBlockDecoder.end();
101         }
102     }

104     @Override
105     protected Object decode(
106             ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer)
107             throws Exception {
108         switch(state) {
109         case READ_COMMON_HEADER:
110             state = readCommonHeader(buffer);
111             if (state == State.FRAME_ERROR) {
112                 if (version != spdyVersion) {
113                     fireProtocolException(ctx, "Unsupported version: " + version);
114                 } else {
115                     fireInvalidFrameException(ctx);
116                 }
117             }
118 
119             // FrameDecoders must consume data when producing frames
120             // All length 0 frames must be generated now
121             if (length == 0) {
122                 if (state == State.READ_DATA_FRAME) {
123                     SpdyDataFrame spdyDataFrame = new DefaultSpdyDataFrame(streamId);
124                     spdyDataFrame.setLast((flags & SPDY_DATA_FLAG_FIN) != 0);
125                     state = State.READ_COMMON_HEADER;
126                     return spdyDataFrame;
127                 }
128                 // There are no length 0 control frames
129                 state = State.READ_COMMON_HEADER;
130             }
131 
132             return null;
133 
134         case READ_CONTROL_FRAME:
135             try {
136                 Object frame = readControlFrame(buffer);
137                 if (frame != null) {
138                     state = State.READ_COMMON_HEADER;
139                 }
140                 return frame;
141             } catch (IllegalArgumentException e) {
142                 state = State.FRAME_ERROR;
143                 fireInvalidFrameException(ctx);
144             }
145             return null;
146 
147         case READ_SETTINGS_FRAME:
148             if (spdySettingsFrame == null) {
149                 // Validate frame length against number of entries
150                 if (buffer.readableBytes() < 4) {
151                     return null;
152                 }
153                 int numEntries = getUnsignedInt(buffer, buffer.readerIndex());
154                 buffer.skipBytes(4);
155                 length -= 4;
156 
157                 // Each ID/Value entry is 8 bytes
158                 if ((length & 0x07) != 0 || length >> 3 != numEntries) {
159                     state = State.FRAME_ERROR;
160                     fireInvalidFrameException(ctx);
161                     return null;
162                 }
163 
164                 spdySettingsFrame = new DefaultSpdySettingsFrame();
165 
166                 boolean clear = (flags & SPDY_SETTINGS_CLEAR) != 0;
167                 spdySettingsFrame.setClearPreviouslyPersistedSettings(clear);
168             }
169 
170             int readableEntries = Math.min(buffer.readableBytes() >> 3, length >> 3);
171             for (int i = 0; i < readableEntries; i ++) {
172                 int ID;
173                 byte ID_flags;
174                 if (version < 3) {
175                     // Chromium Issue 79156
176                     // SPDY setting ids are not written in network byte order
177                     // Read id assuming the architecture is little endian
178                     ID = buffer.readByte() & 0xFF |
179                         (buffer.readByte() & 0xFF) << 8 |
180                         (buffer.readByte() & 0xFF) << 16;
181                     ID_flags = buffer.readByte();
182                 } else {
183                     ID_flags = buffer.readByte();
184                     ID = getUnsignedMedium(buffer, buffer.readerIndex());
185                     buffer.skipBytes(3);
186                 }
187                 int value = getSignedInt(buffer, buffer.readerIndex());
188                 buffer.skipBytes(4);
189 
190                 // Check for invalid ID -- avoid IllegalArgumentException in setValue
191                 if (ID == 0) {
192                     state = State.FRAME_ERROR;
193                     spdySettingsFrame = null;
194                     fireInvalidFrameException(ctx);
195                     return null;
196                 }
197 
198                 if (!spdySettingsFrame.isSet(ID)) {
199                     boolean persistVal = (ID_flags & SPDY_SETTINGS_PERSIST_VALUE) != 0;
200                     boolean persisted  = (ID_flags & SPDY_SETTINGS_PERSISTED) != 0;
201                     spdySettingsFrame.setValue(ID, value, persistVal, persisted);
202                 }
203             }
204 
205             length -= 8 * readableEntries;
206             if (length == 0) {
207                 state = State.READ_COMMON_HEADER;
208                 Object frame = spdySettingsFrame;
209                 spdySettingsFrame = null;
210                 return frame;
211             }
212             return null;
213 
214         case READ_HEADER_BLOCK_FRAME:
215             try {
216                 spdyHeadersFrame = readHeaderBlockFrame(buffer);
217                 if (spdyHeadersFrame != null) {
218                     if (length == 0) {
219                         state = State.READ_COMMON_HEADER;
220                         Object frame = spdyHeadersFrame;
221                         spdyHeadersFrame = null;
222                         return frame;
223                     }
224                     state = State.READ_HEADER_BLOCK;
225                 }
226                 return null;
227             } catch (IllegalArgumentException e) {
228                 state = State.FRAME_ERROR;
229                 fireInvalidFrameException(ctx);
230                 return null;
231             }
232 
233         case READ_HEADER_BLOCK:
234             int compressedBytes = Math.min(buffer.readableBytes(), length);
235             ChannelBuffer compressed = buffer.slice(buffer.readerIndex(), compressedBytes);
236 
237             try {
238                 headerBlockDecoder.decode(compressed, spdyHeadersFrame);
239             } catch (Exception e) {
240                 state = State.FRAME_ERROR;
241                 spdyHeadersFrame = null;
242                 Channels.fireExceptionCaught(ctx, e);
243                 return null;
244             }
245 
246             int readBytes = compressedBytes - compressed.readableBytes();
247             buffer.skipBytes(readBytes);
248             length -= readBytes;
249 
250             if (spdyHeadersFrame != null &&
251                     (spdyHeadersFrame.isInvalid() || spdyHeadersFrame.isTruncated())) {
252                 Object frame = spdyHeadersFrame;
253                 spdyHeadersFrame = null;
254                 if (length == 0) {
255                     headerBlockDecoder.reset();
256                     state = State.READ_COMMON_HEADER;
257                 }
258                 return frame;
259             }
260 
261             if (length == 0) {
262                 Object frame = spdyHeadersFrame;
263                 spdyHeadersFrame = null;
264                 headerBlockDecoder.reset();
265                 state = State.READ_COMMON_HEADER;
266                 return frame;
267             }
268             return null;
269 
270         case READ_DATA_FRAME:
271             if (streamId == 0) {
272                 state = State.FRAME_ERROR;
273                 fireProtocolException(ctx, "Received invalid data frame");
274                 return null;
275             }
276 
277             // Generate data frames that do not exceed maxChunkSize
278             int dataLength = Math.min(maxChunkSize, length);
279 
280             // Wait until entire frame is readable
281             if (buffer.readableBytes() < dataLength) {
282                 return null;
283             }
284 
285             SpdyDataFrame spdyDataFrame = new DefaultSpdyDataFrame(streamId);
286             spdyDataFrame.setData(buffer.readBytes(dataLength));
287             length -= dataLength;
288 
289             if (length == 0) {
290                 spdyDataFrame.setLast((flags & SPDY_DATA_FLAG_FIN) != 0);
291                 state = State.READ_COMMON_HEADER;
292             }
293             return spdyDataFrame;
294 
295         case DISCARD_FRAME:
296             int numBytes = Math.min(buffer.readableBytes(), length);
297             buffer.skipBytes(numBytes);
298             length -= numBytes;
299             if (length == 0) {
300                 state = State.READ_COMMON_HEADER;
301             }
302             return null;
303 
304         case FRAME_ERROR:
305             buffer.skipBytes(buffer.readableBytes());
306             return null;
307 
308         default:
309             throw new Error("Shouldn't reach here.");
310         }
311     }

313     private State readCommonHeader(ChannelBuffer buffer) {
314         // Wait until entire header is readable
315         if (buffer.readableBytes() < SPDY_HEADER_SIZE) {
316             return State.READ_COMMON_HEADER;
317         }
318 
319         int frameOffset  = buffer.readerIndex();
320         int flagsOffset  = frameOffset + SPDY_HEADER_FLAGS_OFFSET;
321         int lengthOffset = frameOffset + SPDY_HEADER_LENGTH_OFFSET;
322         buffer.skipBytes(SPDY_HEADER_SIZE);
323 
324         // Read common header fields
325         boolean control = (buffer.getByte(frameOffset) & 0x80) != 0;
326         flags  = buffer.getByte(flagsOffset);
327         length = getUnsignedMedium(buffer, lengthOffset);
328 
329         if (control) {
330             // Decode control frame common header
331             version = getUnsignedShort(buffer, frameOffset) & 0x7FFF;
332 
333             int typeOffset = frameOffset + SPDY_HEADER_TYPE_OFFSET;
334             type = getUnsignedShort(buffer, typeOffset);
335 
336             streamId = 0; // Default to session Stream-ID
337         } else {
338             // Decode data frame common header
339             version = spdyVersion; // Default to expected version
340 
341             type = SPDY_DATA_FRAME;
342 
343             streamId = getUnsignedInt(buffer, frameOffset);
344         }
345         // Check version first then validity
346         if (version != spdyVersion || !isValidFrameHeader()) {
347             return State.FRAME_ERROR;
348         }
349 
350         // Make sure decoder will produce a frame or consume input
351         State nextState;
352         if (willGenerateFrame()) {
353             switch (type) {
354             case SPDY_DATA_FRAME:
355                 nextState = State.READ_DATA_FRAME;
356                 break;
357 
358             case SPDY_SYN_STREAM_FRAME:
359             case SPDY_SYN_REPLY_FRAME:
360             case SPDY_HEADERS_FRAME:
361                 nextState = State.READ_HEADER_BLOCK_FRAME;
362                 break;
363 
364             case SPDY_SETTINGS_FRAME:
365                 nextState = State.READ_SETTINGS_FRAME;
366                 break;
367 
368             default:
369                 nextState = State.READ_CONTROL_FRAME;
370             }
371         } else if (length != 0) {
372             nextState = State.DISCARD_FRAME;
373         } else {
374             nextState = State.READ_COMMON_HEADER;
375         }
376         return nextState;
377     }

379     private Object readControlFrame(ChannelBuffer buffer) {
380         int streamId;
381         int statusCode;
382         switch (type) {
383         case SPDY_RST_STREAM_FRAME:
384             if (buffer.readableBytes() < 8) {
385                 return null;
386             }
387 
388             streamId = getUnsignedInt(buffer, buffer.readerIndex());
389             statusCode = getSignedInt(buffer, buffer.readerIndex() + 4);
390             buffer.skipBytes(8);
391 
392             return new DefaultSpdyRstStreamFrame(streamId, statusCode);
393 
394         case SPDY_PING_FRAME:
395             if (buffer.readableBytes() < 4) {
396                 return null;
397             }
398 
399             int ID = getSignedInt(buffer, buffer.readerIndex());
400             buffer.skipBytes(4);
401 
402             return new DefaultSpdyPingFrame(ID);
403 
404         case SPDY_GOAWAY_FRAME:
405             int minLength = version < 3 ? 4 : 8;
406             if (buffer.readableBytes() < minLength) {
407                 return null;
408             }
409 
410             int lastGoodStreamID = getUnsignedInt(buffer, buffer.readerIndex());
411             buffer.skipBytes(4);
412 
413             if (version < 3) {
414                 return new DefaultSpdyGoAwayFrame(lastGoodStreamID);
415             }
416 
417             statusCode = getSignedInt(buffer, buffer.readerIndex());
418             buffer.skipBytes(4);
419 
420             return new DefaultSpdyGoAwayFrame(lastGoodStreamID, statusCode);
421 
422         case SPDY_WINDOW_UPDATE_FRAME:
423             if (buffer.readableBytes() < 8) {
424                 return null;
425             }
426 
427             streamId = getUnsignedInt(buffer, buffer.readerIndex());
428             int deltaWindowSize = getUnsignedInt(buffer, buffer.readerIndex() + 4);
429             buffer.skipBytes(8);
430 
431             return new DefaultSpdyWindowUpdateFrame(streamId, deltaWindowSize);
432 
433         default:
434             throw new Error("Shouldn't reach here.");
435         }
436     }

438     private SpdyHeadersFrame readHeaderBlockFrame(ChannelBuffer buffer) {
439         int minLength;
440         int streamId;
441         switch (type) {
442         case SPDY_SYN_STREAM_FRAME:
443             minLength = version < 3 ? 12 : 10;
444             if (buffer.readableBytes() < minLength) {
445                 return null;
446             }
447 
448             int offset = buffer.readerIndex();
449             streamId = getUnsignedInt(buffer, offset);
450             int associatedToStreamID = getUnsignedInt(buffer, offset + 4);
451             byte priority = (byte) (buffer.getByte(offset + 8) >> 5 & 0x07);
452             if (version < 3) {
453                 priority >>= 1;
454             }
455             buffer.skipBytes(10);
456             length -= 10;
457 
458             // SPDY/2 requires 16-bits of padding for empty header blocks
459             if (version < 3 && length == 2 && buffer.getShort(buffer.readerIndex()) == 0) {
460                 buffer.skipBytes(2);
461                 length = 0;
462             }
463 
464             SpdySynStreamFrame spdySynStreamFrame =
465                     new DefaultSpdySynStreamFrame(streamId, associatedToStreamID, priority);
466             spdySynStreamFrame.setLast((flags & SPDY_FLAG_FIN) != 0);
467             spdySynStreamFrame.setUnidirectional((flags & SPDY_FLAG_UNIDIRECTIONAL) != 0);
468 
469             return spdySynStreamFrame;
470 
471         case SPDY_SYN_REPLY_FRAME:
472             minLength = version < 3 ? 8 : 4;
473             if (buffer.readableBytes() < minLength) {
474                 return null;
475             }
476 
477             streamId = getUnsignedInt(buffer, buffer.readerIndex());
478             buffer.skipBytes(4);
479             length -= 4;
480 
481             // SPDY/2 has 16-bits of unused space
482             if (version < 3) {
483                 buffer.skipBytes(2);
484                 length -= 2;
485             }
486 
487             // SPDY/2 requires 16-bits of padding for empty header blocks
488             if (version < 3 && length == 2 && buffer.getShort(buffer.readerIndex()) == 0) {
489                 buffer.skipBytes(2);
490                 length = 0;
491             }
492 
493             SpdySynReplyFrame spdySynReplyFrame = new DefaultSpdySynReplyFrame(streamId);
494             spdySynReplyFrame.setLast((flags & SPDY_FLAG_FIN) != 0);
495 
496             return spdySynReplyFrame;
497 
498         case SPDY_HEADERS_FRAME:
499             if (buffer.readableBytes() < 4) {
500                 return null;
501             }
502 
503             // SPDY/2 allows length 4 frame when there are no name/value pairs
504             if (version < 3 && length > 4 && buffer.readableBytes() < 8) {
505                 return null;
506             }
507 
508             streamId = getUnsignedInt(buffer, buffer.readerIndex());
509             buffer.skipBytes(4);
510             length -= 4;
511 
512             // SPDY/2 has 16-bits of unused space
513             if (version < 3 && length != 0) {
514                 buffer.skipBytes(2);
515                 length -= 2;
516             }
517 
518             // SPDY/2 requires 16-bits of padding for empty header blocks
519             if (version < 3 && length == 2 && buffer.getShort(buffer.readerIndex()) == 0) {
520                 buffer.skipBytes(2);
521                 length = 0;
522             }
523 
524             SpdyHeadersFrame spdyHeadersFrame = new DefaultSpdyHeadersFrame(streamId);
525             spdyHeadersFrame.setLast((flags & SPDY_FLAG_FIN) != 0);
526 
527             return spdyHeadersFrame;
528 
529         default:
530             throw new Error("Shouldn't reach here.");
531         }
532     }

534     private boolean isValidFrameHeader() {
535         switch (type) {
536         case SPDY_DATA_FRAME:
537             return streamId != 0;
538 
539         case SPDY_SYN_STREAM_FRAME:
540             return version < 3 ? length >= 12 : length >= 10;
541 
542         case SPDY_SYN_REPLY_FRAME:
543             return version < 3 ? length >= 8 : length >= 4;
544 
545         case SPDY_RST_STREAM_FRAME:
546             return flags == 0 && length == 8;
547 
548         case SPDY_SETTINGS_FRAME:
549             return length >= 4;
550 
551         case SPDY_PING_FRAME:
552             return length == 4;
553 
554         case SPDY_GOAWAY_FRAME:
555             return version < 3 ? length == 4 : length == 8;
556 
557         case SPDY_HEADERS_FRAME:
558             if (version < 3) {
559                 return length == 4 || length >= 8;
560             } else {
561                 return length >= 4;
562             }
563 
564         case SPDY_WINDOW_UPDATE_FRAME:
565             return length == 8;
566 
567         default:
568             return true;
569         }
570     }

572     private boolean willGenerateFrame() {
573         switch (type) {
574         case SPDY_DATA_FRAME:
575         case SPDY_SYN_STREAM_FRAME:
576         case SPDY_SYN_REPLY_FRAME:
577         case SPDY_RST_STREAM_FRAME:
578         case SPDY_SETTINGS_FRAME:
579         case SPDY_PING_FRAME:
580         case SPDY_GOAWAY_FRAME:
581         case SPDY_HEADERS_FRAME:
582         case SPDY_WINDOW_UPDATE_FRAME:
583             return true;
584 
585         default:
586             return false;
587         }
588     }
