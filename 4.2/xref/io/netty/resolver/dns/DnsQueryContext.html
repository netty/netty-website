85      DnsQueryContext(Channel channel,
86                      InetSocketAddress nameServerAddr,
87                      DnsQueryContextManager queryContextManager,
88                      DnsQueryLifecycleObserver queryLifecycleObserver,
89                      int maxPayLoadSize,
90                      boolean recursionDesired,
91                      long queryTimeoutMillis,
92                      DnsQuestion question,
93                      DnsRecord[] additionals,
94                      Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise,
95                      Bootstrap socketBootstrap,
96                      boolean retryWithTcpOnTimeout) {
97          this.channel = checkNotNull(channel, "channel");
98          this.queryContextManager = checkNotNull(queryContextManager, "queryContextManager");
99          this.queryLifecycleObserver = checkNotNull(queryLifecycleObserver, "queryLifecycleObserver");
100         this.nameServerAddr = checkNotNull(nameServerAddr, "nameServerAddr");
101         this.question = checkNotNull(question, "question");
102         this.additionals = checkNotNull(additionals, "additionals");
103         this.promise = checkNotNull(promise, "promise");
104         this.recursionDesired = recursionDesired;
105         this.queryTimeoutMillis = queryTimeoutMillis;
106         this.socketBootstrap = socketBootstrap;
107         this.retryWithTcpOnTimeout = retryWithTcpOnTimeout;
108 
109         if (maxPayLoadSize > 0 &&
110                 // Only add the extra OPT record if there is not already one. This is required as only one is allowed
111                 // as per RFC:
112                 //  - https://datatracker.ietf.org/doc/html/rfc6891#section-6.1.1
113                 !hasOptRecord(additionals)) {
114             optResource = new AbstractDnsOptPseudoRrRecord(maxPayLoadSize, 0, 0) {
115                 // We may want to remove this in the future and let the user just specify the opt record in the query.
116             };
117         } else {
118             optResource = null;
119         }
120     }

122     private static boolean hasOptRecord(DnsRecord[] additionals) {
123         if (additionals != null && additionals.length > 0) {
124             for (DnsRecord additional: additionals) {
125                 if (additional.type() == DnsRecordType.OPT) {
126                     return true;
127                 }
128             }
129         }
130         return false;
131     }

167     /**
168      * Write the query and return the {@link ChannelFuture} that is completed once the write completes.
169      *
170      * @param flush                 {@code true} if {@link Channel#flush()} should be called as well.
171      */
172     final void writeQuery(boolean flush) {
173         assert id == Integer.MIN_VALUE : this.getClass().getSimpleName() +
174                 ".writeQuery(...) can only be executed once.";
175 
176         if ((id = queryContextManager.add(nameServerAddr, this)) == -1) {
177             // We did exhaust the id space, fail the query
178             IllegalStateException e = new IllegalStateException("query ID space exhausted: " + question());
179             finishFailure("failed to send a query via " + protocol(), e, false);
180             queryLifecycleObserver.queryWritten(nameServerAddr, channel.newFailedFuture(e));
181         }
182 
183         // Ensure we remove the id from the QueryContextManager once the query completes.
184         promise.addListener(new FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>>() {
185             @Override
186             public void operationComplete(Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
187                 // Cancel the timeout task.
188                 Future<?> timeoutFuture = DnsQueryContext.this.timeoutFuture;
189                 if (timeoutFuture != null) {
190                     DnsQueryContext.this.timeoutFuture = null;
191                     timeoutFuture.cancel(false);
192                 }
193 
194                 Throwable cause = future.cause();
195                 if (cause instanceof DnsNameResolverTimeoutException || cause instanceof CancellationException) {
196                     // This query was failed due a timeout or cancellation. Let's delay the removal of the id to reduce
197                     // the risk of reusing the same id again while the remote nameserver might send the response after
198                     // the timeout.
199                     channel.eventLoop().schedule(new Runnable() {
200                         @Override
201                         public void run() {
202                             removeFromContextManager(nameServerAddr);
203                         }
204                     }, ID_REUSE_ON_TIMEOUT_DELAY_MILLIS, TimeUnit.MILLISECONDS);
205                 } else {
206                     // Remove the id from the manager as soon as the query completes. This may be because of success,
207                     // failure or cancellation
208                     removeFromContextManager(nameServerAddr);
209                 }
210             }
211         });
212         final DnsQuestion question = question();
213         final DnsQuery query = newQuery(id, nameServerAddr);
214 
215         query.setRecursionDesired(recursionDesired);
216 
217         query.addRecord(DnsSection.QUESTION, question);
218 
219         for (DnsRecord record: additionals) {
220             query.addRecord(DnsSection.ADDITIONAL, record);
221         }
222 
223         if (optResource != null) {
224             query.addRecord(DnsSection.ADDITIONAL, optResource);
225         }
226 
227         if (logger.isDebugEnabled()) {
228             logger.debug("{} WRITE: {}, [{}: {}], {}",
229                     channel, protocol(), id, nameServerAddr, question);
230         }
231 
232         ChannelFuture f = sendQuery(query, flush);
233         queryLifecycleObserver.queryWritten(nameServerAddr, f);
234     }
235 
236     private void removeFromContextManager(InetSocketAddress nameServerAddr) {
237         DnsQueryContext self = queryContextManager.remove(nameServerAddr, id);
238 
239         assert self == this : "Removed DnsQueryContext is not the correct instance";
240     }

242     private ChannelFuture sendQuery(final DnsQuery query, final boolean flush) {
243         final ChannelPromise writePromise = channel.newPromise();
244         writeQuery(query, flush, writePromise);
245         return writePromise;
246     }
247 
248     private void writeQuery(final DnsQuery query,
249                             final boolean flush, ChannelPromise promise) {
250         final ChannelFuture writeFuture = flush ? channel.writeAndFlush(query, promise) :
251                 channel.write(query, promise);
252         if (writeFuture.isDone()) {
253             onQueryWriteCompletion(queryTimeoutMillis, writeFuture);
254         } else {
255             writeFuture.addListener(new ChannelFutureListener() {
256                 @Override
257                 public void operationComplete(ChannelFuture future) {
258                     onQueryWriteCompletion(queryTimeoutMillis, writeFuture);
259                 }
260             });
261         }
262     }
263 
264     private void onQueryWriteCompletion(final long queryTimeoutMillis,
265                                         ChannelFuture writeFuture) {
266         if (!writeFuture.isSuccess()) {
267             finishFailure("failed to send a query '" + id + "' via " + protocol(), writeFuture.cause(), false);
268             return;
269         }
270 
271         // Schedule a query timeout task if necessary.
272         if (queryTimeoutMillis > 0) {
273             timeoutFuture = channel.eventLoop().schedule(new Runnable() {
274                 @Override
275                 public void run() {
276                     if (promise.isDone()) {
277                         // Received a response before the query times out.
278                         return;
279                     }
280 
281                     finishFailure("query '" + id + "' via " + protocol() + " timed out after " +
282                             queryTimeoutMillis + " milliseconds", null, true);
283                 }
284             }, queryTimeoutMillis, TimeUnit.MILLISECONDS);
285         }
286     }

288     /**
289      * Notifies the original {@link Promise} that the response for the query was received.
290      * This method takes ownership of passed {@link AddressedEnvelope}.
291      */
292     void finishSuccess(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope, boolean truncated) {
293         // Check if the response was not truncated or if a fallback to TCP is possible.
294         if (!truncated || !retryWithTcp(envelope)) {
295             final DnsResponse res = envelope.content();
296             if (res.count(DnsSection.QUESTION) != 1) {
297                 logger.warn("{} Received a DNS response with invalid number of questions. Expected: 1, found: {}",
298                         channel, envelope);
299             } else if (!question().equals(res.recordAt(DnsSection.QUESTION))) {
300                 logger.warn("{} Received a mismatching DNS response. Expected: [{}], found: {}",
301                         channel, question(), envelope);
302             } else if (trySuccess(envelope)) {
303                 return; // Ownership transferred, don't release
304             }
305             envelope.release();
306         }
307     }
308 
309     @SuppressWarnings("unchecked")
310     private boolean trySuccess(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope) {
311         return promise.trySuccess((AddressedEnvelope<DnsResponse, InetSocketAddress>) envelope);
312     }

314     /**
315      * Notifies the original {@link Promise} that the query completes because of an failure.
316      */
317     final boolean finishFailure(String message, Throwable cause, boolean timeout) {
318         if (promise.isDone()) {
319             return false;
320         }
321         final DnsQuestion question = question();
322 
323         final StringBuilder buf = new StringBuilder(message.length() + 128);
324         buf.append('[')
325            .append(id)
326            .append(": ")
327            .append(nameServerAddr)
328            .append("] ")
329            .append(question)
330            .append(' ')
331            .append(message)
332            .append(" (no stack trace available)");
333 
334         final DnsNameResolverException e;
335         if (timeout) {
336             // This was caused by a timeout so use DnsNameResolverTimeoutException to allow the user to
337             // handle it special (like retry the query).
338             e = new DnsNameResolverTimeoutException(nameServerAddr, question, buf.toString());
339             if (retryWithTcpOnTimeout && retryWithTcp(e)) {
340                 // We did successfully retry with TCP.
341                 return false;
342             }
343         } else {
344             e = new DnsNameResolverException(nameServerAddr, question, buf.toString(), cause);
345         }
346         return promise.tryFailure(e);
347     }

349     /**
350      * Retry the original query with TCP if possible.
351      *
352      * @param originalResult    the result of the original {@link DnsQueryContext}.
353      * @return                  {@code true} if retry via TCP is supported and so the ownership of
354      *                          {@code originalResult} was transferred, {@code false} otherwise.
355      */
356     private boolean retryWithTcp(final Object originalResult) {
357         if (socketBootstrap == null) {
358             return false;
359         }
360 
361         socketBootstrap.connect(nameServerAddr).addListener(new ChannelFutureListener() {
362             @Override
363             public void operationComplete(ChannelFuture future) {
364                 if (!future.isSuccess()) {
365                     logger.debug("{} Unable to fallback to TCP [{}: {}]",
366                             future.channel(), id, nameServerAddr, future.cause());
367 
368                     // TCP fallback failed, just use the truncated response or error.
369                     finishOriginal(originalResult, future);
370                     return;
371                 }
372                 final Channel tcpCh = future.channel();
373                 Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise =
374                         tcpCh.eventLoop().newPromise();
375                 final TcpDnsQueryContext tcpCtx = new TcpDnsQueryContext(tcpCh,
376                         (InetSocketAddress) tcpCh.remoteAddress(), queryContextManager, queryLifecycleObserver, 0,
377                         recursionDesired, queryTimeoutMillis, question(), additionals, promise);
378                 tcpCh.pipeline().addLast(TCP_ENCODER);
379                 tcpCh.pipeline().addLast(new TcpDnsResponseDecoder());
380                 tcpCh.pipeline().addLast(new ChannelInboundHandlerAdapter() {
381                     @Override
382                     public void channelRead(ChannelHandlerContext ctx, Object msg) {
383                         Channel tcpCh = ctx.channel();
384                         DnsResponse response = (DnsResponse) msg;
385                         int queryId = response.id();
386 
387                         if (logger.isDebugEnabled()) {
388                             logger.debug("{} RECEIVED: TCP [{}: {}], {}", tcpCh, queryId,
389                                     tcpCh.remoteAddress(), response);
390                         }
391 
392                         DnsQueryContext foundCtx = queryContextManager.get(nameServerAddr, queryId);
393                         if (foundCtx != null && foundCtx.isDone()) {
394                             logger.debug("{} Received a DNS response for a query that was timed out or cancelled " +
395                                     ": TCP [{}: {}]", tcpCh, queryId, nameServerAddr);
396                             response.release();
397                         } else if (foundCtx == tcpCtx) {
398                             tcpCtx.finishSuccess(new AddressedEnvelopeAdapter(
399                                     (InetSocketAddress) ctx.channel().remoteAddress(),
400                                     (InetSocketAddress) ctx.channel().localAddress(),
401                                     response), false);
402                         } else {
403                             response.release();
404                             tcpCtx.finishFailure("Received TCP DNS response with unexpected ID", null, false);
405                             if (logger.isDebugEnabled()) {
406                                 logger.debug("{} Received a DNS response with an unexpected ID: TCP [{}: {}]",
407                                         tcpCh, queryId, tcpCh.remoteAddress());
408                             }
409                         }
410                     }
411 
412                     @Override
413                     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
414                         if (tcpCtx.finishFailure(
415                                 "TCP fallback error", cause, false) && logger.isDebugEnabled()) {
416                             logger.debug("{} Error during processing response: TCP [{}: {}]",
417                                     ctx.channel(), id,
418                                     ctx.channel().remoteAddress(), cause);
419                         }
420                     }
421                 });
422 
423                 promise.addListener(
424                         new FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>>() {
425                             @Override
426                             public void operationComplete(
427                                     Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
428                                 if (future.isSuccess()) {
429                                     finishSuccess(future.getNow(), false);
430                                     // Release the original result.
431                                     ReferenceCountUtil.release(originalResult);
432                                 } else {
433                                     // TCP fallback failed, just use the truncated response or error.
434                                     finishOriginal(originalResult, future);
435                                 }
436                                 tcpCh.close();
437                             }
438                         });
439                 tcpCtx.writeQuery(true);
440             }
441         });
442         return true;
443     }
444 
445     @SuppressWarnings("unchecked")
446     private void finishOriginal(Object originalResult, Future<?> future) {
447         if (originalResult instanceof Throwable) {
448             Throwable error = (Throwable) originalResult;
449             ThrowableUtil.addSuppressed(error, future.cause());
450             promise.tryFailure(error);
451         } else {
452             finishSuccess((AddressedEnvelope<? extends DnsResponse, InetSocketAddress>) originalResult, false);
453         }
454     }

456     private static final class AddressedEnvelopeAdapter implements AddressedEnvelope<DnsResponse, InetSocketAddress> {
457         private final InetSocketAddress sender;
458         private final InetSocketAddress recipient;
459         private final DnsResponse response;
460 
461         AddressedEnvelopeAdapter(InetSocketAddress sender, InetSocketAddress recipient, DnsResponse response) {
462             this.sender = sender;
463             this.recipient = recipient;
464             this.response = response;
465         }
466 
467         @Override
468         public DnsResponse content() {
469             return response;
470         }
471 
472         @Override
473         public InetSocketAddress sender() {
474             return sender;
475         }
476 
477         @Override
478         public InetSocketAddress recipient() {
479             return recipient;
480         }
481 
482         @Override
483         public AddressedEnvelope<DnsResponse, InetSocketAddress> retain() {
484             response.retain();
485             return this;
486         }
487 
488         @Override
489         public AddressedEnvelope<DnsResponse, InetSocketAddress> retain(int increment) {
490             response.retain(increment);
491             return this;
492         }
493 
494         @Override
495         public AddressedEnvelope<DnsResponse, InetSocketAddress> touch() {
496             response.touch();
497             return this;
498         }
499 
500         @Override
501         public AddressedEnvelope<DnsResponse, InetSocketAddress> touch(Object hint) {
502             response.touch(hint);
503             return this;
504         }
505 
506         @Override
507         public int refCnt() {
508             return response.refCnt();
509         }
510 
511         @Override
512         public boolean release() {
513             return response.release();
514         }
515 
516         @Override
517         public boolean release(int decrement) {
518             return response.release(decrement);
519         }
520 
521         @Override
522         public boolean equals(Object obj) {
523             if (this == obj) {
524                 return true;
525             }
526 
527             if (!(obj instanceof AddressedEnvelope)) {
528                 return false;
529             }
530 
531             @SuppressWarnings("unchecked")
532             final AddressedEnvelope<?, SocketAddress> that = (AddressedEnvelope<?, SocketAddress>) obj;
533             if (sender() == null) {
534                 if (that.sender() != null) {
535                     return false;
536                 }
537             } else if (!sender().equals(that.sender())) {
538                 return false;
539             }
540 
541             if (recipient() == null) {
542                 if (that.recipient() != null) {
543                     return false;
544                 }
545             } else if (!recipient().equals(that.recipient())) {
546                 return false;
547             }
548 
549             return response.equals(obj);
550         }
551 
552         @Override
553         public int hashCode() {
554             int hashCode = response.hashCode();
555             if (sender() != null) {
556                 hashCode = hashCode * 31 + sender().hashCode();
557             }
558             if (recipient() != null) {
559                 hashCode = hashCode * 31 + recipient().hashCode();
560             }
561             return hashCode;
562         }
