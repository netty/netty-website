15  package io.netty.handler.codec.http2;
16  
17  public interface Http2FrameSizePolicy {
18      /**
19       * Sets the maximum allowed frame size. Attempts to write frames longer than this maximum will fail.
20       * <p>
21       * This value is used to represent
22       * <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_MAX_FRAME_SIZE</a>. This method should
23       * only be called by Netty (not users) as a result of a receiving a {@code SETTINGS} frame.
24       */
25      void maxFrameSize(int max) throws Http2Exception;
26  
27      /**
28       * Gets the maximum allowed frame size.
29       * <p>
30       * This value is used to represent
31       * <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_MAX_FRAME_SIZE</a>. The initial value
32       * defined by the RFC is unlimited but enforcing a lower limit is generally permitted.
33       * {@link Http2CodecUtil#DEFAULT_MAX_FRAME_SIZE} can be used as a more conservative default.
34       */
35      int maxFrameSize();
36  }
