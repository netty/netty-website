16  package io.netty.handler.codec.http2;
17  
18  /**
19   * An ack for a previously received {@link Http2SettingsFrame}.
20   * <p>
21   * The <a href="https://tools.ietf.org/html/rfc7540#section-6.5">HTTP/2 protocol</a> enforces that ACKs are applied in
22   * order, so this ACK will apply to the earliest received and not yet ACKed {@link Http2SettingsFrame} frame.
23   */
24  public interface Http2SettingsAckFrame extends Http2Frame {
25      Http2SettingsAckFrame INSTANCE = new DefaultHttp2SettingsAckFrame();
26  
27      @Override
28      String name();
29  }
