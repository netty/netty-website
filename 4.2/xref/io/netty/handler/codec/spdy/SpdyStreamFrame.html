16  package io.netty.handler.codec.spdy;
17  
18  /**
19   * A SPDY Protocol Frame that is associated with an individual SPDY Stream
20   */
21  public interface SpdyStreamFrame extends SpdyFrame {
22  
23      /**
24       * Returns the Stream-ID of this frame.
25       */
26      int streamId();
27  
28      /**
29       * Sets the Stream-ID of this frame.  The Stream-ID must be positive.
30       */
31      SpdyStreamFrame setStreamId(int streamID);
32  
33      /**
34       * Returns {@code true} if this frame is the last frame to be transmitted
35       * on the stream.
36       */
37      boolean isLast();
38  
39      /**
40       * Sets if this frame is the last frame to be transmitted on the stream.
41       */
42      SpdyStreamFrame setLast(boolean last);
43  }
