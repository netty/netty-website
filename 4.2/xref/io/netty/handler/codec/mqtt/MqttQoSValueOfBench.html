73      int[] types;
74      long next;
75      long mask;
76  
77      @Setup
78      public void initDataSet() {
79          types = DATASET;
80          next = 0;
81          mask = types.length - 1;
82          if (Integer.bitCount(types.length) != 1) {
83              throw new AssertionError("The data set should contains power of 2 items");
84          }
85      }
86  
87      @Benchmark
88      public MqttQoS getViaArray() {
89          long next = this.next;
90          int nextIndex = (int) (next & mask);
91          MqttQoS mqttQoS = arrayValueOf(types[nextIndex]);
92          this.next = next + 1;
93          return mqttQoS;
94      }
95  
96      @Benchmark
97      public MqttQoS getViaSwitch() {
98          long next = this.next;
99          int nextIndex = (int) (next & mask);
100         MqttQoS mqttQoS = MqttQoS.valueOf(types[nextIndex]);
101         this.next = next + 1;
102         return mqttQoS;
103     }
104 
105     private static final MqttQoS[] VALUES;
106 
107     static {
108         VALUES = new MqttQoS[129];
109         for (MqttQoS value : MqttQoS.values()) {
110             VALUES[value.value()] = value;
111         }
112     }
113 
114     public static MqttQoS arrayValueOf(int value) {
115         MqttQoS mqttQoS = null;
116         try {
117             mqttQoS = VALUES[value];
118             return mqttQoS;
119         } catch (ArrayIndexOutOfBoundsException ignored) {
120             // nop
121         }
122         if (mqttQoS == null) {
123             throw new IllegalArgumentException("invalid QoS: " + value);
124         }
125         return mqttQoS;
126     }
