16  package io.netty.handler.codec.http3;
17  
18  import static io.netty.util.internal.ObjectUtil.checkNotNull;
19  
20  class QpackHeaderField {
21  
22      /**
23       * <a href="https://www.rfc-editor.org/rfc/rfc9204.html#name-dynamic-table-size">
24       *     Section 3.2.1 Dynamic Table Size</a>.
25       * The size of an entry is the sum of its name's length in bytes, its
26       * value's length in bytes, and 32.
27       */
28      static final int ENTRY_OVERHEAD = 32;
29  
30      static long sizeOf(CharSequence name, CharSequence value) {
31          return name.length() + value.length() + ENTRY_OVERHEAD;
32      }
33  
34      final CharSequence name;
35      final CharSequence value;
36  
37      // This constructor can only be used if name and value are ISO-8859-1 encoded.
38      QpackHeaderField(CharSequence name, CharSequence value) {
39          this.name = checkNotNull(name, "name");
40          this.value = checkNotNull(value, "value");
41      }
42  
43      long size() {
44          return sizeOf(name, value);
45      }
46  
47      @Override
48      public String toString() {
49          return name + ": " + value;
50      }
51  }
