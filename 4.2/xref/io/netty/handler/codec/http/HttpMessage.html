16  package io.netty.handler.codec.http;
17  
18  
19  /**
20   * An interface that defines an HTTP message, providing common properties for
21   * {@link HttpRequest} and {@link HttpResponse}.
22   *
23   * @see HttpResponse
24   * @see HttpRequest
25   * @see HttpHeaders
26   */
27  public interface HttpMessage extends HttpObject {
28  
29      /**
30       * @deprecated Use {@link #protocolVersion()} instead.
31       */
32      @Deprecated
33      HttpVersion getProtocolVersion();
34  
35      /**
36       * Returns the protocol version of this {@link HttpMessage}
37       */
38      HttpVersion protocolVersion();
39  
40      /**
41       * Set the protocol version of this {@link HttpMessage}
42       */
43      HttpMessage setProtocolVersion(HttpVersion version);
44  
45      /**
46       * Returns the headers of this message.
47       */
48      HttpHeaders headers();
49  }
