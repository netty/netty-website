17  package io.netty.handler.codec.socksx.v4;
18  
19  import io.netty.buffer.ByteBuf;
20  import io.netty.buffer.ByteBufUtil;
21  import io.netty.channel.ChannelHandler.Sharable;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.handler.codec.MessageToByteEncoder;
24  import io.netty.util.NetUtil;
25  
26  /**
27   * Encodes a {@link Socks4CommandRequest} into a {@link ByteBuf}.
28   */
29  @Sharable
30  public final class Socks4ClientEncoder extends MessageToByteEncoder<Socks4CommandRequest> {
31  
32      /**
33       * The singleton instance of {@link Socks4ClientEncoder}
34       */
35      public static final Socks4ClientEncoder INSTANCE = new Socks4ClientEncoder();
36  
37      private static final byte[] IPv4_DOMAIN_MARKER = {0x00, 0x00, 0x00, 0x01};
38  
39      private Socks4ClientEncoder() {
40          super(Socks4CommandRequest.class);
41      }
42  
43      @Override
44      protected void encode(ChannelHandlerContext ctx, Socks4CommandRequest msg, ByteBuf out) throws Exception {
45          out.writeByte(msg.version().byteValue());
46          out.writeByte(msg.type().byteValue());
47          ByteBufUtil.writeShortBE(out, msg.dstPort());
48          if (NetUtil.isValidIpV4Address(msg.dstAddr())) {
49              out.writeBytes(NetUtil.createByteArrayFromIpAddressString(msg.dstAddr()));
50              ByteBufUtil.writeAscii(out, msg.userId());
51              out.writeByte(0);
52          } else {
53              out.writeBytes(IPv4_DOMAIN_MARKER);
54              ByteBufUtil.writeAscii(out, msg.userId());
55              out.writeByte(0);
56              ByteBufUtil.writeAscii(out, msg.dstAddr());
57              out.writeByte(0);
58          }
59      }
60  }
