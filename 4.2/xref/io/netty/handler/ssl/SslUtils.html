198     private static boolean isTLSv13SupportedByJDK0(Provider provider) {
199         try {
200             return arrayContains(newInitContext(provider)
201                     .getSupportedSSLParameters().getProtocols(), SslProtocols.TLS_v1_3);
202         } catch (Throwable cause) {
203             logger.debug("Unable to detect if JDK SSLEngine with provider {} supports TLSv1.3, assuming no",
204                     provider, cause);
205             return false;
206         }
207     }

230     private static SSLContext newInitContext(Provider provider)
231             throws NoSuchAlgorithmException, KeyManagementException {
232         final SSLContext context;
233         if (provider == null) {
234             context = SSLContext.getInstance("TLS");
235         } else {
236             context = SSLContext.getInstance("TLS", provider);
237         }
238         context.init(null, new TrustManager[0], null);
239         return context;
240     }

304     /**
305      * Return how much bytes can be read out of the encrypted data. Be aware that this method will not increase
306      * the readerIndex of the given {@link ByteBuf}.
307      *
308      * @param   buffer
309      *                  The {@link ByteBuf} to read from.
310      * @return length
311      *                  The length of the encrypted packet that is included in the buffer or
312      *                  {@link #SslUtils#NOT_ENOUGH_DATA} if not enough data is present in the
313      *                  {@link ByteBuf}. This will return {@link SslUtils#NOT_ENCRYPTED} if
314      *                  the given {@link ByteBuf} is not encrypted at all.
315      */
316     static int getEncryptedPacketLength(ByteBuf buffer, int offset) {
317         int packetLength = 0;
318 
319         // SSLv3 or TLS - Check ContentType
320         boolean tls;
321         switch (buffer.getUnsignedByte(offset)) {
322             case SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC:
323             case SSL_CONTENT_TYPE_ALERT:
324             case SSL_CONTENT_TYPE_HANDSHAKE:
325             case SSL_CONTENT_TYPE_APPLICATION_DATA:
326             case SSL_CONTENT_TYPE_EXTENSION_HEARTBEAT:
327                 tls = true;
328                 break;
329             default:
330                 // SSLv2 or bad data
331                 tls = false;
332         }
333 
334         if (tls) {
335             // SSLv3 or TLS or GMSSLv1.0 or GMSSLv1.1 - Check ProtocolVersion
336             int majorVersion = buffer.getUnsignedByte(offset + 1);
337             int version = buffer.getShort(offset + 1);
338             if (majorVersion == 3 || version == GMSSL_PROTOCOL_VERSION) {
339                 // SSLv3 or TLS or GMSSLv1.0 or GMSSLv1.1
340                 packetLength = unsignedShortBE(buffer, offset + 3) + SSL_RECORD_HEADER_LENGTH;
341                 if (packetLength <= SSL_RECORD_HEADER_LENGTH) {
342                     // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
343                     tls = false;
344                 }
345             } else if (version == DTLS_1_0 || version == DTLS_1_2 || version == DTLS_1_3) {
346                 if (buffer.readableBytes() < offset + DTLS_RECORD_HEADER_LENGTH) {
347                     return NOT_ENOUGH_DATA;
348                 }
349                 // length is the last 2 bytes in the 13 byte header.
350                 packetLength = unsignedShortBE(buffer, offset + DTLS_RECORD_HEADER_LENGTH - 2) +
351                         DTLS_RECORD_HEADER_LENGTH;
352             } else {
353                 // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
354                 tls = false;
355             }
356         }
357 
358         if (!tls) {
359             // SSLv2 or bad data - Check the version
360             int headerLength = (buffer.getUnsignedByte(offset) & 0x80) != 0 ? 2 : 3;
361             int majorVersion = buffer.getUnsignedByte(offset + headerLength + 1);
362             if (majorVersion == 2 || majorVersion == 3) {
363                 // SSLv2
364                 packetLength = headerLength == 2 ?
365                         (shortBE(buffer, offset) & 0x7FFF) + 2 : (shortBE(buffer, offset) & 0x3FFF) + 3;
366                 if (packetLength <= headerLength) {
367                     return NOT_ENOUGH_DATA;
368                 }
369             } else {
370                 return NOT_ENCRYPTED;
371             }
372         }
373         return packetLength;
374     }

376     // Reads a big-endian unsigned short integer from the buffer
377     @SuppressWarnings("deprecation")
378     private static int unsignedShortBE(ByteBuf buffer, int offset) {
379         int value = buffer.getUnsignedShort(offset);
380         if (buffer.order() == ByteOrder.LITTLE_ENDIAN) {
381             value = Integer.reverseBytes(value) >>> Short.SIZE;
382         }

411     static int getEncryptedPacketLength(ByteBuffer[] buffers, int offset) {
412         ByteBuffer buffer = buffers[offset];
413 
414         // Check if everything we need is in one ByteBuffer. If so we can make use of the fast-path.
415         if (buffer.remaining() >= SSL_RECORD_HEADER_LENGTH) {
416             return getEncryptedPacketLength(buffer);
417         }
418 
419         // We need to copy 5 bytes into a temporary buffer so we can parse out the packet length easily.
420         ByteBuffer tmp = ByteBuffer.allocate(5);
421 
422         do {
423             buffer = buffers[offset++].duplicate();
424             if (buffer.remaining() > tmp.remaining()) {
425                 buffer.limit(buffer.position() + tmp.remaining());
426             }
427             tmp.put(buffer);
428         } while (tmp.hasRemaining());
429 
430         // Done, flip the buffer so we can read from it.
431         tmp.flip();
432         return getEncryptedPacketLength(tmp);
433     }

435     private static int getEncryptedPacketLength(ByteBuffer buffer) {
436         int packetLength = 0;
437         int pos = buffer.position();
438         // SSLv3 or TLS - Check ContentType
439         boolean tls;
440         switch (unsignedByte(buffer.get(pos))) {
441             case SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC:
442             case SSL_CONTENT_TYPE_ALERT:
443             case SSL_CONTENT_TYPE_HANDSHAKE:
444             case SSL_CONTENT_TYPE_APPLICATION_DATA:
445             case SSL_CONTENT_TYPE_EXTENSION_HEARTBEAT:
446                 tls = true;
447                 break;
448             default:
449                 // SSLv2 or bad data
450                 tls = false;
451         }
452 
453         if (tls) {
454             // SSLv3 or TLS or GMSSLv1.0 or GMSSLv1.1 - Check ProtocolVersion
455             int majorVersion = unsignedByte(buffer.get(pos + 1));
456             if (majorVersion == 3 || buffer.getShort(pos + 1) == GMSSL_PROTOCOL_VERSION) {
457                 // SSLv3 or TLS or GMSSLv1.0 or GMSSLv1.1
458                 packetLength = unsignedShortBE(buffer, pos + 3) + SSL_RECORD_HEADER_LENGTH;
459                 if (packetLength <= SSL_RECORD_HEADER_LENGTH) {
460                     // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
461                     tls = false;
462                 }
463             } else {
464                 // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
465                 tls = false;
466             }
467         }
468 
469         if (!tls) {
470             // SSLv2 or bad data - Check the version
471             int headerLength = (unsignedByte(buffer.get(pos)) & 0x80) != 0 ? 2 : 3;
472             int majorVersion = unsignedByte(buffer.get(pos + headerLength + 1));
473             if (majorVersion == 2 || majorVersion == 3) {
474                 // SSLv2
475                 packetLength = headerLength == 2 ?
476                         (shortBE(buffer, pos) & 0x7FFF) + 2 : (shortBE(buffer, pos) & 0x3FFF) + 3;
477                 if (packetLength <= headerLength) {
478                     return NOT_ENOUGH_DATA;
479                 }
480             } else {
481                 return NOT_ENCRYPTED;
482             }
483         }
484         return packetLength;
485     }

487     static void handleHandshakeFailure(ChannelHandlerContext ctx, Throwable cause, boolean notify) {
488         // We have may haven written some parts of data before an exception was thrown so ensure we always flush.
489         // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
490         ctx.flush();
491         if (notify) {
492             ctx.fireUserEventTriggered(new SslHandshakeCompletionEvent(cause));
493         }
494         ctx.close();
495     }

497     /**
498      * Fills the {@link ByteBuf} with zero bytes.
499      */
500     static void zeroout(ByteBuf buffer) {
501         if (!buffer.isReadOnly()) {
502             buffer.setZero(0, buffer.capacity());
503         }
504     }

507      * Fills the {@link ByteBuf} with zero bytes and releases it.
508      */
509     static void zerooutAndRelease(ByteBuf buffer) {
510         zeroout(buffer);
511         buffer.release();
512     }

514     /**
515      * Same as {@link Base64#encode(ByteBuf, boolean)} but allows the use of a custom {@link ByteBufAllocator}.
516      *
517      * @see Base64#encode(ByteBuf, boolean)
518      */
519     static ByteBuf toBase64(ByteBufAllocator allocator, ByteBuf src) {
520         ByteBuf dst = Base64.encode(src, src.readerIndex(),
521                 src.readableBytes(), true, Base64Dialect.STANDARD, allocator);
522         src.readerIndex(src.writerIndex());
523         return dst;
524     }

526     /**
527      * Validate that the given hostname can be used in SNI extension.
528      */
529     static boolean isValidHostNameForSNI(String hostname) {
530         // See  https://datatracker.ietf.org/doc/html/rfc6066#section-3
531         return hostname != null &&
532                // SNI HostName has to be a FQDN according to TLS SNI Extension spec (see [1]),
533                // which means that is has to have at least a host name and a domain part.
534                hostname.indexOf('.') > 0 &&
535                !hostname.endsWith(".") && !hostname.startsWith("/") &&
536                !NetUtil.isValidIpV4Address(hostname) &&
537                !NetUtil.isValidIpV6Address(hostname);
538     }

540     /**
541      * Returns {@code true} if the given cipher (in openssl format) is for TLSv1.3, {@code false} otherwise.
542      */
543     static boolean isTLSv13Cipher(String cipher) {
544         // See https://tools.ietf.org/html/rfc8446#appendix-B.4
545         return TLSV13_CIPHERS.contains(cipher);
546     }
