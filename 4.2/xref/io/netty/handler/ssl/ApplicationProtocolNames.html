17  package io.netty.handler.ssl;
18  
19  /**
20   * Provides a set of protocol names used in ALPN and NPN.
21   *
22   * @see <a href="https://tools.ietf.org/html/rfc7540#section-11.1">RFC7540 (HTTP/2)</a>
23   * @see <a href="https://tools.ietf.org/html/rfc7301#section-6">RFC7301 (TLS ALPN Extension)</a>
24   * @see <a href="https://tools.ietf.org/html/draft-agl-tls-nextprotoneg-04#section-7">TLS NPN Extension Draft</a>
25   */
26  public final class ApplicationProtocolNames {
27  
28      /**
29       * {@code "h2"}: HTTP version 2
30       */
31      public static final String HTTP_2 = "h2";
32  
33      /**
34       * {@code "http/1.1"}: HTTP version 1.1
35       */
36      public static final String HTTP_1_1 = "http/1.1";
37  
38      /**
39       * {@code "spdy/3.1"}: SPDY version 3.1
40       */
41      public static final String SPDY_3_1 = "spdy/3.1";
42  
43      /**
44       * {@code "spdy/3"}: SPDY version 3
45       */
46      public static final String SPDY_3 = "spdy/3";
47  
48      /**
49       * {@code "spdy/2"}: SPDY version 2
50       */
51      public static final String SPDY_2 = "spdy/2";
52  
53      /**
54       * {@code "spdy/1"}: SPDY version 1
55       */
56      public static final String SPDY_1 = "spdy/1";
57  
58      private ApplicationProtocolNames() { }
59  }
