16  package io.netty.handler.pcap;
17  
18  import io.netty.buffer.ByteBuf;
19  
20  final class UDPPacket {
21  
22      private static final short UDP_HEADER_SIZE = 8;
23  
24      private UDPPacket() {
25          // Prevent outside initialization
26      }
27  
28      /**
29       * Write UDP Packet
30       *
31       * @param byteBuf ByteBuf where Packet data will be set
32       * @param payload Payload of this Packet
33       * @param srcPort Source Port
34       * @param dstPort Destination Port
35       */
36      static void writePacket(ByteBuf byteBuf, ByteBuf payload, int srcPort, int dstPort) {
37          byteBuf.writeShort(srcPort); // Source Port
38          byteBuf.writeShort(dstPort); // Destination Port
39          byteBuf.writeShort(UDP_HEADER_SIZE + payload.readableBytes()); // UDP Header Length + Payload Length
40          byteBuf.writeShort(0x0001);  // Checksum
41          byteBuf.writeBytes(payload); //  Payload of Data
42      }
43  }
