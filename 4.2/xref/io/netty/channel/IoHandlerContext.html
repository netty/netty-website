16  package io.netty.channel;
17  
18  import io.netty.util.concurrent.ThreadAwareExecutor;
19  
20  /**
21   * The context for an {@link IoHandler} that is run by an {@link ThreadAwareExecutor}.
22   * All methods  <strong>MUST</strong> be executed on the {@link ThreadAwareExecutor} thread
23   * (which means {@link ThreadAwareExecutor#isExecutorThread(Thread)} (Thread)} must return {@code true}).
24   */
25  public interface IoHandlerContext {
26      /**
27       * Returns {@code true} if blocking for IO is allowed or if we should try to do a non-blocking request for IO to be
28       * ready.
29       *
30       * @return {@code true} if allowed, {@code false} otherwise.
31       */
32      boolean canBlock();
33  
34      /**
35       * Returns the amount of time left until the scheduled task with the closest deadline should run.
36       *
37       * @param currentTimeNanos  the current nanos.
38       * @return                  nanos
39       */
40      long delayNanos(long currentTimeNanos);
41  
42      /**
43       * Returns the absolute point in time at which the next
44       * closest scheduled task should run or {@code -1} if nothing is scheduled to run.
45       *
46       * @return deadline.
47       */
48      long deadlineNanos();
49  }
