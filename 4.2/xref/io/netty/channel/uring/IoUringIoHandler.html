43  /**
44   * {@link IoHandler} which is implemented in terms of the Linux-specific {@code io_uring} API.
45   */
46  public final class IoUringIoHandler implements IoHandler {
47      private static final InternalLogger logger = InternalLoggerFactory.getInstance(IoUringIoHandler.class);
48      private static final short RING_CLOSE = 1;
49  
50      private final RingBuffer ringBuffer;
51      private final IntObjectMap<DefaultIoUringIoRegistration> registrations;
52      // The maximum number of bytes for an InetAddress / Inet6Address
53      private final byte[] inet4AddressArray = new byte[SockaddrIn.IPV4_ADDRESS_LENGTH];
54      private final byte[] inet6AddressArray = new byte[SockaddrIn.IPV6_ADDRESS_LENGTH];
55  
56      private final AtomicBoolean eventfdAsyncNotify = new AtomicBoolean();
57      private final FileDescriptor eventfd;
58      private final long eventfdReadBuf;
59  
60      private long eventfdReadSubmitted;
61      private boolean eventFdClosing;
62      private volatile boolean shuttingDown;
63      private boolean closeCompleted;
64      private int nextRegistrationId = Integer.MIN_VALUE;
65  
66      // these two ids are used internally any so can't be used by nextRegistrationId().
67      private static final int EVENTFD_ID = Integer.MAX_VALUE;
68      private static final int RINGFD_ID = EVENTFD_ID - 1;
69  
70      IoUringIoHandler(RingBuffer ringBuffer) {
71          // Ensure that we load all native bits as otherwise it may fail when try to use native methods in IovArray
72          IoUring.ensureAvailability();
73          this.ringBuffer = requireNonNull(ringBuffer, "ringBuffer");
74          registrations = new IntObjectHashMap<>();
75          eventfd = Native.newBlockingEventFd();
76          eventfdReadBuf = PlatformDependent.allocateMemory(8);
77      }
78  
79      @Override
80      public int run(IoExecutionContext context) {
81          SubmissionQueue submissionQueue = ringBuffer.ioUringSubmissionQueue();
82          CompletionQueue completionQueue = ringBuffer.ioUringCompletionQueue();
83          if (!completionQueue.hasCompletions() && context.canBlock()) {
84              if (eventfdReadSubmitted == 0) {
85                  submitEventFdRead();
86              }
87              if (context.deadlineNanos() != -1) {
88                  submitTimeout(context);
89              }
90              submissionQueue.submitAndWait();
91          } else {
92              submissionQueue.submit();
93          }
94          return completionQueue.process(this::handle);
95      }
96  
97      private void handle(int res, int flags, int id, byte op, short data) {
98          if (id == EVENTFD_ID) {
99              handleEventFdRead();
100             return;
101         }
102         if (id == RINGFD_ID) {
103             if (op == Native.IORING_OP_NOP && data == RING_CLOSE) {
104                 completeRingClose();
105             }
106             return;
107         }
108         DefaultIoUringIoRegistration registration = registrations.get(id);
109         if (registration == null) {
110             logger.debug("ignoring {} completion for unknown registration (id={}, res={})",
111                     Native.opToStr(op), id, res);
112             return;
113         }
114         registration.handle(res, flags, op, data);
115     }
116 
117     private void handleEventFdRead() {
118         eventfdReadSubmitted = 0;
119         if (!eventFdClosing) {
120             eventfdAsyncNotify.set(false);
121             submitEventFdRead();
122         }
123     }
124 
125     private void submitEventFdRead() {
126         SubmissionQueue submissionQueue = ringBuffer.ioUringSubmissionQueue();
127         eventfdReadSubmitted = submissionQueue.addEventFdRead(
128                 eventfd.intValue(), eventfdReadBuf, 0, 8, EVENTFD_ID, (short) 0);
129     }
130 
131     private void submitTimeout(IoExecutionContext context) {
132         long delayNanos = context.delayNanos(System.nanoTime());
133         ringBuffer.ioUringSubmissionQueue().addTimeout(
134                 ringBuffer.fd(), delayNanos, RINGFD_ID, (short) 0);
135     }

137     @Override
138     public void prepareToDestroy() {
139         shuttingDown = true;
140         CompletionQueue completionQueue = ringBuffer.ioUringCompletionQueue();
141         SubmissionQueue submissionQueue = ringBuffer.ioUringSubmissionQueue();
142 
143         List<DefaultIoUringIoRegistration> copy = new ArrayList<>(registrations.values());
144         // Ensure all previously submitted IOs get to complete before closing all fds.
145         submissionQueue.addNop(ringBuffer.fd(), Native.IOSQE_IO_DRAIN, RINGFD_ID, (short) 0);
146 
147         for (DefaultIoUringIoRegistration registration: copy) {
148             registration.close();
149             if (submissionQueue.count() > 0) {
150                 submissionQueue.submit();
151             }
152             if (completionQueue.hasCompletions()) {
153                 completionQueue.process(this::handle);
154             }
155         }
156     }
157 
158     @Override
159     public void destroy() {
160         SubmissionQueue submissionQueue = ringBuffer.ioUringSubmissionQueue();
161         CompletionQueue completionQueue = ringBuffer.ioUringCompletionQueue();
162         drainEventFd();
163         if (submissionQueue.remaining() < 2) {
164             // We need to submit 2 linked operations. Since they are linked, we cannot allow a submit-call to
165             // separate them. We don't have enough room (< 2) in the queue, so we submit now to make more room.
166             submissionQueue.submit();
167         }
168         // Try to drain all the IO from the queue first...
169         submissionQueue.addNop(ringBuffer.fd(), Native.IOSQE_IO_DRAIN, RINGFD_ID, (short) 0);
170         // ... but only wait for 200 milliseconds on this
171         submissionQueue.addLinkTimeout(ringBuffer.fd(), TimeUnit.MILLISECONDS.toNanos(200), RINGFD_ID, (short) 0);
172         submissionQueue.submitAndWait();
173         completionQueue.process(this::handle);
174         completeRingClose();
175     }
176 
177     // We need to prevent the race condition where a wakeup event is submitted to a file descriptor that has
178     // already been freed (and potentially reallocated by the OS). Because submitted events is gated on the
179     // `eventfdAsyncNotify` flag we can close the gate but may need to read any outstanding events that have
180     // (or will) be written.
181     private void drainEventFd() {
182         CompletionQueue completionQueue = ringBuffer.ioUringCompletionQueue();
183         SubmissionQueue submissionQueue = ringBuffer.ioUringSubmissionQueue();
184         assert !eventFdClosing;
185         eventFdClosing = true;
186         boolean eventPending = eventfdAsyncNotify.getAndSet(true);
187         if (eventPending) {
188             // There is an event that has been or will be written by another thread, so we must wait for the event.
189             // Make sure we're actually listening for writes to the event fd.
190             while (eventfdReadSubmitted == 0) {
191                 submitEventFdRead();
192                 submissionQueue.submit();
193             }
194             // Drain the eventfd of the pending wakup.
195             class DrainFdEventCallback implements CompletionCallback {
196                 boolean eventFdDrained;
197 
198                 @Override
199                 public void handle(int res, int flags, int id, byte op, short data) {
200                     if (id == EVENTFD_ID) {
201                         eventFdDrained = true;
202                     }
203                     IoUringIoHandler.this.handle(res, flags, id, op, data);
204                 }
205             }
206             final DrainFdEventCallback handler = new DrainFdEventCallback();
207             completionQueue.process(handler);
208             while (!handler.eventFdDrained) {
209                 submissionQueue.submitAndWait();
210                 completionQueue.process(handler);
211             }
212         }
213         // We've consumed any pending eventfd read and `eventfdAsyncNotify` should never
214         // transition back to false, thus we should never have any more events written.
215         // So, if we have a read event pending, we can cancel it.
216         if (eventfdReadSubmitted != 0) {
217             submissionQueue.addCancel(eventfd.intValue(), eventfdReadSubmitted, EVENTFD_ID);
218             eventfdReadSubmitted = 0;
219             submissionQueue.submit();
220         }
221     }
222 
223     private void completeRingClose() {
224         if (closeCompleted) {
225             // already done.
226             return;
227         }
228         closeCompleted = true;
229         ringBuffer.close();
230         try {
231             eventfd.close();
232         } catch (IOException e) {
233             logger.warn("Failed to close eventfd", e);
234         }
235         PlatformDependent.freeMemory(eventfdReadBuf);
236     }

238     @Override
239     public IoRegistration register(IoEventLoop eventLoop, IoHandle handle) throws Exception {
240         IoUringIoHandle ioHandle = cast(handle);
241         if (shuttingDown) {
242             throw new RejectedExecutionException("IoEventLoop is shutting down");
243         }
244         DefaultIoUringIoRegistration registration = new DefaultIoUringIoRegistration(eventLoop, ioHandle);
245         for (;;) {
246             int id = nextRegistrationId();
247             DefaultIoUringIoRegistration old = registrations.put(id, registration);
248             if (old != null) {
249                 assert old.handle != registration.handle;
250                 registrations.put(id, old);
251             } else {
252                 registration.setId(id);
253                 break;
254             }
255         }
256 
257         ringBuffer.ioUringSubmissionQueue().incrementHandledFds();
258         return registration;
259     }
260 
261     private int nextRegistrationId() {
262         int id;
263         do {
264             id = nextRegistrationId++;
265         } while (id == RINGFD_ID || id == EVENTFD_ID);
266         return id;
267     }

269     private final class DefaultIoUringIoRegistration extends AtomicBoolean implements IoUringIoRegistration {
270         private final IoEventLoop eventLoop;
271         private final IoUringIoEvent event = new IoUringIoEvent(0, 0, (byte) 0, (short) 0);
272         final IoUringIoHandle handle;
273 
274         private boolean removeLater;
275         private int outstandingCompletions;
276         private int id;
277 
278         DefaultIoUringIoRegistration(IoEventLoop eventLoop, IoUringIoHandle handle) {
279             this.eventLoop = eventLoop;
280             this.handle = handle;
281         }
282 
283         void setId(int id) {
284             this.id = id;
285         }
286 
287         @Override
288         public long submit(IoOps ops) {
289             IoUringIoOps ioOps = (IoUringIoOps) ops;
290             long udata = UserData.encode(id, ioOps.opcode(), ioOps.data());
291             if (!isValid()) {
292                 return udata;
293             }
294             if (eventLoop.inEventLoop()) {
295                 submit0(ioOps, udata);
296             } else {
297                 eventLoop.execute(() -> submit0(ioOps, udata));
298             }
299             return udata;
300         }
301 
302         private void submit0(IoUringIoOps ioOps, long udata) {
303             ringBuffer.ioUringSubmissionQueue().enqueueSqe(ioOps.opcode(), ioOps.flags(), ioOps.ioPrio(),
304                     ioOps.rwFlags(), ioOps.fd(), ioOps.bufferAddress(), ioOps.length(), ioOps.offset(), udata);
305             outstandingCompletions++;
306         }
307 
308         @Override
309         public IoUringIoHandler ioHandler() {
310             return IoUringIoHandler.this;
311         }
312 
313         @Override
314         public boolean isValid() {
315             return !get();
316         }
317 
318         @Override
319         public void cancel() {
320             if (getAndSet(true)) {
321                 return;
322             }
323             if (eventLoop.inEventLoop()) {
324                 tryRemove();
325             } else {
326                 eventLoop.execute(this::tryRemove);
327             }
328         }
329 
330         private void tryRemove() {
331             if (outstandingCompletions > 0) {
332                 // We have some completions outstanding, we will remove the id <-> registration mapping
333                 // once these are done.
334                 removeLater = true;
335                 return;
336             }
337             remove();
338         }
339 
340         private void remove() {
341             DefaultIoUringIoRegistration old = registrations.remove(id);
342             assert old == this;
343             ringBuffer.ioUringSubmissionQueue().decrementHandledFds();
344         }
345 
346         void close() {
347             assert eventLoop.inEventLoop();
348             try {
349                 cancel();
350             } catch (Exception e) {
351                 logger.debug("Exception during canceling " + this, e);
352             }
353             try {
354                 handle.close();
355             } catch (Exception e) {
356                 logger.debug("Exception during closing " + handle, e);
357             }
358         }
359 
360         void handle(int res, int flags, byte op, short data) {
361             event.update(res, flags, op, data);
362             handle.handle(this, event);
363             if (--outstandingCompletions == 0 && removeLater) {
364                 // No more outstanding completions, remove the fd <-> registration mapping now.
365                 removeLater = false;
366                 remove();
367             }
368         }
369     }
370 
371     private static IoUringIoHandle cast(IoHandle handle) {
372         if (handle instanceof IoUringIoHandle) {
373             return (IoUringIoHandle) handle;
374         }
375         throw new IllegalArgumentException("IoHandle of type " + StringUtil.simpleClassName(handle) + " not supported");
376     }
377 
378     @Override
379     public void wakeup(IoEventLoop eventLoop) {
380         if (!eventLoop.inEventLoop() && !eventfdAsyncNotify.getAndSet(true)) {
381             // write to the eventfd which will then trigger an eventfd read completion.
382             Native.eventFdWrite(eventfd.intValue(), 1L);
383         }
384     }

387     public boolean isCompatible(Class<? extends IoHandle> handleType) {
388         return IoUringIoHandle.class.isAssignableFrom(handleType);
389     }

405     /**

410     public static IoHandlerFactory newFactory() {
411         IoUring.ensureAvailability();
412         return () -> new IoUringIoHandler(Native.createRingBuffer());

415     /**
416      * Create a new {@link IoHandlerFactory} that can be used to create {@link IoUringIoHandler}s.
417      * Each {@link IoUringIoHandler} will use a ring of size {@code ringSize}.
418      *
419      * @param  ringSize     the size of the ring.
420      * @return              factory
421      */
422     public static IoHandlerFactory newFactory(int ringSize) {
423         IoUring.ensureAvailability();
424         ObjectUtil.checkPositive(ringSize, "ringSize");
425         return () -> new IoUringIoHandler(Native.createRingBuffer());
