16  package io.netty.channel.sctp;
17  
18  import com.sun.nio.sctp.SctpStandardSocketOptions.InitMaxStreams;
19  import io.netty.channel.ChannelOption;
20  
21  import java.net.SocketAddress;
22  
23  /**
24   * Option for configuring the SCTP transport
25   */
26  public final class SctpChannelOption<T> extends ChannelOption<T> {
27  
28      public static final ChannelOption<Boolean> SCTP_DISABLE_FRAGMENTS =
29              valueOf(SctpChannelOption.class, "SCTP_DISABLE_FRAGMENTS");
30      public static final ChannelOption<Boolean> SCTP_EXPLICIT_COMPLETE =
31              valueOf(SctpChannelOption.class, "SCTP_EXPLICIT_COMPLETE");
32      public static final ChannelOption<Integer> SCTP_FRAGMENT_INTERLEAVE =
33              valueOf(SctpChannelOption.class, "SCTP_FRAGMENT_INTERLEAVE");
34      public static final ChannelOption<InitMaxStreams> SCTP_INIT_MAXSTREAMS =
35              valueOf(SctpChannelOption.class, "SCTP_INIT_MAXSTREAMS");
36  
37      public static final ChannelOption<Boolean> SCTP_NODELAY =
38              valueOf(SctpChannelOption.class, "SCTP_NODELAY");
39      public static final ChannelOption<SocketAddress> SCTP_PRIMARY_ADDR =
40              valueOf(SctpChannelOption.class, "SCTP_PRIMARY_ADDR");
41      public static final ChannelOption<SocketAddress> SCTP_SET_PEER_PRIMARY_ADDR =
42              valueOf(SctpChannelOption.class, "SCTP_SET_PEER_PRIMARY_ADDR");
43  
44      @SuppressWarnings({ "unused", "deprecation" })
45      private SctpChannelOption() {
46          super(null);
47      }
48  }
