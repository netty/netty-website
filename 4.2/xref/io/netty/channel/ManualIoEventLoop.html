37  /**
38   * {@link IoEventLoop} implementation that is owned by the user and so needs to be driven by the user manually with the
39   * given {@link Thread}. This means that the user is responsible to call either {@link #runNow()} or {@link #run(long)}
40   * to execute IO and tasks that were submitted to this {@link IoEventLoop}.
41   * <p>
42   * This is for <strong>advanced use-cases only</strong>, where the user wants to own the {@link Thread} that drives the
43   * {@link IoEventLoop} to also do other work. Care must be taken that the {@link #runNow() or
44   * {@link #waitAndRun()}} methods are called in a timely fashion.
45   */
46  public final class ManualIoEventLoop extends AbstractScheduledEventExecutor implements IoEventLoop {
47      private static final int ST_STARTED = 4;
48      private static final int ST_SHUTTING_DOWN = 5;
49      private static final int ST_SHUTDOWN = 6;
50      private static final int ST_TERMINATED = 7;
51  
52      private final AtomicInteger state = new AtomicInteger();
53      private final Promise<?> terminationFuture = new DefaultPromise<Void>(GlobalEventExecutor.INSTANCE);
54      private final Queue<Runnable> taskQueue = PlatformDependent.newMpscQueue();
55      private final IoHandlerContext nonBlockingContext = new IoHandlerContext() {
56          @Override
57          public boolean canBlock() {
58              assert inEventLoop();
59              return false;
60          }
61  
62          @Override
63          public long delayNanos(long currentTimeNanos) {
64              assert inEventLoop();
65              return 0;
66          }
67  
68          @Override
69          public long deadlineNanos() {
70              assert inEventLoop();
71              return -1;
72          }
73      };

84      /**
85       * Create a new {@link IoEventLoop} that is owned by the user and so needs to be driven by the user with the given
86       * {@link Thread}. This means that the user is responsible to call either {@link #runNow()} or
87       * {@link #run(long)} to execute IO or tasks that were submitted to this {@link IoEventLoop}.
88       *
89       * @param owningThread      the {@link Thread} that executes the IO and tasks for this {@link IoEventLoop}. The
90       *                          user will use this {@link Thread} to call {@link #runNow()} or {@link #run(long)} to
91       *                          make progress.
92       * @param factory           the {@link IoHandlerFactory} that will be used to create the {@link IoHandler} that is
93       *                          used by this {@link IoEventLoop}.
94       */
95      public ManualIoEventLoop(Thread owningThread, IoHandlerFactory factory) {
96          this.owningThread = Objects.requireNonNull(owningThread, "owningThread");
97          this.handler = factory.newHandler(this);
98      }

100     private int runAllTasks() {
101         assert inEventLoop();
102         int numRun = 0;
103         boolean fetchedAll;
104         do {
105             fetchedAll = fetchFromScheduledTaskQueue(taskQueue);
106             for (;;) {
107                 Runnable task = taskQueue.poll();
108                 if (task == null) {
109                     break;
110                 }
111                 safeExecute(task);
112                 numRun++;
113             }
114         } while (!fetchedAll); // keep on processing until we fetched all scheduled tasks.
115         if (numRun > 0) {
116             lastExecutionTime = getCurrentTimeNanos();
117         }
118         return numRun;
119     }
120 
121     private int run(IoHandlerContext context) {
122         if (!initialized) {
123             initialized = true;
124             handler.initialize();
125         }
126         if (isShuttingDown()) {
127             if (terminationFuture.isDone()) {
128                 // Already completely terminated
129                 return 0;
130             }
131             // Run all tasks before prepare to destroy.
132             int run = runAllTasks();
133             handler.prepareToDestroy();
134             if (confirmShutdown()) {
135                 // Destroy the handler now and run all remaining tasks.
136                 try {
137                     handler.destroy();
138                     for (;;) {
139                         int r = runAllTasks();
140                         run += r;
141                         if (r == 0) {
142                             break;
143                         }
144                     }
145                 } finally {
146                     state.set(ST_TERMINATED);
147                     terminationFuture.setSuccess(null);
148                 }
149             }
150             return run;
151         }
152         int run = handler.run(context);
153         // Now run all tasks.
154         return run + runAllTasks();
155     }

157     /**
158      * Run all ready IO and tasks for this {@link IoEventLoop}.
159      * This methods will <strong>NOT</strong> block and wait for IO / tasks to be ready, it will just
160      * return directly if there is nothing to do.
161      * <p>
162      * <strong>Must be called from the owning {@link Thread} that was passed as an parameter on construction.</strong>
163      *
164      * @return the number of IO and tasks executed.
165      */
166     public int runNow() {
167         checkCurrentThread();
168         return run(nonBlockingContext);
169     }
170 
171     /**
172      * Run all ready IO and tasks for this {@link IoEventLoop}.
173      * This methods will block and wait for IO / tasks to be ready if there is nothing to process atm for the given
174      * {@code waitNanos}.
175      * <p>
176      * <strong>Must be called from the owning {@link Thread} that was passed as an parameter on construction.</strong>
177      *
178      * @param waitNanos the maximum amount of nanoseconds to wait before returning.
179      * @return          the number of IO and tasks executed.
180      */
181     public int run(long waitNanos) {
182         checkCurrentThread();
183         blockingContext.maxBlockingNanos = waitNanos;
184         return run(blockingContext);
185     }
186 
187     private void checkCurrentThread() {
188         if (!inEventLoop(Thread.currentThread())) {
189             throw new IllegalStateException();
190         }
191     }
192 
193     /**
194      * Force a wakeup and so the {@link #run(long)} method will unblock and return even if there was nothing to do.
195      */
196     public void wakeup() {
197         if (isShuttingDown()) {
198             return;
199         }
200         handler.wakeup();
201     }

227     @Override
228     public Future<IoRegistration> register(final IoHandle handle) {
229         Promise<IoRegistration> promise = newPromise();
230         if (inEventLoop()) {
231             registerForIo0(handle, promise);
232         } else {
233             execute(() -> registerForIo0(handle, promise));
234         }
235 
236         return promise;
237     }

239     private void registerForIo0(final IoHandle handle, Promise<IoRegistration> promise) {
240         assert inEventLoop();
241         final IoRegistration registration;
242         try {
243             registration = handler.register(handle);
244         } catch (Exception e) {
245             promise.setFailure(e);
246             return;
247         }
248         promise.setSuccess(registration);
249     }

275     private void shutdown0(long quietPeriod, long timeout, int shutdownState) {
276         boolean inEventLoop = inEventLoop();
277         boolean wakeup;
278         int oldState;
279         for (;;) {
280             if (isShuttingDown()) {
281                 return;
282             }
283             int newState;
284             wakeup = true;
285             oldState = state.get();
286             if (inEventLoop) {
287                 newState = shutdownState;
288             } else if (oldState == ST_STARTED) {
289                 newState = shutdownState;
290             } else {
291                 newState = oldState;
292                 wakeup = false;
293             }
294 
295             if (state.compareAndSet(oldState, newState)) {
296                 break;
297             }
298         }
299         if (quietPeriod != -1) {
300             gracefulShutdownQuietPeriod = quietPeriod;
301         }
302         if (timeout != -1) {
303             gracefulShutdownTimeout = timeout;
304         }
305 
306         if (wakeup) {
307             handler.wakeup();
308         }
309     }

311     @Override
312     public Future<?> shutdownGracefully(long quietPeriod, long timeout, TimeUnit unit) {
313         ObjectUtil.checkPositiveOrZero(quietPeriod, "quietPeriod");
314         if (timeout < quietPeriod) {
315             throw new IllegalArgumentException(
316                     "timeout: " + timeout + " (expected >= quietPeriod (" + quietPeriod + "))");
317         }
318         ObjectUtil.checkNotNull(unit, "unit");
319 
320         shutdown0(unit.toNanos(quietPeriod), unit.toNanos(timeout), ST_SHUTTING_DOWN);
321         return terminationFuture();
322     }

355     @Override
356     public void execute(Runnable command) {
357         Objects.requireNonNull(command, "command");
358         boolean inEventLoop = inEventLoop();
359         if (inEventLoop) {
360             if (isShutdown()) {
361                 throw new RejectedExecutionException("event executor terminated");
362             }
363         }
364         taskQueue.add(command);
365         if (!inEventLoop) {
366             if (isShutdown()) {
367                 boolean reject = false;
368                 try {
369                     if (taskQueue.remove(command)) {
370                         reject = true;
371                     }
372                 } catch (UnsupportedOperationException e) {
373                     // The task queue does not support removal so the best thing we can do is to just move on and
374                     // hope we will be able to pick-up the task before its completely terminated.
375                     // In worst case we will log on termination.
376                 }
377                 if (reject) {
378                     throw new RejectedExecutionException("event executor terminated");
379                 }
380             }
381             handler.wakeup();
382         }
383     }

385     private boolean hasTasks() {
386         return !taskQueue.isEmpty();
387     }

389     private boolean confirmShutdown() {
390         if (!isShuttingDown()) {
391             return false;
392         }
393 
394         if (!inEventLoop()) {
395             throw new IllegalStateException("must be invoked from an event loop");
396         }
397 
398         cancelScheduledTasks();
399 
400         if (gracefulShutdownStartTime == 0) {
401             gracefulShutdownStartTime = getCurrentTimeNanos();
402         }
403 
404         if (runAllTasks() > 0) {
405             if (isShutdown()) {
406                 // Executor shut down - no new tasks anymore.
407                 return true;
408             }
409 
410             // There were tasks in the queue. Wait a little bit more until no tasks are queued for the quiet period or
411             // terminate if the quiet period is 0.
412             // See https://github.com/netty/netty/issues/4241
413             if (gracefulShutdownQuietPeriod == 0) {
414                 return true;
415             }
416             return false;
417         }
418 
419         final long nanoTime = getCurrentTimeNanos();
420 
421         if (isShutdown() || nanoTime - gracefulShutdownStartTime > gracefulShutdownTimeout) {
422             return true;
423         }
424 
425         if (nanoTime - lastExecutionTime <= gracefulShutdownQuietPeriod) {
426             try {
427                 Thread.sleep(100);
428             } catch (InterruptedException e) {
429                 // Ignore
430             }
431 
432             return false;
433         }
434 
435         // No tasks were added for last quiet period - hopefully safe to shut down.
436         // (Hopefully because we really cannot make a guarantee that there will be no execute() calls by a user.)
437         return true;
438     }

471     private void throwIfInEventLoop(String method) {
472         if (inEventLoop()) {
473             throw new RejectedExecutionException(
474                     "Calling " + method + " from within the EventLoop is not allowed as it would deadlock");
475         }

478     private final class BlockingIoHandlerContext implements IoHandlerContext {
479         long maxBlockingNanos = Long.MAX_VALUE;
480 
481         @Override
482         public boolean canBlock() {
483             assert inEventLoop();
484             return !hasTasks() && !hasScheduledTasks();
485         }
486 
487         @Override
488         public long delayNanos(long currentTimeNanos) {
489             assert inEventLoop();
490             return ManualIoEventLoop.this.delayNanos(currentTimeNanos, maxBlockingNanos);
491         }
492 
493         @Override
494         public long deadlineNanos() {
495             assert inEventLoop();
496             long next = nextScheduledTaskDeadlineNanos();
497             if (next == -1) {
498                 return maxBlockingNanos;
499             }
500             return Math.min(next, maxBlockingNanos);
501         }
