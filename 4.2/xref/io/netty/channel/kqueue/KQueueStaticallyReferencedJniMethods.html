16  package io.netty.channel.kqueue;
17  
18  /**
19   * This class is necessary to break the following cyclic dependency:
20   * <ol>
21   * <li>JNI_OnLoad</li>
22   * <li>JNI Calls FindClass because RegisterNatives (used to register JNI methods) requires a class</li>
23   * <li>FindClass loads the class, but static members variables of that class attempt to call a JNI method which has not
24   * yet been registered.</li>
25   * <li>java.lang.UnsatisfiedLinkError is thrown because native method has not yet been registered.</li>
26   * </ol>
27   * Static members which call JNI methods must not be declared in this class!
28   */
29  final class KQueueStaticallyReferencedJniMethods {
30      private KQueueStaticallyReferencedJniMethods() { }
31  
32      static native short evAdd();
33      static native short evEnable();
34      static native short evDisable();
35      static native short evDelete();
36      static native short evClear();
37      static native short evEOF();
38      static native short evError();
39  
40      // data/hint fflags for EVFILT_SOCK, shared with userspace.
41      static native short noteReadClosed();
42      static native short noteConnReset();
43      static native short noteDisconnected();
44  
45      static native short evfiltRead();
46      static native short evfiltWrite();
47      static native short evfiltUser();
48      static native short evfiltSock();
49  
50      // Flags for connectx(2)
51      static native int connectResumeOnReadWrite();
52      static native int connectDataIdempotent();
53  
54      // Sysctl values.
55      static native int fastOpenClient();
56      static native int fastOpenServer();
57  }
