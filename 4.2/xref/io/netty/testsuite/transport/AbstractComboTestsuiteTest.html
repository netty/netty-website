16  package io.netty.testsuite.transport;
17  
18  import io.netty.bootstrap.AbstractBootstrap;
19  import io.netty.buffer.ByteBufAllocator;
20  import io.netty.testsuite.util.TestUtils;
21  import io.netty.util.internal.StringUtil;
22  import io.netty.util.internal.logging.InternalLogger;
23  import io.netty.util.internal.logging.InternalLoggerFactory;
24  import org.junit.jupiter.api.TestInfo;
25  
26  import java.util.List;
27  
28  public abstract class AbstractComboTestsuiteTest<SB extends AbstractBootstrap<?, ?>,
29          CB extends AbstractBootstrap<?, ?>> {
30      protected final InternalLogger logger = InternalLoggerFactory.getInstance(getClass());
31      protected volatile CB cb;
32      protected volatile SB sb;
33  
34      protected abstract List<TestsuitePermutation.BootstrapComboFactory<SB, CB>> newFactories();
35  
36      protected List<ByteBufAllocator> newAllocators() {
37          return TestsuitePermutation.allocator();
38      }
39  
40      protected void run(TestInfo testInfo, Runner<SB, CB> runner) throws Throwable {
41          List<TestsuitePermutation.BootstrapComboFactory<SB, CB>> combos = newFactories();
42          String methodName = TestUtils.testMethodName(testInfo);
43          for (ByteBufAllocator allocator: newAllocators()) {
44              int i = 0;
45              for (TestsuitePermutation.BootstrapComboFactory<SB, CB> e: combos) {
46                  sb = e.newServerInstance();
47                  cb = e.newClientInstance();
48                  configure(sb, cb, allocator);
49                  logger.info(String.format(
50                          "Running: %s %d of %d (%s + %s) with %s",
51                          methodName, ++ i, combos.size(), sb, cb, StringUtil.simpleClassName(allocator)));
52                  runner.run(sb, cb);
53              }
54          }
55      }
56  
57      protected abstract void configure(SB sb, CB cb, ByteBufAllocator allocator);
58  
59      public interface Runner<SB extends AbstractBootstrap<?, ?>, CB extends AbstractBootstrap<?, ?>> {
60          void run(SB sb, CB cb) throws Throwable;
61      }
62  }
