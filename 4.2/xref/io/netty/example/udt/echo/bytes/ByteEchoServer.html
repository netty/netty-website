16  package io.netty.example.udt.echo.bytes;
17  
18  import io.netty.bootstrap.ServerBootstrap;
19  import io.netty.channel.ChannelFuture;
20  import io.netty.channel.ChannelInitializer;
21  import io.netty.channel.ChannelOption;
22  import io.netty.channel.EventLoopGroup;
23  import io.netty.channel.MultiThreadIoEventLoopGroup;
24  import io.netty.channel.nio.NioIoHandler;
25  import io.netty.channel.udt.UdtChannel;
26  import io.netty.channel.udt.nio.NioUdtProvider;
27  import io.netty.handler.logging.LogLevel;
28  import io.netty.handler.logging.LoggingHandler;
29  import io.netty.util.concurrent.DefaultThreadFactory;
30  
31  import java.util.concurrent.ThreadFactory;
32  
33  /**
34   * UDT Byte Stream Server
35   * <p>
36   * Echoes back any received data from a client.
37   */
38  public final class ByteEchoServer {
39  
40      static final int PORT = Integer.parseInt(System.getProperty("port", "8007"));
41  
42      public static void main(String[] args) throws Exception {
43          final ThreadFactory factory = new DefaultThreadFactory("udt");
44          final EventLoopGroup group = new MultiThreadIoEventLoopGroup(
45                  1, factory, NioIoHandler.newFactory(NioUdtProvider.BYTE_PROVIDER));
46  
47          // Configure the server.
48          try {
49              final ServerBootstrap boot = new ServerBootstrap();
50              boot.group(group)
51                      .channelFactory(NioUdtProvider.BYTE_ACCEPTOR)
52                      .option(ChannelOption.SO_BACKLOG, 10)
53                      .handler(new LoggingHandler(LogLevel.INFO))
54                      .childHandler(new ChannelInitializer<UdtChannel>() {
55                          @Override
56                          public void initChannel(final UdtChannel ch)
57                                  throws Exception {
58                              ch.pipeline().addLast(
59                                      new LoggingHandler(LogLevel.INFO),
60                                      new ByteEchoServerHandler());
61                          }
62                      });
63              // Start the server.
64              final ChannelFuture future = boot.bind(PORT).sync();
65              // Wait until the server socket is closed.
66              future.channel().closeFuture().sync();
67          } finally {
68              // Shut down all event loops to terminate all threads.
69              group.shutdownGracefully();
70          }
71      }
72  }
