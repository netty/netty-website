61      public static void main(String[] args) throws Exception {
62          String postSimple, postFile, get;
63          if (BASE_URL.endsWith("/")) {
64              postSimple = BASE_URL + "formpost";
65              postFile = BASE_URL + "formpostmultipart";
66              get = BASE_URL + "formget";
67          } else {
68              postSimple = BASE_URL + "/formpost";
69              postFile = BASE_URL + "/formpostmultipart";
70              get = BASE_URL + "/formget";
71          }
72  
73          URI uriSimple = new URI(postSimple);
74          String scheme = uriSimple.getScheme() == null? "http" : uriSimple.getScheme();
75          String host = uriSimple.getHost() == null? "127.0.0.1" : uriSimple.getHost();
76          int port = uriSimple.getPort();
77          if (port == -1) {
78              if ("http".equalsIgnoreCase(scheme)) {
79                  port = 80;
80              } else if ("https".equalsIgnoreCase(scheme)) {
81                  port = 443;
82              }
83          }
84  
85          if (!"http".equalsIgnoreCase(scheme) && !"https".equalsIgnoreCase(scheme)) {
86              System.err.println("Only HTTP(S) is supported.");
87              return;
88          }
89  
90          final boolean ssl = "https".equalsIgnoreCase(scheme);
91          final SslContext sslCtx;
92          if (ssl) {
93              sslCtx = SslContextBuilder.forClient()
94                  .trustManager(InsecureTrustManagerFactory.INSTANCE).build();
95          } else {
96              sslCtx = null;
97          }
98  
99          URI uriFile = new URI(postFile);
100         File file = new File(FILE);
101         if (!file.canRead()) {
102             throw new FileNotFoundException(FILE);
103         }
104 
105         // Configure the client.
106         EventLoopGroup group = new MultiThreadIoEventLoopGroup(NioIoHandler.newFactory());
107 
108         // setup the factory: here using a mixed memory/disk based on size threshold
109         HttpDataFactory factory = new DefaultHttpDataFactory(DefaultHttpDataFactory.MINSIZE); // Disk if MINSIZE exceed
110 
111         DiskFileUpload.deleteOnExitTemporaryFile = true; // should delete file on exit (in normal exit)
112         DiskFileUpload.baseDirectory = null; // system temp directory
113         DiskAttribute.deleteOnExitTemporaryFile = true; // should delete file on exit (in normal exit)
114         DiskAttribute.baseDirectory = null; // system temp directory
115 
116         try {
117             Bootstrap b = new Bootstrap();
118             b.group(group).channel(NioSocketChannel.class).handler(new HttpUploadClientInitializer(sslCtx));
119 
120             // Simple Get form: no factory used (not usable)
121             List<Entry<String, String>> headers = formget(b, host, port, get, uriSimple);
122             if (headers == null) {
123                 factory.cleanAllHttpData();
124                 return;
125             }
126 
127             // Simple Post form: factory used for big attributes
128             List<InterfaceHttpData> bodylist = formpost(b, host, port, uriSimple, file, factory, headers);
129             if (bodylist == null) {
130                 factory.cleanAllHttpData();
131                 return;
132             }
133 
134             // Multipart Post form: factory used
135             formpostmultipart(b, host, port, uriFile, factory, headers, bodylist);
136         } finally {
137             // Shut down executor threads to exit.
138             group.shutdownGracefully();
139 
140             // Really clean all temporary files if they still exist
141             factory.cleanAllHttpData();
142         }
143     }

145     /**
146      * Standard usage of HTTP API in Netty without file Upload (get is not able to achieve File upload
147      * due to limitation on request size).
148      *
149      * @return the list of headers that will be used in every example after
150      **/
151     private static List<Entry<String, String>> formget(
152             Bootstrap bootstrap, String host, int port, String get, URI uriSimple) throws Exception {
153         // XXX /formget
154         // No use of HttpPostRequestEncoder since not a POST
155         Channel channel = bootstrap.connect(host, port).sync().channel();
156 
157         // Prepare the HTTP request.
158         QueryStringEncoder encoder = new QueryStringEncoder(get);
159         // add Form attribute
160         encoder.addParam("getform", "GET");
161         encoder.addParam("info", "first value");
162         encoder.addParam("secondinfo", "secondvalue ���&");
163         // not the big one since it is not compatible with GET size
164         // encoder.addParam("thirdinfo", textArea);
165         encoder.addParam("thirdinfo", "third value\r\ntest second line\r\n\r\nnew line\r\n");
166         encoder.addParam("Send", "Send");
167 
168         URI uriGet = new URI(encoder.toString());
169         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, uriGet.toASCIIString());
170         HttpHeaders headers = request.headers();
171         headers.set(HttpHeaderNames.HOST, host);
172         headers.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
173         headers.set(HttpHeaderNames.ACCEPT_ENCODING, HttpHeaderValues.GZIP + "," + HttpHeaderValues.DEFLATE);
174 
175         headers.set(HttpHeaderNames.ACCEPT_CHARSET, "ISO-8859-1,utf-8;q=0.7,*;q=0.7");
176         headers.set(HttpHeaderNames.ACCEPT_LANGUAGE, "fr");
177         headers.set(HttpHeaderNames.REFERER, uriSimple.toString());
178         headers.set(HttpHeaderNames.USER_AGENT, "Netty Simple Http Client side");
179         headers.set(HttpHeaderNames.ACCEPT, "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8");
180 
181         //connection will not close but needed
182         // headers.set("Connection","keep-alive");
183         // headers.set("Keep-Alive","300");
184 
185         headers.set(
186                 HttpHeaderNames.COOKIE, ClientCookieEncoder.STRICT.encode(
187                         new DefaultCookie("my-cookie", "foo"),
188                         new DefaultCookie("another-cookie", "bar"))
189         );
190 
191         // send request
192         channel.write(request);
193         channel.writeAndFlush(LastHttpContent.EMPTY_LAST_CONTENT);
194 
195         // Wait for the server to close the connection.
196         channel.closeFuture().sync();
197 
198         // convert headers to list
199         return headers.entries();
200     }

202     /**
203      * Standard post without multipart but already support on Factory (memory management)
204      *
205      * @return the list of HttpData object (attribute and file) to be reused on next post
206      */
207     private static List<InterfaceHttpData> formpost(
208             Bootstrap bootstrap,
209             String host, int port, URI uriSimple, File file, HttpDataFactory factory,
210             List<Entry<String, String>> headers) throws Exception {
211         // XXX /formpost
212         // Start the connection attempt.
213         ChannelFuture future = bootstrap.connect(SocketUtils.socketAddress(host, port));
214         // Wait until the connection attempt succeeds or fails.
215         Channel channel = future.sync().channel();
216 
217         // Prepare the HTTP request.
218         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.POST, uriSimple.toASCIIString());
219 
220         // Use the PostBody encoder
221         HttpPostRequestEncoder bodyRequestEncoder =
222                 new HttpPostRequestEncoder(factory, request, false);  // false => not multipart
223 
224         // it is legal to add directly header or cookie into the request until finalize
225         for (Entry<String, String> entry : headers) {
226             request.headers().set(entry.getKey(), entry.getValue());
227         }
228 
229         // add Form attribute
230         bodyRequestEncoder.addBodyAttribute("getform", "POST");
231         bodyRequestEncoder.addBodyAttribute("info", "first value");
232         bodyRequestEncoder.addBodyAttribute("secondinfo", "secondvalue ���&");
233         bodyRequestEncoder.addBodyAttribute("thirdinfo", textArea);
234         bodyRequestEncoder.addBodyAttribute("fourthinfo", textAreaLong);
235         bodyRequestEncoder.addBodyFileUpload("myfile", file, "application/x-zip-compressed", false);
236 
237         // finalize request
238         request = bodyRequestEncoder.finalizeRequest();
239 
240         // Create the bodylist to be reused on the last version with Multipart support
241         List<InterfaceHttpData> bodylist = bodyRequestEncoder.getBodyListAttributes();
242 
243         // send request
244         channel.write(request);
245 
246         // test if request was chunked and if so, finish the write
247         if (bodyRequestEncoder.isChunked()) { // could do either request.isChunked()
248             // either do it through ChunkedWriteHandler
249             channel.write(bodyRequestEncoder);
250         }
251         channel.flush();
252 
253         // Do not clear here since we will reuse the InterfaceHttpData on the next request
254         // for the example (limit action on client side). Take this as a broadcast of the same
255         // request on both Post actions.
256         //
257         // On standard program, it is clearly recommended to clean all files after each request
258         // bodyRequestEncoder.cleanFiles();
259 
260         // Wait for the server to close the connection.
261         channel.closeFuture().sync();
262         return bodylist;
263     }

265     /**
266      * Multipart example
267      */
268     private static void formpostmultipart(
269             Bootstrap bootstrap, String host, int port, URI uriFile, HttpDataFactory factory,
270             Iterable<Entry<String, String>> headers, List<InterfaceHttpData> bodylist) throws Exception {
271         // XXX /formpostmultipart
272         // Start the connection attempt.
273         ChannelFuture future = bootstrap.connect(SocketUtils.socketAddress(host, port));
274         // Wait until the connection attempt succeeds or fails.
275         Channel channel = future.sync().channel();
276 
277         // Prepare the HTTP request.
278         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.POST, uriFile.toASCIIString());
279 
280         // Use the PostBody encoder
281         HttpPostRequestEncoder bodyRequestEncoder =
282                 new HttpPostRequestEncoder(factory, request, true); // true => multipart
283 
284         // it is legal to add directly header or cookie into the request until finalize
285         for (Entry<String, String> entry : headers) {
286             request.headers().set(entry.getKey(), entry.getValue());
287         }
288 
289         // add Form attribute from previous request in formpost()
290         bodyRequestEncoder.setBodyHttpDatas(bodylist);
291 
292         // finalize request
293         bodyRequestEncoder.finalizeRequest();
294 
295         // send request
296         channel.write(request);
297 
298         // test if request was chunked and if so, finish the write
299         if (bodyRequestEncoder.isChunked()) {
300             channel.write(bodyRequestEncoder);
301         }
302         channel.flush();
303 
304         // Now no more use of file representation (and list of HttpData)
305         bodyRequestEncoder.cleanFiles();
306 
307         // Wait for the server to close the connection.
308         channel.closeFuture().sync();
309     }
