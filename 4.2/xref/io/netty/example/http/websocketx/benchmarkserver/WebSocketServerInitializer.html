16  package io.netty.example.http.websocketx.benchmarkserver;
17  
18  import io.netty.channel.ChannelInitializer;
19  import io.netty.channel.ChannelPipeline;
20  import io.netty.channel.socket.SocketChannel;
21  import io.netty.handler.codec.http.HttpObjectAggregator;
22  import io.netty.handler.codec.http.HttpServerCodec;
23  import io.netty.handler.ssl.SslContext;
24  
25  /**
26   */
27  public class WebSocketServerInitializer extends ChannelInitializer<SocketChannel> {
28  
29      private final SslContext sslCtx;
30  
31      public WebSocketServerInitializer(SslContext sslCtx) {
32          this.sslCtx = sslCtx;
33      }
34  
35      @Override
36      public void initChannel(SocketChannel ch) throws Exception {
37          ChannelPipeline pipeline = ch.pipeline();
38          if (sslCtx != null) {
39              pipeline.addLast(sslCtx.newHandler(ch.alloc()));
40          }
41          pipeline.addLast(new HttpServerCodec());
42          pipeline.addLast(new HttpObjectAggregator(65536));
43          pipeline.addLast(new WebSocketServerHandler());
44      }
45  }
