16  package io.netty.example.factorial;
17  
18  import io.netty.channel.ChannelHandlerContext;
19  import io.netty.channel.ChannelInboundMessageHandlerAdapter;
20  
21  import java.math.BigInteger;
22  import java.util.Formatter;
23  import java.util.logging.Level;
24  import java.util.logging.Logger;
25  
26  /**
27   * Handler for a server-side channel.  This handler maintains stateful
28   * information which is specific to a certain channel using member variables.
29   * Therefore, an instance of this handler can cover only one channel.  You have
30   * to create a new handler instance whenever you create a new channel and insert
31   * this handler  to avoid a race condition.
32   */
33  public class FactorialServerHandler extends ChannelInboundMessageHandlerAdapter<BigInteger> {
34  
35      private static final Logger logger = Logger.getLogger(
36              FactorialServerHandler.class.getName());
37  
38      private BigInteger lastMultiplier = new BigInteger("1");
39      private BigInteger factorial = new BigInteger("1");
40  
41      @Override
42      public void messageReceived(
43              ChannelHandlerContext ctx, BigInteger msg) throws Exception {
44          // Calculate the cumulative factorial and send it to the client.
45          lastMultiplier = msg;
46          factorial = factorial.multiply(msg);
47          ctx.write(factorial);
48      }
49  
50      @Override
51      public void channelInactive(
52              ChannelHandlerContext ctx) throws Exception {
53          logger.info(new Formatter().format(
54                  "Factorial of %,d is: %,d", lastMultiplier, factorial).toString());
55      }
56  
57      @Override
58      public void exceptionCaught(
59              ChannelHandlerContext ctx, Throwable cause) throws Exception {
60          logger.log(
61                  Level.WARNING,
62                  "Unexpected exception from downstream.", cause);
63          ctx.close();
64      }
65  }
