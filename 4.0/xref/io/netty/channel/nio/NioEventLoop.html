44  /**
45   * {@link SingleThreadEventLoop} implementation which register the {@link Channel}'s to a
46   * {@link Selector} and so does the multi-plexing of these in the event loop.

48   */

77      NioEventLoop(
78              NioEventLoopGroup parent, ThreadFactory threadFactory,
79              ChannelTaskScheduler scheduler, SelectorProvider selectorProvider) {
80          super(parent, threadFactory, scheduler);
81          if (selectorProvider == null) {
82              throw new NullPointerException("selectorProvider");
83          }
84          provider = selectorProvider;
85          selector = openSelector();
86      }

88      private Selector openSelector() {
89          try {
90              return provider.openSelector();

92              throw new ChannelException("failed to open a new selector", e);
93          }

102     /**
103      * Registers an arbitrary {@link SelectableChannel}, not necessarily created by Netty, to the {@link Selector}
104      * of this event loop.  Once the specified {@link SelectableChannel} is registered, the specified {@code task} will
105      * be executed by this event loop when the {@link SelectableChannel} is ready.
106      */
107     public void register(final SelectableChannel ch, final int interestOps, final NioTask<?> task) {
108         if (ch == null) {
109             throw new NullPointerException("ch");
110         }
111         if (interestOps == 0) {
112             throw new IllegalArgumentException("interestOps must be non-zero.");
113         }
114         if ((interestOps & ~ch.validOps()) != 0) {
115             throw new IllegalArgumentException(
116                     "invalid interestOps: " + interestOps + "(validOps: " + ch.validOps() + ')');
117         }
118         if (task == null) {
119             throw new NullPointerException("task");
120         }
121 
122         if (isShutdown()) {
123             throw new IllegalStateException("event loop shut down");
124         }
125 
126         try {
127             ch.register(selector, interestOps, task);
128         } catch (Exception e) {
129             throw new EventLoopException("failed to register a channel", e);
130         }
131     }

133     void executeWhenWritable(AbstractNioChannel channel, NioTask<SelectableChannel> task) {
134         if (channel == null) {
135             throw new NullPointerException("channel");
136         }
137 
138         if (isShutdown()) {
139             throw new IllegalStateException("event loop shut down");
140         }
141 
142         SelectionKey key = channel.selectionKey();
143         channel.writableTasks.offer(task);
144 
145         int interestOps = key.interestOps();
146         if ((interestOps & SelectionKey.OP_WRITE) == 0) {
147             key.interestOps(interestOps | SelectionKey.OP_WRITE);
148         }
149     }

151     public void rebuildSelector() {
152         if (!inEventLoop()) {
153             execute(new Runnable() {
154                 @Override
155                 public void run() {
156                     rebuildSelector();
157                 }
158             });
159             return;
160         }
161 
162         final Selector oldSelector = selector;
163         final Selector newSelector;
164 
165         if (oldSelector == null) {
166             return;
167         }
168 
169         try {
170             newSelector = Selector.open();
171         } catch (Exception e) {
172             logger.warn("Failed to create a new Selector.", e);
173             return;
174         }
175 
176         // Register all channels to the new Selector.
177         int nChannels = 0;
178         for (;;) {
179             try {
180                 for (SelectionKey key: oldSelector.keys()) {
181                     Object a = key.attachment();
182                     try {
183                         if (key.channel().keyFor(newSelector) != null) {
184                             continue;
185                         }
186 
187                         int interestOps = key.interestOps();
188                         key.cancel();
189                         key.channel().register(newSelector, interestOps, a);
190                         nChannels ++;
191                     } catch (Exception e) {
192                         logger.warn("Failed to re-register a Channel to the new Selector.", e);
193                         if (a instanceof AbstractNioChannel) {
194                             AbstractNioChannel ch = (AbstractNioChannel) a;
195                             ch.unsafe().close(ch.unsafe().voidFuture());
196                         } else {
197                             @SuppressWarnings("unchecked")
198                             NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
199                             invokeChannelUnregistered(task, key, e);
200                         }
201                     }
202                 }
203             } catch (ConcurrentModificationException e) {
204                 // Probably due to concurrent modification of the key set.
205                 continue;
206             }
207 
208             break;
209         }
210 
211         selector = newSelector;
212 
213         try {
214             // time to close the old selector as everything else is registered to the new one
215             oldSelector.close();
216         } catch (Throwable t) {
217             if (logger.isWarnEnabled()) {
218                 logger.warn("Failed to close the old Selector.", t);
219             }
220         }
221 
222         logger.info("Migrated " + nChannels + " channel(s) to the new Selector.");
223     }

225     @Override
226     protected void run() {
227         Selector selector = this.selector;
228         int selectReturnsImmediately = 0;
229 
230         // use 80% of the timeout for measure
231         long minSelectTimeout = SelectorUtil.SELECT_TIMEOUT_NANOS / 100 * 80;
232 
233         for (;;) {
234             wakenUp.set(false);
235 
236             try {
237                 long beforeSelect = System.nanoTime();
238                 int selected = SelectorUtil.select(selector);
239 
240                 if (SelectorUtil.EPOLL_BUG_WORKAROUND) {
241                     if (selected == 0) {
242                         long timeBlocked = System.nanoTime()  - beforeSelect;
243                         if (timeBlocked < minSelectTimeout) {
244                             // returned before the minSelectTimeout elapsed with nothing select.
245                             // this may be the cause of the jdk epoll(..) bug, so increment the counter
246                             // which we use later to see if its really the jdk bug.
247                             selectReturnsImmediately ++;
248                         } else {
249                             selectReturnsImmediately = 0;
250                         }
251                         if (selectReturnsImmediately == 10) {
252                             // The selector returned immediately for 10 times in a row,
253                             // so recreate one selector as it seems like we hit the
254                             // famous epoll(..) jdk bug.
255                             rebuildSelector();
256                             selector = this.selector;
257                             selectReturnsImmediately = 0;
258 
259                             // try to select again
260                             continue;
261                         }
262                     } else {
263                         // reset counter
264                         selectReturnsImmediately = 0;
265                     }
266                 }
267 
268                 // 'wakenUp.compareAndSet(false, true)' is always evaluated
269                 // before calling 'selector.wakeup()' to reduce the wake-up
270                 // overhead. (Selector.wakeup() is an expensive operation.)
271                 //
272                 // However, there is a race condition in this approach.
273                 // The race condition is triggered when 'wakenUp' is set to
274                 // true too early.
275                 //
276                 // 'wakenUp' is set to true too early if:
277                 // 1) Selector is waken up between 'wakenUp.set(false)' and
278                 //    'selector.select(...)'. (BAD)
279                 // 2) Selector is waken up between 'selector.select(...)' and
280                 //    'if (wakenUp.get()) { ... }'. (OK)
281                 //
282                 // In the first case, 'wakenUp' is set to true and the
283                 // following 'selector.select(...)' will wake up immediately.
284                 // Until 'wakenUp' is set to false again in the next round,
285                 // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
286                 // any attempt to wake up the Selector will fail, too, causing
287                 // the following 'selector.select(...)' call to block
288                 // unnecessarily.
289                 //
290                 // To fix this problem, we wake up the selector again if wakenUp
291                 // is true immediately after selector.select(...).
292                 // It is inefficient in that it wakes up the selector for both
293                 // the first case (BAD - wake-up required) and the second case
294                 // (OK - no wake-up required).
295 
296                 if (wakenUp.get()) {
297                     selector.wakeup();
298                 }
299 
300                 cancelledKeys = 0;
301 
302                 processSelectedKeys();
303                 selector = this.selector;
304 
305                 runAllTasks();
306                 selector = this.selector;
307 
308                 if (isShutdown()) {
309                     closeAll();
310                     if (confirmShutdown()) {
311                         break;
312                     }
313                 }
314             } catch (Throwable t) {
315                 logger.warn(
316                         "Unexpected exception in the selector loop.", t);
317 
318                 // Prevent possible consecutive immediate failures that lead to
319                 // excessive CPU consumption.
320                 try {
321                     Thread.sleep(1000);
322                 } catch (InterruptedException e) {
323                     // Ignore.
324                 }
325             }
326         }
327     }

338     void cancel(SelectionKey key) {
339         key.cancel();
340         cancelledKeys ++;
341         if (cancelledKeys >= CLEANUP_INTERVAL) {
342             cancelledKeys = 0;
343             cleanedCancelledKeys = true;
344             SelectorUtil.cleanupKeys(selector);
345         }
346     }

348     private void processSelectedKeys() {
349         Set<SelectionKey> selectedKeys = selector.selectedKeys();
350         // check if the set is empty and if so just return to not create garbage by
351         // creating a new Iterator every time even if there is nothing to process.
352         // See https://github.com/netty/netty/issues/597
353         if (selectedKeys.isEmpty()) {
354             return;
355         }
356 
357         Iterator<SelectionKey> i;
358         cleanedCancelledKeys = false;
359         boolean clearSelectedKeys = true;
360         try {
361             for (i = selectedKeys.iterator(); i.hasNext();) {
362                 final SelectionKey k = i.next();
363                 final Object a = k.attachment();
364                 if (a instanceof AbstractNioChannel) {
365                     processSelectedKey(k, (AbstractNioChannel) a);
366                 } else {
367                     @SuppressWarnings("unchecked")
368                     NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
369                     processSelectedKey(k, task);
370                 }
371 
372                 if (cleanedCancelledKeys) {
373                     // Create the iterator again to avoid ConcurrentModificationException
374                     if (selectedKeys.isEmpty()) {
375                         clearSelectedKeys = false;
376                         break;
377                     } else {
378                         i = selectedKeys.iterator();
379                     }
380                 }
381             }
382         } finally {
383             if (clearSelectedKeys) {
384                 selectedKeys.clear();
385             }
386         }
387     }

389     private static void processSelectedKey(SelectionKey k, AbstractNioChannel ch) {
390         final NioUnsafe unsafe = ch.unsafe();
391         if (!k.isValid()) {
392             // close the channel if the key is not valid anymore
393             unsafe.close(unsafe.voidFuture());
394             return;
395         }
396 
397         int readyOps = -1;
398         try {
399             readyOps = k.readyOps();
400             if ((readyOps & (SelectionKey.OP_READ | SelectionKey.OP_ACCEPT)) != 0 || readyOps == 0) {
401                 unsafe.read();
402                 if (!ch.isOpen()) {
403                     // Connection already closed - no need to handle write.
404                     return;
405                 }
406             }
407             if ((readyOps & SelectionKey.OP_WRITE) != 0) {
408                 processWritable(k, ch);
409             }
410             if ((readyOps & SelectionKey.OP_CONNECT) != 0) {
411                 // remove OP_CONNECT as otherwise Selector.select(..) will always return without blocking
412                 // See https://github.com/netty/netty/issues/924
413                 int ops = k.interestOps();
414                 ops &= ~SelectionKey.OP_CONNECT;
415                 k.interestOps(ops);
416 
417                 unsafe.finishConnect();
418             }
419         } catch (CancelledKeyException e) {
420             if (readyOps != -1 && (readyOps & SelectionKey.OP_WRITE) != 0) {
421                 unregisterWritableTasks(ch);
422             }
423             unsafe.close(unsafe.voidFuture());
424         }
425     }

427     private static void processWritable(SelectionKey k, AbstractNioChannel ch) {
428         if (ch.writableTasks.isEmpty()) {
429             ch.unsafe().flushNow();
430         } else {
431             NioTask<SelectableChannel> task;
432             for (;;) {
433                 task = ch.writableTasks.poll();
434                 if (task == null) { break; }
435                 processSelectedKey(ch.selectionKey(), task);
436             }
437             k.interestOps(k.interestOps() | SelectionKey.OP_WRITE);
438         }
439     }
440 
441     private static void unregisterWritableTasks(AbstractNioChannel ch) {
442         NioTask<SelectableChannel> task;
443         for (;;) {
444             task = ch.writableTasks.poll();
445             if (task == null) {
446                 break;
447             } else {
448                 invokeChannelUnregistered(task, ch.selectionKey(), null);
449             }
450         }
451     }

453     private static void processSelectedKey(SelectionKey k, NioTask<SelectableChannel> task) {
454         int state = 0;
455         try {
456             task.channelReady(k.channel(), k);
457             state = 1;
458         } catch (Exception e) {
459             k.cancel();
460             invokeChannelUnregistered(task, k, e);
461             state = 2;
462         } finally {
463             switch (state) {
464             case 0:
465                 k.cancel();
466                 invokeChannelUnregistered(task, k, null);
467                 break;
468             case 1:
469                 if (!k.isValid()) { // Cancelled by channelReady()
470                     invokeChannelUnregistered(task, k, null);
471                 }
472                 break;
473             }
474         }
475     }

477     private void closeAll() {
478         SelectorUtil.cleanupKeys(selector);
479         Set<SelectionKey> keys = selector.keys();
480         Collection<AbstractNioChannel> channels = new ArrayList<AbstractNioChannel>(keys.size());
481         for (SelectionKey k: keys) {
482             Object a = k.attachment();
483             if (a instanceof AbstractNioChannel) {
484                 channels.add((AbstractNioChannel) a);
485             } else {
486                 k.cancel();
487                 @SuppressWarnings("unchecked")
488                 NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
489                 invokeChannelUnregistered(task, k, null);
490             }
491         }
492 
493         for (AbstractNioChannel ch: channels) {
494             unregisterWritableTasks(ch);
495             ch.unsafe().close(ch.unsafe().voidFuture());
496         }
497     }

499     private static void invokeChannelUnregistered(NioTask<SelectableChannel> task, SelectionKey k, Throwable cause) {
500         try {
501             task.channelUnregistered(k.channel(), cause);
502         } catch (Exception e) {
503             logger.warn("Unexpected exception while running NioTask.channelUnregistered()", e);
504         }
505     }

507     @Override
508     protected void wakeup(boolean inEventLoop) {
509         if (!inEventLoop && wakenUp.compareAndSet(false, true)) {
510             selector.wakeup();
511         }

514     void selectNow() throws IOException {
515         try {
516             selector.selectNow();
517         } finally {
518             // restore wakup state if needed
519             if (wakenUp.get()) {
520                 selector.wakeup();
521             }
522         }
