69      public DefaultChannelPipeline(AbstractChannel channel) {
70          if (channel == null) {
71              throw new NullPointerException("channel");
72          }
73          this.channel = channel;
74  
75          TailHandler tailHandler = new TailHandler();
76          tail = new DefaultChannelHandlerContext(this, null, generateName(tailHandler), tailHandler);
77  
78          HeadHandler headHandler = new HeadHandler(channel.unsafe());
79          head = new DefaultChannelHandlerContext(this, null, generateName(headHandler), headHandler);
80  
81          head.next = tail;
82          tail.prev = head;
83      }

219     @Override
220     public ChannelPipeline addFirst(EventExecutorGroup executor, ChannelHandler... handlers) {
221         if (handlers == null) {
222             throw new NullPointerException("handlers");
223         }
224         if (handlers.length == 0 || handlers[0] == null) {
225             return this;
226         }
227 
228         int size;
229         for (size = 1; size < handlers.length; size ++) {
230             if (handlers[size] == null) {
231                 break;
232             }
233         }
234 
235         for (int i = size - 1; i >= 0; i --) {
236             ChannelHandler h = handlers[i];
237             addFirst(executor, generateName(h), h);
238         }
239 
240         return this;
241     }

249     public ChannelPipeline addLast(EventExecutorGroup executor, ChannelHandler... handlers) {
250         if (handlers == null) {
251             throw new NullPointerException("handlers");
252         }
253 
254         for (ChannelHandler h: handlers) {
255             if (h == null) {
256                 break;
257             }
258             addLast(executor, generateName(h), h);
259         }
260 
261         return this;
262     }

264     private String generateName(ChannelHandler handler) {
265         WeakHashMap<Class<?>, String> cache = nameCaches[(int) (Thread.currentThread().getId() % nameCaches.length)];
266         Class<?> handlerType = handler.getClass();
267         String name;
268         synchronized (cache) {
269             name = cache.get(handlerType);
270             if (name == null) {
271                 name = StringUtil.simpleClassName(handlerType) + "#0";
272                 cache.put(handlerType, name);
273             }
274         }
275 
276         synchronized (this) {
277             // It's not very likely for a user to put more than one handler of the same type, but make sure to avoid
278             // any name conflicts.  Note that we don't cache the names generated here.
279             if (name2ctx.containsKey(name)) {
280                 String baseName = name.substring(0, name.length() - 1); // Strip the trailing '0'.
281                 for (int i = 1;; i ++) {
282                     String newName = baseName + i;
283                     if (!name2ctx.containsKey(newName)) {
284                         name = newName;
285                         break;
286                     }
287                 }
288             }
289         }
290 
291         return name;
292     }

311     private DefaultChannelHandlerContext remove(final DefaultChannelHandlerContext ctx) {
312         assert ctx != head && ctx != tail;
313 
314         DefaultChannelHandlerContext context;
315         Future<?> future;
316 
317         synchronized (this) {
318             if (!ctx.channel().isRegistered() || ctx.executor().inEventLoop()) {
319                 remove0(ctx);
320                 return ctx;
321             } else {
322                future = ctx.executor().submit(new Runnable() {
323                    @Override
324                    public void run() {
325                        synchronized (DefaultChannelPipeline.this) {
326                            remove0(ctx);
327                        }
328                    }
329                });
330                context = ctx;
331             }
332         }
333 
334         // Run the following 'waiting' code outside of the above synchronized block
335         // in order to avoid deadlock
336 
337         waitForFuture(future);
338 
339         return context;
340     }

342     void remove0(DefaultChannelHandlerContext ctx) {
343         DefaultChannelHandlerContext prev = ctx.prev;
344         DefaultChannelHandlerContext next = ctx.next;
345         prev.next = next;
346         next.prev = prev;
347         name2ctx.remove(ctx.name());
348         callHandlerRemoved(ctx);
349     }

385     private ChannelHandler replace(
386             final DefaultChannelHandlerContext ctx, final String newName,
387             ChannelHandler newHandler) {
388 
389         assert ctx != head && ctx != tail;
390 
391         Future<?> future;
392         synchronized (this) {
393             boolean sameName = ctx.name().equals(newName);
394             if (!sameName) {
395                 checkDuplicateName(newName);
396             }
397 
398             final DefaultChannelHandlerContext newCtx =
399                     new DefaultChannelHandlerContext(this, ctx.executor, newName, newHandler);
400 
401             if (!newCtx.channel().isRegistered() || newCtx.executor().inEventLoop()) {
402                 replace0(ctx, newName, newCtx);
403                 return ctx.handler();
404             } else {
405                 future = newCtx.executor().submit(new Runnable() {
406                     @Override
407                     public void run() {
408                         synchronized (DefaultChannelPipeline.this) {
409                             replace0(ctx, newName, newCtx);
410                         }
411                     }
412                 });
413             }
414         }
415 
416         // Run the following 'waiting' code outside of the above synchronized block
417         // in order to avoid deadlock
418 
419         waitForFuture(future);
420 
421         return ctx.handler();
422     }

424     private void replace0(DefaultChannelHandlerContext oldCtx, String newName,
425                           DefaultChannelHandlerContext newCtx) {
426         checkMultiplicity(newCtx);
427 
428         DefaultChannelHandlerContext prev = oldCtx.prev;
429         DefaultChannelHandlerContext next = oldCtx.next;
430         newCtx.prev = prev;
431         newCtx.next = next;
432 
433         // Finish the replacement of oldCtx with newCtx in the linked list.
434         // Note that this doesn't mean events will be sent to the new handler immediately
435         // because we are currently at the event handler thread and no more than one handler methods can be invoked
436         // at the same time (we ensured that in replace().)
437         prev.next = newCtx;
438         next.prev = newCtx;
439 
440         if (!oldCtx.name().equals(newName)) {
441             name2ctx.remove(oldCtx.name());
442         }
443         name2ctx.put(newName, newCtx);
444 
445         // update the reference to the replacement so forward of buffered content will work correctly
446         oldCtx.prev = newCtx;
447         oldCtx.next = newCtx;
448 
449         // Invoke newHandler.handlerAdded() first (i.e. before oldHandler.handlerRemoved() is invoked)
450         // because callHandlerRemoved() will trigger inboundBufferUpdated() or flush() on newHandler and those
451         // event handlers must be called after handlerAdded().
452         callHandlerAdded(newCtx);
453         callHandlerRemoved(oldCtx);
454     }
455 
456     private static void checkMultiplicity(ChannelHandlerContext ctx) {
457         ChannelHandler handler = ctx.handler();
458         if (handler instanceof ChannelHandlerAdapter) {
459             ChannelHandlerAdapter h = (ChannelHandlerAdapter) handler;
460             if (!h.isSharable() && h.added) {
461                 throw new ChannelPipelineException(
462                         h.getClass().getName() +
463                         " is not a @Sharable handler, so can't be added or removed multiple times.");
464             }
465             h.added = true;
466         }
467     }
468 
469     private void callHandlerAdded(final ChannelHandlerContext ctx) {
470         if (ctx.channel().isRegistered() && !ctx.executor().inEventLoop()) {
471             ctx.executor().execute(new Runnable() {
472                 @Override
473                 public void run() {
474                     callHandlerAdded0(ctx);
475                 }
476             });
477             return;
478         }
479         callHandlerAdded0(ctx);
480     }
481 
482     private void callHandlerAdded0(final ChannelHandlerContext ctx) {
483         try {
484             ctx.handler().handlerAdded(ctx);
485         } catch (Throwable t) {
486             boolean removed = false;
487             try {
488                 remove((DefaultChannelHandlerContext) ctx);
489                 removed = true;
490             } catch (Throwable t2) {
491                 if (logger.isWarnEnabled()) {
492                     logger.warn("Failed to remove a handler: " + ctx.name(), t2);
493                 }
494             }
495 
496             if (removed) {
497                 fireExceptionCaught(new ChannelPipelineException(
498                         ctx.handler().getClass().getName() +
499                         ".handlerAdded() has thrown an exception; removed.", t));
500             } else {
501                 fireExceptionCaught(new ChannelPipelineException(
502                         ctx.handler().getClass().getName() +
503                         ".handlerAdded() has thrown an exception; also failed to remove.", t));
504             }
505         }
506     }
507 
508     private void callHandlerRemoved(final DefaultChannelHandlerContext ctx) {
509         if (ctx.channel().isRegistered() && !ctx.executor().inEventLoop()) {
510             ctx.executor().execute(new Runnable() {
511                 @Override
512                 public void run() {
513                     callHandlerRemoved0(ctx);
514                  }
515             });
516             return;
517         }
518         callHandlerRemoved0(ctx);
519     }
520 
521     private void callHandlerRemoved0(final DefaultChannelHandlerContext ctx) {
522         // Notify the complete removal.
523         try {
524             ctx.handler().handlerRemoved(ctx);
525             ctx.setRemoved();
526         } catch (Throwable t) {
527             fireExceptionCaught(new ChannelPipelineException(
528                     ctx.handler().getClass().getName() + ".handlerRemoved() has thrown an exception.", t));
529         }
530     }

532     /**
533      * Waits for a future to finish.  If the task is interrupted, then the current thread will be interrupted.
534      * It is expected that the task performs any appropriate locking.
535      * <p>
536      * If the internal call throws a {@link Throwable}, but it is not an instance of {@link Error} or
537      * {@link RuntimeException}, then it is wrapped inside a {@link ChannelPipelineException} and that is
538      * thrown instead.</p>
539      *
540      * @param future wait for this future
541      * @see Future#get()
542      * @throws Error if the task threw this.
543      * @throws RuntimeException if the task threw this.
544      * @throws ChannelPipelineException with a {@link Throwable} as a cause, if the task threw another type of
545      *         {@link Throwable}.
546      */
547     private static void waitForFuture(Future<?> future) {
548         try {
549             future.get();
550         } catch (ExecutionException ex) {
551             // In the arbitrary case, we can throw Error, RuntimeException, and Exception
552             PlatformDependent.throwException(ex.getCause());
553         } catch (InterruptedException ex) {
554             // Interrupt the calling thread (note that this method is not called from the event loop)
555             Thread.currentThread().interrupt();
556         }
557     }

627     @Override
628     public ChannelHandlerContext context(ChannelHandler handler) {
629         if (handler == null) {
630             throw new NullPointerException("handler");
631         }
632 
633         DefaultChannelHandlerContext ctx = head.next;
634         for (;;) {
635 
636             if (ctx == null) {
637                 return null;
638             }
639 
640             if (ctx.handler() == handler) {
641                 return ctx;
642             }
643 
644             ctx = ctx.next;
645         }
646     }

700     @Override
701     public String toString() {
702         StringBuilder buf = new StringBuilder();
703         buf.append(StringUtil.simpleClassName(this));
704         buf.append('{');
705         DefaultChannelHandlerContext ctx = head.next;
706         for (;;) {
707             if (ctx == tail) {
708                 break;
709             }
710 
711             buf.append('(');
712             buf.append(ctx.name());
713             buf.append(" = ");
714             buf.append(ctx.handler().getClass().getName());
715             buf.append(')');
716 
717             ctx = ctx.next;
718             if (ctx == tail) {
719                 break;
720             }
721 
722             buf.append(", ");
723         }
724         buf.append('}');
725         return buf.toString();
726     }
