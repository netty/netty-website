31  /**
32   * A simple array-backed list that holds one or more messages.
33   *
34   * <h3>Recycling a {@link MessageList}</h3>
35   * <p>
36   * A {@link MessageList} is internally managed by a thread-local object pool to keep the GC pressure minimal.
37   * To return the {@link MessageList} to the pool, you must call one of the following methods: {@link #recycle()},
38   * {@link #releaseAllAndRecycle()}, or {@link #releaseAllAndRecycle(int)}.  If the list is returned to the pool (i.e.
39   * recycled), it will be reused when you attempts to get a {@link MessageList} via {@link #newInstance()}.
40   * </p><p>
41   * If you don't think recycling a {@link MessageList} isn't worth, it is also fine not to recycle it.  Because of this
42   * relaxed contract, you can also decide not to wrap your code with a {@code try-finally} block only to recycle a
43   * list.  However, if you decided to recycle it, you must keep in mind that:
44   * <ul>
45   * <li>you must make sure you do not access the list once it has been recycled.</li>
46   * <li>If you are given with a {@link MessageList} as a parameter of your handler, it means it is your handler's
47   *     responsibility to release the messages in it and to recycle it.</li>
48   * </ul>
49   * </p>
50   *
51   * @param <T> the type of the contained messages
52   */
53  public final class MessageList<T> implements Iterable<T> {

222     /**
223      * Add the message to this {@link MessageList} and return itself.
224      */
225     public MessageList<T> add(T value) {
226         if (value == null) {
227             throw new NullPointerException("value");
228         }
229         modifications ++;
230         int oldSize = size;
231         int newSize = oldSize + 1;
232         ensureCapacity(newSize);
233         elements[oldSize] = value;
234         size = newSize;
235         if (byteBufsOnly && !(value instanceof ByteBuf)) {
236             byteBufsOnly = false;
237         }
238         return this;
239     }

251     /**
252      * Add the messages contained in the array, using the given src index and src length, to this {@link MessageList}
253      * and return itself.
254      */
255     public MessageList<T> add(T[] src, int srcIdx, int srcLen) {
256         if (src == null) {
257             throw new NullPointerException("src");
258         }
259 
260         modifications ++;
261 
262         int dstIdx = size;
263         final int newSize = dstIdx + srcLen;
264         ensureCapacity(newSize);
265 
266         final int srcEndIdx = srcIdx + srcLen;
267         int i = srcIdx;
268         try {
269             if (byteBufsOnly) {
270                 while (i < srcEndIdx) {
271                     T m = src[srcIdx];
272                     if (m == null) {
273                         throw new NullPointerException("src[" + srcIdx + ']');
274                     }
275 
276                     elements[dstIdx ++] = m;
277                     i ++;
278 
279                     if (!(m instanceof ByteBuf)) {
280                         byteBufsOnly = false;
281                         break;
282                     }
283                 }
284             }
285 
286             for (; i < srcEndIdx; i ++) {
287                 T m = src[srcIdx];
288                 if (m == null) {
289                     throw new NullPointerException("src[" + srcIdx + ']');
290                 }
291 
292                 elements[dstIdx ++] = m;
293             }
294         } finally {
295             if (dstIdx != newSize) {
296                 // Could not finish iteration.
297                 Arrays.fill(elements, size, dstIdx, null);
298             }
299         }
300 
301         assert dstIdx == newSize : String.format("dstIdx(%d) != newSize(%d)", dstIdx, newSize);
302         size = newSize;
303 
304         return this;
305     }

314     /**
315      * Add the messages contained in the given {@link MessageList}, using the given src index and src length, to this
316      * {@link MessageList} and return itself.
317      */
318     public MessageList<T> add(MessageList<T> src, int srcIdx, int srcLen) {
319         if (src == null) {
320             throw new NullPointerException("src");
321         }
322 
323         if (srcIdx > src.size - srcLen) {
324             throw new IndexOutOfBoundsException(String.format(
325                     "srcIdx(%d) + srcLen(%d) > src.size(%d)", srcIdx, srcLen, src.size));
326         }
327 
328         modifications ++;
329 
330         final int dstIdx = size;
331         final int newSize = dstIdx + srcLen;
332         ensureCapacity(newSize);
333 
334         byteBufsOnly &= src.byteBufsOnly;
335         System.arraycopy(src.elements, srcIdx, elements, dstIdx, srcLen);
336 
337         size = newSize;
338         return this;
339     }

389     /**
390      * Iterates over the messages in this list with the specified {@code processor}.
391      *
392      * @return {@code -1} if the processor iterated to or beyond the end of the readable bytes.
393      *         If the {@code processor} raised {@link MessageListProcessor#ABORT}, the last-visited index will be
394      *         returned.
395      */
396     public int forEach(MessageListProcessor<? super T> proc) {
397         if (proc == null) {
398             throw new NullPointerException("proc");
399         }
400 
401         final int size = this.size;
402         if (size == 0) {
403             return -1;
404         }
405 
406         @SuppressWarnings("unchecked")
407         MessageListProcessor<T> p = (MessageListProcessor<T>) proc;
408 
409         int i = 0;
410         try {
411             do {
412                 i += p.process(elements[i]);
413             } while (i < size);
414         } catch (Signal abort) {
415             abort.expect(MessageListProcessor.ABORT);
416             return i;
417         } catch (Exception e) {
418             PlatformDependent.throwException(e);
419         }
420 
421         return -1;
422     }

424     /**
425      * Iterates over the messages in this list with the specified {@code processor}.
426      *
427      * @return {@code -1} if the processor iterated to or beyond the end of the specified area.
428      *         If the {@code processor} raised {@link MessageListProcessor#ABORT}, the last-visited index will be
429      *         returned.
430      */
431     public int forEach(int index, int length, MessageListProcessor<? super T> proc) {
432         checkRange(index, length);
433         if (proc == null) {
434             throw new NullPointerException("proc");
435         }
436 
437         if (size == 0) {
438             return -1;
439         }
440 
441         @SuppressWarnings("unchecked")
442         MessageListProcessor<T> p = (MessageListProcessor<T>) proc;
443 
444         final int end = index + length;
445 
446         int i = index;
447         try {
448             do {
449                 i += p.process(elements[i]);
450             } while (i < end);
451         } catch (Signal abort) {
452             abort.expect(MessageListProcessor.ABORT);
453             return i;
454         } catch (Exception e) {
455             PlatformDependent.throwException(e);
456         }
457 
458         return -1;
459     }

487     public boolean containsOnly(Class<?> clazz) {
488         if (clazz == ByteBuf.class) {
489             return byteBufsOnly;
490         }
491         for (int i = 0; i < size; i++) {
492             if (!clazz.isInstance(elements[i])) {
493                 return false;
494             }

499     private void ensureCapacity(int capacity) {
500         if (elements.length >= capacity) {
501             return;
502         }
503 
504         T[] newElements = newArray(normalizeCapacity(capacity));
505         System.arraycopy(elements, 0, newElements, 0, size);
506         elements = newElements;
507     }
508 
509     private static int normalizeCapacity(int initialCapacity) {
510         if (initialCapacity <= MIN_INITIAL_CAPACITY) {
511             initialCapacity = MIN_INITIAL_CAPACITY;
512         } else {
513             initialCapacity |= initialCapacity >>>  1;
514             initialCapacity |= initialCapacity >>>  2;
515             initialCapacity |= initialCapacity >>>  4;
516             initialCapacity |= initialCapacity >>>  8;
517             initialCapacity |= initialCapacity >>> 16;
518             initialCapacity ++;
519 
520             if (initialCapacity < 0) {
521                 initialCapacity >>>= 1;
522             }
523         }
524         return initialCapacity;
525     }

544     private final class MessageListIterator implements Iterator<T> {
545         private int index;
546         private int expectedModifications = modifications;
547 
548         private void checkConcurrentModifications() {
549             if (expectedModifications != modifications) {
550                 throw new ConcurrentModificationException();
551             }
552             if (index > size) {
553                 throw new ConcurrentModificationException();
554             }
555         }
556 
557         @Override
558         public boolean hasNext() {
559             return index < size;
560         }
561 
562         @Override
563         public T next() {
564             checkConcurrentModifications();
565             if (hasNext()) {
566                 return elements[index ++];
567             }
568             throw new NoSuchElementException();
569         }

571         @Override
572         public void remove() {
573            throw new UnsupportedOperationException();
574         }
