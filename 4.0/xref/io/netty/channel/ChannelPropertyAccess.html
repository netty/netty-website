16  package io.netty.channel;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.buffer.ByteBufAllocator;
20  
21  /**
22   * Provides common methods between {@link Channel} and {@link ChannelHandlerContext}.
23   */
24  interface ChannelPropertyAccess {
25  
26      /**
27       * Return the assigned {@link ChannelPipeline}
28       */
29      ChannelPipeline pipeline();
30  
31      /**
32       * Return the assigned {@link ByteBufAllocator} which will be used for allocate {@link ByteBuf}s.
33       */
34      ByteBufAllocator alloc();
35  
36      /**
37       * Create a new {@link ChannelPromise}
38       */
39      ChannelPromise newPromise();
40  
41      /**
42       * Create a new {@link ChannelFuture} which is marked as successes already. So {@link ChannelFuture#isSuccess()}
43       * will return {@code true}. All {@link ChannelFutureListener} added to it will be notified directly. Also
44       * every call of blocking methods will just return without blocking.
45       */
46      ChannelFuture newSucceededFuture();
47  
48      /**
49       * Create a new {@link ChannelFuture} which is marked as fakued already. So {@link ChannelFuture#isSuccess()}
50       * will return {@code false}. All {@link ChannelFutureListener} added to it will be notified directly. Also
51       * every call of blocking methods will just return without blocking.
52       */
53      ChannelFuture newFailedFuture(Throwable cause);
54  }
