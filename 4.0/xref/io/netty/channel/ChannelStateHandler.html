16  package io.netty.channel;
17  
18  /**
19   * {@link ChannelHandler} which adds callbacks for state changes. This allows the user
20   * to hook in to state changes easily.
21   */
22  public interface ChannelStateHandler extends ChannelHandler {
23  
24      /**
25       * The {@link Channel} of the {@link ChannelHandlerContext} was registered with its {@link EventLoop}
26       */
27      void channelRegistered(ChannelHandlerContext ctx) throws Exception;
28  
29      /**
30       * The {@link Channel} of the {@link ChannelHandlerContext} was unregistered from its {@link EventLoop}
31       */
32      void channelUnregistered(ChannelHandlerContext ctx) throws Exception;
33  
34      /**
35       * The {@link Channel} of the {@link ChannelHandlerContext} is now active
36       */
37      void channelActive(ChannelHandlerContext ctx) throws Exception;
38  
39      /**
40       * The {@link Channel} of the {@link ChannelHandlerContext} was registered is now inactive and reached its
41       * end of lifetime.
42       */
43      void channelInactive(ChannelHandlerContext ctx) throws Exception;
44  
45      /**
46       * Invoked when a {@link ChannelHandlerContext#read()} is finished and the inbound buffer of this handler will not
47       * be updated until another {@link ChannelHandlerContext#read()} request is issued.
48       */
49      void channelReadSuspended(ChannelHandlerContext ctx) throws Exception;
50  
51      /**
52       * The inbound buffer of the {@link ChannelHandlerContext} was updated with new data.
53       * This means something may be ready to get processed by the actual {@link ChannelStateHandler}
54       * implementation. It's up to the implementation to consume it or keep it in the buffer
55       * to wait for more data and consume it later.
56       */
57      void inboundBufferUpdated(ChannelHandlerContext ctx) throws Exception;
58  
59      /**
60       * Gets called if an user event was triggered.
61       */
62      void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception;
63  }
