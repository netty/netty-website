16  package io.netty.channel.aio;
17  
18  import io.netty.channel.Channel;
19  import io.netty.channel.EventLoop;
20  
21  import java.nio.channels.CompletionHandler;
22  
23  /**
24   * Special {@link CompletionHandler} which makes sure that the callback methods gets executed in the {@link EventLoop}
25   */
26  public abstract class AioCompletionHandler<V, A extends Channel> implements CompletionHandler<V, A> {
27  
28      /**
29       * See {@link CompletionHandler#completed(Object, Object)}
30       */
31      protected abstract void completed0(V result, A channel);
32  
33      /**
34       * Set {@link CompletionHandler#failed(Throwable, Object)}
35       */
36      protected abstract void failed0(Throwable exc, A channel);
37  
38      // According to JDK AIO documentation, the ExecutorService a user specified must not call the Runnable given by
39      // JDK AIO implementation directly.  However, we violates that rull by calling Runnable.run() directly for
40      // optimization purposes, and it can result in infinite recursion in combination with the fact that the JDK AIO
41      // implementation often makes recursive invocations.  Therefore, we must make sure we don't go too deep in the
42      // stack.
43      private static final int MAX_STACK_DEPTH = 8;
44      private static final ThreadLocal<Integer> STACK_DEPTH = new ThreadLocal<Integer>() {
45          @Override
46          protected Integer initialValue() {
47              return 0;
48          }
49      };
50  
51      @Override
52      public final void completed(final V result, final A channel) {
53          EventLoop loop = channel.eventLoop();
54          if (loop.inEventLoop()) {
55              Integer d = STACK_DEPTH.get();
56              if (d < MAX_STACK_DEPTH) {
57                  STACK_DEPTH.set(d + 1);
58                  try {
59                      completed0(result, channel);
60                  } finally {
61                      STACK_DEPTH.set(d);
62                  }
63              } else {
64                  // schedule it with a special runnable to make sure we keep the right
65                  // order and exist the recursive call to prevent stackoverflow
66                  loop.execute(new AioEventLoop.RecursionBreakingRunnable() {
67                      @Override
68                      public void run() {
69                          completed0(result, channel);
70                      }
71                  });
72              }
73          } else {
74              loop.execute(new Runnable() {
75                  @Override
76                  public void run() {
77                      completed0(result, channel);
78                  }
79              });
80          }
81      }
82  
83      @Override
84      public final void failed(final Throwable exc, final A channel) {
85          EventLoop loop = channel.eventLoop();
86          if (loop.inEventLoop()) {
87              Integer d = STACK_DEPTH.get();
88              if (d < MAX_STACK_DEPTH) {
89                  STACK_DEPTH.set(d + 1);
90                  try {
91                      failed0(exc, channel);
92                  } finally {
93                      STACK_DEPTH.set(d);
94                  }
95              } else {
96                  // schedule it with a special runnable to make sure we keep the right
97                  // order and exist the recursive call to prevent stackoverflow
98                  loop.execute(new AioEventLoop.RecursionBreakingRunnable() {
99                      @Override
100                     public void run() {
101                         failed0(exc, channel);
102                     }
103                 });
104             }
105         } else {
106             loop.execute(new Runnable() {
107                 @Override
108                 public void run() {
109                     failed0(exc, channel);
110                 }
111             });
112         }
113     }
114 }
