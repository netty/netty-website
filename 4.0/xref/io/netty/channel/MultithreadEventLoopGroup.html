16  package io.netty.channel;
17  
18  import io.netty.util.concurrent.DefaultThreadFactory;
19  import io.netty.util.concurrent.MultithreadEventExecutorGroup;
20  import io.netty.util.internal.SystemPropertyUtil;
21  import io.netty.util.internal.logging.InternalLogger;
22  import io.netty.util.internal.logging.InternalLoggerFactory;
23  
24  import java.util.concurrent.ThreadFactory;
25  
26  /**
27   * Abstract base class for {@link EventLoopGroup} implementations that handles their tasks with multiple threads at
28   * the same time.
29   */
30  public abstract class MultithreadEventLoopGroup extends MultithreadEventExecutorGroup implements EventLoopGroup {
31  
32      private static final InternalLogger logger = InternalLoggerFactory.getInstance(MultithreadEventLoopGroup.class);
33  
34      private static final int DEFAULT_EVENT_LOOP_THREADS;
35  
36      static {
37          DEFAULT_EVENT_LOOP_THREADS = Math.max(1, SystemPropertyUtil.getInt(
38                  "io.netty.eventLoopThreads", Runtime.getRuntime().availableProcessors() * 2));
39  
40          if (logger.isDebugEnabled()) {
41              logger.debug("io.netty.eventLoopThreads: {}", DEFAULT_EVENT_LOOP_THREADS);
42          }
43      }
44  
45      protected MultithreadEventLoopGroup(int nThreads, ThreadFactory threadFactory, Object... args) {
46          super(nThreads == 0? DEFAULT_EVENT_LOOP_THREADS : nThreads, threadFactory, args);
47      }
48  
49      @Override
50      protected ThreadFactory newDefaultThreadFactory() {
51          return new DefaultThreadFactory(getClass(), Thread.MAX_PRIORITY);
52      }
53  
54      @Override
55      public EventLoop next() {
56          return (EventLoop) super.next();
57      }
58  
59      @Override
60      public ChannelFuture register(Channel channel) {
61          return next().register(channel);
62      }
63  
64      @Override
65      public ChannelFuture register(Channel channel, ChannelPromise promise) {
66          return next().register(channel, promise);
67      }
68  }
