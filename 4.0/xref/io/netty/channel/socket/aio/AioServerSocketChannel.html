168     @Override

170         Runnable task = super.doRegister();
171         if (ch == null) {
172             AsynchronousServerSocketChannel channel =
173                     newSocket(((AioEventLoopGroup) eventLoop().parent()).channelGroup());
174             ch = channel;

180     private static final class AcceptHandler
181             extends AioCompletionHandler<AsynchronousSocketChannel, AioServerSocketChannel> {
182 
183         @Override
184         protected void completed0(AsynchronousSocketChannel ch, AioServerSocketChannel channel) {
185             channel.acceptInProgress = false;
186 
187             ChannelPipeline pipeline = channel.pipeline();
188             MessageBuf<Object> buffer = pipeline.inboundMessageBuffer();
189 
190             if (buffer.refCnt() == 0) {
191                 try {
192                     ch.close();
193                 } catch (IOException e) {
194                     logger.warn(
195                             "Failed to close a socket which was accepted while its server socket is being closed",
196                             e);
197                 }
198                 return;
199             }
200 
201             // create the socket add it to the buffer and fire the event
202             buffer.add(new AioSocketChannel(channel, null, ch));
203             pipeline.fireInboundBufferUpdated();
204             pipeline.fireChannelReadSuspended();
205         }
206 
207         @Override
208         protected void failed0(Throwable t, AioServerSocketChannel channel) {
209             channel.acceptInProgress = false;
210             boolean asyncClosed = false;
211             if (t instanceof AsynchronousCloseException) {
212                 asyncClosed = true;
213                 channel.closed = true;
214             }
215             // check if the exception was thrown because the channel was closed before
216             // log something
217             if (channel.isOpen() && ! asyncClosed) {
218                 logger.warn("Failed to create a new channel from an accepted socket.", t);
219                 channel.pipeline().fireExceptionCaught(t);
220             }
221         }
222     }
