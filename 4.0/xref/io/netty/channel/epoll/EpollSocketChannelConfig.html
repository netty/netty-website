59      @SuppressWarnings("unchecked")
60      @Override
61      public <T> T getOption(ChannelOption<T> option) {
62          if (option == SO_RCVBUF) {
63              return (T) Integer.valueOf(getReceiveBufferSize());
64          }
65          if (option == SO_SNDBUF) {
66              return (T) Integer.valueOf(getSendBufferSize());
67          }
68          if (option == TCP_NODELAY) {
69              return (T) Boolean.valueOf(isTcpNoDelay());
70          }
71          if (option == SO_KEEPALIVE) {
72              return (T) Boolean.valueOf(isKeepAlive());
73          }
74          if (option == SO_REUSEADDR) {
75              return (T) Boolean.valueOf(isReuseAddress());
76          }
77          if (option == SO_LINGER) {
78              return (T) Integer.valueOf(getSoLinger());
79          }
80          if (option == IP_TOS) {
81              return (T) Integer.valueOf(getTrafficClass());
82          }
83          if (option == ALLOW_HALF_CLOSURE) {
84              return (T) Boolean.valueOf(isAllowHalfClosure());
85          }
86          if (option == EpollChannelOption.TCP_CORK) {
87              return (T) Boolean.valueOf(isTcpCork());
88          }
89          if (option == EpollChannelOption.TCP_NOTSENT_LOWAT) {
90              return (T) Long.valueOf(getTcpNotSentLowAt());
91          }
92          if (option == EpollChannelOption.TCP_KEEPIDLE) {
93              return (T) Integer.valueOf(getTcpKeepIdle());
94          }
95          if (option == EpollChannelOption.TCP_KEEPINTVL) {
96              return (T) Integer.valueOf(getTcpKeepIntvl());
97          }
98          if (option == EpollChannelOption.TCP_KEEPCNT) {
99              return (T) Integer.valueOf(getTcpKeepCnt());
100         }
101         if (option == EpollChannelOption.TCP_USER_TIMEOUT) {
102             return (T) Integer.valueOf(getTcpUserTimeout());
103         }
104         if (option == EpollChannelOption.TCP_QUICKACK) {
105             return (T) Boolean.valueOf(isTcpQuickAck());
106         }
107         return super.getOption(option);
108     }

110     @Override
111     public <T> boolean setOption(ChannelOption<T> option, T value) {
112         validate(option, value);
113 
114         if (option == SO_RCVBUF) {
115             setReceiveBufferSize((Integer) value);
116         } else if (option == SO_SNDBUF) {
117             setSendBufferSize((Integer) value);
118         } else if (option == TCP_NODELAY) {
119             setTcpNoDelay((Boolean) value);
120         } else if (option == SO_KEEPALIVE) {
121             setKeepAlive((Boolean) value);
122         } else if (option == SO_REUSEADDR) {
123             setReuseAddress((Boolean) value);
124         } else if (option == SO_LINGER) {
125             setSoLinger((Integer) value);
126         } else if (option == IP_TOS) {
127             setTrafficClass((Integer) value);
128         } else if (option == ALLOW_HALF_CLOSURE) {
129             setAllowHalfClosure((Boolean) value);
130         } else if (option == EpollChannelOption.TCP_CORK) {
131             setTcpCork((Boolean) value);
132         } else if (option == EpollChannelOption.TCP_NOTSENT_LOWAT) {
133             setTcpNotSentLowAt((Long) value);
134         } else if (option == EpollChannelOption.TCP_KEEPIDLE) {
135             setTcpKeepIdle((Integer) value);
136         } else if (option == EpollChannelOption.TCP_KEEPCNT) {
137             setTcpKeepCntl((Integer) value);
138         } else if (option == EpollChannelOption.TCP_KEEPINTVL) {
139             setTcpKeepIntvl((Integer) value);
140         } else if (option == EpollChannelOption.TCP_USER_TIMEOUT) {
141             setTcpUserTimeout((Integer) value);
142         } else if (option == EpollChannelOption.TCP_MD5SIG) {
143             @SuppressWarnings("unchecked")
144             final Map<InetAddress, byte[]> m = (Map<InetAddress, byte[]>) value;
145             setTcpMd5Sig(m);
146         } else if (option == EpollChannelOption.TCP_QUICKACK) {
147             setTcpQuickAck((Boolean) value);
148         } else {
149             return super.setOption(option, value);
150         }
151 
152         return true;
153     }
