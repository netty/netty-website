62  /**
63   * An implementation of {@link SslContext} which works with libraries that support the
64   * <a href="https://www.openssl.org/">OpenSsl</a> C library API.
65   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
66   *
67   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
68   * which depends upon the instance of this class is released. Otherwise if any method of
69   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
70   */
71  public abstract class ReferenceCountedOpenSslContext extends SslContext implements ReferenceCounted {
72      private static final InternalLogger logger =
73              InternalLoggerFactory.getInstance(ReferenceCountedOpenSslContext.class);
74      /**
75       * To make it easier for users to replace JDK implemention with OpenSsl version we also use
76       * {@code jdk.tls.rejectClientInitiatedRenegotiation} to allow disabling client initiated renegotiation.
77       * Java8+ uses this system property as well.
78       * <p>
79       * See also <a href="http://blog.ivanristic.com/2014/03/ssl-tls-improvements-in-java-8.html">
80       * Significant SSL/TLS improvements in Java 8</a>
81       */
82      private static final boolean JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION =
83              AccessController.doPrivileged(new PrivilegedAction<Boolean>() {
84                  @Override
85                  public Boolean run() {
86                      return SystemPropertyUtil.getBoolean("jdk.tls.rejectClientInitiatedRenegotiation", false);
87                  }

153         // XXX: Make sure to sync this list with JdkSslEngineFactory.
154         Collections.addAll(
155                 ciphers,

201     ReferenceCountedOpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
202                                    OpenSslApplicationProtocolNegotiator apn, long sessionCacheSize,
203                                    long sessionTimeout, int mode, Certificate[] keyCertChain,
204                                    ClientAuth clientAuth, boolean startTls, boolean leakDetection) throws SSLException {
205         super(startTls);
206 
207         OpenSsl.ensureAvailability();
208 
209         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
210             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
211         }
212         leak = leakDetection ? leakDetector.track(this) : null;
213         this.mode = mode;
214         this.clientAuth = isServer() ? checkNotNull(clientAuth, "clientAuth") : ClientAuth.NONE;
215 
216         if (mode == SSL.SSL_MODE_SERVER) {
217             rejectRemoteInitiatedRenegotiation =
218                     JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION;
219         }
220         this.keyCertChain = keyCertChain == null ? null : keyCertChain.clone();
221         final List<String> convertedCiphers;
222         if (ciphers == null) {
223             convertedCiphers = null;
224         } else {
225             convertedCiphers = new ArrayList<String>();
226             for (String c : ciphers) {
227                 if (c == null) {
228                     break;
229                 }
230 
231                 String converted = CipherSuiteConverter.toOpenSsl(c);
232                 if (converted != null) {
233                     c = converted;
234                 }
235                 convertedCiphers.add(c);
236             }
237         }
238 
239         unmodifiableCiphers = Arrays.asList(checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
240                 convertedCiphers, DEFAULT_CIPHERS, OpenSsl.availableCipherSuites()));
241 
242         this.apn = checkNotNull(apn, "apn");
243 
244         // Allocate a new APR pool.
245         aprPool = Pool.create(0);
246 
247         // Create a new SSL_CTX and configure it.
248         boolean success = false;
249         try {
250             synchronized (ReferenceCountedOpenSslContext.class) {
251                 try {
252                     ctx = SSLContext.make(aprPool, SSL.SSL_PROTOCOL_ALL, mode);
253                 } catch (Exception e) {
254                     throw new SSLException("failed to create an SSL_CTX", e);
255                 }
256 
257                 SSLContext.setOptions(ctx, SSL.SSL_OP_ALL);
258                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv2);
259                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv3);
260                 SSLContext.setOptions(ctx, SSL.SSL_OP_CIPHER_SERVER_PREFERENCE);
261                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_ECDH_USE);
262                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_DH_USE);
263                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
264 
265                 // We do not support compression as the moment so we should explicitly disable it.
266                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_COMPRESSION);
267 
268                 // Disable ticket support by default to be more inline with SSLEngineImpl of the JDK.
269                 // This also let SSLSession.getId() work the same way for the JDK implementation and the OpenSSLEngine.
270                 // If tickets are supported SSLSession.getId() will only return an ID on the server-side if it could
271                 // make use of tickets.
272                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_TICKET);
273 
274                 // We need to enable SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER as the memory address may change between
275                 // calling OpenSSLEngine.wrap(...).
276                 // See https://github.com/netty/netty-tcnative/issues/100
277                 SSLContext.setMode(ctx, SSLContext.getMode(ctx) | SSL.SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
278 
279                 if (DH_KEY_LENGTH != null) {
280                     SSLContext.setTmpDHLength(ctx, DH_KEY_LENGTH);
281                 }
282 
283                 /* List the ciphers that are permitted to negotiate. */
284                 try {
285                     SSLContext.setCipherSuite(ctx, CipherSuiteConverter.toOpenSsl(unmodifiableCiphers));
286                 } catch (SSLException e) {
287                     throw e;
288                 } catch (Exception e) {
289                     throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
290                 }
291 
292                 List<String> nextProtoList = apn.protocols();
293                 /* Set next protocols for next protocol negotiation extension, if specified */
294                 if (!nextProtoList.isEmpty()) {
295                     String[] protocols = nextProtoList.toArray(new String[nextProtoList.size()]);
296                     int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
297 
298                     switch (apn.protocol()) {
299                         case NPN:
300                             SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
301                             break;
302                         case ALPN:
303                             SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
304                             break;
305                         case NPN_AND_ALPN:
306                             SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
307                             SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
308                             break;
309                         default:
310                             throw new Error();
311                     }
312                 }
313 
314                 /* Set session cache size, if specified */
315                 if (sessionCacheSize > 0) {
316                     this.sessionCacheSize = sessionCacheSize;
317                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
318                 } else {
319                     // Get the default session cache size using SSLContext.setSessionCacheSize()
320                     this.sessionCacheSize = sessionCacheSize = SSLContext.setSessionCacheSize(ctx, 20480);
321                     // Revert the session cache size to the default value.
322                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
323                 }
324 
325                 /* Set session timeout, if specified */
326                 if (sessionTimeout > 0) {
327                     this.sessionTimeout = sessionTimeout;
328                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
329                 } else {
330                     // Get the default session timeout using SSLContext.setSessionCacheTimeout()
331                     this.sessionTimeout = sessionTimeout = SSLContext.setSessionCacheTimeout(ctx, 300);
332                     // Revert the session timeout to the default value.
333                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
334                 }
335             }
336             success = true;
337         } finally {
338             if (!success) {
339                 release();
340             }
341         }
342     }

344     private static int opensslSelectorFailureBehavior(ApplicationProtocolConfig.SelectorFailureBehavior behavior) {
345         switch (behavior) {
346             case NO_ADVERTISE:

348             case CHOOSE_MY_LAST_PROTOCOL:
349                 return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
350             default:
351                 throw new Error();
352         }

451     // IMPORTANT: This method must only be called from either the constructor or the finalizer as a user MUST never
452     //            get access to an OpenSslSessionContext after this method was called to prevent the user from
453     //            producing a segfault.
454     final void destroy() {
455         synchronized (ReferenceCountedOpenSslContext.class) {
456             if (ctx != 0) {
457                 SSLContext.free(ctx);
458                 ctx = 0;
459             }
460 
461             // Guard against multiple destroyPools() calls triggered by construction exception and finalize() later
462             if (aprPool != 0) {
463                 Pool.destroy(aprPool);
464                 aprPool = 0;
465             }
466         }
467     }

495     /**
496      * Translate a {@link ApplicationProtocolConfig} object to a
497      * {@link OpenSslApplicationProtocolNegotiator} object.
498      *
499      * @param config The configuration which defines the translation
500      * @return The results of the translation
501      */
502     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
503         if (config == null) {
504             return NONE_PROTOCOL_NEGOTIATOR;
505         }
506 
507         switch (config.protocol()) {
508             case NONE:
509                 return NONE_PROTOCOL_NEGOTIATOR;
510             case ALPN:
511             case NPN:
512             case NPN_AND_ALPN:
513                 switch (config.selectedListenerFailureBehavior()) {
514                     case CHOOSE_MY_LAST_PROTOCOL:
515                     case ACCEPT:
516                         switch (config.selectorFailureBehavior()) {
517                             case CHOOSE_MY_LAST_PROTOCOL:
518                             case NO_ADVERTISE:
519                                 return new OpenSslDefaultApplicationProtocolNegotiator(
520                                         config);
521                             default:
522                                 throw new UnsupportedOperationException(
523                                         new StringBuilder("OpenSSL provider does not support ")
524                                                 .append(config.selectorFailureBehavior())
525                                                 .append(" behavior").toString());
526                         }
527                     default:
528                         throw new UnsupportedOperationException(
529                                 new StringBuilder("OpenSSL provider does not support ")
530                                         .append(config.selectedListenerFailureBehavior())
531                                         .append(" behavior").toString());
532                 }
533             default:
534                 throw new Error();
535         }
536     }

573     abstract static class AbstractCertificateVerifier implements CertificateVerifier {
574         private final OpenSslEngineMap engineMap;
575 
576         AbstractCertificateVerifier(OpenSslEngineMap engineMap) {
577             this.engineMap = engineMap;
578         }
579 
580         @Override
581         public final int verify(long ssl, byte[][] chain, String auth) {
582             X509Certificate[] peerCerts = certificates(chain);
583             final ReferenceCountedOpenSslEngine engine = engineMap.get(ssl);
584             try {
585                 verify(engine, peerCerts, auth);
586                 return CertificateVerifier.X509_V_OK;
587             } catch (Throwable cause) {
588                 logger.debug("verification of certificate failed", cause);
589                 SSLHandshakeException e = new SSLHandshakeException("General OpenSslEngine problem");
590                 e.initCause(cause);
591                 engine.handshakeException = e;
592 
593                 // Try to extract the correct error code that should be used.
594                 if (cause instanceof OpenSslCertificateException) {
595                     // This will never return a negative error code as its validated when constructing the
596                     // OpenSslCertificateException.
597                     return ((OpenSslCertificateException) cause).errorCode();
598                 }
599                 if (cause instanceof CertificateExpiredException) {
600                     return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
601                 }
602                 if (cause instanceof CertificateNotYetValidException) {
603                     return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
604                 }
605                 if (PlatformDependent.javaVersion() >= 7) {
606                     if (cause instanceof CertificateRevokedException) {
607                         return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
608                     }
609 
610                     // The X509TrustManagerImpl uses a Validator which wraps a CertPathValidatorException into
611                     // an CertificateException. So we need to handle the wrapped CertPathValidatorException to be
612                     // able to send the correct alert.
613                     Throwable wrapped = cause.getCause();
614                     while (wrapped != null) {
615                         if (wrapped instanceof CertPathValidatorException) {
616                             CertPathValidatorException ex = (CertPathValidatorException) wrapped;
617                             CertPathValidatorException.Reason reason = ex.getReason();
618                             if (reason == CertPathValidatorException.BasicReason.EXPIRED) {
619                                 return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
620                             }
621                             if (reason == CertPathValidatorException.BasicReason.NOT_YET_VALID) {
622                                 return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
623                             }
624                             if (reason == CertPathValidatorException.BasicReason.REVOKED) {
625                                 return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
626                             }
627                         }
628                         wrapped = wrapped.getCause();
629                     }
630                 }
631 
632                 // Could not detect a specific error code to use, so fallback to a default code.
633                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
634             }
635         }
636 
637         abstract void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts,
638                              String auth) throws Exception;
639     }

660     static void setKeyMaterial(long ctx, X509Certificate[] keyCertChain, PrivateKey key, String keyPassword)
661             throws SSLException {
662          /* Load the certificate file and private key. */
663         long keyBio = 0;
664         long keyCertChainBio = 0;
665         long keyCertChainBio2 = 0;
666         PemEncoded encoded = null;
667         try {
668             // Only encode one time
669             encoded = PemX509Certificate.toPEM(ByteBufAllocator.DEFAULT, true, keyCertChain);
670             keyCertChainBio = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
671             keyCertChainBio2 = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
672 
673             if (key != null) {
674                 keyBio = toBIO(key);
675             }
676 
677             SSLContext.setCertificateBio(
678                     ctx, keyCertChainBio, keyBio,
679                     keyPassword == null ? StringUtil.EMPTY_STRING : keyPassword);
680             // We may have more then one cert in the chain so add all of them now.
681             SSLContext.setCertificateChainBio(ctx, keyCertChainBio2, true);
682         } catch (SSLException e) {
683             throw e;
684         } catch (Exception e) {
685             throw new SSLException("failed to set certificate and key", e);
686         } finally {
687             freeBio(keyBio);
688             freeBio(keyCertChainBio);
689             freeBio(keyCertChainBio2);
690             if (encoded != null) {
691                 encoded.release();
692             }
693         }
694     }

742     static long toBIO(ByteBufAllocator allocator, PemEncoded pem) throws Exception {
743         try {
744             // We can turn direct buffers straight into BIOs. No need to
745             // make a yet another copy.
746             ByteBuf content = pem.content();
747 
748             if (content.isDirect()) {
749                 return newBIO(content.slice().retain());
750             }
751 
752             ByteBuf buffer = allocator.directBuffer(content.readableBytes());
753             try {
754                 buffer.writeBytes(content, content.readerIndex(), content.readableBytes());
755                 return newBIO(buffer.slice().retain());
756             } finally {
757                 try {
758                     // If the contents of the ByteBuf is sensitive (e.g. a PrivateKey) we
759                     // need to zero out the bytes of the copy before we're releasing it.
760                     if (pem.isSensitive()) {
761                         SslUtils.zeroout(buffer);
762                     }
763                 } finally {
764                     buffer.release();
765                 }
766             }
767         } finally {
768             pem.release();
769         }
770     }

772     private static long newBIO(ByteBuf buffer) throws Exception {
773         try {
774             long bio = SSL.newMemBIO();
775             int readable = buffer.readableBytes();
776             if (SSL.writeToBIO(bio, OpenSsl.memoryAddress(buffer) + buffer.readerIndex(), readable) != readable) {
777                 SSL.freeBIO(bio);
778                 throw new IllegalStateException("Could not write data to memory BIO");
779             }
780             return bio;
781         } finally {
782             buffer.release();
783         }
