19  import io.netty.channel.ChannelHandlerContext;

21  /**
22   * Constants for SSL packets.
23   */
24  final class SslUtils {
25  
26      /**
27       * change cipher spec
28       */
29      public static final int SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC = 20;
30  
31      /**
32       * alert
33       */
34      public static final int SSL_CONTENT_TYPE_ALERT = 21;
35  
36      /**
37       * handshake
38       */
39      public static final int SSL_CONTENT_TYPE_HANDSHAKE = 22;
40  
41      /**
42       * application data
43       */
44      public static final int SSL_CONTENT_TYPE_APPLICATION_DATA = 23;
45  
46      /**
47       * the length of the ssl record header (in bytes)
48       */
49      public static final int SSL_RECORD_HEADER_LENGTH = 5;
50  
51      /**
52       * Return how much bytes can be read out of the encrypted data. Be aware that this method will not increase
53       * the readerIndex of the given {@link ByteBuf}.
54       *
55       * @param   buffer
56       *                  The {@link ByteBuf} to read from. Be aware that it must have at least
57       *                  {@link #SSL_RECORD_HEADER_LENGTH} bytes to read,
58       *                  otherwise it will throw an {@link IllegalArgumentException}.
59       * @return length
60       *                  The length of the encrypted packet that is included in the buffer. This will
61       *                  return {@code -1} if the given {@link ByteBuf} is not encrypted at all.
62       * @throws IllegalArgumentException
63       *                  Is thrown if the given {@link ByteBuf} has not at least {@link #SSL_RECORD_HEADER_LENGTH}
64       *                  bytes to read.
65       */
66      static int getEncryptedPacketLength(ByteBuf buffer, int offset) {
67          int packetLength = 0;
68  
69          // SSLv3 or TLS - Check ContentType
70          boolean tls;
71          switch (buffer.getUnsignedByte(offset)) {
72              case SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC:
73              case SSL_CONTENT_TYPE_ALERT:
74              case SSL_CONTENT_TYPE_HANDSHAKE:
75              case SSL_CONTENT_TYPE_APPLICATION_DATA:
76                  tls = true;
77                  break;
78              default:
79                  // SSLv2 or bad data
80                  tls = false;
81          }
82  
83          if (tls) {
84              // SSLv3 or TLS - Check ProtocolVersion
85              int majorVersion = buffer.getUnsignedByte(offset + 1);
86              if (majorVersion == 3) {
87                  // SSLv3 or TLS
88                  packetLength = buffer.getUnsignedShort(offset + 3) + SSL_RECORD_HEADER_LENGTH;
89                  if (packetLength <= SSL_RECORD_HEADER_LENGTH) {
90                      // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
91                      tls = false;
92                  }
93              } else {
94                  // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
95                  tls = false;
96              }
97          }
98  
99          if (!tls) {
100             // SSLv2 or bad data - Check the version
101             int headerLength = (buffer.getUnsignedByte(offset) & 0x80) != 0 ? 2 : 3;
102             int majorVersion = buffer.getUnsignedByte(offset + headerLength + 1);
103             if (majorVersion == 2 || majorVersion == 3) {
104                 // SSLv2
105                 if (headerLength == 2) {
106                     packetLength = (buffer.getShort(offset) & 0x7FFF) + 2;
107                 } else {
108                     packetLength = (buffer.getShort(offset) & 0x3FFF) + 3;
109                 }
110                 if (packetLength <= headerLength) {
111                     return -1;
112                 }
113             } else {
114                 return -1;
115             }
116         }
117         return packetLength;
118     }

120     static void notifyHandshakeFailure(ChannelHandlerContext ctx, Throwable cause) {
121         // We have may haven written some parts of data before an exception was thrown so ensure we always flush.
122         // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
123         ctx.flush();
124         ctx.fireUserEventTriggered(new SslHandshakeCompletionEvent(cause));
125         ctx.close();
126     }
