32  /**
33   * A server-side {@link SslContext} which uses OpenSSL's SSL/TLS implementation.

35  public final class OpenSslServerContext extends SslContext {

37      private static final InternalLogger logger = InternalLoggerFactory.getInstance(OpenSslServerContext.class);
38      private static final List<String> DEFAULT_CIPHERS;
39  
40      static {
41          List<String> ciphers = new ArrayList<String>();
42          // XXX: Make sure to sync this list with JdkSslEngineFactory.
43          Collections.addAll(
44                  ciphers,
45                  "ECDHE-RSA-AES128-GCM-SHA256",
46                  "ECDHE-RSA-RC4-SHA",
47                  "ECDHE-RSA-AES128-SHA",
48                  "ECDHE-RSA-AES256-SHA",
49                  "AES128-GCM-SHA256",
50                  "RC4-SHA",
51                  "RC4-MD5",
52                  "AES128-SHA",
53                  "AES256-SHA",
54                  "DES-CBC3-SHA");
55          DEFAULT_CIPHERS = Collections.unmodifiableList(ciphers);
56  
57          if (logger.isDebugEnabled()) {
58              logger.debug("Default cipher suite (OpenSSL): " + ciphers);
59          }
60      }

65      private final List<String> unmodifiableCiphers = Collections.unmodifiableList(ciphers);

96      /**
97       * Creates a new instance.
98       *
99       * @param certChainFile an X.509 certificate chain file in PEM format
100      * @param keyFile a PKCS#8 private key file in PEM format
101      * @param keyPassword the password of the {@code keyFile}.
102      *                    {@code null} if it's not password-protected.
103      * @param ciphers the cipher suites to enable, in the order of preference.
104      *                {@code null} to use the default cipher suites.
105      * @param nextProtocols the application layer protocols to accept, in the order of preference.
106      *                      {@code null} to disable TLS NPN/ALPN extension.
107      * @param sessionCacheSize the size of the cache used for storing SSL session objects.
108      *                         {@code 0} to use the default value.
109      * @param sessionTimeout the timeout for the cached SSL session objects, in seconds.
110      *                       {@code 0} to use the default value.
111      */
112     public OpenSslServerContext(
113             File certChainFile, File keyFile, String keyPassword,
114             Iterable<String> ciphers, Iterable<String> nextProtocols,
115             long sessionCacheSize, long sessionTimeout) throws SSLException {
116 
117         OpenSsl.ensureAvailability();
118 
119         if (certChainFile == null) {
120             throw new NullPointerException("certChainFile");
121         }
122         if (!certChainFile.isFile()) {
123             throw new IllegalArgumentException("certChainFile is not a file: " + certChainFile);
124         }
125         if (keyFile == null) {
126             throw new NullPointerException("keyPath");
127         }
128         if (!keyFile.isFile()) {
129             throw new IllegalArgumentException("keyPath is not a file: " + keyFile);
130         }
131         if (ciphers == null) {
132             ciphers = DEFAULT_CIPHERS;
133         }
134 
135         if (keyPassword == null) {
136             keyPassword = "";
137         }
138         if (nextProtocols == null) {
139             nextProtocols = Collections.emptyList();
140         }
141 
142         for (String c: ciphers) {
143             if (c == null) {
144                 break;
145             }
146             this.ciphers.add(c);
147         }
148 
149         List<String> nextProtoList = new ArrayList<String>();
150         for (String p: nextProtocols) {
151             if (p == null) {
152                 break;
153             }
154             nextProtoList.add(p);
155         }
156         this.nextProtocols = Collections.unmodifiableList(nextProtoList);
157 
158         // Allocate a new APR pool.
159         aprPool = Pool.create(0);
160 
161         // Create a new SSL_CTX and configure it.
162         boolean success = false;
163         try {
164             synchronized (OpenSslServerContext.class) {
165                 try {
166                     ctx = SSLContext.make(aprPool, SSL.SSL_PROTOCOL_ALL, SSL.SSL_MODE_SERVER);
167                 } catch (Exception e) {
168                     throw new SSLException("failed to create an SSL_CTX", e);
169                 }
170 
171                 SSLContext.setOptions(ctx, SSL.SSL_OP_ALL);
172                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv2);
173                 SSLContext.setOptions(ctx, SSL.SSL_OP_CIPHER_SERVER_PREFERENCE);
174                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_ECDH_USE);
175                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_DH_USE);
176                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
177 
178                 /* List the ciphers that the client is permitted to negotiate. */
179                 try {
180                     // Convert the cipher list into a colon-separated string.
181                     StringBuilder cipherBuf = new StringBuilder();
182                     for (String c: this.ciphers) {
183                         cipherBuf.append(c);
184                         cipherBuf.append(':');
185                     }
186                     cipherBuf.setLength(cipherBuf.length() - 1);
187 
188                     SSLContext.setCipherSuite(ctx, cipherBuf.toString());
189                 } catch (SSLException e) {
190                     throw e;
191                 } catch (Exception e) {
192                     throw new SSLException("failed to set cipher suite: " + this.ciphers, e);
193                 }
194 
195                 /* Set certificate verification policy. */
196                 SSLContext.setVerify(ctx, SSL.SSL_CVERIFY_NONE, 10);
197 
198                 /* Load the certificate file and private key. */
199                 try {
200                     if (!SSLContext.setCertificate(
201                             ctx, certChainFile.getPath(), keyFile.getPath(), keyPassword, SSL.SSL_AIDX_RSA)) {
202                         throw new SSLException("failed to set certificate: " +
203                                 certChainFile + " and " + keyFile + " (" + SSL.getLastError() + ')');
204                     }
205                 } catch (SSLException e) {
206                     throw e;
207                 } catch (Exception e) {
208                     throw new SSLException("failed to set certificate: " + certChainFile + " and " + keyFile, e);
209                 }
210 
211                 /* Load the certificate chain. We must skip the first cert since it was loaded above. */
212                 if (!SSLContext.setCertificateChainFile(ctx, certChainFile.getPath(), true)) {
213                     String error = SSL.getLastError();
214                     if (!error.startsWith(OpenSsl.IGNORABLE_ERROR_PREFIX)) {
215                         throw new SSLException(
216                                 "failed to set certificate chain: " + certChainFile + " (" + SSL.getLastError() + ')');
217                     }
218                 }
219 
220                 /* Set next protocols for next protocol negotiation extension, if specified */
221                 if (!nextProtoList.isEmpty()) {
222                     // Convert the protocol list into a comma-separated string.
223                     StringBuilder nextProtocolBuf = new StringBuilder();
224                     for (String p: nextProtoList) {
225                         nextProtocolBuf.append(p);
226                         nextProtocolBuf.append(',');
227                     }
228                     nextProtocolBuf.setLength(nextProtocolBuf.length() - 1);
229 
230                     SSLContext.setNextProtos(ctx, nextProtocolBuf.toString());
231                 }
232 
233                 /* Set session cache size, if specified */
234                 if (sessionCacheSize > 0) {
235                     this.sessionCacheSize = sessionCacheSize;
236                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
237                 } else {
238                     // Get the default session cache size using SSLContext.setSessionCacheSize()
239                     this.sessionCacheSize = sessionCacheSize = SSLContext.setSessionCacheSize(ctx, 20480);
240                     // Revert the session cache size to the default value.
241                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
242                 }
243 
244                 /* Set session timeout, if specified */
245                 if (sessionTimeout > 0) {
246                     this.sessionTimeout = sessionTimeout;
247                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
248                 } else {
249                     // Get the default session timeout using SSLContext.setSessionCacheTimeout()
250                     this.sessionTimeout = sessionTimeout = SSLContext.setSessionCacheTimeout(ctx, 300);
251                     // Revert the session timeout to the default value.
252                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
253                 }
254             }
255             success = true;
256         } finally {
257             if (!success) {
258                 destroyPools();
259             }
260         }
261 
262         stats = new OpenSslSessionStats(ctx);
263     }

304     /**
305      * Returns a new server-side {@link javax.net.ssl.SSLEngine} with the current configuration.
306      */
307     @Override
308     public SSLEngine newEngine(ByteBufAllocator alloc) {
309         if (nextProtocols.isEmpty()) {
310             return new OpenSslEngine(ctx, alloc, null);
311         } else {
312             return new OpenSslEngine(ctx, alloc, nextProtocols.get(nextProtocols.size() - 1));
313         }
314     }

331     @Override

333     protected void finalize() throws Throwable {
334         super.finalize();
335         synchronized (OpenSslServerContext.class) {
336             if (ctx != 0) {
337                 SSLContext.free(ctx);
338             }
339         }
