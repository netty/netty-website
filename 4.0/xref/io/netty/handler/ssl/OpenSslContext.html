124     OpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
125                    OpenSslApplicationProtocolNegotiator apn, long sessionCacheSize,
126                    long sessionTimeout, int mode) throws SSLException {
127         OpenSsl.ensureAvailability();
128 
129         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
130             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
131         }
132         this.mode = mode;
133 
134         if (mode == SSL.SSL_MODE_SERVER) {
135             rejectRemoteInitiatedRenegotiation =
136                     JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION;
137         }
138         final List<String> convertedCiphers;
139         if (ciphers == null) {
140             convertedCiphers = null;
141         } else {
142             convertedCiphers = new ArrayList<String>();
143             for (String c: ciphers) {
144                 if (c == null) {
145                     break;
146                 }
147 
148                 String converted = CipherSuiteConverter.toOpenSsl(c);
149                 if (converted != null) {
150                     c = converted;
151                 }
152                 convertedCiphers.add(c);
153             }
154         }
155 
156         unmodifiableCiphers = Arrays.asList(checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
157                 convertedCiphers, DEFAULT_CIPHERS, OpenSsl.availableCipherSuites()));
158 
159         this.apn = checkNotNull(apn, "apn");
160 
161         // Allocate a new APR pool.
162         aprPool = Pool.create(0);
163 
164         // Create a new SSL_CTX and configure it.
165         boolean success = false;
166         try {
167             synchronized (OpenSslContext.class) {
168                 try {
169                     ctx = SSLContext.make(aprPool, SSL.SSL_PROTOCOL_ALL, mode);
170                 } catch (Exception e) {
171                     throw new SSLException("failed to create an SSL_CTX", e);
172                 }
173 
174                 SSLContext.setOptions(ctx, SSL.SSL_OP_ALL);
175                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv2);
176                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv3);
177                 SSLContext.setOptions(ctx, SSL.SSL_OP_CIPHER_SERVER_PREFERENCE);
178                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_ECDH_USE);
179                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_DH_USE);
180                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
181 
182                 /* List the ciphers that are permitted to negotiate. */
183                 try {
184                     SSLContext.setCipherSuite(ctx, CipherSuiteConverter.toOpenSsl(unmodifiableCiphers));
185                 } catch (SSLException e) {
186                     throw e;
187                 } catch (Exception e) {
188                     throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
189                 }
190 
191                 List<String> nextProtoList = apn.protocols();
192                 /* Set next protocols for next protocol negotiation extension, if specified */
193                 if (!nextProtoList.isEmpty()) {
194                     String[] protocols = nextProtoList.toArray(new String[nextProtoList.size()]);
195                     int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
196 
197                     switch (apn.protocol()) {
198                     case NPN:
199                         SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
200                         break;
201                     case ALPN:
202                         SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
203                         break;
204                     case NPN_AND_ALPN:
205                         SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
206                         SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
207                         break;
208                     default:
209                         throw new Error();
210                     }
211                 }
212 
213                 /* Set session cache size, if specified */
214                 if (sessionCacheSize > 0) {
215                     this.sessionCacheSize = sessionCacheSize;
216                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
217                 } else {
218                     // Get the default session cache size using SSLContext.setSessionCacheSize()
219                     this.sessionCacheSize = sessionCacheSize = SSLContext.setSessionCacheSize(ctx, 20480);
220                     // Revert the session cache size to the default value.
221                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
222                 }
223 
224                 /* Set session timeout, if specified */
225                 if (sessionTimeout > 0) {
226                     this.sessionTimeout = sessionTimeout;
227                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
228                 } else {
229                     // Get the default session timeout using SSLContext.setSessionCacheTimeout()
230                     this.sessionTimeout = sessionTimeout = SSLContext.setSessionCacheTimeout(ctx, 300);
231                     // Revert the session timeout to the default value.
232                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
233                 }
234             }
235             success = true;
236         } finally {
237             if (!success) {
238                 destroy();
239             }
240         }
241     }

243     private static int opensslSelectorFailureBehavior(SelectorFailureBehavior behavior) {
244         switch (behavior) {
245         case NO_ADVERTISE:
246             return SSL.SSL_SELECTOR_FAILURE_NO_ADVERTISE;
247         case CHOOSE_MY_LAST_PROTOCOL:
248             return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
249         default:
250             throw new Error();
251         }

352     protected final void destroy() {
353         synchronized (OpenSslContext.class) {
354             if (ctx != 0) {
355                 SSLContext.free(ctx);
356                 ctx = 0;
357             }
358 
359             // Guard against multiple destroyPools() calls triggered by construction exception and finalize() later
360             if (aprPool != 0) {
361                 Pool.destroy(aprPool);
362                 aprPool = 0;
363             }
364         }
365     }

367     protected static X509Certificate[] certificates(byte[][] chain) {
368         X509Certificate[] peerCerts = new X509Certificate[chain.length];
369         for (int i = 0; i < peerCerts.length; i++) {
370             peerCerts[i] = new OpenSslX509Certificate(chain[i]);
371         }
372         return peerCerts;

384     /**
385      * Translate a {@link ApplicationProtocolConfig} object to a
386      * {@link OpenSslApplicationProtocolNegotiator} object.
387      * @param config The configuration which defines the translation
388      * @return The results of the translation
389      */
390     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
391         if (config == null) {
392             return NONE_PROTOCOL_NEGOTIATOR;
393         }
394 
395         switch (config.protocol()) {
396         case NONE:
397             return NONE_PROTOCOL_NEGOTIATOR;
398         case ALPN:
399         case NPN:
400         case NPN_AND_ALPN:
401             switch (config.selectedListenerFailureBehavior()) {
402             case CHOOSE_MY_LAST_PROTOCOL:
403             case ACCEPT:
404                 switch (config.selectorFailureBehavior()) {
405                 case CHOOSE_MY_LAST_PROTOCOL:
406                 case NO_ADVERTISE:
407                     return new OpenSslDefaultApplicationProtocolNegotiator(
408                             config);
409                 default:
410                     throw new UnsupportedOperationException(
411                             new StringBuilder("OpenSSL provider does not support ")
412                                     .append(config.selectorFailureBehavior())
413                                     .append(" behavior").toString());
414                 }
415             default:
416                 throw new UnsupportedOperationException(
417                         new StringBuilder("OpenSSL provider does not support ")
418                                 .append(config.selectedListenerFailureBehavior())
419                                 .append(" behavior").toString());
420             }
421         default:
422             throw new Error();
423         }
424     }

430     abstract class AbstractCertificateVerifier implements CertificateVerifier {
431         @Override
432         public final boolean verify(long ssl, byte[][] chain, String auth) {
433             X509Certificate[] peerCerts = certificates(chain);
434             final OpenSslEngine engine = engineMap.remove(ssl);
435             try {
436                 verify(engine, peerCerts, auth);
437                 return true;
438             } catch (Throwable cause) {
439                 logger.debug("verification of certificate failed", cause);
440                 SSLHandshakeException e = new SSLHandshakeException("General OpenSslEngine problem");
441                 e.initCause(cause);
442                 engine.handshakeException = e;
443             }
444             return false;
445         }
446 
447         abstract void verify(OpenSslEngine engine, X509Certificate[] peerCerts, String auth) throws Exception;
448     }
