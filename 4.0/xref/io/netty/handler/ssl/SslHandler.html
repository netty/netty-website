64  /**
65   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
66   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
67   * to the <strong>"SecureChat"</strong> example in the distribution or the web
68   * site for the detailed usage.
69   *
70   * <h3>Beginning the handshake</h3>
71   * <p>
72   * You must make sure not to write a message while the handshake is in progress unless you are
73   * renegotiating.  You will be notified by the {@link Future} which is
74   * returned by the {@link #handshakeFuture()} method when the handshake
75   * process succeeds or fails.
76   * <p>
77   * Beside using the handshake {@link ChannelFuture} to get notified about the completation of the handshake it's
78   * also possible to detect it by implement the
79   * {@link ChannelInboundHandler#userEventTriggered(ChannelHandlerContext, Object)}
80   * method and check for a {@link SslHandshakeCompletionEvent}.
81   *
82   * <h3>Handshake</h3>
83   * <p>
84   * The handshake will be automaticly issued for you once the {@link Channel} is active and
85   * {@link SSLEngine#getUseClientMode()} returns {@code true}.
86   * So no need to bother with it by your self.
87   *
88   * <h3>Closing the session</h3>
89   * <p>
90   * To close the SSL session, the {@link #close()} method should be
91   * called to send the {@code close_notify} message to the remote peer.  One
92   * exception is when you close the {@link Channel} - {@link SslHandler}
93   * intercepts the close request and send the {@code close_notify} message
94   * before the channel closure automatically.  Once the SSL session is closed,
95   * it is not reusable, and consequently you should create a new
96   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
97   * following section.
98   *
99   * <h3>Restarting the session</h3>
100  * <p>
101  * To restart the SSL session, you must remove the existing closed
102  * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
103  * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
104  * and start the handshake process as described in the first section.
105  *
106  * <h3>Implementing StartTLS</h3>
107  * <p>
108  * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
109  * communication pattern that secures the wire in the middle of the plaintext
110  * connection.  Please note that it is different from SSL &middot; TLS, that
111  * secures the wire from the beginning of the connection.  Typically, StartTLS
112  * is composed of three steps:
113  * <ol>
114  * <li>Client sends a StartTLS request to server.</li>
115  * <li>Server sends a StartTLS response to client.</li>
116  * <li>Client begins SSL handshake.</li>
117  * </ol>
118  * If you implement a server, you need to:
119  * <ol>
120  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
121  *     to {@code true},</li>
122  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
123  * <li>write a StartTLS response.</li>
124  * </ol>
125  * Please note that you must insert {@link SslHandler} <em>before</em> sending
126  * the StartTLS response.  Otherwise the client can send begin SSL handshake
127  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
128  * data corruption.
129  * <p>
130  * The client-side implementation is much simpler.
131  * <ol>
132  * <li>Write a StartTLS request,</li>
133  * <li>wait for the StartTLS response,</li>
134  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
135  *     to {@code false},</li>
136  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
137  * <li>Initiate SSL handshake.</li>
138  * </ol>
139  *
140  * <h3>Known issues</h3>
141  * <p>
142  * Because of a known issue with the current implementation of the SslEngine that comes
143  * with Java it may be possible that you see blocked IO-Threads while a full GC is done.
144  * <p>
145  * So if you are affected you can workaround this problem by adjust the cache settings
146  * like shown below:
147  *
148  * <pre>
149  *     SslContext context = ...;
150  *     context.getServerSessionContext().setSessionCacheSize(someSaneSize);
151  *     context.getServerSessionContext().setSessionTime(someSameTimeout);
152  * </pre>
153  * <p>
154  * What values to use here depends on the nature of your application and should be set
155  * based on monitoring and debugging of it.
156  * For more details see
157  * <a href="https://github.com/netty/netty/issues/832">#832</a> in our issue tracker.
158  */

315         final ChannelHandlerContext ctx = this.ctx;
316         ctx.executor().execute(new Runnable() {

320                 try {

323                 } catch (Exception e) {

327                 }
328             }

400     @Override
401     public void flush(ChannelHandlerContext ctx) throws Exception {
402         // Do not encrypt the first write request if this handler is
403         // created with startTLS flag turned on.
404         if (startTls && !sentFirstMessage) {
405             sentFirstMessage = true;
406             for (;;) {
407                 PendingWrite pendingWrite = pendingUnencryptedWrites.poll();
408                 if (pendingWrite == null) {
409                     break;
410                 }
411                 ctx.write(pendingWrite.msg(), (ChannelPromise) pendingWrite.recycleAndGet());
412             }
413             ctx.flush();
414             return;
415         }
416         if (pendingUnencryptedWrites.isEmpty()) {
417             pendingUnencryptedWrites.add(PendingWrite.newInstance(Unpooled.EMPTY_BUFFER, null));
418         }
419         wrap(ctx, false);
420         ctx.flush();
421     }

423     private void wrap(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
424         ByteBuf out = null;
425         ChannelPromise promise = null;
426         try {
427             for (;;) {
428                 PendingWrite pending = pendingUnencryptedWrites.peek();
429                 if (pending == null) {
430                     break;
431                 }
432                 if (out == null) {
433                     out = ctx.alloc().buffer(maxPacketBufferSize);
434                 }
435 
436                 if (!(pending.msg() instanceof ByteBuf)) {
437                     ctx.write(pending.msg(), (ChannelPromise) pending.recycleAndGet());
438                     pendingUnencryptedWrites.remove();
439                     continue;
440                 }
441 
442                 ByteBuf buf = (ByteBuf) pending.msg();
443                 SSLEngineResult result = wrap(engine, buf, out);
444 
445                 if (!buf.isReadable()) {
446                     buf.release();
447                     promise = (ChannelPromise) pending.recycleAndGet();
448                     pendingUnencryptedWrites.remove();
449                 } else {
450                     promise = null;
451                 }
452 
453                 if (result.getStatus() == Status.CLOSED) {
454                     // SSLEngine has been closed already.
455                     // Any further write attempts should be denied.
456                     for (;;) {
457                         PendingWrite w = pendingUnencryptedWrites.poll();
458                         if (w == null) {
459                             break;
460                         }
461                         w.failAndRecycle(SSLENGINE_CLOSED);
462                     }
463                     return;
464                 } else {
465                     switch (result.getHandshakeStatus()) {
466                         case NEED_TASK:
467                             runDelegatedTasks();
468                             break;
469                         case FINISHED:
470                             setHandshakeSuccess();
471                             // deliberate fall-through
472                         case NOT_HANDSHAKING:
473                         case NEED_WRAP:
474                             finishWrap(ctx, out, promise, inUnwrap);
475                             promise = null;
476                             out = null;
477                             break;
478                         case NEED_UNWRAP:
479                             return;
480                         default:
481                             throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
482                     }
483                 }
484             }
485         } catch (SSLException e) {
486             setHandshakeFailure(e);
487             throw e;
488         } finally {
489             finishWrap(ctx, out, promise, inUnwrap);
490         }
491     }
492 
493     private void finishWrap(ChannelHandlerContext ctx, ByteBuf out, ChannelPromise promise, boolean inUnwrap) {
494         if (out == null) {
495             out = Unpooled.EMPTY_BUFFER;
496         } else if (!out.isReadable()) {
497             out.release();
498             out = Unpooled.EMPTY_BUFFER;
499         }
500 
501         if (promise != null) {
502             ctx.write(out, promise);
503         } else {
504             ctx.write(out);
505         }
506 
507         if (inUnwrap) {
508             needsFlush = true;
509         }
510     }

512     private void wrapNonAppData(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
513         ByteBuf out = null;
514         try {
515             for (;;) {
516                 if (out == null) {
517                     out = ctx.alloc().buffer(maxPacketBufferSize);
518                 }
519                 SSLEngineResult result = wrap(engine, Unpooled.EMPTY_BUFFER, out);
520 
521                 if (result.bytesProduced() > 0) {
522                     ctx.write(out);
523                     if (inUnwrap) {
524                         needsFlush = true;
525                     }
526                     out = null;
527                 }
528 
529                 switch (result.getHandshakeStatus()) {
530                     case FINISHED:
531                         setHandshakeSuccess();
532                         break;
533                     case NEED_TASK:
534                         runDelegatedTasks();
535                         break;
536                     case NEED_UNWRAP:
537                         if (!inUnwrap) {
538                             unwrap(ctx);
539                         }
540                         break;
541                     case NEED_WRAP:
542                         break;
543                     case NOT_HANDSHAKING:
544                         // Workaround for TLS False Start problem reported at:
545                         // https://github.com/netty/netty/issues/1108#issuecomment-14266970
546                         if (!inUnwrap) {
547                             unwrap(ctx);
548                         }
549                         break;
550                     default:
551                         throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
552                 }
553 
554                 if (result.bytesProduced() == 0) {
555                     break;
556                 }
557             }
558         } catch (SSLException e) {
559             setHandshakeFailure(e);
560             throw e;
561         }  finally {
562             if (out != null) {
563                 out.release();
564             }
565         }
566     }

568     private SSLEngineResult wrap(SSLEngine engine, ByteBuf in, ByteBuf out) throws SSLException {
569         ByteBuffer in0 = in.nioBuffer();
570         for (;;) {
571             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
572             SSLEngineResult result = engine.wrap(in0, out0);
573             in.skipBytes(result.bytesConsumed());
574             out.writerIndex(out.writerIndex() + result.bytesProduced());
575 
576             switch (result.getStatus()) {
577                 case BUFFER_OVERFLOW:
578                     out.ensureWritable(maxPacketBufferSize);
579                     break;
580                 default:
581                     return result;
582             }
583         }
584     }

594     @Override
595     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
596         if (ignoreException(cause)) {
597             // It is safe to ignore the 'connection reset by peer' or
598             // 'broken pipe' error after sending close_notify.
599             if (logger.isDebugEnabled()) {
600                 logger.debug(
601                         "Swallowing a harmless 'connection reset by peer / broken pipe' error that occurred " +
602                                 "while writing close_notify in response to the peer's close_notify", cause);
603             }
604 
605             // Close the connection explicitly just in case the transport
606             // did not close the connection automatically.
607             if (ctx.channel().isActive()) {
608                 ctx.close();
609             }
610         } else {
611             ctx.fireExceptionCaught(cause);
612         }
613     }

615     /**
616      * Checks if the given {@link Throwable} can be ignore and just "swallowed"
617      *
618      * When an ssl connection is closed a close_notify message is sent.
619      * After that the peer also sends close_notify however, it's not mandatory to receive
620      * the close_notify. The party who sent the initial close_notify can close the connection immediately
621      * then the peer will get connection reset error.
622      *
623      */
624     private boolean ignoreException(Throwable t) {
625         if (!(t instanceof SSLException) && t instanceof IOException && sslCloseFuture.isDone()) {
626             String message = String.valueOf(t.getMessage()).toLowerCase();
627 
628             // first try to match connection reset / broke peer based on the regex. This is the fastest way
629             // but may fail on different jdk impls or OS's
630             if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
631                 return true;
632             }
633 
634             // Inspect the StackTraceElements to see if it was a connection reset / broken pipe or not
635             StackTraceElement[] elements = t.getStackTrace();
636             for (StackTraceElement element: elements) {
637                 String classname = element.getClassName();
638                 String methodname = element.getMethodName();
639 
640                 // skip all classes that belong to the io.netty package
641                 if (classname.startsWith("io.netty.")) {
642                     continue;
643                 }
644 
645                 // check if the method name is read if not skip it
646                 if (!"read".equals(methodname)) {
647                     continue;
648                 }
649 
650                 // This will also match against SocketInputStream which is used by openjdk 7 and maybe
651                 // also others
652                 if (IGNORABLE_CLASS_IN_STACK.matcher(classname).matches()) {
653                     return true;
654                 }
655 
656                 try {
657                     // No match by now.. Try to load the class via classloader and inspect it.
658                     // This is mainly done as other JDK implementations may differ in name of
659                     // the impl.
660                     Class<?> clazz = getClass().getClassLoader().loadClass(classname);
661 
662                     if (SocketChannel.class.isAssignableFrom(clazz)
663                             || DatagramChannel.class.isAssignableFrom(clazz)) {
664                         return true;
665                     }
666 
667                     // also match against SctpChannel via String matching as it may not present.
668                     if (PlatformDependent.javaVersion() >= 7
669                             && "com.sun.nio.sctp.SctpChannel".equals(clazz.getSuperclass().getName())) {
670                         return true;
671                     }
672                 } catch (ClassNotFoundException e) {
673                     // This should not happen just ignore
674                 }
675             }
676         }
677 
678         return false;
679     }

681     /**
682      * Returns {@code true} if the given {@link ByteBuf} is encrypted. Be aware that this method
683      * will not increase the readerIndex of the given {@link ByteBuf}.
684      *
685      * @param   buffer
686      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
687      *                  otherwise it will throw an {@link IllegalArgumentException}.

700     /**
701      * Return how much bytes can be read out of the encrypted data. Be aware that this method will not increase
702      * the readerIndex of the given {@link ByteBuf}.
703      *
704      * @param   buffer
705      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
706      *                  otherwise it will throw an {@link IllegalArgumentException}.
707      * @return length
708      *                  The length of the encrypted packet that is included in the buffer. This will
709      *                  return {@code -1} if the given {@link ByteBuf} is not encrypted at all.
710      * @throws IllegalArgumentException
711      *                  Is thrown if the given {@link ByteBuf} has not at least 5 bytes to read.
712      */
713     private static int getEncryptedPacketLength(ByteBuf buffer, int offset) {
714         int packetLength = 0;
715 
716         // SSLv3 or TLS - Check ContentType
717         boolean tls;
718         switch (buffer.getUnsignedByte(offset)) {
719             case 20:  // change_cipher_spec
720             case 21:  // alert
721             case 22:  // handshake
722             case 23:  // application_data
723                 tls = true;
724                 break;
725             default:
726                 // SSLv2 or bad data
727                 tls = false;
728         }
729 
730         if (tls) {
731             // SSLv3 or TLS - Check ProtocolVersion
732             int majorVersion = buffer.getUnsignedByte(offset + 1);
733             if (majorVersion == 3) {
734                 // SSLv3 or TLS
735                 packetLength = buffer.getUnsignedShort(offset + 3) + 5;
736                 if (packetLength <= 5) {
737                     // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
738                     tls = false;
739                 }
740             } else {
741                 // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
742                 tls = false;
743             }
744         }
745 
746         if (!tls) {
747             // SSLv2 or bad data - Check the version
748             boolean sslv2 = true;
749             int headerLength = (buffer.getUnsignedByte(offset) & 0x80) != 0 ? 2 : 3;
750             int majorVersion = buffer.getUnsignedByte(offset + headerLength + 1);
751             if (majorVersion == 2 || majorVersion == 3) {
752                 // SSLv2
753                 if (headerLength == 2) {
754                     packetLength = (buffer.getShort(offset) & 0x7FFF) + 2;
755                 } else {
756                     packetLength = (buffer.getShort(offset) & 0x3FFF) + 3;
757                 }
758                 if (packetLength <= headerLength) {
759                     sslv2 = false;
760                 }
761             } else {
762                 sslv2 = false;
763             }
764 
765             if (!sslv2) {
766                 return -1;
767             }
768         }
769         return packetLength;
770     }

772     @Override
773     protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws SSLException {
774         final int startOffset = in.readerIndex();
775         final int endOffset = in.writerIndex();
776         int offset = startOffset;
777 
778         // If we calculated the length of the current SSL record before, use that information.
779         if (packetLength > 0) {
780             if (endOffset - startOffset < packetLength) {
781                 return;
782             } else {
783                 offset += packetLength;
784                 packetLength = 0;
785             }
786         }
787 
788         boolean nonSslRecord = false;
789 
790         for (;;) {
791             final int readableBytes = endOffset - offset;
792             if (readableBytes < 5) {
793                 break;
794             }
795 
796             final int packetLength = getEncryptedPacketLength(in, offset);
797             if (packetLength == -1) {
798                 nonSslRecord = true;
799                 break;
800             }
801 
802             assert packetLength > 0;
803 
804             if (packetLength > readableBytes) {
805                 // wait until the whole packet can be read
806                 this.packetLength = packetLength;
807                 break;
808             }
809 
810             offset += packetLength;
811         }
812 
813         final int length = offset - startOffset;
814         if (length > 0) {
815             // The buffer contains one or more full SSL records.
816             // Slice out the whole packet so unwrap will only be called with complete packets.
817             // Also directly reset the packetLength. This is needed as unwrap(..) may trigger
818             // decode(...) again via:
819             // 1) unwrap(..) is called
820             // 2) wrap(...) is called from within unwrap(...)
821             // 3) wrap(...) calls unwrapLater(...)
822             // 4) unwrapLater(...) calls decode(...)
823             //
824             // See https://github.com/netty/netty/issues/1534
825             in.skipBytes(length);
826             ByteBuffer buffer = in.nioBuffer(startOffset, length);
827             unwrap(ctx, buffer, out);
828         }
829 
830         if (nonSslRecord) {
831             // Not an SSL/TLS packet
832             NotSslRecordException e = new NotSslRecordException(
833                     "not an SSL/TLS record: " + ByteBufUtil.hexDump(in));
834             in.skipBytes(in.readableBytes());
835             ctx.fireExceptionCaught(e);
836             setHandshakeFailure(e);
837         }
838     }

850         RecyclableArrayList out = RecyclableArrayList.newInstance();
851         try {
852             unwrap(ctx, Unpooled.EMPTY_BUFFER.nioBuffer(), out);

862     private void unwrap(ChannelHandlerContext ctx, ByteBuffer packet, List<Object> out) throws SSLException {
863         boolean wrapLater = false;
864         int totalProduced = 0;
865         try {
866             for (;;) {
867                 if (decodeOut == null) {
868                     decodeOut = ctx.alloc().buffer(packet.remaining());
869                 }
870 
871                 final SSLEngineResult result = unwrap(engine, packet, decodeOut);
872                 final Status status = result.getStatus();
873                 final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
874                 final int produced = result.bytesProduced();
875                 final int consumed = result.bytesConsumed();
876 
877                 totalProduced += produced;
878                 if (status == Status.CLOSED) {
879                     // notify about the CLOSED state of the SSLEngine. See #137
880                     sslCloseFuture.trySuccess(ctx.channel());
881                     break;
882                 }
883 
884                 switch (handshakeStatus) {
885                     case NEED_UNWRAP:
886                         break;
887                     case NEED_WRAP:
888                         wrapNonAppData(ctx, true);
889                         break;
890                     case NEED_TASK:
891                         runDelegatedTasks();
892                         break;
893                     case FINISHED:
894                         setHandshakeSuccess();
895                         wrapLater = true;
896                         continue;
897                     case NOT_HANDSHAKING:
898                         break;
899                     default:
900                         throw new IllegalStateException("Unknown handshake status: " + handshakeStatus);
901                 }
902 
903                 if (status == Status.BUFFER_UNDERFLOW || consumed == 0 && produced == 0) {
904                     break;
905                 }
906             }
907 
908             if (wrapLater) {
909                 wrap(ctx, true);
910             }
911         } catch (SSLException e) {
912             setHandshakeFailure(e);
913             throw e;
914         } finally {
915             if (totalProduced > 0) {
916                 ByteBuf decodeOut = this.decodeOut;
917                 this.decodeOut = null;
918                 out.add(decodeOut);
919             }
920         }
921     }

923     private static SSLEngineResult unwrap(SSLEngine engine, ByteBuffer in, ByteBuf out) throws SSLException {
924         int overflows = 0;
925         for (;;) {
926             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
927             SSLEngineResult result = engine.unwrap(in, out0);
928             out.writerIndex(out.writerIndex() + result.bytesProduced());
929             switch (result.getStatus()) {
930                 case BUFFER_OVERFLOW:
931                     int max = engine.getSession().getApplicationBufferSize();
932                     switch (overflows ++) {
933                         case 0:
934                             out.ensureWritable(Math.min(max, in.remaining()));
935                             break;
936                         default:
937                             out.ensureWritable(max);
938                     }
939                     break;
940                 default:
941                     return result;
942             }
943         }

946     /**
947      * Fetches all delegated tasks from the {@link SSLEngine} and runs them via the {@link #delegatedTaskExecutor}.
948      * If the {@link #delegatedTaskExecutor} is {@link ImmediateExecutor}, just call {@link Runnable#run()} directly
949      * instead of using {@link Executor#execute(Runnable)}.  Otherwise, run the tasks via
950      * the {@link #delegatedTaskExecutor} and wait until the tasks are finished.
951      */
952     private void runDelegatedTasks() {
953         if (delegatedTaskExecutor == ImmediateExecutor.INSTANCE) {
954             for (;;) {
955                 Runnable task = engine.getDelegatedTask();
956                 if (task == null) {
957                     break;
958                 }
959 
960                 task.run();
961             }
962         } else {
963             final List<Runnable> tasks = new ArrayList<Runnable>(2);
964             for (;;) {
965                 final Runnable task = engine.getDelegatedTask();
966                 if (task == null) {
967                     break;
968                 }
969 
970                 tasks.add(task);
971             }
972 
973             if (tasks.isEmpty()) {
974                 return;
975             }
976 
977             final CountDownLatch latch = new CountDownLatch(1);
978             delegatedTaskExecutor.execute(new Runnable() {
979                 @Override
980                 public void run() {
981                     try {
982                         for (Runnable task: tasks) {
983                             task.run();
984                         }
985                     } catch (Exception e) {
986                         ctx.fireExceptionCaught(e);
987                     } finally {
988                         latch.countDown();
989                     }
990                 }
991             });
992 
993             boolean interrupted = false;
994             while (latch.getCount() != 0) {
995                 try {
996                     latch.await();
997                 } catch (InterruptedException e) {
998                     // Interrupt later.
999                     interrupted = true;
1000                 }
1001             }
1002 
1003             if (interrupted) {
1004                 Thread.currentThread().interrupt();
1005             }
1006         }
1007     }

1090         if (handshakeTimeoutMillis > 0) {
1091             timeoutFuture = ctx.executor().schedule(new Runnable() {
1092                 @Override
1093                 public void run() {

1095                         return;

1107                 if (timeoutFuture != null) {
