58      static final String PROTOCOL = "TLS";
59      static final String[] PROTOCOLS;
60      static final List<String> DEFAULT_CIPHERS;
61      static final Set<String> SUPPORTED_CIPHERS;
62  
63      static {
64          SSLContext context;
65          int i;
66          try {
67              context = SSLContext.getInstance(PROTOCOL);
68              context.init(null, null, null);
69          } catch (Exception e) {
70              throw new Error("failed to initialize the default SSL context", e);
71          }
72  
73          SSLEngine engine = context.createSSLEngine();
74  
75          // Choose the sensible default list of protocols.
76          final String[] supportedProtocols = engine.getSupportedProtocols();
77          Set<String> supportedProtocolsSet = new HashSet<String>(supportedProtocols.length);
78          for (i = 0; i < supportedProtocols.length; ++i) {
79              supportedProtocolsSet.add(supportedProtocols[i]);
80          }
81          List<String> protocols = new ArrayList<String>();
82          addIfSupported(
83                  supportedProtocolsSet, protocols,
84                  "TLSv1.2", "TLSv1.1", "TLSv1");
85  
86          if (!protocols.isEmpty()) {
87              PROTOCOLS = protocols.toArray(new String[protocols.size()]);
88          } else {
89              PROTOCOLS = engine.getEnabledProtocols();
90          }
91  
92          // Choose the sensible default list of cipher suites.
93          final String[] supportedCiphers = engine.getSupportedCipherSuites();
94          SUPPORTED_CIPHERS = new HashSet<String>(supportedCiphers.length);
95          for (i = 0; i < supportedCiphers.length; ++i) {
96              SUPPORTED_CIPHERS.add(supportedCiphers[i]);
97          }
98          List<String> ciphers = new ArrayList<String>();
99          addIfSupported(
100                 SUPPORTED_CIPHERS, ciphers,
101                 // XXX: Make sure to sync this list with OpenSslEngineFactory.
102                 // GCM (Galois/Counter Mode) requires JDK 8.
103                 "TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384",
104                 "TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256",
105                 "TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256",
106                 "TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA",
107                 // AES256 requires JCE unlimited strength jurisdiction policy files.
108                 "TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA",
109                 // GCM (Galois/Counter Mode) requires JDK 8.
110                 "TLS_RSA_WITH_AES_128_GCM_SHA256",
111                 "TLS_RSA_WITH_AES_128_CBC_SHA",
112                 // AES256 requires JCE unlimited strength jurisdiction policy files.
113                 "TLS_RSA_WITH_AES_256_CBC_SHA");
114 
115         if (ciphers.isEmpty()) {
116             // Use the default from JDK as fallback.
117             for (String cipher : engine.getEnabledCipherSuites()) {
118                 if (cipher.contains("_RC4_")) {
119                     continue;
120                 }
121                 ciphers.add(cipher);
122             }
123         }
124         DEFAULT_CIPHERS = Collections.unmodifiableList(ciphers);
125 
126         if (logger.isDebugEnabled()) {
127             logger.debug("Default protocols (JDK): {} ", Arrays.asList(PROTOCOLS));
128             logger.debug("Default cipher suites (JDK): {}", DEFAULT_CIPHERS);
129         }
130     }

132     private static void addIfSupported(Set<String> supported, List<String> enabled, String... names) {
133         for (String n: names) {
134             if (supported.contains(n)) {
135                 enabled.add(n);
136             }
137         }
138     }

176     JdkSslContext(SSLContext sslContext, boolean isClient, Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
177                   JdkApplicationProtocolNegotiator apn, ClientAuth clientAuth, boolean startTls) {
178         super(startTls);
179         this.apn = checkNotNull(apn, "apn");
180         this.clientAuth = checkNotNull(clientAuth, "clientAuth");
181         cipherSuites = checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
182                 ciphers, DEFAULT_CIPHERS, SUPPORTED_CIPHERS);
183         unmodifiableCipherSuites = Collections.unmodifiableList(Arrays.asList(cipherSuites));
184         this.sslContext = checkNotNull(sslContext, "sslContext");
185         this.isClient = isClient;
186     }

237     private SSLEngine configureAndWrapEngine(SSLEngine engine) {
238         engine.setEnabledCipherSuites(cipherSuites);
239         engine.setEnabledProtocols(PROTOCOLS);
240         engine.setUseClientMode(isClient());
241         if (isServer()) {
242             switch (clientAuth) {
243                 case OPTIONAL:
244                     engine.setWantClientAuth(true);
245                     break;
246                 case REQUIRE:
247                     engine.setNeedClientAuth(true);
248                     break;
249             }
250         }
251         return apn.wrapperFactory().wrapSslEngine(engine, apn, isServer());
252     }

259     /**
260      * Translate a {@link ApplicationProtocolConfig} object to a {@link JdkApplicationProtocolNegotiator} object.
261      * @param config The configuration which defines the translation
262      * @param isServer {@code true} if a server {@code false} otherwise.
263      * @return The results of the translation
264      */
265     static JdkApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config, boolean isServer) {
266         if (config == null) {
267             return JdkDefaultApplicationProtocolNegotiator.INSTANCE;
268         }
269 
270         switch(config.protocol()) {
271         case NONE:
272             return JdkDefaultApplicationProtocolNegotiator.INSTANCE;
273         case ALPN:
274             if (isServer) {
275                 switch(config.selectorFailureBehavior()) {
276                 case FATAL_ALERT:
277                     return new JdkAlpnApplicationProtocolNegotiator(true, config.supportedProtocols());
278                 case NO_ADVERTISE:
279                     return new JdkAlpnApplicationProtocolNegotiator(false, config.supportedProtocols());
280                 default:
281                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
282                     .append(config.selectorFailureBehavior()).append(" failure behavior").toString());
283                 }
284             } else {
285                 switch(config.selectedListenerFailureBehavior()) {
286                 case ACCEPT:
287                     return new JdkAlpnApplicationProtocolNegotiator(false, config.supportedProtocols());
288                 case FATAL_ALERT:
289                     return new JdkAlpnApplicationProtocolNegotiator(true, config.supportedProtocols());
290                 default:
291                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
292                     .append(config.selectedListenerFailureBehavior()).append(" failure behavior").toString());
293                 }
294             }
295         case NPN:
296             if (isServer) {
297                 switch(config.selectedListenerFailureBehavior()) {
298                 case ACCEPT:
299                     return new JdkNpnApplicationProtocolNegotiator(false, config.supportedProtocols());
300                 case FATAL_ALERT:
301                     return new JdkNpnApplicationProtocolNegotiator(true, config.supportedProtocols());
302                 default:
303                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
304                     .append(config.selectedListenerFailureBehavior()).append(" failure behavior").toString());
305                 }
306             } else {
307                 switch(config.selectorFailureBehavior()) {
308                 case FATAL_ALERT:
309                     return new JdkNpnApplicationProtocolNegotiator(true, config.supportedProtocols());
310                 case NO_ADVERTISE:
311                     return new JdkNpnApplicationProtocolNegotiator(false, config.supportedProtocols());
312                 default:
313                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
314                     .append(config.selectorFailureBehavior()).append(" failure behavior").toString());
315                 }
316             }
317         default:
318             throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
319             .append(config.protocol()).append(" protocol").toString());
320         }
321     }
