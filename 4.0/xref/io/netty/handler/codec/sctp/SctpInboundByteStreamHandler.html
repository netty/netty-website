17  package io.netty.handler.codec.sctp;
18  
19  import io.netty.buffer.ByteBuf;
20  import io.netty.channel.ChannelHandlerContext;
21  import io.netty.channel.ChannelInboundMessageHandlerAdapter;
22  import io.netty.channel.sctp.SctpMessage;
23  import io.netty.handler.codec.CodecException;
24  
25  /**
26   * A ChannelHandler which receives {@link SctpMessage}s which belong to a application protocol form a specific
27   * SCTP Stream  and decode it as {@link ByteBuf}.
28   */
29  public class SctpInboundByteStreamHandler extends ChannelInboundMessageHandlerAdapter<SctpMessage> {
30      private final int protocolIdentifier;
31      private final int streamIdentifier;
32  
33      /**
34       * @param streamIdentifier   accepted stream number, this should be >=0 or <= max stream number of the association.
35       * @param protocolIdentifier supported application protocol.
36       */
37      public SctpInboundByteStreamHandler(int protocolIdentifier, int streamIdentifier) {
38          this.protocolIdentifier = protocolIdentifier;
39          this.streamIdentifier = streamIdentifier;
40      }
41  
42      @Override
43      protected final boolean acceptInboundMessage(Object msg) throws Exception {
44          if (super.acceptInboundMessage(msg)) {
45              return acceptInboundMessage((SctpMessage) msg);
46          }
47          return false;
48      }
49  
50      protected boolean acceptInboundMessage(SctpMessage msg) {
51          return msg.protocolIdentifier() == protocolIdentifier && msg.streamIdentifier() == streamIdentifier;
52      }
53  
54      @Override
55      protected void messageReceived(ChannelHandlerContext ctx, SctpMessage msg) throws Exception {
56          if (!msg.isComplete()) {
57              throw new CodecException(String.format("Received SctpMessage is not complete, please add %s in the " +
58                      "pipeline before this handler", SctpMessageCompletionHandler.class.getSimpleName()));
59          }
60  
61          ctx.nextInboundByteBuffer().writeBytes(msg.data());
62          ctx.fireInboundBufferUpdated();
63      }
64  }
