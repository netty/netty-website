16  package io.netty.handler.codec.spdy;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelFuture;
20  import io.netty.channel.ChannelFutureListener;
21  import io.netty.channel.ChannelHandlerContext;
22  import io.netty.handler.codec.MessageToByteEncoder;
23  import io.netty.handler.codec.UnsupportedMessageTypeException;
24  
25  import java.util.Set;
26  
27  import static io.netty.handler.codec.spdy.SpdyCodecUtil.*;
28  
29  /**
30   * Encodes a SPDY Frame into a {@link ByteBuf}.
31   */
32  public class SpdyFrameEncoder extends MessageToByteEncoder<SpdyFrame> {
33  
34      private final int version;
35      private final SpdyHeaderBlockEncoder headerBlockEncoder;
36  
37      /**
38       * Creates a new instance with the specified {@code version} and the
39       * default {@code compressionLevel (6)}, {@code windowBits (15)},
40       * and {@code memLevel (8)}.
41       */
42      public SpdyFrameEncoder(int version) {
43          this(version, 6, 15, 8);
44      }
45  
46      /**
47       * Creates a new instance with the specified parameters.
48       */
49      public SpdyFrameEncoder(int version, int compressionLevel, int windowBits, int memLevel) {
50          this(version, SpdyHeaderBlockEncoder.newInstance(
51                      version, compressionLevel, windowBits, memLevel));
52      }
53  
54      protected SpdyFrameEncoder(int version, SpdyHeaderBlockEncoder headerBlockEncoder) {
55          if (version < SpdyConstants.SPDY_MIN_VERSION || version > SpdyConstants.SPDY_MAX_VERSION) {
56              throw new IllegalArgumentException(
57                      "unknown version: " + version);
58          }
59          this.version = version;
60          this.headerBlockEncoder = headerBlockEncoder;
61      }
62  
63      @Override
64      public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
65          ctx.channel().closeFuture().addListener(new ChannelFutureListener() {
66              @Override
67              public void operationComplete(ChannelFuture future) throws Exception {
68                  synchronized (headerBlockEncoder) {
69                      headerBlockEncoder.end();
70                  }
71              }
72          });
73      }
74  
75      @Override
76      protected void encode(ChannelHandlerContext ctx, SpdyFrame msg, ByteBuf out) throws Exception {
77          if (msg instanceof SpdyDataFrame) {
78  
79              SpdyDataFrame spdyDataFrame = (SpdyDataFrame) msg;
80              ByteBuf data = spdyDataFrame.content();
81              byte flags = spdyDataFrame.isLast() ? SPDY_DATA_FLAG_FIN : 0;
82              out.ensureWritable(SPDY_HEADER_SIZE + data.readableBytes());
83              out.writeInt(spdyDataFrame.getStreamId() & 0x7FFFFFFF);
84              out.writeByte(flags);
85              out.writeMedium(data.readableBytes());
86              out.writeBytes(data, data.readerIndex(), data.readableBytes());
87  
88          } else if (msg instanceof SpdySynStreamFrame) {
89  
90              SpdySynStreamFrame spdySynStreamFrame = (SpdySynStreamFrame) msg;
91              ByteBuf data = headerBlockEncoder.encode(ctx, spdySynStreamFrame);
92              try {
93                  byte flags = spdySynStreamFrame.isLast() ? SPDY_FLAG_FIN : 0;
94                  if (spdySynStreamFrame.isUnidirectional()) {
95                      flags |= SPDY_FLAG_UNIDIRECTIONAL;
96                  }
97                  int headerBlockLength = data.readableBytes();
98                  int length;
99                  if (version < 3) {
100                     length = headerBlockLength == 0 ? 12 : 10 + headerBlockLength;
101                 } else {
102                     length = 10 + headerBlockLength;
103                 }
104                 out.ensureWritable(SPDY_HEADER_SIZE + length);
105                 out.writeShort(version | 0x8000);
106                 out.writeShort(SPDY_SYN_STREAM_FRAME);
107                 out.writeByte(flags);
108                 out.writeMedium(length);
109                 out.writeInt(spdySynStreamFrame.getStreamId());
110                 out.writeInt(spdySynStreamFrame.getAssociatedToStreamId());
111                 if (version < 3) {
112                     // Restrict priorities for SPDY/2 to between 0 and 3
113                     byte priority = spdySynStreamFrame.getPriority();
114                     if (priority > 3) {
115                         priority = 3;
116                     }
117                     out.writeShort((priority & 0xFF) << 14);
118                 } else {
119                     out.writeShort((spdySynStreamFrame.getPriority() & 0xFF) << 13);
120                 }
121                 if (version < 3 && data.readableBytes() == 0) {
122                     out.writeShort(0);
123                 }
124                 out.writeBytes(data, data.readerIndex(), headerBlockLength);
125             } finally {
126                 data.release();
127             }
128 
129         } else if (msg instanceof SpdySynReplyFrame) {
130 
131             SpdySynReplyFrame spdySynReplyFrame = (SpdySynReplyFrame) msg;
132             ByteBuf data = headerBlockEncoder.encode(ctx, spdySynReplyFrame);
133             try {
134                 byte flags = spdySynReplyFrame.isLast() ? SPDY_FLAG_FIN : 0;
135                 int headerBlockLength = data.readableBytes();
136                 int length;
137                 if (version < 3) {
138                     length = headerBlockLength == 0 ? 8 : 6 + headerBlockLength;
139                 } else {
140                     length = 4 + headerBlockLength;
141                 }
142                 out.ensureWritable(SPDY_HEADER_SIZE + length);
143                 out.writeShort(version | 0x8000);
144                 out.writeShort(SPDY_SYN_REPLY_FRAME);
145                 out.writeByte(flags);
146                 out.writeMedium(length);
147                 out.writeInt(spdySynReplyFrame.getStreamId());
148                 if (version < 3) {
149                     if (headerBlockLength == 0) {
150                         out.writeInt(0);
151                     } else {
152                         out.writeShort(0);
153                     }
154                 }
155                 out.writeBytes(data, data.readerIndex(), headerBlockLength);
156             } finally {
157                 data.release();
158             }
159 
160         } else if (msg instanceof SpdyRstStreamFrame) {
161 
162             SpdyRstStreamFrame spdyRstStreamFrame = (SpdyRstStreamFrame) msg;
163             out.ensureWritable(SPDY_HEADER_SIZE + 8);
164             out.writeShort(version | 0x8000);
165             out.writeShort(SPDY_RST_STREAM_FRAME);
166             out.writeInt(8);
167             out.writeInt(spdyRstStreamFrame.getStreamId());
168             out.writeInt(spdyRstStreamFrame.getStatus().getCode());
169 
170         } else if (msg instanceof SpdySettingsFrame) {
171 
172             SpdySettingsFrame spdySettingsFrame = (SpdySettingsFrame) msg;
173             byte flags = spdySettingsFrame.clearPreviouslyPersistedSettings() ?
174                 SPDY_SETTINGS_CLEAR : 0;
175             Set<Integer> IDs = spdySettingsFrame.getIds();
176             int numEntries = IDs.size();
177             int length = 4 + numEntries * 8;
178             out.ensureWritable(SPDY_HEADER_SIZE + length);
179             out.writeShort(version | 0x8000);
180             out.writeShort(SPDY_SETTINGS_FRAME);
181             out.writeByte(flags);
182             out.writeMedium(length);
183             out.writeInt(numEntries);
184             for (Integer ID: IDs) {
185                 int id = ID.intValue();
186                 byte ID_flags = 0;
187                 if (spdySettingsFrame.isPersistValue(id)) {
188                     ID_flags |= SPDY_SETTINGS_PERSIST_VALUE;
189                 }
190                 if (spdySettingsFrame.isPersisted(id)) {
191                     ID_flags |= SPDY_SETTINGS_PERSISTED;
192                 }
193                 if (version < 3) {
194                     // Chromium Issue 79156
195                     // SPDY setting ids are not written in network byte order
196                     // Write id assuming the architecture is little endian
197                     out.writeByte(id       & 0xFF);
198                     out.writeByte(id >>  8 & 0xFF);
199                     out.writeByte(id >> 16 & 0xFF);
200                     out.writeByte(ID_flags);
201                 } else {
202                     out.writeByte(ID_flags);
203                     out.writeMedium(id);
204                 }
205                 out.writeInt(spdySettingsFrame.getValue(id));
206             }
207 
208         } else if (msg instanceof SpdyPingFrame) {
209 
210             SpdyPingFrame spdyPingFrame = (SpdyPingFrame) msg;
211             out.ensureWritable(SPDY_HEADER_SIZE + 4);
212             out.writeShort(version | 0x8000);
213             out.writeShort(SPDY_PING_FRAME);
214             out.writeInt(4);
215             out.writeInt(spdyPingFrame.getId());
216 
217         } else if (msg instanceof SpdyGoAwayFrame) {
218 
219             SpdyGoAwayFrame spdyGoAwayFrame = (SpdyGoAwayFrame) msg;
220             int length = version < 3 ? 4 : 8;
221             out.ensureWritable(SPDY_HEADER_SIZE + length);
222             out.writeShort(version | 0x8000);
223             out.writeShort(SPDY_GOAWAY_FRAME);
224             out.writeInt(length);
225             out.writeInt(spdyGoAwayFrame.getLastGoodStreamId());
226             if (version >= 3) {
227                 out.writeInt(spdyGoAwayFrame.getStatus().getCode());
228             }
229 
230         } else if (msg instanceof SpdyHeadersFrame) {
231 
232             SpdyHeadersFrame spdyHeadersFrame = (SpdyHeadersFrame) msg;
233             ByteBuf data = headerBlockEncoder.encode(ctx, spdyHeadersFrame);
234             try {
235                 byte flags = spdyHeadersFrame.isLast() ? SPDY_FLAG_FIN : 0;
236                 int headerBlockLength = data.readableBytes();
237                 int length;
238                 if (version < 3) {
239                     length = headerBlockLength == 0 ? 4 : 6 + headerBlockLength;
240                 } else {
241                     length = 4 + headerBlockLength;
242                 }
243                 out.ensureWritable(SPDY_HEADER_SIZE + length);
244                 out.writeShort(version | 0x8000);
245                 out.writeShort(SPDY_HEADERS_FRAME);
246                 out.writeByte(flags);
247                 out.writeMedium(length);
248                 out.writeInt(spdyHeadersFrame.getStreamId());
249                 if (version < 3 && headerBlockLength != 0) {
250                     out.writeShort(0);
251                 }
252                 out.writeBytes(data, data.readerIndex(), headerBlockLength);
253             } finally {
254                 data.release();
255             }
256 
257         } else if (msg instanceof SpdyWindowUpdateFrame) {
258 
259             SpdyWindowUpdateFrame spdyWindowUpdateFrame = (SpdyWindowUpdateFrame) msg;
260             out.ensureWritable(SPDY_HEADER_SIZE + 8);
261             out.writeShort(version | 0x8000);
262             out.writeShort(SPDY_WINDOW_UPDATE_FRAME);
263             out.writeInt(8);
264             out.writeInt(spdyWindowUpdateFrame.getStreamId());
265             out.writeInt(spdyWindowUpdateFrame.getDeltaWindowSize());
266         } else {
267             throw new UnsupportedMessageTypeException(msg);
268         }
269     }
