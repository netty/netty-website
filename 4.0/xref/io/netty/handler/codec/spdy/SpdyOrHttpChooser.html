33  /**
34   * {@link ChannelInboundHandler} which is responsible to setup the {@link ChannelPipeline} either for
35   * HTTP or SPDY. This offers an easy way for users to support both at the same time while not care to
36   * much about the low-level details.
37   */
38  public abstract class SpdyOrHttpChooser extends ByteToMessageDecoder {
39  
40      // TODO: Replace with generic NPN handler
41  
42      public enum SelectedProtocol {
43          SPDY_3("spdy/3"),
44          SPDY_3_1("spdy/3.1"),
45          HTTP_1_1("http/1.1"),
46          HTTP_1_0("http/1.0"),
47          UNKNOWN("Unknown");
48  
49          private final String name;
50  
51          SelectedProtocol(String defaultName) {
52              name = defaultName;
53          }
54  
55          public String protocolName() {
56              return name;
57          }
58  
59          /**
60           * Get an instance of this enum based on the protocol name returned by the NPN server provider
61           *
62           * @param name
63           *            the protocol name
64           * @return the SelectedProtocol instance
65           */
66          public static SelectedProtocol protocol(String name) {
67              for (SelectedProtocol protocol : SelectedProtocol.values()) {
68                  if (protocol.protocolName().equals(name)) {
69                      return protocol;
70                  }
71              }
72              return UNKNOWN;
73          }
74      }
75  
76      private final int maxSpdyContentLength;
77      private final int maxHttpContentLength;
78  
79      protected SpdyOrHttpChooser(int maxSpdyContentLength, int maxHttpContentLength) {
80          this.maxSpdyContentLength = maxSpdyContentLength;
81          this.maxHttpContentLength = maxHttpContentLength;
82      }
83  
84      /**
85       * Return the {@link SelectedProtocol} for the {@link SSLEngine}. If its not known yet implementations MUST return
86       * {@link SelectedProtocol#UNKNOWN}.
87       *
88       */
89      protected abstract SelectedProtocol getProtocol(SSLEngine engine);
90  
91      @Override
92      protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
93          if (initPipeline(ctx)) {
94              // When we reached here we can remove this handler as its now clear
95              // what protocol we want to use
96              // from this point on. This will also take care of forward all
97              // messages.
98              ctx.pipeline().remove(this);
99          }
100     }
101 
102     private boolean initPipeline(ChannelHandlerContext ctx) {
103         // Get the SslHandler from the ChannelPipeline so we can obtain the
104         // SslEngine from it.
105         SslHandler handler = ctx.pipeline().get(SslHandler.class);
106         if (handler == null) {
107             // SslHandler is needed by SPDY by design.
108             throw new IllegalStateException("SslHandler is needed for SPDY");
109         }
110 
111         SelectedProtocol protocol = getProtocol(handler.engine());
112         switch (protocol) {
113         case UNKNOWN:
114             // Not done with choosing the protocol, so just return here for now,
115             return false;
116         case SPDY_3:
117             addSpdyHandlers(ctx, SpdyVersion.SPDY_3);
118             break;
119         case SPDY_3_1:
120             addSpdyHandlers(ctx, SpdyVersion.SPDY_3_1);
121             break;
122         case HTTP_1_0:
123         case HTTP_1_1:
124             addHttpHandlers(ctx);
125             break;
126         default:
127             throw new IllegalStateException("Unknown SelectedProtocol");
128         }
129         return true;
130     }
131 
132     /**
133      * Add all {@link ChannelHandler}'s that are needed for SPDY with the given version.
134      */
135     protected void addSpdyHandlers(ChannelHandlerContext ctx, SpdyVersion version) {
136         ChannelPipeline pipeline = ctx.pipeline();
137         pipeline.addLast("spdyDecoder", new SpdyFrameDecoder(version));
138         pipeline.addLast("spdyEncoder", new SpdyFrameEncoder(version));
139         pipeline.addLast("spdySessionHandler", new SpdySessionHandler(version, true));
140         pipeline.addLast("spdyHttpEncoder", new SpdyHttpEncoder(version));
141         pipeline.addLast("spdyHttpDecoder", new SpdyHttpDecoder(version, maxSpdyContentLength));
142         pipeline.addLast("spdyStreamIdHandler", new SpdyHttpResponseStreamIdHandler());
143         pipeline.addLast("httpRquestHandler", createHttpRequestHandlerForSpdy());
144     }
145 
146     /**
147      * Add all {@link ChannelHandler}'s that are needed for HTTP.
148      */
149     protected void addHttpHandlers(ChannelHandlerContext ctx) {
150         ChannelPipeline pipeline = ctx.pipeline();
151         pipeline.addLast("httpRquestDecoder", new HttpRequestDecoder());
152         pipeline.addLast("httpResponseEncoder", new HttpResponseEncoder());
153         pipeline.addLast("httpChunkAggregator", new HttpObjectAggregator(maxHttpContentLength));
154         pipeline.addLast("httpRquestHandler", createHttpRequestHandlerForHttp());
155     }
156 
157     /**
158      * Create the {@link ChannelInboundHandler} that is responsible for handling the http requests
159      * when the {@link SelectedProtocol} was {@link SelectedProtocol#HTTP_1_0} or
160      * {@link SelectedProtocol#HTTP_1_1}
161      */
162     protected abstract ChannelInboundHandler createHttpRequestHandlerForHttp();
163 
164     /**
165      * Create the {@link ChannelInboundHandler} that is responsible for handling the http responses
166      * when the {@link SelectedProtocol} was {@link SelectedProtocol#SPDY_3} or
167      * {@link SelectedProtocol#SPDY_3_1}.
168      *
169      * By default this getMethod will just delecate to {@link #createHttpRequestHandlerForHttp()}, but sub-classes may
170      * override this to change the behaviour.
171      */
172     protected ChannelInboundHandler createHttpRequestHandlerForSpdy() {
173         return createHttpRequestHandlerForHttp();
174     }
