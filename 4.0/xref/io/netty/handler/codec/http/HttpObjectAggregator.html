32  /**
33   * A {@link ChannelHandler} that aggregates an {@link HttpMessage}
34   * and its following {@link HttpContent}s into a single {@link HttpMessage} with
35   * no following {@link HttpContent}s.  It is useful when you don't want to take
36   * care of HTTP messages whose transfer encoding is 'chunked'.  Insert this
37   * handler after {@link HttpObjectDecoder} in the {@link ChannelPipeline}:
38   * <pre>
39   * {@link ChannelPipeline} p = ...;
40   * ...
41   * p.addLast("decoder", new {@link HttpRequestDecoder}());
42   * p.addLast("aggregator", <b>new {@link HttpObjectAggregator}(1048576)</b>);
43   * ...
44   * p.addLast("encoder", new {@link HttpResponseEncoder}());
45   * p.addLast("handler", new HttpRequestHandler());
46   * </pre>
47   */
48  public class HttpObjectAggregator extends MessageToMessageDecoder<HttpObject> {
49      public static final int DEFAULT_MAX_COMPOSITEBUFFER_COMPONENTS = 1024;
50      private static final ByteBuf CONTINUE = Unpooled.copiedBuffer(
51              "HTTP/1.1 100 Continue\r\n\r\n", CharsetUtil.US_ASCII);
52  
53      private final int maxContentLength;
54      private FullHttpMessage currentMessage;
55  
56      private int maxCumulationBufferComponents = DEFAULT_MAX_COMPOSITEBUFFER_COMPONENTS;
57      private ChannelHandlerContext ctx;
58  
59      /**
60       * Creates a new instance.
61       *
62       * @param maxContentLength
63       *        the maximum length of the aggregated content.
64       *        If the length of the aggregated content exceeds this value,
65       *        a {@link TooLongFrameException} will be raised.
66       */
67      public HttpObjectAggregator(int maxContentLength) {
68          if (maxContentLength <= 0) {
69              throw new IllegalArgumentException(
70                      "maxContentLength must be a positive integer: " +
71                      maxContentLength);
72          }
73          this.maxContentLength = maxContentLength;
74      }
75  
76      /**
77       * Returns the maximum number of components in the cumulation buffer.  If the number of
78       * the components in the cumulation buffer exceeds this value, the components of the
79       * cumulation buffer are consolidated into a single component, involving memory copies.
80       * The default value of this property is {@link #DEFAULT_MAX_COMPOSITEBUFFER_COMPONENTS}.
81       */
82      public final int getMaxCumulationBufferComponents() {
83          return maxCumulationBufferComponents;
84      }
85  
86      /**
87       * Sets the maximum number of components in the cumulation buffer.  If the number of
88       * the components in the cumulation buffer exceeds this value, the components of the
89       * cumulation buffer are consolidated into a single component, involving memory copies.
90       * The default value of this property is {@link #DEFAULT_MAX_COMPOSITEBUFFER_COMPONENTS}
91       * and its minimum allowed value is {@code 2}.
92       */
93      public final void setMaxCumulationBufferComponents(int maxCumulationBufferComponents) {
94          if (maxCumulationBufferComponents < 2) {
95              throw new IllegalArgumentException(
96                      "maxCumulationBufferComponents: " + maxCumulationBufferComponents +
97                      " (expected: >= 2)");
98          }
99  
100         if (ctx == null) {
101             this.maxCumulationBufferComponents = maxCumulationBufferComponents;
102         } else {
103             throw new IllegalStateException(
104                     "decoder properties cannot be changed once the decoder is added to a pipeline.");
105         }
106     }
107 
108     @Override
109     protected Object decode(ChannelHandlerContext ctx, HttpObject msg) throws Exception {
110         FullHttpMessage currentMessage = this.currentMessage;
111 
112         if (msg instanceof HttpMessage) {
113             assert currentMessage == null;
114 
115             HttpMessage m = (HttpMessage) msg;
116 
117             // Handle the 'Expect: 100-continue' header if necessary.
118             // TODO: Respond with 413 Request Entity Too Large
119             //   and discard the traffic or close the connection.
120             //       No need to notify the upstream handlers - just log.
121             //       If decoding a response, just throw an exception.
122             if (is100ContinueExpected(m)) {
123                 ctx.write(CONTINUE.duplicate());
124             }
125 
126             if (!m.getDecoderResult().isSuccess()) {
127                 removeTransferEncodingChunked(m);
128                 this.currentMessage = null;
129                 BufUtil.retain(m);
130                 return m;
131             }
132             if (msg instanceof HttpRequest) {
133                 HttpRequest header = (HttpRequest) msg;
134                 this.currentMessage = currentMessage = new DefaultFullHttpRequest(header.getProtocolVersion(),
135                         header.getMethod(), header.getUri(), Unpooled.compositeBuffer(maxCumulationBufferComponents));
136             } else if (msg instanceof HttpResponse) {
137                 HttpResponse header = (HttpResponse) msg;
138                 this.currentMessage = currentMessage = new DefaultFullHttpResponse(
139                         header.getProtocolVersion(), header.getStatus(),
140                         Unpooled.compositeBuffer(maxCumulationBufferComponents));
141             } else {
142                 throw new Error();
143             }
144 
145             currentMessage.headers().set(m.headers());
146 
147             // A streamed message - initialize the cumulative buffer, and wait for incoming chunks.
148             removeTransferEncodingChunked(currentMessage);
149             return null;
150 
151         } else if (msg instanceof HttpContent) {
152             assert currentMessage != null;
153 
154             // Merge the received chunk into the content of the current message.
155             HttpContent chunk = (HttpContent) msg;
156             CompositeByteBuf content = (CompositeByteBuf) currentMessage.data();
157 
158             if (content.readableBytes() > maxContentLength - chunk.data().readableBytes()) {
159                 // TODO: Respond with 413 Request Entity Too Large
160                 //   and discard the traffic or close the connection.
161                 //       No need to notify the upstream handlers - just log.
162                 //       If decoding a response, just throw an exception.
163                 throw new TooLongFrameException(
164                         "HTTP content length exceeded " + maxContentLength +
165                         " bytes.");
166             }
167 
168             // Append the content of the chunk
169             if (chunk.data().isReadable()) {
170                 chunk.retain();
171                 content.addComponent(chunk.data());
172                 content.writerIndex(content.writerIndex() + chunk.data().readableBytes());
173             }
174 
175             final boolean last;
176             if (!chunk.getDecoderResult().isSuccess()) {
177                 currentMessage.setDecoderResult(
178                         DecoderResult.partialFailure(chunk.getDecoderResult().cause()));
179                 last = true;
180             } else {
181                 last = chunk instanceof LastHttpContent;
182             }
183 
184             if (last) {
185                 this.currentMessage = null;
186 
187                 // Merge trailing headers into the message.
188                 if (chunk instanceof LastHttpContent) {
189                     LastHttpContent trailer = (LastHttpContent) chunk;
190                     currentMessage.headers().add(trailer.trailingHeaders());
191                 }
192 
193                 // Set the 'Content-Length' header.
194                 currentMessage.headers().set(
195                         HttpHeaders.Names.CONTENT_LENGTH,
196                         String.valueOf(content.readableBytes()));
197 
198                 // All done
199                 return currentMessage;
200             } else {
201                 return null;
202             }
203         } else {
204             throw new Error();
205         }
206     }
207 
208     @Override
209     public void beforeAdd(ChannelHandlerContext ctx) throws Exception {
210         this.ctx = ctx;
211     }
