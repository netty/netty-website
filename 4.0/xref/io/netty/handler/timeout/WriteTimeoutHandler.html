16  package io.netty.handler.timeout;
17  
18  import io.netty.bootstrap.ServerBootstrap;
19  import io.netty.channel.Channel;
20  import io.netty.channel.ChannelDuplexHandler;
21  import io.netty.channel.ChannelFuture;
22  import io.netty.channel.ChannelFutureListener;
23  import io.netty.channel.ChannelHandlerContext;
24  import io.netty.channel.ChannelInitializer;
25  import io.netty.channel.ChannelOutboundHandlerAdapter;
26  import io.netty.channel.ChannelPromise;
27  
28  import java.util.concurrent.ScheduledFuture;
29  import java.util.concurrent.TimeUnit;
30  
31  /**
32   * Raises a {@link WriteTimeoutException} when no data was written within a
33   * certain period of time.
34   *
35   * <pre>
36   * // The connection is closed when there is no outbound traffic
37   * // for 30 seconds.
38   *
39   * public class MyChannelInitializer extends {@link ChannelInitializer}&lt{@link Channel}&gt {
40   *     public void initChannel({@link Channel} channel) {
41   *         channel.pipeline().addLast("writeTimeoutHandler", new {@link WriteTimeoutHandler}(30);
42   *         channel.pipeline().addLast("myHandler", new MyHandler());
43   *     }
44   * }
45   *
46   * // Handler should handle the {@link WriteTimeoutException}.
47   * public class MyHandler extends {@link ChannelDuplexHandler} {
48   *     {@code @Override}
49   *     public void exceptionCaught({@link ChannelHandlerContext} ctx, {@link Throwable} cause)
50   *             throws {@link Exception} {
51   *         if (cause instanceof {@link WriteTimeoutException}) {
52   *             // do something
53   *         } else {
54   *             super.exceptionCaught(ctx, cause);
55   *         }
56   *     }
57   * }
58   *
59   * {@link ServerBootstrap} bootstrap = ...;
60   * ...
61   * bootstrap.childHandler(new MyChannelInitializer());
62   * ...
63   * </pre>
64   * @see ReadTimeoutHandler
65   * @see IdleStateHandler
66   */
67  public class WriteTimeoutHandler extends ChannelOutboundHandlerAdapter {
68  
69      private final long timeoutMillis;
70  
71      private boolean closed;
72  
73      /**
74       * Creates a new instance.
75       *
76       * @param timeoutSeconds
77       *        write timeout in seconds
78       */
79      public WriteTimeoutHandler(int timeoutSeconds) {
80          this(timeoutSeconds, TimeUnit.SECONDS);
81      }
82  
83      /**
84       * Creates a new instance.
85       *
86       * @param timeout
87       *        write timeout
88       * @param unit
89       *        the {@link TimeUnit} of {@code timeout}
90       */
91      public WriteTimeoutHandler(long timeout, TimeUnit unit) {
92          if (unit == null) {
93              throw new NullPointerException("unit");
94          }
95  
96          if (timeout <= 0) {
97              timeoutMillis = 0;
98          } else {
99              timeoutMillis = Math.max(unit.toMillis(timeout), 1);
100         }
101     }
102 
103     @Override
104     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
105         scheduleTimeout(ctx, promise);
106         ctx.write(msg, promise);
107     }
108 
109     private void scheduleTimeout(final ChannelHandlerContext ctx, final ChannelPromise future) {
110         if (timeoutMillis > 0) {
111             // Schedule a timeout.
112             final ScheduledFuture<?> sf = ctx.executor().schedule(new Runnable() {
113                 @Override
114                 public void run() {
115                     // Was not written yet so issue a write timeout
116                     // The future itself will be failed with a ClosedChannelException once the close() was issued
117                     // See https://github.com/netty/netty/issues/2159
118                     if (!future.isDone()) {
119                         try {
120                             writeTimedOut(ctx);
121                         } catch (Throwable t) {
122                             ctx.fireExceptionCaught(t);
123                         }
124                     }
125                 }
126             }, timeoutMillis, TimeUnit.MILLISECONDS);
127 
128             // Cancel the scheduled timeout if the flush future is complete.
129             future.addListener(new ChannelFutureListener() {
130                 @Override
131                 public void operationComplete(ChannelFuture future) throws Exception {
132                     sf.cancel(false);
133                 }
134             });
135         }
136     }
137 
138     /**
139      * Is called when a write timeout was detected
140      */
141     protected void writeTimedOut(ChannelHandlerContext ctx) throws Exception {
142         if (!closed) {
143             ctx.fireExceptionCaught(WriteTimeoutException.INSTANCE);
144             ctx.close();
145             closed = true;
146         }
147     }
148 }
