47      protected PoolArena(PooledByteBufAllocator parent, int pageSize, int maxOrder, int pageShifts, int chunkSize) {
48          this.parent = parent;
49          this.pageSize = pageSize;
50          this.maxOrder = maxOrder;
51          this.pageShifts = pageShifts;
52          this.chunkSize = chunkSize;
53          subpageOverflowMask = ~(pageSize - 1);
54  
55          tinySubpagePools = newSubpagePoolArray(512 >>> 4);
56          for (int i = 0; i < tinySubpagePools.length; i ++) {
57              tinySubpagePools[i] = newSubpagePoolHead(pageSize);
58          }
59  
60          smallSubpagePools = newSubpagePoolArray(pageShifts - 9);
61          for (int i = 0; i < smallSubpagePools.length; i ++) {
62              smallSubpagePools[i] = newSubpagePoolHead(pageSize);
63          }
64  
65          q100 = new PoolChunkList<T>(this, null, 100, Integer.MAX_VALUE);
66          q075 = new PoolChunkList<T>(this, q100, 75, 100);
67          q050 = new PoolChunkList<T>(this, q075, 50, 100);
68          q025 = new PoolChunkList<T>(this, q050, 25, 75);

70          qInit = new PoolChunkList<T>(this, q000, Integer.MIN_VALUE, 25);

72          q100.prevList = q075;
73          q075.prevList = q050;
74          q050.prevList = q025;
75          q025.prevList = q000;

98      private void allocate(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity) {
99          final int normCapacity = normalizeCapacity(reqCapacity);
100         if ((normCapacity & subpageOverflowMask) == 0) { // capacity < pageSize
101             int tableIdx;
102             PoolSubpage<T>[] table;
103             if ((normCapacity & 0xFFFFFE00) == 0) { // < 512
104                 tableIdx = normCapacity >>> 4;
105                 table = tinySubpagePools;
106             } else {
107                 tableIdx = 0;
108                 int i = normCapacity >>> 10;
109                 while (i != 0) {
110                     i >>>= 1;
111                     tableIdx ++;
112                 }
113                 table = smallSubpagePools;
114             }
115 
116             synchronized (this) {
117                 final PoolSubpage<T> head = table[tableIdx];
118                 final PoolSubpage<T> s = head.next;
119                 if (s != head) {
120                     assert s.doNotDestroy && s.elemSize == normCapacity;
121                     long handle = s.allocate();
122                     assert handle >= 0;
123                     s.chunk.initBufWithSubpage(buf, handle, reqCapacity);
124                     return;
125                 }
126             }
127         } else if (normCapacity > chunkSize) {
128             allocateHuge(buf, reqCapacity);
129             return;
130         }
131 
132         allocateNormal(buf, reqCapacity, normCapacity);
133     }

135     private synchronized void allocateNormal(PooledByteBuf<T> buf, int reqCapacity, int normCapacity) {
136         if (q050.allocate(buf, reqCapacity, normCapacity) || q025.allocate(buf, reqCapacity, normCapacity) ||
137             q000.allocate(buf, reqCapacity, normCapacity) || qInit.allocate(buf, reqCapacity, normCapacity) ||
138             q075.allocate(buf, reqCapacity, normCapacity) || q100.allocate(buf, reqCapacity, normCapacity)) {
139             return;
140         }
141 
142         // Add a new chunk.
143         PoolChunk<T> c = newChunk(pageSize, maxOrder, pageShifts, chunkSize);
144         long handle = c.allocate(normCapacity);
145         assert handle > 0;
146         c.initBuf(buf, handle, reqCapacity);
147         qInit.add(c);
148     }

162     PoolSubpage<T> findSubpagePoolHead(int elemSize) {
163         int tableIdx;
164         PoolSubpage<T>[] table;
165         if ((elemSize & 0xFFFFFE00) == 0) { // < 512
166             tableIdx = elemSize >>> 4;
167             table = tinySubpagePools;
168         } else {
169             tableIdx = 0;
170             elemSize >>>= 10;
171             while (elemSize != 0) {
172                 elemSize >>>= 1;
173                 tableIdx ++;
174             }
175             table = smallSubpagePools;
176         }
177 
178         return table[tableIdx];
179     }

181     private int normalizeCapacity(int reqCapacity) {
182         if (reqCapacity < 0) {
183             throw new IllegalArgumentException("capacity: " + reqCapacity + " (expected: 0+)");
184         }
185         if (reqCapacity >= chunkSize) {
186             return reqCapacity;
187         }
188 
189         if ((reqCapacity & 0xFFFFFE00) != 0) { // >= 512
190             // Doubled
191 
192             int normalizedCapacity = reqCapacity;
193             normalizedCapacity |= normalizedCapacity >>>  1;
194             normalizedCapacity |= normalizedCapacity >>>  2;
195             normalizedCapacity |= normalizedCapacity >>>  4;
196             normalizedCapacity |= normalizedCapacity >>>  8;
197             normalizedCapacity |= normalizedCapacity >>> 16;
198             normalizedCapacity ++;
199 
200             if (normalizedCapacity < 0) {
201                 normalizedCapacity >>>= 1;
202             }
203 
204             return normalizedCapacity;
205         }
206 
207         // Quantum-spaced
208         if ((reqCapacity & 15) == 0) {
209             return reqCapacity;
210         }
211 
212         return (reqCapacity & ~15) + 16;
213     }

215     void reallocate(PooledByteBuf<T> buf, int newCapacity, boolean freeOldMemory) {
216         if (newCapacity < 0 || newCapacity > buf.maxCapacity()) {
217             throw new IllegalArgumentException("newCapacity: " + newCapacity);
218         }
219 
220         int oldCapacity = buf.length;
221         if (oldCapacity == newCapacity) {
222             return;
223         }
224 
225         PoolChunk<T> oldChunk = buf.chunk;
226         long oldHandle = buf.handle;
227         T oldMemory = buf.memory;
228         int oldOffset = buf.offset;
229 
230         int readerIndex = buf.readerIndex();
231         int writerIndex = buf.writerIndex();
232 
233         allocate(parent.threadCache.get(), buf, newCapacity);
234         if (newCapacity > oldCapacity) {
235             memoryCopy(
236                     oldMemory, oldOffset + readerIndex,
237                     buf.memory, buf.offset + readerIndex, writerIndex - readerIndex);
238         } else if (newCapacity < oldCapacity) {
239             if (readerIndex < newCapacity) {
240                 if (writerIndex > newCapacity) {
241                     writerIndex = newCapacity;
242                 }
243                 memoryCopy(
244                         oldMemory, oldOffset + readerIndex,
245                         buf.memory, buf.offset + readerIndex, writerIndex - readerIndex);
246             } else {
247                 readerIndex = writerIndex = newCapacity;
248             }
249         }
250 
251         buf.setIndex(readerIndex, writerIndex);
252 
253         if (freeOldMemory) {
254             free(oldChunk, oldHandle);
255         }
256     }

370     static final class DirectArena extends PoolArena<ByteBuffer> {
371 
372         private static final boolean HAS_UNSAFE = PlatformDependent.hasUnsafe();
373 
374         DirectArena(PooledByteBufAllocator parent, int pageSize, int maxOrder, int pageShifts, int chunkSize) {
375             super(parent, pageSize, maxOrder, pageShifts, chunkSize);
376         }
377 
378         @Override
379         protected PoolChunk<ByteBuffer> newChunk(int pageSize, int maxOrder, int pageShifts, int chunkSize) {
380             return new PoolChunk<ByteBuffer>(
381                     this, ByteBuffer.allocateDirect(chunkSize), pageSize, maxOrder, pageShifts, chunkSize);
382         }
383 
384         @Override
385         protected PoolChunk<ByteBuffer> newUnpooledChunk(int capacity) {
386             return new PoolChunk<ByteBuffer>(this, ByteBuffer.allocateDirect(capacity), capacity);
387         }
388 
389         @Override
390         protected void destroyChunk(PoolChunk<ByteBuffer> chunk) {
391             PlatformDependent.freeDirectBuffer(chunk.memory);
392         }
393 
394         @Override
395         protected PooledByteBuf<ByteBuffer> newByteBuf(int maxCapacity) {
396             if (HAS_UNSAFE) {
397                 return PooledUnsafeDirectByteBuf.newInstance(maxCapacity);
398             } else {
399                 return PooledDirectByteBuf.newInstance(maxCapacity);
400             }
401         }
402 
403         @Override
404         protected void memoryCopy(ByteBuffer src, int srcOffset, ByteBuffer dst, int dstOffset, int length) {
405             if (length == 0) {
406                 return;
407             }
408 
409             if (HAS_UNSAFE) {
410                 PlatformDependent.copyMemory(
411                         PlatformDependent.directBufferAddress(src) + srcOffset,
412                         PlatformDependent.directBufferAddress(dst) + dstOffset, length);
413             } else {
414                 // We must duplicate the NIO buffers because they may be accessed by other Netty buffers.
415                 src = src.duplicate();
416                 dst = dst.duplicate();
417                 src.position(srcOffset).limit(srcOffset + length);
418                 dst.position(dstOffset);
419                 dst.put(src);
420             }
421         }
