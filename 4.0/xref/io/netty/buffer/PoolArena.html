49      protected PoolArena(PooledByteBufAllocator parent, int pageSize, int maxOrder, int pageShifts, int chunkSize) {
50          this.parent = parent;
51          this.pageSize = pageSize;
52          this.maxOrder = maxOrder;
53          this.pageShifts = pageShifts;
54          this.chunkSize = chunkSize;
55          subpageOverflowMask = ~(pageSize - 1);
56          tinySubpagePools = newSubpagePoolArray(numTinySubpagePools);
57          for (int i = 0; i < tinySubpagePools.length; i ++) {
58              tinySubpagePools[i] = newSubpagePoolHead(pageSize);
59          }
60  
61          numSmallSubpagePools = pageShifts - 9;
62          smallSubpagePools = newSubpagePoolArray(numSmallSubpagePools);
63          for (int i = 0; i < smallSubpagePools.length; i ++) {
64              smallSubpagePools[i] = newSubpagePoolHead(pageSize);
65          }
66  
67          q100 = new PoolChunkList<T>(this, null, 100, Integer.MAX_VALUE);
68          q075 = new PoolChunkList<T>(this, q100, 75, 100);
69          q050 = new PoolChunkList<T>(this, q075, 50, 100);
70          q025 = new PoolChunkList<T>(this, q050, 25, 75);
71          q000 = new PoolChunkList<T>(this, q025, 1, 50);
72          qInit = new PoolChunkList<T>(this, q000, Integer.MIN_VALUE, 25);
73  
74          q100.prevList = q075;
75          q075.prevList = q050;
76          q050.prevList = q025;
77          q025.prevList = q000;
78          q000.prevList = null;
79          qInit.prevList = qInit;
80      }

82      private PoolSubpage<T> newSubpagePoolHead(int pageSize) {
83          PoolSubpage<T> head = new PoolSubpage<T>(pageSize);
84          head.prev = head;
85          head.next = head;

106     static int smallIdx(int normCapacity) {
107         int tableIdx = 0;
108         int i = normCapacity >>> 10;
109         while (i != 0) {
110             i >>>= 1;
111             tableIdx ++;
112         }
113         return tableIdx;

116     // capacity < pageSize
117     boolean isTinyOrSmall(int normCapacity) {
118         return (normCapacity & subpageOverflowMask) == 0;

122     static boolean isTiny(int normCapacity) {
123         return (normCapacity & 0xFFFFFE00) == 0;

126     private void allocate(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity) {
127         final int normCapacity = normalizeCapacity(reqCapacity);
128         if (isTinyOrSmall(normCapacity)) { // capacity < pageSize
129             int tableIdx;
130             PoolSubpage<T>[] table;
131             if (isTiny(normCapacity)) { // < 512
132                 if (cache.allocateTiny(this, buf, reqCapacity, normCapacity)) {
133                     // was able to allocate out of the cache so move on
134                     return;
135                 }
136                 tableIdx = tinyIdx(normCapacity);
137                 table = tinySubpagePools;
138             } else {
139                 if (cache.allocateSmall(this, buf, reqCapacity, normCapacity)) {
140                     // was able to allocate out of the cache so move on
141                     return;
142                 }
143                 tableIdx = smallIdx(normCapacity);
144                 table = smallSubpagePools;
145             }
146 
147             synchronized (this) {
148                 final PoolSubpage<T> head = table[tableIdx];
149                 final PoolSubpage<T> s = head.next;
150                 if (s != head) {
151                     assert s.doNotDestroy && s.elemSize == normCapacity;
152                     long handle = s.allocate();
153                     assert handle >= 0;
154                     s.chunk.initBufWithSubpage(buf, handle, reqCapacity);
155                     return;
156                 }
157             }
158         } else if (normCapacity <= chunkSize) {
159             if (cache.allocateNormal(this, buf, reqCapacity, normCapacity)) {
160                 // was able to allocate out of the cache so move on
161                 return;
162             }
163         } else {
164             // Huge allocations are never served via the cache so just call allocateHuge
165             allocateHuge(buf, reqCapacity);
166             return;
167         }
168         allocateNormal(buf, reqCapacity, normCapacity);
169     }

171     private synchronized void allocateNormal(PooledByteBuf<T> buf, int reqCapacity, int normCapacity) {
172         if (q050.allocate(buf, reqCapacity, normCapacity) || q025.allocate(buf, reqCapacity, normCapacity) ||
173             q000.allocate(buf, reqCapacity, normCapacity) || qInit.allocate(buf, reqCapacity, normCapacity) ||
174             q075.allocate(buf, reqCapacity, normCapacity) || q100.allocate(buf, reqCapacity, normCapacity)) {
175             return;
176         }
177 
178         // Add a new chunk.
179         PoolChunk<T> c = newChunk(pageSize, maxOrder, pageShifts, chunkSize);
180         long handle = c.allocate(normCapacity);
181         assert handle > 0;
182         c.initBuf(buf, handle, reqCapacity);
183         qInit.add(c);
184     }

186     private void allocateHuge(PooledByteBuf<T> buf, int reqCapacity) {
187         buf.initUnpooled(newUnpooledChunk(reqCapacity), reqCapacity);

190     void free(PoolChunk<T> chunk, long handle, int normCapacity) {
191         if (chunk.unpooled) {
192             destroyChunk(chunk);
193         } else {
194             PoolThreadCache cache = parent.threadCache.get();
195             if (cache.add(this, chunk, handle, normCapacity)) {
196                 // cached so not free it.
197                 return;
198             }
199             synchronized (this) {
200                 chunk.parent.free(chunk, handle);
201             }
202         }
203     }

205     PoolSubpage<T> findSubpagePoolHead(int elemSize) {
206         int tableIdx;
207         PoolSubpage<T>[] table;
208         if (isTiny(elemSize)) { // < 512
209             tableIdx = elemSize >>> 4;
210             table = tinySubpagePools;
211         } else {
212             tableIdx = 0;
213             elemSize >>>= 10;
214             while (elemSize != 0) {
215                 elemSize >>>= 1;
216                 tableIdx ++;
217             }
218             table = smallSubpagePools;
219         }
220 
221         return table[tableIdx];
222     }

224     int normalizeCapacity(int reqCapacity) {
225         if (reqCapacity < 0) {
226             throw new IllegalArgumentException("capacity: " + reqCapacity + " (expected: 0+)");
227         }
228         if (reqCapacity >= chunkSize) {
229             return reqCapacity;
230         }
231 
232         if (!isTiny(reqCapacity)) { // >= 512
233             // Doubled
234 
235             int normalizedCapacity = reqCapacity;
236             normalizedCapacity |= normalizedCapacity >>>  1;
237             normalizedCapacity |= normalizedCapacity >>>  2;
238             normalizedCapacity |= normalizedCapacity >>>  4;
239             normalizedCapacity |= normalizedCapacity >>>  8;
240             normalizedCapacity |= normalizedCapacity >>> 16;
241             normalizedCapacity ++;
242 
243             if (normalizedCapacity < 0) {
244                 normalizedCapacity >>>= 1;
245             }
246 
247             return normalizedCapacity;
248         }
249 
250         // Quantum-spaced
251         if ((reqCapacity & 15) == 0) {
252             return reqCapacity;
253         }
254 
255         return (reqCapacity & ~15) + 16;
256     }

258     void reallocate(PooledByteBuf<T> buf, int newCapacity, boolean freeOldMemory) {
259         if (newCapacity < 0 || newCapacity > buf.maxCapacity()) {
260             throw new IllegalArgumentException("newCapacity: " + newCapacity);
261         }
262 
263         int oldCapacity = buf.length;
264         if (oldCapacity == newCapacity) {
265             return;
266         }
267 
268         PoolChunk<T> oldChunk = buf.chunk;
269         long oldHandle = buf.handle;
270         T oldMemory = buf.memory;
271         int oldOffset = buf.offset;
272         int oldMaxLength = buf.maxLength;
273         int readerIndex = buf.readerIndex();
274         int writerIndex = buf.writerIndex();
275 
276         allocate(parent.threadCache.get(), buf, newCapacity);
277         if (newCapacity > oldCapacity) {
278             memoryCopy(
279                     oldMemory, oldOffset,
280                     buf.memory, buf.offset, oldCapacity);
281         } else if (newCapacity < oldCapacity) {
282             if (readerIndex < newCapacity) {
283                 if (writerIndex > newCapacity) {
284                     writerIndex = newCapacity;
285                 }
286                 memoryCopy(
287                         oldMemory, oldOffset + readerIndex,
288                         buf.memory, buf.offset + readerIndex, writerIndex - readerIndex);
289             } else {
290                 readerIndex = writerIndex = newCapacity;
291             }
292         }
293 
294         buf.setIndex(readerIndex, writerIndex);
295 
296         if (freeOldMemory) {
297             free(oldChunk, oldHandle, oldMaxLength);
298         }
299     }

418     static final class DirectArena extends PoolArena<ByteBuffer> {
419 
420         private static final boolean HAS_UNSAFE = PlatformDependent.hasUnsafe();
421 
422         DirectArena(PooledByteBufAllocator parent, int pageSize, int maxOrder, int pageShifts, int chunkSize) {
423             super(parent, pageSize, maxOrder, pageShifts, chunkSize);
424         }
425 
426         @Override
427         boolean isDirect() {
428             return true;
429         }
430 
431         @Override
432         protected PoolChunk<ByteBuffer> newChunk(int pageSize, int maxOrder, int pageShifts, int chunkSize) {
433             return new PoolChunk<ByteBuffer>(
434                     this, ByteBuffer.allocateDirect(chunkSize), pageSize, maxOrder, pageShifts, chunkSize);
435         }
436 
437         @Override
438         protected PoolChunk<ByteBuffer> newUnpooledChunk(int capacity) {
439             return new PoolChunk<ByteBuffer>(this, ByteBuffer.allocateDirect(capacity), capacity);
440         }
441 
442         @Override
443         protected void destroyChunk(PoolChunk<ByteBuffer> chunk) {
444             PlatformDependent.freeDirectBuffer(chunk.memory);
445         }
446 
447         @Override
448         protected PooledByteBuf<ByteBuffer> newByteBuf(int maxCapacity) {
449             if (HAS_UNSAFE) {
450                 return PooledUnsafeDirectByteBuf.newInstance(maxCapacity);
451             } else {
452                 return PooledDirectByteBuf.newInstance(maxCapacity);
453             }
454         }
455 
456         @Override
457         protected void memoryCopy(ByteBuffer src, int srcOffset, ByteBuffer dst, int dstOffset, int length) {
458             if (length == 0) {
459                 return;
460             }
461 
462             if (HAS_UNSAFE) {
463                 PlatformDependent.copyMemory(
464                         PlatformDependent.directBufferAddress(src) + srcOffset,
465                         PlatformDependent.directBufferAddress(dst) + dstOffset, length);
466             } else {
467                 // We must duplicate the NIO buffers because they may be accessed by other Netty buffers.
468                 src = src.duplicate();
469                 dst = dst.duplicate();
470                 src.position(srcOffset).limit(srcOffset + length);
471                 dst.position(dstOffset);
472                 dst.put(src);
473             }
474         }
