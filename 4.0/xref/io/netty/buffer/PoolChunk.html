19  final class PoolChunk<T> {
20      private static final int ST_UNUSED = 0;
21      private static final int ST_BRANCH = 1;
22      private static final int ST_ALLOCATED = 2;
23      private static final int ST_ALLOCATED_SUBPAGE = ST_ALLOCATED | 1;

54      PoolChunk(PoolArena<T> arena, T memory, int pageSize, int maxOrder, int pageShifts, int chunkSize) {
55          unpooled = false;
56          this.arena = arena;
57          this.memory = memory;
58          this.pageSize = pageSize;
59          this.pageShifts = pageShifts;
60          this.chunkSize = chunkSize;
61          subpageOverflowMask = ~(pageSize - 1);
62          freeBytes = chunkSize;
63  
64          int chunkSizeInPages = chunkSize >>> pageShifts;
65          maxSubpageAllocs = 1 << maxOrder;
66  
67          // Generate the memory map.
68          memoryMap = new int[maxSubpageAllocs << 1];
69          int memoryMapIndex = 1;
70          for (int i = 0; i <= maxOrder; i ++) {
71              int runSizeInPages = chunkSizeInPages >>> i;
72              for (int j = 0; j < chunkSizeInPages; j += runSizeInPages) {
73                  //noinspection PointlessBitwiseExpression
74                  memoryMap[memoryMapIndex ++] = j << 17 | runSizeInPages << 2 | ST_UNUSED;
75              }
76          }
77  
78          subpages = newSubpageArray(maxSubpageAllocs);
79      }

89          pageSize = 0;
90          pageShifts = 0;
91          chunkSize = size;

100     int usage() {
101         if (freeBytes == 0) {
102             return 100;
103         }
104 
105         int freePercentage = (int) (freeBytes * 100L / chunkSize);
106         if (freePercentage == 0) {
107             return 99;
108         }
109         return 100 - freePercentage;
110     }

112     long allocate(int normCapacity) {
113         int firstVal = memoryMap[1];
114         if ((normCapacity & subpageOverflowMask) != 0) { // >= pageSize
115             return allocateRun(normCapacity, 1, firstVal);
116         } else {
117             return allocateSubpage(normCapacity, 1, firstVal);
118         }
119     }

121     private long allocateRun(int normCapacity, int curIdx, int val) {
122         for (;;) {
123             if ((val & ST_ALLOCATED) != 0) { // state == ST_ALLOCATED || state == ST_ALLOCATED_SUBPAGE
124                 return -1;
125             }
126 
127             if ((val & ST_BRANCH) != 0) { // state == ST_BRANCH
128                 int nextIdx = curIdx << 1 ^ nextRandom();
129                 long res = allocateRun(normCapacity, nextIdx, memoryMap[nextIdx]);
130                 if (res > 0) {
131                     return res;
132                 }
133 
134                 curIdx = nextIdx ^ 1;
135                 val = memoryMap[curIdx];
136                 continue;
137             }
138 
139             // state == ST_UNUSED
140             return allocateRunSimple(normCapacity, curIdx, val);
141         }
142     }
143 
144     private long allocateRunSimple(int normCapacity, int curIdx, int val) {
145         int runLength = runLength(val);
146         if (normCapacity > runLength) {
147             return -1;
148         }
149 
150         for (;;) {
151             if (normCapacity == runLength) {
152                 // Found the run that fits.
153                 // Note that capacity has been normalized already, so we don't need to deal with
154                 // the values that are not power of 2.
155                 memoryMap[curIdx] = val & ~3 | ST_ALLOCATED;
156                 freeBytes -= runLength;
157                 return curIdx;
158             }
159 
160             int nextIdx = curIdx << 1 ^ nextRandom();
161             int unusedIdx = nextIdx ^ 1;
162 
163             memoryMap[curIdx] = val & ~3 | ST_BRANCH;
164             //noinspection PointlessBitwiseExpression
165             memoryMap[unusedIdx] = memoryMap[unusedIdx] & ~3 | ST_UNUSED;
166 
167             runLength >>>= 1;
168             curIdx = nextIdx;
169             val = memoryMap[curIdx];
170         }
171     }

173     private long allocateSubpage(int normCapacity, int curIdx, int val) {
174         int state = val & 3;
175         if (state == ST_BRANCH) {
176             int nextIdx = curIdx << 1 ^ nextRandom();
177             long res = branchSubpage(normCapacity, nextIdx);
178             if (res > 0) {
179                 return res;
180             }
181 
182             return branchSubpage(normCapacity, nextIdx ^ 1);
183         }
184 
185         if (state == ST_UNUSED) {
186             return allocateSubpageSimple(normCapacity, curIdx, val);
187         }
188 
189         if (state == ST_ALLOCATED_SUBPAGE) {
190             PoolSubpage<T> subpage = subpages[subpageIdx(curIdx)];
191             int elemSize = subpage.elemSize;
192             if (normCapacity != elemSize) {
193                 return -1;
194             }
195 
196             return subpage.allocate();
197         }
198 
199         return -1;
200     }

202     private long allocateSubpageSimple(int normCapacity, int curIdx, int val) {
203         int runLength = runLength(val);
204         for (;;) {
205             if (runLength == pageSize) {
206                 memoryMap[curIdx] = val & ~3 | ST_ALLOCATED_SUBPAGE;
207                 freeBytes -= runLength;
208 
209                 int subpageIdx = subpageIdx(curIdx);
210                 PoolSubpage<T> subpage = subpages[subpageIdx];
211                 if (subpage == null) {
212                     subpage = new PoolSubpage<T>(this, curIdx, runOffset(val), pageSize, normCapacity);
213                     subpages[subpageIdx] = subpage;
214                 } else {
215                     subpage.init(normCapacity);
216                 }
217                 arena.addSubpage(subpage);
218                 return subpage.allocate();
219             }
220 
221             int nextIdx = curIdx << 1 ^ nextRandom();
222             int unusedIdx = nextIdx ^ 1;
223 
224             memoryMap[curIdx] = val & ~3 | ST_BRANCH;
225             //noinspection PointlessBitwiseExpression
226             memoryMap[unusedIdx] = memoryMap[unusedIdx] & ~3 | ST_UNUSED;
227 
228             runLength >>>= 1;
229             curIdx = nextIdx;
230             val = memoryMap[curIdx];
231         }
232     }

234     private long branchSubpage(int normCapacity, int nextIdx) {
235         int nextVal = memoryMap[nextIdx];
236         if ((nextVal & 3) != ST_ALLOCATED) {
237             return allocateSubpage(normCapacity, nextIdx, nextVal);
238         }
239         return -1;
240     }

242     void free(long handle) {
243         int memoryMapIdx = (int) handle;
244         int bitmapIdx = (int) (handle >>> 32);
245 
246         int val = memoryMap[memoryMapIdx];
247         int state = val & 3;
248         if (state == ST_ALLOCATED_SUBPAGE) {
249             assert bitmapIdx != 0;
250             PoolSubpage<T> subpage = subpages[subpageIdx(memoryMapIdx)];
251             assert subpage != null && subpage.doNotDestroy;
252             if (subpage.free(bitmapIdx & 0x3FFFFFFF)) {
253                 return;
254             }
255         } else {
256             assert state == ST_ALLOCATED : "state: " + state;
257             assert bitmapIdx == 0;
258         }
259 
260         freeBytes += runLength(val);
261 
262         for (;;) {
263             //noinspection PointlessBitwiseExpression
264             memoryMap[memoryMapIdx] = val & ~3 | ST_UNUSED;
265             if (memoryMapIdx == 1) {
266                 assert freeBytes == chunkSize;
267                 return;
268             }
269 
270             if ((memoryMap[siblingIdx(memoryMapIdx)] & 3) != ST_UNUSED) {
271                 break;
272             }
273 
274             memoryMapIdx = parentIdx(memoryMapIdx);
275             val = memoryMap[memoryMapIdx];
276         }
277     }

279     void initBuf(PooledByteBuf<T> buf, long handle, int reqCapacity) {
280         int memoryMapIdx = (int) handle;
281         int bitmapIdx = (int) (handle >>> 32);
282         if (bitmapIdx == 0) {
283             int val = memoryMap[memoryMapIdx];
284             assert (val & 3) == ST_ALLOCATED : String.valueOf(val & 3);
285             buf.init(this, handle, runOffset(val), reqCapacity, runLength(val));
286         } else {
287             initBufWithSubpage(buf, handle, bitmapIdx, reqCapacity);
288         }
289     }
290 
291     void initBufWithSubpage(PooledByteBuf<T> buf, long handle, int reqCapacity) {
292         initBufWithSubpage(buf, handle, (int) (handle >>> 32), reqCapacity);
293     }

295     private void initBufWithSubpage(PooledByteBuf<T> buf, long handle, int bitmapIdx, int reqCapacity) {
296         assert bitmapIdx != 0;
297 
298         int memoryMapIdx = (int) handle;
299         int val = memoryMap[memoryMapIdx];
300         assert (val & 3) == ST_ALLOCATED_SUBPAGE;
301 
302         PoolSubpage<T> subpage = subpages[subpageIdx(memoryMapIdx)];
303         assert subpage.doNotDestroy;
304         assert reqCapacity <= subpage.elemSize;
305 
306         buf.init(
307                 this, handle,
308                 runOffset(val) + (bitmapIdx & 0x3FFFFFFF) * subpage.elemSize, reqCapacity, subpage.elemSize);
309     }

311     private static int parentIdx(int memoryMapIdx) {
312         return memoryMapIdx >>> 1;
313     }

315     private static int siblingIdx(int memoryMapIdx) {
316         return memoryMapIdx ^ 1;
317     }

319     private int runLength(int val) {
320         return (val >>> 2 & 0x7FFF) << pageShifts;
321     }
322 
323     private int runOffset(int val) {
324         return val >>> 17 << pageShifts;
325     }
326 
327     private int subpageIdx(int memoryMapIdx) {
328         return memoryMapIdx - maxSubpageAllocs;
329     }
330 
331     private int nextRandom() {
332         random = random * multiplier + addend & mask;
333         return (int) (random >>> 47) & 1;
334     }

336     public String toString() {
337         StringBuilder buf = new StringBuilder();
338         buf.append("Chunk(");
339         buf.append(Integer.toHexString(System.identityHashCode(this)));
340         buf.append(": ");
341         buf.append(usage());
342         buf.append("%, ");
343         buf.append(chunkSize - freeBytes);
344         buf.append('/');
345         buf.append(chunkSize);
346         buf.append(')');
347         return buf.toString();
348     }
