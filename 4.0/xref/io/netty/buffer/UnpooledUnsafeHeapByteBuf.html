159         if (PlatformDependent.javaVersion() >= 7) {
160             // Only do on java7+ as the needed Unsafe call was only added there.
161             ensureWritable(length);
162             int wIndex = writerIndex;
163             _setZero(wIndex, length);
164             writerIndex = wIndex + length;
165             return this;
166         }
167         return super.writeZero(length);
168     }
