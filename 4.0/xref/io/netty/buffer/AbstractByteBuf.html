186     @Override
187     public ByteBuf discardReadBytes() {
188         ensureAccessible();
189         if (readerIndex == 0) {
190             return this;
191         }
192 
193         if (readerIndex != writerIndex) {
194             setBytes(0, this, readerIndex, writerIndex - readerIndex);
195             writerIndex -= readerIndex;
196             adjustMarkers(readerIndex);
197             readerIndex = 0;
198         } else {
199             adjustMarkers(readerIndex);
200             writerIndex = readerIndex = 0;
201         }
202         return this;
203     }

205     @Override
206     public ByteBuf discardSomeReadBytes() {
207         ensureAccessible();
208         if (readerIndex == 0) {
209             return this;
210         }
211 
212         if (readerIndex == writerIndex) {
213             adjustMarkers(readerIndex);
214             writerIndex = readerIndex = 0;
215             return this;
216         }
217 
218         if (readerIndex >= capacity() >>> 1) {
219             setBytes(0, this, readerIndex, writerIndex - readerIndex);
220             writerIndex -= readerIndex;
221             adjustMarkers(readerIndex);
222             readerIndex = 0;
223         }
224         return this;
225     }

227     protected final void adjustMarkers(int decrement) {
228         int markedReaderIndex = this.markedReaderIndex;
229         if (markedReaderIndex <= decrement) {
230             this.markedReaderIndex = 0;
231             int markedWriterIndex = this.markedWriterIndex;
232             if (markedWriterIndex <= decrement) {
233                 this.markedWriterIndex = 0;
234             } else {
235                 this.markedWriterIndex = markedWriterIndex - decrement;
236             }
237         } else {
238             this.markedReaderIndex = markedReaderIndex - decrement;
239             markedWriterIndex -= decrement;
240         }
241     }

253     private void ensureWritable0(int minWritableBytes) {
254         if (minWritableBytes <= writableBytes()) {
255             return;
256         }
257 
258         if (minWritableBytes > maxCapacity - writerIndex) {
259             throw new IndexOutOfBoundsException(String.format(
260                     "writerIndex(%d) + minWritableBytes(%d) exceeds maxCapacity(%d): %s",
261                     writerIndex, minWritableBytes, maxCapacity, this));
262         }
263 
264         // Normalize the current capacity to the power of 2.
265         int newCapacity = calculateNewCapacity(writerIndex + minWritableBytes);
266 
267         // Adjust to the new capacity.
268         capacity(newCapacity);
269     }

271     @Override
272     public int ensureWritable(int minWritableBytes, boolean force) {
273         if (minWritableBytes < 0) {
274             throw new IllegalArgumentException(String.format(
275                     "minWritableBytes: %d (expected: >= 0)", minWritableBytes));
276         }
277 
278         if (minWritableBytes <= writableBytes()) {
279             return 0;
280         }
281 
282         if (minWritableBytes > maxCapacity - writerIndex) {
283             if (force) {
284                 if (capacity() == maxCapacity()) {
285                     return 1;
286                 }
287 
288                 capacity(maxCapacity());
289                 return 3;
290             }
291         }
292 
293         // Normalize the current capacity to the power of 2.
294         int newCapacity = calculateNewCapacity(writerIndex + minWritableBytes);
295 
296         // Adjust to the new capacity.
297         capacity(newCapacity);
298         return 2;
299     }
300 
301     private int calculateNewCapacity(int minNewCapacity) {
302         final int maxCapacity = this.maxCapacity;
303         final int threshold = 1048576 * 4; // 4 MiB page
304 
305         if (minNewCapacity == threshold) {
306             return threshold;
307         }
308 
309         // If over threshold, do not double but just increase by threshold.
310         if (minNewCapacity > threshold) {
311             int newCapacity = minNewCapacity / threshold * threshold;
312             if (newCapacity > maxCapacity - threshold) {
313                 newCapacity = maxCapacity;
314             } else {
315                 newCapacity += threshold;
316             }
317             return newCapacity;
318         }
319 
320         // Not over threshold. Double up to 4 MiB, starting from 64.
321         int newCapacity = 64;
322         while (newCapacity < minNewCapacity) {
323             newCapacity <<= 1;
324         }
325 
326         return Math.min(newCapacity, maxCapacity);
327     }

552     @Override
553     public ByteBuf setZero(int index, int length) {
554         if (length == 0) {
555             return this;
556         }
557 
558         checkIndex(index, length);
559 
560         int nLong = length >>> 3;
561         int nBytes = length & 7;
562         for (int i = nLong; i > 0; i --) {
563             _setLong(index, 0);
564             index += 8;
565         }
566         if (nBytes == 4) {
567             _setInt(index, 0);
568             // Not need to update the index as we not will use it after this.
569         } else if (nBytes < 4) {
570             for (int i = nBytes; i > 0; i --) {
571                 _setByte(index, (byte) 0);
572                 index ++;
573             }
574         } else {
575             _setInt(index, 0);
576             index += 4;
577             for (int i = nBytes - 4; i > 0; i --) {
578                 _setByte(index, (byte) 0);
579                 index ++;
580             }
581         }
582         return this;
583     }
