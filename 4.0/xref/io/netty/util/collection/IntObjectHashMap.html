23  /**
24   * A hash map implementation of {@link IntObjectMap} that uses open addressing for keys. To minimize
25   * the memory footprint, this class uses open addressing rather than chaining. Collisions are
26   * resolved using double hashing.
27   *
28   * @param <V> The value type stored in the map.
29   */
30  public class IntObjectHashMap<V> implements IntObjectMap<V>, Iterable<IntObjectMap.Entry<V>> {
31  
32      /** State indicating that a slot is available.*/
33      private static final byte AVAILABLE = 0;
34  
35      /** State indicating that a slot is occupied. */
36      private static final byte OCCUPIED = 1;
37  
38      /** State indicating that a slot was removed. */
39      private static final byte REMOVED = 2;
40  
41      /** Default initial capacity. Used if not specified in the constructor */
42      private static final int DEFAULT_CAPACITY = 11;
43  
44      /** Default load factor. Used if not specified in the constructor */
45      private static final float DEFAULT_LOAD_FACTOR = 0.5f;
46  
47      /** The maximum number of elements allowed without allocating more space. */
48      private int maxSize;
49  
50      /** The load factor for the map. Used to calculate {@link maxSize}. */
51      private final float loadFactor;
52  
53      private byte[] states;
54      private int[] keys;
55      private V[] values;
56      private int size;
57      private int available;
58  
59      public IntObjectHashMap() {
60          this(DEFAULT_CAPACITY, DEFAULT_LOAD_FACTOR);
61      }

63      public IntObjectHashMap(int initialCapacity) {
64          this(initialCapacity, DEFAULT_LOAD_FACTOR);
65      }

67      public IntObjectHashMap(int initialCapacity, float loadFactor) {
68          if (initialCapacity < 1) {
69              throw new IllegalArgumentException("initialCapacity must be >= 1");
70          }
71          if (loadFactor <= 0.0f) {
72              throw new IllegalArgumentException("loadFactor must be > 0");
73          }
74  
75          this.loadFactor = loadFactor;
76  
77          // Allocate the arrays.
78          states = new byte[initialCapacity];
79          keys = new int[initialCapacity];
80          @SuppressWarnings("unchecked")
81          V[] temp = (V[]) new Object[initialCapacity];
82          values = temp;
83  
84          // Initialize the maximum size value.
85          maxSize = calcMaxSize(initialCapacity);
86  
87          // Initialize the available element count
88          available = initialCapacity - size;
89      }

91      @Override
92      public V get(int key) {
93          int index = indexOf(key);
94          return index < 0 ? null : values[index];
95      }
96  
97      @Override
98      public V put(int key, V value) {
99          int hash = hash(key);
100         int capacity = capacity();
101         int index = hash % capacity;
102         int increment = 1 + (hash % (capacity - 2));
103         final int startIndex = index;
104         int firstRemovedIndex = -1;
105         do {
106             switch (states[index]) {
107                 case AVAILABLE:
108                     // We only stop probing at a AVAILABLE node, since the value may still exist
109                     // beyond
110                     // a REMOVED node.
111                     if (firstRemovedIndex != -1) {
112                         // We encountered a REMOVED node prior. Store the entry there so that
113                         // retrieval
114                         // will be faster.
115                         insertAt(firstRemovedIndex, key, value);
116                         return null;
117                     }
118 
119                     // No REMOVED node, just store the entry here.
120                     insertAt(index, key, value);
121                     return null;
122                 case OCCUPIED:
123                     if (keys[index] == key) {
124                         V previousValue = values[index];
125                         insertAt(index, key, value);
126                         return previousValue;
127                     }
128                     break;
129                 case REMOVED:
130                     // Check for first removed index.
131                     if (firstRemovedIndex == -1) {
132                         firstRemovedIndex = index;
133                     }
134                     break;
135                 default:
136                     throw new AssertionError("Invalid state: " + states[index]);
137             }
138 
139             // REMOVED or OCCUPIED but wrong key, keep probing ...
140             index += increment;
141             if (index >= capacity) {
142                 // Handle wrap-around by decrement rather than mod.
143                 index -= capacity;
144             }
145         } while (index != startIndex);
146 
147         if (firstRemovedIndex == -1) {
148             // Should never happen.
149             throw new AssertionError("Unable to insert");
150         }
151 
152         // Never found a AVAILABLE slot, just use the first REMOVED.
153         insertAt(firstRemovedIndex, key, value);
154         return null;
155     }

157     @Override
158     public void putAll(IntObjectMap<V> sourceMap) {
159         if (sourceMap instanceof IntObjectHashMap) {
160             // Optimization - iterate through the arrays.
161             IntObjectHashMap<V> source = (IntObjectHashMap<V>) sourceMap;
162             int i = -1;
163             while ((i = source.nextEntryIndex(i + 1)) >= 0) {
164                 put(source.keys[i], source.values[i]);
165             }
166             return;
167         }
168 
169         // Otherwise, just add each entry.
170         for (Entry<V> entry : sourceMap.entries()) {
171             put(entry.key(), entry.value());
172         }
173     }

177         int index = indexOf(key);
178         if (index < 0) {

184         return prev;

210     @Override
211     public boolean containsValue(V value) {
212         int i = -1;
213         while ((i = nextEntryIndex(i + 1)) >= 0) {
214             V next = values[i];
215             if (value == next || (value != null && value.equals(next))) {
216                 return true;
217             }
218         }
219         return false;

247     /**
248      * Copies the occupied entries from the source to the target array.
249      */
250     private void copyEntries(Object sourceArray, Object targetArray) {
251         int sourceIx = -1;
252         int targetIx = 0;
253         while ((sourceIx = nextEntryIndex(sourceIx + 1)) >= 0) {
254             Object obj = Array.get(sourceArray, sourceIx);
255             Array.set(targetArray, targetIx++, obj);
256         }
257     }

259     /**
260      * Locates the index for the given key. This method probes using double hashing.
261      *
262      * @param key the key for an entry in the map.
263      * @return the index where the key was found, or {@code -1} if no entry is found for that key.
264      */
265     private int indexOf(int key) {
266         int hash = hash(key);
267         int capacity = capacity();
268         int increment = 1 + (hash % (capacity - 2));
269         int index = hash % capacity;
270         int startIndex = index;
271         do {
272             switch(states[index]) {
273                 case AVAILABLE:
274                     // It's available, so no chance that this value exists anywhere in the map.
275                     return -1;
276                 case OCCUPIED:
277                     if (key == keys[index]) {
278                         // Found it!
279                         return index;
280                     }
281                     break;
282                 default:
283                     break;
284             }
285 
286             // REMOVED or OCCUPIED but wrong key, keep probing ...
287             index += increment;
288             if (index >= capacity) {
289                 // Handle wrap-around by decrement rather than mod.
290                 index -= capacity;
291             }
292         } while (index != startIndex);
293 
294         // Got back to the beginning. Not found.
295         return -1;
296     }

299      * Determines the current capacity (i.e. size of the arrays).
300      */

302         return keys.length;
303     }

305     /**
306      * Creates a hash value for the given key.
307      */
308     private int hash(int key) {
309         // Just make sure the integer is positive.
310         return key & Integer.MAX_VALUE;
311     }

313     /**
314      * Performs an insert of the key/value at the given index position. If necessary, performs a
315      * rehash of the map.
316      *
317      * @param index the index at which to insert the key/value
318      * @param key the entry key
319      * @param value the entry value
320      */
321     private void insertAt(int index, int key, V value) {
322         byte state = states[index];
323         if (state != OCCUPIED) {
324             // Added a new mapping, increment the size.
325             size++;
326 
327             if (state == AVAILABLE) {
328                 // Consumed a OCCUPIED slot, decrement the number of available slots.
329                 available--;
330             }
331         }
332 
333         keys[index] = key;
334         values[index] = value;
335         states[index] = OCCUPIED;
336 
337         if (size > maxSize) {
338             // Need to grow the arrays.
339             // TODO: consider using the next prime greater than capacity * 2.
340             rehash(capacity() * 2);
341         } else if (available == 0) {
342             // Open addressing requires that we have at least 1 slot available. Need to refresh
343             // the arrays to clear any removed elements.
344             rehash(capacity());
345         }
346     }
347 
348     /**
349      * Marks the entry at the given index position as {@link REMOVED} and sets the value to
350      * {@code null}.
351      * <p>
352      * TODO: consider performing re-compaction.
353      *
354      * @param index the index position of the element to remove.
355      */
356     private void removeAt(int index) {
357         if (states[index] == OCCUPIED) {
358             size--;
359         }
360         states[index] = REMOVED;
361         values[index] = null;
362     }

364     /**
365      * Calculates the maximum size allowed before rehashing.
366      */
367     private int calcMaxSize(int capacity) {
368         // Clip the upper bound so that there will always be at least one
369         // available slot.
370         int upperBound = capacity - 1;
371         return Math.min(upperBound, (int) (capacity * loadFactor));
372     }

374     /**
375      * Rehashes the map for the given capacity.
376      *
377      * @param newCapacity the new capacity for the map.
378      */
379     private void rehash(int newCapacity) {
380         int oldCapacity = capacity();
381         int[] oldKeys = keys;
382         V[] oldVals = values;
383         byte[] oldStates = states;
384 
385         // New states array is automatically initialized to AVAILABLE (i.e. 0 == AVAILABLE).
386         states = new byte[newCapacity];
387         keys = new int[newCapacity];
388         @SuppressWarnings("unchecked")
389         V[] temp = (V[]) new Object[newCapacity];
390         values = temp;
391 
392         size = 0;
393         available = newCapacity;
394         maxSize = calcMaxSize(newCapacity);
395 
396         // Insert the new states.
397         for (int i = 0; i < oldCapacity; ++i) {
398             if (oldStates[i] == OCCUPIED) {
399                 put(oldKeys[i], oldVals[i]);
400             }
401         }
402     }

404     /**
405      * Returns the next index of the next entry in the map.
406      *
407      * @param index the index at which to begin the search.
408      * @return the index of the next entry, or {@code -1} if not found.
409      */
410     private int nextEntryIndex(int index) {
411         int capacity = capacity();
412         for (; index < capacity; ++index) {
413             if (states[index] == OCCUPIED) {
414                 return index;
415             }
416         }
417         return -1;
418     }

420     /**
421      * Iterator for traversing the entries in this map.
422      */
423     private final class IteratorImpl implements Iterator<Entry<V>> {
424         int prevIndex;
425         int nextIndex;
426 
427         IteratorImpl() {
428             prevIndex = -1;
429             nextIndex = nextEntryIndex(0);
430         }
431 
432         @Override
433         public boolean hasNext() {
434             return nextIndex >= 0;
435         }
436 
437         @Override
438         public Entry<V> next() {
439             if (!hasNext()) {
440                 throw new NoSuchElementException();
441             }
442 
443             prevIndex = nextIndex;
444             nextIndex = nextEntryIndex(nextIndex + 1);
445             return new EntryImpl(prevIndex);
446         }
447 
448         @Override
449         public void remove() {
450             if (prevIndex < 0) {
451                 throw new IllegalStateException("Next must be called before removing.");
452             }
453             removeAt(prevIndex);
454             prevIndex = -1;
455         }
456     }
