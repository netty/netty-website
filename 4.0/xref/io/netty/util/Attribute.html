16  package io.netty.util;
17  
18  /**
19   * An attribute which allows to store an value reference. It may be updated atomically and so is thread-safe.
20   *
21   * @param <T>   the type of the value it holds.
22   */
23  public interface Attribute<T> {
24      /**
25       * Returns the current value, which may be {@code null}
26       */
27      T get();
28  
29      /**
30       * Sets the value
31       */
32      void set(T value);
33  
34      /**
35       *  Atomically sets to the given value and returns the old value which may be {@code null} if non was set before.
36       */
37      T getAndSet(T value);
38  
39      /**
40       *  Atomically sets to the given value if this {@link Attribute} does not contain a value at the moment.
41       *  If it was not possible to set the value as it contains a value it will just return the current value.
42       */
43      T setIfAbsent(T value);
44  
45      /**
46       * Atomically sets the value to the given updated value if the current value == the expected value.
47       * If it the set was successful it returns {@code true} otherwise {@code false}.
48       */
49      boolean compareAndSet(T oldValue, T newValue);
50  
51      /**
52       * Remove the current value which is stored in this {@link Attribute}, which means after this call {@link #get()}
53       * will return {@code null}.
54       */
55      void remove();
56  }
