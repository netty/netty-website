35  /**
36   * A {@link Timer} optimized for approximated I/O timeout scheduling.
37   *
38   * <h3>Tick Duration</h3>
39   *
40   * As described with 'approximated', this timer does not execute the scheduled
41   * {@link TimerTask} on time.  {@link HashedWheelTimer}, on every tick, will
42   * check if there are any {@link TimerTask}s behind the schedule and execute
43   * them.
44   * <p>
45   * You can increase or decrease the accuracy of the execution timing by
46   * specifying smaller or larger tick duration in the constructor.  In most
47   * network applications, I/O timeout does not need to be accurate.  Therefore,
48   * the default tick duration is 100 milliseconds and you will not need to try
49   * different configurations in most cases.
50   *
51   * <h3>Ticks per Wheel (Wheel Size)</h3>
52   *
53   * {@link HashedWheelTimer} maintains a data structure called 'wheel'.
54   * To put simply, a wheel is a hash table of {@link TimerTask}s whose hash
55   * function is 'dead line of the task'.  The default number of ticks per wheel
56   * (i.e. the size of the wheel) is 512.  You could specify a larger value
57   * if you are going to schedule a lot of timeouts.
58   *
59   * <h3>Do not create many instances.</h3>
60   *
61   * {@link HashedWheelTimer} creates a new thread whenever it is instantiated and
62   * started.  Therefore, you should make sure to create only one instance and
63   * share it across your application.  One of the common mistakes, that makes
64   * your application unresponsive, is to create a new instance for every connection.
65   *
66   * <h3>Implementation Details</h3>
67   *
68   * {@link HashedWheelTimer} is based on
69   * <a href="http://cseweb.ucsd.edu/users/varghese/">George Varghese</a> and
70   * Tony Lauck's paper,
71   * <a href="http://cseweb.ucsd.edu/users/varghese/PAPERS/twheel.ps.Z">'Hashed
72   * and Hierarchical Timing Wheels: data structures to efficiently implement a
73   * timer facility'</a>.  More comprehensive slides are located
74   * <a href="http://www.cse.wustl.edu/~cdgill/courses/cs6874/TimingWheels.ppt">here</a>.
75   */
76  public class HashedWheelTimer implements Timer {

166     /**
167      * Creates a new timer.
168      *
169      * @param threadFactory  a {@link ThreadFactory} that creates a
170      *                       background {@link Thread} which is dedicated to
171      *                       {@link TimerTask} execution.
172      * @param tickDuration   the duration between tick
173      * @param unit           the time unit of the {@code tickDuration}
174      * @param ticksPerWheel  the size of the wheel
175      * @throws NullPointerException     if either of {@code threadFactory} and {@code unit} is {@code null}
176      * @throws IllegalArgumentException if either of {@code tickDuration} and {@code ticksPerWheel} is <= 0
177      */
178     public HashedWheelTimer(
179             ThreadFactory threadFactory,
180             long tickDuration, TimeUnit unit, int ticksPerWheel) {
181 
182         if (threadFactory == null) {
183             throw new NullPointerException("threadFactory");
184         }
185         if (unit == null) {
186             throw new NullPointerException("unit");
187         }
188         if (tickDuration <= 0) {
189             throw new IllegalArgumentException("tickDuration must be greater than 0: " + tickDuration);
190         }
191         if (ticksPerWheel <= 0) {
192             throw new IllegalArgumentException("ticksPerWheel must be greater than 0: " + ticksPerWheel);
193         }
194 
195         // Normalize ticksPerWheel to power of two and initialize the wheel.
196         wheel = createWheel(ticksPerWheel);
197         mask = wheel.length - 1;
198 
199         // Convert tickDuration to milliseconds.
200         this.tickDuration = tickDuration = unit.toMillis(tickDuration);
201 
202         // Prevent overflow.
203         if (tickDuration == Long.MAX_VALUE || tickDuration >= Long.MAX_VALUE / wheel.length) {
204             throw new IllegalArgumentException("tickDuration is too long: " + tickDuration + ' ' + unit);
205         }
206 
207         workerThread = threadFactory.newThread(worker);
208     }

210     @SuppressWarnings("unchecked")
211     private static Set<HashedWheelTimeout>[] createWheel(int ticksPerWheel) {
212         if (ticksPerWheel <= 0) {
213             throw new IllegalArgumentException(
214                     "ticksPerWheel must be greater than 0: " + ticksPerWheel);
215         }
216         if (ticksPerWheel > 1073741824) {
217             throw new IllegalArgumentException(
218                     "ticksPerWheel may not be greater than 2^30: " + ticksPerWheel);
219         }
220 
221         ticksPerWheel = normalizeTicksPerWheel(ticksPerWheel);
222         Set<HashedWheelTimeout>[] wheel = new Set[ticksPerWheel];
223         for (int i = 0; i < wheel.length; i ++) {
224             wheel[i] = Collections.newSetFromMap(
225                     PlatformDependent.<HashedWheelTimeout, Boolean>newConcurrentHashMap());
226         }
227         return wheel;
228     }
229 
230     private static int normalizeTicksPerWheel(int ticksPerWheel) {
231         int normalizedTicksPerWheel = 1;
232         while (normalizedTicksPerWheel < ticksPerWheel) {
233             normalizedTicksPerWheel <<= 1;
234         }
235         return normalizedTicksPerWheel;
236     }

238     /**
239      * Starts the background thread explicitly.  The background thread will
240      * start automatically on demand even if you did not call this method.
241      *
242      * @throws IllegalStateException if this timer has been
243      *                               {@linkplain #stop() stopped} already
244      */
245     public void start() {
246         switch (workerState.get()) {
247             case WORKER_STATE_INIT:
248                 if (workerState.compareAndSet(WORKER_STATE_INIT, WORKER_STATE_STARTED)) {
249                     workerThread.start();
250                 }
251                 break;
252             case WORKER_STATE_STARTED:
253                 break;
254             case WORKER_STATE_SHUTDOWN:
255                 throw new IllegalStateException("cannot be started once stopped");
256             default:
257                 throw new Error("Invalid WorkerState");
258         }
259     }

261     @Override
262     public Set<Timeout> stop() {
263         if (Thread.currentThread() == workerThread) {
264             throw new IllegalStateException(
265                     HashedWheelTimer.class.getSimpleName() +
266                             ".stop() cannot be called from " +
267                             TimerTask.class.getSimpleName());
268         }
269 
270         if (!workerState.compareAndSet(WORKER_STATE_STARTED, WORKER_STATE_SHUTDOWN)) {
271             // workerState can be 0 or 2 at this moment - let it always be 2.
272             workerState.set(WORKER_STATE_SHUTDOWN);
273             return Collections.emptySet();
274         }
275 
276         boolean interrupted = false;
277         while (workerThread.isAlive()) {
278             workerThread.interrupt();
279             try {
280                 workerThread.join(100);
281             } catch (InterruptedException e) {
282                 interrupted = true;
283             }
284         }
285 
286         if (interrupted) {
287             Thread.currentThread().interrupt();
288         }
289 
290         leak.close();
291 
292         Set<Timeout> unprocessedTimeouts = new HashSet<Timeout>();
293         for (Set<HashedWheelTimeout> bucket: wheel) {
294             unprocessedTimeouts.addAll(bucket);
295             bucket.clear();
296         }
297 
298         return Collections.unmodifiableSet(unprocessedTimeouts);
299     }

301     @Override
302     public Timeout newTimeout(TimerTask task, long delay, TimeUnit unit) {
303         final long currentTime = System.currentTimeMillis();
304 
305         if (task == null) {
306             throw new NullPointerException("task");
307         }
308         if (unit == null) {
309             throw new NullPointerException("unit");
310         }
311 
312         start();
313 
314         delay = unit.toMillis(delay);
315         HashedWheelTimeout timeout = new HashedWheelTimeout(task, currentTime + delay);
316         scheduleTimeout(timeout, delay);
317         return timeout;
318     }

320     void scheduleTimeout(HashedWheelTimeout timeout, long delay) {
321         // Prepare the required parameters to schedule the timeout object.
322         long relativeIndex = (delay + tickDuration - 1) / tickDuration;
323         // if the previous line had an overflow going on, then we’ll just schedule this timeout
324         // one tick early; that shouldn’t matter since we’re talking 270 years here
325         if (relativeIndex < 0) {
326             relativeIndex = delay / tickDuration;
327         }
328         if (relativeIndex == 0) {
329             relativeIndex = 1;
330         }
331         if ((relativeIndex & mask) == 0) {
332             relativeIndex--;
333         }
334         final long remainingRounds = relativeIndex / wheel.length;
335 
336         // Add the timeout to the wheel.
337         lock.readLock().lock();
338         try {
339             if (workerState.get() == WORKER_STATE_SHUTDOWN) {
340                 throw new IllegalStateException("Cannot enqueue after shutdown");
341             }
342             final int stopIndex = (int) ((wheelCursor + relativeIndex) & mask);
343             timeout.stopIndex = stopIndex;
344             timeout.remainingRounds = remainingRounds;
345             wheel[stopIndex].add(timeout);
346         } finally {
347             lock.readLock().unlock();
348         }
349     }

351     private final class Worker implements Runnable {
352 
353         private long startTime;
354         private long tick;
355 
356         Worker() {
357         }
358 
359         @Override
360         public void run() {
361             List<HashedWheelTimeout> expiredTimeouts =
362                     new ArrayList<HashedWheelTimeout>();
363 
364             startTime = System.currentTimeMillis();
365             tick = 1;
366 
367             while (workerState.get() == WORKER_STATE_STARTED) {
368                 final long deadline = waitForNextTick();
369                 if (deadline > 0) {
370                     fetchExpiredTimeouts(expiredTimeouts, deadline);
371                     notifyExpiredTimeouts(expiredTimeouts);
372                 }
373             }
374         }
375 
376         private void fetchExpiredTimeouts(
377                 List<HashedWheelTimeout> expiredTimeouts, long deadline) {
378 
379             // Find the expired timeouts and decrease the round counter
380             // if necessary.  Note that we don't send the notification
381             // immediately to make sure the listeners are called without
382             // an exclusive lock.
383             lock.writeLock().lock();
384             try {
385                 int newWheelCursor = wheelCursor = wheelCursor + 1 & mask;
386                 fetchExpiredTimeouts(expiredTimeouts, wheel[newWheelCursor].iterator(), deadline);
387             } finally {
388                 lock.writeLock().unlock();
389             }
390         }
391 
392         private void fetchExpiredTimeouts(
393                 List<HashedWheelTimeout> expiredTimeouts,
394                 Iterator<HashedWheelTimeout> i, long deadline) {
395 
396             List<HashedWheelTimeout> slipped = null;
397             while (i.hasNext()) {
398                 HashedWheelTimeout timeout = i.next();
399                 if (timeout.remainingRounds <= 0) {
400                     i.remove();
401                     if (timeout.deadline <= deadline) {
402                         expiredTimeouts.add(timeout);
403                     } else {
404                         // Handle the case where the timeout is put into a wrong
405                         // place, usually one tick earlier.  For now, just add
406                         // it to a temporary list - we will reschedule it in a
407                         // separate loop.
408                         if (slipped == null) {
409                             slipped = new ArrayList<HashedWheelTimeout>();
410                         }
411                         slipped.add(timeout);
412                     }
413                 } else {
414                     timeout.remainingRounds --;
415                 }
416             }
417 
418             // Reschedule the slipped timeouts.
419             if (slipped != null) {
420                 for (HashedWheelTimeout timeout: slipped) {
421                     scheduleTimeout(timeout, timeout.deadline - deadline);
422                 }
423             }
424         }
425 
426         private void notifyExpiredTimeouts(
427                 List<HashedWheelTimeout> expiredTimeouts) {
428             // Notify the expired timeouts.
429             for (int i = expiredTimeouts.size() - 1; i >= 0; i --) {
430                 expiredTimeouts.get(i).expire();
431             }
432 
433             // Clean up the temporary list.
434             expiredTimeouts.clear();
435         }
436 
437         /**
438          * calculate goal nanoTime from startTime and current tick number,
439          * then wait until that goal has been reached.
440          * @return Long.MIN_VALUE if received a shutdown request,
441          * current time otherwise (with Long.MIN_VALUE changed by +1)
442          */
443         private long waitForNextTick() {
444             long deadline = startTime + tickDuration * tick;
445 
446             for (;;) {
447                 final long currentTime = System.currentTimeMillis();
448                 long sleepTimeMs = (deadline - currentTime + 999999) / 1000000;
449 
450                 if (sleepTimeMs <= 0) {
451                     tick += 1;
452                     if (currentTime == Long.MIN_VALUE) {
453                         return -Long.MAX_VALUE;
454                     } else {
455                         return currentTime;
456                     }
457                 }
458 
459                 // Check if we run on windows, as if thats the case we will need
460                 // to round the sleepTime as workaround for a bug that only affect
461                 // the JVM if it runs on windows.
462                 //
463                 // See https://github.com/netty/netty/issues/356
464                 if (PlatformDependent.isWindows()) {
465                     sleepTimeMs = (sleepTimeMs / 10) * 10;
466                 }
467 
468                 try {
469                     Thread.sleep(sleepTimeMs);
470                 } catch (InterruptedException e) {
471                     if (workerState.get() == WORKER_STATE_SHUTDOWN) {
472                         return Long.MIN_VALUE;
473                     }
474                 }
475             }
476         }
477     }

479     private final class HashedWheelTimeout implements Timeout {
480 
481         private static final int ST_INIT = 0;
482         private static final int ST_CANCELLED = 1;
483         private static final int ST_EXPIRED = 2;
484 
485         private final TimerTask task;
486         final long deadline;
487         volatile int stopIndex;
488         volatile long remainingRounds;
489         private final AtomicInteger state = new AtomicInteger(ST_INIT);
490 
491         HashedWheelTimeout(TimerTask task, long deadline) {
492             this.task = task;
493             this.deadline = deadline;
494         }
495 
496         @Override
497         public Timer timer() {
498             return HashedWheelTimer.this;
499         }
500 
501         @Override
502         public TimerTask task() {
503             return task;
504         }
505 
506         @Override
507         public boolean cancel() {
508             if (!state.compareAndSet(ST_INIT, ST_CANCELLED)) {
509                 return false;
510             }
511 
512             wheel[stopIndex].remove(this);
513             return true;
514         }
515 
516         @Override
517         public boolean isCancelled() {
518             return state.get() == ST_CANCELLED;
519         }
520 
521         @Override
522         public boolean isExpired() {
523             return state.get() != ST_INIT;
524         }
525 
526         public void expire() {
527             if (!state.compareAndSet(ST_INIT, ST_EXPIRED)) {
528                 return;
529             }
530 
531             try {
532                 task.run(this);
533             } catch (Throwable t) {
534                 if (logger.isWarnEnabled()) {
535                     logger.warn("An exception was thrown by " + TimerTask.class.getSimpleName() + '.', t);
536                 }
537             }
538         }
539 
540         @Override
541         public String toString() {
542             long currentTime = System.currentTimeMillis();
543             long remaining = deadline - currentTime;
544 
545             StringBuilder buf = new StringBuilder(192);
546             buf.append(getClass().getSimpleName());
547             buf.append('(');
548 
549             buf.append("deadline: ");
550             if (remaining > 0) {
551                 buf.append(remaining);
552                 buf.append(" ms later, ");
553             } else if (remaining < 0) {
554                 buf.append(-remaining);
555                 buf.append(" ms ago, ");
556             } else {
557                 buf.append("now, ");
558             }
559 
560             if (isCancelled()) {
561                 buf.append(", cancelled");
562             }
563 
564             return buf.append(')').toString();
565         }
566     }
