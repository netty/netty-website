37  /**
38   * An {@link java.util.concurrent.ExecutorService} for running {@link ForkJoinTask}s.
39   * A {@code ForkJoinPool} provides the entry point for submissions
40   * from non-{@code ForkJoinTask} clients, as well as management and
41   * monitoring operations.
42   *
43   * <p>A {@code ForkJoinPool} differs from other kinds of {@link
44   * java.util.concurrent.ExecutorService} mainly by virtue of employing
45   * <em>work-stealing</em>: all threads in the pool attempt to find and
46   * execute tasks submitted to the pool and/or created by other active
47   * tasks (eventually blocking waiting for work if none exist). This
48   * enables efficient processing when most tasks spawn other subtasks
49   * (as do most {@code ForkJoinTask}s), as well as when many small
50   * tasks are submitted to the pool from external clients.  Especially
51   * when setting <em>asyncMode</em> to true in constructors, {@code
52   * ForkJoinPool}s may also be appropriate for use with event-style
53   * tasks that are never joined.
54   *
55   * <p>A static {@link #commonPool()} is available and appropriate for
56   * most applications. The common pool is used by any ForkJoinTask that
57   * is not explicitly submitted to a specified pool. Using the common
58   * pool normally reduces resource usage (its threads are slowly
59   * reclaimed during periods of non-use, and reinstated upon subsequent
60   * use).
61   *
62   * <p>For applications that require separate or custom pools, a {@code
63   * ForkJoinPool} may be constructed with a given target parallelism
64   * level; by default, equal to the number of available processors. The
65   * pool attempts to maintain enough active (or available) threads by
66   * dynamically adding, suspending, or resuming internal worker
67   * threads, even if some tasks are stalled waiting to join
68   * others. However, no such adjustments are guaranteed in the face of
69   * blocked I/O or other unmanaged synchronization. The nested {@link
70   * io.netty.util.internal.chmv8.ForkJoinPool.ManagedBlocker} interface enables extension of the kinds of
71   * synchronization accommodated.
72   *
73   * <p>In addition to execution and lifecycle control methods, this
74   * class provides status check methods (for example
75   * {@link #getStealCount}) that are intended to aid in developing,
76   * tuning, and monitoring fork/join applications. Also, method
77   * {@link #toString} returns indications of pool state in a
78   * convenient form for informal monitoring.
79   *
80   * <p>As is the case with other ExecutorServices, there are three
81   * main task execution methods summarized in the following table.
82   * These are designed to be used primarily by clients not already
83   * engaged in fork/join computations in the current pool.  The main
84   * forms of these methods accept instances of {@code ForkJoinTask},
85   * but overloaded forms also allow mixed execution of plain {@code
86   * Runnable}- or {@code Callable}- based activities as well.  However,
87   * tasks that are already executing in a pool should normally instead
88   * use the within-computation forms listed in the table unless using
89   * async event-style tasks that are not usually joined, in which case
90   * there is little difference among choice of methods.
91   *
92   * <table BORDER CELLPADDING=3 CELLSPACING=1>
93   *  <tr>
94   *    <td></td>
95   *    <td ALIGN=CENTER> <b>Call from non-fork/join clients</b></td>
96   *    <td ALIGN=CENTER> <b>Call from within fork/join computations</b></td>
97   *  </tr>
98   *  <tr>
99   *    <td> <b>Arrange async execution</td>
100  *    <td> {@link #execute(ForkJoinTask)}</td>
101  *    <td> {@link ForkJoinTask#fork}</td>
102  *  </tr>
103  *  <tr>
104  *    <td> <b>Await and obtain result</td>
105  *    <td> {@link #invoke(ForkJoinTask)}</td>
106  *    <td> {@link ForkJoinTask#invoke}</td>
107  *  </tr>
108  *  <tr>
109  *    <td> <b>Arrange exec and obtain Future</td>
110  *    <td> {@link #submit(ForkJoinTask)}</td>
111  *    <td> {@link ForkJoinTask#fork} (ForkJoinTasks <em>are</em> Futures)</td>
112  *  </tr>
113  * </table>
114  *
115  * <p>The common pool is by default constructed with default
116  * parameters, but these may be controlled by setting three {@link
117  * System#getProperty system properties} with prefix {@code
118  * java.util.concurrent.ForkJoinPool.common}: {@code parallelism} --
119  * an integer greater than zero, {@code threadFactory} -- the class
120  * name of a {@link io.netty.util.internal.chmv8.ForkJoinPool.ForkJoinWorkerThreadFactory}, and {@code
121  * exceptionHandler} -- the class name of a {@link
122  * java.lang.Thread.UncaughtExceptionHandler
123  * Thread.UncaughtExceptionHandler}. Upon any error in establishing
124  * these settings, default parameters are used.
125  *
126  * <p><b>Implementation notes</b>: This implementation restricts the
127  * maximum number of running threads to 32767. Attempts to create
128  * pools with greater than the maximum number result in
129  * {@code IllegalArgumentException}.
130  *
131  * <p>This implementation rejects submitted tasks (that is, by throwing
132  * {@link java.util.concurrent.RejectedExecutionException}) only when the pool is shut down
133  * or internal resources have been exhausted.
134  *
135  * @since 1.7
136  * @author Doug Lea
137  */

141     /*
142      * Implementation Overview
143      *
144      * This class and its nested classes provide the main
145      * functionality and control for a set of worker threads:
146      * Submissions from non-FJ threads enter into submission queues.
147      * Workers take these tasks and typically split them into subtasks
148      * that may be stolen by other workers.  Preference rules give
149      * first priority to processing tasks from their own queues (LIFO
150      * or FIFO, depending on mode), then to randomized FIFO steals of
151      * tasks in other queues.
152      *
153      * WorkQueues
154      * ==========
155      *
156      * Most operations occur within work-stealing queues (in nested
157      * class WorkQueue).  These are special forms of Deques that
158      * support only three of the four possible end-operations -- push,
159      * pop, and poll (aka steal), under the further constraints that
160      * push and pop are called only from the owning thread (or, as
161      * extended here, under a lock), while poll may be called from
162      * other threads.  (If you are unfamiliar with them, you probably
163      * want to read Herlihy and Shavit's book "The Art of
164      * Multiprocessor programming", chapter 16 describing these in
165      * more detail before proceeding.)  The main work-stealing queue
166      * design is roughly similar to those in the papers "Dynamic
167      * Circular Work-Stealing Deque" by Chase and Lev, SPAA 2005
168      * (http://research.sun.com/scalable/pubs/index.html) and
169      * "Idempotent work stealing" by Michael, Saraswat, and Vechev,
170      * PPoPP 2009 (http://portal.acm.org/citation.cfm?id=1504186).
171      * The main differences ultimately stem from GC requirements that
172      * we null out taken slots as soon as we can, to maintain as small
173      * a footprint as possible even in programs generating huge
174      * numbers of tasks. To accomplish this, we shift the CAS
175      * arbitrating pop vs poll (steal) from being on the indices
176      * ("base" and "top") to the slots themselves.  So, both a
177      * successful pop and poll mainly entail a CAS of a slot from
178      * non-null to null.  Because we rely on CASes of references, we
179      * do not need tag bits on base or top.  They are simple ints as
180      * used in any circular array-based queue (see for example
181      * ArrayDeque).  Updates to the indices must still be ordered in a
182      * way that guarantees that top == base means the queue is empty,
183      * but otherwise may err on the side of possibly making the queue
184      * appear nonempty when a push, pop, or poll have not fully
185      * committed. Note that this means that the poll operation,
186      * considered individually, is not wait-free. One thief cannot
187      * successfully continue until another in-progress one (or, if
188      * previously empty, a push) completes.  However, in the
189      * aggregate, we ensure at least probabilistic non-blockingness.
190      * If an attempted steal fails, a thief always chooses a different
191      * random victim target to try next. So, in order for one thief to
192      * progress, it suffices for any in-progress poll or new push on
193      * any empty queue to complete. (This is why we normally use
194      * method pollAt and its variants that try once at the apparent
195      * base index, else consider alternative actions, rather than
196      * method poll.)
197      *
198      * This approach also enables support of a user mode in which local
199      * task processing is in FIFO, not LIFO order, simply by using
200      * poll rather than pop.  This can be useful in message-passing
201      * frameworks in which tasks are never joined.  However neither
202      * mode considers affinities, loads, cache localities, etc, so
203      * rarely provide the best possible performance on a given
204      * machine, but portably provide good throughput by averaging over
205      * these factors.  (Further, even if we did try to use such
206      * information, we do not usually have a basis for exploiting it.
207      * For example, some sets of tasks profit from cache affinities,
208      * but others are harmed by cache pollution effects.)
209      *
210      * WorkQueues are also used in a similar way for tasks submitted
211      * to the pool. We cannot mix these tasks in the same queues used
212      * for work-stealing (this would contaminate lifo/fifo
213      * processing). Instead, we randomly associate submission queues
214      * with submitting threads, using a form of hashing.  The
215      * ThreadLocal Submitter class contains a value initially used as
216      * a hash code for choosing existing queues, but may be randomly
217      * repositioned upon contention with other submitters.  In
218      * essence, submitters act like workers except that they are
219      * restricted to executing local tasks that they submitted (or in
220      * the case of CountedCompleters, others with the same root task).
221      * However, because most shared/external queue operations are more
222      * expensive than internal, and because, at steady state, external
223      * submitters will compete for CPU with workers, ForkJoinTask.join
224      * and related methods disable them from repeatedly helping to
225      * process tasks if all workers are active.  Insertion of tasks in
226      * shared mode requires a lock (mainly to protect in the case of
227      * resizing) but we use only a simple spinlock (using bits in
228      * field qlock), because submitters encountering a busy queue move
229      * on to try or create other queues -- they block only when
230      * creating and registering new queues.
231      *
232      * Management
233      * ==========
234      *
235      * The main throughput advantages of work-stealing stem from
236      * decentralized control -- workers mostly take tasks from
237      * themselves or each other. We cannot negate this in the
238      * implementation of other management responsibilities. The main
239      * tactic for avoiding bottlenecks is packing nearly all
240      * essentially atomic control state into two volatile variables
241      * that are by far most often read (not written) as status and
242      * consistency checks.
243      *
244      * Field "ctl" contains 64 bits holding all the information needed
245      * to atomically decide to add, inactivate, enqueue (on an event
246      * queue), dequeue, and/or re-activate workers.  To enable this
247      * packing, we restrict maximum parallelism to (1<<15)-1 (which is
248      * far in excess of normal operating range) to allow ids, counts,
249      * and their negations (used for thresholding) to fit into 16bit
250      * fields.
251      *
252      * Field "plock" is a form of sequence lock with a saturating
253      * shutdown bit (similarly for per-queue "qlocks"), mainly
254      * protecting updates to the workQueues array, as well as to
255      * enable shutdown.  When used as a lock, it is normally only very
256      * briefly held, so is nearly always available after at most a
257      * brief spin, but we use a monitor-based backup strategy to
258      * block when needed.
259      *
260      * Recording WorkQueues.  WorkQueues are recorded in the
261      * "workQueues" array that is created upon first use and expanded
262      * if necessary.  Updates to the array while recording new workers
263      * and unrecording terminated ones are protected from each other
264      * by a lock but the array is otherwise concurrently readable, and
265      * accessed directly.  To simplify index-based operations, the
266      * array size is always a power of two, and all readers must
267      * tolerate null slots. Worker queues are at odd indices. Shared
268      * (submission) queues are at even indices, up to a maximum of 64
269      * slots, to limit growth even if array needs to expand to add
270      * more workers. Grouping them together in this way simplifies and
271      * speeds up task scanning.
272      *
273      * All worker thread creation is on-demand, triggered by task
274      * submissions, replacement of terminated workers, and/or
275      * compensation for blocked workers. However, all other support
276      * code is set up to work with other policies.  To ensure that we
277      * do not hold on to worker references that would prevent GC, ALL
278      * accesses to workQueues are via indices into the workQueues
279      * array (which is one source of some of the messy code
280      * constructions here). In essence, the workQueues array serves as
281      * a weak reference mechanism. Thus for example the wait queue
282      * field of ctl stores indices, not references.  Access to the
283      * workQueues in associated methods (for example signalWork) must
284      * both index-check and null-check the IDs. All such accesses
285      * ignore bad IDs by returning out early from what they are doing,
286      * since this can only be associated with termination, in which
287      * case it is OK to give up.  All uses of the workQueues array
288      * also check that it is non-null (even if previously
289      * non-null). This allows nulling during termination, which is
290      * currently not necessary, but remains an option for
291      * resource-revocation-based shutdown schemes. It also helps
292      * reduce JIT issuance of uncommon-trap code, which tends to
293      * unnecessarily complicate control flow in some methods.
294      *
295      * Event Queuing. Unlike HPC work-stealing frameworks, we cannot
296      * let workers spin indefinitely scanning for tasks when none can
297      * be found immediately, and we cannot start/resume workers unless
298      * there appear to be tasks available.  On the other hand, we must
299      * quickly prod them into action when new tasks are submitted or
300      * generated. In many usages, ramp-up time to activate workers is
301      * the main limiting factor in overall performance (this is
302      * compounded at program start-up by JIT compilation and
303      * allocation). So we try to streamline this as much as possible.
304      * We park/unpark workers after placing in an event wait queue
305      * when they cannot find work. This "queue" is actually a simple
306      * Treiber stack, headed by the "id" field of ctl, plus a 15bit
307      * counter value (that reflects the number of times a worker has
308      * been inactivated) to avoid ABA effects (we need only as many
309      * version numbers as worker threads). Successors are held in
310      * field WorkQueue.nextWait.  Queuing deals with several intrinsic
311      * races, mainly that a task-producing thread can miss seeing (and
312      * signalling) another thread that gave up looking for work but
313      * has not yet entered the wait queue. We solve this by requiring
314      * a full sweep of all workers (via repeated calls to method
315      * scan()) both before and after a newly waiting worker is added
316      * to the wait queue. During a rescan, the worker might release
317      * some other queued worker rather than itself, which has the same
318      * net effect. Because enqueued workers may actually be rescanning
319      * rather than waiting, we set and clear the "parker" field of
320      * WorkQueues to reduce unnecessary calls to unpark.  (This
321      * requires a secondary recheck to avoid missed signals.)  Note
322      * the unusual conventions about Thread.interrupts surrounding
323      * parking and other blocking: Because interrupts are used solely
324      * to alert threads to check termination, which is checked anyway
325      * upon blocking, we clear status (using Thread.interrupted)
326      * before any call to park, so that park does not immediately
327      * return due to status being set via some other unrelated call to
328      * interrupt in user code.
329      *
330      * Signalling.  We create or wake up workers only when there
331      * appears to be at least one task they might be able to find and
332      * execute. However, many other threads may notice the same task
333      * and each signal to wake up a thread that might take it. So in
334      * general, pools will be over-signalled.  When a submission is
335      * added or another worker adds a task to a queue that has fewer
336      * than two tasks, they signal waiting workers (or trigger
337      * creation of new ones if fewer than the given parallelism level
338      * -- signalWork), and may leave a hint to the unparked worker to
339      * help signal others upon wakeup).  These primary signals are
340      * buttressed by others (see method helpSignal) whenever other
341      * threads scan for work or do not have a task to process.  On
342      * most platforms, signalling (unpark) overhead time is noticeably
343      * long, and the time between signalling a thread and it actually
344      * making progress can be very noticeably long, so it is worth
345      * offloading these delays from critical paths as much as
346      * possible.
347      *
348      * Trimming workers. To release resources after periods of lack of
349      * use, a worker starting to wait when the pool is quiescent will
350      * time out and terminate if the pool has remained quiescent for a
351      * given period -- a short period if there are more threads than
352      * parallelism, longer as the number of threads decreases. This
353      * will slowly propagate, eventually terminating all workers after
354      * periods of non-use.
355      *
356      * Shutdown and Termination. A call to shutdownNow atomically sets
357      * a plock bit and then (non-atomically) sets each worker's
358      * qlock status, cancels all unprocessed tasks, and wakes up
359      * all waiting workers.  Detecting whether termination should
360      * commence after a non-abrupt shutdown() call requires more work
361      * and bookkeeping. We need consensus about quiescence (i.e., that
362      * there is no more work). The active count provides a primary
363      * indication but non-abrupt shutdown still requires a rechecking
364      * scan for any workers that are inactive but not queued.
365      *
366      * Joining Tasks
367      * =============
368      *
369      * Any of several actions may be taken when one worker is waiting
370      * to join a task stolen (or always held) by another.  Because we
371      * are multiplexing many tasks on to a pool of workers, we can't
372      * just let them block (as in Thread.join).  We also cannot just
373      * reassign the joiner's run-time stack with another and replace
374      * it later, which would be a form of "continuation", that even if
375      * possible is not necessarily a good idea since we sometimes need
376      * both an unblocked task and its continuation to progress.
377      * Instead we combine two tactics:
378      *
379      *   Helping: Arranging for the joiner to execute some task that it
380      *      would be running if the steal had not occurred.
381      *
382      *   Compensating: Unless there are already enough live threads,
383      *      method tryCompensate() may create or re-activate a spare
384      *      thread to compensate for blocked joiners until they unblock.
385      *
386      * A third form (implemented in tryRemoveAndExec) amounts to
387      * helping a hypothetical compensator: If we can readily tell that
388      * a possible action of a compensator is to steal and execute the
389      * task being joined, the joining thread can do so directly,
390      * without the need for a compensation thread (although at the
391      * expense of larger run-time stacks, but the tradeoff is
392      * typically worthwhile).
393      *
394      * The ManagedBlocker extension API can't use helping so relies
395      * only on compensation in method awaitBlocker.
396      *
397      * The algorithm in tryHelpStealer entails a form of "linear"
398      * helping: Each worker records (in field currentSteal) the most
399      * recent task it stole from some other worker. Plus, it records
400      * (in field currentJoin) the task it is currently actively
401      * joining. Method tryHelpStealer uses these markers to try to
402      * find a worker to help (i.e., steal back a task from and execute
403      * it) that could hasten completion of the actively joined task.
404      * In essence, the joiner executes a task that would be on its own
405      * local deque had the to-be-joined task not been stolen. This may
406      * be seen as a conservative variant of the approach in Wagner &
407      * Calder "Leapfrogging: a portable technique for implementing
408      * efficient futures" SIGPLAN Notices, 1993
409      * (http://portal.acm.org/citation.cfm?id=155354). It differs in
410      * that: (1) We only maintain dependency links across workers upon
411      * steals, rather than use per-task bookkeeping.  This sometimes
412      * requires a linear scan of workQueues array to locate stealers,
413      * but often doesn't because stealers leave hints (that may become
414      * stale/wrong) of where to locate them.  It is only a hint
415      * because a worker might have had multiple steals and the hint
416      * records only one of them (usually the most current).  Hinting
417      * isolates cost to when it is needed, rather than adding to
418      * per-task overhead.  (2) It is "shallow", ignoring nesting and
419      * potentially cyclic mutual steals.  (3) It is intentionally
420      * racy: field currentJoin is updated only while actively joining,
421      * which means that we miss links in the chain during long-lived
422      * tasks, GC stalls etc (which is OK since blocking in such cases
423      * is usually a good idea).  (4) We bound the number of attempts
424      * to find work (see MAX_HELP) and fall back to suspending the
425      * worker and if necessary replacing it with another.
426      *
427      * Helping actions for CountedCompleters are much simpler: Method
428      * helpComplete can take and execute any task with the same root
429      * as the task being waited on. However, this still entails some
430      * traversal of completer chains, so is less efficient than using
431      * CountedCompleters without explicit joins.
432      *
433      * It is impossible to keep exactly the target parallelism number
434      * of threads running at any given time.  Determining the
435      * existence of conservatively safe helping targets, the
436      * availability of already-created spares, and the apparent need
437      * to create new spares are all racy, so we rely on multiple
438      * retries of each.  Compensation in the apparent absence of
439      * helping opportunities is challenging to control on JVMs, where
440      * GC and other activities can stall progress of tasks that in
441      * turn stall out many other dependent tasks, without us being
442      * able to determine whether they will ever require compensation.
443      * Even though work-stealing otherwise encounters little
444      * degradation in the presence of more threads than cores,
445      * aggressively adding new threads in such cases entails risk of
446      * unwanted positive feedback control loops in which more threads
447      * cause more dependent stalls (as well as delayed progress of
448      * unblocked threads to the point that we know they are available)
449      * leading to more situations requiring more threads, and so
450      * on. This aspect of control can be seen as an (analytically
451      * intractable) game with an opponent that may choose the worst
452      * (for us) active thread to stall at any time.  We take several
453      * precautions to bound losses (and thus bound gains), mainly in
454      * methods tryCompensate and awaitJoin.
455      *
456      * Common Pool
457      * ===========
458      *
459      * The static common Pool always exists after static
460      * initialization.  Since it (or any other created pool) need
461      * never be used, we minimize initial construction overhead and
462      * footprint to the setup of about a dozen fields, with no nested
463      * allocation. Most bootstrapping occurs within method
464      * fullExternalPush during the first submission to the pool.
465      *
466      * When external threads submit to the common pool, they can
467      * perform some subtask processing (see externalHelpJoin and
468      * related methods).  We do not need to record whether these
469      * submissions are to the common pool -- if not, externalHelpJoin
470      * returns quickly (at the most helping to signal some common pool
471      * workers). These submitters would otherwise be blocked waiting
472      * for completion, so the extra effort (with liberally sprinkled
473      * task status checks) in inapplicable cases amounts to an odd
474      * form of limited spin-wait before blocking in ForkJoinTask.join.
475      *
476      * Style notes
477      * ===========
478      *
479      * There is a lot of representation-level coupling among classes
480      * ForkJoinPool, ForkJoinWorkerThread, and ForkJoinTask.  The
481      * fields of WorkQueue maintain data structures managed by
482      * ForkJoinPool, so are directly accessed.  There is little point
483      * trying to reduce this, since any associated future changes in
484      * representations will need to be accompanied by algorithmic
485      * changes anyway. Several methods intrinsically sprawl because
486      * they must accumulate sets of consistent reads of volatiles held
487      * in local variables.  Methods signalWork() and scan() are the
488      * main bottlenecks, so are especially heavily
489      * micro-optimized/mangled.  There are lots of inline assignments
490      * (of form "while ((local = field) != 0)") which are usually the
491      * simplest way to ensure the required read orderings (which are
492      * sometimes critical). This leads to a "C"-like style of listing
493      * declarations of these locals at the heads of methods or blocks.
494      * There are several occurrences of the unusual "do {} while
495      * (!cas...)"  which is the simplest way to force an update of a
496      * CAS'ed variable. There are also other coding oddities (including
497      * several unnecessary-looking hoisted null checks) that help
498      * some methods perform reasonably even when interpreted (not
499      * compiled).
500      *
501      * The order of declarations in this file is:
502      * (1) Static utility functions
503      * (2) Nested (static) classes
504      * (3) Static fields
505      * (4) Fields, along with constants used when unpacking some of them
506      * (5) Internal control methods
507      * (6) Callbacks and other support for ForkJoinTask methods
508      * (7) Exported methods
509      * (8) Static block initializing statics in minimally dependent order
510      */

587     /**
588      * Queues supporting work-stealing as well as external task
589      * submission. See above for main rationale and algorithms.
590      * Implementation relies heavily on "Unsafe" intrinsics
591      * and selective use of "volatile":
592      *
593      * Field "base" is the index (mod array.length) of the least valid
594      * queue slot, which is always the next position to steal (poll)
595      * from if nonempty. Reads and writes require volatile orderings
596      * but not CAS, because updates are only performed after slot
597      * CASes.
598      *
599      * Field "top" is the index (mod array.length) of the next queue
600      * slot to push to or pop from. It is written only by owner thread
601      * for push, or under lock for external/shared push, and accessed
602      * by other threads only after reading (volatile) base.  Both top
603      * and base are allowed to wrap around on overflow, but (top -
604      * base) (or more commonly -(base - top) to force volatile read of
605      * base before top) still estimates size. The lock ("qlock") is
606      * forced to -1 on termination, causing all further lock attempts
607      * to fail. (Note: we don't need CAS for termination state because
608      * upon pool shutdown, all shared-queues will stop being used
609      * anyway.)  Nearly all lock bodies are set up so that exceptions
610      * within lock bodies are "impossible" (modulo JVM errors that
611      * would cause failure anyway.)
612      *
613      * The array slots are read and written using the emulation of
614      * volatiles/atomics provided by Unsafe. Insertions must in
615      * general use putOrderedObject as a form of releasing store to
616      * ensure that all writes to the task object are ordered before
617      * its publication in the queue.  All removals entail a CAS to
618      * null.  The array is always a power of two. To ensure safety of
619      * Unsafe array operations, all accesses perform explicit null
620      * checks and implicit bounds checks via power-of-two masking.
621      *
622      * In addition to basic queuing support, this class contains
623      * fields described elsewhere to control execution. It turns out
624      * to work better memory-layout-wise to include them in this class
625      * rather than a separate class.
626      *
627      * Performance on most platforms is very sensitive to placement of
628      * instances of both WorkQueues and their arrays -- we absolutely
629      * do not want multiple WorkQueue instances or multiple queue
630      * arrays sharing cache lines. (It would be best for queue objects
631      * and their arrays to share, but there is nothing available to
632      * help arrange that).  Unfortunately, because they are recorded
633      * in a common array, WorkQueue instances are often moved to be
634      * adjacent by garbage collectors. To reduce impact, we use field
635      * padding that works OK on common platforms; this effectively
636      * trades off slightly slower average field access for the sake of
637      * avoiding really bad worst-case access. (Until better JVM
638      * support is in place, this padding is dependent on transient
639      * properties of JVM field layout rules.) We also take care in
640      * allocating, sizing and resizing the array. Non-shared queue
641      * arrays are initialized by workers before use. Others are
642      * allocated on first use.
643      */

647          * Must be a power of two; at least 4, but should be larger to
648          * reduce or eliminate cacheline sharing among queues.

650          * the fact that JVMs often place arrays in locations that
651          * share GC bookkeeping (especially cardmarks) such that
652          * per-write accesses encounter serious memory contention.
653          */

922          * If present, removes from queue and executes the given task,
923          * or any other cancelled task. Returns (true) on any CAS
924          * or consistency check failure so caller can retry.
925          *
926          * @return false if no progress can be made, else true
927          */
928         final boolean tryRemoveAndExec(ForkJoinTask<?> task) {
929             boolean stat = true, removed = false, empty = true;
930             ForkJoinTask<?>[] a; int m, s, b, n;
931             if ((a = array) != null && (m = a.length - 1) >= 0 &&
932                 (n = (s = top) - (b = base)) > 0) {
933                 for (ForkJoinTask<?> t;;) {           // traverse from s to b
934                     int j = ((--s & m) << ASHIFT) + ABASE;
935                     t = (ForkJoinTask<?>)U.getObjectVolatile(a, j);
936                     if (t == null)                    // inconsistent length
937                         break;
938                     else if (t == task) {
939                         if (s + 1 == top) {           // pop
940                             if (!U.compareAndSwapObject(a, j, task, null))
941                                 break;
942                             top = s;
943                             removed = true;
944                         }
945                         else if (base == b)           // replace with proxy
946                             removed = U.compareAndSwapObject(a, j, task,
947                                                              new EmptyTask());
948                         break;
949                     }
950                     else if (t.status >= 0)
951                         empty = false;
952                     else if (s + 1 == top) {          // pop and throw away
953                         if (U.compareAndSwapObject(a, j, t, null))
954                             top = s;
955                         break;
956                     }
957                     if (--n == 0) {
958                         if (!empty && base == b)
959                             stat = false;
960                         break;
961                     }
962                 }
963             }
964             if (removed)
965                 task.doExec();
966             return stat;

1159      *
1160      * When convenient, we can extract the upper 32 bits of counts and
1161      * the lower 32 bits of queue state, u = (int)(ctl >>> 32) and e =
1162      * (int)ctl.  The ec field is never accessed alone, but always
1163      * together with id and st. The offsets of counts by the target
1164      * parallelism and the positionings of fields makes it possible to
1165      * perform the most common checks via sign tests of fields: When
1166      * ac is negative, there are not enough active workers, when tc is
1167      * negative, there are not enough total workers, and when e is
1168      * negative, the pool is terminating.  To deal with these possibly
1169      * negative fields, we use casts in and out of "short" and/or
1170      * signed shifts to maintain signedness.
1171      *
1172      * When a thread is queued (inactivated), its eventCount field is
1173      * set negative, which is the only way to tell if a worker is
1174      * prevented from executing tasks, even though it must continue to
1175      * scan for them to avoid queuing races. Note however that
1176      * eventCount updates lag releases so usage requires care.
1177      *
1178      * Field plock is an int packed with:
1179      * SHUTDOWN: true if shutdown is enabled (1 bit)
1180      * SEQ:  a sequence lock, with PL_LOCK bit set if locked (30 bits)
1181      * SIGNAL: set when threads may be waiting on the lock (1 bit)
1182      *
1183      * The sequence number enables simple consistency checks:
1184      * Staleness of read-only operations on the workQueues array can
1185      * be checked by comparing plock before vs after the reads.
1186      */

1263     /**
1264      * Acquires the plock lock to protect worker array and related
1265      * updates. This method is called only if an initial CAS on plock
1266      * fails. This acts as a spinlock for normal cases, but falls back
1267      * to builtin monitor to block when (rarely) needed. This would be
1268      * a terrible idea for a highly contended lock, but works fine as
1269      * a more conservative alternative to a pure spinlock.
1270      */
1271     private int acquirePlock() {
1272         int spins = PL_SPINS, r = 0, ps, nps;
1273         for (;;) {
1274             if (((ps = plock) & PL_LOCK) == 0 &&
1275                 U.compareAndSwapInt(this, PLOCK, ps, nps = ps + PL_LOCK))
1276                 return nps;
1277             else if (r == 0) { // randomize spins if possible
1278                 Thread t = Thread.currentThread(); WorkQueue w; Submitter z;
1279                 if ((t instanceof ForkJoinWorkerThread) &&
1280                     (w = ((ForkJoinWorkerThread)t).workQueue) != null)
1281                     r = w.seed;
1282                 else if ((z = submitters.get()) != null)
1283                     r = z.seed;
1284                 else
1285                     r = 1;
1286             }
1287             else if (spins >= 0) {
1288                 r ^= r << 1; r ^= r >>> 3; r ^= r << 10; // xorshift
1289                 if (r >= 0)
1290                     --spins;
1291             }
1292             else if (U.compareAndSwapInt(this, PLOCK, ps, ps | PL_SIGNAL)) {
1293                 synchronized (this) {
1294                     if ((plock & PL_SIGNAL) != 0) {
1295                         try {
1296                             wait();
1297                         } catch (InterruptedException ie) {
1298                             try {
1299                                 Thread.currentThread().interrupt();
1300                             } catch (SecurityException ignore) {
1301                             }
1302                         }
1303                     }
1304                     else
1305                         notifyAll();
1306                 }
1307             }
1308         }
1309     }

1351     /**
1352      * Callback from ForkJoinWorkerThread to establish and record its
1353      * WorkQueue. To avoid scanning bias due to packing entries in
1354      * front of the workQueues array, we treat the array as a simple
1355      * power-of-two hash table using per-thread seed as hash,
1356      * expanding as needed.
1357      *
1358      * @param wt the worker thread
1359      * @return the worker's queue
1360      */
1361     final WorkQueue registerWorker(ForkJoinWorkerThread wt) {
1362         Thread.UncaughtExceptionHandler handler; WorkQueue[] ws; int s, ps;
1363         wt.setDaemon(true);
1364         if ((handler = ueh) != null)
1365             wt.setUncaughtExceptionHandler(handler);
1366         do {} while (!U.compareAndSwapInt(this, INDEXSEED, s = indexSeed,
1367                                           s += SEED_INCREMENT) ||
1368                      s == 0); // skip 0
1369         WorkQueue w = new WorkQueue(this, wt, config >>> 16, s);
1370         if (((ps = plock) & PL_LOCK) != 0 ||
1371             !U.compareAndSwapInt(this, PLOCK, ps, ps += PL_LOCK))
1372             ps = acquirePlock();
1373         int nps = (ps & SHUTDOWN) | ((ps + PL_LOCK) & ~SHUTDOWN);
1374         try {
1375             if ((ws = workQueues) != null) {    // skip if shutting down
1376                 int n = ws.length, m = n - 1;
1377                 int r = (s << 1) | 1;           // use odd-numbered indices
1378                 if (ws[r &= m] != null) {       // collision
1379                     int probes = 0;             // step by approx half size
1380                     int step = (n <= 4) ? 2 : ((n >>> 1) & EVENMASK) + 2;
1381                     while (ws[r = (r + step) & m] != null) {
1382                         if (++probes >= n) {
1383                             workQueues = ws = Arrays.copyOf(ws, n <<= 1);
1384                             m = n - 1;
1385                             probes = 0;
1386                         }
1387                     }
1388                 }
1389                 w.eventCount = w.poolIndex = r; // volatile write orders
1390                 ws[r] = w;
1391             }
1392         } finally {
1393             if (!U.compareAndSwapInt(this, PLOCK, ps, nps))
1394                 releasePlock(nps);
1395         }
1396         wt.setName(workerNamePrefix.concat(Integer.toString(w.poolIndex)));
1397         return w;
1398     }

1400     /**
1401      * Final callback from terminating worker, as well as upon failure
1402      * to construct or start a worker.  Removes record of worker from
1403      * array, and adjusts counts. If pool is shutting down, tries to
1404      * complete termination.
1405      *
1406      * @param wt the worker thread or null if construction failed
1407      * @param ex the exception causing failure, or null if none
1408      */
1409     final void deregisterWorker(ForkJoinWorkerThread wt, Throwable ex) {
1410         WorkQueue w = null;
1411         if (wt != null && (w = wt.workQueue) != null) {
1412             int ps;
1413             w.qlock = -1;                // ensure set
1414             long ns = w.nsteals, sc;     // collect steal count
1415             do {} while (!U.compareAndSwapLong(this, STEALCOUNT,
1416                                                sc = stealCount, sc + ns));
1417             if (((ps = plock) & PL_LOCK) != 0 ||
1418                 !U.compareAndSwapInt(this, PLOCK, ps, ps += PL_LOCK))
1419                 ps = acquirePlock();
1420             int nps = (ps & SHUTDOWN) | ((ps + PL_LOCK) & ~SHUTDOWN);
1421             try {
1422                 int idx = w.poolIndex;
1423                 WorkQueue[] ws = workQueues;
1424                 if (ws != null && idx >= 0 && idx < ws.length && ws[idx] == w)
1425                     ws[idx] = null;
1426             } finally {
1427                 if (!U.compareAndSwapInt(this, PLOCK, ps, nps))
1428                     releasePlock(nps);
1429             }
1430         }
1431 
1432         long c;                          // adjust ctl counts
1433         do {} while (!U.compareAndSwapLong
1434                      (this, CTL, c = ctl, (((c - AC_UNIT) & AC_MASK) |
1435                                            ((c - TC_UNIT) & TC_MASK) |
1436                                            (c & ~(AC_MASK|TC_MASK)))));
1437 
1438         if (!tryTerminate(false, false) && w != null && w.array != null) {
1439             w.cancelAll();               // cancel remaining tasks
1440             WorkQueue[] ws; WorkQueue v; Thread p; int u, i, e;
1441             while ((u = (int)((c = ctl) >>> 32)) < 0 && (e = (int)c) >= 0) {
1442                 if (e > 0) {             // activate or create replacement
1443                     if ((ws = workQueues) == null ||
1444                         (i = e & SMASK) >= ws.length ||
1445                         (v = ws[i]) == null)
1446                         break;
1447                     long nc = (((long)(v.nextWait & E_MASK)) |
1448                                ((long)(u + UAC_UNIT) << 32));
1449                     if (v.eventCount != (e | INT_SIGN))
1450                         break;
1451                     if (U.compareAndSwapLong(this, CTL, c, nc)) {
1452                         v.eventCount = (e + E_SEQ) & E_MASK;
1453                         if ((p = v.parker) != null)
1454                             U.unpark(p);
1455                         break;
1456                     }
1457                 }
1458                 else {
1459                     if ((short)u < 0)
1460                         tryAddWorker();
1461                     break;
1462                 }
1463             }
1464         }
1465         if (ex == null)                     // help clean refs on way out
1466             ForkJoinTask.helpExpungeStaleExceptions();
1467         else                                // rethrow
1468             ForkJoinTask.rethrow(ex);
1469     }

1474      * Unless shutting down, adds the given task to a submission queue
1475      * at submitter's current queue index (modulo submission
1476      * range). Only the most common path is directly handled in this
1477      * method. All others are relayed to fullExternalPush.
1478      *
1479      * @param task the task. Caller must ensure non-null.
1480      */
1481     final void externalPush(ForkJoinTask<?> task) {
1482         WorkQueue[] ws; WorkQueue q; Submitter z; int m; ForkJoinTask<?>[] a;
1483         if ((z = submitters.get()) != null && plock > 0 &&
1484             (ws = workQueues) != null && (m = (ws.length - 1)) >= 0 &&
1485             (q = ws[m & z.seed & SQMASK]) != null &&
1486             U.compareAndSwapInt(q, QLOCK, 0, 1)) { // lock
1487             int b = q.base, s = q.top, n, an;
1488             if ((a = q.array) != null && (an = a.length) > (n = s + 1 - b)) {
1489                 int j = (((an - 1) & s) << ASHIFT) + ABASE;
1490                 U.putOrderedObject(a, j, task);
1491                 q.top = s + 1;                     // push on to deque
1492                 q.qlock = 0;
1493                 if (n <= 2)
1494                     signalWork(q);
1495                 return;
1496             }
1497             q.qlock = 0;
1498         }
1499         fullExternalPush(task);

1502     /**
1503      * Full version of externalPush. This method is called, among
1504      * other times, upon the first submission of the first task to the
1505      * pool, so must perform secondary initialization.  It also
1506      * detects first submission by an external thread by looking up
1507      * its ThreadLocal, and creates a new shared queue if the one at
1508      * index if empty or contended. The plock lock body must be
1509      * exception-free (so no try/finally) so we optimistically
1510      * allocate new queues outside the lock and throw them away if
1511      * (very rarely) not needed.
1512      *
1513      * Secondary initialization occurs when plock is zero, to create
1514      * workQueue array and set plock to a valid value.  This lock body
1515      * must also be exception-free. Because the plock seq value can
1516      * eventually wrap around zero, this method harmlessly fails to
1517      * reinitialize if workQueues exists, while still advancing plock.
1518      */
1519     private void fullExternalPush(ForkJoinTask<?> task) {
1520         int r = 0; // random index seed
1521         for (Submitter z = submitters.get();;) {
1522             WorkQueue[] ws; WorkQueue q; int ps, m, k;
1523             if (z == null) {
1524                 if (U.compareAndSwapInt(this, INDEXSEED, r = indexSeed,
1525                                         r += SEED_INCREMENT) && r != 0)
1526                     submitters.set(z = new Submitter(r));
1527             }
1528             else if (r == 0) {                  // move to a different index
1529                 r = z.seed;
1530                 r ^= r << 13;                   // same xorshift as WorkQueues
1531                 r ^= r >>> 17;
1532                 z.seed = r ^ (r << 5);
1533             }
1534             else if ((ps = plock) < 0)
1535                 throw new RejectedExecutionException();
1536             else if (ps == 0 || (ws = workQueues) == null ||
1537                      (m = ws.length - 1) < 0) { // initialize workQueues
1538                 int p = config & SMASK;         // find power of two table size
1539                 int n = (p > 1) ? p - 1 : 1;    // ensure at least 2 slots
1540                 n |= n >>> 1; n |= n >>> 2;  n |= n >>> 4;
1541                 n |= n >>> 8; n |= n >>> 16; n = (n + 1) << 1;
1542                 WorkQueue[] nws = ((ws = workQueues) == null || ws.length == 0 ?
1543                                    new WorkQueue[n] : null);
1544                 if (((ps = plock) & PL_LOCK) != 0 ||
1545                     !U.compareAndSwapInt(this, PLOCK, ps, ps += PL_LOCK))
1546                     ps = acquirePlock();
1547                 if (((ws = workQueues) == null || ws.length == 0) && nws != null)
1548                     workQueues = nws;
1549                 int nps = (ps & SHUTDOWN) | ((ps + PL_LOCK) & ~SHUTDOWN);
1550                 if (!U.compareAndSwapInt(this, PLOCK, ps, nps))
1551                     releasePlock(nps);
1552             }
1553             else if ((q = ws[k = r & m & SQMASK]) != null) {
1554                 if (q.qlock == 0 && U.compareAndSwapInt(q, QLOCK, 0, 1)) {
1555                     ForkJoinTask<?>[] a = q.array;
1556                     int s = q.top;
1557                     boolean submitted = false;
1558                     try {                      // locked version of push
1559                         if ((a != null && a.length > s + 1 - q.base) ||
1560                             (a = q.growArray()) != null) {   // must presize
1561                             int j = (((a.length - 1) & s) << ASHIFT) + ABASE;
1562                             U.putOrderedObject(a, j, task);
1563                             q.top = s + 1;
1564                             submitted = true;
1565                         }
1566                     } finally {
1567                         q.qlock = 0;  // unlock
1568                     }
1569                     if (submitted) {
1570                         signalWork(q);
1571                         return;
1572                     }
1573                 }
1574                 r = 0; // move on failure
1575             }
1576             else if (((ps = plock) & PL_LOCK) == 0) { // create new queue
1577                 q = new WorkQueue(this, null, SHARED_QUEUE, r);
1578                 if (((ps = plock) & PL_LOCK) != 0 ||
1579                     !U.compareAndSwapInt(this, PLOCK, ps, ps += PL_LOCK))
1580                     ps = acquirePlock();
1581                 if ((ws = workQueues) != null && k < ws.length && ws[k] == null)
1582                     ws[k] = q;
1583                 int nps = (ps & SHUTDOWN) | ((ps + PL_LOCK) & ~SHUTDOWN);
1584                 if (!U.compareAndSwapInt(this, PLOCK, ps, nps))
1585                     releasePlock(nps);
1586             }
1587             else
1588                 r = 0; // try elsewhere while lock held
1589         }
1590     }

1647     /**
1648      * Scans for and, if found, returns one task, else possibly
1649      * inactivates the worker. This method operates on single reads of
1650      * volatile state and is designed to be re-invoked continuously,
1651      * in part because it returns upon detecting inconsistencies,
1652      * contention, or state changes that indicate possible success on
1653      * re-invocation.
1654      *
1655      * The scan searches for tasks across queues (starting at a random
1656      * index, and relying on registerWorker to irregularly scatter
1657      * them within array to avoid bias), checking each at least twice.
1658      * The scan terminates upon either finding a non-empty queue, or
1659      * completing the sweep. If the worker is not inactivated, it
1660      * takes and returns a task from this queue. Otherwise, if not
1661      * activated, it signals workers (that may include itself) and
1662      * returns so caller can retry. Also returns for true if the
1663      * worker array may have changed during an empty scan.  On failure
1664      * to find a task, we take one of the following actions, after
1665      * which the caller will retry calling this method unless
1666      * terminated.
1667      *
1668      * * If pool is terminating, terminate the worker.
1669      *
1670      * * If not already enqueued, try to inactivate and enqueue the
1671      * worker on wait queue. Or, if inactivating has caused the pool
1672      * to be quiescent, relay to idleAwaitWork to possibly shrink
1673      * pool.
1674      *
1675      * * If already enqueued and none of the above apply, possibly
1676      * park awaiting signal, else lingering to help scan and signal.
1677      *
1678      * * If a non-empty queue discovered or left as a hint,
1679      * help wake up other workers before return.
1680      *
1681      * @param w the worker (via its WorkQueue)
1682      * @return a task or null if none found
1683      */
1684     private final ForkJoinTask<?> scan(WorkQueue w) {
1685         WorkQueue[] ws; int m;
1686         int ps = plock;                          // read plock before ws
1687         if (w != null && (ws = workQueues) != null && (m = ws.length - 1) >= 0) {
1688             int ec = w.eventCount;               // ec is negative if inactive
1689             int r = w.seed; r ^= r << 13; r ^= r >>> 17; w.seed = r ^= r << 5;
1690             w.hint = -1;                         // update seed and clear hint
1691             int j = ((m + m + 1) | MIN_SCAN) & MAX_SCAN;
1692             do {
1693                 WorkQueue q; ForkJoinTask<?>[] a; int b;
1694                 if ((q = ws[(r + j) & m]) != null && (b = q.base) - q.top < 0 &&
1695                     (a = q.array) != null) {     // probably nonempty
1696                     int i = (((a.length - 1) & b) << ASHIFT) + ABASE;
1697                     ForkJoinTask<?> t = (ForkJoinTask<?>)
1698                         U.getObjectVolatile(a, i);
1699                     if (q.base == b && ec >= 0 && t != null &&
1700                         U.compareAndSwapObject(a, i, t, null)) {

2181      * programmers, frameworks, tools, or languages have little or no

2183      * method, we ask users only about tradeoffs in overhead vs
2184      * expected throughput and its variance, rather than how finely to
2185      * partition tasks.

2187      * In a steady state strict (tree-structured) computation, each
2188      * thread makes available for stealing enough tasks for other
2189      * threads to remain active. Inductively, if all threads play by
2190      * the same rules, each thread should make available only a
2191      * constant number of tasks.
2192      *

2194      * would require immediate replenishment upon each steal to
2195      * maintain enough tasks, which is infeasible.  Further,
2196      * partitionings/granularities of offered tasks should minimize
2197      * steal rates, which in general means that threads nearer the top
2198      * of computation tree should generate more than those nearer the
2199      * bottom. In perfect steady state, each thread is at
2200      * approximately the same level of computation tree. However,
2201      * producing extra tasks amortizes the uncertainty of progress and
2202      * diffusion assumptions.

2275                     }
2276                 }
2277                 return true;

2280                 WorkQueue[] ws; WorkQueue w;

2285                         if ((w = ws[i]) != null) {

2292                         }
2293                     }
2294                 }

2298                     WorkQueue[] ws; WorkQueue w; Thread wt;
2299                     if ((ws = workQueues) != null) {
2300                         int n = ws.length;
2301                         for (int i = 0; i < n; ++i) {
2302                             if ((w = ws[i]) != null) {
2303                                 w.qlock = -1;
2304                                 if (pass > 0) {
2305                                     w.cancelAll();
2306                                     if (pass > 1 && (wt = w.owner) != null) {
2307                                         if (!wt.isInterrupted()) {
2308                                             try {
2309                                                 wt.interrupt();
2310                                             } catch (Throwable ignore) {
2311                                             }
2312                                         }
2313                                         U.unpark(wt);
2314                                     }
2315                                 }
2316                             }
2317                         }
2318                         // Wake up workers parked on event queue
2319                         int i, e; long cc; Thread p;
2320                         while ((e = (int)(cc = ctl) & E_MASK) != 0 &&
2321                                (i = e & SMASK) < n && i >= 0 &&
2322                                (w = ws[i]) != null) {

2324                                        ((cc + AC_UNIT) & AC_MASK) |
2325                                        (cc & (TC_MASK|STOP_BIT)));
2326                             if (w.eventCount == (e | INT_SIGN) &&

2329                                 w.qlock = -1;
