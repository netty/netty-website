135         //
136         // * <  0  - Don't use cleaner, and inherit max direct memory from java. In this case the
137         //           "practical max direct memory" would be 2 * max memory as defined by the JDK.
138         // * == 0  - Use cleaner, Netty will not enforce max memory, and instead will defer to JDK.

504     private static void incrementMemoryCounter(int capacity) {
505         if (DIRECT_MEMORY_COUNTER != null) {
506             for (;;) {
507                 long usedMemory = DIRECT_MEMORY_COUNTER.get();
508                 long newUsedMemory = usedMemory + capacity;
509                 if (newUsedMemory > DIRECT_MEMORY_LIMIT) {
510                     throw new OutOfDirectMemoryError("failed to allocate " + capacity
511                             + " byte(s) of direct memory (used: " + usedMemory + ", max: " + DIRECT_MEMORY_LIMIT + ')');
512                 }
513                 if (DIRECT_MEMORY_COUNTER.compareAndSet(usedMemory, newUsedMemory)) {
514                     break;
515                 }
516             }
517         }
518     }
519 
520     private static void decrementMemoryCounter(int capacity) {
521         if (DIRECT_MEMORY_COUNTER != null) {
522             long usedMemory = DIRECT_MEMORY_COUNTER.addAndGet(-capacity);
523             assert usedMemory >= 0;
524         }
525     }

544     private static final class Mpsc {
545         private static final boolean USE_MPSC_CHUNKED_ARRAY_QUEUE;
546 
547         private Mpsc() {
548         }
549 
550         static {
551             Object unsafe = null;
552             if (hasUnsafe()) {
553                 // jctools goes through its own process of initializing unsafe; of
554                 // course, this requires permissions which might not be granted to calling code, so we
555                 // must mark this block as privileged too
556                 unsafe = AccessController.doPrivileged(new PrivilegedAction<Object>() {
557                     @Override
558                     public Object run() {
559                         // force JCTools to initialize unsafe
560                         return UnsafeAccess.UNSAFE;
561                     }
562                 });
563             }
564 
565             if (unsafe == null) {
566                 logger.debug("org.jctools-core.MpscChunkedArrayQueue: unavailable");
567                 USE_MPSC_CHUNKED_ARRAY_QUEUE = false;
568             } else {
569                 logger.debug("org.jctools-core.MpscChunkedArrayQueue: available");
570                 USE_MPSC_CHUNKED_ARRAY_QUEUE = true;
571             }
572         }
573 
574         static <T> Queue<T> newMpscQueue(final int maxCapacity) {
575             if (USE_MPSC_CHUNKED_ARRAY_QUEUE) {
576                 // Calculate the max capacity which can not be bigger then MAX_ALLOWED_MPSC_CAPACITY.
577                 // This is forced by the MpscChunkedArrayQueue implementation as will try to round it
578                 // up to the next power of two and so will overflow otherwise.
579                 final int capacity =
580                         Math.max(Math.min(maxCapacity, MAX_ALLOWED_MPSC_CAPACITY), MIN_MAX_MPSC_CAPACITY);
581                 return new MpscChunkedArrayQueue<T>(MPSC_CHUNK_SIZE, capacity);
582             } else {
583                 return new MpscLinkedAtomicQueue<T>();
584             }
585         }
586     }
