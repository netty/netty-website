16  package io.netty.util.concurrent;
17  
18  import java.util.concurrent.Callable;
19  import java.util.concurrent.ScheduledExecutorService;
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * The {@link EventExecutor} is a special {@link ScheduledExecutorService} which comes
24   * with some handy methods to see if a {@link Thread} is executed in a event loop.
25   * Beside this it also extends the {@link EventExecutorGroup} to allow a generic way to
26   * access methods.
27   *
28   */
29  public interface EventExecutor extends EventExecutorGroup, ScheduledExecutorService {
30  
31      /**
32       * Returns a reference to itself.
33       */
34      @Override
35      EventExecutor next();
36  
37      /**
38       * Return the {@link EventExecutorGroup} which is the parent of this {@link EventExecutor},
39       */
40      EventExecutorGroup parent();
41  
42      /**
43       * Calls {@link #inEventLoop(Thread)} with {@link Thread#currentThread()} as argument
44       */
45      boolean inEventLoop();
46  
47      /**
48       * Return {@code true} if the given {@link Thread} is executed in the event loop,
49       * {@code false} otherwise.
50       */
51      boolean inEventLoop(Thread thread);
52  
53      /**
54       * Return a new {@link Promise}.
55       */
56      <V> Promise<V> newPromise();
57  
58      /**
59       * Create a new {@link Future} which is marked as successes already. So {@link Future#isSuccess()}
60       * will return {@code true}. All {@link FutureListener} added to it will be notified directly. Also
61       * every call of blocking methods will just return without blocking.
62       */
63      <V> Future<V> newSucceededFuture(V result);
64  
65      /**
66       * Create a new {@link Future} which is marked as fakued already. So {@link Future#isSuccess()}
67       * will return {@code false}. All {@link FutureListener} added to it will be notified directly. Also
68       * every call of blocking methods will just return without blocking.
69       */
70      <V> Future<V> newFailedFuture(Throwable cause);
71  
72      @Override
73      Future<?> submit(Runnable task);
74  
75      @Override
76      <T> Future<T> submit(Runnable task, T result);
77  
78      @Override
79      <T> Future<T> submit(Callable<T> task);
80  
81      @Override
82      ScheduledFuture<?> schedule(Runnable command, long delay, TimeUnit unit);
83  
84      @Override
85      <V> ScheduledFuture<V> schedule(Callable<V> callable, long delay, TimeUnit unit);
86  
87      @Override
88      ScheduledFuture<?> scheduleAtFixedRate(Runnable command, long initialDelay, long period, TimeUnit unit);
89  
90      @Override
91      ScheduledFuture<?> scheduleWithFixedDelay(Runnable command, long initialDelay, long delay, TimeUnit unit);
92  }
