161     /**
162      *
163      * @param factory the factory used to create InterfaceHttpData
164      * @param request the request to encode
165      * @param multipart True if the FORM is a ENCTYPE="multipart/form-data"
166      * @param charset the charset to use as default
167     +  @param encoderMode the mode for the encoder to use. See {@link EncoderMode} for the details.
168      * @throws NullPointerException for request or charset or factory
169      * @throws ErrorDataEncoderException if the request is not a POST
170      */
171     public HttpPostRequestEncoder(HttpDataFactory factory, HttpRequest request, boolean multipart,
172                                    Charset charset, EncoderMode encoderMode) throws ErrorDataEncoderException {
173         if (factory == null) {
174             throw new NullPointerException("factory");
175         }
176         if (request == null) {
177             throw new NullPointerException("request");
178         }
179         if (charset == null) {
180             throw new NullPointerException("charset");
181         }
182         HttpMethod method = request.getMethod();
183         if (!(method.equals(HttpMethod.POST) || method.equals(HttpMethod.PUT) || method.equals(HttpMethod.PATCH))) {
184             throw new ErrorDataEncoderException("Cannot create a Encoder if not a POST");
185         }
186         this.request = request;
187         this.charset = charset;
188         this.factory = factory;
189         this.encoderMode = encoderMode;
190         // Fill default values
191         bodyListDatas = new ArrayList<InterfaceHttpData>();
192         // default mode
193         isLastChunk = false;
194         isLastChunkSent = false;
195         isMultipart = multipart;
196         multipartHttpDatas = new ArrayList<InterfaceHttpData>();
197         if (isMultipart) {
198             initDataMultipart();
199         }
200     }

283         currentFileUpload = null;

311     /**
312      * Add a file as a FileUpload
313      * @param name the name of the parameter
314      * @param file the file to be uploaded (if not Multipart mode, only the filename will be included)
315      * @param contentType the associated contentType for the File
316      * @param isText True if this file should be transmitted in Text format (else binary)
317      * @throws NullPointerException for name and file
318      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
319      */
320     public void addBodyFileUpload(String name, File file, String contentType, boolean isText)
321     throws ErrorDataEncoderException {
322         if (name == null) {
323             throw new NullPointerException("name");
324         }
325         if (file == null) {
326             throw new NullPointerException("file");
327         }
328         String scontentType = contentType;
329         String contentTransferEncoding = null;
330         if (contentType == null) {
331             if (isText) {
332                 scontentType = HttpPostBodyUtil.DEFAULT_TEXT_CONTENT_TYPE;
333             } else {
334                 scontentType = HttpPostBodyUtil.DEFAULT_BINARY_CONTENT_TYPE;
335             }
336         }
337         if (!isText) {
338             contentTransferEncoding = HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value();
339         }
340         FileUpload fileUpload = factory.createFileUpload(request, name, file.getName(),
341                 scontentType, contentTransferEncoding, null, file.length());
342         try {
343             fileUpload.setContent(file);
344         } catch (IOException e) {
345             throw new ErrorDataEncoderException(e);
346         }
347         addBodyHttpData(fileUpload);
348     }

350     /**
351      * Add a series of Files associated with one File parameter (implied Mixed mode in Multipart)
352      * @param name the name of the parameter
353      * @param file the array of files
354      * @param contentType the array of content Types associated with each file
355      * @param isText the array of isText attribute (False meaning binary mode) for each file
356      * @throws NullPointerException also throws if array have different sizes
357      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
358      */
359     public void addBodyFileUploads(String name, File[] file, String[] contentType, boolean[] isText)
360     throws ErrorDataEncoderException {
361         if (file.length != contentType.length && file.length != isText.length) {
362             throw new NullPointerException("Different array length");
363         }
364         for (int i = 0; i < file.length; i++) {
365             addBodyFileUpload(name, file[i], contentType[i], isText[i]);
366         }
367     }

369     /**
370      * Add the InterfaceHttpData to the Body list
371      * @throws NullPointerException for data
372      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
373      */
374     public void addBodyHttpData(InterfaceHttpData data)
375     throws ErrorDataEncoderException {
376         if (headerFinalized) {
377             throw new ErrorDataEncoderException("Cannot add value once finalized");
378         }
379         if (data == null) {
380             throw new NullPointerException("data");
381         }
382         bodyListDatas.add(data);
383         if (! isMultipart) {
384             if (data instanceof Attribute) {
385                 Attribute attribute = (Attribute) data;
386                 try {
387                     // name=value& with encoded name and attribute
388                     String key = encodeAttribute(attribute.getName(), charset);
389                     String value = encodeAttribute(attribute.getValue(), charset);
390                     Attribute newattribute = factory.createAttribute(request, key, value);
391                     multipartHttpDatas.add(newattribute);
392                     globalBodySize += newattribute.getName().length() + 1 +
393                         newattribute.length() + 1;
394                 } catch (IOException e) {
395                     throw new ErrorDataEncoderException(e);
396                 }
397             } else if (data instanceof FileUpload) {
398                 // since not Multipart, only name=filename => Attribute
399                 FileUpload fileUpload = (FileUpload) data;
400                 // name=filename& with encoded name and filename
401                 String key = encodeAttribute(fileUpload.getName(), charset);
402                 String value = encodeAttribute(fileUpload.getFilename(), charset);
403                 Attribute newattribute = factory.createAttribute(request, key, value);
404                 multipartHttpDatas.add(newattribute);
405                 globalBodySize += newattribute.getName().length() + 1 +
406                     newattribute.length() + 1;
407             }
408             return;
409         }
410         /*
411          * Logic:
412          * if not Attribute:
413          *      add Data to body list
414          *      if (duringMixedMode)
415          *          add endmixedmultipart delimiter
416          *          currentFileUpload = null
417          *          duringMixedMode = false;
418          *      add multipart delimiter, multipart body header and Data to multipart list
419          *      reset currentFileUpload, duringMixedMode
420          * if FileUpload: take care of multiple file for one field => mixed mode
421          *      if (duringMixeMode)
422          *          if (currentFileUpload.name == data.name)
423          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
424          *          else
425          *              add endmixedmultipart delimiter, multipart body header and Data to multipart list
426          *              currentFileUpload = data
427          *              duringMixedMode = false;
428          *      else
429          *          if (currentFileUpload.name == data.name)
430          *              change multipart body header of previous file into multipart list to
431          *                      mixedmultipart start, mixedmultipart body header
432          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
433          *              duringMixedMode = true
434          *          else
435          *              add multipart delimiter, multipart body header and Data to multipart list
436          *              currentFileUpload = data
437          *              duringMixedMode = false;
438          * Do not add last delimiter! Could be:
439          * if duringmixedmode: endmixedmultipart + endmultipart
440          * else only endmultipart
441          */
442         if (data instanceof Attribute) {
443             if (duringMixedMode) {
444                 InternalAttribute internal = new InternalAttribute(charset);
445                 internal.addValue("\r\n--" + multipartMixedBoundary + "--");
446                 multipartHttpDatas.add(internal);
447                 multipartMixedBoundary = null;
448                 currentFileUpload = null;
449                 duringMixedMode = false;
450             }
451             InternalAttribute internal = new InternalAttribute(charset);
452             if (!multipartHttpDatas.isEmpty()) {
453                 // previously a data field so CRLF
454                 internal.addValue("\r\n");
455             }
456             internal.addValue("--" + multipartDataBoundary + "\r\n");
457             // content-disposition: form-data; name="field1"
458             Attribute attribute = (Attribute) data;
459             internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
460                     HttpPostBodyUtil.FORM_DATA + "; " +
461                     HttpPostBodyUtil.NAME + "=\"" + attribute.getName() + "\"\r\n");
462             Charset localcharset = attribute.getCharset();
463             if (localcharset != null) {
464                 // Content-Type: charset=charset
465                 internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " +
466                         HttpPostBodyUtil.DEFAULT_TEXT_CONTENT_TYPE + "; " +
467                         HttpHeaders.Values.CHARSET + '=' + localcharset + "\r\n");
468             }
469             // CRLF between body header and data
470             internal.addValue("\r\n");
471             multipartHttpDatas.add(internal);
472             multipartHttpDatas.add(data);
473             globalBodySize += attribute.length() + internal.size();
474         } else if (data instanceof FileUpload) {
475             FileUpload fileUpload = (FileUpload) data;
476             InternalAttribute internal = new InternalAttribute(charset);
477             if (!multipartHttpDatas.isEmpty()) {
478                 // previously a data field so CRLF
479                 internal.addValue("\r\n");
480             }
481             boolean localMixed;
482             if (duringMixedMode) {
483                 if (currentFileUpload != null &&
484                         currentFileUpload.getName().equals(fileUpload.getName())) {
485                     // continue a mixed mode
486 
487                     localMixed = true;
488                 } else {
489                     // end a mixed mode
490 
491                     // add endmixedmultipart delimiter, multipart body header and
492                     // Data to multipart list
493                     internal.addValue("--" + multipartMixedBoundary + "--");
494                     multipartHttpDatas.add(internal);
495                     multipartMixedBoundary = null;
496                     // start a new one (could be replaced if mixed start again from here
497                     internal = new InternalAttribute(charset);
498                     internal.addValue("\r\n");
499                     localMixed = false;
500                     // new currentFileUpload and no more in Mixed mode
501                     currentFileUpload = fileUpload;
502                     duringMixedMode = false;
503                 }
504             } else {
505                 if (currentFileUpload != null &&
506                         currentFileUpload.getName().equals(fileUpload.getName())) {
507                     // create a new mixed mode (from previous file)
508 
509                     // change multipart body header of previous file into multipart list to
510                     // mixedmultipart start, mixedmultipart body header
511 
512                     // change Internal (size()-2 position in multipartHttpDatas)
513                     // from (line starting with *)
514                     // --AaB03x
515                     // * Content-Disposition: form-data; name="files"; filename="file1.txt"
516                     // Content-Type: text/plain
517                     // to (lines starting with *)
518                     // --AaB03x
519                     // * Content-Disposition: form-data; name="files"
520                     // * Content-Type: multipart/mixed; boundary=BbC04y
521                     // *
522                     // * --BbC04y
523                     // * Content-Disposition: file; filename="file1.txt"
524                     // Content-Type: text/plain
525                     initMixedMultipart();
526                     InternalAttribute pastAttribute =
527                         (InternalAttribute) multipartHttpDatas.get(multipartHttpDatas.size() - 2);
528                     // remove past size
529                     globalBodySize -= pastAttribute.size();
530                     String replacement = HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
531                         HttpPostBodyUtil.FORM_DATA + "; " + HttpPostBodyUtil.NAME + "=\"" +
532                         fileUpload.getName() + "\"\r\n";
533                     replacement += HttpHeaders.Names.CONTENT_TYPE + ": " +
534                         HttpPostBodyUtil.MULTIPART_MIXED + "; " + HttpHeaders.Values.BOUNDARY +
535                         '=' + multipartMixedBoundary + "\r\n\r\n";
536                     replacement += "--" + multipartMixedBoundary + "\r\n";
537                     replacement += HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
538                         HttpPostBodyUtil.FILE + "; " + HttpPostBodyUtil.FILENAME + "=\"" +
539                         fileUpload.getFilename() + "\"\r\n";
540                     pastAttribute.setValue(replacement, 1);
541                     // update past size
542                     globalBodySize += pastAttribute.size();
543 
544                     // now continue
545                     // add mixedmultipart delimiter, mixedmultipart body header and
546                     // Data to multipart list
547                     localMixed = true;
548                     duringMixedMode = true;
549                 } else {
550                     // a simple new multipart
551                     //add multipart delimiter, multipart body header and Data to multipart list
552                     localMixed = false;
553                     currentFileUpload = fileUpload;
554                     duringMixedMode = false;
555                 }
556             }
557 
558             if (localMixed) {
559                 // add mixedmultipart delimiter, mixedmultipart body header and
560                 // Data to multipart list
561                 internal.addValue("--" + multipartMixedBoundary + "\r\n");
562                 // Content-Disposition: file; filename="file1.txt"
563                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
564                         HttpPostBodyUtil.FILE + "; " + HttpPostBodyUtil.FILENAME + "=\"" +
565                         fileUpload.getFilename() +
566                         "\"\r\n");
567             } else {
568                 internal.addValue("--" + multipartDataBoundary + "\r\n");
569                 // Content-Disposition: form-data; name="files"; filename="file1.txt"
570                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
571                         HttpPostBodyUtil.FORM_DATA + "; " + HttpPostBodyUtil.NAME + "=\"" +
572                         fileUpload.getName() + "\"; " +
573                         HttpPostBodyUtil.FILENAME + "=\"" +
574                         fileUpload.getFilename() +
575                         "\"\r\n");
576             }
577             // Content-Type: image/gif
578             // Content-Type: text/plain; charset=ISO-8859-1
579             // Content-Transfer-Encoding: binary
580             internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " +
581                     fileUpload.getContentType());
582             String contentTransferEncoding = fileUpload.getContentTransferEncoding();
583             if (contentTransferEncoding != null &&
584                     contentTransferEncoding.equals(
585                             HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value())) {
586                 internal.addValue("\r\n" + HttpHeaders.Names.CONTENT_TRANSFER_ENCODING +
587                         ": " + HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value() +
588                         "\r\n\r\n");
589             } else if (fileUpload.getCharset() != null) {
590                 internal.addValue("; " + HttpHeaders.Values.CHARSET + '=' +
591                         fileUpload.getCharset() + "\r\n\r\n");
592             } else {
593                 internal.addValue("\r\n\r\n");
594             }
595             multipartHttpDatas.add(internal);
596             multipartHttpDatas.add(data);
597             globalBodySize += fileUpload.length() + internal.size();
598         }
599     }

606     /**
607      * Finalize the request by preparing the Header in the request and
608      * returns the request ready to be sent.<br>
609      * Once finalized, no data must be added.<br>
610      * If the request does not need chunk (isChunked() == false),
611      * this request is the only object to send to
612      * the remote server.
613      *
614      * @return the request object (chunked or not according to size of body)
615      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
616      */
617     public HttpRequest finalizeRequest() throws ErrorDataEncoderException {
618         HttpHeaders headers = request.headers();
619         // Finalize the multipartHttpDatas
620         if (! headerFinalized) {
621             if (isMultipart) {
622                 InternalAttribute internal = new InternalAttribute(charset);
623                 if (duringMixedMode) {
624                     internal.addValue("\r\n--" + multipartMixedBoundary + "--");
625                 }
626                 internal.addValue("\r\n--" + multipartDataBoundary + "--\r\n");
627                 multipartHttpDatas.add(internal);
628                 multipartMixedBoundary = null;
629                 currentFileUpload = null;
630                 duringMixedMode = false;
631                 globalBodySize += internal.size();
632             }
633             headerFinalized = true;
634         } else {
635             throw new ErrorDataEncoderException("Header already encoded");
636         }
637         List<String> contentTypes = headers.getAll(HttpHeaders.Names.CONTENT_TYPE);
638         List<String> transferEncoding = headers.getAll(HttpHeaders.Names.TRANSFER_ENCODING);
639         if (contentTypes != null) {
640             headers.remove(HttpHeaders.Names.CONTENT_TYPE);
641             for (String contentType: contentTypes) {
642                 // "multipart/form-data; boundary=--89421926422648"
643                 if (contentType.toLowerCase().startsWith(
644                         HttpHeaders.Values.MULTIPART_FORM_DATA)) {
645                     // ignore
646                 } else if (contentType.toLowerCase().startsWith(
647                         HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED)) {
648                     // ignore
649                 } else {
650                     headers.add(HttpHeaders.Names.CONTENT_TYPE, contentType);
651                 }
652             }
653         }
654         if (isMultipart) {
655             String value = HttpHeaders.Values.MULTIPART_FORM_DATA + "; " +
656                 HttpHeaders.Values.BOUNDARY + '=' + multipartDataBoundary;
657             headers.add(HttpHeaders.Names.CONTENT_TYPE, value);
658         } else {
659             // Not multipart
660             headers.add(HttpHeaders.Names.CONTENT_TYPE,
661                     HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED);
662         }
663         // Now consider size for chunk or not
664         long realSize = globalBodySize;
665         if (isMultipart) {
666             iterator = multipartHttpDatas.listIterator();
667         } else {
668             realSize -= 1; // last '&' removed
669             iterator = multipartHttpDatas.listIterator();
670         }
671         headers.set(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(realSize));
672         if (realSize > HttpPostBodyUtil.chunkSize || isMultipart) {
673             isChunked = true;
674             if (transferEncoding != null) {
675                 headers.remove(HttpHeaders.Names.TRANSFER_ENCODING);
676                 for (String v: transferEncoding) {
677                     if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
678                         // ignore
679                     } else {
680                         headers.add(HttpHeaders.Names.TRANSFER_ENCODING, v);
681                     }
682                 }
683             }
684             headers.add(HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED);
685             request.setContent(ChannelBuffers.EMPTY_BUFFER);
686         } else {
687             // get the only one body and set it to the request
688             HttpChunk chunk = nextChunk();
689             request.setContent(chunk.getContent());
690         }
691         return request;
692     }

738     /**
739      *
740      * @return the next ChannelBuffer to send as a HttpChunk and modifying currentBuffer
741      * accordingly
742      */
743     private ChannelBuffer fillChannelBuffer() {
744         int length = currentBuffer.readableBytes();
745         if (length > HttpPostBodyUtil.chunkSize) {
746             ChannelBuffer slice =
747                 currentBuffer.slice(currentBuffer.readerIndex(), HttpPostBodyUtil.chunkSize);
748             currentBuffer.skipBytes(HttpPostBodyUtil.chunkSize);
749             return slice;
750         } else {
751             // to continue
752             ChannelBuffer slice = currentBuffer;
753             currentBuffer = null;
754             return slice;
755         }

758     /**
759      * From the current context (currentBuffer and currentData), returns the next HttpChunk
760      * (if possible) trying to get sizeleft bytes more into the currentBuffer.
761      * This is the Multipart version.
762      *
763      * @param sizeleft the number of bytes to try to get from currentData
764      * @return the next HttpChunk or null if not enough bytes were found
765      * @throws ErrorDataEncoderException if the encoding is in error
766      */
767     private HttpChunk encodeNextChunkMultipart(int sizeleft) throws ErrorDataEncoderException {
768         if (currentData == null) {
769             return null;
770         }
771         ChannelBuffer buffer;
772         if (currentData instanceof InternalAttribute) {
773 
774             buffer = ((InternalAttribute) currentData).toChannelBuffer();
775             currentData = null;
776         } else {
777             if (currentData instanceof Attribute) {
778                 try {
779                     buffer = ((Attribute) currentData).getChunk(sizeleft);
780                 } catch (IOException e) {
781                     throw new ErrorDataEncoderException(e);
782                 }
783             } else {
784                 try {
785                     buffer = ((HttpData) currentData).getChunk(sizeleft);
786                 } catch (IOException e) {
787                     throw new ErrorDataEncoderException(e);
788                 }
789             }
790             if (buffer.capacity() == 0) {
791                 // end for current InterfaceHttpData, need more data
792                 currentData = null;
793                 return null;
794             }
795         }
796         if (currentBuffer == null) {
797             currentBuffer = buffer;
798         } else {
799             currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
800                 buffer);
801         }
802         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
803             currentData = null;
804             return null;
805         }
806         buffer = fillChannelBuffer();
807         return new DefaultHttpChunk(buffer);
808     }

810     /**
811      * From the current context (currentBuffer and currentData), returns the next HttpChunk
812      * (if possible) trying to get sizeleft bytes more into the currentBuffer.
813      * This is the UrlEncoded version.
814      *
815      * @param sizeleft the number of bytes to try to get from currentData
816      * @return the next HttpChunk or null if not enough bytes were found
817      * @throws ErrorDataEncoderException if the encoding is in error
818      */
819     private HttpChunk encodeNextChunkUrlEncoded(int sizeleft) throws ErrorDataEncoderException {
820         if (currentData == null) {
821             return null;
822         }
823         int size = sizeleft;
824         ChannelBuffer buffer;
825         if (isKey) {
826             // get name
827             String key = currentData.getName();
828             buffer = ChannelBuffers.wrappedBuffer(key.getBytes());
829             isKey = false;
830             if (currentBuffer == null) {
831                 currentBuffer = ChannelBuffers.wrappedBuffer(
832                         buffer, ChannelBuffers.wrappedBuffer("=".getBytes()));
833                 //continue
834                 size -= buffer.readableBytes() + 1;
835             } else {
836                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
837                     buffer, ChannelBuffers.wrappedBuffer("=".getBytes()));
838                 //continue
839                 size -= buffer.readableBytes() + 1;
840             }
841             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
842                 buffer = fillChannelBuffer();
843                 return new DefaultHttpChunk(buffer);
844             }
845         }
846         try {
847             buffer = ((HttpData) currentData).getChunk(size);
848         } catch (IOException e) {
849             throw new ErrorDataEncoderException(e);
850         }
851         ChannelBuffer delimiter = null;
852         if (buffer.readableBytes() < size) {
853             // delimiter
854             isKey = true;
855             delimiter = iterator.hasNext() ?
856                     ChannelBuffers.wrappedBuffer("&".getBytes()) :
857                         null;
858         }
859         if (buffer.capacity() == 0) {
860             // end for current InterfaceHttpData, need potentially more data
861             currentData = null;
862             if (currentBuffer == null) {
863                 currentBuffer = delimiter;
864             } else {
865                 if (delimiter != null) {
866                     currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
867                         delimiter);
868                 }
869             }
870             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
871                 buffer = fillChannelBuffer();
872                 return new DefaultHttpChunk(buffer);
873             }
874             return null;
875         }
876         if (currentBuffer == null) {
877             if (delimiter != null) {
878                 currentBuffer = ChannelBuffers.wrappedBuffer(buffer,
879                     delimiter);
880             } else {
881                 currentBuffer = buffer;
882             }
883         } else {
884             if (delimiter != null) {
885                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
886                     buffer, delimiter);
887             } else {
888                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
889                         buffer);
890             }
891         }
892         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
893             // end for current InterfaceHttpData, need more data
894             currentData = null;
895             isKey = true;
896             return null;
897         }
898         buffer = fillChannelBuffer();
899         // size = 0
900         return new DefaultHttpChunk(buffer);
901     }

907     /**
908      * Returns the next available HttpChunk. The caller is responsible to test if this chunk is the
909      * last one (isLast()), in order to stop calling this method.
910      *
911      * @return the next available HttpChunk
912      * @throws ErrorDataEncoderException if the encoding is in error
913      */
914     public HttpChunk nextChunk() throws ErrorDataEncoderException {
915         if (isLastChunk) {
916             isLastChunkSent = true;
917             return new DefaultHttpChunk(ChannelBuffers.EMPTY_BUFFER);
918         }
919         ChannelBuffer buffer;
920         int size = HttpPostBodyUtil.chunkSize;
921         // first test if previous buffer is not empty
922         if (currentBuffer != null) {
923             size -= currentBuffer.readableBytes();
924         }
925         if (size <= 0) {
926             //NextChunk from buffer
927             buffer = fillChannelBuffer();
928             return new DefaultHttpChunk(buffer);
929         }
930         // size > 0
931         if (currentData != null) {
932             // continue to read data
933             if (isMultipart) {
934                 HttpChunk chunk = encodeNextChunkMultipart(size);
935                 if (chunk != null) {
936                     return chunk;
937                 }
938             } else {
939                 HttpChunk chunk = encodeNextChunkUrlEncoded(size);
940                 if (chunk != null) {
941                     //NextChunk Url from currentData
942                     return chunk;
943                 }
944             }
945             size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
946         }
947         if (! iterator.hasNext()) {
948             isLastChunk = true;
949             //NextChunk as last non empty from buffer
950             buffer = currentBuffer;
951             currentBuffer = null;
952             return new DefaultHttpChunk(buffer);
953         }
954         while (size > 0 && iterator.hasNext()) {
955             currentData = iterator.next();
956             HttpChunk chunk;
957             if (isMultipart) {
958                 chunk = encodeNextChunkMultipart(size);
959             } else {
960                 chunk = encodeNextChunkUrlEncoded(size);
961             }
962             if (chunk == null) {
963                 // not enough
964                 size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
965                 continue;
966             }
967             //NextChunk from data
968             return chunk;
969         }
970         // end since no more data
971         isLastChunk = true;
972         if (currentBuffer == null) {
973             isLastChunkSent = true;
974             //LastChunk with no more data
975             return new DefaultHttpChunk(ChannelBuffers.EMPTY_BUFFER);
976         }
977         //Previous LastChunk with no more data
978         buffer = currentBuffer;
979         currentBuffer = null;
980         return new DefaultHttpChunk(buffer);
981     }
