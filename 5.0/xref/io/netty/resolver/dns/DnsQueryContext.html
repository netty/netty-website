16  package io.netty5.resolver.dns;
17  
18  import io.netty5.channel.AddressedEnvelope;
19  import io.netty5.channel.Channel;
20  import io.netty5.handler.codec.dns.AbstractDnsOptPseudoRrRecord;
21  import io.netty5.handler.codec.dns.DnsQuery;
22  import io.netty5.handler.codec.dns.DnsQuestion;
23  import io.netty5.handler.codec.dns.DnsRecord;
24  import io.netty5.handler.codec.dns.DnsResponse;
25  import io.netty5.handler.codec.dns.DnsSection;
26  import io.netty5.util.ReferenceCountUtil;
27  import io.netty5.util.concurrent.Future;
28  import io.netty5.util.concurrent.FutureListener;
29  import io.netty5.util.concurrent.Promise;
30  import io.netty5.util.internal.logging.InternalLogger;
31  import io.netty5.util.internal.logging.InternalLoggerFactory;
32  
33  import java.net.InetSocketAddress;
34  import java.util.concurrent.TimeUnit;
35  
36  import static java.util.Objects.requireNonNull;
37  
38  abstract class DnsQueryContext implements FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>> {
39  
40      private static final InternalLogger logger = InternalLoggerFactory.getInstance(DnsQueryContext.class);
41  
42      private final DnsNameResolver parent;
43      private final Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise;
44      private final int id;
45      private final DnsQuestion question;
46      private final DnsRecord[] additionals;
47      private final DnsRecord optResource;
48      private final InetSocketAddress nameServerAddr;
49  
50      private final boolean recursionDesired;
51      private volatile Future<?> timeoutFuture;
52  
53      DnsQueryContext(DnsNameResolver parent,
54                      InetSocketAddress nameServerAddr,
55                      DnsQuestion question,
56                      DnsRecord[] additionals,
57                      Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise) {
58  
59          this.parent = requireNonNull(parent, "parent");
60          this.nameServerAddr = requireNonNull(nameServerAddr, "nameServerAddr");
61          this.question = requireNonNull(question, "question");
62          this.additionals = requireNonNull(additionals, "additionals");
63          this.promise = requireNonNull(promise, "promise");
64          recursionDesired = parent.isRecursionDesired();
65          id = parent.queryContextManager.add(this);
66  
67          // Ensure we remove the id from the QueryContextManager once the query completes.
68          promise.asFuture().addListener(this);
69  
70          if (parent.isOptResourceEnabled()) {
71              optResource = new AbstractDnsOptPseudoRrRecord(parent.maxPayloadSize(), 0, 0) {
72                  // We may want to remove this in the future and let the user just specify the opt record in the query.
73              };
74          } else {
75              optResource = null;
76          }
77      }
78  
79      InetSocketAddress nameServerAddr() {
80          return nameServerAddr;
81      }
82  
83      DnsQuestion question() {
84          return question;
85      }
86  
87      DnsNameResolver parent() {
88          return parent;
89      }
90  
91      protected abstract DnsQuery newQuery(int id);
92      protected abstract Channel channel();
93      protected abstract String protocol();
94  
95      void query(boolean flush, Promise<Void> writePromise) {
96          final DnsQuestion question = question();
97          final InetSocketAddress nameServerAddr = nameServerAddr();
98          final DnsQuery query = newQuery(id);
99  
100         query.setRecursionDesired(recursionDesired);
101 
102         query.addRecord(DnsSection.QUESTION, question);
103 
104         for (DnsRecord record: additionals) {
105             query.addRecord(DnsSection.ADDITIONAL, record);
106         }
107 
108         if (optResource != null) {
109             query.addRecord(DnsSection.ADDITIONAL, optResource);
110         }
111 
112         if (logger.isDebugEnabled()) {
113             logger.debug("{} WRITE: {}, [{}: {}], {}", channel(), protocol(), id, nameServerAddr, question);
114         }
115 
116         sendQuery(query, flush, writePromise);
117     }
118 
119     private void sendQuery(final DnsQuery query, final boolean flush, final Promise<Void> writePromise) {
120         if (parent.channelReadyPromise.isSuccess()) {
121             writeQuery(query, flush, writePromise);
122         } else if (parent.channelReadyPromise.isFailed()) {
123             failQuery(query, parent.channelReadyPromise.cause(), writePromise);
124         } else {
125             parent.channelReadyPromise.asFuture().addListener(future -> {
126                 if (future.isSuccess()) {
127                     // If the query is done in a late fashion (as the channel was not ready yet) we always flush
128                     // to ensure we did not race with a previous flush() that was done when the Channel was not
129                     // ready yet.
130                     writeQuery(query, true, writePromise);
131                 } else {
132                     failQuery(query, future.cause(), writePromise);
133                 }
134             });
135         }
136     }
137 
138     private void failQuery(DnsQuery query, Throwable cause, Promise<Void> writePromise) {
139         try {
140             promise.tryFailure(cause);
141             writePromise.setFailure(cause);
142         } finally {
143             ReferenceCountUtil.release(query);
144         }
145     }
146 
147     private void writeQuery(final DnsQuery query, final boolean flush, final Promise<Void> writePromise) {
148         final Future<Void> writeFuture = flush ? channel().writeAndFlush(query) :
149                 channel().write(query);
150         if (writeFuture.isDone()) {
151             onQueryWriteCompletion(writeFuture, writePromise);
152         } else {
153             writeFuture.addListener(future ->
154                     onQueryWriteCompletion(future, writePromise));
155         }
156     }
157 
158     private void onQueryWriteCompletion(Future<?> writeFuture, Promise<Void> writePromise) {
159         if (writeFuture.isFailed()) {
160             writePromise.setFailure(writeFuture.cause());
161             tryFailure("failed to send a query via " + protocol(), writeFuture.cause(), false);
162             return;
163         }
164         writePromise.setSuccess(null);
165         // Schedule a query timeout task if necessary.
166         final long queryTimeoutMillis = parent.queryTimeoutMillis();
167         if (queryTimeoutMillis > 0) {
168             timeoutFuture = parent.ch.executor().schedule(() -> {
169                 if (promise.isDone()) {
170                     // Received a response before the query times out.
171                     return;
172                 }
173 
174                 tryFailure("query via " + protocol() + " timed out after " +
175                         queryTimeoutMillis + " milliseconds", null, true);
176                 }, queryTimeoutMillis, TimeUnit.MILLISECONDS);
177         }
178     }
179 
180     /**
181      * Takes ownership of passed envelope
182      */
183     void finish(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope) {
184         final DnsResponse res = envelope.content();
185         if (res.count(DnsSection.QUESTION) != 1) {
186             logger.warn("Received a DNS response with invalid number of questions: {}", envelope);
187         } else if (!question().equals(res.recordAt(DnsSection.QUESTION))) {
188             logger.warn("Received a mismatching DNS response: {}", envelope);
189         } else if (trySuccess(envelope)) {
190             return; // Ownership transferred, don't release
191         }
192         ReferenceCountUtil.release(envelope);
193     }
194 
195     @SuppressWarnings("unchecked")
196     private boolean trySuccess(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope) {
197         return promise.trySuccess((AddressedEnvelope<DnsResponse, InetSocketAddress>) envelope);
198     }
199 
200     boolean tryFailure(String message, Throwable cause, boolean timeout) {
201         if (promise.isDone()) {
202             return false;
203         }
204         final InetSocketAddress nameServerAddr = nameServerAddr();
205 
206         final StringBuilder buf = new StringBuilder(message.length() + 64);
207         buf.append('[')
208            .append(nameServerAddr)
209            .append("] ")
210            .append(message)
211            .append(" (no stack trace available)");
212 
213         final DnsNameResolverException e;
214         if (timeout) {
215             // This was caused by an timeout so use DnsNameResolverTimeoutException to allow the user to
216             // handle it special (like retry the query).
217             e = new DnsNameResolverTimeoutException(nameServerAddr, question(), buf.toString());
218         } else {
219             e = new DnsNameResolverException(nameServerAddr, question(), buf.toString(), cause);
220         }
221         return promise.tryFailure(e);
222     }
223 
224     @Override
225     public void operationComplete(Future<? extends AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
226         // Cancel the timeout task.
227         final Future<?> timeoutFuture = this.timeoutFuture;
228         if (timeoutFuture != null) {
229             this.timeoutFuture = null;
230             timeoutFuture.cancel();
231         }
232 
233         // Remove the id from the manager as soon as the query completes. This may be because of success, failure or
234         // cancellation
235         parent.queryContextManager.remove(nameServerAddr, id);
236     }
