161     /**
162      * Returns {@code true} if any {@link NetworkInterface} supports {@code IPv6}, {@code false} otherwise.
163      */
164     private static boolean anyInterfaceSupportsIpV6() {
165         try {
166             Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
167             while (interfaces.hasMoreElements()) {
168                 NetworkInterface iface = interfaces.nextElement();
169                 Enumeration<InetAddress> addresses = iface.getInetAddresses();
170                 while (addresses.hasMoreElements()) {
171                     InetAddress inetAddress = addresses.nextElement();
172                     if (inetAddress instanceof Inet6Address && !inetAddress.isAnyLocalAddress() &&
173                         !inetAddress.isLoopbackAddress() && !inetAddress.isLinkLocalAddress()) {
174                         return true;
175                     }
176                 }
177             }
178         } catch (SocketException e) {
179             logger.debug("Unable to detect if any interface supports IPv6, assuming IPv4-only", e);
180             // ignore
181         }
182         return false;
183     }

185     private static final DatagramDnsResponseDecoder DATAGRAM_DECODER = new DatagramDnsResponseDecoder() {
186         @Override
187         protected DnsResponse decodeResponse(ChannelHandlerContext ctx, DatagramPacket packet) throws Exception {
188             DnsResponse response = super.decodeResponse(ctx, packet);
189             if (packet.content().isReadable()) {
190                 // If there is still something to read we did stop parsing because of a truncated message.
191                 // This can happen if we enabled EDNS0 but our MTU is not big enough to handle all the
192                 // data.
193                 response.setTruncated(true);
194 
195                 if (logger.isDebugEnabled()) {
196                     logger.debug(
197                             "{} RECEIVED: UDP truncated packet received, consider adjusting maxPayloadSize for the {}.",
198                             ctx.channel(), StringUtil.simpleClassName(DnsNameResolver.class));
199                 }
200             }
201             return response;
202         }
203     };

378     DnsNameResolver(
379             EventLoop eventLoop,
380             ChannelFactory<? extends DatagramChannel> channelFactory,
381             ChannelFactory<? extends SocketChannel> socketChannelFactory,
382             final DnsCache resolveCache,
383             final DnsCnameCache cnameCache,
384             final AuthoritativeDnsServerCache authoritativeDnsServerCache,
385             SocketAddress localAddress,
386             DnsQueryLifecycleObserverFactory dnsQueryLifecycleObserverFactory,
387             long queryTimeoutMillis,
388             ResolvedAddressTypes resolvedAddressTypes,
389             boolean recursionDesired,
390             int maxQueriesPerResolve,
391             int maxPayloadSize,
392             boolean optResourceEnabled,
393             HostsFileEntriesResolver hostsFileEntriesResolver,
394             DnsServerAddressStreamProvider dnsServerAddressStreamProvider,
395             String[] searchDomains,
396             int ndots,
397             boolean decodeIdn,
398             boolean completeOncePreferredResolved) {
399         super(eventLoop);
400         this.queryTimeoutMillis = queryTimeoutMillis > 0
401             ? queryTimeoutMillis
402             : TimeUnit.SECONDS.toMillis(DEFAULT_OPTIONS.timeout());
403         this.resolvedAddressTypes = resolvedAddressTypes != null ? resolvedAddressTypes : DEFAULT_RESOLVE_ADDRESS_TYPES;
404         this.recursionDesired = recursionDesired;
405         this.maxQueriesPerResolve = maxQueriesPerResolve > 0 ? maxQueriesPerResolve : DEFAULT_OPTIONS.attempts();
406         this.maxPayloadSize = checkPositive(maxPayloadSize, "maxPayloadSize");
407         this.optResourceEnabled = optResourceEnabled;
408         this.hostsFileEntriesResolver = requireNonNull(hostsFileEntriesResolver, "hostsFileEntriesResolver");
409         this.dnsServerAddressStreamProvider =
410                 requireNonNull(dnsServerAddressStreamProvider, "dnsServerAddressStreamProvider");
411         this.resolveCache = requireNonNull(resolveCache, "resolveCache");
412         this.cnameCache = requireNonNull(cnameCache, "cnameCache");
413         this.dnsQueryLifecycleObserverFactory =
414                 requireNonNull(dnsQueryLifecycleObserverFactory, "dnsQueryLifecycleObserverFactory");
415         this.searchDomains = searchDomains != null ? searchDomains.clone() : DEFAULT_SEARCH_DOMAINS;
416         this.ndots = ndots >= 0 ? ndots : DEFAULT_OPTIONS.ndots();
417         this.decodeIdn = decodeIdn;
418         this.completeOncePreferredResolved = completeOncePreferredResolved;
419         this.socketChannelFactory = socketChannelFactory;
420         switch (this.resolvedAddressTypes) {
421             case IPV4_ONLY:
422                 supportsAAAARecords = false;
423                 supportsARecords = true;
424                 resolveRecordTypes = IPV4_ONLY_RESOLVED_RECORD_TYPES;
425                 resolvedInternetProtocolFamilies = IPV4_ONLY_RESOLVED_PROTOCOL_FAMILIES;
426                 break;
427             case IPV4_PREFERRED:
428                 supportsAAAARecords = true;
429                 supportsARecords = true;
430                 resolveRecordTypes = IPV4_PREFERRED_RESOLVED_RECORD_TYPES;
431                 resolvedInternetProtocolFamilies = IPV4_PREFERRED_RESOLVED_PROTOCOL_FAMILIES;
432                 break;
433             case IPV6_ONLY:
434                 supportsAAAARecords = true;
435                 supportsARecords = false;
436                 resolveRecordTypes = IPV6_ONLY_RESOLVED_RECORD_TYPES;
437                 resolvedInternetProtocolFamilies = IPV6_ONLY_RESOLVED_PROTOCOL_FAMILIES;
438                 break;
439             case IPV6_PREFERRED:
440                 supportsAAAARecords = true;
441                 supportsARecords = true;
442                 resolveRecordTypes = IPV6_PREFERRED_RESOLVED_RECORD_TYPES;
443                 resolvedInternetProtocolFamilies = IPV6_PREFERRED_RESOLVED_PROTOCOL_FAMILIES;
444                 break;
445             default:
446                 throw new IllegalArgumentException("Unknown ResolvedAddressTypes " + resolvedAddressTypes);
447         }
448         preferredAddressType = preferredAddressType(this.resolvedAddressTypes);
449         this.authoritativeDnsServerCache = requireNonNull(authoritativeDnsServerCache, "authoritativeDnsServerCache");
450         nameServerComparator = new NameServerComparator(preferredAddressType.addressType());
451 
452         Bootstrap b = new Bootstrap();
453         b.group(executor());
454         b.channelFactory(channelFactory);
455         channelReadyPromise = executor().newPromise();
456         final DnsResponseHandler responseHandler = new DnsResponseHandler(channelReadyPromise);
457         b.handler(new ChannelInitializer<DatagramChannel>() {
458             @Override
459             protected void initChannel(DatagramChannel ch) {
460                 ch.pipeline().addLast(DATAGRAM_ENCODER, DATAGRAM_DECODER, responseHandler);
461                 ch.closeFuture().addListener(closeFuture -> {
462                     resolveCache.clear();
463                     cnameCache.clear();
464                     authoritativeDnsServerCache.clear();
465                 });
466             }
467         });
468         b.option(ChannelOption.RCVBUF_ALLOCATOR, new FixedRecvBufferAllocator(maxPayloadSize));
469 
470         try {
471             Future<Void> future;
472             if (localAddress == null) {
473                 b.option(ChannelOption.DATAGRAM_CHANNEL_ACTIVE_ON_REGISTRATION, true);
474                 ch = b.createUnregistered();
475                 future = ch.register();
476             } else {
477                 ch = b.createUnregistered();
478                 future = ch.bind(localAddress);
479             }
480             if (future.isFailed()) {
481                 throw future.cause();
482             } else if (!future.isSuccess()) {
483                 future.addListener(f -> {
484                     Throwable cause = f.cause();
485                     if (cause != null) {
486                         channelReadyPromise.tryFailure(cause);
487                     }
488                 });
489             }
490         } catch (Error | RuntimeException e) {
491             throw e;
492         } catch (Throwable cause) {
493             throw new IllegalStateException("Unable to create / register Channel", cause);
494         }
495     }

507         }

519     /**
520      * Creates a new {@link DnsServerAddressStream} to following a redirected DNS query. By overriding this
521      * it provides the opportunity to sort the name servers before following a redirected DNS query.
522      *
523      * @param hostname the hostname.
524      * @param nameservers The addresses of the DNS servers which are used in the event of a redirect. This may
525      *                    contain resolved and unresolved addresses so the used {@link DnsServerAddressStream} must
526      *                    allow unresolved addresses if you want to include these as well.
527      * @return A {@link DnsServerAddressStream} which will be used to follow the DNS redirect or {@code null} if
528      *         none should be followed.
529      */
530     protected DnsServerAddressStream newRedirectDnsServerStream(
531             @SuppressWarnings("unused") String hostname, List<InetSocketAddress> nameservers) {
532         DnsServerAddressStream cached = authoritativeDnsServerCache().get(hostname);
533         if (cached == null || cached.size() == 0) {
534             // If there is no cache hit (which may be the case for example when a NoopAuthoritativeDnsServerCache
535             // is used), we will just directly use the provided nameservers.
536             nameservers.sort(nameServerComparator);
537             return new SequentialDnsServerAddressStream(nameservers, 0);
538         }
539         return cached;
540     }

808      * {@link #query(DnsQuestion)}, this method handles redirection, CNAMEs and multiple name servers.
809      * If the specified {@link DnsQuestion} is {@code A} or {@code AAAA}, this method looks up the configured
810      * {@link HostsFileEntries} before sending a query to the name servers. If a match is found in the
811      * {@link HostsFileEntries}, a synthetic {@code A} or {@code AAAA} record will be returned.
812      *
813      * @param question the question
814      * @param additionals additional records ({@code OPT})
815      * @param promise the {@link Promise} which will be fulfilled when the resolution is finished
816      *
817      * @return the list of the {@link DnsRecord}s as the result of the resolution
818      */
819     public final Future<List<DnsRecord>> resolveAll(DnsQuestion question, Iterable<DnsRecord> additionals,
820                                                     Promise<List<DnsRecord>> promise) {

822         return resolveAll(question, additionalsArray, promise);

825     private Future<List<DnsRecord>> resolveAll(DnsQuestion question, DnsRecord[] additionals,
826                                                Promise<List<DnsRecord>> promise) {
827         requireNonNull(question, "question");
828         requireNonNull(promise, "promise");
829 
830         // Respect /etc/hosts as well if the record type is A or AAAA.
831         final DnsRecordType type = question.type();
832         final String hostname = question.name();
833 
834         if (type == DnsRecordType.A || type == DnsRecordType.AAAA) {
835             final List<InetAddress> hostsFileEntries = resolveHostsFileEntries(hostname);
836             if (hostsFileEntries != null) {
837                 List<DnsRecord> result = new ArrayList<DnsRecord>();
838                 for (InetAddress hostsFileEntry : hostsFileEntries) {
839                     ByteBuf content = null;
840                     if (hostsFileEntry instanceof Inet4Address) {
841                         if (type == DnsRecordType.A) {
842                             content = Unpooled.wrappedBuffer(hostsFileEntry.getAddress());
843                         }
844                     } else if (hostsFileEntry instanceof Inet6Address) {
845                         if (type == DnsRecordType.AAAA) {
846                             content = Unpooled.wrappedBuffer(hostsFileEntry.getAddress());
847                         }
848                     }
849                     if (content != null) {
850                         // Our current implementation does not support reloading the hosts file,
851                         // so use a fairly large TTL (1 day, i.e. 86400 seconds).
852                         result.add(new DefaultDnsRawRecord(hostname, type, 86400, content));
853                     }
854                 }
855 
856                 if (!result.isEmpty()) {
857                     trySuccess(promise, result);
858                     return promise.asFuture();
859                 }
860             }
861         }
862 
863         // It was not A/AAAA question or there was no entry in /etc/hosts.
864         final DnsServerAddressStream nameServerAddrs =
865                 dnsServerAddressStreamProvider.nameServerAddressStream(hostname);
866         new DnsRecordResolveContext(this, promise, question, additionals, nameServerAddrs, maxQueriesPerResolve)
867                 .resolve(promise);
868         return promise.asFuture();
869     }

871     private static DnsRecord[] toArray(Iterable<DnsRecord> additionals, boolean validateType) {
872         requireNonNull(additionals, "additionals");
873         if (additionals instanceof Collection) {
874             Collection<DnsRecord> records = (Collection<DnsRecord>) additionals;
875             for (DnsRecord r: additionals) {
876                 validateAdditional(r, validateType);
877             }
878             return records.toArray(new DnsRecord[0]);
879         }
880 
881         Iterator<DnsRecord> additionalsIt = additionals.iterator();
882         if (!additionalsIt.hasNext()) {
883             return EMPTY_ADDITIONALS;
884         }
885         List<DnsRecord> records = new ArrayList<>();
886         do {
887             DnsRecord r = additionalsIt.next();
888             validateAdditional(r, validateType);
889             records.add(r);
890         } while (additionalsIt.hasNext());
891 
892         return records.toArray(new DnsRecord[0]);
893     }

895     private static void validateAdditional(DnsRecord record, boolean validateType) {
896         requireNonNull(record, "record");
897         if (validateType && record instanceof DnsRawRecord) {
898             throw new IllegalArgumentException("DnsRawRecord implementations not allowed: " + record);
899         }
900     }

906     /**
907      * Hook designed for extensibility so one can pass a different cache on each resolution attempt
908      * instead of using the global one.
909      */
910     protected void doResolve(String inetHost,
911                              DnsRecord[] additionals,
912                              Promise<InetAddress> promise,
913                              DnsCache resolveCache) throws Exception {
914         if (inetHost == null || inetHost.isEmpty()) {
915             // If an empty hostname is used we should use "localhost", just like InetAddress.getByName(...) does.
916             promise.setSuccess(loopbackAddress());
917             return;
918         }
919         final InetAddress address = NetUtil.createInetAddressFromIpAddressString(inetHost);
920         if (address != null) {
921             // The inetHost is actually an ipaddress.
922             promise.setSuccess(address);
923             return;
924         }
925 
926         final String hostname = hostname(inetHost);
927 
928         InetAddress hostsFileEntry = resolveHostsFileEntry(hostname);
929         if (hostsFileEntry != null) {
930             promise.setSuccess(hostsFileEntry);
931             return;
932         }
933 
934         if (!doResolveCached(hostname, additionals, promise, resolveCache)) {
935             doResolveUncached(hostname, additionals, promise, resolveCache, true);
936         }
937     }

939     private boolean doResolveCached(String hostname,
940                                     DnsRecord[] additionals,
941                                     Promise<InetAddress> promise,
942                                     DnsCache resolveCache) {
943         final List<? extends DnsCacheEntry> cachedEntries = resolveCache.get(hostname, additionals);
944         if (cachedEntries == null || cachedEntries.isEmpty()) {
945             return false;
946         }
947 
948         Throwable cause = cachedEntries.get(0).cause();
949         if (cause == null) {
950             final int numEntries = cachedEntries.size();
951             // Find the first entry with the preferred address type.
952             for (InternetProtocolFamily f : resolvedInternetProtocolFamilies) {
953                 for (int i = 0; i < numEntries; i++) {
954                     final DnsCacheEntry e = cachedEntries.get(i);
955                     if (f.addressType().isInstance(e.address())) {
956                         trySuccess(promise, e.address());
957                         return true;
958                     }
959                 }
960             }
961             return false;
962         } else {
963             tryFailure(promise, cause);
964             return true;
965         }
966     }

968     static <T> boolean trySuccess(Promise<T> promise, T result) {
969         final boolean notifiedRecords = promise.trySuccess(result);
970         if (!notifiedRecords) {
971             // There is nothing really wrong with not be able to notify the promise as we may have raced here because
972             // of multiple queries that have been executed. Log it with trace level anyway just in case the user
973             // wants to better understand what happened.
974             logger.trace("Failed to notify success ({}) to a promise: {}", result, promise);
975         }
976         return notifiedRecords;
977     }
978 
979     private static void tryFailure(Promise<?> promise, Throwable cause) {
980         if (!promise.tryFailure(cause)) {
981             // There is nothing really wrong with not be able to notify the promise as we may have raced here because
982             // of multiple queries that have been executed. Log it with trace level anyway just in case the user
983             // wants to better understand what happened.
984             logger.trace("Failed to notify failure to a promise: {}", promise, cause);
985         }
986     }

988     private void doResolveUncached(String hostname,
989                                    DnsRecord[] additionals,
990                                    final Promise<InetAddress> promise,
991                                    DnsCache resolveCache, boolean completeEarlyIfPossible) {
992         final Promise<List<InetAddress>> allPromise = executor().newPromise();
993         doResolveAllUncached(hostname, additionals, promise, allPromise, resolveCache, true);
994         allPromise.asFuture().addListener(future -> {

997             } else {
998                 tryFailure(promise, future.cause());
999             }
1000         });

1012     protected void doResolveAll(String inetHost,
1013                                 DnsRecord[] additionals,
1014                                 Promise<List<InetAddress>> promise,
1015                                 DnsCache resolveCache) throws Exception {
1016         if (inetHost == null || inetHost.isEmpty()) {
1017             // If an empty hostname is used we should use "localhost", just like InetAddress.getAllByName(...) does.
1018             promise.setSuccess(Collections.singletonList(loopbackAddress()));
1019             return;
1020         }
1021         final InetAddress address = NetUtil.createInetAddressFromIpAddressString(inetHost);
1022         if (address != null) {
1023             // The unresolvedAddress was created via a String that contains an ipaddress.
1024             promise.setSuccess(Collections.singletonList(address));
1025             return;
1026         }
1027 
1028         final String hostname = hostname(inetHost);
1029 
1030         List<InetAddress> hostsFileEntries = resolveHostsFileEntries(hostname);
1031         if (hostsFileEntries != null) {
1032             promise.setSuccess(hostsFileEntries);
1033             return;
1034         }
1035 
1036         if (!doResolveAllCached(hostname, additionals, promise, resolveCache, resolvedInternetProtocolFamilies)) {
1037             doResolveAllUncached(hostname, additionals, promise, promise,
1038                                  resolveCache, completeOncePreferredResolved);
1039         }

1042     static boolean doResolveAllCached(String hostname,
1043                                       DnsRecord[] additionals,
1044                                       Promise<List<InetAddress>> promise,
1045                                       DnsCache resolveCache,
1046                                       InternetProtocolFamily[] resolvedInternetProtocolFamilies) {
1047         final List<? extends DnsCacheEntry> cachedEntries = resolveCache.get(hostname, additionals);
1048         if (cachedEntries == null || cachedEntries.isEmpty()) {
1049             return false;
1050         }
1051 
1052         Throwable cause = cachedEntries.get(0).cause();
1053         if (cause == null) {
1054             List<InetAddress> result = null;
1055             final int numEntries = cachedEntries.size();
1056             for (InternetProtocolFamily f : resolvedInternetProtocolFamilies) {
1057                 for (int i = 0; i < numEntries; i++) {
1058                     final DnsCacheEntry e = cachedEntries.get(i);
1059                     if (f.addressType().isInstance(e.address())) {
1060                         if (result == null) {
1061                             result = new ArrayList<>(numEntries);
1062                         }
1063                         result.add(e.address());
1064                     }
1065                 }
1066             }
1067             if (result != null) {
1068                 trySuccess(promise, result);
1069                 return true;
1070             }
1071             return false;
1072         } else {
1073             tryFailure(promise, cause);
1074             return true;
1075         }
1076     }

1240     private final class DnsResponseHandler implements ChannelHandler {
1241 
1242         private final Promise<Channel> channelActivePromise;
1243 
1244         DnsResponseHandler(Promise<Channel> channelActivePromise) {
1245             this.channelActivePromise = channelActivePromise;
1246         }
1247 
1248         @Override
1249         public void channelRead(ChannelHandlerContext ctx, Object msg) {
1250             final DatagramDnsResponse res = (DatagramDnsResponse) msg;
1251             final int queryId = res.id();
1252 
1253             if (logger.isDebugEnabled()) {
1254                 logger.debug("{} RECEIVED: UDP [{}: {}], {}", ch, queryId, res.sender(), res);
1255             }
1256 
1257             final DnsQueryContext qCtx = queryContextManager.get(res.sender(), queryId);
1258             if (qCtx == null) {
1259                 logger.debug("Received a DNS response with an unknown ID: UDP [{}: {}]", ch, queryId);
1260                 res.release();
1261                 return;
1262             }
1263 
1264             // Check if the response was truncated and if we can fallback to TCP to retry.
1265             if (!res.isTruncated() || socketChannelFactory == null) {
1266                 qCtx.finish(res);
1267                 return;
1268             }
1269 
1270             Bootstrap bs = new Bootstrap();
1271             bs.option(ChannelOption.SO_REUSEADDR, true)
1272             .group(executor())
1273             .channelFactory(socketChannelFactory)
1274             .handler(TCP_ENCODER);
1275             bs.connect(res.sender()).addListener(future -> {
1276                 if (future.isFailed()) {
1277                     if (logger.isDebugEnabled()) {
1278                         logger.debug("{} Unable to fallback to TCP [{}]", queryId, future.cause());
1279                     }
1280 
1281                     // TCP fallback failed, just use the truncated response.
1282                     qCtx.finish(res);
1283                     return;
1284                 }
1285                 final Channel channel = future.getNow();
1286 
1287                 Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise =
1288                         channel.executor().newPromise();
1289                 final TcpDnsQueryContext tcpCtx = new TcpDnsQueryContext(DnsNameResolver.this, channel,
1290                         (InetSocketAddress) channel.remoteAddress(), qCtx.question(),
1291                         EMPTY_ADDITIONALS, promise);
1292 
1293                 channel.pipeline().addLast(new TcpDnsResponseDecoder());
1294                 channel.pipeline().addLast(new ChannelHandler() {
1295                     @Override
1296                     public void channelRead(ChannelHandlerContext ctx1, Object msg1) {
1297                         Channel channel = ctx1.channel();
1298                         DnsResponse response = (DnsResponse) msg1;
1299                         int queryId1 = response.id();
1300 
1301                         if (logger.isDebugEnabled()) {
1302                             logger.debug("{} RECEIVED: TCP [{}: {}], {}", channel, queryId1,
1303                                     channel.remoteAddress(), response);
1304                         }
1305 
1306                         DnsQueryContext foundCtx = queryContextManager.get(res.sender(), queryId1);
1307                         if (foundCtx == tcpCtx) {
1308                             tcpCtx.finish(new AddressedEnvelopeAdapter(
1309                                     (InetSocketAddress) ctx1.channel().remoteAddress(),
1310                                     (InetSocketAddress) ctx1.channel().localAddress(),
1311                                     response));
1312                         } else {
1313                             response.release();
1314                             tcpCtx.tryFailure("Received TCP DNS response with unexpected ID", null, false);
1315                             logger.debug("Received a DNS response with an unexpected ID: TCP [{}: {}]",
1316                                     channel, queryId);
1317                         }
1318                     }
1319 
1320                     @Override
1321                     public void exceptionCaught(ChannelHandlerContext ctx1, Throwable cause) {
1322                         if (tcpCtx.tryFailure("TCP fallback error", cause, false) && logger.isDebugEnabled()) {
1323                             logger.debug("{} Error during processing response: TCP [{}: {}]",
1324                                     ctx1.channel(), queryId,
1325                                     ctx1.channel().remoteAddress(), cause);
1326                         }
1327                     }
1328                 });
1329 
1330                 promise.asFuture().addListener(addressEnvelopeFuture -> {
1331                     channel.close();
1332 
1333                     if (addressEnvelopeFuture.isSuccess()) {
1334                         qCtx.finish(addressEnvelopeFuture.getNow());
1335                         res.release();
1336                     } else {
1337                         // TCP fallback failed, just use the truncated response.
1338                         qCtx.finish(res);
1339                     }
1340                 });
1341                 tcpCtx.query(true, channel.newPromise());
1342             });
1343         }

1345         @Override
1346         public void channelActive(ChannelHandlerContext ctx) {
1347             ctx.fireChannelActive();
1348             channelActivePromise.trySuccess(ctx.channel());
1349         }

1351         @Override
1352         public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
1353             if (cause instanceof CorruptedFrameException) {
1354                 logger.debug("Unable to decode DNS response: UDP [{}]", ctx.channel(), cause);
1355             } else {
1356                 logger.warn("Unexpected exception: UDP [{}]", ctx.channel(), cause);
1357             }
1358         }
