179     private static final FutureAndDelay CANCELLED = new FutureAndDelay(CANCELLED_FUTURE, Integer.MIN_VALUE);

181     // Two years are supported by all our EventLoop implementations and so safe to use as maximum.
182     // See also: https://github.com/netty/netty/commit/b47fb817991b42ec8808c7d26538f3f2464e1fa6
183     static final int MAX_SUPPORTED_TTL_SECS = (int) TimeUnit.DAYS.toSeconds(365 * 2);

185     private final ConcurrentMap<String, Entries> resolveCache = new ConcurrentHashMap<>();
186 
187     /**
188      * Remove everything from the cache.
189      */
190     final void clear() {
191         while (!resolveCache.isEmpty()) {
192             for (Iterator<Entry<String, Entries>> i = resolveCache.entrySet().iterator(); i.hasNext();) {
193                 Map.Entry<String, Entries> e = i.next();
194                 i.remove();
195 
196                 e.getValue().clearAndCancel();
197             }
198         }
199     }
200 
201     /**
202      * Clear all entries (if anything exists) for the given hostname and return {@code true} if anything was removed.
203      */
204     final boolean clear(String hostname) {
205         Entries entries = resolveCache.remove(hostname);
206         return entries != null && entries.clearAndCancel();
207     }
208 
209     /**
210      * Returns all caches entries for the given hostname.
211      */
212     final List<? extends E> get(String hostname) {
213         Entries entries = resolveCache.get(hostname);
214         return entries == null ? null : entries.get();
215     }
216 
217     /**
218      * Cache a value for the given hostname that will automatically expire once the TTL is reached.
219      */
220     final void cache(String hostname, E value, int ttl, EventLoop loop) {
221         Entries entries = resolveCache.get(hostname);
222         if (entries == null) {
223             entries = new Entries(hostname);
224             Entries oldEntries = resolveCache.putIfAbsent(hostname, entries);
225             if (oldEntries != null) {
226                 entries = oldEntries;
227             }
228         }
229         entries.add(value, ttl, loop);
230     }
231 
232     /**
233      * Return the number of hostnames for which we have cached something.
234      */
235     final int size() {
236         return resolveCache.size();
237     }
238 
239     /**
240      * Returns {@code true} if this entry should replace all other entries that are already cached for the hostname.
241      */
242     protected abstract boolean shouldReplaceAll(E entry);
243 
244     /**
245      * Sort the {@link List} for a {@code hostname} before caching these.
246      */
247     protected void sortEntries(
248             @SuppressWarnings("unused") String hostname, @SuppressWarnings("unused") List<E> entries) {
249         // NOOP.
250     }
251 
252     /**
253      * Returns {@code true} if both entries are equal.
254      */
255     protected abstract boolean equals(E entry, E otherEntry);

257     // Directly extend AtomicReference for intrinsics and also to keep memory overhead low.
258     private final class Entries extends AtomicReference<List<E>> implements Runnable {
259 
260         private final String hostname;
261         // Needs to be package-private to be able to access it via the AtomicReferenceFieldUpdater
262         volatile FutureAndDelay expirationFuture;
263 
264         Entries(String hostname) {
265             super(Collections.emptyList());
266             this.hostname = hostname;
267         }
268 
269         void add(E e, int ttl, EventLoop loop) {
270             if (!shouldReplaceAll(e)) {
271                 for (;;) {
272                     List<E> entries = get();
273                     if (!entries.isEmpty()) {
274                         final E firstEntry = entries.get(0);
275                         if (shouldReplaceAll(firstEntry)) {
276                             assert entries.size() == 1;
277 
278                             if (compareAndSet(entries, singletonList(e))) {
279                                 scheduleCacheExpirationIfNeeded(ttl, loop);
280                                 return;
281                             } else {
282                                 // Need to try again as CAS failed
283                                 continue;
284                             }
285                         }
286 
287                         // Create a new List for COW semantics
288                         List<E> newEntries = new ArrayList<>(entries.size() + 1);
289                         int i = 0;
290                         E replacedEntry = null;
291                         do {
292                             E entry = entries.get(i);
293                             // Only add old entry if the address is not the same as the one we try to add as well.
294                             // In this case we will skip it and just add the new entry as this may have
295                             // more up-to-date data and cancel the old after we were able to update the cache.
296                             if (!Cache.this.equals(e, entry)) {
297                                 newEntries.add(entry);
298                             } else {
299                                 replacedEntry = entry;
300                                 newEntries.add(e);
301 
302                                 ++i;
303                                 for (; i < entries.size(); ++i) {
304                                     newEntries.add(entries.get(i));
305                                 }
306                                 break;
307                             }
308                         } while (++i < entries.size());
309                         if (replacedEntry == null) {
310                             newEntries.add(e);
311                         }
312                         sortEntries(hostname, newEntries);
313 
314                         if (compareAndSet(entries, Collections.unmodifiableList(newEntries))) {
315                             scheduleCacheExpirationIfNeeded(ttl, loop);
316                             return;
317                         }
318                     } else if (compareAndSet(entries, singletonList(e))) {
319                         scheduleCacheExpirationIfNeeded(ttl, loop);
320                         return;
321                     }
322                 }
323             } else {
324                 set(singletonList(e));
325                 scheduleCacheExpirationIfNeeded(ttl, loop);
326             }
327         }
328 
329         private void scheduleCacheExpirationIfNeeded(int ttl, EventLoop loop) {
330             for (;;) {
331                 // We currently don't calculate a new TTL when we need to retry the CAS as we don't expect this to
332                 // be invoked very concurrently and also we use SECONDS anyway. If this ever becomes a problem
333                 // we can reconsider.
334                 FutureAndDelay oldFuture = FUTURE_UPDATER.get(this);
335                 if (oldFuture == null || oldFuture.getDelay(TimeUnit.SECONDS) > ttl) {
336                     Future<?> newFuture = loop.schedule(this, ttl, TimeUnit.SECONDS);
337                     // It is possible that
338                     // 1. task will fire in between this line, or
339                     // 2. multiple timers may be set if there is concurrency
340                     // (1) Shouldn't be a problem because we will fail the CAS and then the next loop will see CANCELLED
341                     //     so the ttl will not be less, and we will bail out of the loop.
342                     // (2) This is a trade-off to avoid concurrency resulting in contention on a synchronized block.
343                     if (FUTURE_UPDATER.compareAndSet(this, oldFuture, new FutureAndDelay(newFuture, ttl))) {
344                         if (oldFuture != null) {
345                             oldFuture.cancel();
346                         }
347                         break;
348                     } else {
349                         // There was something else scheduled in the meantime... Cancel and try again.
350                         newFuture.cancel();
351                     }
352                 } else {
353                     break;
354                 }
355             }
356         }
357 
358         boolean clearAndCancel() {
359             List<E> entries = getAndSet(Collections.emptyList());
360             if (entries.isEmpty()) {
361                 return false;
362             }
363 
364             FutureAndDelay expirationFuture = FUTURE_UPDATER.getAndSet(this, CANCELLED);
365             if (expirationFuture != null) {
366                 expirationFuture.cancel();
367             }
368 
369             return true;
370         }
371 
372         @Override
373         public void run() {
374             // We always remove all entries for a hostname once one entry expire. This is not the
375             // most efficient to do but this way we can guarantee that if a DnsResolver
376             // be configured to prefer one ip family over the other we will not return unexpected
377             // results to the enduser if one of the A or AAAA records has different TTL settings.
378             //
379             // As a TTL is just a hint of the maximum time a cache is allowed to cache stuff it's
380             // completely fine to remove the entry even if the TTL is not reached yet.
381             //
382             // See https://github.com/netty/netty/issues/7329
383             resolveCache.remove(hostname, this);
384 
385             clearAndCancel();
386         }
387     }
388 
389     private static final class FutureAndDelay implements Delayed {
390         final Future<?> future;
391         final long deadlineNanos;
392 
393         private FutureAndDelay(Future<?> future, int ttl) {
394             this.future = Objects.requireNonNull(future, "future");
395             deadlineNanos = System.nanoTime() + TimeUnit.SECONDS.toNanos(ttl);
396         }
397 
398         void cancel() {
399             future.cancel();
400         }
401 
402         @Override
403         public long getDelay(TimeUnit unit) {
404             return unit.convert(deadlineNanos - System.nanoTime(), TimeUnit.NANOSECONDS);
405         }
406 
407         @Override
408         public int compareTo(Delayed other) {
409             return Long.compare(deadlineNanos, other.getDelay(TimeUnit.NANOSECONDS));
410         }
411 
412         @Override
413         public boolean equals(Object o) {
414             return o instanceof FutureAndDelay && compareTo((FutureAndDelay) o) == 0;
415         }
416 
417         @Override
418         public int hashCode() {
419             int result = future.hashCode();
420             result = 31 * result + (int) (deadlineNanos ^ deadlineNanos >>> 32);
421             return result;
422         }
423     }
