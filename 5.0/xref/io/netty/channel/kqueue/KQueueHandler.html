44  /**
45   * {@link IoHandler} which uses kqueue under the covers. Only works on BSD!
46   */
47  @UnstableApi
48  public final class KQueueHandler implements IoHandler {
49      private static final InternalLogger logger = InternalLoggerFactory.getInstance(KQueueHandler.class);
50      private static final AtomicIntegerFieldUpdater<KQueueHandler> WAKEN_UP_UPDATER =
51              AtomicIntegerFieldUpdater.newUpdater(KQueueHandler.class, "wakenUp");
52      private static final int KQUEUE_WAKE_UP_IDENT = 0;
53  
54      static {
55          // Ensure JNI is initialized by the time this class is loaded by this time!
56          // We use unix-common methods in this class which are backed by JNI methods.
57          KQueue.ensureAvailability();
58      }
59  
60      private final boolean allowGrowing;
61      private final FileDescriptor kqueueFd;
62      private final KQueueEventArray changeList;
63      private final KQueueEventArray eventList;
64      private final SelectStrategy selectStrategy;
65      private final IovArray iovArray = new IovArray();
66      private final IntSupplier selectNowSupplier = this::kqueueWaitNow;
67      private final IntObjectMap<AbstractKQueueChannel> channels = new IntObjectHashMap<AbstractKQueueChannel>(4096);
68  
69      private volatile int wakenUp;
70  
71      private static AbstractKQueueChannel cast(Channel channel) {
72          if (channel instanceof AbstractKQueueChannel) {
73              return (AbstractKQueueChannel) channel;
74          }
75          throw new IllegalArgumentException("Channel of type " + StringUtil.simpleClassName(channel) + " not supported");
76      }
77  
78      private KQueueHandler() {
79          this(0, DefaultSelectStrategyFactory.INSTANCE.newSelectStrategy());
80      }
81  
82      private KQueueHandler(int maxEvents, SelectStrategy strategy) {
83          selectStrategy = requireNonNull(strategy, "strategy");
84          this.kqueueFd = Native.newKQueue();
85          if (maxEvents == 0) {
86              allowGrowing = true;
87              maxEvents = 4096;
88          } else {
89              allowGrowing = false;
90          }
91          changeList = new KQueueEventArray(maxEvents);
92          eventList = new KQueueEventArray(maxEvents);
93          int result = Native.keventAddUserEvent(kqueueFd.intValue(), KQUEUE_WAKE_UP_IDENT);
94          if (result < 0) {
95              destroy();
96              throw new IllegalStateException("kevent failed to add user event with errno: " + (-result));
97          }
98      }

111                                               final SelectStrategyFactory selectStrategyFactory) {
112         checkPositiveOrZero(maxEvents, "maxEvents");
113         requireNonNull(selectStrategyFactory, "selectStrategyFactory");
114         return () -> new KQueueHandler(maxEvents, selectStrategyFactory.newSelectStrategy());

117     @Override
118     public void register(Channel channel) {
119         final AbstractKQueueChannel kQueueChannel = cast(channel);
120         final int id = kQueueChannel.fd().intValue();
121         AbstractKQueueChannel old = channels.put(id, kQueueChannel);
122         // We either expect to have no Channel in the map with the same FD or that the FD of the old Channel is already
123         // closed.
124         assert old == null || !old.isOpen();
125 
126         kQueueChannel.register0(new KQueueRegistration() {
127             @Override
128             public void evSet(short filter, short flags, int fflags) {
129                 KQueueHandler.this.evSet(kQueueChannel, filter, flags, fflags);
130             }
131 
132             @Override
133             public IovArray cleanArray() {
134                 return KQueueHandler.this.cleanArray();
135             }
136         });
137     }
138 
139     @Override
140     public void deregister(Channel channel) throws Exception {
141         AbstractKQueueChannel kQueueChannel = cast(channel);
142         int fd = kQueueChannel.fd().intValue();
143 
144         AbstractKQueueChannel old = channels.remove(fd);
145         if (old != null && old != kQueueChannel) {
146             // The Channel mapping was already replaced due FD reuse, put back the stored Channel.
147             channels.put(fd, old);
148 
149             // If we found another Channel in the map that is mapped to the same FD the given Channel MUST be closed.
150             assert !kQueueChannel.isOpen();
151         } else if (kQueueChannel.isOpen()) {
152             // Remove the filters. This is only needed if it's still open as otherwise it will be automatically
153             // removed once the file-descriptor is closed.
154             //
155             // See also https://www.freebsd.org/cgi/man.cgi?query=kqueue&sektion=2
156             kQueueChannel.unregisterFilters();
157         }
158 
159         kQueueChannel.deregister0();
160     }
161 
162     private void evSet(AbstractKQueueChannel ch, short filter, short flags, int fflags) {
163         changeList.evSet(ch, filter, flags, fflags);
164     }
165 
166     private IovArray cleanArray() {
167         iovArray.clear();
168         return iovArray;
169     }

171     @Override
172     public void wakeup(boolean inEventLoop) {
173         if (!inEventLoop && WAKEN_UP_UPDATER.compareAndSet(this, 0, 1)) {
174             wakeup();
175         }
176     }
177 
178     private void wakeup() {
179         Native.keventTriggerUserEvent(kqueueFd.intValue(), KQUEUE_WAKE_UP_IDENT);
180         // Note that the result may return an error (e.g. errno = EBADF after the event loop has been shutdown).
181         // So it is not very practical to assert the return value is always >= 0.
182     }
183 
184     private int kqueueWait(IoExecutionContext context, boolean oldWakeup) throws IOException {
185         // If a task was submitted when wakenUp value was 1, the task didn't get a chance to produce wakeup event.
186         // So we need to check task queue again before calling kqueueWait. If we don't, the task might be pended
187         // until kqueueWait was timed out. It might be pended until idle timeout if IdleStateHandler existed
188         // in pipeline.
189         if (oldWakeup && !context.canBlock()) {
190             return kqueueWaitNow();
191         }
192 
193         long totalDelay = context.delayNanos(System.nanoTime());
194         int delaySeconds = (int) min(totalDelay / 1000000000L, Integer.MAX_VALUE);
195         return kqueueWait(delaySeconds, (int) min(totalDelay - delaySeconds * 1000000000L, Integer.MAX_VALUE));
196     }
197 
198     private int kqueueWaitNow() throws IOException {
199         return kqueueWait(0, 0);
200     }
201 
202     private int kqueueWait(int timeoutSec, int timeoutNs) throws IOException {
203         int numEvents = Native.keventWait(kqueueFd.intValue(), changeList, eventList, timeoutSec, timeoutNs);
204         changeList.clear();
205         return numEvents;
206     }
207 
208     private void processReady(int ready) {
209         for (int i = 0; i < ready; ++i) {
210             final short filter = eventList.filter(i);
211             final short flags = eventList.flags(i);
212             final int fd = eventList.fd(i);
213             if (filter == Native.EVFILT_USER || (flags & Native.EV_ERROR) != 0) {
214                 // EV_ERROR is returned if the FD is closed synchronously (which removes from kqueue) and then
215                 // we later attempt to delete the filters from kqueue.
216                 assert filter != Native.EVFILT_USER ||
217                         (filter == Native.EVFILT_USER && fd == KQUEUE_WAKE_UP_IDENT);
218                 continue;
219             }
220 
221             AbstractKQueueChannel channel = channels.get(fd);
222             if (channel == null) {
223                 // This may happen if the channel has already been closed, and it will be removed from kqueue anyways.
224                 // We also handle EV_ERROR above to skip this even early if it is a result of a referencing a closed and
225                 // thus removed from kqueue FD.
226                 logger.warn("events[{}]=[{}, {}] had no channel!", i, fd, filter);
227                 continue;
228             }
229 
230             AbstractKQueueUnsafe unsafe = (AbstractKQueueUnsafe) channel.unsafe();
231             // First check for EPOLLOUT as we may need to fail the connect Promise before try
232             // to read from the file descriptor.
233             if (filter == Native.EVFILT_WRITE) {
234                 unsafe.writeReady();
235             } else if (filter == Native.EVFILT_READ) {
236                 // Check READ before EOF to ensure all data is read before shutting down the input.
237                 unsafe.readReady(eventList.data(i));
238             } else if (filter == Native.EVFILT_SOCK && (eventList.fflags(i) & Native.NOTE_RDHUP) != 0) {
239                 unsafe.readEOF();
240             }
241 
242             // Check if EV_EOF was set, this will notify us for connection-reset in which case
243             // we may close the channel directly or try to read more data depending on the state of the
244             // Channel and also depending on the AbstractKQueueChannel subtype.
245             if ((flags & Native.EV_EOF) != 0) {
246                 unsafe.readEOF();
247             }
248         }
249     }

251     @Override
252     public int run(IoExecutionContext context) {
253         int handled = 0;
254         try {
255             int strategy = selectStrategy.calculateStrategy(selectNowSupplier, !context.canBlock());
256             switch (strategy) {
257                 case SelectStrategy.CONTINUE:
258                     return 0;
259 
260                 case SelectStrategy.BUSY_WAIT:
261                     // fall-through to SELECT since the busy-wait is not supported with kqueue
262 
263                 case SelectStrategy.SELECT:
264                     strategy = kqueueWait(context, WAKEN_UP_UPDATER.getAndSet(this, 0) == 1);
265 
266                     // 'wakenUp.compareAndSet(false, true)' is always evaluated
267                     // before calling 'selector.wakeup()' to reduce the wake-up
268                     // overhead. (Selector.wakeup() is an expensive operation.)
269                     //
270                     // However, there is a race condition in this approach.
271                     // The race condition is triggered when 'wakenUp' is set to
272                     // true too early.
273                     //
274                     // 'wakenUp' is set to true too early if:
275                     // 1) Selector is waken up between 'wakenUp.set(false)' and
276                     //    'selector.select(...)'. (BAD)
277                     // 2) Selector is waken up between 'selector.select(...)' and
278                     //    'if (wakenUp.get()) { ... }'. (OK)
279                     //
280                     // In the first case, 'wakenUp' is set to true and the
281                     // following 'selector.select(...)' will wake up immediately.
282                     // Until 'wakenUp' is set to false again in the next round,
283                     // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
284                     // any attempt to wake up the Selector will fail, too, causing
285                     // the following 'selector.select(...)' call to block
286                     // unnecessarily.
287                     //
288                     // To fix this problem, we wake up the selector again if wakenUp
289                     // is true immediately after selector.select(...).
290                     // It is inefficient in that it wakes up the selector for both
291                     // the first case (BAD - wake-up required) and the second case
292                     // (OK - no wake-up required).
293 
294                     if (wakenUp == 1) {
295                         wakeup();
296                     }
297                     // fallthrough
298                 default:
299             }
300 
301             if (strategy > 0) {
302                 handled = strategy;
303                 processReady(strategy);
304             }
305             if (allowGrowing && strategy == eventList.capacity()) {
306                 //increase the size of the array as we needed the whole space for the events
307                 eventList.realloc(false);
308             }
309         } catch (Error e) {
310             throw e;
311         } catch (Throwable t) {
312             handleLoopException(t);
313         }
314         return handled;
315     }

317     @Override
318     public void destroy() {
319         try {
320             try {
321                 kqueueFd.close();
322             } catch (IOException e) {
323                 logger.warn("Failed to close the kqueue fd.", e);
324             }
325         } finally {
326             // Cleanup all native memory!
327             changeList.free();
328             eventList.free();
329         }
330     }
331 
332     @Override
333     public void prepareToDestroy() {
334         try {
335             kqueueWaitNow();
336         } catch (IOException e) {
337             // ignore on close
338         }
339 
340         // Using the intermediate collection to prevent ConcurrentModificationException.
341         // In the `close()` method, the channel is deleted from `channels` map.
342         AbstractKQueueChannel[] localChannels = channels.values().toArray(new AbstractKQueueChannel[0]);
343 
344         for (AbstractKQueueChannel ch: localChannels) {
345             ch.unsafe().close(ch.newPromise());
346         }
347     }
348 
349     private static void handleLoopException(Throwable t) {
350         logger.warn("Unexpected exception in the selector loop.", t);
351 
352         // Prevent possible consecutive immediate failures that lead to
353         // excessive CPU consumption.
354         try {
355             Thread.sleep(1000);
356         } catch (InterruptedException e) {
357             // Ignore.
358         }
359     }
