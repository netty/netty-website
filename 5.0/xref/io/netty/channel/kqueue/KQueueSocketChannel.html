16  package io.netty5.channel.kqueue;
17  
18  import io.netty5.buffer.ByteBuf;
19  import io.netty5.channel.Channel;
20  import io.netty5.channel.ChannelOutboundBuffer;
21  import io.netty5.channel.EventLoop;
22  import io.netty5.channel.socket.ServerSocketChannel;
23  import io.netty5.channel.socket.SocketChannel;
24  import io.netty5.channel.unix.IovArray;
25  import io.netty5.util.concurrent.GlobalEventExecutor;
26  import io.netty5.util.internal.UnstableApi;
27  
28  import java.net.InetSocketAddress;
29  import java.net.SocketAddress;
30  import java.util.concurrent.Executor;
31  
32  @UnstableApi
33  public final class KQueueSocketChannel extends AbstractKQueueStreamChannel implements SocketChannel {
34      private final KQueueSocketChannelConfig config;
35  
36      public KQueueSocketChannel(EventLoop eventLoop) {
37          super(null, eventLoop, BsdSocket.newSocketStream(), false);
38          config = new KQueueSocketChannelConfig(this);
39      }
40  
41      public KQueueSocketChannel(EventLoop eventLoop, int fd) {
42          super(eventLoop, new BsdSocket(fd));
43          config = new KQueueSocketChannelConfig(this);
44      }
45  
46      KQueueSocketChannel(Channel parent, EventLoop eventLoop, BsdSocket fd, InetSocketAddress remoteAddress) {
47          super(parent, eventLoop, fd, remoteAddress);
48          config = new KQueueSocketChannelConfig(this);
49      }
50  
51      @Override
52      public InetSocketAddress remoteAddress() {
53          return (InetSocketAddress) super.remoteAddress();
54      }
55  
56      @Override
57      public InetSocketAddress localAddress() {
58          return (InetSocketAddress) super.localAddress();
59      }
60  
61      @Override
62      public KQueueSocketChannelConfig config() {
63          return config;
64      }
65  
66      @Override
67      public ServerSocketChannel parent() {
68          return (ServerSocketChannel) super.parent();
69      }
70  
71      @Override
72      protected boolean doConnect0(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
73          if (config.isTcpFastOpenConnect()) {
74              ChannelOutboundBuffer outbound = unsafe().outboundBuffer();
75              outbound.addFlush();
76              Object curr;
77              if ((curr = outbound.current()) instanceof ByteBuf) {
78                  ByteBuf initialData = (ByteBuf) curr;
79                  // Don't bother with TCP FastOpen if we don't have any initial data to send anyway.
80                  if (initialData.isReadable()) {
81                      IovArray iov = new IovArray(config.getAllocator().directBuffer());
82                      try {
83                          iov.add(initialData, initialData.readerIndex(), initialData.readableBytes());
84                          int bytesSent = socket.connectx(
85                                  (InetSocketAddress) localAddress, (InetSocketAddress) remoteAddress, iov, true);
86                          writeFilter(true);
87                          outbound.removeBytes(Math.abs(bytesSent));
88                          // The `connectx` method returns a negative number if connection is in-progress.
89                          // So we should return `true` to indicate that connection was established, if it's positive.
90                          return bytesSent > 0;
91                      } finally {
92                          iov.release();
93                      }
94                  }
95              }
96          }
97          return super.doConnect0(remoteAddress, localAddress);
98      }
99  
100     @Override
101     protected AbstractKQueueUnsafe newUnsafe() {
102         return new KQueueSocketChannelUnsafe();
103     }
104 
105     private final class KQueueSocketChannelUnsafe extends KQueueStreamUnsafe {
106         @Override
107         protected Executor prepareToClose() {
108             try {
109                 // Check isOpen() first as otherwise it will throw a RuntimeException
110                 // when call getSoLinger() as the fd is not valid anymore.
111                 if (isOpen() && config().getSoLinger() > 0) {
112                     // We need to cancel this key of the channel so we may not end up in a eventloop spin
113                     // because we try to read or write until the actual close happens which may be later due
114                     // SO_LINGER handling.
115                     // See https://github.com/netty/netty/issues/4449
116                     doDeregister();
117                     return GlobalEventExecutor.INSTANCE;
118                 }
119             } catch (Throwable ignore) {
120                 // Ignore the error as the underlying channel may be closed in the meantime and so
121                 // getSoLinger() may produce an exception. In this case we just return null.
122                 // See https://github.com/netty/netty/issues/4449
123             }
124             return null;
125         }
126     }
127 }
