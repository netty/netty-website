33  /**
34   * A nexus to a network socket or a component which is capable of I/O
35   * operations such as read, write, connect, and bind.
36   * <p>
37   * A channel provides a user:
38   * <ul>
39   * <li>the current state of the channel (e.g. is it open? is it connected?),</li>
40   * <li>the {@linkplain ChannelConfig configuration parameters} of the channel (e.g. receive buffer size),</li>
41   * <li>the I/O operations that the channel supports (e.g. read, write, connect, and bind), and</li>
42   * <li>the {@link ChannelPipeline} which handles all I/O events and requests
43   *     associated with the channel.</li>
44   * </ul>
45   *
46   * <h3>All I/O operations are asynchronous.</h3>
47   * <p>
48   * All I/O operations in Netty are asynchronous.  It means any I/O calls will
49   * return immediately with no guarantee that the requested I/O operation has
50   * been completed at the end of the call.  Instead, you will be returned with
51   * a {@link Future} instance which will notify you when the requested I/O
52   * operation has succeeded, failed, or canceled.
53   *
54   * <h3>Channels are hierarchical</h3>
55   * <p>
56   * A {@link Channel} can have a {@linkplain #parent() parent} depending on
57   * how it was created.  For instance, a {@link SocketChannel}, that was accepted
58   * by {@link ServerSocketChannel}, will return the {@link ServerSocketChannel}
59   * as its parent on {@link #parent()}.
60   * <p>
61   * The semantics of the hierarchical structure depends on the transport
62   * implementation where the {@link Channel} belongs to.  For example, you could
63   * write a new {@link Channel} implementation that creates the sub-channels that
64   * share one socket connection, as <a href="http://beepcore.org/">BEEP</a> and
65   * <a href="https://en.wikipedia.org/wiki/Secure_Shell">SSH</a> do.
66   *
67   * <h3>Downcast to access transport-specific operations</h3>
68   * <p>
69   * Some transports exposes additional operations that is specific to the
70   * transport.  Down-cast the {@link Channel} to sub-type to invoke such
71   * operations.  For example, with the old I/O datagram transport, multicast
72   * join / leave operations are provided by {@link DatagramChannel}.
73   *
74   * <h3>Release resources</h3>
75   * <p>
76   * It is important to call {@link #close()} to release all
77   * resources once you are done with the {@link Channel}. This ensures all resources are
78   * released in a proper way, i.e. filehandles.
79   */
80  public interface Channel extends AttributeMap, ChannelOutboundInvoker, Comparable<Channel> {
81  
82      /**
83       * Returns the globally unique identifier of this {@link Channel}.
84       */
85      ChannelId id();
86  
87      /**
88       * Return the {@link EventLoop} this {@link Channel} was registered to.
89       */
90      @Override
91      EventLoop executor();
92  
93      /**
94       * Returns the parent of this channel.
95       *
96       * @return the parent channel.
97       *         {@code null} if this channel does not have a parent channel.
98       */
99      Channel parent();
100 
101     /**
102      * Returns the configuration of this channel.
103      */
104     ChannelConfig config();
105 
106     /**
107      * Returns {@code true} if the {@link Channel} is open and may get active later
108      */
109     boolean isOpen();
110 
111     /**
112      * Returns {@code true} if the {@link Channel} is registered with an {@link EventLoop}.
113      */
114     boolean isRegistered();
115 
116     /**
117      * Return {@code true} if the {@link Channel} is active and so connected.
118      */
119     boolean isActive();
120 
121     /**
122      * Return the {@link ChannelMetadata} of the {@link Channel} which describe the nature of the {@link Channel}.
123      */
124     ChannelMetadata metadata();
125 
126     /**
127      * Returns the local address where this channel is bound to.  The returned
128      * {@link SocketAddress} is supposed to be down-cast into more concrete
129      * type such as {@link InetSocketAddress} to retrieve the detailed
130      * information.
131      *
132      * @return the local address of this channel.
133      *         {@code null} if this channel is not bound.
134      */
135     SocketAddress localAddress();
136 
137     /**
138      * Returns the remote address where this channel is connected to.  The
139      * returned {@link SocketAddress} is supposed to be down-cast into more
140      * concrete type such as {@link InetSocketAddress} to retrieve the detailed
141      * information.
142      *
143      * @return the remote address of this channel.
144      *         {@code null} if this channel is not connected.
145      *         If this channel is not connected but it can receive messages
146      *         from arbitrary remote addresses (e.g. {@link DatagramChannel},
147      *         use {@link DatagramPacket#recipient()} to determine
148      *         the origination of the received message as this method will
149      *         return {@code null}.
150      */
151     SocketAddress remoteAddress();
152 
153     /**
154      * Returns the {@link Future} which will be notified when this
155      * channel is closed.  This method always returns the same future instance.
156      */
157     Future<Void> closeFuture();
158 
159     /**
160      * Returns {@code true} if and only if the I/O thread will perform the
161      * requested write operation immediately.  Any write requests made when
162      * this method returns {@code false} are queued until the I/O thread is
163      * ready to process the queued write requests.
164      */
165     default boolean isWritable() {
166         ChannelOutboundBuffer buf = unsafe().outboundBuffer();
167         return buf != null && buf.isWritable();
168     }
169 
170     /**
171      * Get how many bytes can be written until {@link #isWritable()} returns {@code false}.
172      * This quantity will always be non-negative. If {@link #isWritable()} is {@code false} then 0.
173      */
174     default long bytesBeforeUnwritable() {
175         ChannelOutboundBuffer buf = unsafe().outboundBuffer();
176         // isWritable() is currently assuming if there is no outboundBuffer then the channel is not writable.
177         // We should be consistent with that here.
178         return buf != null ? buf.bytesBeforeUnwritable() : 0;
179     }
180 
181     /**
182      * Get how many bytes must be drained from underlying buffers until {@link #isWritable()} returns {@code true}.
183      * This quantity will always be non-negative. If {@link #isWritable()} is {@code true} then 0.
184      */
185     default long bytesBeforeWritable() {
186         ChannelOutboundBuffer buf = unsafe().outboundBuffer();
187         // isWritable() is currently assuming if there is no outboundBuffer then the channel is not writable.
188         // We should be consistent with that here.
189         return buf != null ? buf.bytesBeforeWritable() : Long.MAX_VALUE;
190     }
191 
192     /**
193      * Returns an <em>internal-use-only</em> object that provides unsafe operations.
194      */
195     Unsafe unsafe();
196 
197     /**
198      * Return the assigned {@link ChannelPipeline}.
199      */
200     ChannelPipeline pipeline();
201 
202     /**
203      * Return the assigned {@link ByteBufAllocator} which will be used to allocate {@link ByteBuf}s.
204      * @deprecated Use {@link #bufferAllocator()}
205      */
206     @Deprecated
207     default ByteBufAllocator alloc() {
208         return config().getAllocator();
209     }
210 
211     /**
212      * Return the assigned {@link BufferAllocator} which will be used to allocate {@link io.netty5.buffer.api.Buffer}s.
213      */
214     default BufferAllocator bufferAllocator() {
215         return config().getBufferAllocator();
216     }
217 
218     @Override
219     default Channel read() {
220         pipeline().read();
221         return this;
222     }
223 
224     @Override
225     default Future<Void> bind(SocketAddress localAddress) {
226         return pipeline().bind(localAddress);
227     }
228 
229     @Override
230     default Future<Void> connect(SocketAddress remoteAddress) {
231         return pipeline().connect(remoteAddress);
232     }
233 
234     @Override
235     default Future<Void> connect(SocketAddress remoteAddress, SocketAddress localAddress) {
236         return pipeline().connect(remoteAddress, localAddress);
237     }
238 
239     @Override
240     default Future<Void> disconnect() {
241         return pipeline().disconnect();
242     }
243 
244     @Override
245     default Future<Void> close() {
246         return pipeline().close();
247     }
248 
249     @Override
250     default Future<Void> register() {
251         return pipeline().register();
252     }
253 
254     @Override
255     default Future<Void> deregister() {
256         return pipeline().deregister();
257     }
258 
259     @Override
260     default Future<Void> write(Object msg) {
261         return pipeline().write(msg);
262     }
263 
264     @Override
265     default Future<Void> writeAndFlush(Object msg) {
266         return pipeline().writeAndFlush(msg);
267     }
268 
269     @Override
270     default Channel flush() {
271         pipeline().flush();
272         return this;
273     }
274 
275     /**
276      * <em>Unsafe</em> operations that should <em>never</em> be called from user-code. These methods
277      * are only provided to implement the actual transport, and must be invoked from an I/O thread except for the
278      * following methods:
279      * <ul>
280      *   <li>{@link #localAddress()}</li>
281      *   <li>{@link #remoteAddress()}</li>
282      *   <li>{@link #closeForcibly()}</li>
283      *   <li>{@link #register(Promise)}</li>
284      *   <li>{@link #deregister(Promise)}</li>
285      * </ul>
286      */
287     interface Unsafe {
288 
289         /**
290          * Return the assigned {@link RecvBufferAllocator.Handle} which will be used to allocate {@link ByteBuf}'s when
291          * receiving data.
292          */
293         RecvBufferAllocator.Handle recvBufAllocHandle();
294 
295         /**
296          * Return the {@link SocketAddress} to which is bound local or
297          * {@code null} if none.
298          */
299         SocketAddress localAddress();
300 
301         /**
302          * Return the {@link SocketAddress} to which is bound remote or
303          * {@code null} if none is bound yet.
304          */
305         SocketAddress remoteAddress();
306 
307         /**
308          * Register the {@link Channel} of the {@link Promise} and notify
309          * the {@link Future} once the registration was complete.
310          */
311         void register(Promise<Void> promise);
312 
313         /**
314          * Bind the {@link SocketAddress} to the {@link Channel} of the {@link Promise} and notify
315          * it once its done.
316          */
317         void bind(SocketAddress localAddress, Promise<Void> promise);
318 
319         /**
320          * Connect the {@link Channel} of the given {@link Future} with the given remote {@link SocketAddress}.
321          * If a specific local {@link SocketAddress} should be used it need to be given as argument. Otherwise just
322          * pass {@code null} to it.
323          *
324          * The {@link Promise} will get notified once the connect operation was complete.
325          */
326         void connect(SocketAddress remoteAddress, SocketAddress localAddress, Promise<Void> promise);
327 
328         /**
329          * Disconnect the {@link Channel} of the {@link Future} and notify the {@link Promise} once the
330          * operation was complete.
331          */
332         void disconnect(Promise<Void> promise);
333 
334         /**
335          * Close the {@link Channel} of the {@link Promise} and notify the {@link Promise} once the
336          * operation was complete.
337          */
338         void close(Promise<Void> promise);
339 
340         /**
341          * Closes the {@link Channel} immediately without firing any events.  Probably only useful
342          * when registration attempt failed.
343          */
344         void closeForcibly();
345 
346         /**
347          * Deregister the {@link Channel} of the {@link Promise} from {@link EventLoop} and notify the
348          * {@link Promise} once the operation was complete.
349          */
350         void deregister(Promise<Void> promise);
351 
352         /**
353          * Schedules a read operation that fills the inbound buffer of the first {@link ChannelHandler} in the
354          * {@link ChannelPipeline}.  If there's already a pending read operation, this method does nothing.
355          */
356         void beginRead();
357 
358         /**
359          * Schedules a write operation.
360          */
361         void write(Object msg, Promise<Void> promise);
362 
363         /**
364          * Flush out all write operations scheduled via {@link #write(Object, Promise)}.
365          */
366         void flush();
367 
368         /**
369          * Returns the {@link ChannelOutboundBuffer} of the {@link Channel} where the pending write requests are stored.
370          */
371         ChannelOutboundBuffer outboundBuffer();
372     }
373 }
