138         final Selector unwrappedSelector;

142             this.unwrappedSelector = unwrappedSelector;
143             this.selector = unwrappedSelector;

152     private SelectorTuple openSelector() {
153         final Selector unwrappedSelector;
154         try {
155             unwrappedSelector = provider.openSelector();
156         } catch (IOException e) {
157             throw new ChannelException("failed to open a new selector", e);
158         }
159 
160         if (DISABLE_KEY_SET_OPTIMIZATION) {
161             return new SelectorTuple(unwrappedSelector);
162         }
163 
164         Object maybeSelectorImplClass = AccessController.doPrivileged((PrivilegedAction<Object>) () -> {
165             try {
166                 return Class.forName(
167                         "sun.nio.ch.SelectorImpl",
168                         false,
169                         PlatformDependent.getSystemClassLoader());
170             } catch (Throwable cause) {
171                 return cause;
172             }
173         });
174 
175         if (!(maybeSelectorImplClass instanceof Class) ||
176             // ensure the current selector implementation is what we can instrument.
177             !((Class<?>) maybeSelectorImplClass).isAssignableFrom(unwrappedSelector.getClass())) {
178             if (maybeSelectorImplClass instanceof Throwable) {
179                 Throwable t = (Throwable) maybeSelectorImplClass;
180                 logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, t);
181             }
182             return new SelectorTuple(unwrappedSelector);
183         }
184 
185         final Class<?> selectorImplClass = (Class<?>) maybeSelectorImplClass;
186         final SelectedSelectionKeySet selectedKeySet = new SelectedSelectionKeySet();
187 
188         Object maybeException = AccessController.doPrivileged((PrivilegedAction<Object>) () -> {
189             try {
190                 Field selectedKeysField = selectorImplClass.getDeclaredField("selectedKeys");
191                 Field publicSelectedKeysField = selectorImplClass.getDeclaredField("publicSelectedKeys");
192 
193                 if (PlatformDependent.hasUnsafe()) {
194                     // Let us try to use sun.misc.Unsafe to replace the SelectionKeySet.
195                     // This allows us to also do this in Java9+ without any extra flags.
196                     long selectedKeysFieldOffset = PlatformDependent.objectFieldOffset(selectedKeysField);
197                     long publicSelectedKeysFieldOffset =
198                             PlatformDependent.objectFieldOffset(publicSelectedKeysField);
199 
200                     if (selectedKeysFieldOffset != -1 && publicSelectedKeysFieldOffset != -1) {
201                         PlatformDependent.putObject(
202                                 unwrappedSelector, selectedKeysFieldOffset, selectedKeySet);
203                         PlatformDependent.putObject(
204                                 unwrappedSelector, publicSelectedKeysFieldOffset, selectedKeySet);
205                         return null;
206                     }
207                     // We could not retrieve the offset, lets try reflection as last-resort.
208                 }
209 
210                 Throwable cause = ReflectionUtil.trySetAccessible(selectedKeysField, true);
211                 if (cause != null) {
212                     return cause;
213                 }
214                 cause = ReflectionUtil.trySetAccessible(publicSelectedKeysField, true);
215                 if (cause != null) {
216                     return cause;
217                 }
218 
219                 selectedKeysField.set(unwrappedSelector, selectedKeySet);
220                 publicSelectedKeysField.set(unwrappedSelector, selectedKeySet);
221                 return null;
222             } catch (NoSuchFieldException | IllegalAccessException e) {
223                 return e;
224             }
225         });
226 
227         if (maybeException instanceof Exception) {
228             selectedKeys = null;
229             Exception e = (Exception) maybeException;
230             logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, e);
231             return new SelectorTuple(unwrappedSelector);
232         }
233         selectedKeys = selectedKeySet;
234         logger.trace("instrumented a special java.util.Set into: {}", unwrappedSelector);
235         return new SelectorTuple(unwrappedSelector,
236                                  new SelectedSelectionKeySetSelector(unwrappedSelector, selectedKeySet));
237     }

246     /**
247      * Registers an arbitrary {@link SelectableChannel}, not necessarily created by Netty, to the {@link Selector}
248      * of this event loop.  Once the specified {@link SelectableChannel} is registered, the specified {@code task} will
249      * be executed by this event loop when the {@link SelectableChannel} is ready.
250      */
251     public void register(final SelectableChannel ch, final int interestOps, final NioTask<?> task) {
252         requireNonNull(ch, "ch");
253         if (interestOps == 0) {
254             throw new IllegalArgumentException("interestOps must be non-zero.");
255         }
256         if ((interestOps & ~ch.validOps()) != 0) {
257             throw new IllegalArgumentException(
258                     "invalid interestOps: " + interestOps + "(validOps: " + ch.validOps() + ')');
259         }
260         requireNonNull(task, "task");
261 
262         register0(ch, interestOps, task);
263     }
264 
265     private void register0(SelectableChannel ch, int interestOps, NioTask<?> task) {
266         try {
267             ch.register(unwrappedSelector, interestOps, task);
268         } catch (Exception e) {
269             throw new EventLoopException("failed to register a channel", e);
270         }
271     }

273     /**
274      * Replaces the current {@link Selector} of this event loop with newly created {@link Selector}s to work
275      * around the infamous epoll 100% CPU bug.
276      */
277     void rebuildSelector() {
278         final Selector oldSelector = selector;
279         final SelectorTuple newSelectorTuple;
280 
281         if (oldSelector == null) {
282             return;
283         }
284 
285         try {
286             newSelectorTuple = openSelector();
287         } catch (Exception e) {
288             logger.warn("Failed to create a new Selector.", e);
289             return;
290         }
291 
292         // Register all channels to the new Selector.
293         int nChannels = 0;
294         for (SelectionKey key: oldSelector.keys()) {
295             Object a = key.attachment();
296             try {
297                 if (!key.isValid() || key.channel().keyFor(newSelectorTuple.unwrappedSelector) != null) {
298                     continue;
299                 }
300 
301                 int interestOps = key.interestOps();
302                 key.cancel();
303                 SelectionKey newKey = key.channel().register(newSelectorTuple.unwrappedSelector, interestOps, a);
304                 if (a instanceof AbstractNioChannel) {
305                     // Update SelectionKey
306                     ((AbstractNioChannel) a).selectionKey = newKey;
307                 }
308                 nChannels ++;
309             } catch (Exception e) {
310                 logger.warn("Failed to re-register a Channel to the new Selector.", e);
311                 if (a instanceof AbstractNioChannel) {
312                     AbstractNioChannel ch = (AbstractNioChannel) a;
313                     ch.unsafe().close(ch.newPromise());
314                 } else {
315                     @SuppressWarnings("unchecked")
316                     NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
317                     invokeChannelUnregistered(task, key, e);
318                 }
319             }
320         }
321 
322         selector = newSelectorTuple.selector;
323         unwrappedSelector = newSelectorTuple.unwrappedSelector;
324 
325         try {
326             // time to close the old selector as everything else is registered to the new one
327             oldSelector.close();
328         } catch (Throwable t) {
329             if (logger.isWarnEnabled()) {
330                 logger.warn("Failed to close the old Selector.", t);
331             }
332         }
333 
334         if (logger.isInfoEnabled()) {
335             logger.info("Migrated " + nChannels + " channel(s) to the new Selector.");
336         }
337     }

339     private static AbstractNioChannel cast(Channel channel) {
340         if (channel instanceof AbstractNioChannel) {
341             return (AbstractNioChannel) channel;
342         }
343         throw new IllegalArgumentException("Channel of type " + StringUtil.simpleClassName(channel) + " not supported");
344     }

346     @Override
347     public void register(Channel channel) throws Exception {
348         AbstractNioChannel nioChannel = cast(channel);
349         boolean selected = false;
350         for (;;) {
351             try {
352                 nioChannel.selectionKey = nioChannel.javaChannel().register(unwrappedSelector(), 0, nioChannel);
353                 return;
354             } catch (CancelledKeyException e) {
355                 if (!selected) {
356                     // Force the Selector to select now as the "canceled" SelectionKey may still be
357                     // cached and not removed because no Select.select(..) operation was called yet.
358                     selectNow();
359                     selected = true;
360                 } else {
361                     // We forced a select operation on the selector before but the SelectionKey is still cached
362                     // for whatever reason. JDK bug ?
363                     throw e;
364                 }
365             }
366         }
367     }
368 
369     @Override
370     public void deregister(Channel channel) {
371         AbstractNioChannel nioChannel = cast(channel);
372         cancel(nioChannel.selectionKey());
373     }

375     @Override
376     public int run(IoExecutionContext runner) {
377         int handled = 0;
378         try {
379             try {
380                 switch (selectStrategy.calculateStrategy(selectNowSupplier, !runner.canBlock())) {
381                     case SelectStrategy.CONTINUE:
382                         return 0;
383 
384                     case SelectStrategy.BUSY_WAIT:
385                         // fall-through to SELECT since the busy-wait is not supported with NIO
386 
387                     case SelectStrategy.SELECT:
388                         select(runner, wakenUp.getAndSet(false));
389 
390                         // 'wakenUp.compareAndSet(false, true)' is always evaluated
391                         // before calling 'selector.wakeup()' to reduce the wake-up
392                         // overhead. (Selector.wakeup() is an expensive operation.)
393                         //
394                         // However, there is a race condition in this approach.
395                         // The race condition is triggered when 'wakenUp' is set to
396                         // true too early.
397                         //
398                         // 'wakenUp' is set to true too early if:
399                         // 1) Selector is waken up between 'wakenUp.set(false)' and
400                         //    'selector.select(...)'. (BAD)
401                         // 2) Selector is waken up between 'selector.select(...)' and
402                         //    'if (wakenUp.get()) { ... }'. (OK)
403                         //
404                         // In the first case, 'wakenUp' is set to true and the
405                         // following 'selector.select(...)' will wake up immediately.
406                         // Until 'wakenUp' is set to false again in the next round,
407                         // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
408                         // any attempt to wake up the Selector will fail, too, causing
409                         // the following 'selector.select(...)' call to block
410                         // unnecessarily.
411                         //
412                         // To fix this problem, we wake up the selector again if wakenUp
413                         // is true immediately after selector.select(...).
414                         // It is inefficient in that it wakes up the selector for both
415                         // the first case (BAD - wake-up required) and the second case
416                         // (OK - no wake-up required).
417 
418                         if (wakenUp.get()) {
419                             selector.wakeup();
420                         }
421                         // fall through
422                     default:
423                 }
424             } catch (IOException e) {
425                 // If we receive an IOException here its because the Selector is messed up. Let's rebuild
426                 // the selector and retry. https://github.com/netty/netty/issues/8566
427                 rebuildSelector();
428                 handleLoopException(e);
429                 return 0;
430             }
431 
432             cancelledKeys = 0;
433             needsToSelectAgain = false;
434             handled = processSelectedKeys();
435         } catch (Error e) {
436             throw e;
437         } catch (Throwable t) {
438             handleLoopException(t);
439         }
440         return handled;
441     }
442 
443     private static void handleLoopException(Throwable t) {
444         logger.warn("Unexpected exception in the selector loop.", t);
445 
446         // Prevent possible consecutive immediate failures that lead to
447         // excessive CPU consumption.
448         try {
449             Thread.sleep(1000);
450         } catch (InterruptedException e) {
451             // Ignore.
452         }
453     }

459             return processSelectedKeysPlain(selector.selectedKeys());

472     private void cancel(SelectionKey key) {
473         key.cancel();
474         cancelledKeys ++;
475         if (cancelledKeys >= CLEANUP_INTERVAL) {
476             cancelledKeys = 0;
477             needsToSelectAgain = true;
478         }
479     }

481     private int processSelectedKeysPlain(Set<SelectionKey> selectedKeys) {
482         // check if the set is empty and if so just return to not create garbage by
483         // creating a new Iterator every time even if there is nothing to process.
484         // See https://github.com/netty/netty/issues/597
485         if (selectedKeys.isEmpty()) {
486             return 0;
487         }
488 
489         Iterator<SelectionKey> i = selectedKeys.iterator();
490         int handled = 0;
491         for (;;) {
492             final SelectionKey k = i.next();
493             i.remove();
494 
495             processSelectedKey(k);
496             ++handled;
497 
498             if (!i.hasNext()) {
499                 break;
500             }
501 
502             if (needsToSelectAgain) {
503                 selectAgain();
504                 selectedKeys = selector.selectedKeys();
505 
506                 // Create the iterator again to avoid ConcurrentModificationException
507                 if (selectedKeys.isEmpty()) {
508                     break;
509                 } else {
510                     i = selectedKeys.iterator();
511                 }
512             }
513         }
514         return handled;
515     }
516 
517     private int processSelectedKeysOptimized() {
518         int handled = 0;
519         for (int i = 0; i < selectedKeys.size; ++i) {
520             final SelectionKey k = selectedKeys.keys[i];
521             // null out entry in the array to allow to have it GC'ed once the Channel close
522             // See https://github.com/netty/netty/issues/2363
523             selectedKeys.keys[i] = null;
524 
525             processSelectedKey(k);
526             ++handled;
527 
528             if (needsToSelectAgain) {
529                 // null out entries in the array to allow to have it GC'ed once the Channel close
530                 // See https://github.com/netty/netty/issues/2363
531                 selectedKeys.reset(i + 1);
532 
533                 selectAgain();
534                 i = -1;
535             }
536         }
537         return handled;
538     }

540     private void processSelectedKey(SelectionKey k) {
541         final Object a = k.attachment();
542 
543         if (a instanceof AbstractNioChannel) {
544             processSelectedKey(k, (AbstractNioChannel) a);
545         } else {
546             @SuppressWarnings("unchecked")
547             NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
548             processSelectedKey(k, task);
549         }
550     }
551 
552     private void processSelectedKey(SelectionKey k, AbstractNioChannel ch) {
553         final AbstractNioChannel.NioUnsafe unsafe = ch.unsafe();
554         if (!k.isValid()) {
555 
556             // close the channel if the key is not valid anymore
557             unsafe.close(ch.newPromise());
558             return;
559         }
560 
561         try {
562             int readyOps = k.readyOps();
563             // We first need to call finishConnect() before try to trigger a read(...) or write(...) as otherwise
564             // the NIO JDK channel implementation may throw a NotYetConnectedException.
565             if ((readyOps & SelectionKey.OP_CONNECT) != 0) {
566                 // remove OP_CONNECT as otherwise Selector.select(..) will always return without blocking
567                 // See https://github.com/netty/netty/issues/924
568                 int ops = k.interestOps();
569                 ops &= ~SelectionKey.OP_CONNECT;
570                 k.interestOps(ops);
571 
572                 unsafe.finishConnect();
573             }
574 
575             // Process OP_WRITE first as we may be able to write some queued buffers and so free memory.
576             if ((readyOps & SelectionKey.OP_WRITE) != 0) {
577                 // Call forceFlush which will also take care of clear the OP_WRITE once there is nothing left to write
578                 ch.unsafe().forceFlush();
579             }
580 
581             // Also check for readOps of 0 to workaround possible JDK bug which may otherwise lead
582             // to a spin loop
583             if ((readyOps & (SelectionKey.OP_READ | SelectionKey.OP_ACCEPT)) != 0 || readyOps == 0) {
584                 unsafe.read();
585             }
586         } catch (CancelledKeyException ignored) {
587             unsafe.close(ch.newPromise());
588         }
589     }

591     private static void processSelectedKey(SelectionKey k, NioTask<SelectableChannel> task) {
592         int state = 0;
593         try {
594             task.channelReady(k.channel(), k);
595             state = 1;
596         } catch (Exception e) {
597             k.cancel();
598             invokeChannelUnregistered(task, k, e);
599             state = 2;
600         } finally {
601             switch (state) {
602             case 0:
603                 k.cancel();
604                 invokeChannelUnregistered(task, k, null);
605                 break;
606             case 1:
607                 if (!k.isValid()) { // Cancelled by channelReady()
608                     invokeChannelUnregistered(task, k, null);
609                 }
610                 break;
611             default:
612                  break;
613             }
614         }
615     }

617     @Override
618     public void prepareToDestroy() {
619         selectAgain();
620         Set<SelectionKey> keys = selector.keys();
621         Collection<AbstractNioChannel> channels = new ArrayList<>(keys.size());
622         for (SelectionKey k: keys) {
623             Object a = k.attachment();
624             if (a instanceof AbstractNioChannel) {
625                 channels.add((AbstractNioChannel) a);
626             } else {
627                 k.cancel();
628                 @SuppressWarnings("unchecked")
629                 NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
630                 invokeChannelUnregistered(task, k, null);
631             }
632         }
633 
634         for (AbstractNioChannel ch: channels) {
635             ch.unsafe().close(ch.newPromise());
636         }
637     }

639     private static void invokeChannelUnregistered(NioTask<SelectableChannel> task, SelectionKey k, Throwable cause) {

641             task.channelUnregistered(k.channel(), cause);
642         } catch (Exception e) {

644         }

658     private int selectNow() throws IOException {
659         try {
660             return selector.selectNow();
661         } finally {
662             // restore wakeup state if needed
663             if (wakenUp.get()) {
664                 selector.wakeup();
665             }
666         }
667     }

669     private void select(IoExecutionContext runner, boolean oldWakenUp) throws IOException {
670         Selector selector = this.selector;
671         try {
672             int selectCnt = 0;
673             long currentTimeNanos = System.nanoTime();
674             long selectDeadLineNanos = currentTimeNanos + runner.delayNanos(currentTimeNanos);
675 
676             for (;;) {
677                 long timeoutMillis = (selectDeadLineNanos - currentTimeNanos + 500000L) / 1000000L;
678                 if (timeoutMillis <= 0) {
679                     if (selectCnt == 0) {
680                         selector.selectNow();
681                         selectCnt = 1;
682                     }
683                     break;
684                 }
685 
686                 // If a task was submitted when wakenUp value was true, the task didn't get a chance to call
687                 // Selector#wakeup. So we need to check task queue again before executing select operation.
688                 // If we don't, the task might be pended until select operation was timed out.
689                 // It might be pended until idle timeout if IdleStateHandler existed in pipeline.
690                 if (!runner.canBlock() && wakenUp.compareAndSet(false, true)) {
691                     selector.selectNow();
692                     selectCnt = 1;
693                     break;
694                 }
695 
696                 int selectedKeys = selector.select(timeoutMillis);
697                 selectCnt ++;
698 
699                 if (selectedKeys != 0 || oldWakenUp || wakenUp.get() || !runner.canBlock()) {
700                     // - Selected something,
701                     // - waken up by user, or
702                     // - the task queue has a pending task.
703                     // - a scheduled task is ready for processing
704                     break;
705                 }
706                 if (Thread.interrupted()) {
707                     // Thread was interrupted so reset selected keys and break so we not run into a busy loop.
708                     // As this is most likely a bug in the handler of the user or it's client library we will
709                     // also log it.
710                     //
711                     // See https://github.com/netty/netty/issues/2426
712                     if (logger.isDebugEnabled()) {
713                         logger.debug("Selector.select() returned prematurely because " +
714                                 "Thread.currentThread().interrupt() was called. Use " +
715                                 "NioHandler.shutdownGracefully() to shutdown the NioHandler.");
716                     }
717                     selectCnt = 1;
718                     break;
719                 }
720 
721                 long time = System.nanoTime();
722                 if (time - TimeUnit.MILLISECONDS.toNanos(timeoutMillis) >= currentTimeNanos) {
723                     // timeoutMillis elapsed without anything selected.
724                     selectCnt = 1;
725                 } else if (SELECTOR_AUTO_REBUILD_THRESHOLD > 0 &&
726                         selectCnt >= SELECTOR_AUTO_REBUILD_THRESHOLD) {
727                     // The code exists in an extra method to ensure the method is not too big to inline as this
728                     // branch is not very likely to get hit very frequently.
729                     selector = selectRebuildSelector(selectCnt);
730                     selectCnt = 1;
731                     break;
732                 }
733 
734                 currentTimeNanos = time;
735             }
736 
737             if (selectCnt > MIN_PREMATURE_SELECTOR_RETURNS) {
738                 if (logger.isDebugEnabled()) {
739                     logger.debug("Selector.select() returned prematurely {} times in a row for Selector {}.",
740                             selectCnt - 1, selector);
741                 }
742             }
743         } catch (CancelledKeyException e) {
744             if (logger.isDebugEnabled()) {
745                 logger.debug(CancelledKeyException.class.getSimpleName() + " raised by a Selector {} - JDK bug?",
746                         selector, e);
747             }
748             // Harmless exception - log anyway
749         }
750     }
751 
752     private Selector selectRebuildSelector(int selectCnt) throws IOException {
753         // The selector returned prematurely many times in a row.
754         // Rebuild the selector to work around the problem.
755         logger.warn(
756                 "Selector.select() returned prematurely {} times in a row; rebuilding Selector {}.",
757                 selectCnt, selector);
758 
759         rebuildSelector();
760         Selector selector = this.selector;
761 
762         // Select again to populate selectedKeys.
763         selector.selectNow();
764         return selector;
765     }
766 
767     private void selectAgain() {
768         needsToSelectAgain = false;
769         try {
770             selector.selectNow();
771         } catch (Throwable t) {
772             logger.warn("Failed to update SelectionKeys.", t);
773         }
774     }
