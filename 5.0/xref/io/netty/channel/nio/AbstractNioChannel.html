59      private final Runnable clearReadPendingRunnable = this::clearReadPending0;

61      /**
62       * The future of the current connection attempt.  If not null, subsequent
63       * connection attempts will fail.
64       */
65      private Promise<Void> connectPromise;
66      private Future<?> connectTimeoutFuture;
67      private SocketAddress requestedRemoteAddress;

69      /**
70       * Create a new instance
71       *
72       * @param parent            the parent {@link Channel} by which this instance was created. May be {@code null}
73       * @param eventLoop         the {@link EventLoop} to use for all I/O.
74       * @param ch                the underlying {@link SelectableChannel} on which it operates
75       * @param readInterestOp    the ops to set to receive data from the {@link SelectableChannel}
76       */
77      protected AbstractNioChannel(Channel parent, EventLoop eventLoop, SelectableChannel ch, int readInterestOp) {
78          super(parent, eventLoop);
79          this.ch = ch;
80          this.readInterestOp = readInterestOp;
81          try {
82              ch.configureBlocking(false);
83          } catch (IOException e) {
84              try {
85                  ch.close();
86              } catch (IOException e2) {
87                  logger.warn(
88                              "Failed to close a partially initialized socket.", e2);
89              }
90  
91              throw new ChannelException("Failed to enter non-blocking mode.", e);
92          }
93      }

97          return ch.isOpen();

147     /**
148      * Set read pending to {@code false}.
149      */
150     protected final void clearReadPending() {
151         if (isRegistered()) {
152             EventLoop eventLoop = executor();
153             if (eventLoop.inEventLoop()) {
154                 clearReadPending0();
155             } else {
156                 eventLoop.execute(clearReadPendingRunnable);
157             }
158         } else {
159             // Best effort if we are not registered yet clear readPending. This happens during channel initialization.
160             // NB: We only set the boolean field instead of calling clearReadPending0(), because the SelectionKey is
161             // not set yet so it would produce an assertion failure.
162             readPending = false;
163         }
164     }

166     private void setReadPending0(boolean readPending) {
167         this.readPending = readPending;
168         if (!readPending) {
169             ((AbstractNioUnsafe) unsafe()).removeReadOp();
170         }
171     }

173     private void clearReadPending0() {
174         readPending = false;
175         ((AbstractNioUnsafe) unsafe()).removeReadOp();

179      * Special {@link Unsafe} sub-type which allows to access the underlying {@link SelectableChannel}
180      */
181     public interface NioUnsafe extends Unsafe {
182         /**
183          * Return underlying {@link SelectableChannel}
184          */
185         SelectableChannel ch();

187         /**
188          * Finish connect

200     protected abstract class AbstractNioUnsafe extends AbstractUnsafe implements NioUnsafe {
201 
202         protected final void removeReadOp() {
203             SelectionKey key = selectionKey();
204             // Check first if the key is still valid as it may be canceled as part of the deregistration
205             // from the EventLoop
206             // See https://github.com/netty/netty/issues/2104
207             if (key == null || !key.isValid()) {
208                 return;
209             }
210             int interestOps = key.interestOps();
211             if ((interestOps & readInterestOp) != 0) {
212                 // only remove readInterestOp if needed
213                 key.interestOps(interestOps & ~readInterestOp);
214             }
215         }

217         @Override
218         public final SelectableChannel ch() {
219             return javaChannel();
220         }

222         @Override
223         public final void connect(
224                 final SocketAddress remoteAddress, final SocketAddress localAddress, Promise<Void> promise) {
225             if (!promise.setUncancellable() || !ensureOpen(promise)) {
226                 return;
227             }
228 
229             try {
230                 if (connectPromise != null) {
231                     // Already a connect in process.
232                     throw new ConnectionPendingException();
233                 }
234 
235                 boolean wasActive = isActive();
236                 if (doConnect(remoteAddress, localAddress)) {
237                     fulfillConnectPromise(promise, wasActive);
238                 } else {
239                     connectPromise = promise;
240                     requestedRemoteAddress = remoteAddress;
241 
242                     // Schedule connect timeout.
243                     int connectTimeoutMillis = config().getConnectTimeoutMillis();
244                     if (connectTimeoutMillis > 0) {
245                         connectTimeoutFuture = executor().schedule(() -> {
246                             Promise<Void> connectPromise = AbstractNioChannel.this.connectPromise;
247                             if (connectPromise != null && !connectPromise.isDone()
248                                     && connectPromise.tryFailure(new ConnectTimeoutException(
249                                     "connection timed out: " + remoteAddress))) {
250                                 close(newPromise());
251                             }
252                         }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
253                     }
254 
255                     promise.asFuture().addListener(future -> {
256                         if (future.isCancelled()) {
257                             if (connectTimeoutFuture != null) {
258                                 connectTimeoutFuture.cancel();
259                             }
260                             connectPromise = null;
261                             close(newPromise());
262                         }
263                     });
264                 }
265             } catch (Throwable t) {
266                 promise.tryFailure(annotateConnectException(t, remoteAddress));
267                 closeIfClosed();
268             }
269         }
270 
271         private void fulfillConnectPromise(Promise<Void> promise, boolean wasActive) {
272             if (promise == null) {
273                 // Closed via cancellation and the promise has been notified already.
274                 return;
275             }
276 
277             // Get the state as trySuccess() may trigger an ChannelFutureListeners that will close the Channel.
278             // We still need to ensure we call fireChannelActive() in this case.
279             boolean active = isActive();
280 
281             // trySuccess() will return false if a user cancelled the connection attempt.
282             boolean promiseSet = promise.trySuccess(null);
283 
284             // Regardless if the connection attempt was cancelled, channelActive() event should be triggered,
285             // because what happened is what happened.
286             if (!wasActive && active) {
287                 pipeline().fireChannelActive();
288                 readIfIsAutoRead();
289             }
290 
291             // If a user cancelled the connection attempt, close the channel, which is followed by channelInactive().
292             if (!promiseSet) {
293                 close(newPromise());
294             }
295         }
296 
297         private void fulfillConnectPromise(Promise<Void> promise, Throwable cause) {
298             if (promise == null) {
299                 // Closed via cancellation and the promise has been notified already.
300                 return;
301             }
302 
303             // Use tryFailure() instead of setFailure() to avoid the race against cancel().
304             promise.tryFailure(cause);
305             closeIfClosed();
306         }
307 
308         @Override
309         public final void finishConnect() {
310             // Note this method is invoked by the event loop only if the connection attempt was
311             // neither cancelled nor timed out.
312 
313             assert executor().inEventLoop();
314 
315             try {
316                 boolean wasActive = isActive();
317                 doFinishConnect();
318                 fulfillConnectPromise(connectPromise, wasActive);
319             } catch (Throwable t) {
320                 fulfillConnectPromise(connectPromise, annotateConnectException(t, requestedRemoteAddress));
321             } finally {
322                 // Check for null as the connectTimeoutFuture is only created if a connectTimeoutMillis > 0 is used
323                 // See https://github.com/netty/netty/issues/1770
324                 if (connectTimeoutFuture != null) {
325                     connectTimeoutFuture.cancel();
326                 }
327                 connectPromise = null;
328             }
329         }
330 
331         @Override
332         protected final void flush0() {
333             // Flush immediately only when there's no pending flush.
334             // If there's a pending flush operation, event loop will call forceFlush() later,
335             // and thus there's no need to call it now.
336             if (!isFlushPending()) {
337                 super.flush0();
338             }
339         }
340 
341         @Override
342         public final void forceFlush() {
343             // directly call super.flush0() to force a flush now
344             super.flush0();
345         }
346 
347         private boolean isFlushPending() {
348             SelectionKey selectionKey = selectionKey();
349             return selectionKey != null && selectionKey.isValid()
350                     && (selectionKey.interestOps() & SelectionKey.OP_WRITE) != 0;
351         }
352     }
353 
354     @Override
355     protected void doBeginRead() throws Exception {
356         // Channel.read() or ChannelHandlerContext.read() was called
357         final SelectionKey selectionKey = this.selectionKey;
358         if (!selectionKey.isValid()) {
359             return;
360         }
361 
362         readPending = true;
363 
364         final int interestOps = selectionKey.interestOps();
365         if ((interestOps & readInterestOp) == 0) {
366             selectionKey.interestOps(interestOps | readInterestOp);
367         }
368     }

380     /**
381      * Returns an off-heap copy of the specified {@link ByteBuf}, and releases the original one.
382      * Note that this method does not create an off-heap copy if the allocation / deallocation cost is too high,
383      * but just returns the original {@link ByteBuf}..
384      */
385     protected final ByteBuf newDirectBuffer(ByteBuf buf) {
386         final int readableBytes = buf.readableBytes();
387         if (readableBytes == 0) {
388             ReferenceCountUtil.safeRelease(buf);
389             return Unpooled.EMPTY_BUFFER;
390         }
391 
392         final ByteBufAllocator alloc = alloc();
393         if (alloc.isDirectBufferPooled()) {
394             ByteBuf directBuf = alloc.directBuffer(readableBytes);
395             directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
396             ReferenceCountUtil.safeRelease(buf);
397             return directBuf;
398         }
399 
400         final ByteBuf directBuf = ByteBufUtil.threadLocalDirectBuffer();
401         if (directBuf != null) {
402             directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
403             ReferenceCountUtil.safeRelease(buf);
404             return directBuf;
405         }
406 
407         // Allocating and deallocating an unpooled direct buffer is very expensive; give up.
408         return buf;
409     }
410 
411     /**
412      * Returns an off-heap copy of the specified {@link ByteBuf}, and releases the specified holder.
413      * The caller must ensure that the holder releases the original {@link ByteBuf} when the holder is released by
414      * this method.  Note that this method does not create an off-heap copy if the allocation / deallocation cost is
415      * too high, but just returns the original {@link ByteBuf}..
416      */
417     protected final ByteBuf newDirectBuffer(ReferenceCounted holder, ByteBuf buf) {
418         final int readableBytes = buf.readableBytes();
419         if (readableBytes == 0) {
420             ReferenceCountUtil.safeRelease(holder);
421             return Unpooled.EMPTY_BUFFER;
422         }
423 
424         final ByteBufAllocator alloc = alloc();
425         if (alloc.isDirectBufferPooled()) {
426             ByteBuf directBuf = alloc.directBuffer(readableBytes);
427             directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
428             ReferenceCountUtil.safeRelease(holder);
429             return directBuf;
430         }
431 
432         final ByteBuf directBuf = ByteBufUtil.threadLocalDirectBuffer();
433         if (directBuf != null) {
434             directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
435             ReferenceCountUtil.safeRelease(holder);
436             return directBuf;
437         }
438 
439         // Allocating and deallocating an unpooled direct buffer is very expensive; give up.
440         if (holder != buf) {
441             // Ensure to call holder.release() to give the holder a chance to release other resources than its content.
442             buf.retain();
443             ReferenceCountUtil.safeRelease(holder);
444         }
445 
446         return buf;
447     }

449     /**
450      * Allocates a new off-heap copy of the given buffer, unless the cost of doing so is too high.
451      * The given buffer is closed if a copy is created, or returned directly.
452      *
453      * @param buf The buffer to copy.
454      * @return Probably an off-heap copy of the given buffer.
455      */
456     protected final Buffer newDirectBuffer(Buffer buf) {
457         if (buf.readableBytes() == 0) {
458             // Don't bother allocating a zero-sized buffer. They will not cause IO anyway.
459             return buf;
460         }
461 
462         BufferAllocator bufferAllocator = bufferAllocator();
463         if (bufferAllocator.getAllocationType() != StandardAllocationTypes.OFF_HEAP) {
464             bufferAllocator = DefaultBufferAllocators.offHeapAllocator();
465         }
466         if (bufferAllocator.isPooling()) {
467             try (buf) {
468                 return bufferAllocator.allocate(buf.readableBytes()).writeBytes(buf);
469             }
470         }
471         return buf; // Un-pooled off-heap allocation is too expensive. Give up.
472     }
473 
474     /**
475      * Allocates a new off-heap copy of the given buffer, unless the cost of doing so is too high.
476      * The given holder is closed regardless.
477      *
478      * @param buf The buffer to copy.
479      * @return Probably an off-heap copy of the given buffer.
480      */
481     protected final Buffer newDirectBuffer(Resource<?> holder, Buffer buf) {
482         try (holder) {
483             BufferAllocator bufferAllocator = bufferAllocator();
484             if (bufferAllocator.getAllocationType() != StandardAllocationTypes.OFF_HEAP) {
485                 bufferAllocator = DefaultBufferAllocators.offHeapAllocator();
486             }
487             if (bufferAllocator.isPooling()) {
488                 return bufferAllocator.allocate(buf.readableBytes()).writeBytes(buf);
489             }
490             // Un-pooled off-heap allocation is too expensive. Give up.
491             // Use split() to grab the readable part of the buffer; the remainder will be closed along with its holder.
492             return buf.split();
493         }
494     }

496     @Override
497     protected void doClose() throws Exception {
498         Promise<Void> promise = connectPromise;
499         if (promise != null) {
500             // Use tryFailure() instead of setFailure() to avoid the race against cancel().
501             promise.tryFailure(new ClosedChannelException());
502             connectPromise = null;
503         }
504 
505         Future<?> future = connectTimeoutFuture;
506         if (future != null) {
507             future.cancel();
508             connectTimeoutFuture = null;
509         }
510     }
