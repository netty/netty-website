34  @SuppressWarnings("unchecked")
35  @UnstableApi
36  public abstract class AbstractCoalescingBufferQueue {
37      private static final InternalLogger logger = InternalLoggerFactory.getInstance(AbstractCoalescingBufferQueue.class);
38      private final ArrayDeque<Object> bufAndListenerPairs;
39      private final PendingBytesTracker tracker;
40      private int readableBytes;
41  
42      /**
43       * Create a new instance.
44       *
45       * @param channel the {@link Channel} which will have the {@link Channel#isWritable()} reflect the amount of queued
46       *                buffers or {@code null} if there is no writability state updated.
47       * @param initSize the initial size of the underlying queue.
48       */
49      protected AbstractCoalescingBufferQueue(Channel channel, int initSize) {
50          bufAndListenerPairs = new ArrayDeque<>(initSize);
51          tracker = channel == null ? null : PendingBytesTracker.newTracker(channel);
52      }
53  
54      /**
55       * Add a buffer to the front of the queue and associate a promise with it that should be completed when
56       * all the buffer's bytes have been consumed from the queue and written.
57       * @param buf to add to the head of the queue
58       * @param promise to complete when all the bytes have been consumed and written, can be void.
59       */
60      public final void addFirst(ByteBuf buf, Promise<Void> promise) {
61          addFirst(buf, f -> f.cascadeTo(promise));
62      }
63  
64      private void addFirst(ByteBuf buf, FutureListener<Void> listener) {
65          if (listener != null) {
66              bufAndListenerPairs.addFirst(listener);
67          }
68          bufAndListenerPairs.addFirst(buf);
69          incrementReadableBytes(buf.readableBytes());
70      }
71  
72      /**
73       * Add a buffer to the end of the queue.
74       */
75      public final void add(ByteBuf buf) {
76          add(buf, (FutureListener<Void>) null);
77      }
78  
79      /**
80       * Add a buffer to the end of the queue and associate a promise with it that should be completed when
81       * all the buffer's bytes have been consumed from the queue and written.
82       * @param buf to add to the tail of the queue
83       * @param promise to complete when all the bytes have been consumed and written, can be void.
84       */
85      public final void add(ByteBuf buf, Promise<Void> promise) {
86          // buffers are added before promises so that we naturally 'consume' the entire buffer during removal
87          // before we complete it's promise.
88          add(buf, f -> f.cascadeTo(promise));
89      }
90  
91      /**
92       * Add a buffer to the end of the queue and associate a listener with it that should be completed when
93       * all the buffers  bytes have been consumed from the queue and written.
94       * @param buf to add to the tail of the queue
95       * @param listener to notify when all the bytes have been consumed and written, can be {@code null}.
96       */
97      public final void add(ByteBuf buf, FutureListener<Void> listener) {
98          // buffers are added before promises so that we naturally 'consume' the entire buffer during removal
99          // before we complete it's promise.
100         bufAndListenerPairs.add(buf);
101         if (listener != null) {
102             bufAndListenerPairs.add(listener);
103         }
104         incrementReadableBytes(buf.readableBytes());
105     }
106 
107     /**
108      * Remove the first {@link ByteBuf} from the queue.
109      * @param aggregatePromise used to aggregate the promises and listeners for the returned buffer.
110      * @return the first {@link ByteBuf} from the queue.
111      */
112     public final ByteBuf removeFirst(Promise<Void> aggregatePromise) {
113         Object entry = bufAndListenerPairs.poll();
114         if (entry == null) {
115             return null;
116         }
117         assert entry instanceof ByteBufConvertible;
118         ByteBuf result = ((ByteBufConvertible) entry).asByteBuf();
119 
120         decrementReadableBytes(result.readableBytes());
121 
122         entry = bufAndListenerPairs.peek();
123         if (entry instanceof FutureListener) {
124             aggregatePromise.asFuture().addListener((FutureListener<Void>) entry);
125             bufAndListenerPairs.poll();
126         }
127         return result;
128     }
129 
130     /**
131      * Remove a {@link ByteBuf} from the queue with the specified number of bytes. Any added buffer who's bytes are
132      * fully consumed during removal will have it's promise completed when the passed aggregate {@link Promise}
133      * completes.
134      *
135      * @param alloc The allocator used if a new {@link ByteBuf} is generated during the aggregation process.
136      * @param bytes the maximum number of readable bytes in the returned {@link ByteBuf}, if {@code bytes} is greater
137      *              than {@link #readableBytes} then a buffer of length {@link #readableBytes} is returned.
138      * @param aggregatePromise used to aggregate the promises and listeners for the constituent buffers.
139      * @return a {@link ByteBuf} composed of the enqueued buffers.
140      */
141     public final ByteBuf remove(ByteBufAllocator alloc, int bytes, Promise<Void> aggregatePromise) {
142         checkPositiveOrZero(bytes, "bytes");
143         requireNonNull(aggregatePromise, "aggregatePromise");
144 
145         // Use isEmpty rather than readableBytes==0 as we may have a promise associated with an empty buffer.
146         if (bufAndListenerPairs.isEmpty()) {
147             assert readableBytes == 0;
148             return removeEmptyValue();
149         }
150         bytes = Math.min(bytes, readableBytes);
151 
152         ByteBuf toReturn = null;
153         ByteBuf entryBuffer = null;
154         int originalBytes = bytes;
155         try {
156             for (;;) {
157                 Object entry = bufAndListenerPairs.poll();
158                 if (entry == null) {
159                     break;
160                 }
161                 if (entry instanceof FutureListener) {
162                     aggregatePromise.asFuture().addListener((FutureListener<Void>) entry);
163                     continue;
164                 }
165                 entryBuffer = (ByteBuf) entry;
166                 if (entryBuffer.readableBytes() > bytes) {
167                     // Add the buffer back to the queue as we can't consume all of it.
168                     bufAndListenerPairs.addFirst(entryBuffer);
169                     if (bytes > 0) {
170                         // Take a slice of what we can consume and retain it.
171                         entryBuffer = entryBuffer.readRetainedSlice(bytes);
172                         toReturn = toReturn == null ? composeFirst(alloc, entryBuffer)
173                                                     : compose(alloc, toReturn, entryBuffer);
174                         bytes = 0;
175                     }
176                     break;
177                 } else {
178                     bytes -= entryBuffer.readableBytes();
179                     toReturn = toReturn == null ? composeFirst(alloc, entryBuffer)
180                                                 : compose(alloc, toReturn, entryBuffer);
181                 }
182                 entryBuffer = null;
183             }
184         } catch (Throwable cause) {
185             safeRelease(entryBuffer);
186             safeRelease(toReturn);
187             aggregatePromise.setFailure(cause);
188             throw cause;
189         }
190         decrementReadableBytes(originalBytes - bytes);
191         return toReturn;
192     }

194     /**
195      * The number of readable bytes.
196      */
197     public final int readableBytes() {
198         return readableBytes;
199     }
200 
201     /**
202      * Are there pending buffers in the queue.
203      */
204     public final boolean isEmpty() {
205         return bufAndListenerPairs.isEmpty();
206     }
207 
208     /**
209      *  Release all buffers in the queue and complete all listeners and promises.
210      */
211     public final void releaseAndFailAll(ChannelOutboundInvoker invoker, Throwable cause) {
212         releaseAndCompleteAll(invoker.newFailedFuture(cause));
213     }
214 
215     /**
216      * Copy all pending entries in this queue into the destination queue.
217      * @param dest to copy pending buffers to.
218      */
219     public final void copyTo(AbstractCoalescingBufferQueue dest) {
220         dest.bufAndListenerPairs.addAll(bufAndListenerPairs);
221         dest.incrementReadableBytes(readableBytes);
222     }
223 
224     /**
225      * Writes all remaining elements in this queue.
226      * @param ctx The context to write all elements to.
227      */
228     public final void writeAndRemoveAll(ChannelHandlerContext ctx) {
229         Throwable pending = null;
230         ByteBuf previousBuf = null;
231         for (;;) {
232             Object entry = bufAndListenerPairs.poll();
233             try {
234                 if (entry == null) {
235                     if (previousBuf != null) {
236                         decrementReadableBytes(previousBuf.readableBytes());
237                         // If the write fails we want to at least propagate the exception through the ChannelPipeline
238                         // as otherwise the user will not be made aware of the failure at all.
239                         ctx.write(previousBuf)
240                            .addListener(ctx.channel(), ChannelFutureListeners.FIRE_EXCEPTION_ON_FAILURE);
241                     }
242                     break;
243                 }
244 
245                 if (entry instanceof ByteBufConvertible) {
246                     if (previousBuf != null) {
247                         decrementReadableBytes(previousBuf.readableBytes());
248                         // If the write fails we want to at least propagate the exception through the ChannelPipeline
249                         // as otherwise the user will not be made aware of the failure at all.
250                         ctx.write(previousBuf)
251                            .addListener(ctx.channel(), ChannelFutureListeners.FIRE_EXCEPTION_ON_FAILURE);
252                     }
253                     previousBuf = ((ByteBufConvertible) entry).asByteBuf();
254                 } else if (entry instanceof Promise) {
255                     decrementReadableBytes(previousBuf.readableBytes());
256                     ctx.write(previousBuf).cascadeTo((Promise<? super Void>) entry);
257                     previousBuf = null;
258                 } else {
259                     decrementReadableBytes(previousBuf.readableBytes());
260                     ctx.write(previousBuf).addListener((FutureListener<Void>) entry);
261                     previousBuf = null;
262                 }
263             } catch (Throwable t) {
264                 if (pending == null) {
265                     pending = t;
266                 } else {
267                     logger.info("Throwable being suppressed because Throwable {} is already pending", pending, t);
268                 }
269             }
270         }
271         if (pending != null) {
272             throw new IllegalStateException(pending);
273         }
274     }

276     @Override
277     public String toString() {
278         return "bytes: " + readableBytes + " buffers: " + (size() >> 1);
279     }

281     /**
282      * Calculate the result of {@code current + next}.
283      */
284     protected abstract ByteBuf compose(ByteBufAllocator alloc, ByteBuf cumulation, ByteBuf next);
285 
286     /**
287      * Compose {@code cumulation} and {@code next} into a new {@link CompositeByteBuf}.
288      */
289     protected final ByteBuf composeIntoComposite(ByteBufAllocator alloc, ByteBuf cumulation, ByteBuf next) {
290         // Create a composite buffer to accumulate this pair and potentially all the buffers
291         // in the queue. Using +2 as we have already dequeued current and next.
292         CompositeByteBuf composite = alloc.compositeBuffer(size() + 2);
293         try {
294             composite.addComponent(true, cumulation);
295             composite.addComponent(true, next);
296         } catch (Throwable cause) {
297             composite.release();
298             safeRelease(next);
299             throw cause;
300         }
301         return composite;
302     }
303 
304     /**
305      * Compose {@code cumulation} and {@code next} into a new {@link ByteBufAllocator#ioBuffer()}.
306      * @param alloc The allocator to use to allocate the new buffer.
307      * @param cumulation The current cumulation.
308      * @param next The next buffer.
309      * @return The result of {@code cumulation + next}.
310      */
311     protected final ByteBuf copyAndCompose(ByteBufAllocator alloc, ByteBuf cumulation, ByteBuf next) {
312         ByteBuf newCumulation = alloc.ioBuffer(cumulation.readableBytes() + next.readableBytes());
313         try {
314             newCumulation.writeBytes(cumulation).writeBytes(next);
315         } catch (Throwable cause) {
316             newCumulation.release();
317             safeRelease(next);
318             throw cause;
319         }
320         cumulation.release();
321         next.release();
322         return newCumulation;
323     }

347     private void releaseAndCompleteAll(Future<Void> future) {
348         Throwable pending = null;
349         for (;;) {
350             Object entry = bufAndListenerPairs.poll();
351             if (entry == null) {
352                 break;
353             }
354             try {
355                 if (entry instanceof ByteBufConvertible) {
356                     ByteBuf buffer = ((ByteBufConvertible) entry).asByteBuf();
357                     decrementReadableBytes(buffer.readableBytes());
358                     safeRelease(buffer);
359                 } else {
360                     ((FutureListener<Void>) entry).operationComplete(future);
361                 }
362             } catch (Throwable t) {
363                 if (pending == null) {
364                     pending = t;
365                 } else {
366                     logger.info("Throwable being suppressed because Throwable {} is already pending", pending, t);
367                 }
368             }
369         }
370         if (pending != null) {
371             throw new IllegalStateException(pending);
372         }
373     }
374 
375     private void incrementReadableBytes(int increment) {
376         int nextReadableBytes = readableBytes + increment;
377         if (nextReadableBytes < readableBytes) {
378             throw new IllegalStateException("buffer queue length overflow: " + readableBytes + " + " + increment);
379         }
380         readableBytes = nextReadableBytes;
381         if (tracker != null) {
382             tracker.incrementPendingOutboundBytes(increment);
383         }
384     }
385 
386     private void decrementReadableBytes(int decrement) {
387         readableBytes -= decrement;
388         assert readableBytes >= 0;
389         if (tracker != null) {
390             tracker.decrementPendingOutboundBytes(decrement);
391         }
392     }
