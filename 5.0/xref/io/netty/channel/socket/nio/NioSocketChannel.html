59      private static SocketChannel newSocket(SelectorProvider provider) {
60          try {
61              //  Use the {@link SelectorProvider} to open {@link SocketChannel} and so remove condition in
62              //  {@link SelectorProvider#provider()} which is called by each SocketChannel.open() otherwise.
63              //
64              //  See <a href="https://github.com/netty/netty/issues/2308">#2308</a>.
65              return provider.openSocketChannel();
66          } catch (IOException e) {
67              throw new ChannelException("Failed to open a socket.", e);
68          }
69      }

201     @Override
202     public Future<Void> shutdown(final Promise<Void> promise) {
203         Future<Void> shutdownOutputFuture = shutdownOutput();
204         if (shutdownOutputFuture.isDone()) {
205             shutdownOutputDone(promise, shutdownOutputFuture);
206         } else {
207             shutdownOutputFuture.addListener(promise, this::shutdownOutputDone);
208         }
209         return promise.asFuture();
210     }
211 
212     private void shutdownOutputDone(Promise<Void> promise, Future<?> shutdownOutputFuture) {
213         Future<Void> shutdownInputFuture = shutdownInput();
214         if (shutdownInputFuture.isDone()) {
215             shutdownDone(shutdownOutputFuture, shutdownInputFuture, promise);
216         } else {
217             shutdownInputFuture.addListener(shutdownInputFuture1 ->
218                     shutdownDone(shutdownOutputFuture, shutdownInputFuture1, promise));
219         }
220     }
221 
222     private static void shutdownDone(Future<?> shutdownOutputFuture,
223                                      Future<?> shutdownInputFuture,
224                                      Promise<Void> promise) {
225         Throwable shutdownOutputCause = shutdownOutputFuture.cause();
226         Throwable shutdownInputCause = shutdownInputFuture.cause();
227         if (shutdownOutputCause != null) {
228             if (shutdownInputCause != null) {
229                 logger.debug("Exception suppressed because a previous exception occurred.",
230                         shutdownInputCause);
231             }
232             promise.setFailure(shutdownOutputCause);
233         } else if (shutdownInputCause != null) {
234             promise.setFailure(shutdownInputCause);
235         } else {
236             promise.setSuccess(null);
237         }
238     }
239     private void shutdownInput0(Promise<Void> promise) {
240         try {
241             shutdownInput0();
242             promise.setSuccess(null);
243         } catch (Throwable t) {
244             promise.setFailure(t);
245         }
246     }

271     @Override
272     protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
273         if (localAddress != null) {
274             doBind0(localAddress);
275         }
276 
277         boolean success = false;
278         try {
279             boolean connected = SocketUtils.connect(javaChannel(), remoteAddress);
280             if (!connected) {
281                 selectionKey().interestOps(SelectionKey.OP_CONNECT);
282             }
283             success = true;
284             return connected;
285         } finally {
286             if (!success) {
287                 doClose();
288             }
289         }
290     }

342     private void adjustMaxBytesPerGatheringWrite(int attempted, int written, int oldMaxBytesPerGatheringWrite) {
343         // By default we track the SO_SNDBUF when ever it is explicitly set. However some OSes may dynamically change
344         // SO_SNDBUF (and other characteristics that determine how much data can be written at once) so we should try
345         // make a best effort to adjust as OS behavior changes.
346         if (attempted == written) {
347             if (attempted << 1 > oldMaxBytesPerGatheringWrite) {
348                 ((NioSocketChannelConfig) config).setMaxBytesPerGatheringWrite(attempted << 1);
349             }
350         } else if (attempted > MAX_BYTES_PER_GATHERING_WRITE_ATTEMPTED_LOW_THRESHOLD && written < attempted >>> 1) {
351             ((NioSocketChannelConfig) config).setMaxBytesPerGatheringWrite(attempted >>> 1);
352         }
353     }

355     @Override
356     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
357         SocketChannel ch = javaChannel();
358         int writeSpinCount = config().getWriteSpinCount();
359         do {
360             if (in.isEmpty()) {
361                 // All written so clear OP_WRITE
362                 clearOpWrite();
363                 // Directly return here so incompleteWrite(...) is not called.
364                 return;
365             }
366 
367             // Ensure the pending writes are made of ByteBufs only.
368             int maxBytesPerGatheringWrite = ((NioSocketChannelConfig) config).getMaxBytesPerGatheringWrite();
369             ByteBuffer[] nioBuffers = in.nioBuffers(1024, maxBytesPerGatheringWrite);
370             int nioBufferCnt = in.nioBufferCount();
371 
372             // Always use nioBuffers() to workaround data-corruption.
373             // See https://github.com/netty/netty/issues/2761
374             switch (nioBufferCnt) {
375                 case 0:
376                     // We have something else beside ByteBuffers to write so fallback to normal writes.
377                     writeSpinCount -= doWrite0(in);
378                     break;
379                 case 1: {
380                     // Only one ByteBuf so use non-gathering write
381                     // Zero length buffers are not added to nioBuffers by ChannelOutboundBuffer, so there is no need
382                     // to check if the total size of all the buffers is non-zero.
383                     ByteBuffer buffer = nioBuffers[0];
384                     int attemptedBytes = buffer.remaining();
385                     final int localWrittenBytes = ch.write(buffer);
386                     if (localWrittenBytes <= 0) {
387                         incompleteWrite(true);
388                         return;
389                     }
390                     adjustMaxBytesPerGatheringWrite(attemptedBytes, localWrittenBytes, maxBytesPerGatheringWrite);
391                     in.removeBytes(localWrittenBytes);
392                     --writeSpinCount;
393                     break;
394                 }
395                 default: {
396                     // Zero length buffers are not added to nioBuffers by ChannelOutboundBuffer, so there is no need
397                     // to check if the total size of all the buffers is non-zero.
398                     // We limit the max amount to int above so cast is safe
399                     long attemptedBytes = in.nioBufferSize();
400                     final long localWrittenBytes = ch.write(nioBuffers, 0, nioBufferCnt);
401                     if (localWrittenBytes <= 0) {
402                         incompleteWrite(true);
403                         return;
404                     }
405                     // Casting to int is safe because we limit the total amount of data in the nioBuffers to int above.
406                     adjustMaxBytesPerGatheringWrite((int) attemptedBytes, (int) localWrittenBytes,
407                             maxBytesPerGatheringWrite);
408                     in.removeBytes(localWrittenBytes);
409                     --writeSpinCount;
410                     break;
411                 }
412             }
413         } while (writeSpinCount > 0);
414 
415         incompleteWrite(writeSpinCount < 0);
416     }

423     private final class NioSocketChannelUnsafe extends NioByteUnsafe {
424         @Override
425         protected Executor prepareToClose() {
426             try {
427                 if (javaChannel().isOpen() && config().getSoLinger() > 0) {
428                     // We need to cancel this key of the channel so we may not end up in a eventloop spin
429                     // because we try to read or write until the actual close happens which may be later due
430                     // SO_LINGER handling.
431                     // See https://github.com/netty/netty/issues/4449
432                     doDeregister();
433                     return GlobalEventExecutor.INSTANCE;
434                 }
435             } catch (Throwable ignore) {
436                 // Ignore the error as the underlying channel may be closed in the meantime and so
437                 // getSoLinger() may produce an exception. In this case we just return null.
438                 // See https://github.com/netty/netty/issues/4449
439             }
440             return null;
441         }
442     }

444     private final class NioSocketChannelConfig extends DefaultSocketChannelConfig {
445         private volatile int maxBytesPerGatheringWrite = Integer.MAX_VALUE;
446         private NioSocketChannelConfig(NioSocketChannel channel, Socket javaSocket) {
447             super(channel, javaSocket);
448             calculateMaxBytesPerGatheringWrite();
449         }

458             super.setSendBufferSize(sendBufferSize);

467             }

475             }
476             return super.getOption(option);
477         }

484         void setMaxBytesPerGatheringWrite(int maxBytesPerGatheringWrite) {
485             this.maxBytesPerGatheringWrite = maxBytesPerGatheringWrite;
486         }

488         int getMaxBytesPerGatheringWrite() {
489             return maxBytesPerGatheringWrite;
490         }
491 
492         private void calculateMaxBytesPerGatheringWrite() {
493             // Multiply by 2 to give some extra space in case the OS can process write data faster than we can provide.
494             int newSendBufferSize = getSendBufferSize() << 1;
495             if (newSendBufferSize > 0) {
496                 setMaxBytesPerGatheringWrite(newSendBufferSize);
497             }
498         }

501             return ((NioSocketChannel) channel).javaChannel();
502         }
