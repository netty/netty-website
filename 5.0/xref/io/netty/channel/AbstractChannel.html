246     /**
247      * Returns the {@link String} representation of this channel.  The returned
248      * string contains the {@linkplain #hashCode() ID}, {@linkplain #localAddress() local address},
249      * and {@linkplain #remoteAddress() remote address} of this channel for
250      * easier identification.
251      */
252     @Override
253     public String toString() {
254         boolean active = isActive();
255         if (strValActive == active && strVal != null) {
256             return strVal;
257         }
258 
259         SocketAddress remoteAddr = remoteAddress();
260         SocketAddress localAddr = localAddress();
261         if (remoteAddr != null) {
262             StringBuilder buf = new StringBuilder(96)
263                 .append("[id: 0x")
264                 .append(id.asShortText())
265                 .append(", L:")
266                 .append(localAddr)
267                 .append(active? " - " : " ! ")
268                 .append("R:")
269                 .append(remoteAddr)
270                 .append(']');
271             strVal = buf.toString();
272         } else if (localAddr != null) {
273             StringBuilder buf = new StringBuilder(64)
274                 .append("[id: 0x")
275                 .append(id.asShortText())
276                 .append(", L:")
277                 .append(localAddr)
278                 .append(']');
279             strVal = buf.toString();
280         } else {
281             StringBuilder buf = new StringBuilder(16)
282                 .append("[id: 0x")
283                 .append(id.asShortText())
284                 .append(']');
285             strVal = buf.toString();
286         }
287 
288         strValActive = active;
289         return strVal;
290     }

339         @Override
340         public final void register(final Promise<Void> promise) {
341             assertEventLoop();
342 
343             if (isRegistered()) {
344                 promise.setFailure(new IllegalStateException("registered to an event loop already"));
345                 return;
346             }
347 
348             try {
349                 // check if the channel is still open as it could be closed in the mean time when the register
350                 // call was outside of the eventLoop
351                 if (!promise.setUncancellable() || !ensureOpen(promise)) {
352                     return;
353                 }
354                 boolean firstRegistration = neverRegistered;
355                 doRegister();
356                 neverRegistered = false;
357                 registered = true;
358 
359                 safeSetSuccess(promise);
360                 pipeline.fireChannelRegistered();
361                 // Only fire a channelActive if the channel has never been registered. This prevents firing
362                 // multiple channel actives if the channel is deregistered and re-registered.
363                 if (isActive()) {
364                     if (firstRegistration) {
365                         pipeline.fireChannelActive();
366                     }
367                     readIfIsAutoRead();
368                 }
369             } catch (Throwable t) {
370                 // Close the channel directly to avoid FD leak.
371                 closeForcibly();
372                 closePromise.setClosed();
373                 safeSetFailure(promise, t);
374             }
375         }

377         @Override
378         public final void bind(final SocketAddress localAddress, final Promise<Void> promise) {
379             assertEventLoop();
380 
381             if (!promise.setUncancellable() || !ensureOpen(promise)) {
382                 return;
383             }
384 
385             // See: https://github.com/netty/netty/issues/576
386             if (Boolean.TRUE.equals(config().getOption(ChannelOption.SO_BROADCAST)) &&
387                 localAddress instanceof InetSocketAddress &&
388                 !((InetSocketAddress) localAddress).getAddress().isAnyLocalAddress() &&
389                 !PlatformDependent.isWindows() && !PlatformDependent.maybeSuperUser()) {
390                 // Warn a user about the fact that a non-root user can't receive a
391                 // broadcast packet on *nix if the socket is bound on non-wildcard address.
392                 logger.warn(
393                         "A non-root user can't receive a broadcast packet if the socket " +
394                         "is not bound to a wildcard address; binding to a non-wildcard " +
395                         "address (" + localAddress + ") anyway as requested.");
396             }
397 
398             boolean wasActive = isActive();
399             try {
400                 doBind(localAddress);
401             } catch (Throwable t) {
402                 safeSetFailure(promise, t);
403                 closeIfClosed();
404                 return;
405             }
406 
407             if (!wasActive && isActive()) {
408                 invokeLater(() -> {
409                     pipeline.fireChannelActive();
410                     readIfIsAutoRead();
411                 });
412             }
413 
414             safeSetSuccess(promise);
415         }

417         @Override
418         public final void disconnect(final Promise<Void> promise) {
419             assertEventLoop();
420 
421             if (!promise.setUncancellable()) {
422                 return;
423             }
424 
425             boolean wasActive = isActive();
426             try {
427                 doDisconnect();
428                 // Reset remoteAddress and localAddress
429                 remoteAddress = null;
430                 localAddress = null;
431             } catch (Throwable t) {
432                 safeSetFailure(promise, t);
433                 closeIfClosed();
434                 return;
435             }
436 
437             if (wasActive && !isActive()) {
438                 invokeLater(pipeline::fireChannelInactive);
439             }
440 
441             safeSetSuccess(promise);
442             closeIfClosed(); // doDisconnect() might have closed the channel
443         }

464         /**
465          * Shutdown the output portion of the corresponding {@link Channel}.
466          * For example this will clean up the {@link ChannelOutboundBuffer} and not allow any more writes.
467          * @param cause The cause which may provide rational for the shutdown.
468          */
469         private void shutdownOutput(final Promise<Void> promise, Throwable cause) {
470             if (!promise.setUncancellable()) {
471                 return;
472             }
473 
474             final ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
475             if (outboundBuffer == null) {
476                 promise.setFailure(new ClosedChannelException());
477                 return;
478             }
479             this.outboundBuffer = null; // Disallow adding any messages and flushes to outboundBuffer.
480 
481             final Throwable shutdownCause = cause == null ?
482                     new ChannelOutputShutdownException("Channel output shutdown") :
483                     new ChannelOutputShutdownException("Channel output shutdown", cause);
484             // When a side enables SO_LINGER and calls showdownOutput(...) to start TCP half-closure
485             // we can not call doDeregister here because we should ensure this side in fin_wait2 state
486             // can still receive and process the data which is send by another side in the close_wait state。
487             // See https://github.com/netty/netty/issues/11981
488             try {
489                 // The shutdown function does not block regardless of the SO_LINGER setting on the socket
490                 // so we don't need to use GlobalEventExecutor to execute the shutdown
491                 doShutdownOutput();
492                 promise.setSuccess(null);
493             } catch (Throwable err) {
494                 promise.setFailure(err);
495             } finally {
496                 closeOutboundBufferForShutdown(pipeline, outboundBuffer, shutdownCause);
497             }
498         }

500         private void closeOutboundBufferForShutdown(
501                 ChannelPipeline pipeline, ChannelOutboundBuffer buffer, Throwable cause) {
502             buffer.failFlushed(cause, false);
503             buffer.close(cause, true);
504             pipeline.fireUserEventTriggered(ChannelOutputShutdownEvent.INSTANCE);
505         }

507         private void close(final Promise<Void> promise, final Throwable cause,
508                            final ClosedChannelException closeCause, final boolean notify) {
509             if (!promise.setUncancellable()) {
510                 return;
511             }
512 
513             if (closeInitiated) {
514                 if (closePromise.isDone()) {
515                     // Closed already.
516                     safeSetSuccess(promise);
517                 } else {
518                     // This means close() was called before, so we just register a listener and return
519                     closePromise.addListener(promise, (p, future) -> p.setSuccess(null));
520                 }
521                 return;
522             }
523 
524             closeInitiated = true;
525 
526             final boolean wasActive = isActive();
527             final ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
528             this.outboundBuffer = null; // Disallow adding any messages and flushes to outboundBuffer.
529             Executor closeExecutor = prepareToClose();
530             if (closeExecutor != null) {
531                 closeExecutor.execute(() -> {
532                     try {
533                         // Execute the close.
534                         doClose0(promise);
535                     } finally {
536                         // Call invokeLater so closeAndDeregister is executed in the EventLoop again!
537                         invokeLater(() -> {
538                             if (outboundBuffer != null) {
539                                 // Fail all the queued messages
540                                 outboundBuffer.failFlushed(cause, notify);
541                                 outboundBuffer.close(closeCause);
542                             }
543                             fireChannelInactiveAndDeregister(wasActive);
544                         });
545                     }
546                 });
547             } else {
548                 try {
549                     // Close the channel and fail the queued messages in all cases.
550                     doClose0(promise);
551                 } finally {
552                     if (outboundBuffer != null) {
553                         // Fail all the queued messages.
554                         outboundBuffer.failFlushed(cause, notify);
555                         outboundBuffer.close(closeCause);
556                     }
557                 }
558                 if (inFlush0) {
559                     invokeLater(() -> fireChannelInactiveAndDeregister(wasActive));
560                 } else {
561                     fireChannelInactiveAndDeregister(wasActive);
562                 }
563             }
564         }
565 
566         private void doClose0(Promise<Void> promise) {
567             try {
568                 doClose();
569                 closePromise.setClosed();
570                 safeSetSuccess(promise);
571             } catch (Throwable t) {
572                 closePromise.setClosed();
573                 safeSetFailure(promise, t);
574             }
575         }

577         private void fireChannelInactiveAndDeregister(final boolean wasActive) {
578             deregister(newPromise(), wasActive && !isActive());

597         private void deregister(final Promise<Void> promise, final boolean fireChannelInactive) {
598             if (!promise.setUncancellable()) {
599                 return;
600             }
601 
602             if (!registered) {
603                 safeSetSuccess(promise);
604                 return;
605             }
606 
607             // As a user may call deregister() from within any method while doing processing in the ChannelPipeline,
608             // we need to ensure we do the actual deregister operation later. This is needed as for example,
609             // we may be in the ByteToMessageDecoder.callDecode(...) method and so still try to do processing in
610             // the old EventLoop while the user already registered the Channel to a new EventLoop. Without delay,
611             // the deregister operation this could lead to have a handler invoked by different EventLoop and so
612             // threads.
613             //
614             // See:
615             // https://github.com/netty/netty/issues/4435
616             invokeLater(() -> {
617                 try {
618                     doDeregister();
619                 } catch (Throwable t) {
620                     logger.warn("Unexpected exception occurred while deregistering a channel.", t);
621                 } finally {
622                     if (fireChannelInactive) {
623                         pipeline.fireChannelInactive();
624                     }
625                     // Some transports like local and AIO does not allow the deregistration of
626                     // an open channel. Their doDeregister() calls close(). Consequently,
627                     // close() calls deregister() again - no need to fire channelUnregistered, so check
628                     // if it was registered.
629                     if (registered) {
630                         registered = false;
631                         pipeline.fireChannelUnregistered();
632 
633                         if (!isOpen()) {
634                             // Remove all handlers from the ChannelPipeline. This is needed to ensure
635                             // handlerRemoved(...) is called and so resources are released.
636                             while (!pipeline.isEmpty()) {
637                                 try {
638                                     pipeline.removeLast();
639                                 } catch (NoSuchElementException ignore) {
640                                     // try again as there may be a race when someone outside the EventLoop removes
641                                     // handlers concurrently as well.
642                                 }
643                             }
644                         }
645                     }
646                     safeSetSuccess(promise);
647                 }
648             });
649         }

668         @Override
669         public final void write(Object msg, Promise<Void> promise) {
670             assertEventLoop();
671 
672             ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
673             if (outboundBuffer == null) {
674                 try {
675                     // release message now to prevent resource-leak
676                     ReferenceCountUtil.release(msg);
677                 } finally {
678                     // If the outboundBuffer is null we know the channel was closed and so
679                     // need to fail the future right away. If it is not null the handling of the rest
680                     // will be done in flush0()
681                     // See https://github.com/netty/netty/issues/2362
682                     safeSetFailure(promise,
683                             newClosedChannelException(initialCloseCause, "write(Object, Promise)"));
684                 }
685                 return;
686             }
687 
688             int size;
689             try {
690                 msg = filterOutboundMessage(msg);
691                 if (estimatorHandler == null) {
692                     estimatorHandler = config().getMessageSizeEstimator().newHandle();
693                 }
694                 size = estimatorHandler.size(msg);
695                 if (size < 0) {
696                     size = 0;
697                 }
698             } catch (Throwable t) {
699                 try {
700                     ReferenceCountUtil.release(msg);
701                 } finally {
702                     safeSetFailure(promise, t);
703                 }
704                 return;
705             }
706 
707             outboundBuffer.addMessage(msg, size, promise);
708         }

710         @Override
711         public final void flush() {
712             assertEventLoop();
713 
714             ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
715             if (outboundBuffer == null) {
716                 return;
717             }
718 
719             outboundBuffer.addFlush();
720             flush0();
721         }
722 
723         protected void flush0() {
724             if (inFlush0) {
725                 // Avoid re-entrance
726                 return;
727             }
728 
729             final ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
730             if (outboundBuffer == null || outboundBuffer.isEmpty()) {
731                 return;
732             }
733 
734             inFlush0 = true;
735 
736             // Mark all pending write requests as failure if the channel is inactive.
737             if (!isActive()) {
738                 try {
739                     // Check if we need to generate the exception at all.
740                     if (!outboundBuffer.isEmpty()) {
741                         if (isOpen()) {
742                             outboundBuffer.failFlushed(new NotYetConnectedException(), true);
743                         } else {
744                             // Do not trigger channelWritabilityChanged because the channel is closed already.
745                             outboundBuffer.failFlushed(newClosedChannelException(initialCloseCause, "flush0()"), false);
746                         }
747                     }
748                 } finally {
749                     inFlush0 = false;
750                 }
751                 return;
752             }
753 
754             try {
755                 doWrite(outboundBuffer);
756             } catch (Throwable t) {
757                 handleWriteError(t);
758             } finally {
759                 inFlush0 = false;
760             }
761         }
762 
763         protected final void handleWriteError(Throwable t) {
764             if (t instanceof IOException && config().isAutoClose()) {
765                 /**
766                  * Just call {@link #close(Promise, Throwable, boolean)} here which will take care of
767                  * failing all flushed messages and also ensure the actual close of the underlying transport
768                  * will happen before the promises are notified.
769                  *
770                  * This is needed as otherwise {@link #isActive()} , {@link #isOpen()} and {@link #isWritable()}
771                  * may still return {@code true} even if the channel should be closed as result of the exception.
772                  */
773                 initialCloseCause = t;
774                 close(newPromise(), t, newClosedChannelException(t, "flush0()"), false);
775             } else {
776                 try {
777                     shutdownOutput(newPromise(), t);
778                 } catch (Throwable t2) {
779                     initialCloseCause = t;
780                     close(newPromise(), t2, newClosedChannelException(t, "flush0()"), false);
781                 }
782             }
783         }

785         private ClosedChannelException newClosedChannelException(Throwable cause, String method) {
786             ClosedChannelException exception =
787                     StacklessClosedChannelException.newInstance(AbstractUnsafe.class, method);
788             if (cause != null) {
789                 exception.initCause(cause);
790             }
791             return exception;

828         private void invokeLater(Runnable task) {
829             try {
830                 // This method is used by outbound operation implementations to trigger an inbound event later.
831                 // They do not trigger an inbound event immediately because an outbound operation might have been
832                 // triggered by another inbound event handler method.  If fired immediately, the call stack
833                 // will look like this for example:
834                 //
835                 //   handlerA.inboundBufferUpdated() - (1) an inbound handler method closes a connection.
836                 //   -> handlerA.ctx.close()
837                 //      -> channel.unsafe.close()
838                 //         -> handlerA.channelInactive() - (2) another inbound handler method called while in (1) yet
839                 //
840                 // which means the execution of two inbound handler methods of the same handler overlap undesirably.
841                 executor().execute(task);
842             } catch (RejectedExecutionException e) {
843                 logger.warn("Can't invoke task later as EventLoop rejected it", e);
844             }
845         }

847         /**
848          * Appends the remote address to the message of the exceptions caused by connection attempt failure.
849          */
850         protected final Throwable annotateConnectException(Throwable cause, SocketAddress remoteAddress) {
851             if (cause instanceof ConnectException) {
852                 return new AnnotatedConnectException((ConnectException) cause, remoteAddress);
853             }
854             if (cause instanceof NoRouteToHostException) {
855                 return new AnnotatedNoRouteToHostException((NoRouteToHostException) cause, remoteAddress);
856             }
857             if (cause instanceof SocketException) {
858                 return new AnnotatedSocketException((SocketException) cause, remoteAddress);
859             }
860 
861             return cause;
862         }
