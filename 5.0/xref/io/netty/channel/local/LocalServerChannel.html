16  package io.netty5.channel.local;
17  
18  import io.netty5.channel.AbstractServerChannel;
19  import io.netty5.channel.ChannelConfig;
20  import io.netty5.channel.ChannelPipeline;
21  import io.netty5.channel.DefaultChannelConfig;
22  import io.netty5.channel.EventLoop;
23  import io.netty5.channel.EventLoopGroup;
24  import io.netty5.channel.PreferHeapByteBufAllocator;
25  import io.netty5.channel.RecvBufferAllocator;
26  import io.netty5.channel.ServerChannel;
27  import io.netty5.channel.ServerChannelRecvBufferAllocator;
28  import io.netty5.util.concurrent.Promise;
29  
30  import java.net.SocketAddress;
31  import java.util.ArrayDeque;
32  import java.util.Queue;
33  
34  /**
35   * A {@link ServerChannel} for the local transport which allows in VM communication.
36   */
37  public class LocalServerChannel extends AbstractServerChannel {
38  
39      private final ChannelConfig config =
40              new DefaultChannelConfig(this, new ServerChannelRecvBufferAllocator()) { };
41      private final Queue<Object> inboundBuffer = new ArrayDeque<>();
42      private volatile int state; // 0 - open, 1 - active, 2 - closed
43      private volatile LocalAddress localAddress;
44      private volatile boolean acceptInProgress;
45  
46      public LocalServerChannel(EventLoop eventLoop, EventLoopGroup childEventLoopGroup) {
47          super(eventLoop, childEventLoopGroup);
48          config().setAllocator(new PreferHeapByteBufAllocator(config.getAllocator()));
49      }
50  
51      @Override
52      public ChannelConfig config() {
53          return config;
54      }
55  
56      @Override
57      public LocalAddress localAddress() {
58          return (LocalAddress) super.localAddress();
59      }
60  
61      @Override
62      public LocalAddress remoteAddress() {
63          return (LocalAddress) super.remoteAddress();
64      }
65  
66      @Override
67      public boolean isOpen() {
68          return state < 2;
69      }
70  
71      @Override
72      public boolean isActive() {
73          return state == 1;
74      }
75  
76      @Override
77      protected SocketAddress localAddress0() {
78          return localAddress;
79      }
80  
81      @Override
82      protected void doBind(SocketAddress localAddress) throws Exception {
83          this.localAddress = LocalChannelRegistry.register(this, this.localAddress, localAddress);
84          state = 1;
85      }
86  
87      @Override
88      protected void doClose() throws Exception {
89          if (state <= 1) {
90              // Update all internal state before the closeFuture is notified.
91              if (localAddress != null) {
92                  LocalChannelRegistry.unregister(localAddress);
93                  localAddress = null;
94              }
95              state = 2;
96          }
97      }
98  
99      @Override
100     protected void doBeginRead() throws Exception {
101         if (acceptInProgress) {
102             return;
103         }
104 
105         Queue<Object> inboundBuffer = this.inboundBuffer;
106         if (inboundBuffer.isEmpty()) {
107             acceptInProgress = true;
108             return;
109         }
110 
111         readInbound();
112     }
113 
114     LocalChannel serve(final LocalChannel peer) {
115         final LocalChannel child = newLocalChannel(peer);
116         if (executor().inEventLoop()) {
117             serve0(child);
118         } else {
119             executor().execute(() -> serve0(child));
120         }
121         return child;
122     }
123 
124     private void readInbound() {
125         RecvBufferAllocator.Handle handle = unsafe().recvBufAllocHandle();
126         handle.reset(config());
127         ChannelPipeline pipeline = pipeline();
128         do {
129             Object m = inboundBuffer.poll();
130             if (m == null) {
131                 break;
132             }
133             pipeline.fireChannelRead(m);
134         } while (handle.continueReading());
135 
136         pipeline.fireChannelReadComplete();
137         readIfIsAutoRead();
138     }
139 
140     /**
141      * A factory method for {@link LocalChannel}s. Users may override it
142      * to create custom instances of {@link LocalChannel}s.
143      */
144     protected LocalChannel newLocalChannel(LocalChannel peer) {
145         return new LocalChannel(this, childEventLoopGroup().next(), peer);
146     }
147 
148     private void serve0(final LocalChannel child) {
149         inboundBuffer.add(child);
150         if (acceptInProgress) {
151             acceptInProgress = false;
152 
153             readInbound();
154         }
155     }
156 
157     @Override
158     protected AbstractUnsafe newUnsafe() {
159         return new DefaultServerUnsafe();
160     }
161 
162     private final class DefaultServerUnsafe extends AbstractUnsafe implements LocalChannelUnsafe {
163         @Override
164         public void connect(SocketAddress remoteAddress, SocketAddress localAddress, Promise<Void> promise) {
165             safeSetFailure(promise, new UnsupportedOperationException());
166         }
167 
168         @Override
169         public void register0() {
170         }
171 
172         @Override
173         public void deregister0() {
174         }
175 
176         @Override
177         public Promise<Void> newPromise() {
178             return LocalServerChannel.this.newPromise();
179         }
180     }
181 }
