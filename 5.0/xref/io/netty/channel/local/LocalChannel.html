61      private final Runnable readTask = () -> {
62          // Ensure the inboundBuffer is not empty as readInbound() will always call fireChannelReadComplete()
63          if (!inboundBuffer.isEmpty()) {
64              readInbound();
65          }
66      };

153     @Override
154     protected void doClose() throws Exception {
155         final LocalChannel peer = this.peer;
156         State oldState = state;
157         try {
158             if (oldState != State.CLOSED) {
159                 // Update all internal state before the closeFuture is notified.
160                 if (localAddress != null) {
161                     if (parent() == null) {
162                         LocalChannelRegistry.unregister(localAddress);
163                     }
164                     localAddress = null;
165                 }
166 
167                 // State change must happen before finishPeerRead to ensure writes are released either in doWrite or
168                 // channelRead.
169                 state = State.CLOSED;
170 
171                 // Preserve order of event and force a read operation now before the close operation is processed.
172                 if (writeInProgress && peer != null) {
173                     finishPeerRead(peer);
174                 }
175 
176                 Promise<Void> promise = connectPromise;
177                 if (promise != null) {
178                     // Use tryFailure() instead of setFailure() to avoid the race against cancel().
179                     promise.tryFailure(new ClosedChannelException());
180                     connectPromise = null;
181                 }
182             }
183 
184             if (peer != null) {
185                 this.peer = null;
186                 // Always call peer.eventLoop().execute() even if peer.eventLoop().inEventLoop() is true.
187                 // This ensures that if both channels are on the same event loop, the peer's channelInActive
188                 // event is triggered *after* this peer's channelInActive event
189                 EventLoop peerEventLoop = peer.executor();
190                 final boolean peerIsActive = peer.isActive();
191                 try {
192                     peerEventLoop.execute(() -> peer.tryClose(peerIsActive));
193                 } catch (Throwable cause) {
194                     logger.warn("Releasing Inbound Queues for channels {}-{} because exception occurred!",
195                             this, peer, cause);
196                     if (peerEventLoop.inEventLoop()) {
197                         peer.releaseInboundBuffers();
198                     } else {
199                         // inboundBuffers is a SPSC so we may leak if the event loop is shutdown prematurely or
200                         // rejects the close Runnable but give a best effort.
201                         peer.close();
202                     }
203                     throw cause;
204                 }
205             }
206         } finally {
207             // Release all buffers if the Channel was already registered in the past and if it was not closed before.
208             if (oldState != null && oldState != State.CLOSED) {
209                 // We need to release all the buffers that may be put into our inbound queue since we closed the Channel
210                 // to ensure we not leak any memory. This is fine as it basically gives the same guarantees as TCP which
211                 // means even if the promise was notified before its not really guaranteed that the "remote peer" will
212                 // see the buffer at all.
213                 releaseInboundBuffers();
214             }
215         }
216     }
217 
218     private void tryClose(boolean isActive) {
219         if (isActive) {
220             unsafe().close(newPromise());
221         } else {
222             releaseInboundBuffers();
223 
224             Promise<Void> promise = connectPromise;
225             if (promise != null) {
226                 // Use tryFailure() instead of setFailure() to avoid the race against cancel().
227                 promise.tryFailure(new ClosedChannelException());
228                 connectPromise = null;
229             }
230         }
231     }
232 
233     private void readInbound() {
234         RecvBufferAllocator.Handle handle = unsafe().recvBufAllocHandle();
235         handle.reset(config());
236         ChannelPipeline pipeline = pipeline();
237         do {
238             Object received = inboundBuffer.poll();
239             if (received == null) {
240                 break;
241             }
242             pipeline.fireChannelRead(received);
243         } while (handle.continueReading());
244 
245         pipeline.fireChannelReadComplete();
246         readIfIsAutoRead();
247     }

249     private static final class ReaderStackDepth {
250         private int stackDepth;
251 
252         boolean incrementIfPossible() {
253             if (stackDepth == MAX_READER_STACK_DEPTH) {
254                 return false;
255             }
256             stackDepth++;
257             return true;
258         }
259 
260         void decrement() {
261             stackDepth--;
262         }
263     }
264 
265     private static final FastThreadLocal<ReaderStackDepth> STACK_DEPTH = new FastThreadLocal<>() {
266         @Override
267         protected ReaderStackDepth initialValue() throws Exception {
268             return new ReaderStackDepth();
269         }
270     };
271 
272     @Override
273     protected void doBeginRead() throws Exception {
274         if (readInProgress) {
275             return;
276         }
277 
278         Queue<Object> inboundBuffer = this.inboundBuffer;
279         if (inboundBuffer.isEmpty()) {
280             readInProgress = true;
281             return;
282         }
283 
284         final ReaderStackDepth readerStackDepth = STACK_DEPTH.get();
285         if (readerStackDepth.incrementIfPossible()) {
286             try {
287                 readInbound();
288             } finally {
289                 readerStackDepth.decrement();
290             }
291         } else {
292             try {
293                 executor().execute(readTask);
294             } catch (Throwable cause) {
295                 logger.warn("Closing Local channels {}-{} because exception occurred!", this, peer, cause);
296                 close();
297                 peer.close();
298                 throw cause;
299             }
300         }
301     }
302 
303     @Override
304     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
305         switch (state) {
306         case OPEN:
307         case BOUND:
308             throw new NotYetConnectedException();
309         case CLOSED:
310             throw new ClosedChannelException();
311         case CONNECTED:
312             break;
313         }
314 
315         final LocalChannel peer = this.peer;
316 
317         writeInProgress = true;
318         try {
319             ClosedChannelException exception = null;
320             for (;;) {
321                 Object msg = in.current();
322                 if (msg == null) {
323                     break;
324                 }
325                 try {
326                     // It is possible the peer could have closed while we are writing, and in this case we should
327                     // simulate real socket behavior and ensure the write operation is failed.
328                     if (peer.state == State.CONNECTED) {
329                         peer.inboundBuffer.add(ReferenceCountUtil.retain(msg));
330                         in.remove();
331                     } else {
332                         if (exception == null) {
333                             exception = new ClosedChannelException();
334                         }
335                         in.remove(exception);
336                     }
337                 } catch (Throwable cause) {
338                     in.remove(cause);
339                 }
340             }
341         } finally {
342             // The following situation may cause trouble:
343             // 1. Write (with promise X)
344             // 2. promise X is completed when in.remove() is called, and a listener on this promise calls close()
345             // 3. Then the close event will be executed for the peer before the write events, when the write events
346             // actually happened before the close event.
347             writeInProgress = false;
348         }
349 
350         finishPeerRead(peer);
351     }
352 
353     private void finishPeerRead(final LocalChannel peer) {
354         // If the peer is also writing, then we must schedule the event on the event loop to preserve read order.
355         if (peer.executor() == executor() && !peer.writeInProgress) {
356             finishPeerRead0(peer);
357         } else {
358             runFinishPeerReadTask(peer);
359         }
360     }
361 
362     private void runFinishPeerReadTask(final LocalChannel peer) {
363         // If the peer is writing, we must wait until after reads are completed for that peer before we can read. So
364         // we keep track of the task, and coordinate later that our read can't happen until the peer is done.
365         final Runnable finishPeerReadTask = () -> finishPeerRead0(peer);
366         try {
367             if (peer.writeInProgress) {
368                 peer.finishReadFuture = peer.executor().submit(finishPeerReadTask);
369             } else {
370                 peer.executor().execute(finishPeerReadTask);
371             }
372         } catch (Throwable cause) {
373             logger.warn("Closing Local channels {}-{} because exception occurred!", this, peer, cause);
374             close();
375             peer.close();
376             throw cause;
377         }
378     }
379 
380     private void releaseInboundBuffers() {
381         assert executor() == null || executor().inEventLoop();
382         readInProgress = false;
383         Queue<Object> inboundBuffer = this.inboundBuffer;
384         Object msg;
385         while ((msg = inboundBuffer.poll()) != null) {
386             ReferenceCountUtil.release(msg);
387         }
388     }
389 
390     private void finishPeerRead0(LocalChannel peer) {
391         Future<?> peerFinishReadFuture = peer.finishReadFuture;
392         if (peerFinishReadFuture != null) {
393             if (!peerFinishReadFuture.isDone()) {
394                 runFinishPeerReadTask(peer);
395                 return;
396             } else {
397                 // Lazy unset to make sure we don't prematurely unset it while scheduling a new task.
398                 FINISH_READ_FUTURE_UPDATER.compareAndSet(peer, peerFinishReadFuture, null);
399             }
400         }
401         // We should only set readInProgress to false if there is any data that was read as otherwise we may miss to
402         // forward data later on.
403         if (peer.readInProgress && !peer.inboundBuffer.isEmpty()) {
404             peer.readInProgress = false;
405             peer.readInbound();
406         }
407     }

409     private class LocalUnsafe extends AbstractUnsafe implements LocalChannelUnsafe {

411         @Override
412         public void connect(final SocketAddress remoteAddress,
413                 SocketAddress localAddress, final Promise<Void> promise) {
414             if (!promise.setUncancellable() || !ensureOpen(promise)) {
415                 return;
416             }
417 
418             if (state == State.CONNECTED) {
419                 Exception cause = new AlreadyConnectedException();
420                 safeSetFailure(promise, cause);
421                 pipeline().fireExceptionCaught(cause);
422                 return;
423             }
424 
425             if (connectPromise != null) {
426                 throw new ConnectionPendingException();
427             }
428 
429             connectPromise = promise;
430 
431             if (state != State.BOUND) {
432                 // Not bound yet and no localAddress specified - get one.
433                 if (localAddress == null) {
434                     localAddress = new LocalAddress(LocalChannel.this);
435                 }
436             }
437 
438             if (localAddress != null) {
439                 try {
440                     doBind(localAddress);
441                 } catch (Throwable t) {
442                     safeSetFailure(promise, t);
443                     close(newPromise());
444                     return;
445                 }
446             }
447 
448             Channel boundChannel = LocalChannelRegistry.get(remoteAddress);
449             if (!(boundChannel instanceof LocalServerChannel)) {
450                 Exception cause = new ConnectException("connection refused: " + remoteAddress);
451                 safeSetFailure(promise, cause);
452                 close(newPromise());
453                 return;
454             }
455 
456             LocalServerChannel serverChannel = (LocalServerChannel) boundChannel;
457             peer = serverChannel.serve(LocalChannel.this);
458         }
459 
460         @Override
461         public void register0() {
462             // Check if both peer and parent are non-null because this channel was created by a LocalServerChannel.
463             // This is needed as a peer may not be null also if a LocalChannel was connected before and
464             // deregistered / registered later again.
465             //
466             // See https://github.com/netty/netty/issues/2400
467             if (peer != null && parent() != null) {
468                 // Store the peer in a local variable as it may be set to null if doClose() is called.
469                 // See https://github.com/netty/netty/issues/2144
470                 final LocalChannel peer = LocalChannel.this.peer;
471                 state = State.CONNECTED;
472 
473                 peer.remoteAddress = parent() == null ? null : parent().localAddress();
474                 peer.state = State.CONNECTED;
475 
476                 // Always call peer.eventLoop().execute() even if peer.eventLoop().inEventLoop() is true.
477                 // This ensures that if both channels are on the same event loop, the peer's channelActive
478                 // event is triggered *after* this channel's channelRegistered event, so that this channel's
479                 // pipeline is fully initialized by ChannelInitializer before any channelRead events.
480                 peer.executor().execute(() -> {
481                     Promise<Void> promise = peer.connectPromise;
482 
483                     // Only trigger fireChannelActive() if the promise was not null and was not completed yet.
484                     // connectPromise may be set to null if doClose() was called in the meantime.
485                     if (promise != null && promise.trySuccess(null)) {
486                         peer.pipeline().fireChannelActive();
487                         peer.readIfIsAutoRead();
488                     }
489                 });
490             }
491         }
