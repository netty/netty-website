124     @Override
125     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
126         int maxMessagesPerWrite = maxMessagesPerWrite();
127         while (maxMessagesPerWrite > 0) {
128             Object msg = in.current();
129             if (msg == null) {
130                 break;
131             }
132 
133             try {
134                 boolean done = false;
135                 for (int i = config().getWriteSpinCount(); i > 0; --i) {
136                     if (doWriteMessage(msg)) {
137                         done = true;
138                         break;
139                     }
140                 }
141 
142                 if (done) {
143                     in.remove();
144                     maxMessagesPerWrite--;
145                 } else {
146                     break;
147                 }
148             } catch (IOException e) {
149                 maxMessagesPerWrite--;
150 
151                 // Continue on write error as a DatagramChannel can write to multiple remote peers
152                 //
153                 // See https://github.com/netty/netty/issues/2665
154                 in.remove(e);
155             }
156         }
157 
158         if (in.isEmpty()) {
159             // Did write all messages.
160             clearFlag(Native.EPOLLOUT);
161         } else {
162             // Did not write all messages.
163             setFlag(Native.EPOLLOUT);
164         }
165     }

167     private boolean doWriteMessage(Object msg) throws Exception {
168         final Object data;
169         DomainSocketAddress remoteAddress;
170         if (msg instanceof AddressedEnvelope) {
171             @SuppressWarnings("unchecked")
172             AddressedEnvelope<?, DomainSocketAddress> envelope =
173                     (AddressedEnvelope<?, DomainSocketAddress>) msg;
174             data = envelope.content();
175             remoteAddress = envelope.recipient();
176         } else {
177             data = msg;
178             remoteAddress = null;
179         }
180 
181         if (data instanceof Buffer) {
182             return doWriteBufferMessage((Buffer) data, remoteAddress);
183         }
184         return doWriteByteBufMessage((ByteBuf) data, remoteAddress);
185     }
186 
187     private boolean doWriteBufferMessage(Buffer data, DomainSocketAddress remoteAddress) throws IOException {
188         final int initialReadableBytes = data.readableBytes();
189         if (initialReadableBytes == 0) {
190             return true;
191         }
192 
193         if (data.countReadableComponents() > 1) {
194             IovArray array = registration().cleanIovArray();
195             data.forEachReadable(0, array);
196             int count = array.count();
197             assert count != 0;
198 
199             final long writtenBytes;
200             if (remoteAddress == null) {
201                 writtenBytes = socket.writevAddresses(array.memoryAddress(0), count);
202             } else {
203                 writtenBytes = socket.sendToAddressesDomainSocket(
204                         array.memoryAddress(0), count, remoteAddress.path().getBytes(UTF_8));
205             }
206             return writtenBytes > 0;
207         }
208         if (remoteAddress == null) {
209             data.forEachReadable(0, (index, component) -> {
210                 int written = socket.writeAddress(component.readableNativeAddress(), 0, component.readableBytes());
211                 component.skipReadable(written);
212                 return false;
213             });
214         } else {
215             data.forEachReadable(0, (index, component) -> {
216                 int written = socket.sendToAddressDomainSocket(
217                         component.readableNativeAddress(), 0, component.readableBytes(),
218                         remoteAddress.path().getBytes(UTF_8));
219                 component.skipReadable(written);
220                 return false;
221             });
222         }
223         return data.readableBytes() < initialReadableBytes;
224     }

226     private boolean doWriteByteBufMessage(ByteBuf data, DomainSocketAddress remoteAddress) throws IOException {
227         final int dataLen = data.readableBytes();
228         if (dataLen == 0) {
229             return true;
230         }
231 
232         final long writtenBytes;
233         if (data.hasMemoryAddress()) {
234             long memoryAddress = data.memoryAddress();
235             if (remoteAddress == null) {
236                 writtenBytes = socket.writeAddress(memoryAddress, data.readerIndex(), data.writerIndex());
237             } else {
238                 writtenBytes = socket.sendToAddressDomainSocket(memoryAddress, data.readerIndex(), data.writerIndex(),
239                                                                 remoteAddress.path().getBytes(UTF_8));
240             }
241         } else if (data.nioBufferCount() > 1) {
242             IovArray array = registration().cleanIovArray();
243             array.add(data, data.readerIndex(), data.readableBytes());
244             int cnt = array.count();
245             assert cnt != 0;
246 
247             if (remoteAddress == null) {
248                 writtenBytes = socket.writevAddresses(array.memoryAddress(0), cnt);
249             } else {
250                 writtenBytes = socket.sendToAddressesDomainSocket(array.memoryAddress(0), cnt,
251                                                                   remoteAddress.path().getBytes(UTF_8));
252             }
253         } else {
254             ByteBuffer nioData = data.internalNioBuffer(data.readerIndex(), data.readableBytes());
255             if (remoteAddress == null) {
256                 writtenBytes = socket.write(nioData, nioData.position(), nioData.limit());
257             } else {
258                 writtenBytes = socket.sendToDomainSocket(nioData, nioData.position(), nioData.limit(),
259                                                          remoteAddress.path().getBytes(UTF_8));
260             }
261         }
262 
263         return writtenBytes > 0;
264     }

266     @Override
267     protected Object filterOutboundMessage(Object msg) {
268         if (msg instanceof DomainDatagramPacket) {
269             DomainDatagramPacket packet = (DomainDatagramPacket) msg;
270             ByteBuf content = packet.content();
271             return UnixChannelUtil.isBufferCopyNeededForWrite(content) ?
272                     new DomainDatagramPacket(newDirectBuffer(packet, content), packet.recipient()) : msg;
273         }
274         if (msg instanceof BufferDomainDatagramPacket) {
275             BufferDomainDatagramPacket packet = (BufferDomainDatagramPacket) msg;
276             Buffer content = packet.content();
277             return UnixChannelUtil.isBufferCopyNeededForWrite(content) ?
278                     new BufferDomainDatagramPacket(newDirectBuffer(packet, content), packet.recipient()) : msg;
279         }
280 
281         if (msg instanceof Buffer) {
282             Buffer buf = (Buffer) msg;
283             return UnixChannelUtil.isBufferCopyNeededForWrite(buf) ? newDirectBuffer(buf) : buf;
284         }
285         if (msg instanceof ByteBufConvertible) {
286             ByteBuf buf = ((ByteBufConvertible) msg).asByteBuf();
287             return UnixChannelUtil.isBufferCopyNeededForWrite(buf) ? newDirectBuffer(buf) : buf;
288         }
289 
290         if (msg instanceof AddressedEnvelope) {
291             @SuppressWarnings("unchecked")
292             AddressedEnvelope<Object, SocketAddress> e = (AddressedEnvelope<Object, SocketAddress>) msg;
293             SocketAddress recipient = e.recipient();
294             if (recipient == null || recipient instanceof DomainSocketAddress) {
295                 DomainSocketAddress domainRecipient = (DomainSocketAddress) recipient;
296                 if (e.content() instanceof Buffer) {
297                     Buffer buf = (Buffer) e.content();
298                     if (UnixChannelUtil.isBufferCopyNeededForWrite(buf)) {
299                         try {
300                             return new DefaultBufferAddressedEnvelope<>(newDirectBuffer(buf), domainRecipient);
301                         } finally {
302                             ReferenceCountUtil.release(e);
303                         }
304                     }
305                     return e;
306                 }
307                 if (e.content() instanceof ByteBufConvertible) {
308                     ByteBuf content = ((ByteBufConvertible) e.content()).asByteBuf();
309                     return UnixChannelUtil.isBufferCopyNeededForWrite(content) ?
310                             new DefaultByteBufAddressedEnvelope<>(newDirectBuffer(e, content), domainRecipient) : e;
311                 }
312             }
313         }
314 
315         throw new UnsupportedOperationException(
316                 "unsupported message type: " + StringUtil.simpleClassName(msg) + EXPECTED_TYPES);
317     }

367     final class EpollDomainDatagramChannelUnsafe extends AbstractEpollUnsafe {
368 
369         @Override
370         void epollInReady() {
371             assert executor().inEventLoop();
372             final DomainDatagramChannelConfig config = config();
373             if (shouldBreakEpollInReady(config)) {
374                 clearEpollIn0();
375                 return;
376             }
377             final EpollRecvBufferAllocatorHandle allocHandle = recvBufAllocHandle();
378 
379             final ChannelPipeline pipeline = pipeline();
380             final boolean useBufferApi = config.getRecvBufferAllocatorUseBuffer();
381             allocHandle.reset(config);
382             epollInBefore();
383 
384             try {
385                 Throwable exception = useBufferApi ?
386                         doReadBuffer(allocHandle, pipeline) :
387                         doReadByteBuf(allocHandle, pipeline);
388 
389                 allocHandle.readComplete();
390                 pipeline.fireChannelReadComplete();
391 
392                 if (exception != null) {
393                     pipeline.fireExceptionCaught(exception);
394                 }
395                 readIfIsAutoRead();
396             } finally {
397                 epollInFinally(config);
398             }
399         }
400 
401         private Throwable doReadBuffer(EpollRecvBufferAllocatorHandle allocHandle, ChannelPipeline pipeline) {
402             BufferAllocator allocator = config().getBufferAllocator();
403             if (allocator.getAllocationType() != StandardAllocationTypes.OFF_HEAP) {
404                 allocator = DefaultBufferAllocators.offHeapAllocator();
405             }
406             Buffer buf = null;
407             try {
408                 boolean connected = isConnected();
409                 do {
410                     buf = allocHandle.allocate(allocator);
411                     allocHandle.attemptedBytesRead(buf.writableBytes());
412 
413                     final BufferDomainDatagramPacket packet;
414                     if (connected) {
415                         doReadBytes(buf);
416                         if (allocHandle.lastBytesRead() <= 0) {
417                             // nothing was read, release the buffer.
418                             buf.close();
419                             break;
420                         }
421                         packet = new BufferDomainDatagramPacket(buf, (DomainSocketAddress) localAddress(),
422                                 (DomainSocketAddress) remoteAddress());
423                     } else {
424                         final RecvFromAddressDomainSocket recvFrom = new RecvFromAddressDomainSocket(socket);
425                         buf.forEachWritable(0, recvFrom);
426                         final DomainDatagramSocketAddress remoteAddress = recvFrom.remoteAddress();
427 
428                         if (remoteAddress == null) {
429                             allocHandle.lastBytesRead(-1);
430                             buf.close();
431                             break;
432                         }
433                         DomainSocketAddress localAddress = remoteAddress.localAddress();
434                         if (localAddress == null) {
435                             localAddress = (DomainSocketAddress) localAddress();
436                         }
437                         allocHandle.lastBytesRead(remoteAddress.receivedAmount());
438                         buf.skipWritable(allocHandle.lastBytesRead());
439 
440                         packet = new BufferDomainDatagramPacket(buf, localAddress, remoteAddress);
441                     }
442 
443                     allocHandle.incMessagesRead(1);
444 
445                     readPending = false;
446                     pipeline.fireChannelRead(packet);
447 
448                     buf = null;
449 
450                     // We use the TRUE_SUPPLIER as it is also ok to read less than what we did try to read (as long
451                     // as we read anything).
452                 } while (allocHandle.continueReading(UncheckedBooleanSupplier.TRUE_SUPPLIER));
453             } catch (Throwable t) {
454                 if (buf != null) {
455                     buf.close();
456                 }
457                 return t;
458             }
459             return null;
460         }
461 
462         private Throwable doReadByteBuf(EpollRecvBufferAllocatorHandle allocHandle, ChannelPipeline pipeline) {
463             final ByteBufAllocator allocator = config().getAllocator();
464             ByteBuf byteBuf = null;
465             try {
466                 boolean connected = isConnected();
467                 do {
468                     byteBuf = allocHandle.allocate(allocator);
469                     allocHandle.attemptedBytesRead(byteBuf.writableBytes());
470 
471                     final DomainDatagramPacket packet;
472                     if (connected) {
473                         allocHandle.lastBytesRead(doReadBytes(byteBuf));
474                         if (allocHandle.lastBytesRead() <= 0) {
475                             // nothing was read, release the buffer.
476                             byteBuf.release();
477                             break;
478                         }
479                         packet = new DomainDatagramPacket(byteBuf, (DomainSocketAddress) localAddress(),
480                                 (DomainSocketAddress) remoteAddress());
481                     } else {
482                         final DomainDatagramSocketAddress remoteAddress;
483                         if (byteBuf.hasMemoryAddress()) {
484                             // has a memory address so use optimized call
485                             remoteAddress = socket.recvFromAddressDomainSocket(byteBuf.memoryAddress(),
486                                     byteBuf.writerIndex(), byteBuf.capacity());
487                         } else {
488                             ByteBuffer nioData = byteBuf.internalNioBuffer(
489                                     byteBuf.writerIndex(), byteBuf.writableBytes());
490                             remoteAddress =
491                                     socket.recvFromDomainSocket(nioData, nioData.position(), nioData.limit());
492                         }
493 
494                         if (remoteAddress == null) {
495                             allocHandle.lastBytesRead(-1);
496                             byteBuf.release();
497                             break;
498                         }
499                         DomainSocketAddress localAddress = remoteAddress.localAddress();
500                         if (localAddress == null) {
501                             localAddress = (DomainSocketAddress) localAddress();
502                         }
503                         allocHandle.lastBytesRead(remoteAddress.receivedAmount());
504                         byteBuf.writerIndex(byteBuf.writerIndex() + allocHandle.lastBytesRead());
505 
506                         packet = new DomainDatagramPacket(byteBuf, localAddress, remoteAddress);
507                     }
508 
509                     allocHandle.incMessagesRead(1);
510 
511                     readPending = false;
512                     pipeline.fireChannelRead(packet);
513 
514                     byteBuf = null;
515 
516                     // We use the TRUE_SUPPLIER as it is also ok to read less than what we did try to read (as long
517                     // as we read anything).
518                 } while (allocHandle.continueReading(UncheckedBooleanSupplier.TRUE_SUPPLIER));
519             } catch (Throwable t) {
520                 if (byteBuf != null) {
521                     byteBuf.release();
522                 }
523                 return t;
524             }
525             return null;
526         }
