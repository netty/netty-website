158     @Override
159     protected void doClose() throws Exception {
160         active = false;
161         // Even if we allow half closed sockets we should give up on reading. Otherwise we may allow a read attempt on a
162         // socket which has not even been connected yet. This has been observed to block during unit tests.
163         inputClosedSeenErrorOnRead = true;
164         try {
165             Promise<Void> promise = connectPromise;
166             if (promise != null) {
167                 // Use tryFailure() instead of setFailure() to avoid the race against cancel().
168                 promise.tryFailure(new ClosedChannelException());
169                 connectPromise = null;
170             }
171 
172             Future<?> future = connectTimeoutFuture;
173             if (future != null) {
174                 future.cancel();
175                 connectTimeoutFuture = null;
176             }
177 
178             if (isRegistered()) {
179                 // Need to check if we are on the EventLoop as doClose() may be triggered by the GlobalEventExecutor
180                 // if SO_LINGER is used.
181                 //
182                 // See https://github.com/netty/netty/issues/7159
183                 EventLoop loop = executor();
184                 if (loop.inEventLoop()) {
185                     doDeregister();
186                 } else {
187                     loop.execute(() -> {
188                         try {
189                             doDeregister();
190                         } catch (Throwable cause) {
191                             pipeline().fireExceptionCaught(cause);
192                         }
193                     });
194                 }
195             }
196         } finally {
197             socket.close();
198         }
199     }

217         // Just in case the previous EventLoop was shutdown abruptly, or an event is still pending on the old EventLoop

230     @Override
231     protected final void doBeginRead() throws Exception {
232         // Channel.read() or ChannelHandlerContext.read() was called
233         final AbstractEpollUnsafe unsafe = (AbstractEpollUnsafe) unsafe();
234         unsafe.readPending = true;
235 
236         // We must set the read flag here as it is possible the user didn't read in the last read loop, the
237         // executeEpollInReadyRunnable could read nothing, and if the user doesn't explicitly call read they will
238         // never get data after this.
239         setFlag(Native.EPOLLIN);
240 
241         // If EPOLL ET mode is enabled and auto read was toggled off on the last read loop then we may not be notified
242         // again if we didn't consume all the data. So we force a read operation here if there maybe more data.
243         if (unsafe.maybeMoreDataToRead) {
244             unsafe.executeEpollInReadyRunnable(config());
245         }
246     }

260     final void clearEpollIn() {
261         // Only clear if registered with an EventLoop as otherwise
262         if (isRegistered()) {
263             final EventLoop loop = executor();
264             final AbstractEpollUnsafe unsafe = (AbstractEpollUnsafe) unsafe();
265             if (loop.inEventLoop()) {
266                 unsafe.clearEpollIn0();
267             } else {
268                 // schedule a task to clear the EPOLLIN as it is not safe to modify it directly
269                 loop.execute(() -> {
270                     if (!unsafe.readPending && !config().isAutoRead()) {
271                         // Still no read triggered so clear it now
272                         unsafe.clearEpollIn0();
273                     }
274                 });
275             }
276         } else  {
277             // The EventLoop is not registered atm so just update the flags so the correct value
278             // will be used once the channel is registered
279             flags &= ~Native.EPOLLIN;
280         }
281     }

299     /**
300      * Returns an off-heap copy of the specified {@link ByteBuf}, and releases the specified holder.
301      * The caller must ensure that the holder releases the original {@link ByteBuf} when the holder is released by
302      * this method.
303      */
304     protected final ByteBuf newDirectBuffer(Object holder, ByteBuf buf) {
305         final int readableBytes = buf.readableBytes();
306         if (readableBytes == 0) {
307             ReferenceCountUtil.release(holder);
308             return Unpooled.EMPTY_BUFFER;
309         }
310 
311         final ByteBufAllocator alloc = alloc();
312         if (alloc.isDirectBufferPooled()) {
313             return newDirectBuffer0(holder, buf, alloc, readableBytes);
314         }
315 
316         final ByteBuf directBuf = ByteBufUtil.threadLocalDirectBuffer();
317         if (directBuf == null) {
318             return newDirectBuffer0(holder, buf, alloc, readableBytes);
319         }
320 
321         directBuf.writeBytes(buf, buf.readerIndex(), readableBytes);
322         ReferenceCountUtil.safeRelease(holder);
323         return directBuf;
324     }

326     private static ByteBuf newDirectBuffer0(Object holder, ByteBuf buf, ByteBufAllocator alloc, int capacity) {
327         final ByteBuf directBuf = alloc.directBuffer(capacity);
328         directBuf.writeBytes(buf, buf.readerIndex(), capacity);
329         ReferenceCountUtil.safeRelease(holder);
330         return directBuf;
331     }

340     /**
341      * Returns an off-heap copy of the given {@link Buffer}, and then closes the {@code holder} under the assumption
342      * that it owned (or was itself) the buffer.
343      */
344     protected final Buffer newDirectBuffer(Resource<?> holder, Buffer buf) {
345         BufferAllocator allocator = bufferAllocator();
346         if (allocator.getAllocationType() != StandardAllocationTypes.OFF_HEAP) {
347             allocator = DefaultBufferAllocators.offHeapAllocator();
348         }
349         try (holder) {
350             int readableBytes = buf.readableBytes();
351             Buffer directCopy = allocator.allocate(readableBytes);
352             if (readableBytes > 0) {
353                 directCopy.writeBytes(buf);
354             }
355             return directCopy;
356         }

365     /**
366      * Read bytes into the given {@link ByteBuf} and return the amount.
367      */
368     protected final int doReadBytes(ByteBuf byteBuf) throws Exception {
369         int writerIndex = byteBuf.writerIndex();
370         int localReadAmount;
371         unsafe().recvBufAllocHandle().attemptedBytesRead(byteBuf.writableBytes());
372         if (byteBuf.hasMemoryAddress()) {
373             localReadAmount = socket.readAddress(byteBuf.memoryAddress(), writerIndex, byteBuf.capacity());
374         } else {
375             ByteBuffer buf = byteBuf.internalNioBuffer(writerIndex, byteBuf.writableBytes());
376             localReadAmount = socket.read(buf, buf.position(), buf.limit());
377         }
378         if (localReadAmount > 0) {
379             byteBuf.writerIndex(writerIndex + localReadAmount);
380         }
381         return localReadAmount;
382     }
383 
384     /**
385      * Read bytes into the given {@link Buffer} and return the amount.
386      */
387     protected final void doReadBytes(Buffer buffer) throws Exception {
388         unsafe().recvBufAllocHandle().attemptedBytesRead(buffer.writableBytes());
389         buffer.forEachWritable(0, (index, component) -> {
390             long address = component.writableNativeAddress();
391             assert address != 0;
392             int localReadAmount = socket.readAddress(address, 0, component.writableBytes());
393             unsafe().recvBufAllocHandle().lastBytesRead(localReadAmount);
394             if (localReadAmount > 0) {
395                 component.skipWritable(localReadAmount);
396             }
397             return false;
398         });
399     }
400 
401     protected final int doWriteBytes(ChannelOutboundBuffer in, ByteBuf buf) throws Exception {
402         if (buf.hasMemoryAddress()) {
403             int localFlushedAmount = socket.writeAddress(buf.memoryAddress(), buf.readerIndex(), buf.writerIndex());
404             if (localFlushedAmount > 0) {
405                 in.removeBytes(localFlushedAmount);
406                 return 1;
407             }
408         } else {
409             final ByteBuffer nioBuf = buf.nioBufferCount() == 1 ?
410                     buf.internalNioBuffer(buf.readerIndex(), buf.readableBytes()) : buf.nioBuffer();
411             int localFlushedAmount = socket.write(nioBuf, nioBuf.position(), nioBuf.limit());
412             if (localFlushedAmount > 0) {
413                 nioBuf.position(nioBuf.position() + localFlushedAmount);
414                 in.removeBytes(localFlushedAmount);
415                 return 1;
416             }
417         }
418         return WRITE_STATUS_SNDBUF_FULL;
419     }
420 
421     protected final int doWriteBytes(ChannelOutboundBuffer in, Buffer buf) throws Exception {
422         int initialReadableBytes = buf.readableBytes();
423         buf.forEachReadable(0, (index, component) -> {
424             long address = component.readableNativeAddress();
425             assert address != 0;
426             int written = socket.writeAddress(address, 0, component.readableBytes());
427             if (written > 0) {
428                 component.skipReadable(written);
429             }
430             return false;
431         });
432         int readableBytesLeft = buf.readableBytes();
433         if (readableBytesLeft < initialReadableBytes) {
434             int bytesWritten = initialReadableBytes - readableBytesLeft;
435             buf.skipReadable(-bytesWritten); // Restore read offset for ChannelOutboundBuffer.
436             in.removeBytes(bytesWritten);
437             return 1; // Some data was written to the socket.
438         }
439         return WRITE_STATUS_SNDBUF_FULL;
440     }

442     /**
443      * Write bytes to the socket, with or without a remote address.
444      * Used for datagram and TCP client fast open writes.
445      */
446     final long doWriteOrSendBytes(ByteBuf data, InetSocketAddress remoteAddress, boolean fastOpen)
447             throws IOException {
448         assert !(fastOpen && remoteAddress == null) : "fastOpen requires a remote address";
449         if (data.hasMemoryAddress()) {
450             long memoryAddress = data.memoryAddress();
451             if (remoteAddress == null) {
452                 return socket.writeAddress(memoryAddress, data.readerIndex(), data.writerIndex());
453             }
454             return socket.sendToAddress(memoryAddress, data.readerIndex(), data.writerIndex(),
455                     remoteAddress.getAddress(), remoteAddress.getPort(), fastOpen);
456         }
457 
458         if (data.nioBufferCount() > 1) {
459             IovArray array = registration().cleanIovArray();
460             array.add(data, data.readerIndex(), data.readableBytes());
461             int cnt = array.count();
462             assert cnt != 0;
463 
464             if (remoteAddress == null) {
465                 return socket.writevAddresses(array.memoryAddress(0), cnt);
466             }
467             return socket.sendToAddresses(array.memoryAddress(0), cnt,
468                     remoteAddress.getAddress(), remoteAddress.getPort(), fastOpen);
469         }
470 
471         ByteBuffer nioData = data.internalNioBuffer(data.readerIndex(), data.readableBytes());
472         if (remoteAddress == null) {
473             return socket.write(nioData, nioData.position(), nioData.limit());
474         }
475         return socket.sendTo(nioData, nioData.position(), nioData.limit(),
476                 remoteAddress.getAddress(), remoteAddress.getPort(), fastOpen);
477     }

479     /**
480      * Write bytes to the socket, with or without a remote address.
481      * Used for datagram and TCP client fast open writes.
482      */
483     final long doWriteOrSendBytes(Buffer data, InetSocketAddress remoteAddress, boolean fastOpen)
484             throws IOException {
485         assert !(fastOpen && remoteAddress == null) : "fastOpen requires a remote address";
486 
487         IovArray array = registration().cleanIovArray();
488         data.forEachReadable(0, array);
489         int count = array.count();
490         assert count != 0;
491         if (remoteAddress == null) {
492             return socket.writevAddresses(array.memoryAddress(0), count);
493         }
494         return socket.sendToAddresses(array.memoryAddress(0), count,
495                                       remoteAddress.getAddress(), remoteAddress.getPort(), fastOpen);
496     }

498     protected abstract class AbstractEpollUnsafe extends AbstractUnsafe {
499         boolean readPending;
500         boolean maybeMoreDataToRead;
501         private EpollRecvBufferAllocatorHandle allocHandle;
502         private final Runnable epollInReadyRunnable = new Runnable() {
503             @Override
504             public void run() {
505                 epollInReadyRunnablePending = false;
506                 epollInReady();
507             }
508         };
509 
510         /**
511          * Called once EPOLLIN event is ready to be processed
512          */
513         abstract void epollInReady();
514 
515         final void epollInBefore() {
516             maybeMoreDataToRead = false;
517         }
518 
519         final void epollInFinally(ChannelConfig config) {
520             maybeMoreDataToRead = allocHandle.maybeMoreDataToRead();
521 
522             if (allocHandle.isReceivedRdHup() || readPending && maybeMoreDataToRead) {
523                 // trigger a read again as there may be something left to read and because of epoll ET we
524                 // will not get notified again until we read everything from the socket
525                 //
526                 // It is possible the last fireChannelRead call could cause the user to call read() again, or if
527                 // autoRead is true the call to channelReadComplete would also call read, but maybeMoreDataToRead is set
528                 // to false before every read operation to prevent re-entry into epollInReady() we will not read from
529                 // the underlying OS again unless the user happens to call read again.
530                 executeEpollInReadyRunnable(config);
531             } else if (!readPending && !config.isAutoRead()) {
532                 // Check if there is a readPending which was not processed yet.
533                 // This could be for two reasons:
534                 // * The user called Channel.read() or ChannelHandlerContext.read() in channelRead(...) method
535                 // * The user called Channel.read() or ChannelHandlerContext.read() in channelReadComplete(...) method
536                 //
537                 // See https://github.com/netty/netty/issues/2254
538                 clearEpollIn();
539             }
540         }
541 
542         final void executeEpollInReadyRunnable(ChannelConfig config) {
543             if (epollInReadyRunnablePending || !isActive() || shouldBreakEpollInReady(config)) {
544                 return;
545             }
546             epollInReadyRunnablePending = true;
547             executor().execute(epollInReadyRunnable);
548         }
549 
550         /**
551          * Called once EPOLLRDHUP event is ready to be processed
552          */
553         final void epollRdHupReady() {
554             // This must happen before we attempt to read. This will ensure reading continues until an error occurs.
555             recvBufAllocHandle().receivedRdHup();
556 
557             if (isActive()) {
558                 // If it is still active, we need to call epollInReady as otherwise we may miss to
559                 // read pending data from the underlying file descriptor.
560                 // See https://github.com/netty/netty/issues/3709
561                 epollInReady();
562             } else {
563                 // Just to be safe make sure the input marked as closed.
564                 shutdownInput(true);
565             }
566 
567             // Clear the EPOLLRDHUP flag to prevent continuously getting woken up on this event.
568             clearEpollRdHup();
569         }
570 
571         /**
572          * Clear the {@link Native#EPOLLRDHUP} flag from EPOLL, and close on failure.
573          */
574         private void clearEpollRdHup() {
575             try {
576                 clearFlag(Native.EPOLLRDHUP);
577             } catch (IOException e) {
578                 pipeline().fireExceptionCaught(e);
579                 close(newPromise());
580             }
581         }
582 
583         /**
584          * Shutdown the input side of the channel.
585          */
586         void shutdownInput(boolean rdHup) {
587             if (!socket.isInputShutdown()) {
588                 if (isAllowHalfClosure(config())) {
589                     try {
590                         socket.shutdown(true, false);
591                     } catch (IOException ignored) {
592                         // We attempted to shutdown and failed, which means the input has already effectively been
593                         // shutdown.
594                         fireEventAndClose(ChannelInputShutdownEvent.INSTANCE);
595                         return;
596                     } catch (NotYetConnectedException ignore) {
597                         // We attempted to shutdown and failed, which means the input has already effectively been
598                         // shutdown.
599                     }
600                     clearEpollIn();
601                     pipeline().fireUserEventTriggered(ChannelInputShutdownEvent.INSTANCE);
602                 } else {
603                     close(newPromise());
604                 }
605             } else if (!rdHup) {
606                 inputClosedSeenErrorOnRead = true;
607                 pipeline().fireUserEventTriggered(ChannelInputShutdownReadComplete.INSTANCE);
608             }
609         }
610 
611         private void fireEventAndClose(Object evt) {
612             pipeline().fireUserEventTriggered(evt);
613             close(newPromise());
614         }

632         @Override
633         protected final void flush0() {
634             // Flush immediately only when there's no pending flush.
635             // If there's a pending flush operation, event loop will call forceFlush() later,
636             // and thus there's no need to call it now.
637             if (!isFlagSet(Native.EPOLLOUT)) {
638                 super.flush0();
639             }

642         /**
643          * Called once a EPOLLOUT event is ready to be processed
644          */
645         final void epollOutReady() {
646             if (connectPromise != null) {
647                 // pending connect which is now complete so handle it.
648                 finishConnect();
649             } else if (!socket.isOutputShutdown()) {
650                 // directly call super.flush0() to force a flush now
651                 super.flush0();
652             }
653         }

655         protected final void clearEpollIn0() {
656             assert executor().inEventLoop();
657             try {
658                 readPending = false;
659                 clearFlag(Native.EPOLLIN);
660             } catch (IOException e) {
661                 // When this happens there is something completely wrong with either the filedescriptor or epoll,
662                 // so fire the exception through the pipeline and close the Channel.
663                 pipeline().fireExceptionCaught(e);
664                 unsafe().close(newPromise());
665             }
666         }

668         @Override
669         public void connect(
670                 final SocketAddress remoteAddress, final SocketAddress localAddress, final Promise<Void> promise) {
671             if (!promise.setUncancellable() || !ensureOpen(promise)) {
672                 return;
673             }
674 
675             try {
676                 if (connectPromise != null) {
677                     throw new ConnectionPendingException();
678                 }
679 
680                 boolean wasActive = isActive();
681                 if (doConnect(remoteAddress, localAddress)) {
682                     fulfillConnectPromise(promise, wasActive);
683                 } else {
684                     connectPromise = promise;
685                     requestedRemoteAddress = remoteAddress;
686 
687                     // Schedule connect timeout.
688                     int connectTimeoutMillis = config().getConnectTimeoutMillis();
689                     if (connectTimeoutMillis > 0) {
690                         connectTimeoutFuture = executor().schedule(() -> {
691                             Promise<Void> connectPromise = AbstractEpollChannel.this.connectPromise;
692                             if (connectPromise != null && !connectPromise.isDone()
693                                     && connectPromise.tryFailure(new ConnectTimeoutException(
694                                     "connection timed out: " + remoteAddress))) {
695                                 close(newPromise());
696                             }
697                         }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
698                     }
699 
700                     promise.asFuture().addListener(future -> {
701                         if (future.isCancelled()) {
702                             if (connectTimeoutFuture != null) {
703                                 connectTimeoutFuture.cancel();
704                             }
705                             connectPromise = null;
706                             close(newPromise());
707                         }
708                     });
709                 }
710             } catch (Throwable t) {
711                 closeIfClosed();
712                 promise.tryFailure(annotateConnectException(t, remoteAddress));
713             }
714         }
715 
716         private void fulfillConnectPromise(Promise<Void> promise, boolean wasActive) {
717             if (promise == null) {
718                 // Closed via cancellation and the promise has been notified already.
719                 return;
720             }
721             active = true;
722 
723             // Get the state as trySuccess() may trigger an ChannelFutureListeners that will close the Channel.
724             // We still need to ensure we call fireChannelActive() in this case.
725             boolean active = isActive();
726 
727             // trySuccess() will return false if a user cancelled the connection attempt.
728             boolean promiseSet = promise.trySuccess(null);
729 
730             // Regardless if the connection attempt was cancelled, channelActive() event should be triggered,
731             // because what happened is what happened.
732             if (!wasActive && active) {
733                 pipeline().fireChannelActive();
734                 readIfIsAutoRead();
735             }
736 
737             // If a user cancelled the connection attempt, close the channel, which is followed by channelInactive().
738             if (!promiseSet) {
739                 close(newPromise());
740             }
741         }
742 
743         private void fulfillConnectPromise(Promise<Void> promise, Throwable cause) {
744             if (promise == null) {
745                 // Closed via cancellation and the promise has been notified already.
746                 return;
747             }
748 
749             // Use tryFailure() instead of setFailure() to avoid the race against cancel().
750             promise.tryFailure(cause);
751             closeIfClosed();
752         }
753 
754         private void finishConnect() {
755             // Note this method is invoked by the event loop only if the connection attempt was
756             // neither cancelled nor timed out.
757 
758             assert executor().inEventLoop();
759 
760             boolean connectStillInProgress = false;
761             try {
762                 boolean wasActive = isActive();
763                 if (!doFinishConnect()) {
764                     connectStillInProgress = true;
765                     return;
766                 }
767                 fulfillConnectPromise(connectPromise, wasActive);
768             } catch (Throwable t) {
769                 fulfillConnectPromise(connectPromise, annotateConnectException(t, requestedRemoteAddress));
770             } finally {
771                 if (!connectStillInProgress) {
772                     // Check for null as the connectTimeoutFuture is only created if a connectTimeoutMillis > 0 is used
773                     // See https://github.com/netty/netty/issues/1770
774                     if (connectTimeoutFuture != null) {
775                         connectTimeoutFuture.cancel();
776                     }
777                     connectPromise = null;
778                 }
779             }
780         }
781 
782         /**
783          * Finish the connect
784          */
785         private boolean doFinishConnect() throws Exception {
786             if (socket.finishConnect()) {
787                 clearFlag(Native.EPOLLOUT);
788                 if (requestedRemoteAddress instanceof InetSocketAddress) {
789                     remote = computeRemoteAddr((InetSocketAddress) requestedRemoteAddress, socket.remoteAddress());
790                 }
791                 requestedRemoteAddress = null;
792 
793                 return true;
794             }
795             setFlag(Native.EPOLLOUT);
796             return false;
797         }
798     }

800     @Override
801     protected void doBind(SocketAddress local) throws Exception {
802         if (local instanceof InetSocketAddress) {
803             checkResolvable((InetSocketAddress) local);
804         }
805         socket.bind(local);
806         this.local = socket.localAddress();
807     }
808 
809     /**
810      * Connect to the remote peer
811      */
812     protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
813         if (localAddress instanceof InetSocketAddress) {
814             checkResolvable((InetSocketAddress) localAddress);
815         }
816 
817         InetSocketAddress remoteSocketAddr = remoteAddress instanceof InetSocketAddress
818                 ? (InetSocketAddress) remoteAddress : null;
819         if (remoteSocketAddr != null) {
820             checkResolvable(remoteSocketAddr);
821         }
822 
823         if (remote != null) {
824             // Check if already connected before trying to connect. This is needed as connect(...) will not return -1
825             // and set errno to EISCONN if a previous connect(...) attempt was setting errno to EINPROGRESS and finished
826             // later.
827             throw new AlreadyConnectedException();
828         }
829 
830         if (localAddress != null) {
831             socket.bind(localAddress);
832         }
833 
834         boolean connected = doConnect0(remoteAddress);
835         if (connected) {
836             remote = remoteSocketAddr == null ?
837                     remoteAddress : computeRemoteAddr(remoteSocketAddr, socket.remoteAddress());
838         }
839         // We always need to set the localAddress even if not connected yet as the bind already took place.
840         //
841         // See https://github.com/netty/netty/issues/3463
842         local = socket.localAddress();
843         return connected;
844     }
845 
846     boolean doConnect0(SocketAddress remote) throws Exception {
847         boolean success = false;
848         try {
849             boolean connected = socket.connect(remote);
850             if (!connected) {
851                 setFlag(Native.EPOLLOUT);
852             }
853             success = true;
854             return connected;
855         } finally {
856             if (!success) {
857                 doClose();
858             }
859         }
860     }

864         return local;

869         return remote;
