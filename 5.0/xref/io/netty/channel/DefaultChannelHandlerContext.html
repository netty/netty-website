30      // This class keeps an integer member field 'skipFlags' whose each bit tells if the corresponding handler method
31      // is annotated with @Skip. 'skipFlags' is retrieved in runtime via the reflection API and is cached.
32      // The following constants signify which bit of 'skipFlags' corresponds to which handler method:
33  
34      static final int MASK_HANDLER_ADDED = 1;
35      static final int MASK_HANDLER_REMOVED = 1 << 1;
36      private static final int MASK_EXCEPTION_CAUGHT = 1 << 2;
37      private static final int MASK_CHANNEL_REGISTERED = 1 << 3;
38      private static final int MASK_CHANNEL_ACTIVE = 1 << 4;
39      private static final int MASK_CHANNEL_INACTIVE = 1 << 5;
40      private static final int MASK_CHANNEL_READ = 1 << 6;
41      private static final int MASK_CHANNEL_READ_COMPLETE = 1 << 7;
42      private static final int MASK_CHANNEL_WRITABILITY_CHANGED = 1 << 8;
43      private static final int MASK_USER_EVENT_TRIGGERED = 1 << 9;
44      private static final int MASK_BIND = 1 << 10;
45      private static final int MASK_CONNECT = 1 << 11;
46      private static final int MASK_DISCONNECT = 1 << 12;
47      private static final int MASK_CLOSE = 1 << 13;
48      private static final int MASK_READ = 1 << 14;
49      private static final int MASK_WRITE = 1 << 15;
50      private static final int MASK_FLUSH = 1 << 16;
51  
52      /**
53       * Cache the result of the costly generation of {@link #skipFlags} in the partitioned synchronized
54       * {@link WeakHashMap}.
55       */
56      @SuppressWarnings("unchecked")
57      private static final WeakHashMap<Class<?>, Integer>[] skipFlagsCache =
58              new WeakHashMap[Runtime.getRuntime().availableProcessors()];
59  
60      static {
61          for (int i = 0; i < skipFlagsCache.length; i ++) {
62              skipFlagsCache[i] = new WeakHashMap<Class<?>, Integer>();
63          }
64      }
65  
66      /**
67       * Returns an integer bitset that tells which handler methods were annotated with {@link Skip}.
68       * It gets the value from {@link #skipFlagsCache} if an handler of the same type were queried before.
69       * Otherwise, it delegates to {@link #skipFlags0(Class)} to get it.
70       */
71      private static int skipFlags(ChannelHandler handler) {
72          WeakHashMap<Class<?>, Integer> cache =
73                  skipFlagsCache[(int) (Thread.currentThread().getId() % skipFlagsCache.length)];
74          Class<? extends ChannelHandler> handlerType = handler.getClass();
75          int flagsVal;
76          synchronized (cache) {
77              Integer flags = cache.get(handlerType);
78              if (flags != null) {
79                  flagsVal = flags;
80              } else {
81                  flagsVal = skipFlags0(handlerType);
82                  cache.put(handlerType, Integer.valueOf(flagsVal));
83              }
84          }
85  
86          return flagsVal;
87      }
88  
89      /**
90       * Determines the {@link #skipFlags} of the specified {@code handlerType} using the reflection API.
91       */
92      private static int skipFlags0(Class<? extends ChannelHandler> handlerType) {
93          int flags = 0;
94          try {
95              if (handlerType.getMethod(
96                      "handlerAdded", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
97                  flags |= MASK_HANDLER_ADDED;
98              }
99              if (handlerType.getMethod(
100                     "handlerRemoved", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
101                 flags |= MASK_HANDLER_REMOVED;
102             }
103             if (handlerType.getMethod(
104                     "exceptionCaught", ChannelHandlerContext.class, Throwable.class).isAnnotationPresent(Skip.class)) {
105                 flags |= MASK_EXCEPTION_CAUGHT;
106             }
107             if (handlerType.getMethod(
108                     "channelRegistered", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
109                 flags |= MASK_CHANNEL_REGISTERED;
110             }
111             if (handlerType.getMethod(
112                     "channelActive", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
113                 flags |= MASK_CHANNEL_ACTIVE;
114             }
115             if (handlerType.getMethod(
116                     "channelInactive", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
117                 flags |= MASK_CHANNEL_INACTIVE;
118             }
119             if (handlerType.getMethod(
120                     "channelRead", ChannelHandlerContext.class, Object.class).isAnnotationPresent(Skip.class)) {
121                 flags |= MASK_CHANNEL_READ;
122             }
123             if (handlerType.getMethod(
124                     "channelReadComplete", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
125                 flags |= MASK_CHANNEL_READ_COMPLETE;
126             }
127             if (handlerType.getMethod(
128                     "channelWritabilityChanged", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
129                 flags |= MASK_CHANNEL_WRITABILITY_CHANGED;
130             }
131             if (handlerType.getMethod(
132                     "userEventTriggered", ChannelHandlerContext.class, Object.class).isAnnotationPresent(Skip.class)) {
133                 flags |= MASK_USER_EVENT_TRIGGERED;
134             }
135             if (handlerType.getMethod(
136                     "bind", ChannelHandlerContext.class,
137                     SocketAddress.class, ChannelPromise.class).isAnnotationPresent(Skip.class)) {
138                 flags |= MASK_BIND;
139             }
140             if (handlerType.getMethod(
141                     "connect", ChannelHandlerContext.class, SocketAddress.class, SocketAddress.class,
142                     ChannelPromise.class).isAnnotationPresent(Skip.class)) {
143                 flags |= MASK_CONNECT;
144             }
145             if (handlerType.getMethod(
146                     "disconnect", ChannelHandlerContext.class, ChannelPromise.class).isAnnotationPresent(Skip.class)) {
147                 flags |= MASK_DISCONNECT;
148             }
149             if (handlerType.getMethod(
150                     "close", ChannelHandlerContext.class, ChannelPromise.class).isAnnotationPresent(Skip.class)) {
151                 flags |= MASK_CLOSE;
152             }
153             if (handlerType.getMethod(
154                     "read", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
155                 flags |= MASK_READ;
156             }
157             if (handlerType.getMethod(
158                     "write", ChannelHandlerContext.class,
159                     Object.class, ChannelPromise.class).isAnnotationPresent(Skip.class)) {
160                 flags |= MASK_WRITE;
161 
162                 // flush() is skipped only when write() is also skipped to avoid the situation where
163                 // flush() is handled by the event loop before write() in staged execution.
164                 if (handlerType.getMethod(
165                         "flush", ChannelHandlerContext.class).isAnnotationPresent(Skip.class)) {
166                     flags |= MASK_FLUSH;
167                 }
168             }
169         } catch (Exception e) {
170             // Should never reach here.
171             PlatformDependent.throwException(e);
172         }
173 
174         return flags;
175     }

199     DefaultChannelHandlerContext(
200             DefaultChannelPipeline pipeline, ChannelHandlerInvoker invoker, String name, ChannelHandler handler) {
201 
202         if (name == null) {
203             throw new NullPointerException("name");
204         }
205         if (handler == null) {
206             throw new NullPointerException("handler");
207         }
208 
209         channel = pipeline.channel;
210         this.pipeline = pipeline;
211         this.name = name;
212         this.handler = handler;
213 
214         skipFlags = skipFlags(handler);
215 
216         if (invoker == null) {
217             this.invoker = channel.unsafe().invoker();
218         } else {
219             this.invoker = invoker;
220         }
221     }

223     /** Invocation initiated by {@link DefaultChannelPipeline#teardownAll()}}. */
224     void teardown() {
225         EventExecutor executor = executor();
226         if (executor.inEventLoop()) {
227             teardown0();
228         } else {
229             executor.execute(new Runnable() {
230                 @Override
231                 public void run() {
232                     teardown0();
233                 }
234             });
235         }
236     }
237 
238     private void teardown0() {
239         DefaultChannelHandlerContext prev = this.prev;
240         if (prev != null) {
241             synchronized (pipeline) {
242                 pipeline.remove0(this);
243             }
244             prev.teardown();
245         }
246     }

471     private DefaultChannelHandlerContext findContextInbound(int mask) {
472         DefaultChannelHandlerContext ctx = this;
473         do {
474             ctx = ctx.next;
475         } while ((ctx.skipFlags & mask) != 0);
476         return ctx;
477     }

479     private DefaultChannelHandlerContext findContextOutbound(int mask) {
480         DefaultChannelHandlerContext ctx = this;
481         do {
482             ctx = ctx.prev;
483         } while ((ctx.skipFlags & mask) != 0);
484         return ctx;
485     }
