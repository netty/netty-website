95          NONE, // no renegotiation
96          CLIENT_INITIATED, // renegotiation from client

100     protected static class Renegotiation {
101         static final Renegotiation NONE = new Renegotiation(RenegotiationType.NONE, null);
102 
103         final RenegotiationType type;
104         final String cipherSuite;
105 
106         Renegotiation(RenegotiationType type, String cipherSuite) {
107             this.type = type;
108             this.cipherSuite = cipherSuite;
109         }
110 
111         @Override
112         public String toString() {
113             if (type == RenegotiationType.NONE) {
114                 return "NONE";
115             }
116 
117             return type + "(" + cipherSuite + ')';
118         }

121     public static Collection<Object[]> data() throws Exception {
122         List<SslContext> serverContexts = new ArrayList<>();
123         serverContexts.add(SslContextBuilder.forServer(CERT_FILE, KEY_FILE)
124                                             .sslProvider(SslProvider.JDK)
125                                             // As we test renegotiation we should use a protocol that support it.
126                                             .protocols("TLSv1.2")
127                                             .build());
128 
129         List<SslContext> clientContexts = new ArrayList<>();
130         clientContexts.add(SslContextBuilder.forClient()
131                                             .sslProvider(SslProvider.JDK)
132                                             .trustManager(CERT_FILE)
133                                             // As we test renegotiation we should use a protocol that support it.
134                                             .protocols("TLSv1.2")
135                                             .build());
136 
137         boolean hasOpenSsl = OpenSsl.isAvailable();
138         if (hasOpenSsl) {
139             serverContexts.add(SslContextBuilder.forServer(CERT_FILE, KEY_FILE)
140                                                 .sslProvider(SslProvider.OPENSSL)
141                                                 // As we test renegotiation we should use a protocol that support it.
142                                                 .protocols("TLSv1.2")
143                                                 .build());
144             clientContexts.add(SslContextBuilder.forClient()
145                                                 .sslProvider(SslProvider.OPENSSL)
146                                                 .trustManager(CERT_FILE)
147                                                 // As we test renegotiation we should use a protocol that support it.
148                                                 .protocols("TLSv1.2")
149                                                 .build());
150         } else {
151             logger.warn("OpenSSL is unavailable and thus will not be tested.", OpenSsl.unavailabilityCause());
152         }
153 
154         List<Object[]> params = new ArrayList<>();
155         for (SslContext sc: serverContexts) {
156             for (SslContext cc: clientContexts) {
157                 for (RenegotiationType rt: RenegotiationType.values()) {
158                     if (rt != RenegotiationType.NONE &&
159                         (sc instanceof OpenSslContext || cc instanceof OpenSslContext)) {
160                         // TODO: OpenSslEngine does not support renegotiation yet.
161                         continue;
162                     }
163 
164                     final Renegotiation r;
165                     switch (rt) {
166                         case NONE:
167                             r = Renegotiation.NONE;
168                             break;
169                         case SERVER_INITIATED:
170                             r = new Renegotiation(rt, sc.cipherSuites().get(sc.cipherSuites().size() - 1));
171                             break;
172                         case CLIENT_INITIATED:
173                             r = new Renegotiation(rt, cc.cipherSuites().get(cc.cipherSuites().size() - 1));
174                             break;
175                         default:
176                             throw new Error();
177                     }
178 
179                     for (int i = 0; i < 32; i++) {
180                         params.add(new Object[] {
181                                 sc, cc, r,
182                                 (i & 16) != 0, (i & 8) != 0, (i & 4) != 0, (i & 2) != 0, (i & 1) != 0 });
183                     }
184                 }
185             }
186         }
187 
188         return params;
189     }

249     public void testSslEcho(ServerBootstrap sb, Bootstrap cb) throws Throwable {
250         final ExecutorService delegatedTaskExecutor = Executors.newCachedThreadPool();
251         reset();
252 
253         sb.childOption(ChannelOption.AUTO_READ, autoRead);
254         cb.option(ChannelOption.AUTO_READ, autoRead);
255 
256         sb.childHandler(new ChannelInitializer<Channel>() {
257             @Override
258             public void initChannel(Channel sch) {
259                 serverChannel = sch;
260 
261                 if (serverUsesDelegatedTaskExecutor) {
262                     SSLEngine sse = serverCtx.newEngine(sch.alloc());
263                     serverSslHandler = new SslHandler(sse, delegatedTaskExecutor);
264                 } else {
265                     serverSslHandler = serverCtx.newHandler(sch.alloc());
266                 }
267                 serverSslHandler.setHandshakeTimeoutMillis(0);
268 
269                 sch.pipeline().addLast("ssl", serverSslHandler);
270                 if (useChunkedWriteHandler) {
271                     sch.pipeline().addLast(new ChunkedWriteHandler());
272                 }
273                 sch.pipeline().addLast("serverHandler", serverHandler);
274             }
275         });
276 
277         final CountDownLatch clientHandshakeEventLatch = new CountDownLatch(1);
278         cb.handler(new ChannelInitializer<Channel>() {
279             @Override
280             public void initChannel(Channel sch) {
281                 clientChannel = sch;
282 
283                 if (clientUsesDelegatedTaskExecutor) {
284                     SSLEngine cse = clientCtx.newEngine(sch.alloc());
285                     clientSslHandler = new SslHandler(cse, delegatedTaskExecutor);
286                 } else {
287                     clientSslHandler = clientCtx.newHandler(sch.alloc());
288                 }
289                 clientSslHandler.setHandshakeTimeoutMillis(0);
290 
291                 sch.pipeline().addLast("ssl", clientSslHandler);
292                 if (useChunkedWriteHandler) {
293                     sch.pipeline().addLast(new ChunkedWriteHandler());
294                 }
295                 sch.pipeline().addLast("clientHandler", clientHandler);
296                 sch.pipeline().addLast(new ChannelHandler() {
297                     @Override
298                     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) {
299                         if (evt instanceof SslHandshakeCompletionEvent) {
300                             clientHandshakeEventLatch.countDown();
301                         }
302                         ctx.fireUserEventTriggered(evt);
303                     }
304                 });
305             }
306         });
307 
308         final Channel sc = sb.bind().get();
309         cb.connect(sc.localAddress()).sync();
310 
311         final Future<Channel> clientHandshakeFuture = clientSslHandler.handshakeFuture();
312 
313         // Wait for the handshake to complete before we flush anything. SslHandler should flush non-application data.
314         clientHandshakeFuture.sync();
315         clientHandshakeEventLatch.await();
316 
317         clientChannel.writeAndFlush(Unpooled.wrappedBuffer(data, 0, FIRST_MESSAGE_SIZE));
318         clientSendCounter.set(FIRST_MESSAGE_SIZE);
319 
320         boolean needsRenegotiation = renegotiation.type == RenegotiationType.CLIENT_INITIATED;
321         Future<Channel> renegoFuture = null;
322         while (clientSendCounter.get() < data.length) {
323             int clientSendCounterVal = clientSendCounter.get();
324             int length = Math.min(random.nextInt(1024 * 64), data.length - clientSendCounterVal);
325             ByteBuf buf = Unpooled.wrappedBuffer(data, clientSendCounterVal, length);
326             if (useCompositeByteBuf) {
327                 buf = Unpooled.compositeBuffer().addComponent(true, buf);
328             }
329 
330             Future<Void> future = clientChannel.writeAndFlush(buf);
331             clientSendCounter.set(clientSendCounterVal += length);
332             future.sync();
333 
334             if (needsRenegotiation && clientSendCounterVal >= data.length / 2) {
335                 needsRenegotiation = false;
336                 clientSslHandler.engine().setEnabledCipherSuites(new String[] { renegotiation.cipherSuite });
337                 renegoFuture = clientSslHandler.renegotiate();
338                 logStats("CLIENT RENEGOTIATES");
339                 assertThat(renegoFuture, is(not(sameInstance(clientHandshakeFuture))));
340             }
341         }
342 
343         // Ensure all data has been exchanged.
344         while (clientRecvCounter.get() < data.length) {
345             if (serverException.get() != null) {
346                 break;
347             }
348             if (serverException.get() != null) {
349                 break;
350             }
351 
352             try {
353                 Thread.sleep(50);
354             } catch (InterruptedException e) {
355                 // Ignore.
356             }
357         }
358 
359         while (serverRecvCounter.get() < data.length) {
360             if (serverException.get() != null) {
361                 break;
362             }
363             if (clientException.get() != null) {
364                 break;
365             }
366 
367             try {
368                 Thread.sleep(50);
369             } catch (InterruptedException e) {
370                 // Ignore.
371             }
372         }
373 
374         // Wait until renegotiation is done.
375         if (renegoFuture != null) {
376             renegoFuture.sync();
377         }
378         if (serverHandler.renegoFuture != null) {
379             serverHandler.renegoFuture.sync();
380         }
381 
382         serverChannel.close().awaitUninterruptibly();
383         clientChannel.close().awaitUninterruptibly();
384         sc.close().awaitUninterruptibly();
385         delegatedTaskExecutor.shutdown();
386 
387         if (serverException.get() != null && !(serverException.get() instanceof IOException)) {
388             throw serverException.get();
389         }
390         if (clientException.get() != null && !(clientException.get() instanceof IOException)) {
391             throw clientException.get();
392         }
393         if (serverException.get() != null) {
394             throw serverException.get();
395         }
396         if (clientException.get() != null) {
397             throw clientException.get();
398         }
399 
400         // When renegotiation is done, at least the initiating side should be notified.
401         try {
402             switch (renegotiation.type) {
403             case SERVER_INITIATED:
404                 assertThat(serverSslHandler.engine().getSession().getCipherSuite(), is(renegotiation.cipherSuite));
405                 assertThat(serverNegoCounter.get(), is(2));
406                 assertThat(clientNegoCounter.get(), anyOf(is(1), is(2)));
407                 break;
408             case CLIENT_INITIATED:
409                 assertThat(serverNegoCounter.get(), anyOf(is(1), is(2)));
410                 assertThat(clientSslHandler.engine().getSession().getCipherSuite(), is(renegotiation.cipherSuite));
411                 assertThat(clientNegoCounter.get(), is(2));
412                 break;
413             case NONE:
414                 assertThat(serverNegoCounter.get(), is(1));
415                 assertThat(clientNegoCounter.get(), is(1));
416             }
417         } finally {
418             logStats("STATS");
419         }
420     }

422     private void reset() {
423         clientException.set(null);
424         serverException.set(null);
425 
426         clientSendCounter.set(0);
427         clientRecvCounter.set(0);
428         serverRecvCounter.set(0);
429 
430         clientNegoCounter.set(0);
431         serverNegoCounter.set(0);
432 
433         clientChannel = null;
434         serverChannel = null;
435 
436         clientSslHandler = null;
437         serverSslHandler = null;
438     }

440     void logStats(String message) {
441         logger.debug(
442                 "{}:\n" +
443                 "\tclient { sent: {}, rcvd: {}, nego: {}, cipher: {} },\n" +
444                 "\tserver { rcvd: {}, nego: {}, cipher: {} }",
445                 message,
446                 clientSendCounter, clientRecvCounter, clientNegoCounter,
447                 clientSslHandler.engine().getSession().getCipherSuite(),
448                 serverRecvCounter, serverNegoCounter,
449                 serverSslHandler.engine().getSession().getCipherSuite());
450     }

452     @Sharable
453     private abstract class EchoHandler extends SimpleChannelInboundHandler<ByteBuf> {
454 
455         protected final AtomicInteger recvCounter;
456         protected final AtomicInteger negoCounter;
457         protected final AtomicReference<Throwable> exception;
458 
459         EchoHandler(
460                 AtomicInteger recvCounter, AtomicInteger negoCounter,
461                 AtomicReference<Throwable> exception) {
462 
463             this.recvCounter = recvCounter;
464             this.negoCounter = negoCounter;
465             this.exception = exception;
466         }
467 
468         @Override
469         public final void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
470             // We intentionally do not ctx.flush() here because we want to verify the SslHandler correctly flushing
471             // non-application and previously flushed writes internally.
472             if (!autoRead) {
473                 ctx.read();
474             }
475             ctx.fireChannelReadComplete();
476         }
477 
478         @Override
479         public final void userEventTriggered(ChannelHandlerContext ctx, Object evt) {
480             if (evt instanceof SslHandshakeCompletionEvent) {
481                 SslHandshakeCompletionEvent handshakeEvt = (SslHandshakeCompletionEvent) evt;
482                 if (handshakeEvt.cause() != null) {
483                     logger.warn("Handshake failed:", handshakeEvt.cause());
484                 }
485                 assertSame(SslHandshakeCompletionEvent.SUCCESS, evt);
486                 negoCounter.incrementAndGet();
487                 logStats("HANDSHAKEN");
488             }
489             ctx.fireUserEventTriggered(evt);
490         }
491 
492         @Override
493         public final void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
494             if (logger.isWarnEnabled()) {
495                 logger.warn("Unexpected exception from the client side:", cause);
496             }
497 
498             exception.compareAndSet(null, cause);
499             ctx.close();
500         }
501     }

503     private class EchoClientHandler extends EchoHandler {
504 
505         EchoClientHandler(
506                 AtomicInteger recvCounter, AtomicInteger negoCounter,
507                 AtomicReference<Throwable> exception) {
508 
509             super(recvCounter, negoCounter, exception);
510         }
511 
512         @Override
513         public void handlerAdded(final ChannelHandlerContext ctx) {
514             if (!autoRead) {
515                 ctx.pipeline().get(SslHandler.class).handshakeFuture().addListener(ctx, (c, f) -> c.read());
516             }
517         }
518 
519         @Override
520         public void messageReceived(ChannelHandlerContext ctx, ByteBuf in) throws Exception {
521             byte[] actual = new byte[in.readableBytes()];
522             in.readBytes(actual);
523 
524             int lastIdx = recvCounter.get();
525             for (int i = 0; i < actual.length; i ++) {
526                 assertEquals(data[i + lastIdx], actual[i]);
527             }
528 
529             recvCounter.addAndGet(actual.length);
530         }
531     }

533     private class EchoServerHandler extends EchoHandler {
534         volatile Future<Channel> renegoFuture;
535 
536         EchoServerHandler(
537                 AtomicInteger recvCounter, AtomicInteger negoCounter,
538                 AtomicReference<Throwable> exception) {
539 
540             super(recvCounter, negoCounter, exception);
541         }
542 
543         @Override
544         public final void channelRegistered(ChannelHandlerContext ctx) {
545             renegoFuture = null;
546         }
547 
548         @Override
549         public void channelActive(final ChannelHandlerContext ctx) throws Exception {
550             if (!autoRead) {
551                 ctx.read();
552             }
553             ctx.fireChannelActive();
554         }
555 
556         @Override
557         public void messageReceived(ChannelHandlerContext ctx, ByteBuf in) throws Exception {
558             byte[] actual = new byte[in.readableBytes()];
559             in.readBytes(actual);
560 
561             int lastIdx = recvCounter.get();
562             for (int i = 0; i < actual.length; i ++) {
563                 assertEquals(data[i + lastIdx], actual[i]);
564             }
565 
566             ByteBuf buf = Unpooled.wrappedBuffer(actual);
567             if (useCompositeByteBuf) {
568                 buf = Unpooled.compositeBuffer().addComponent(true, buf);
569             }
570             ctx.writeAndFlush(buf);
571 
572             recvCounter.addAndGet(actual.length);
573 
574             // Perform server-initiated renegotiation if necessary.
575             if (renegotiation.type == RenegotiationType.SERVER_INITIATED &&
576                 recvCounter.get() > data.length / 2 && renegoFuture == null) {
577 
578                 SslHandler sslHandler = ctx.pipeline().get(SslHandler.class);
579 
580                 Future<Channel> hf = sslHandler.handshakeFuture();
581                 assertThat(hf.isDone(), is(true));
582 
583                 sslHandler.engine().setEnabledCipherSuites(new String[] { renegotiation.cipherSuite });
584                 logStats("SERVER RENEGOTIATES");
585                 renegoFuture = sslHandler.renegotiate();
586                 assertThat(renegoFuture, is(not(sameInstance(hf))));
587                 assertThat(renegoFuture, is(sameInstance(sslHandler.handshakeFuture())));
588                 assertThat(renegoFuture.isDone(), is(false));
589             }
590         }
