58      public void testShutdownOutput(Bootstrap cb) throws Throwable {
59          TestHandler h = new TestHandler();
60          ServerSocket ss = new ServerSocket();
61          Socket s = null;
62          SocketChannel ch = null;
63          try {
64              ss.bind(newSocketAddress());
65              ch = (SocketChannel) cb.handler(h).connect(ss.getLocalSocketAddress()).get();
66              assertTrue(ch.isActive());
67              assertFalse(ch.isOutputShutdown());
68  
69              s = ss.accept();
70              ch.writeAndFlush(Unpooled.wrappedBuffer(new byte[] { 1 })).sync();
71              assertEquals(1, s.getInputStream().read());
72  
73              assertTrue(h.ch.isOpen());
74              assertTrue(h.ch.isActive());
75              assertFalse(h.ch.isInputShutdown());
76              assertFalse(h.ch.isOutputShutdown());
77  
78              // Make the connection half-closed and ensure read() returns -1.
79              ch.shutdownOutput().sync();
80              assertEquals(-1, s.getInputStream().read());
81  
82              assertTrue(h.ch.isOpen());
83              assertTrue(h.ch.isActive());
84              assertFalse(h.ch.isInputShutdown());
85              assertTrue(h.ch.isOutputShutdown());
86  
87              // If half-closed, the peer should be able to write something.
88              s.getOutputStream().write(new byte[] { 1 });
89              assertEquals(1, (int) h.queue.take());
90          } finally {
91              if (s != null) {
92                  s.close();
93              }
94              if (ch != null) {
95                  ch.close();
96              }
97              ss.close();
98          }
99      }

145     public void testWriteAfterShutdownOutputNoWritabilityChange(Bootstrap cb) throws Throwable {
146         final TestHandler h = new TestHandler();
147         ServerSocket ss = new ServerSocket();
148         Socket s = null;
149         SocketChannel ch = null;
150         try {
151             ss.bind(newSocketAddress());
152             cb.option(ChannelOption.WRITE_BUFFER_WATER_MARK, new WriteBufferWaterMark(2, 4));
153             ch = (SocketChannel) cb.handler(h).connect(ss.getLocalSocketAddress()).get();
154             assertTrue(ch.isActive());
155             assertFalse(ch.isOutputShutdown());
156 
157             s = ss.accept();
158 
159             byte[] expectedBytes = { 1, 2, 3, 4, 5, 6 };
160             Future<Void> writeFuture = ch.write(Unpooled.wrappedBuffer(expectedBytes));
161             h.assertWritability(false);
162             ch.flush();
163             writeFuture.sync();
164             h.assertWritability(true);
165             for (byte expectedByte : expectedBytes) {
166                 assertEquals(expectedByte, s.getInputStream().read());
167             }
168 
169             assertTrue(h.ch.isOpen());
170             assertTrue(h.ch.isActive());
171             assertFalse(h.ch.isInputShutdown());
172             assertFalse(h.ch.isOutputShutdown());
173 
174             // Make the connection half-closed and ensure read() returns -1.
175             ch.shutdownOutput().sync();
176             assertEquals(-1, s.getInputStream().read());
177 
178             assertTrue(h.ch.isOpen());
179             assertTrue(h.ch.isActive());
180             assertFalse(h.ch.isInputShutdown());
181             assertTrue(h.ch.isOutputShutdown());
182 
183             try {
184                 // If half-closed, the local endpoint shouldn't be able to write
185                 ch.writeAndFlush(Unpooled.wrappedBuffer(new byte[]{ 2 })).sync();
186                 fail();
187             } catch (Throwable cause) {
188                 checkThrowable(cause.getCause());
189             }
190             assertNull(h.writabilityQueue.poll());
191         } finally {
192             if (s != null) {
193                 s.close();
194             }
195             if (ch != null) {
196                 ch.close();
197             }
198             ss.close();
199         }
200     }

222     private void testShutdownSoLingerNoAssertError0(Bootstrap cb, boolean output) throws Throwable {
223         ServerSocket ss = new ServerSocket();
224         Socket s = null;
225 
226         Channel client = null;
227         try {
228             ss.bind(newSocketAddress());
229             client = cb.option(ChannelOption.SO_LINGER, 1).handler(new ChannelHandler() { })
230                     .connect(ss.getLocalSocketAddress()).get();
231             s = ss.accept();
232 
233             if (output) {
234                 ((SocketChannel) client).shutdownOutput().sync();
235             } else {
236                 ((SocketChannel) client).shutdown().sync();
237             }
238         } finally {
239             if (s != null) {
240                 s.close();
241             }
242             if (client != null) {
243                 client.close();
244             }
245             ss.close();
246         }
247     }
248     private static void checkThrowable(Throwable cause) throws Throwable {
249         // Depending on OIO / NIO both are ok
250         if (!(cause instanceof ClosedChannelException) && !(cause instanceof SocketException)) {
251             throw cause;
252         }

255     private static final class TestHandler extends SimpleChannelInboundHandler<ByteBuf> {
256         volatile SocketChannel ch;
257         final BlockingQueue<Byte> queue = new LinkedBlockingQueue<>();
258         final BlockingDeque<Boolean> writabilityQueue = new LinkedBlockingDeque<>();
259 
260         @Override
261         public void channelWritabilityChanged(ChannelHandlerContext ctx) throws Exception {
262             writabilityQueue.add(ctx.channel().isWritable());
263         }
264 
265         @Override
266         public void channelActive(ChannelHandlerContext ctx) throws Exception {
267             ch = (SocketChannel) ctx.channel();
268         }
269 
270         @Override
271         public void messageReceived(ChannelHandlerContext ctx, ByteBuf msg) throws Exception {
272             queue.offer(msg.readByte());
273         }
274 
275         private void drainWritabilityQueue() throws InterruptedException {
276             while (writabilityQueue.poll(100, TimeUnit.MILLISECONDS) != null) {
277                 // Just drain the queue.
278             }
279         }
280 
281         void assertWritability(boolean isWritable) throws InterruptedException {
282             try {
283                 Boolean writability = writabilityQueue.takeLast();
284                 assertEquals(isWritable, writability);
285                 // TODO(scott): why do we get multiple writability changes here ... race condition?
286                 drainWritabilityQueue();
287             } catch (Throwable c) {
288                 c.printStackTrace();
289             }
290         }
