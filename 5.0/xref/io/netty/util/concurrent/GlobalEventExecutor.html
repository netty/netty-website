35  /**
36   * Single-thread singleton {@link EventExecutor}.  It starts the thread automatically and stops it when there is no
37   * task pending in the task queue for 1 second.  Please note it is not scalable to schedule large number of tasks to
38   * this executor; use a dedicated executor.
39   */
40  public final class GlobalEventExecutor extends AbstractScheduledEventExecutor implements OrderedEventExecutor {
41  
42      private static final InternalLogger logger = InternalLoggerFactory.getInstance(GlobalEventExecutor.class);
43  
44      private static final long SCHEDULE_QUIET_PERIOD_INTERVAL = TimeUnit.SECONDS.toNanos(1);
45  
46      private static final RunnableScheduledFutureAdapter<Void> QUIET_PERIOD_TASK;
47      public static final GlobalEventExecutor INSTANCE;
48  
49      static {
50          INSTANCE = new GlobalEventExecutor();
51          QUIET_PERIOD_TASK = new RunnableScheduledFutureAdapter<>(
52                  INSTANCE, INSTANCE.newPromise(), Executors.callable(() -> {
53                      // NOOP
54                  }, null), deadlineNanos(SCHEDULE_QUIET_PERIOD_INTERVAL), -SCHEDULE_QUIET_PERIOD_INTERVAL);
55  
56          INSTANCE.scheduledTaskQueue().add(QUIET_PERIOD_TASK);
57      }
58  
59      private final BlockingQueue<Runnable> taskQueue = new LinkedBlockingQueue<Runnable>();
60  
61      // because the GlobalEventExecutor is a singleton, tasks submitted to it can come from arbitrary threads and this
62      // can trigger the creation of a thread from arbitrary thread groups; for this reason, the thread factory must not
63      // be sticky about its thread group
64      // visible for testing
65      final ThreadFactory threadFactory;
66      private final TaskRunner taskRunner = new TaskRunner();
67      private final AtomicBoolean started = new AtomicBoolean();
68      volatile Thread thread;
69  
70      private final Future<Void> terminationFuture = DefaultPromise.<Void>newFailedPromise(
71              this, new UnsupportedOperationException()).asFuture();
72  
73      private GlobalEventExecutor() {
74          threadFactory = ThreadExecutorMap.apply(new DefaultThreadFactory(
75                  DefaultThreadFactory.toPoolName(getClass()), false, Thread.NORM_PRIORITY, null), this);
76      }
77  
78      /**
79       * Take the next {@link Runnable} from the task queue and so will block if no task is currently present.
80       *
81       * @return {@code null} if the executor thread has been interrupted or waken up.
82       */
83      private Runnable takeTask() {
84          BlockingQueue<Runnable> taskQueue = this.taskQueue;
85          for (;;) {
86              RunnableScheduledFuture<?> scheduledTask = peekScheduledTask();
87              if (scheduledTask == null) {
88                  Runnable task = null;
89                  try {
90                      task = taskQueue.take();
91                  } catch (InterruptedException e) {
92                      // Ignore
93                  }
94                  return task;
95              } else {
96                  long delayNanos = scheduledTask.delayNanos();
97                  Runnable task = null;
98                  if (delayNanos > 0) {
99                      try {
100                         task = taskQueue.poll(delayNanos, TimeUnit.NANOSECONDS);
101                     } catch (InterruptedException e) {
102                         // Waken up.
103                         return null;
104                     }
105                 }
106                 if (task == null) {
107                     // We need to fetch the scheduled tasks now as otherwise there may be a chance that
108                     // scheduled tasks are never executed if there is always one task in the taskQueue.
109                     // This is for example true for the read task of OIO Transport
110                     // See https://github.com/netty/netty/issues/1614
111                     fetchFromScheduledTaskQueue();
112                     task = taskQueue.poll();
113                 }
114 
115                 if (task != null) {
116                     return task;
117                 }
118             }
119         }
120     }
121 
122     private void fetchFromScheduledTaskQueue() {
123         long nanoTime = AbstractScheduledEventExecutor.nanoTime();
124         Runnable scheduledTask = pollScheduledTask(nanoTime);
125         while (scheduledTask != null) {
126             taskQueue.add(scheduledTask);
127             scheduledTask = pollScheduledTask(nanoTime);
128         }
129     }
130 
131     /**
132      * Return the number of tasks that are pending for processing.
133      */
134     public int pendingTasks() {
135         return taskQueue.size();
136     }
137 
138     /**
139      * Add a task to the task queue, or throws a {@link RejectedExecutionException} if this instance was shutdown
140      * before.
141      */
142     private void addTask(Runnable task) {
143         requireNonNull(task, "task");
144         taskQueue.add(task);
145     }

149         return thread == this.thread;

179         return false;

182     /**
183      * Waits until the worker thread of this executor has no tasks left in its task queue and terminates itself.
184      * Because a new worker thread will be started again when a new task is submitted, this operation is only useful
185      * when you want to ensure that the worker thread is terminated <strong>after</strong> your application is shut
186      * down and there's no chance of submitting a new task afterwards.
187      *
188      * @return {@code true} if and only if the worker thread has been terminated
189      */
190     public boolean awaitInactivity(long timeout, TimeUnit unit) throws InterruptedException {
191         requireNonNull(unit, "unit");
192 
193         final Thread thread = this.thread;
194         if (thread == null) {
195             throw new IllegalStateException("thread was not started");
196         }
197         thread.join(unit.toMillis(timeout));
198         return !thread.isAlive();
199     }
200 
201     @Override
202     public void execute(Runnable task) {
203         requireNonNull(task, "task");
204 
205         addTask(task);
206         if (!inEventLoop()) {
207             startThread();
208         }
209     }
210 
211     private void startThread() {
212         if (started.compareAndSet(false, true)) {
213             final Thread t = threadFactory.newThread(taskRunner);
214             // Set to null to ensure we not create classloader leaks by holds a strong reference to the inherited
215             // classloader.
216             // See:
217             // - https://github.com/netty/netty/issues/7290
218             // - https://bugs.openjdk.java.net/browse/JDK-7008595
219             AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
220                 t.setContextClassLoader(null);
221                 return null;
222             });
223 
224             // Set the thread before starting it as otherwise inEventLoop() may return false and so produce
225             // an assert error.
226             // See https://github.com/netty/netty/issues/4357
227             thread = t;
228             t.start();
229         }
230     }
231 
232     final class TaskRunner implements Runnable {
233         @Override
234         public void run() {
235             for (;;) {
236                 Runnable task = takeTask();
237                 if (task != null) {
238                     try {
239                         task.run();
240                     } catch (Throwable t) {
241                         logger.warn("Unexpected exception from the global event executor: ", t);
242                     }
243 
244                     if (task != QUIET_PERIOD_TASK) {
245                         continue;
246                     }
247                 }
248 
249                 Queue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = scheduledTaskQueue();
250                 // Terminate if there is no task in the queue (except the noop task).
251                 if (taskQueue.isEmpty() && scheduledTaskQueue.size() <= 1) {
252                     // Mark the current thread as stopped.
253                     // The following CAS must always success and must be uncontended,
254                     // because only one thread should be running at the same time.
255                     boolean stopped = started.compareAndSet(true, false);
256                     assert stopped;
257 
258                     // Do not check scheduledTaskQueue because it is not thread-safe and can only be mutated from a
259                     // TaskRunner actively running tasks.
260                     if (taskQueue.isEmpty()) {
261                         // A) No new task was added and thus there's nothing to handle
262                         //    -> safe to terminate because there's nothing left to do
263                         // B) A new thread started and handled all the new tasks.
264                         //    -> safe to terminate the new thread will take care the rest
265                         break;
266                     }
267 
268                     // There are pending tasks added again.
269                     if (!started.compareAndSet(false, true)) {
270                         // startThread() started a new thread and set 'started' to true.
271                         // -> terminate this thread so that the new thread reads from taskQueue exclusively.
272                         break;
273                     }
274 
275                     // New tasks were added, but this worker was faster to set 'started' to true.
276                     // i.e. a new worker thread was not started by startThread().
277                     // -> keep this thread alive to handle the newly added entries.
278                 }
279             }
280         }
281     }
