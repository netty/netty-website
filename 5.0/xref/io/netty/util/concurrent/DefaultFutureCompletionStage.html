213     @Override
214     public <U> FutureCompletionStage<U> thenApplyAsync(Function<? super V, ? extends U> fn, Executor executor) {
215         requireNonNull(fn, "fn");
216         requireNonNull(executor, "executor");
217 
218         Promise<U> promise = executor().newPromise();
219         future.addListener(future -> {
220             Throwable cause = future.cause();
221             if (cause == null) {
222                 V value = future.getNow();
223                 if (executeDirectly(executor)) {
224                     thenApplyAsync0(promise, value, fn);
225                 } else {
226                     safeExecute(executor, () -> thenApplyAsync0(promise, value, fn), promise);
227                 }
228             } else {
229                 promise.setFailure(cause);
230             }
231         });
232         return promise.asFuture().asStage();
233     }

282     @Override
283     public <U, V1> FutureCompletionStage<V1> thenCombineAsync(
284             CompletionStage<? extends U> other, BiFunction<? super V, ? super U, ? extends V1> fn, Executor executor) {
285         requireNonNull(other, "other");
286         requireNonNull(fn, "fn");
287         requireNonNull(executor, "executor");
288 
289         Promise<V1> promise = executor().newPromise();
290         AtomicReference<Object> reference = new AtomicReference<>(Marker.EMPTY);
291 
292         abstract class CombineBiConsumer<T1, T2, T> implements BiConsumer<T, Throwable> {
293             @SuppressWarnings("unchecked")
294             @Override
295             public void accept(T v, Throwable error) {
296                 if (error == null) {
297                     if (!reference.compareAndSet(Marker.EMPTY, v)) {
298                         Object rawValue = reference.get();
299                         if (rawValue == Marker.ERROR) {
300                             return;
301                         }
302                         applyAndNotify0(promise, (T1) v, (T2) rawValue, fn);
303                     }
304                 } else {
305                     if (reference.getAndSet(Marker.ERROR) != Marker.ERROR) {
306                         // Did not fail the promise before, do it now.
307                         promise.setFailure(error);
308                     }
309                 }
310             }
311 
312             abstract void applyAndNotify0(
313                     Promise<V1> promise, T1 value1, T2 value2, BiFunction<? super V, ? super U, ? extends V1> fn);
314         }
315 
316         whenCompleteAsync(new CombineBiConsumer<V, U, V>() {
317             @Override
318             void applyAndNotify0(
319                     Promise<V1> promise, V value1, U value2, BiFunction<? super V, ? super U, ? extends V1> fn) {
320                 applyAndNotify(promise, value1, value2, fn);
321             }
322         }, executor);
323         other.whenCompleteAsync(new CombineBiConsumer<U, V, U>() {
324             @Override
325             void applyAndNotify0(
326                     Promise<V1> promise, U value1, V value2, BiFunction<? super V, ? super U, ? extends V1> fn) {
327                 applyAndNotify(promise, value2, value1, fn);
328             }
329         }, otherExecutor(executor));
330         return promise.asFuture().asStage();
331     }

442             Promise<U> promise, Function<? super V, ? extends CompletionStage<U>> fn, V value) {

566     private abstract static class AtomicBiConsumer<V, U> extends AtomicReference<Object>
567             implements BiConsumer<V, Throwable> {
568         private static final long serialVersionUID = 880039612531973027L;
569 
570         private final Promise<U> promise;
571 
572         AtomicBiConsumer(Promise<U> promise) {
573             super(Marker.EMPTY);
574             this.promise = promise;
575         }
576 
577         @Override
578         public void accept(V v, Throwable error) {
579             if (error == null) {
580                 if (compareAndSet(Marker.EMPTY, v)) {
581                     final U value;
582                     try {
583                         value = apply(v);
584                     } catch (Throwable cause) {
585                         promise.setFailure(cause);
586                         return;
587                     }
588                     promise.setSuccess(value);
589                 }
590             } else if (compareAndSet(Marker.EMPTY, Marker.ERROR)) {
591                 promise.setFailure(error);
592             }
593         }
594 
595         protected abstract U apply(V value);
596     }
