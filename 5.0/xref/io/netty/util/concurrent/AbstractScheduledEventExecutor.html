32  /**
33   * Abstract base class for {@link EventExecutor}s that want to support scheduling.
34   */
35  public abstract class AbstractScheduledEventExecutor extends AbstractEventExecutor {
36      static final long START_TIME = System.nanoTime();
37  
38      private static final Comparator<RunnableScheduledFutureNode<?>> SCHEDULED_FUTURE_TASK_COMPARATOR =
39              Comparable::compareTo;
40      private static final RunnableScheduledFutureNode<?>[]
41              EMPTY_RUNNABLE_SCHEDULED_FUTURE_NODES = new RunnableScheduledFutureNode<?>[0];
42  
43      private PriorityQueue<RunnableScheduledFutureNode<?>> scheduledTaskQueue;
44  
45      protected AbstractScheduledEventExecutor() {
46      }
47  
48      /**
49       * The time elapsed since initialization of this class in nanoseconds. This may return a negative number just like
50       * {@link System#nanoTime()}.
51       */
52      public static long nanoTime() {
53          return System.nanoTime() - START_TIME;
54      }
55  
56      /**
57       * The deadline (in nanoseconds) for a given delay (in nanoseconds).
58       */
59      static long deadlineNanos(long delay) {
60          long deadlineNanos = nanoTime() + delay;
61          // Guard against overflow
62          return deadlineNanos < 0? Long.MAX_VALUE : deadlineNanos;
63      }
64  
65      PriorityQueue<RunnableScheduledFutureNode<?>> scheduledTaskQueue() {
66          if (scheduledTaskQueue == null) {
67              scheduledTaskQueue = new DefaultPriorityQueue<>(
68                      SCHEDULED_FUTURE_TASK_COMPARATOR,
69                      // Use same initial capacity as java.util.PriorityQueue
70                      11);
71          }
72          return scheduledTaskQueue;
73      }
74  
75      private static boolean isNullOrEmpty(Queue<RunnableScheduledFutureNode<?>> queue) {
76          return queue == null || queue.isEmpty();
77      }
78  
79      /**
80       * Cancel all scheduled tasks.
81       * <p>
82       * This method MUST be called only when {@link #inEventLoop()} is {@code true}.
83       */
84      protected final void cancelScheduledTasks() {
85          assert inEventLoop();
86          PriorityQueue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = this.scheduledTaskQueue;
87          if (isNullOrEmpty(scheduledTaskQueue)) {
88              return;
89          }
90  
91          final RunnableScheduledFutureNode<?>[] scheduledTasks =
92                  scheduledTaskQueue.toArray(EMPTY_RUNNABLE_SCHEDULED_FUTURE_NODES);
93  
94          for (RunnableScheduledFutureNode<?> task : scheduledTasks) {
95              task.cancel();
96          }
97  
98          scheduledTaskQueue.clearIgnoringIndexes();
99      }
100 
101     /**
102      * @see #pollScheduledTask(long)
103      */
104     protected final RunnableScheduledFuture<?> pollScheduledTask() {
105         return pollScheduledTask(nanoTime());
106     }
107 
108     /**
109      * Return the {@link Runnable} which is ready to be executed with the given {@code nanoTime}. You should use {@link
110      * #nanoTime()} to retrieve the correct {@code nanoTime}.
111      * <p>
112      * This method MUST be called only when {@link #inEventLoop()} is {@code true}.
113      */
114     protected final RunnableScheduledFuture<?> pollScheduledTask(long nanoTime) {
115         assert inEventLoop();
116 
117         Queue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = this.scheduledTaskQueue;
118         RunnableScheduledFutureNode<?> scheduledTask = scheduledTaskQueue == null? null : scheduledTaskQueue.peek();
119         if (scheduledTask == null) {
120             return null;
121         }
122 
123         if (scheduledTask.deadlineNanos() <= nanoTime) {
124             scheduledTaskQueue.remove();
125             return scheduledTask;
126         }
127         return null;
128     }
129 
130     /**
131      * Return the nanoseconds when the next scheduled task is ready to be run or {@code -1} if no task is scheduled.
132      * <p>
133      * This method MUST be called only when {@link #inEventLoop()} is {@code true}.
134      */
135     protected final long nextScheduledTaskNano() {
136         Queue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = this.scheduledTaskQueue;
137         RunnableScheduledFutureNode<?> scheduledTask = scheduledTaskQueue == null? null : scheduledTaskQueue.peek();
138         if (scheduledTask == null) {
139             return -1;
140         }
141         return Math.max(0, scheduledTask.deadlineNanos() - nanoTime());
142     }
143 
144     final RunnableScheduledFuture<?> peekScheduledTask() {
145         Queue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = this.scheduledTaskQueue;
146         if (scheduledTaskQueue == null) {
147             return null;
148         }
149         return scheduledTaskQueue.peek();
150     }
151 
152     /**
153      * Returns {@code true} if a scheduled task is ready for processing.
154      * <p>
155      * This method MUST be called only when {@link #inEventLoop()} is {@code true}.
156      */
157     protected final boolean hasScheduledTasks() {
158         assert inEventLoop();
159         Queue<RunnableScheduledFutureNode<?>> scheduledTaskQueue = this.scheduledTaskQueue;
160         RunnableScheduledFutureNode<?> scheduledTask = scheduledTaskQueue == null? null : scheduledTaskQueue.peek();
161         return scheduledTask != null && scheduledTask.deadlineNanos() <= nanoTime();
162     }

164     @Override
165     public Future<Void> schedule(Runnable command, long delay, TimeUnit unit) {
166         requireNonNull(command, "command");
167         requireNonNull(unit, "unit");
168         if (delay < 0) {
169             delay = 0;
170         }
171         RunnableScheduledFuture<Void> task = newScheduledTaskFor(
172                 callable(command, null), deadlineNanos(unit.toNanos(delay)), 0);
173         return schedule(task);
174     }
175 
176     @Override
177     public <V> Future<V> schedule(Callable<V> callable, long delay, TimeUnit unit) {
178         requireNonNull(callable, "callable");
179         requireNonNull(unit, "unit");
180         if (delay < 0) {
181             delay = 0;
182         }
183         RunnableScheduledFuture<V> task = newScheduledTaskFor(callable, deadlineNanos(unit.toNanos(delay)), 0);
184         return schedule(task);
185     }
186 
187     @Override
188     public Future<Void> scheduleAtFixedRate(Runnable command, long initialDelay, long period, TimeUnit unit) {
189         requireNonNull(command, "command");
190         requireNonNull(unit, "unit");
191         if (initialDelay < 0) {
192             throw new IllegalArgumentException(
193                     String.format("initialDelay: %d (expected: >= 0)", initialDelay));
194         }
195         if (period <= 0) {
196             throw new IllegalArgumentException(
197                     String.format("period: %d (expected: > 0)", period));
198         }
199 
200         RunnableScheduledFuture<Void> task = newScheduledTaskFor(
201                 callable(command, null), deadlineNanos(unit.toNanos(initialDelay)), unit.toNanos(period));
202         return schedule(task);
203     }
204 
205     @Override
206     public Future<Void> scheduleWithFixedDelay(Runnable command, long initialDelay, long delay, TimeUnit unit) {
207         requireNonNull(command, "command");
208         requireNonNull(unit, "unit");
209         if (initialDelay < 0) {
210             throw new IllegalArgumentException(
211                     String.format("initialDelay: %d (expected: >= 0)", initialDelay));
212         }
213         if (delay <= 0) {
214             throw new IllegalArgumentException(
215                     String.format("delay: %d (expected: > 0)", delay));
216         }
217 
218         RunnableScheduledFuture<Void> task = newScheduledTaskFor(
219                 callable(command, null), deadlineNanos(unit.toNanos(initialDelay)), -unit.toNanos(delay));
220         return schedule(task);
221     }
222 
223     /**
224      * Add the {@link RunnableScheduledFuture} for execution.
225      */
226     protected final <V> Future<V> schedule(final RunnableScheduledFuture<V> task) {
227         if (inEventLoop()) {
228             add0(task);
229         } else {
230             execute(() -> add0(task));
231         }
232         return task;
233     }
234 
235     private <V> void add0(RunnableScheduledFuture<V> task) {
236         final RunnableScheduledFutureNode<V> node;
237         if (task instanceof RunnableScheduledFutureNode) {
238             node = (RunnableScheduledFutureNode<V>) task;
239         } else {
240             node = new DefaultRunnableScheduledFutureNode<V>(task);
241         }
242         scheduledTaskQueue().add(node);
243     }
244 
245     final void removeScheduled(final RunnableScheduledFutureNode<?> task) {
246         if (inEventLoop()) {
247             scheduledTaskQueue().removeTyped(task);
248         } else {
249             execute(() -> removeScheduled(task));
250         }
251     }
252 
253     /**
254      * Returns a new {@link RunnableFuture} build on top of the given {@link Promise} and {@link Callable}.
255      * <p>
256      * This can be used if you want to override {@link #newScheduledTaskFor(Callable, long, long)} and return a
257      * different {@link RunnableFuture}.
258      */
259     protected static <V> RunnableScheduledFuture<V> newRunnableScheduledFuture(
260             AbstractScheduledEventExecutor executor, Promise<V> promise, Callable<V> task,
261             long deadlineNanos, long periodNanos) {
262         return new RunnableScheduledFutureAdapter<V>(executor, promise, task, deadlineNanos, periodNanos);
263     }
264 
265     /**
266      * Returns a {@code RunnableScheduledFuture} for the given values.
267      */
268     protected <V> RunnableScheduledFuture<V> newScheduledTaskFor(
269             Callable<V> callable, long deadlineNanos, long period) {
270         return newRunnableScheduledFuture(this, newPromise(), callable, deadlineNanos, period);
271     }
