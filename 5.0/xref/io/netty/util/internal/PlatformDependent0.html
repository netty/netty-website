710     static boolean equals(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
711         int remainingBytes = length & 7;
712         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
713         final long diff = startPos2 - startPos1;
714         if (length >= 8) {
715             final long end = baseOffset1 + remainingBytes;
716             for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
717                 if (UNSAFE.getLong(bytes1, i) != UNSAFE.getLong(bytes2, i + diff)) {
718                     return false;
719                 }
720             }
721         }
722         if (remainingBytes >= 4) {
723             remainingBytes -= 4;
724             long pos = baseOffset1 + remainingBytes;
725             if (UNSAFE.getInt(bytes1, pos) != UNSAFE.getInt(bytes2, pos + diff)) {
726                 return false;
727             }
728         }
729         final long baseOffset2 = baseOffset1 + diff;
730         if (remainingBytes >= 2) {
731             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2) &&
732                     (remainingBytes == 2 ||
733                     UNSAFE.getByte(bytes1, baseOffset1 + 2) == UNSAFE.getByte(bytes2, baseOffset2 + 2));
734         }
735         return remainingBytes == 0 ||
736                 UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
737     }

739     static int equalsConstantTime(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
740         long result = 0;
741         long remainingBytes = length & 7;
742         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
743         final long end = baseOffset1 + remainingBytes;
744         final long diff = startPos2 - startPos1;
745         for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
746             result |= UNSAFE.getLong(bytes1, i) ^ UNSAFE.getLong(bytes2, i + diff);
747         }
748         if (remainingBytes >= 4) {
749             result |= UNSAFE.getInt(bytes1, baseOffset1) ^ UNSAFE.getInt(bytes2, baseOffset1 + diff);
750             remainingBytes -= 4;
751         }
752         if (remainingBytes >= 2) {
753             long pos = end - remainingBytes;
754             result |= UNSAFE.getChar(bytes1, pos) ^ UNSAFE.getChar(bytes2, pos + diff);
755             remainingBytes -= 2;
756         }
757         if (remainingBytes == 1) {
758             long pos = end - 1;
759             result |= UNSAFE.getByte(bytes1, pos) ^ UNSAFE.getByte(bytes2, pos + diff);
760         }
761         return ConstantTimeUtils.equalsConstantTime(result, 0);
762     }

764     static boolean isZero(byte[] bytes, int startPos, int length) {
765         if (length <= 0) {
766             return true;
767         }
768         final long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
769         int remainingBytes = length & 7;
770         final long end = baseOffset + remainingBytes;
771         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
772             if (UNSAFE.getLong(bytes, i) != 0) {
773                 return false;
774             }
775         }
776 
777         if (remainingBytes >= 4) {
778             remainingBytes -= 4;
779             if (UNSAFE.getInt(bytes, baseOffset + remainingBytes) != 0) {
780                 return false;
781             }
782         }
783         if (remainingBytes >= 2) {
784             return UNSAFE.getChar(bytes, baseOffset) == 0 &&
785                     (remainingBytes == 2 || bytes[startPos + 2] == 0);
786         }
787         return bytes[startPos] == 0;
788     }

790     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
791         int hash = HASH_CODE_ASCII_SEED;
792         long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
793         final int remainingBytes = length & 7;
794         final long end = baseOffset + remainingBytes;
795         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
796             hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
797         }
798         if (remainingBytes == 0) {
799             return hash;
800         }
801         int hcConst = HASH_CODE_C1;
802         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
803             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
804             hcConst = HASH_CODE_C2;
805             baseOffset++;
806         }
807         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
808             hash = hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
809             hcConst = hcConst == HASH_CODE_C1 ? HASH_CODE_C2 : HASH_CODE_C1;
810             baseOffset += 2;
811         }
812         if (remainingBytes >= 4) { // 4, 5, 6, 7
813             return hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
814         }
815         return hash;
816     }

818     static int hashCodeAsciiCompute(long value, int hash) {
819         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
820         // code the same regardless of character case (upper case or lower case hash is the same).
821         return hash * HASH_CODE_C1 +
822                 // Low order int
823                 hashCodeAsciiSanitize((int) value) * HASH_CODE_C2 +
824                 // High order int
825                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32);
826     }
