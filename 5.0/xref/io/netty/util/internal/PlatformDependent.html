126     static {
127         // Here is how the system property is used:
128         //
129         // * <  0  - Don't use cleaner, and inherit max direct memory from java. In this case the
130         //           "practical max direct memory" would be 2 * max memory as defined by the JDK.
131         // * == 0  - Use cleaner, Netty will not enforce max memory, and instead will defer to JDK.
132         // * >  0  - Don't use cleaner. This will limit Netty's total direct memory
133         //           (note: that JDK's direct memory limit is independent of this).
134         long maxDirectMemory = SystemPropertyUtil.getLong("io.netty5.maxDirectMemory", -1);
135 
136         if (maxDirectMemory == 0 || !hasUnsafe() || !PlatformDependent0.hasDirectBufferNoCleanerConstructor()) {
137             USE_DIRECT_BUFFER_NO_CLEANER = false;
138             DIRECT_MEMORY_COUNTER = null;
139         } else {
140             USE_DIRECT_BUFFER_NO_CLEANER = true;
141             if (maxDirectMemory < 0) {
142                 maxDirectMemory = MAX_DIRECT_MEMORY;
143                 if (maxDirectMemory <= 0) {
144                     DIRECT_MEMORY_COUNTER = null;
145                 } else {
146                     DIRECT_MEMORY_COUNTER = new AtomicLong();
147                 }
148             } else {
149                 DIRECT_MEMORY_COUNTER = new AtomicLong();
150             }
151         }
152         logger.debug("-Dio.netty5.maxDirectMemory: {} bytes", maxDirectMemory);
153         DIRECT_MEMORY_LIMIT = maxDirectMemory >= 1 ? maxDirectMemory : MAX_DIRECT_MEMORY;
154 
155         int tryAllocateUninitializedArray =
156                 SystemPropertyUtil.getInt("io.netty5.uninitializedArrayAllocationThreshold", 1024);
157         UNINITIALIZED_ARRAY_ALLOCATION_THRESHOLD = PlatformDependent0.hasAllocateArrayMethod() ?
158                 tryAllocateUninitializedArray : -1;
159         logger.debug("-Dio.netty5.uninitializedArrayAllocationThreshold: {}", UNINITIALIZED_ARRAY_ALLOCATION_THRESHOLD);
160 
161         MAYBE_SUPER_USER = maybeSuperUser0();
162 
163         if (!isAndroid()) {
164             // only direct to method if we are not running on android.
165             // See https://github.com/netty/netty/issues/2604
166             CLEANER = CleanerJava9.isSupported() ? new CleanerJava9() : NOOP;
167         } else {
168             CLEANER = NOOP;
169         }
170 
171         // We should always prefer direct buffers by default if we can use a Cleaner to release direct buffers.
172         DIRECT_BUFFER_PREFERRED = CLEANER != NOOP
173                                   && !SystemPropertyUtil.getBoolean("io.netty5.noPreferDirect", false);
174         if (logger.isDebugEnabled()) {
175             logger.debug("-Dio.netty5.noPreferDirect: {}", !DIRECT_BUFFER_PREFERRED);
176         }
177 
178         /*
179          * We do not want to log this message if unsafe is explicitly disabled. Do not remove the explicit no unsafe
180          * guard.
181          */
182         if (CLEANER == NOOP && !PlatformDependent0.isExplicitNoUnsafe()) {
183             logger.info(
184                     "Your platform does not provide complete low-level API for accessing direct buffers reliably. " +
185                     "Unless explicitly requested, heap buffer will always be preferred to avoid potential system " +
186                     "instability.");
187         }
188 
189         final Set<String> allowedClassifiers = Collections.unmodifiableSet(
190                 new HashSet<String>(Arrays.asList(ALLOWED_LINUX_OS_CLASSIFIERS)));
191         final Set<String> availableClassifiers = new LinkedHashSet<String>();
192         for (final String osReleaseFileName : OS_RELEASE_FILES) {
193             final File file = new File(osReleaseFileName);
194             boolean found = AccessController.doPrivileged((PrivilegedAction<Boolean>) () -> {
195                 try {
196                     if (file.exists()) {
197                         BufferedReader reader = null;
198                         try {
199                             reader = new BufferedReader(
200                                     new InputStreamReader(
201                                             new FileInputStream(file), CharsetUtil.UTF_8));
202 
203                             String line;
204                             while ((line = reader.readLine()) != null) {
205                                 if (line.startsWith(LINUX_ID_PREFIX)) {
206                                     String id = normalizeOsReleaseVariableValue(
207                                             line.substring(LINUX_ID_PREFIX.length()));
208                                     addClassifier(allowedClassifiers, availableClassifiers, id);
209                                 } else if (line.startsWith(LINUX_ID_LIKE_PREFIX)) {
210                                     line = normalizeOsReleaseVariableValue(
211                                             line.substring(LINUX_ID_LIKE_PREFIX.length()));
212                                     addClassifier(allowedClassifiers, availableClassifiers, line.split("[ ]+"));
213                                 }
214                             }
215                         } catch (SecurityException e) {
216                             logger.debug("Unable to read {}", osReleaseFileName, e);
217                         } catch (IOException e) {
218                             logger.debug("Error while reading content of {}", osReleaseFileName, e);
219                         } finally {
220                             if (reader != null) {
221                                 try {
222                                     reader.close();
223                                 } catch (IOException ignored) {
224                                     // Ignore
225                                 }
226                             }
227                         }
228                         // specification states we should only fall back if /etc/os-release does not exist
229                         return true;
230                     }
231                 } catch (SecurityException e) {
232                     logger.debug("Unable to check if {} exists", osReleaseFileName, e);
233                 }
234                 return false;
235             });
236 
237             if (found) {
238                 break;
239             }
240         }
241         LINUX_OS_CLASSIFIERS = Collections.unmodifiableSet(availableClassifiers);
242     }

499     private static long getLongSafe(byte[] bytes, int offset) {
500         if (BIG_ENDIAN_NATIVE_ORDER) {
501             return (long) bytes[offset] << 56 |
502                     ((long) bytes[offset + 1] & 0xff) << 48 |
503                     ((long) bytes[offset + 2] & 0xff) << 40 |
504                     ((long) bytes[offset + 3] & 0xff) << 32 |
505                     ((long) bytes[offset + 4] & 0xff) << 24 |
506                     ((long) bytes[offset + 5] & 0xff) << 16 |
507                     ((long) bytes[offset + 6] & 0xff) <<  8 |
508                     (long) bytes[offset + 7] & 0xff;
509         }
510         return (long) bytes[offset] & 0xff |
511                 ((long) bytes[offset + 1] & 0xff) << 8 |
512                 ((long) bytes[offset + 2] & 0xff) << 16 |
513                 ((long) bytes[offset + 3] & 0xff) << 24 |
514                 ((long) bytes[offset + 4] & 0xff) << 32 |
515                 ((long) bytes[offset + 5] & 0xff) << 40 |
516                 ((long) bytes[offset + 6] & 0xff) << 48 |
517                 (long) bytes[offset + 7] << 56;
518     }

520     private static int getIntSafe(byte[] bytes, int offset) {
521         if (BIG_ENDIAN_NATIVE_ORDER) {
522             return bytes[offset] << 24 |
523                     (bytes[offset + 1] & 0xff) << 16 |
524                     (bytes[offset + 2] & 0xff) << 8 |
525                     bytes[offset + 3] & 0xff;
526         }
527         return bytes[offset] & 0xff |
528                 (bytes[offset + 1] & 0xff) << 8 |
529                 (bytes[offset + 2] & 0xff) << 16 |
530                 bytes[offset + 3] << 24;
531     }

533     private static short getShortSafe(byte[] bytes, int offset) {
534         if (BIG_ENDIAN_NATIVE_ORDER) {
535             return (short) (bytes[offset] << 8 | (bytes[offset + 1] & 0xff));
536         }
537         return (short) (bytes[offset] & 0xff | (bytes[offset + 1] << 8));
538     }

540     /**
541      * Identical to {@link PlatformDependent0#hashCodeAsciiCompute(long, int)} but for {@link CharSequence}.
542      */
543     private static int hashCodeAsciiCompute(CharSequence value, int offset, int hash) {
544         if (BIG_ENDIAN_NATIVE_ORDER) {
545             return hash * HASH_CODE_C1 +
546                     // Low order int
547                     hashCodeAsciiSanitizeInt(value, offset + 4) * HASH_CODE_C2 +
548                     // High order int
549                     hashCodeAsciiSanitizeInt(value, offset);
550         }
551         return hash * HASH_CODE_C1 +
552                 // Low order int
553                 hashCodeAsciiSanitizeInt(value, offset) * HASH_CODE_C2 +
554                 // High order int
555                 hashCodeAsciiSanitizeInt(value, offset + 4);
556     }

558     /**
559      * Identical to {@link PlatformDependent0#hashCodeAsciiSanitize(int)} but for {@link CharSequence}.
560      */
561     private static int hashCodeAsciiSanitizeInt(CharSequence value, int offset) {
562         if (BIG_ENDIAN_NATIVE_ORDER) {
563             // mimic a unsafe.getInt call on a big endian machine
564             return (value.charAt(offset + 3) & 0x1f) |
565                    (value.charAt(offset + 2) & 0x1f) << 8 |
566                    (value.charAt(offset + 1) & 0x1f) << 16 |
567                    (value.charAt(offset) & 0x1f) << 24;
568         }
569         return (value.charAt(offset + 3) & 0x1f) << 24 |
570                (value.charAt(offset + 2) & 0x1f) << 16 |
571                (value.charAt(offset + 1) & 0x1f) << 8 |
572                (value.charAt(offset) & 0x1f);
573     }

575     /**
576      * Identical to {@link PlatformDependent0#hashCodeAsciiSanitize(short)} but for {@link CharSequence}.
577      */
578     private static int hashCodeAsciiSanitizeShort(CharSequence value, int offset) {
579         if (BIG_ENDIAN_NATIVE_ORDER) {
580             // mimic a unsafe.getShort call on a big endian machine
581             return (value.charAt(offset + 1) & 0x1f) |
582                     (value.charAt(offset) & 0x1f) << 8;
583         }
584         return (value.charAt(offset + 1) & 0x1f) << 8 |
585                 (value.charAt(offset) & 0x1f);
586     }

866     /**
867      * Calculate a hash code of a byte array assuming ASCII character encoding.
868      * The resulting hash code will be case insensitive.
869      * <p>
870      * This method assumes that {@code bytes} is equivalent to a {@code byte[]} but just using {@link CharSequence}
871      * for storage. The upper most byte of each {@code char} from {@code bytes} is ignored.
872      * @param bytes The array which contains the data to hash (assumed to be equivalent to a {@code byte[]}).
873      * @return The hash code of {@code bytes} assuming ASCII character encoding.
874      * The resulting hash code will be case insensitive.
875      */
876     public static int hashCodeAscii(CharSequence bytes) {
877         final int length = bytes.length();
878         final int remainingBytes = length & 7;
879         int hash = HASH_CODE_ASCII_SEED;
880         // Benchmarking shows that by just naively looping for inputs 8~31 bytes long we incur a relatively large
881         // performance penalty (only achieve about 60% performance of loop which iterates over each char). So because
882         // of this we take special provisions to unroll the looping for these conditions.
883         if (length >= 32) {
884             for (int i = length - 8; i >= remainingBytes; i -= 8) {
885                 hash = hashCodeAsciiCompute(bytes, i, hash);
886             }
887         } else if (length >= 8) {
888             hash = hashCodeAsciiCompute(bytes, length - 8, hash);
889             if (length >= 16) {
890                 hash = hashCodeAsciiCompute(bytes, length - 16, hash);
891                 if (length >= 24) {
892                     hash = hashCodeAsciiCompute(bytes, length - 24, hash);
893                 }
894             }
895         }
896         if (remainingBytes == 0) {
897             return hash;
898         }
899         int offset = 0;
900         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
901             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitizeByte(bytes.charAt(0));
902             offset = 1;
903         }
904         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
905             hash = hash * (offset == 0 ? HASH_CODE_C1 : HASH_CODE_C2)
906                     + hashCodeAsciiSanitize(hashCodeAsciiSanitizeShort(bytes, offset));
907             offset += 2;
908         }
909         if (remainingBytes >= 4) { // 4, 5, 6, 7
910             return hash * ((offset == 0 | offset == 3) ? HASH_CODE_C1 : HASH_CODE_C2)
911                     + hashCodeAsciiSanitizeInt(bytes, offset);
912         }
913         return hash;
914     }

1111      * <p>
1112      * The computation is not cached, so you probably want to use {@link #maxDirectMemory()} instead.
1113      * <p>
1114      * This will produce debug log output when called.
1115      *
1116      * @return The estimated max direct memory, in bytes.
1117      */
1118     public static long estimateMaxDirectMemory() {
1119         long maxDirectMemory = 0;
1120 
1121         ClassLoader systemClassLoader = null;
1122         try {
1123             systemClassLoader = getSystemClassLoader();
1124 
1125             // When using IBM J9 / Eclipse OpenJ9 we should not use VM.maxDirectMemory() as it not reflects the
1126             // correct value.
1127             // See:
1128             //  - https://github.com/netty/netty/issues/7654
1129             String vmName = SystemPropertyUtil.get("java.vm.name", "").toLowerCase();
1130             if (!vmName.startsWith("ibm j9") &&
1131                     // https://github.com/eclipse/openj9/blob/openj9-0.8.0/runtime/include/vendor_version.h#L53
1132                     !vmName.startsWith("eclipse openj9")) {
1133                 // Try to get from sun.misc.VM.maxDirectMemory() which should be most accurate.
1134                 Class<?> vmClass = Class.forName("sun.misc.VM", true, systemClassLoader);
1135                 Method m = vmClass.getDeclaredMethod("maxDirectMemory");
1136                 maxDirectMemory = ((Number) m.invoke(null)).longValue();
1137             }
1138         } catch (Throwable ignored) {
1139             // Ignore
1140         }
1141 
1142         if (maxDirectMemory > 0) {
1143             return maxDirectMemory;
1144         }
1145 
1146         try {
1147             // Now try to get the JVM option (-XX:MaxDirectMemorySize) and parse it.
1148             // Note that we are using reflection because Android doesn't have these classes.
1149             Class<?> mgmtFactoryClass = Class.forName(
1150                     "java.lang.management.ManagementFactory", true, systemClassLoader);
1151             Class<?> runtimeClass = Class.forName(
1152                     "java.lang.management.RuntimeMXBean", true, systemClassLoader);
1153 
1154             Object runtime = mgmtFactoryClass.getDeclaredMethod("getRuntimeMXBean").invoke(null);
1155 
1156             @SuppressWarnings("unchecked")
1157             List<String> vmArgs = (List<String>) runtimeClass.getDeclaredMethod("getInputArguments").invoke(runtime);
1158             for (int i = vmArgs.size() - 1; i >= 0; i --) {
1159                 Matcher m = MAX_DIRECT_MEMORY_SIZE_ARG_PATTERN.matcher(vmArgs.get(i));
1160                 if (!m.matches()) {
1161                     continue;
1162                 }
1163 
1164                 maxDirectMemory = Long.parseLong(m.group(1));
1165                 switch (m.group(2).charAt(0)) {
1166                     case 'k': case 'K':
1167                         maxDirectMemory *= 1024;
1168                         break;
1169                     case 'm': case 'M':
1170                         maxDirectMemory *= 1024 * 1024;
1171                         break;
1172                     case 'g': case 'G':
1173                         maxDirectMemory *= 1024 * 1024 * 1024;
1174                         break;
1175                     default:
1176                         break;
1177                 }
1178                 break;
1179             }
1180         } catch (Throwable ignored) {
1181             // Ignore
1182         }
1183 
1184         if (maxDirectMemory <= 0) {
1185             maxDirectMemory = Runtime.getRuntime().maxMemory();
1186             logger.debug("maxDirectMemory: {} bytes (maybe)", maxDirectMemory);
1187         } else {
1188             logger.debug("maxDirectMemory: {} bytes", maxDirectMemory);
1189         }
1190 
1191         return maxDirectMemory;
1192     }

1306         String vm = SystemPropertyUtil.get("java.vm.name", "").toLowerCase(Locale.US);

1308         Matcher m = bitPattern.matcher(vm);
