16  package io.netty5.buffer.api;
17  
18  import io.netty5.buffer.api.internal.SendFromSupplier;
19  import io.netty5.util.SafeCloseable;
20  
21  import java.util.function.Function;
22  import java.util.function.Supplier;
23  
24  /**
25   * A temporary holder of a {@link Resource}, used for transferring the ownership of the
26   * resource from one thread to another.
27   * <p>
28   * Prior to the {@code Send} being created, the originating resource is invalidated, to prevent access while it is being
29   * sent. This means it cannot be accessed, closed, or disposed of, while it is in-flight. Once the resource is
30   * {@linkplain #receive() received}, the new ownership is established.
31   * <p>
32   * Care must be taken to ensure that the resource is always received by some thread.
33   * Failure to do so can result in a resource leak.
34   *
35   * @param <T>
36   */
37  public interface Send<T extends Resource<T>> extends SafeCloseable {
38      /**
39       * Construct a {@link Send} based on the given {@link Supplier}. The supplier will be called only once, in the
40       * receiving thread.
41       *
42       * @param concreteObjectType The concrete type of the object being sent. Specifically, the object returned from the
43       *                           {@link Supplier#get()} method must be an instance of this class.
44       * @param supplier           The supplier of the object being sent, which will be called when the object is ready to
45       *                           be received.
46       * @param <T>                The type of object being sent.
47       * @return A {@link Send} which will deliver an object of the given type, from the supplier.
48       */
49      static <T extends Resource<T>> Send<T> sending(Class<T> concreteObjectType, Supplier<? extends T> supplier) {
50          return new SendFromSupplier<>(concreteObjectType, supplier);
51      }
52  
53      /**
54       * Determine if the given candidate object is an instance of a {@link Send} from which an object of the given type
55       * can be received.
56       *
57       * @param type      The type of object we wish to receive.
58       * @param candidate The candidate object that might be a {@link Send} of an object of the given type.
59       * @return {@code true} if the candidate object is a {@link Send} that would deliver an object of the given type,
60       * otherwise {@code false}.
61       */
62      static boolean isSendOf(Class<?> type, Object candidate) {
63          return candidate instanceof Send && ((Send<?>) candidate).referentIsInstanceOf(type);
64      }
65  
66      /**
67       * Receive the {@link Resource} instance being sent, and bind its ownership to the calling thread.
68       * The invalidation of the sent resource in the sending thread happens-before the return of this method.
69       * <p>
70       * This method can only be called once, and will throw otherwise.
71       *
72       * @return The sent resource instance, never {@code null}.
73       * @throws IllegalStateException If this method is called more than once.
74       */
75      T receive();
76  
77      /**
78       * Apply a mapping function to the object being sent. The mapping will occur when the object is received.
79       *
80       * @param type   The result type of the mapping function.
81       * @param mapper The mapping function to apply to the object being sent.
82       * @param <R>    The result type of the mapping function.
83       * @return A new {@link Send} instance that will deliver an object that is the result of the mapping.
84       */
85      default <R extends Resource<R>> Send<R> map(Class<R> type, Function<T, R> mapper) {
86          return sending(type, () -> mapper.apply(receive()));
87      }
88  
89      /**
90       * Discard this {@link Send} and the object it contains.
91       * This has no effect if the send-object has already been received.
92       */
93      @Override
94      void close();
95  
96      /**
97       * Determine if the object received from this {@code Send} is an instance of the given class.
98       *
99       * @param cls The type to check.
100      * @return {@code true} if the object received from this {@code Send} can be assigned fields or variables of the
101      * given type, otherwise false.
102      */
103     boolean referentIsInstanceOf(Class<?> cls);
104 }
