203     private void copyIntoArray(int srcPos, byte[] dest, int destPos, int length) {
204         long destOffset = PlatformDependent.byteArrayBaseOffset();
205         try {
206             PlatformDependent.copyMemory(base, address + srcPos, dest, destOffset + destPos, length);
207         } finally {
208             Reference.reachabilityFence(memory);
209             Reference.reachabilityFence(dest);
210         }

213     @Override
214     public void copyInto(int srcPos, ByteBuffer dest, int destPos, int length) {
215         checkCopyIntoArgs(srcPos, length, destPos, dest.capacity());
216         if (dest.isReadOnly()) {
217             throw new ReadOnlyBufferException();
218         }
219         if (dest.hasArray()) {
220             copyIntoArray(srcPos, dest.array(), dest.arrayOffset() + destPos, length);
221         } else {
222             assert dest.isDirect();
223             long destAddr = PlatformDependent.directBufferAddress(dest);
224             try {
225                 PlatformDependent.copyMemory(base, address + srcPos, null, destAddr + destPos, length);
226             } finally {
227                 Reference.reachabilityFence(memory);
228                 Reference.reachabilityFence(dest);
229             }
230         }
231     }

233     private void checkCopyIntoArgs(int srcPos, int length, int destPos, int destLength) {
234         if (rsize == CLOSED_SIZE) {
235             throw bufferIsClosed(this);
236         }
237         if (srcPos < 0) {
238             throw new IllegalArgumentException("The srcPos cannot be negative: " + srcPos + '.');
239         }
240         checkLength(length);
241         if (rsize < srcPos + length) {
242             throw new IllegalArgumentException("The srcPos + length is beyond the end of the buffer: " +
243                     "srcPos = " + srcPos + ", length = " + length + '.');
244         }
245         if (destPos < 0) {
246             throw new IllegalArgumentException("The destPos cannot be negative: " + destPos + '.');
247         }
248         if (destLength < destPos + length) {
249             throw new IllegalArgumentException("The destPos + length is beyond the end of the destination: " +
250                     "destPos = " + destPos + ", length = " + length + '.');
251         }
252     }

254     @Override
255     public void copyInto(int srcPos, Buffer dest, int destPos, int length) {
256         if (!dest.isAccessible()) {
257             throw bufferIsClosed(dest);
258         }
259         checkCopyIntoArgs(srcPos, length, destPos, dest.capacity());
260         if (dest.readOnly()) {
261             throw bufferIsReadOnly(this);
262         }
263         try {
264             if (dest instanceof UnsafeBuffer) {
265                 UnsafeBuffer destUnsafe = (UnsafeBuffer) dest;
266                 long nativeAddress = destUnsafe.nativeAddress();
267                 if (nativeAddress != 0) {
268                     PlatformDependent.copyMemory(base, address + srcPos, null, nativeAddress + destPos, length);
269                 } else {
270                     PlatformDependent.copyMemory(
271                             base, address + srcPos, destUnsafe.base, destUnsafe.address + destPos, length);
272                 }
273             } else {
274                 Statics.copyToViaReverseLoop(this, srcPos, dest, destPos, length);
275             }
276         } finally {
277             Reference.reachabilityFence(memory);
278             Reference.reachabilityFence(dest);
279         }
280     }

282     @Override
283     public int transferTo(WritableByteChannel channel, int length) throws IOException {
284         if (!isAccessible()) {
285             throw bufferIsClosed(this);
286         }
287         length = Math.min(readableBytes(), length);
288         if (length == 0) {
289             return 0;
290         }
291         checkGet(readerOffset(), length);
292         int bytesWritten = channel.write(readableBuffer().limit(length));
293         skipReadable(bytesWritten);
294         return bytesWritten;
295     }

297     @Override
298     public int transferFrom(ReadableByteChannel channel, int length) throws IOException {
299         if (!isAccessible()) {
300             throw bufferIsClosed(this);
301         }
302         if (readOnly()) {
303             throw bufferIsReadOnly(this);
304         }
305         length = Math.min(writableBytes(), length);
306         if (length == 0) {
307             return 0;
308         }
309         checkSet(writerOffset(), length);
310         int bytesRead = channel.read(writableBuffer().limit(length));
311         if (bytesRead != -1) {
312             skipWritable(bytesRead);
313         }
314         return bytesRead;
315     }

317     /**
318      * Most deployment platforms support unaligned access, and are little-endian.
319      * This allows us to micro-optimise for this common case.
320      */
321     private static final boolean BYTES_BEFORE_USE_LITTLE_ENDIAN =
322             PlatformDependent.isUnaligned() && ByteOrder.nativeOrder() != ByteOrder.BIG_ENDIAN;
323 
324     @Override
325     public int bytesBefore(byte needle) {
326         // For the details of this algorithm, see Hacker's Delight, Chapter 6, Searching Words.
327         // Richard Startin also describes this on his blog: https://richardstartin.github.io/posts/finding-bytes
328         if (!isAccessible()) {
329             throw bufferIsClosed(this);
330         }
331         try {
332             int offset = roff;
333             final int length = woff - roff;
334             final int end = woff;
335             final long addr = address;
336 
337             if (length > 7) {
338                 final long pattern = (needle & 0xFFL) * 0x101010101010101L;
339                 for (final int longEnd = offset + (length >>> 3) * Long.BYTES;
340                      offset < longEnd;
341                      offset += Long.BYTES) {
342                     final long word = BYTES_BEFORE_USE_LITTLE_ENDIAN?
343                             PlatformDependent.getLong(base, addr + offset) :
344                             loadLong(addr + offset);
345 
346                     final long input = word ^ pattern;
347                     final long tmp =
348                             ~((input & 0x7F7F7F7F7F7F7F7FL) + 0x7F7F7F7F7F7F7F7FL | input | 0x7F7F7F7F7F7F7F7FL);
349                     final int binaryPosition = BYTES_BEFORE_USE_LITTLE_ENDIAN?
350                             Long.numberOfTrailingZeros(tmp) :
351                             Long.numberOfLeadingZeros(tmp);
352 
353                     if (binaryPosition < Long.SIZE) {
354                         return offset + (binaryPosition >>> 3) - roff;
355                     }
356                 }
357             }
358             for (; offset < end; offset++) {
359                 if (loadByte(addr + offset) == needle) {
360                     return offset - roff;
361                 }
362             }
363 
364             return -1;
365         } finally {
366             Reference.reachabilityFence(memory);
367         }
368     }

379         }

387         }

391     @Override
392     public ByteCursor openReverseCursor(int fromOffset, int length) {
393         if (rsize == CLOSED_SIZE) {
394             throw bufferIsClosed(this);
395         }
396         if (fromOffset < 0) {
397             throw new IllegalArgumentException("The fromOffset cannot be negative: " + fromOffset + '.');
398         }
399         checkLength(length);
400         if (capacity() <= fromOffset) {
401             throw new IllegalArgumentException("The fromOffset is beyond the end of the buffer: " + fromOffset + '.');
402         }
403         if (fromOffset - length < -1) {
404             throw new IllegalArgumentException("The fromOffset - length would underflow the buffer: " +
405                     "fromOffset = " + fromOffset + ", length = " + length + '.');
406         }
407         return new ReverseUnsafeByteCursor(memory, base, address, fromOffset, length);

410     @Override
411     public Buffer ensureWritable(int size, int minimumGrowth, boolean allowCompaction) {
412         if (!isAccessible()) {
413             throw bufferIsClosed(this);
414         }
415         if (!isOwned()) {
416             throw attachTrace(new IllegalStateException(
417                     "Buffer is not owned. Only owned buffers can call ensureWritable."));
418         }
419         if (size < 0) {
420             throw new IllegalArgumentException("Cannot ensure writable for a negative size: " + size + '.');
421         }
422         if (minimumGrowth < 0) {
423             throw new IllegalArgumentException("The minimum growth cannot be negative: " + minimumGrowth + '.');
424         }
425         if (rsize != wsize) {
426             throw bufferIsReadOnly(this);
427         }
428         if (writableBytes() >= size) {
429             // We already have enough space.
430             return this;
431         }
432 
433         if (allowCompaction && writableBytes() + readerOffset() >= size) {
434             // We can solve this with compaction.
435             return compact();
436         }
437 
438         // Allocate a bigger buffer.
439         long newSize = capacity() + (long) Math.max(size - writableBytes(), minimumGrowth);
440         Statics.assertValidBufferSize(newSize);
441         UnsafeBuffer buffer = (UnsafeBuffer) control.getAllocator().allocate((int) newSize);
442 
443         // Copy contents.
444         try {
445             copyInto(0, buffer, 0, capacity());
446         } finally {
447             Reference.reachabilityFence(memory);
448             Reference.reachabilityFence(buffer.memory);
449         }
450 
451         // Release the old memory, and install the new memory:
452         Drop<UnsafeBuffer> drop = buffer.unsafeGetDrop();
453         disconnectDrop(drop);
454         attachNewMemory(buffer.memory, drop);
455         return this;
456     }
457 
458     private Drop<UnsafeBuffer> disconnectDrop(Drop<UnsafeBuffer> newDrop) {
459         var drop = (Drop<UnsafeBuffer>) unsafeGetDrop();
460         int roff = this.roff;
461         int woff = this.woff;
462         drop.drop(this);
463         unsafeSetDrop(newDrop);
464         this.roff = roff;
465         this.woff = woff;
466         return drop;
467     }

469     private void attachNewMemory(UnsafeMemory memory, Drop<UnsafeBuffer> drop) {
470         this.memory = memory;
471         base = memory.base;
472         baseOffset = 0;
473         address = memory.address;
474         rsize = memory.size;
475         wsize = memory.size;
476         drop.attach(this);
477     }

479     @Override
480     public Buffer split(int splitOffset) {
481         if (splitOffset < 0) {
482             throw new IllegalArgumentException("The split offset cannot be negative: " + splitOffset + '.');
483         }
484         if (capacity() < splitOffset) {
485             throw new IllegalArgumentException("The split offset cannot be greater than the buffer capacity, " +
486                     "but the split offset was " + splitOffset + ", and capacity is " + capacity() + '.');
487         }
488         if (!isAccessible()) {
489             throw attachTrace(bufferIsClosed(this));
490         }
491         if (!isOwned()) {
492             throw attachTrace(new IllegalStateException("Cannot split a buffer that is not owned."));
493         }
494         var drop = unsafeGetDrop().fork();
495         var splitBuffer = new UnsafeBuffer(memory, baseOffset, splitOffset, control, drop);
496         drop.attach(splitBuffer);
497         splitBuffer.woff = Math.min(woff, splitOffset);
498         splitBuffer.roff = Math.min(roff, splitOffset);
499         boolean readOnly = readOnly();
500         if (readOnly) {
501             splitBuffer.makeReadOnly();
502         }
503         // Split preserves const-state.
504         rsize -= splitOffset;
505         baseOffset += splitOffset;
506         address += splitOffset;
507         if (!readOnly) {
508             wsize = rsize;
509         }
510         woff = Math.max(woff, splitOffset) - splitOffset;
511         roff = Math.max(roff, splitOffset) - splitOffset;
512         return splitBuffer;
513     }

515     @Override
516     public Buffer compact() {
517         if (!isAccessible()) {
518             throw attachTrace(bufferIsClosed(this));
519         }
520         if (!isOwned()) {
521             throw attachTrace(new IllegalStateException("Buffer must be owned in order to compact."));
522         }
523         if (readOnly()) {
524             throw new BufferReadOnlyException("Buffer must be writable in order to compact, but was read-only.");
525         }
526         if (roff == 0) {
527             return this;
528         }
529         try {
530             PlatformDependent.copyMemory(base, address + roff, base, address, woff - roff);
531         } finally {
532             Reference.reachabilityFence(memory);
533         }
534         woff -= roff;
535         roff = 0;
536         return this;
537     }

1191         var roff = this.roff;

1193         var readOnly = readOnly();
1194         UnsafeMemory memory = this.memory;
1195         AllocatorControl control = this.control;

1197         int rsize = this.rsize;
1198         return new Owned<UnsafeBuffer>() {
1199             @Override

1201                 UnsafeBuffer copy = new UnsafeBuffer(memory, baseOffset, rsize, control, drop);
1202                 copy.roff = roff;

1205                     copy.makeReadOnly();

1489         Object b = base;
