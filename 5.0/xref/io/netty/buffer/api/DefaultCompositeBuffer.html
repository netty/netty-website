42   * The default implementation of the {@link CompositeBuffer} interface.

44  final class DefaultCompositeBuffer extends ResourceSupport<Buffer, DefaultCompositeBuffer> implements CompositeBuffer {
45      /**
46       * The max array size is JVM implementation dependant, but most seem to settle on {@code Integer.MAX_VALUE - 8}.
47       * We set the max composite buffer capacity to the same, since it would otherwise be impossible to create a
48       * non-composite copy of the buffer.
49       */
50      private static final int MAX_CAPACITY = Integer.MAX_VALUE - 8;
51      private static final Drop<DefaultCompositeBuffer> COMPOSITE_DROP = new Drop<>() {
52          @Override
53          public void drop(DefaultCompositeBuffer buf) {
54              RuntimeException re = null;
55              for (Buffer b : buf.bufs) {
56                  try {
57                      b.close();
58                  } catch (RuntimeException e) {
59                      if (re == null) {
60                          re = e;
61                      } else {
62                          re.addSuppressed(e);
63                      }
64                  }
65              }
66          }

70              return this;

73          @Override
74          public void attach(DefaultCompositeBuffer obj) {
75          }

77          @Override
78          public String toString() {
79              return "COMPOSITE_DROP";
80          }

98      @SafeVarargs
99      public static CompositeBuffer compose(BufferAllocator allocator, Send<Buffer>... sends) {
100         Buffer[] bufs = new Buffer[sends.length];
101         RuntimeException ise = null;
102         for (int i = 0; i < sends.length; i++) {
103             if (ise != null) {
104                 try {
105                     sends[i].close();
106                 } catch (Exception closeExc) {
107                     ise.addSuppressed(closeExc);
108                 }
109             } else {
110                 try {
111                     bufs[i] = sends[i].receive();
112                 } catch (RuntimeException e) {
113                     // We catch RuntimeException instead of IllegalStateException to ensure cleanup always happens
114                     // regardless of the exception thrown.
115                     ise = e;
116                     for (int j = 0; j < i; j++) {
117                         try {
118                             bufs[j].close();
119                         } catch (Exception closeExc) {
120                             ise.addSuppressed(closeExc);
121                         }
122                     }
123                 }
124             }
125         }
126         if (ise != null) {
127             throw ise;
128         }
129         return new DefaultCompositeBuffer(allocator, filterExternalBufs(Arrays.stream(bufs)), COMPOSITE_DROP);
130     }

139     private static Buffer[] filterExternalBufs(Stream<Buffer> refs) {
140         // We filter out all zero-capacity buffers because they wouldn't contribute to the composite buffer anyway,
141         // and also, by ensuring that all constituent buffers contribute to the size of the composite buffer,
142         // we make sure that the number of composite buffers will never become greater than the number of bytes in
143         // the composite buffer.
144         // This restriction guarantees that methods like countComponents, forEachReadable and forEachWritable,
145         // will never overflow their component counts.
146         // Allocating a new array unconditionally also prevents external modification of the array.
147         Buffer[] bufs = refs
148                 .filter(DefaultCompositeBuffer::discardEmpty)
149                 .flatMap(DefaultCompositeBuffer::flattenBuffer)
150                 .toArray(Buffer[]::new);
151         // Make sure there are no duplicates among the buffers.
152         Set<Buffer> duplicatesCheck = Collections.newSetFromMap(new IdentityHashMap<>());
153         duplicatesCheck.addAll(Arrays.asList(bufs));
154         if (duplicatesCheck.size() < bufs.length) {
155             IllegalArgumentException iae = new IllegalArgumentException(
156                     "Cannot create composite buffer with duplicate constituent buffer components.");
157             for (Buffer buf : bufs) {
158                 try {
159                     buf.close();
160                 } catch (Exception closeExc) {
161                     iae.addSuppressed(closeExc);
162                 }
163             }
164             throw iae;
165         }
166         return bufs;
167     }

169     private static boolean discardEmpty(Buffer buf) {
170         if (buf.capacity() > 0) {
171             return true;
172         } else {
173             // If we filter a buffer out, then we must make sure to close it since it's ownership was sent to us.
174             buf.close();
175             return false;
176         }
177     }

179     private static Stream<Buffer> flattenBuffer(Buffer buf) {
180         if (CompositeBuffer.isComposite(buf)) {
181             // Extract components so composite buffers always have non-composite components.
182             var composite = (CompositeBuffer) buf;
183             return Stream.of(composite.decomposeBuffer());
184         }
185         return Stream.of(buf);
186     }

188     private DefaultCompositeBuffer(BufferAllocator allocator, Buffer[] bufs, Drop<DefaultCompositeBuffer> drop) {
189         super(drop);
190         try {
191             this.allocator = Objects.requireNonNull(allocator, "BufferAllocator cannot be null.");
192             if (bufs.length > 0) {
193                 boolean targetReadOnly = bufs[0].readOnly();
194                 for (Buffer buf : bufs) {
195                     if (buf.readOnly() != targetReadOnly) {
196                         throw new IllegalArgumentException("Constituent buffers have inconsistent read-only state.");
197                     }
198                 }
199                 readOnly = targetReadOnly;
200             }
201             this.bufs = bufs;
202             computeBufferOffsets();
203             tornBufAccessors = new TornBufferAccessor(this);
204         } catch (Exception e) {
205             // Always close bufs on exception, since we've taken ownership of them at this point.
206             for (Buffer buf : bufs) {
207                 try {
208                     buf.close();
209                 } catch (Exception closeExc) {
210                     e.addSuppressed(closeExc);
211                 }
212             }
213             throw e;
214         }
215     }

217     private void computeBufferOffsets() {
218         int woff = 0;
219         int roff = 0;
220         if (bufs.length > 0) {
221             boolean woffMidpoint = false;
222             for (Buffer buf : bufs) {
223                 if (buf.writableBytes() == 0) {
224                     woff += buf.capacity();
225                 } else if (!woffMidpoint) {
226                     woff += buf.writerOffset();
227                     woffMidpoint = true;
228                 } else if (buf.writerOffset() != 0) {
229                     throw new IllegalArgumentException(
230                             "The given buffers cannot be composed because they leave an unwritten gap: " +
231                             Arrays.toString(bufs) + '.');
232                 }
233             }
234             boolean roffMidpoint = false;
235             for (Buffer buf : bufs) {
236                 if (buf.readableBytes() == 0 && buf.writableBytes() == 0) {
237                     roff += buf.capacity();
238                 } else if (!roffMidpoint) {
239                     roff += buf.readerOffset();
240                     roffMidpoint = true;
241                 } else if (buf.readerOffset() != 0) {
242                     throw new IllegalArgumentException(
243                             "The given buffers cannot be composed because they leave an unread gap: " +
244                             Arrays.toString(bufs) + '.');
245                 }
246             }
247             assert roff <= woff:
248                     "The given buffers place the read offset ahead of the write offset: " + Arrays.toString(bufs) + '.';
249         }
250         // Commit computed offsets, if any
251         this.woff = woff;
252         this.roff = roff;
253 
254         offsets = new int[bufs.length];
255         long cap = 0;
256         for (int i = 0; i < bufs.length; i++) {
257             offsets[i] = (int) cap;
258             cap += bufs[i].capacity();
259         }
260         if (cap > MAX_CAPACITY) {
261             throw new IllegalArgumentException(
262                     "Combined size of the constituent buffers is too big. " +
263                     "The maximum buffer capacity is " + MAX_CAPACITY + " (Integer.MAX_VALUE - 8), " +
264                     "but the sum of the constituent buffer capacities was " + cap + '.');
265         }
266         capacity = (int) cap;
267     }

374     @Override
375     public CompositeBuffer copy(int offset, int length) {
376         checkLength(length);
377         checkGetBounds(offset, length);
378         if (closed) {
379             throw bufferIsClosed(this);
380         }
381         Buffer choice = (Buffer) chooseBuffer(offset, 0);
382         Buffer[] copies;
383 
384         if (length > 0) {
385             copies = new Buffer[bufs.length];
386             int off = subOffset;
387             int cap = length;
388             int i;
389             int j = 0;
390             for (i = searchOffsets(offset); cap > 0; i++) {
391                 var buf = bufs[i];
392                 int avail = buf.capacity() - off;
393                 copies[j++] = buf.copy(off, Math.min(cap, avail));
394                 cap -= avail;
395                 off = 0;
396             }
397             copies = Arrays.copyOf(copies, j);
398         } else {
399             // Specialize for length == 0, since we must copy from at least one constituent buffer.
400             copies = new Buffer[] { choice.copy(subOffset, 0) };
401         }
402 
403         return new DefaultCompositeBuffer(allocator, copies, COMPOSITE_DROP);
404     }

419     private void copyInto(int srcPos, CopyInto dest, int destPos, int length) {
420         if (!isAccessible()) {
421             throw attachTrace(bufferIsClosed(this));
422         }
423         if (length < 0) {
424             throw new IndexOutOfBoundsException("Length cannot be negative: " + length + '.');
425         }
426         if (srcPos < 0) {
427             throw indexOutOfBounds(srcPos, false);
428         }
429         if (srcPos + length > capacity) {
430             throw indexOutOfBounds(srcPos + length, false);
431         }
432         while (length > 0) {
433             var buf = (Buffer) chooseBuffer(srcPos, 0);
434             int toCopy = Math.min(buf.capacity() - subOffset, length);
435             dest.copyInto(subOffset, buf, destPos, toCopy);
436             srcPos += toCopy;
437             destPos += toCopy;
438             length -= toCopy;
439         }
440     }

447     @Override
448     public void copyInto(int srcPos, Buffer dest, int destPos, int length) {
449         if (!isAccessible()) {
450             throw attachTrace(bufferIsClosed(this));
451         }
452         if (length < 0) {
453             throw new IndexOutOfBoundsException("Length cannot be negative: " + length + '.');
454         }
455         if (srcPos < 0) {
456             throw indexOutOfBounds(srcPos, false);
457         }
458         if (addExact(srcPos, length) > capacity) {
459             throw indexOutOfBounds(srcPos + length, false);
460         }
461         if (dest.readOnly()) {
462             throw bufferIsReadOnly(dest);
463         }
464 
465         // Iterate in reverse to account for src and dest buffer overlap.
466         // todo optimise by delegating to constituent buffers.
467         var cursor = openReverseCursor(srcPos + length - 1, length);
468         while (cursor.readByte()) {
469             dest.setByte(destPos + --length, cursor.getByte());
470         }
471     }

473     @Override
474     public int transferTo(WritableByteChannel channel, int length) throws IOException {
475         if (!isAccessible()) {
476             throw bufferIsClosed(this);
477         }
478         length = Math.min(readableBytes(), length);
479         if (length == 0) {
480             return 0;
481         }
482         checkReadBounds(readerOffset(), length);
483         ByteBufferCollector collector = new ByteBufferCollector(countReadableComponents());
484         forEachReadable(0, collector);
485         ByteBuffer[] byteBuffers = collector.buffers;
486         int bufferCount = countAndPrepareBuffersForChannelIO(length, byteBuffers);
487         int totalBytesWritten = 0;
488         try {
489             if (channel instanceof GatheringByteChannel) {
490                 GatheringByteChannel gatheringChannel = (GatheringByteChannel) channel;
491                 totalBytesWritten = toIntExact(gatheringChannel.write(byteBuffers, 0, bufferCount));
492             } else {
493                 for (int i = 0; i < bufferCount; i++) {
494                     int bytesWritten = channel.write(byteBuffers[i]);
495                     totalBytesWritten = addExact(totalBytesWritten, bytesWritten);
496                 }
497             }
498         } finally {
499             skipReadable(totalBytesWritten);
500         }
501         return totalBytesWritten;
502     }

504     @Override
505     public int transferFrom(ReadableByteChannel channel, int length) throws IOException {
506         if (!isAccessible()) {
507             throw bufferIsClosed(this);
508         }
509         if (readOnly()) {
510             throw bufferIsReadOnly(this);
511         }
512         length = Math.min(writableBytes(), length);
513         if (length == 0) {
514             return 0;
515         }
516         checkWriteBounds(writerOffset(), length);
517         ByteBufferCollector collector = new ByteBufferCollector(countWritableComponents());
518         forEachWritable(0, collector);
519         ByteBuffer[] byteBuffers = collector.buffers;
520         int bufferCount = countAndPrepareBuffersForChannelIO(length, byteBuffers);
521         int totalBytesRead = 0;
522         try {
523             if (channel instanceof ScatteringByteChannel) {
524                 ScatteringByteChannel scatteringChannel = (ScatteringByteChannel) channel;
525                 totalBytesRead = toIntExact(scatteringChannel.read(byteBuffers, 0, bufferCount));
526             } else {
527                 for (int i = 0; i < bufferCount; i++) {
528                     int bytesRead = channel.read(byteBuffers[i]);
529                     if (bytesRead == -1) {
530                         if (i == 0) {
531                             return -1; // If we're end-of-stream on the first read, immediately return -1.
532                         }
533                         break;
534                     }
535                     totalBytesRead = addExact(totalBytesRead, bytesRead);
536                 }
537             }
538         } finally {
539             if (totalBytesRead > 0) { // Don't skipWritable if total is 0 or -1
540                 skipWritable(totalBytesRead);
541             }
542         }
543         return totalBytesRead;
544     }

546     private static int countAndPrepareBuffersForChannelIO(int byteLength, ByteBuffer[] byteBuffers) {
547         int bufferCount = 0;
548         int byteSum = 0;
549         for (ByteBuffer buffer : byteBuffers) {
550             byteSum += buffer.remaining();
551             bufferCount++;
552             if (byteSum >= byteLength) {
553                 int diff = byteSum - byteLength;
554                 if (diff > 0) {
555                     buffer.limit(buffer.limit() - diff);
556                 }
557                 break;
558             }
559         }
560         return bufferCount;
561     }

563     @Override
564     public int bytesBefore(byte needle) {
565         if (!isAccessible()) {
566             throw bufferIsClosed(this);
567         }
568         final int length = readableBytes();
569         for (int i = searchOffsets(readerOffset()), skip = 0; skip < length; i++) {
570             Buffer buf = bufs[i];
571             int found = buf.bytesBefore(needle);
572             if (found != -1) {
573                 return skip + found;
574             }
575             skip += buf.readableBytes();
576         }
577         return -1;

585     @Override
586     public ByteCursor openCursor(int fromOffset, int length) {
587         if (fromOffset < 0) {
588             throw new IllegalArgumentException("The fromOffset cannot be negative: " + fromOffset + '.');
589         }
590         checkLength(length);
591         if (capacity < addExact(fromOffset, length)) {
592             throw new IllegalArgumentException("The fromOffset+length is beyond the end of the buffer: " +
593                                                "fromOffset=" + fromOffset + ", length=" + length + '.');
594         }
595         if (closed) {
596             throw bufferIsClosed(this);
597         }
598         int startBufferIndex = searchOffsets(fromOffset);
599         int off = fromOffset - offsets[startBufferIndex];
600         Buffer startBuf = bufs[startBufferIndex];
601         ByteCursor startCursor = startBuf.openCursor(off, Math.min(startBuf.capacity() - off, length));
602         return new ForwardCompositeByteCursor(bufs, fromOffset, length, startBufferIndex, startCursor);
603     }

605     @Override
606     public ByteCursor openReverseCursor(int fromOffset, int length) {
607         if (fromOffset < 0) {
608             throw new IllegalArgumentException("The fromOffset cannot be negative: " + fromOffset + '.');
609         }
610         checkLength(length);
611         if (fromOffset - length < -1) {
612             throw new IllegalArgumentException("The fromOffset-length would underflow the buffer: " +
613                                                "fromOffset=" + fromOffset + ", length=" + length + '.');
614         }
615         if (closed) {
616             throw bufferIsClosed(this);
617         }
618         int startBufferIndex = searchOffsets(fromOffset);
619         int off = fromOffset - offsets[startBufferIndex];
620         Buffer startBuf = bufs[startBufferIndex];
621         ByteCursor startCursor = startBuf.openReverseCursor(off, Math.min(off + 1, length));
622         return new ReverseCompositeByteCursor(bufs, fromOffset, length, startBufferIndex, startCursor);
623     }

625     @Override
626     public CompositeBuffer ensureWritable(int size, int minimumGrowth, boolean allowCompaction) {
627         if (!isAccessible()) {
628             throw bufferIsClosed(this);
629         }
630         if (!isOwned()) {
631             throw new IllegalStateException("Buffer is not owned. Only owned buffers can call ensureWritable.");
632         }
633         if (size < 0) {
634             throw new IllegalArgumentException("Cannot ensure writable for a negative size: " + size + '.');
635         }
636         if (minimumGrowth < 0) {
637             throw new IllegalArgumentException("The minimum growth cannot be negative: " + minimumGrowth + '.');
638         }
639         if (readOnly) {
640             throw bufferIsReadOnly(this);
641         }
642         if (writableBytes() >= size) {
643             // We already have enough space.
644             return this;
645         }
646 
647         if (allowCompaction && size <= roff) {
648             // Let's see if we can solve some or all of the requested size with compaction.
649             // We always compact as much as is possible, regardless of size. This amortizes our work.
650             int compactableBuffers = 0;
651             for (Buffer buf : bufs) {
652                 if (buf.capacity() != buf.readerOffset()) {
653                     break;
654                 }
655                 compactableBuffers++;
656             }
657             if (compactableBuffers > 0) {
658                 Buffer[] compactable;
659                 if (compactableBuffers < bufs.length) {
660                     compactable = new Buffer[compactableBuffers];
661                     System.arraycopy(bufs, 0, compactable, 0, compactable.length);
662                     System.arraycopy(bufs, compactable.length, bufs, 0, bufs.length - compactable.length);
663                     System.arraycopy(compactable, 0, bufs, bufs.length - compactable.length, compactable.length);
664                 } else {
665                     compactable = bufs;
666                 }
667                 for (Buffer buf : compactable) {
668                     buf.resetOffsets();
669                 }
670                 computeBufferOffsets();
671                 if (writableBytes() >= size) {
672                     // Now we have enough space.
673                     return this;
674                 }
675             } else if (bufs.length == 1) {
676                 // If we only have a single component buffer, then we can safely compact that in-place.
677                 bufs[0].compact();
678                 computeBufferOffsets();
679                 if (writableBytes() >= size) {
680                     // Now we have enough space.
681                     return this;
682                 }
683             }
684         }
685 
686         int growth = Math.max(size - writableBytes(), minimumGrowth);
687         Statics.assertValidBufferSize(capacity() + (long) growth);
688         Buffer extension = allocator.allocate(growth);
689         unsafeExtendWith(extension);
690         return this;
691     }

693     @Override
694     public CompositeBuffer extendWith(Send<Buffer> extension) {
695         Buffer buffer = Objects.requireNonNull(extension, "Extension buffer cannot be null.").receive();
696         if (!isAccessible() || !isOwned()) {
697             buffer.close();
698             if (!isAccessible()) {
699                 throw bufferIsClosed(this);
700             }
701             throw new IllegalStateException("This buffer cannot be extended because it is not in an owned state.");
702         }
703         if (bufs.length > 0 && buffer.readOnly() != readOnly()) {
704             buffer.close();
705             throw new IllegalArgumentException(
706                     "This buffer is " + (readOnly? "read-only" : "writable") + ", " +
707                     "and cannot be extended with a buffer that is " +
708                     (buffer.readOnly()? "read-only." : "writable."));
709         }
710 
711         long extensionCapacity = buffer.capacity();
712         if (extensionCapacity == 0) {
713             // Extending by a zero-sized buffer makes no difference. Especially since it's not allowed to change the
714             // capacity of buffers that are constituents of composite buffers.
715             // This also ensures that methods like countComponents, and forEachReadable, do not have to worry about
716             // overflow in their component counters.
717             buffer.close();
718             return this;
719         }
720 
721         long newSize = capacity() + extensionCapacity;
722         Statics.assertValidBufferSize(newSize);
723 
724         Buffer[] restoreTemp = bufs; // We need this to restore our buffer array, in case offset computations fail.
725         try {
726             if (CompositeBuffer.isComposite(buffer)) {
727                 // If the extension is itself a composite buffer, then extend this one by all the constituent
728                 // component buffers.
729                 CompositeBuffer compositeExtension = (CompositeBuffer) buffer;
730                 Buffer[] addedBuffers = compositeExtension.decomposeBuffer();
731                 Set<Buffer> duplicatesCheck = Collections.newSetFromMap(new IdentityHashMap<>());
732                 duplicatesCheck.addAll(Arrays.asList(bufs));
733                 duplicatesCheck.addAll(Arrays.asList(addedBuffers));
734                 if (duplicatesCheck.size() < bufs.length + addedBuffers.length) {
735                     throw extensionDuplicatesException();
736                 }
737                 int extendAtIndex = bufs.length;
738                 bufs = Arrays.copyOf(bufs, extendAtIndex + addedBuffers.length);
739                 System.arraycopy(addedBuffers, 0, bufs, extendAtIndex, addedBuffers.length);
740                 computeBufferOffsets();
741             } else {
742                 for (Buffer buf : restoreTemp) {
743                     if (buf == buffer) {
744                         throw extensionDuplicatesException();
745                     }
746                 }
747                 unsafeExtendWith(buffer);
748             }
749             if (restoreTemp.length == 0) {
750                 readOnly = buffer.readOnly();
751             }
752         } catch (Exception e) {
753             bufs = restoreTemp;
754             throw e;
755         }
756         return this;
757     }

794         checkSplit(splitOffset);
795         if (bufs.length == 0) {
796             // Splitting a zero-length buffer is trivial.
797             return new DefaultCompositeBuffer(allocator, bufs, unsafeGetDrop());
798         }
799 
800         int i = searchOffsets(splitOffset);
801         int off = splitOffset - offsets[i];
802         Buffer[] splits = Arrays.copyOf(bufs, off == 0? i : 1 + i);
803         bufs = Arrays.copyOfRange(bufs, off == bufs[i].capacity()? 1 + i : i, bufs.length);
804         if (off > 0 && splits.length > 0 && off < splits[splits.length - 1].capacity()) {
805             splits[splits.length - 1] = bufs[0].split(off);
806         }
807         computeBufferOffsets();
808         return buildSplitBuffer(splits);
809     }

874     public CompositeBuffer compact() {
875         if (!isAccessible()) {
876             throw attachTrace(bufferIsClosed(this));
877         }
878         if (!isOwned()) {
879             throw attachTrace(new IllegalStateException("Buffer must be owned in order to compact."));
880         }
881         if (readOnly()) {
882             throw new BufferReadOnlyException("Buffer must be writable in order to compact, but was read-only.");
883         }

885         if (distance == 0) {
886             return this;
887         }
888         int pos = 0;
889         // TODO maybe we can delegate to a copyInto method, once it's more optimised
890         var cursor = openCursor();
891         while (cursor.readByte()) {
892             setByte(pos, cursor.getByte());
893             pos++;
894         }
895         readerOffset(0);
896         writerOffset(woff - distance);
897         return this;
898     }

927     @Override
928     public <E extends Exception> int forEachReadable(int initialIndex, ReadableComponentProcessor<E> processor)
929             throws E {
930         if (!isAccessible()) {
931             throw attachTrace(bufferIsClosed(this));
932         }
933         int readableBytes = readableBytes();
934         if (readableBytes == 0) {
935             return 0;
936         }
937         checkReadBounds(readerOffset(), readableBytes);
938         int visited = 0;
939         for (Buffer buf : bufs) {
940             if (buf.readableBytes() > 0) {
941                 int roffBefore = buf.readerOffset();
942                 int count = buf.forEachReadable(visited + initialIndex, processor);
943                 int roffAfter = buf.readerOffset();
944                 if (roffAfter != roffBefore) { // Check if ReadableComponent.skipReadable was called.
945                     buf.readerOffset(roffBefore); // Reset component offset.
946                     skipReadable(roffAfter - roffBefore); // Then move *composite* buffer offset.
947                 }
948                 if (count > 0) {
949                     visited += count;
950                 } else {
951                     visited = -visited + count;
952                     break;
953                 }
954             }
955         }
956         return visited;
957     }

960     public <E extends Exception> int forEachWritable(int initialIndex, WritableComponentProcessor<E> processor)
961             throws E {
962         if (!isAccessible()) {
963             throw attachTrace(bufferIsClosed(this));
964         }
965         int writableBytes = writableBytes();
966         if (writableBytes == 0) {
967             return 0;
968         }
969         checkWriteBounds(writerOffset(), writableBytes);
970         int visited = 0;
971         for (Buffer buf : bufs) {
972             if (buf.writableBytes() > 0) {
973                 int woffBefore = buf.writerOffset();
974                 int count = buf.forEachWritable(visited + initialIndex, processor);
975                 int woffAfter = buf.writerOffset();
976                 if (woffAfter != woffBefore) { // Check if WritableComponent.skipWritable was called.
977                     buf.writerOffset(woffBefore);
978                     skipWritable(woffAfter - woffBefore);
979                 }
980                 if (count > 0) {
981                     visited += count;
982                 } else {
983                     visited = -visited + count;
984                     break;
985                 }
986             }
987         }
988         return visited;

1258     protected Owned<DefaultCompositeBuffer> prepareSend() {
1259         @SuppressWarnings("unchecked")
1260         Send<Buffer>[] sends = new Send[bufs.length];
1261         try {
1262             for (int i = 0; i < bufs.length; i++) {
1263                 sends[i] = bufs[i].send();
1264             }
1265         } catch (Throwable throwable) {
1266             // Repair our bufs array.
1267             for (int i = 0; i < sends.length; i++) {
1268                 if (sends[i] != null) {
1269                     try {
1270                         bufs[i] = sends[i].receive();
1271                     } catch (Exception e) {
1272                         throwable.addSuppressed(e);
1273                     }
1274                 }
1275             }
1276             throw throwable;
1277         }
1278         boolean readOnly = this.readOnly;
1279         return drop -> {
1280             Buffer[] received = new Buffer[sends.length];
1281             for (int i = 0; i < sends.length; i++) {
1282                 received[i] = sends[i].receive();
1283             }
1284             var composite = new DefaultCompositeBuffer(allocator, received, drop);
1285             composite.readOnly = readOnly;
1286             drop.attach(composite);
1287             return composite;
1288         };
1289     }

1404         int i = searchOffsets(index);
1405         if (i == bufs.length) {
1406             // This happens when the read/write offsets are parked 1 byte beyond the end of the buffer.
1407             // In that case it should not matter what buffer is returned, because it shouldn't be used anyway.
1408             return null;

1413             subOffset = off;
1414             return candidate;
1415         }
