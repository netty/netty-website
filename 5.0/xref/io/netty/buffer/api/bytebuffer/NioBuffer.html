137     @Override

139         int capacity = capacity();
140         checkSet(0, capacity);
141         if (rmem == CLOSED_BUFFER) {
142             throw bufferIsClosed(this);
143         }
144         for (int i = 0; i < capacity; i++) {
145             wmem.put(i, value);

190     @Override
191     public void copyInto(int srcPos, ByteBuffer dest, int destPos, int length) {
192         if (rmem == CLOSED_BUFFER) {
193             throw bufferIsClosed(this);
194         }
195         if (srcPos < 0) {
196             throw new IllegalArgumentException("The srcPos cannot be negative: " + srcPos + '.');
197         }
198         checkLength(length);
199         if (capacity() < srcPos + length) {
200             throw new IllegalArgumentException("The srcPos + length is beyond the end of the buffer: " +
201                     "srcPos = " + srcPos + ", length = " + length + '.');
202         }
203         dest = dest.duplicate().clear();
204         bbput(dest, destPos, rmem, srcPos, length);
205     }

207     @Override
208     public void copyInto(int srcPos, Buffer dest, int destPos, int length) {
209         if (!isAccessible()) {
210             throw attachTrace(bufferIsClosed(this));
211         }
212         if (dest.readOnly()) {
213             throw bufferIsReadOnly(dest);
214         }
215         if (dest instanceof NioBuffer) {
216             var nb = (NioBuffer) dest;
217             nb.checkSet(destPos, length);
218             copyInto(srcPos, nb.wmem, destPos, length);
219             return;
220         }
221 
222         Statics.copyToViaReverseLoop(this, srcPos, dest, destPos, length);
223     }

225     @Override
226     public int transferTo(WritableByteChannel channel, int length) throws IOException {
227         if (!isAccessible()) {
228             throw bufferIsClosed(this);
229         }
230         length = Math.min(readableBytes(), length);
231         if (length == 0) {
232             return 0;
233         }
234         checkGet(readerOffset(), length);
235         int bytesWritten = channel.write(readableBuffer().limit(length));
236         skipReadable(bytesWritten);
237         return bytesWritten;
238     }

240     @Override
241     public int transferFrom(ReadableByteChannel channel, int length) throws IOException {
242         if (!isAccessible()) {
243             throw bufferIsClosed(this);
244         }
245         if (readOnly()) {
246             throw bufferIsReadOnly(this);
247         }
248         length = Math.min(writableBytes(), length);
249         if (length == 0) {
250             return 0;
251         }
252         checkSet(writerOffset(), length);
253         int bytesRead = channel.read(writableBuffer().limit(length));
254         if (bytesRead != -1) {
255             skipWritable(bytesRead);
256         }
257         return bytesRead;
258     }

260     @Override
261     public int bytesBefore(byte needle) {
262         // For the details of this algorithm, see Hacker's Delight, Chapter 6, Searching Words.
263         // Richard Startin also describes this on his blog: https://richardstartin.github.io/posts/finding-bytes
264         if (!isAccessible()) {
265             throw bufferIsClosed(this);
266         }
267         int offset = roff;
268         final int length = woff - roff;
269         final int end = woff;
270 
271         if (length > 7) {
272             final long pattern = (needle & 0xFFL) * 0x101010101010101L;
273             for (final int longEnd = offset + (length >>> 3) * Long.BYTES;
274                  offset < longEnd;
275                  offset += Long.BYTES) {
276                 final long word = rmem.getLong(offset);
277 
278                 long input = word ^ pattern;
279                 long tmp = (input & 0x7F7F7F7F7F7F7F7FL) + 0x7F7F7F7F7F7F7F7FL;
280                 tmp = ~(tmp | input | 0x7F7F7F7F7F7F7F7FL);
281                 final int binaryPosition = Long.numberOfLeadingZeros(tmp);
282 
283                 int index = binaryPosition >>> 3;
284                 if (index < Long.BYTES) {
285                     return offset + index - roff;
286                 }
287             }
288         }
289         for (; offset < end; offset++) {
290             if (rmem.get(offset) == needle) {
291                 return offset - roff;
292             }
293         }
294 
295         return -1;
296     }

304     public ByteCursor openCursor(int fromOffset, int length) {
305         if (rmem == CLOSED_BUFFER) {
306             throw bufferIsClosed(this);
307         }
308         if (fromOffset < 0) {
309             throw new IllegalArgumentException("The fromOffset cannot be negative: " + fromOffset + '.');
310         }
311         checkLength(length);
312         if (capacity() < fromOffset + length) {
313             throw new IllegalArgumentException("The fromOffset + length is beyond the end of the buffer: " +
314                     "fromOffset = " + fromOffset + ", length = " + length + '.');
315         }
316         return new ForwardNioByteCursor(rmem, fromOffset, length);

319     @Override
320     public ByteCursor openReverseCursor(int fromOffset, int length) {
321         if (rmem == CLOSED_BUFFER) {
322             throw bufferIsClosed(this);
323         }
324         if (fromOffset < 0) {
325             throw new IllegalArgumentException("The fromOffset cannot be negative: " + fromOffset + '.');
326         }
327         checkLength(length);
328         if (capacity() <= fromOffset) {
329             throw new IllegalArgumentException("The fromOffset is beyond the end of the buffer: " + fromOffset + '.');
330         }
331         if (fromOffset - length < -1) {
332             throw new IllegalArgumentException("The fromOffset - length would underflow the buffer: " +
333                     "fromOffset = " + fromOffset + ", length = " + length + '.');
334         }
335         return new ReverseNioByteCursor(rmem, fromOffset, length);
336     }

338     @Override
339     public Buffer ensureWritable(int size, int minimumGrowth, boolean allowCompaction) {
340         if (!isAccessible()) {
341             throw bufferIsClosed(this);
342         }
343         if (!isOwned()) {
344             throw attachTrace(new IllegalStateException(
345                     "Buffer is not owned. Only owned buffers can call ensureWritable."));
346         }
347         if (size < 0) {
348             throw new IllegalArgumentException("Cannot ensure writable for a negative size: " + size + '.');
349         }
350         if (minimumGrowth < 0) {
351             throw new IllegalArgumentException("The minimum growth cannot be negative: " + minimumGrowth + '.');
352         }
353         if (rmem != wmem) {
354             throw bufferIsReadOnly(this);
355         }
356         if (writableBytes() >= size) {
357             // We already have enough space.
358             return this;
359         }
360 
361         if (allowCompaction && writableBytes() + readerOffset() >= size) {
362             // We can solve this with compaction.
363             return compact();
364         }
365 
366         // Allocate a bigger buffer.
367         long newSize = capacity() + (long) Math.max(size - writableBytes(), minimumGrowth);
368         Statics.assertValidBufferSize(newSize);
369         NioBuffer buffer = (NioBuffer) control.getAllocator().allocate((int) newSize);
370 
371         // Copy contents.
372         copyInto(0, buffer, 0, capacity());
373 
374         // Release the old memory and install the new:
375         Drop<NioBuffer> drop = buffer.unsafeGetDrop();
376         disconnectDrop(drop);
377         attachNewBuffer(buffer, drop);
378         return this;
379     }
380 
381     private void disconnectDrop(Drop<NioBuffer> newDrop) {
382         var drop = (Drop<NioBuffer>) unsafeGetDrop();
383         int roff = this.roff;
384         int woff = this.woff;
385         drop.drop(this);
386         unsafeSetDrop(newDrop);
387         this.roff = roff;
388         this.woff = woff;
389     }
390 
391     private void attachNewBuffer(NioBuffer buffer, Drop<NioBuffer> drop) {
392         base = buffer.base;
393         rmem = buffer.rmem;
394         wmem = buffer.wmem;
395         drop.attach(this);
396     }

398     @Override
399     public Buffer split(int splitOffset) {
400         if (splitOffset < 0) {
401             throw new IllegalArgumentException("The split offset cannot be negative: " + splitOffset + '.');
402         }
403         if (capacity() < splitOffset) {
404             throw new IllegalArgumentException("The split offset cannot be greater than the buffer capacity, " +
405                     "but the split offset was " + splitOffset + ", and capacity is " + capacity() + '.');
406         }
407         if (!isAccessible()) {
408             throw attachTrace(bufferIsClosed(this));
409         }
410         if (!isOwned()) {
411             throw attachTrace(new IllegalStateException("Cannot split a buffer that is not owned."));
412         }
413         var drop = unsafeGetDrop().fork();
414         var splitByteBuffer = bbslice(rmem, 0, splitOffset);
415         var splitBuffer = new NioBuffer(base, splitByteBuffer, control, drop);
416         drop.attach(splitBuffer);
417         splitBuffer.woff = Math.min(woff, splitOffset);
418         splitBuffer.roff = Math.min(roff, splitOffset);
419         boolean readOnly = readOnly();
420         if (readOnly) {
421             splitBuffer.makeReadOnly();
422         }
423         // Split preserves const-state.
424         rmem = bbslice(rmem, splitOffset, rmem.capacity() - splitOffset);
425         if (!readOnly) {
426             wmem = rmem;
427         }
428         woff = Math.max(woff, splitOffset) - splitOffset;
429         roff = Math.max(roff, splitOffset) - splitOffset;
430         return splitBuffer;
431     }

433     @Override
434     public Buffer compact() {
435         if (!isAccessible()) {
436             throw attachTrace(bufferIsClosed(this));
437         }
438         if (!isOwned()) {
439             throw attachTrace(new IllegalStateException("Buffer must be owned in order to compact."));
440         }
441         if (readOnly()) {
442             throw new BufferReadOnlyException("Buffer must be writable in order to compact, but was read-only.");
443         }
444         if (roff == 0) {
445             return this;
446         }
447         rmem.limit(woff).position(roff).compact().clear();
448         woff -= roff;
449         roff = 0;
450         return this;
451     }

1013     protected Owned<NioBuffer> prepareSend() {
1014         var roff = this.roff;
1015         var woff = this.woff;
1016         var readOnly = readOnly();
1017         ByteBuffer base = this.base;
1018         ByteBuffer rmem = this.rmem;
1019         return new Owned<NioBuffer>() {
1020             @Override
1021             public NioBuffer transferOwnership(Drop<NioBuffer> drop) {
1022                 NioBuffer copy = new NioBuffer(base, rmem, control, drop);
1023                 copy.roff = roff;
1024                 copy.woff = woff;
1025                 if (readOnly) {
1026                     copy.makeReadOnly();
1027                 }
1028                 return copy;
1029             }
1030         };
1031     }

1042     private void checkRead(int index, int size) {
1043         if (index < 0 | woff < index + size) {
1044             throw readAccessCheckException(index, size);
1045         }
1046     }

1048     private void checkGet(int index, int size) {
1049         if (index < 0 | capacity() < index + size) {
1050             throw readAccessCheckException(index, size);
1051         }
1052     }

1054     private void checkWrite(int index, int size) {
1055         if (index < roff | wmem.capacity() < index + size) {
1056             throw writeAccessCheckException(index, size);
1057         }
1058     }
1059 
1060     private void checkSet(int index, int size) {
1061         if (index < 0 | wmem.capacity() < index + size) {
1062             throw writeAccessCheckException(index, size);
1063         }
1064     }

1066     private RuntimeException checkWriteState(IndexOutOfBoundsException ioobe, int offset, int size) {
1067         if (rmem == CLOSED_BUFFER) {
1068             return bufferIsClosed(this);
1069         }
1070         if (wmem != rmem) {
1071             return bufferIsReadOnly(this);
1072         }
1073 
1074         IndexOutOfBoundsException exception = outOfBounds(offset, size);
1075         exception.addSuppressed(ioobe);
1076         return exception;
1077     }
1078 
1079     private RuntimeException readAccessCheckException(int index, int size) {
1080         if (rmem == CLOSED_BUFFER) {
1081             throw bufferIsClosed(this);
1082         }
1083         return outOfBounds(index, size);
1084     }
1085 
1086     private RuntimeException writeAccessCheckException(int index, int size) {
1087         if (rmem == CLOSED_BUFFER) {
1088             throw bufferIsClosed(this);
1089         }
1090         if (wmem != rmem) {
1091             return bufferIsReadOnly(this);
1092         }
1093         return outOfBounds(index, size);
1094     }
1095 
1096     private IndexOutOfBoundsException outOfBounds(int index, int size) {
1097         return new IndexOutOfBoundsException(
1098                 "Access at index " + index + " of size " + size + " is out of bounds: " +
1099                 "[read 0 to " + woff + ", write 0 to " + rmem.capacity() + "].");
1100     }

1121         ForwardNioByteCursor(ByteBuffer rmem, int fromOffset, int length) {
1122             buffer = rmem.duplicate().order(ByteOrder.BIG_ENDIAN);
1123             index = fromOffset;

1125             byteValue = -1;

1136         }
