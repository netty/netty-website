16  package io.netty5.buffer.api;
17  
18  /**
19   * The {@code CompositeBuffer} is a concrete {@link Buffer} implementation that make a number of other buffers appear
20   * as one. A composite buffer behaves the same as a normal, non-composite buffer in every way, so you normally don't
21   * need to handle them specially.
22   * <p>
23   * A composite buffer is constructed using one of the {@code compose} methods:
24   * <ul>
25   *     <li>
26   *         {@link #compose(BufferAllocator, Send[])} creates a composite buffer from the buffers that are sent to it via
27   *         the passed in send objects. Since {@link Send#receive()} transfers ownership, the resulting composite buffer
28   *         will have ownership, because it is guaranteed that there are no other references to its constituent buffers.
29   *     </li>
30   *     <li>
31   *         {@link #compose(BufferAllocator)} creates an empty, zero capacity, composite buffer. Such empty buffers may
32   *         change their {@linkplain #readOnly() read-only} states when they gain their first component.
33   *     </li>
34   * </ul>
35   * Composite buffers can later be extended with internally allocated components, with {@link #ensureWritable(int)},
36   * or with externally allocated buffers, using {@link #extendWith(Send)}.
37   *
38   * <h3>Constituent buffer requirements</h3>
39   *
40   * The buffers that are being composed to form the composite buffer, need to live up to a number of requirements.
41   * Basically, if we imagine that the constituent buffers have their memory regions concatenated together, then the
42   * result needs to make sense.
43   * <p>
44   * The read and write offsets of the constituent buffers must be arranged such that there are no "gaps" when viewed
45   * as a single connected chunk of memory.
46   * Specifically, there can be at most one buffer whose write offset is neither zero nor at capacity,
47   * and all buffers prior to it must have their write offsets at capacity, and all buffers after it must have a
48   * write-offset of zero.
49   * Likewise, there can be at most one buffer whose read offset is neither zero nor at capacity,
50   * and all buffers prior to it must have their read offsets at capacity, and all buffers after it must have a read
51   * offset of zero.
52   * Furthermore, the sum of the read offsets must be less than or equal to the sum of the write-offsets.
53   * <p>
54   * Reads and writes to the composite buffer that modifies the read or write offsets, will also modify the relevant
55   * offsets in the constituent buffers.
56   * <p>
57   * It is not a requirement that the buffers have the same size.
58   * <p>
59   * It is not a requirement that the buffers are allocated by this allocator, but if
60   * {@link Buffer#ensureWritable(int)} is called on the composed buffer, and the composed buffer needs to be
61   * expanded, then this allocator instance will be used for allocation the extra memory.
62   *
63   * <h3>Ownership and Send</h3>
64   *
65   * {@linkplain Resource#send() Sending} a composite buffer implies sending all of its constituent buffers.
66   * For sending to be possible, both the composite buffer itself, and all of its constituent buffers, must be in a
67   * state that permits them being sent. This should be the case by default, as it shouldn't be possible to create
68   * composite buffers that can't be sent.
69   */
70  public interface CompositeBuffer extends Buffer {
71  
72      /**
73       * Compose the given sequence of sends of buffers and present them as a single buffer.
74       * <p>
75       * When a composite buffer is closed, all of its constituent component buffers are closed as well.
76       * <p>
77       * See the class documentation for more information on what is required of the given buffers for composition to be
78       * allowed.
79       *
80       * @param allocator The allocator for the composite buffer. This allocator will be used e.g. to service
81       * {@link #ensureWritable(int)} calls.
82       * @param sends The sent buffers to compose into a single buffer view.
83       * @return A buffer composed of, and backed by, the given buffers.
84       * @throws IllegalStateException if one of the sends have already been received. The remaining buffers and sends
85       * will be closed and discarded, respectively.
86       */
87      @SafeVarargs
88      static CompositeBuffer compose(BufferAllocator allocator, Send<Buffer>... sends) {
89          return DefaultCompositeBuffer.compose(allocator, sends);
90      }
91  
92      /**
93       * Create an empty composite buffer, that has no components. The buffer can be extended with components using either
94       * {@link #ensureWritable(int)} or {@link #extendWith(Send)}.
95       *
96       * @param allocator The allocator for the composite buffer. This allocator will be used e.g. to service
97       * {@link #ensureWritable(int)} calls.
98       * @return A composite buffer that has no components, and has a capacity of zero.
99       */
100     static CompositeBuffer compose(BufferAllocator allocator) {
101         return DefaultCompositeBuffer.compose(allocator);
102     }
103 
104     /**
105      * Check if the given buffer is a {@linkplain #compose(BufferAllocator, Send...) composite} buffer or not.
106      * @param composite The buffer to check.
107      * @return {@code true} if the given buffer was created with {@link #compose(BufferAllocator, Send...)},
108      * {@code false} otherwise.
109      */
110     static boolean isComposite(Buffer composite) {
111         return composite instanceof CompositeBuffer;
112     }
113 
114     /**
115      * Extend this composite buffer with the given extension buffer.
116      * This works as if the extension had originally been included at the end of the list of constituent buffers when
117      * the composite buffer was created.
118      * The extension buffer is added to the end of this composite buffer, which is modified in-place.
119      *
120      * @see #compose(BufferAllocator, Send...)
121      * @param extension The buffer to extend the composite buffer with.
122      * @return This composite buffer instance.
123      */
124     CompositeBuffer extendWith(Send<Buffer> extension);
125 
126     /**
127      * Split this buffer at a component boundary that is less than or equal to the given offset.
128      * <p>
129      * This method behaves the same as {@link #split(int)}, except no components are split.
130      *
131      * @param splitOffset The maximum split offset. The real split offset will be at a component boundary that is less
132      *                   than or equal to this offset.
133      * @return A new buffer with independent and exclusive ownership over the bytes from the beginning to a component
134      * boundary less than or equal to the given offset of this buffer.
135      */
136     CompositeBuffer splitComponentsFloor(int splitOffset);
137 
138     /**
139      * Split this buffer at a component boundary that is greater than or equal to the given offset.
140      * <p>
141      * This method behaves the same as {@link #split(int)}, except no components are split.
142      *
143      * @param splitOffset The minimum split offset. The real split offset will be at a component boundary that is
144      *                   greater than or equal to this offset.
145      * @return A new buffer with independent and exclusive ownership over the bytes from the beginning to a component
146      * boundary greater than or equal to the given offset of this buffer.
147      */
148     CompositeBuffer splitComponentsCeil(int splitOffset);
149 
150     /**
151      * Break a composite buffer into its constituent components.
152      * <p>
153      * This "consumes" the composite buffer, leaving the composite buffer instance as if it had been closed.
154      * The buffers in the returned array are not closed, and become owned by the caller.
155      *
156      * @return An array of the constituent buffer components.
157      */
158     Buffer[] decomposeBuffer();
