69      static {
70          int defaultAlignment = SystemPropertyUtil.getInt(
71                  "io.netty5.allocator.directMemoryCacheAlignment", 0);
72          int defaultPageSize = SystemPropertyUtil.getInt("io.netty5.allocator.pageSize", 8192);
73          Throwable pageSizeFallbackCause = null;
74          try {
75              validateAndCalculatePageShifts(defaultPageSize, defaultAlignment);
76          } catch (Throwable t) {
77              pageSizeFallbackCause = t;
78              defaultPageSize = 8192;
79              defaultAlignment = 0;
80          }
81          DEFAULT_PAGE_SIZE = defaultPageSize;
82          DEFAULT_DIRECT_MEMORY_CACHE_ALIGNMENT = defaultAlignment;
83  
84          int defaultMaxOrder = SystemPropertyUtil.getInt("io.netty5.allocator.maxOrder", 9);
85          Throwable maxOrderFallbackCause = null;
86          try {
87              validateAndCalculateChunkSize(DEFAULT_PAGE_SIZE, defaultMaxOrder);
88          } catch (Throwable t) {
89              maxOrderFallbackCause = t;
90              defaultMaxOrder = 11;
91          }
92          DEFAULT_MAX_ORDER = defaultMaxOrder;
93  
94          // Determine reasonable default for nHeapArena and nDirectArena.
95          // Assuming each arena has 3 chunks, the pool should not consume more than 50% of max memory.
96          final Runtime runtime = Runtime.getRuntime();
97  
98          /*
99           * We use 2 * available processors by default to reduce contention as we use 2 * available processors for the
100          * number of EventLoops in NIO and EPOLL as well. If we choose a smaller number we will run into hot spots as
101          * allocation and de-allocation needs to be synchronized on the PoolArena.
102          *
103          * See https://github.com/netty/netty/issues/3888.
104          */
105         final int defaultMinNumArena = NettyRuntime.availableProcessors() * 2;
106         final int defaultChunkSize = DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER;
107         DEFAULT_NUM_HEAP_ARENA = Math.max(0,
108                 SystemPropertyUtil.getInt(
109                         "io.netty5.allocator.numArenas",
110                         (int) Math.min(
111                                 defaultMinNumArena,
112                                 runtime.maxMemory() / defaultChunkSize / 2 / 3)));
113         DEFAULT_NUM_DIRECT_ARENA = Math.max(0,
114                 SystemPropertyUtil.getInt(
115                         "io.netty5.allocator.numDirectArenas",
116                         (int) Math.min(
117                                 defaultMinNumArena,
118                                 PlatformDependent.maxDirectMemory() / defaultChunkSize / 2 / 3)));
119 
120         // cache sizes
121         DEFAULT_SMALL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty5.allocator.smallCacheSize", 256);
122         DEFAULT_NORMAL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty5.allocator.normalCacheSize", 64);
123 
124         // 32 kb is the default maximum capacity of the cached buffer. Similar to what is explained in
125         // 'Scalable memory allocation using jemalloc'
126         DEFAULT_MAX_CACHED_BUFFER_CAPACITY = SystemPropertyUtil.getInt(
127                 "io.netty5.allocator.maxCachedBufferCapacity", 32 * 1024);
128 
129         // the number of threshold of allocations when cached entries will be freed up if not frequently used
130         DEFAULT_CACHE_TRIM_INTERVAL = SystemPropertyUtil.getInt(
131                 "io.netty5.allocator.cacheTrimInterval", 8192);
132 
133         DEFAULT_CACHE_TRIM_INTERVAL_MILLIS = SystemPropertyUtil.getLong(
134                 "io.netty5.allocator.cacheTrimIntervalMillis", 0);
135 
136         DEFAULT_USE_CACHE_FOR_ALL_THREADS = SystemPropertyUtil.getBoolean(
137                 "io.netty5.allocator.useCacheForAllThreads", false);
138 
139         // Use 1023 by default as we use an ArrayDeque as backing storage which will then allocate an internal array
140         // of 1024 elements. Otherwise, we would allocate 2048 and only use 1024 which is wasteful.
141         DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK = SystemPropertyUtil.getInt(
142                 "io.netty5.allocator.maxCachedByteBuffersPerChunk", 1023);
143 
144         if (logger.isDebugEnabled()) {
145             logger.debug("-Dio.netty5.allocator.numArenas: {}", DEFAULT_NUM_HEAP_ARENA);
146             logger.debug("-Dio.netty5.allocator.numDirectArenas: {}", DEFAULT_NUM_DIRECT_ARENA);
147             if (pageSizeFallbackCause == null) {
148                 logger.debug("-Dio.netty5.allocator.pageSize: {}", DEFAULT_PAGE_SIZE);
149             } else {
150                 logger.debug("-Dio.netty5.allocator.pageSize: {}", DEFAULT_PAGE_SIZE, pageSizeFallbackCause);
151             }
152             if (maxOrderFallbackCause == null) {
153                 logger.debug("-Dio.netty5.allocator.maxOrder: {}", DEFAULT_MAX_ORDER);
154             } else {
155                 logger.debug("-Dio.netty5.allocator.maxOrder: {}", DEFAULT_MAX_ORDER, maxOrderFallbackCause);
156             }
157             logger.debug("-Dio.netty5.allocator.chunkSize: {}", DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER);
158             logger.debug("-Dio.netty5.allocator.smallCacheSize: {}", DEFAULT_SMALL_CACHE_SIZE);
159             logger.debug("-Dio.netty5.allocator.normalCacheSize: {}", DEFAULT_NORMAL_CACHE_SIZE);
160             logger.debug("-Dio.netty5.allocator.maxCachedBufferCapacity: {}", DEFAULT_MAX_CACHED_BUFFER_CAPACITY);
161             logger.debug("-Dio.netty5.allocator.cacheTrimInterval: {}", DEFAULT_CACHE_TRIM_INTERVAL);
162             logger.debug("-Dio.netty5.allocator.cacheTrimIntervalMillis: {}", DEFAULT_CACHE_TRIM_INTERVAL_MILLIS);
163             logger.debug("-Dio.netty5.allocator.useCacheForAllThreads: {}", DEFAULT_USE_CACHE_FOR_ALL_THREADS);
164             logger.debug("-Dio.netty5.allocator.maxCachedByteBuffersPerChunk: {}",
165                     DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK);
166         }
167     }

202     public PooledBufferAllocator(MemoryManager manager, boolean direct, int numArenas, int pageSize, int maxOrder,
203                                  int smallCacheSize, int normalCacheSize,
204                                  boolean useCacheForAllThreads, int directMemoryCacheAlignment) {
205         this.manager = requireNonNull(manager, "MemoryManager");
206         allocationType = direct? StandardAllocationTypes.OFF_HEAP : StandardAllocationTypes.ON_HEAP;
207         threadCache = new PoolThreadLocalCache(useCacheForAllThreads);
208         this.smallCacheSize = smallCacheSize;
209         this.normalCacheSize = normalCacheSize;
210 
211         if (directMemoryCacheAlignment != 0) {
212             if (!PlatformDependent.hasAlignDirectByteBuffer()) {
213                 throw new UnsupportedOperationException("Buffer alignment is not supported. " +
214                         "Either Unsafe or ByteBuffer.alignSlice() must be available.");
215             }
216 
217             // Ensure page size is a whole multiple of the alignment, or bump it to the next whole multiple.
218             pageSize = (int) PlatformDependent.align(pageSize, directMemoryCacheAlignment);
219         }
220 
221         chunkSize = validateAndCalculateChunkSize(pageSize, maxOrder);
222 
223         checkPositiveOrZero(numArenas, "numArenas");
224 
225         checkPositiveOrZero(directMemoryCacheAlignment, "directMemoryCacheAlignment");
226         if (directMemoryCacheAlignment > 0 && !isDirectMemoryCacheAlignmentSupported()) {
227             throw new IllegalArgumentException("directMemoryCacheAlignment is not supported");
228         }
229 
230         if ((directMemoryCacheAlignment & -directMemoryCacheAlignment) != directMemoryCacheAlignment) {
231             throw new IllegalArgumentException("directMemoryCacheAlignment: "
232                     + directMemoryCacheAlignment + " (expected: power of two)");
233         }
234 
235         int pageShifts = validateAndCalculatePageShifts(pageSize, directMemoryCacheAlignment);
236 
237         if (numArenas > 0) {
238             arenas = newArenaArray(numArenas);
239             List<PoolArenaMetric> metrics = new ArrayList<>(arenas.length);
240             for (int i = 0; i < arenas.length; i ++) {
241                 PoolArena arena = new PoolArena(this, manager, allocationType,
242                         pageSize, pageShifts, chunkSize,
243                         directMemoryCacheAlignment);
244                 arenas[i] = arena;
245                 metrics.add(arena);
246             }
247             arenaMetrics = metrics;
248             arenaMetricsView = Collections.unmodifiableList(metrics);
249         } else {
250             arenas = null;
251             arenaMetrics = new ArrayList<>(1);
252             arenaMetricsView = Collections.emptyList();
253         }
254 
255         metric = new PooledBufferAllocatorMetric(this);
256     }

262     private static int validateAndCalculatePageShifts(int pageSize, int alignment) {
263         if (pageSize < MIN_PAGE_SIZE) {
264             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: " + MIN_PAGE_SIZE + ')');
265         }
266 
267         if ((pageSize & pageSize - 1) != 0) {
268             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: power of 2)");
269         }
270 
271         if (pageSize < alignment) {
272             throw new IllegalArgumentException("Alignment cannot be greater than page size. " +
273                     "Alignment: " + alignment + ", page size: " + pageSize + '.');
274         }
275 
276         // Logarithm base 2. At this point we know that pageSize is a power of two.
277         return Integer.SIZE - 1 - Integer.numberOfLeadingZeros(pageSize);
278     }
279 
280     private static int validateAndCalculateChunkSize(int pageSize, int maxOrder) {
281         if (maxOrder > 14) {
282             throw new IllegalArgumentException("maxOrder: " + maxOrder + " (expected: 0-14)");
283         }
284 
285         // Ensure the resulting chunkSize does not overflow.
286         int chunkSize = pageSize;
287         for (int i = maxOrder; i > 0; i--) {
288             if (chunkSize > MAX_CHUNK_SIZE / 2) {
289                 throw new IllegalArgumentException(String.format(
290                         "pageSize (%d) << maxOrder (%d) must not exceed %d", pageSize, maxOrder, MAX_CHUNK_SIZE));
291             }
292             chunkSize <<= 1;
293         }
294         return chunkSize;
295     }

307     @Override
308     public Buffer allocate(int size) {
309         if (closed) {
310             throw allocatorClosedException();
311         }
312         Statics.assertValidBufferSize(size);
313         PooledAllocatorControl control = new PooledAllocatorControl(this);
314         UntetheredMemory memory = allocateUntethered(size);
315         Drop<Buffer> drop = memory.drop();
316         Buffer buffer = manager.recoverMemory(control, memory.memory(), drop);
317         drop.attach(buffer);
318         return buffer.fill((byte) 0);
319     }
320 
321     @Override
322     public Supplier<Buffer> constBufferSupplier(byte[] bytes) {
323         if (closed) {
324             throw allocatorClosedException();
325         }
326         PooledAllocatorControl control = new PooledAllocatorControl(this);
327         Buffer constantBuffer = manager.allocateShared(
328                 control, bytes.length, ArcDrop::wrap, allocationType);
329         constantBuffer.writeBytes(bytes).makeReadOnly();
330         return () -> manager.allocateConstChild(constantBuffer);
331     }

333     UntetheredMemory allocateUntethered(int size) {
334         PoolThreadCache cache = threadCache.get();
335         PoolArena arena = cache.getArena();
336 
337         if (arena != null) {
338             return arena.allocate(cache, size);
339         }
340         return allocateUnpooled(size);
341     }
342 
343     private UntetheredMemory allocateUnpooled(int size) {
344         return new UnpooledUntetheredMemory(this, manager, allocationType, size);
345     }

347     @Override
348     public void close() {
349         closed = true;
350         trimCurrentThreadCache();
351         threadCache.remove();
352         for (int i = 0, arenasLength = arenas.length; i < arenasLength; i++) {
353             PoolArena arena = arenas[i];
354             if (arena != null) {
355                 arena.close();
356                 arenas[i] = null;
357             }
358         }
359         arenaMetrics.clear();
360     }

433     final class PoolThreadLocalCache extends FastThreadLocal<PoolThreadCache> {
434         private final boolean useCacheForAllThreads;
435 
436         PoolThreadLocalCache(boolean useCacheForAllThreads) {
437             this.useCacheForAllThreads = useCacheForAllThreads;
438         }
439 
440         @Override
441         protected synchronized PoolThreadCache initialValue() {
442             final PoolArena arena = leastUsedArena(arenas);
443 
444             final Thread current = Thread.currentThread();
445             final EventExecutor executor = ThreadExecutorMap.currentExecutor();
446             if (useCacheForAllThreads ||
447                 // If the current thread is a FastThreadLocalThread we will always use the cache
448                 current instanceof FastThreadLocalThread ||
449                 // The Thread is used by an EventExecutor, let's use the cache as the chances are good that we
450                 // will allocate a lot!
451                 executor != null) {
452                 final PoolThreadCache cache = new PoolThreadCache(
453                         arena, smallCacheSize, normalCacheSize,
454                         DEFAULT_MAX_CACHED_BUFFER_CAPACITY, DEFAULT_CACHE_TRIM_INTERVAL);
455 
456                 if (DEFAULT_CACHE_TRIM_INTERVAL_MILLIS > 0) {
457                     if (executor != null) {
458                         executor.scheduleAtFixedRate(trimTask, DEFAULT_CACHE_TRIM_INTERVAL_MILLIS,
459                                                      DEFAULT_CACHE_TRIM_INTERVAL_MILLIS, TimeUnit.MILLISECONDS);
460                     }
461                 }
462                 return cache;
463             }
464             // No caching so just use 0 as sizes.
465             return new PoolThreadCache(arena, 0, 0, 0, 0);
466         }
467 
468         @Override
469         protected void onRemoval(PoolThreadCache threadCache) {
470             threadCache.free();
471         }
472     }
473 
474     static PoolArena leastUsedArena(PoolArena[] arenas) {
475         if (arenas == null || arenas.length == 0) {
476             return null;
477         }
478 
479         PoolArena minArena = arenas[0];
480         for (int i = 1; i < arenas.length; i++) {
481             PoolArena arena = arenas[i];
482             if (arena.numThreadCaches.get() < minArena.numThreadCaches.get()) {
483                 minArena = arena;
484             }
485         }
486 
487         return minArena;
488     }

543     private static long usedMemory(PoolArena[] arenas) {
544         if (arenas == null) {
545             return -1;
546         }
547         long used = 0;
548         for (PoolArena arena : arenas) {
549             used += arena.numActiveBytes();
550             if (used < 0) {
551                 return Long.MAX_VALUE;
552             }
553         }
554         return used;
555     }

557     final PoolThreadCache threadCache() {
558         PoolThreadCache cache =  threadCache.get();
559         assert cache != null;

563     /**
564      * Trim thread local cache for the current {@link Thread}, which will give back any cached memory that was not
565      * allocated frequently since the last trim operation.
566      *
567      * Returns {@code true} if a cache for the current {@link Thread} exists and so was trimmed, false otherwise.
568      */
569     public boolean trimCurrentThreadCache() {
570         PoolThreadCache cache = threadCache.getIfExists();
571         if (cache != null) {
572             cache.trim();
573             return true;
574         }
575         return false;
576     }

578     /**
579      * Returns the status of the allocator (which contains all metrics) as string. Be aware this may be expensive
580      * and so should not be called too frequently.
581      */
582     public String dumpStats() {
583         int heapArenasLen = arenas == null ? 0 : arenas.length;
584         StringBuilder buf = new StringBuilder(512)
585                 .append(heapArenasLen)
586                 .append(" arena(s):")
587                 .append(StringUtil.NEWLINE);
588         if (heapArenasLen > 0) {
589             for (PoolArena a: arenas) {
590                 buf.append(a);
591             }
592         }
593 
594         return buf.toString();
595     }
