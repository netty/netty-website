60      static {
61          int defaultAlignment = SystemPropertyUtil.getInt(
62                  "io.netty5.allocator.directMemoryCacheAlignment", 0);
63          int defaultPageSize = SystemPropertyUtil.getInt("io.netty5.allocator.pageSize", 8192);
64          Throwable pageSizeFallbackCause = null;
65          try {
66              validateAndCalculatePageShifts(defaultPageSize, defaultAlignment);
67          } catch (Throwable t) {
68              pageSizeFallbackCause = t;
69              defaultPageSize = 8192;
70              defaultAlignment = 0;
71          }
72          DEFAULT_PAGE_SIZE = defaultPageSize;
73          DEFAULT_DIRECT_MEMORY_CACHE_ALIGNMENT = defaultAlignment;
74  
75          int defaultMaxOrder = SystemPropertyUtil.getInt("io.netty5.allocator.maxOrder", 9);
76          Throwable maxOrderFallbackCause = null;
77          try {
78              validateAndCalculateChunkSize(DEFAULT_PAGE_SIZE, defaultMaxOrder);
79          } catch (Throwable t) {
80              maxOrderFallbackCause = t;
81              defaultMaxOrder = 11;
82          }
83          DEFAULT_MAX_ORDER = defaultMaxOrder;
84  
85          // Determine reasonable default for nHeapArena and nDirectArena.
86          // Assuming each arena has 3 chunks, the pool should not consume more than 50% of max memory.
87          final Runtime runtime = Runtime.getRuntime();
88  
89          /*
90           * We use 2 * available processors by default to reduce contention as we use 2 * available processors for the
91           * number of EventLoops in NIO and EPOLL as well. If we choose a smaller number we will run into hot spots as
92           * allocation and de-allocation needs to be synchronized on the PoolArena.
93           *
94           * See https://github.com/netty/netty/issues/3888.
95           */
96          final int defaultMinNumArena = NettyRuntime.availableProcessors() * 2;
97          final int defaultChunkSize = DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER;
98          DEFAULT_NUM_HEAP_ARENA = Math.max(0,
99                  SystemPropertyUtil.getInt(
100                         "io.netty5.allocator.numHeapArenas",
101                         (int) Math.min(
102                                 defaultMinNumArena,
103                                 runtime.maxMemory() / defaultChunkSize / 2 / 3)));
104         DEFAULT_NUM_DIRECT_ARENA = Math.max(0,
105                 SystemPropertyUtil.getInt(
106                         "io.netty5.allocator.numDirectArenas",
107                         (int) Math.min(
108                                 defaultMinNumArena,
109                                 PlatformDependent.maxDirectMemory() / defaultChunkSize / 2 / 3)));
110 
111         // cache sizes
112         DEFAULT_SMALL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty5.allocator.smallCacheSize", 256);
113         DEFAULT_NORMAL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty5.allocator.normalCacheSize", 64);
114 
115         // 32 kb is the default maximum capacity of the cached buffer. Similar to what is explained in
116         // 'Scalable memory allocation using jemalloc'
117         DEFAULT_MAX_CACHED_BUFFER_CAPACITY = SystemPropertyUtil.getInt(
118                 "io.netty5.allocator.maxCachedBufferCapacity", 32 * 1024);
119 
120         // the number of threshold of allocations when cached entries will be freed up if not frequently used
121         DEFAULT_CACHE_TRIM_INTERVAL = SystemPropertyUtil.getInt(
122                 "io.netty5.allocator.cacheTrimInterval", 8192);
123 
124         DEFAULT_CACHE_TRIM_INTERVAL_MILLIS = SystemPropertyUtil.getLong(
125                 "io.netty5.allocator.cacheTrimIntervalMillis", 0);
126 
127         DEFAULT_USE_CACHE_FOR_ALL_THREADS = SystemPropertyUtil.getBoolean(
128                 "io.netty5.allocator.useCacheForAllThreads", false);
129 
130         // Use 1023 by default as we use an ArrayDeque as backing storage which will then allocate an internal array
131         // of 1024 elements. Otherwise we would allocate 2048 and only use 1024 which is wasteful.
132         DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK = SystemPropertyUtil.getInt(
133                 "io.netty5.allocator.maxCachedByteBuffersPerChunk", 1023);
134 
135         if (logger.isDebugEnabled()) {
136             logger.debug("-Dio.netty5.allocator.numHeapArenas: {}", DEFAULT_NUM_HEAP_ARENA);
137             logger.debug("-Dio.netty5.allocator.numDirectArenas: {}", DEFAULT_NUM_DIRECT_ARENA);
138             if (pageSizeFallbackCause == null) {
139                 logger.debug("-Dio.netty5.allocator.pageSize: {}", DEFAULT_PAGE_SIZE);
140             } else {
141                 logger.debug("-Dio.netty5.allocator.pageSize: {}", DEFAULT_PAGE_SIZE, pageSizeFallbackCause);
142             }
143             if (maxOrderFallbackCause == null) {
144                 logger.debug("-Dio.netty5.allocator.maxOrder: {}", DEFAULT_MAX_ORDER);
145             } else {
146                 logger.debug("-Dio.netty5.allocator.maxOrder: {}", DEFAULT_MAX_ORDER, maxOrderFallbackCause);
147             }
148             logger.debug("-Dio.netty5.allocator.chunkSize: {}", DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER);
149             logger.debug("-Dio.netty5.allocator.smallCacheSize: {}", DEFAULT_SMALL_CACHE_SIZE);
150             logger.debug("-Dio.netty5.allocator.normalCacheSize: {}", DEFAULT_NORMAL_CACHE_SIZE);
151             logger.debug("-Dio.netty5.allocator.maxCachedBufferCapacity: {}", DEFAULT_MAX_CACHED_BUFFER_CAPACITY);
152             logger.debug("-Dio.netty5.allocator.cacheTrimInterval: {}", DEFAULT_CACHE_TRIM_INTERVAL);
153             logger.debug("-Dio.netty5.allocator.cacheTrimIntervalMillis: {}", DEFAULT_CACHE_TRIM_INTERVAL_MILLIS);
154             logger.debug("-Dio.netty5.allocator.useCacheForAllThreads: {}", DEFAULT_USE_CACHE_FOR_ALL_THREADS);
155             logger.debug("-Dio.netty5.allocator.maxCachedByteBuffersPerChunk: {}",
156                     DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK);
157         }
158     }

244     public PooledByteBufAllocator(boolean preferDirect, int nHeapArena, int nDirectArena, int pageSize, int maxOrder,
245                                   int smallCacheSize, int normalCacheSize,
246                                   boolean useCacheForAllThreads, int directMemoryCacheAlignment) {
247         super(preferDirect);
248         threadCache = new PoolThreadLocalCache(useCacheForAllThreads);
249         this.smallCacheSize = smallCacheSize;
250         this.normalCacheSize = normalCacheSize;
251 
252         if (directMemoryCacheAlignment != 0) {
253             if (!PlatformDependent.hasAlignDirectByteBuffer()) {
254                 throw new UnsupportedOperationException("Buffer alignment is not supported. " +
255                         "Either Unsafe or ByteBuffer.alignSlice() must be available.");
256             }
257 
258             // Ensure page size is a whole multiple of the alignment, or bump it to the next whole multiple.
259             pageSize = (int) PlatformDependent.align(pageSize, directMemoryCacheAlignment);
260         }
261 
262         chunkSize = validateAndCalculateChunkSize(pageSize, maxOrder);
263 
264         checkPositiveOrZero(nHeapArena, "nHeapArena");
265         checkPositiveOrZero(nDirectArena, "nDirectArena");
266 
267         checkPositiveOrZero(directMemoryCacheAlignment, "directMemoryCacheAlignment");
268         if (directMemoryCacheAlignment > 0 && !isDirectMemoryCacheAlignmentSupported()) {
269             throw new IllegalArgumentException("directMemoryCacheAlignment is not supported");
270         }
271 
272         if ((directMemoryCacheAlignment & -directMemoryCacheAlignment) != directMemoryCacheAlignment) {
273             throw new IllegalArgumentException("directMemoryCacheAlignment: "
274                     + directMemoryCacheAlignment + " (expected: power of two)");
275         }
276 
277         int pageShifts = validateAndCalculatePageShifts(pageSize, directMemoryCacheAlignment);
278 
279         if (nHeapArena > 0) {
280             heapArenas = newArenaArray(nHeapArena);
281             List<PoolArenaMetric> metrics = new ArrayList<>(heapArenas.length);
282             for (int i = 0; i < heapArenas.length; i ++) {
283                 PoolArena.HeapArena arena = new PoolArena.HeapArena(this,
284                         pageSize, pageShifts, chunkSize);
285                 heapArenas[i] = arena;
286                 metrics.add(arena);
287             }
288             heapArenaMetrics = Collections.unmodifiableList(metrics);
289         } else {
290             heapArenas = null;
291             heapArenaMetrics = Collections.emptyList();
292         }
293 
294         if (nDirectArena > 0) {
295             directArenas = newArenaArray(nDirectArena);
296             List<PoolArenaMetric> metrics = new ArrayList<>(directArenas.length);
297             for (int i = 0; i < directArenas.length; i ++) {
298                 PoolArena.DirectArena arena = new PoolArena.DirectArena(
299                         this, pageSize, pageShifts, chunkSize, directMemoryCacheAlignment);
300                 directArenas[i] = arena;
301                 metrics.add(arena);
302             }
303             directArenaMetrics = Collections.unmodifiableList(metrics);
304         } else {
305             directArenas = null;
306             directArenaMetrics = Collections.emptyList();
307         }
308         metric = new PooledByteBufAllocatorMetric(this);
309     }

316     private static int validateAndCalculatePageShifts(int pageSize, int alignment) {
317         if (pageSize < MIN_PAGE_SIZE) {
318             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: " + MIN_PAGE_SIZE + ')');
319         }
320 
321         if ((pageSize & pageSize - 1) != 0) {
322             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: power of 2)");
323         }
324 
325         if (pageSize < alignment) {
326             throw new IllegalArgumentException("Alignment cannot be greater than page size. " +
327                     "Alignment: " + alignment + ", page size: " + pageSize + '.');
328         }
329 
330         // Logarithm base 2. At this point we know that pageSize is a power of two.
331         return Integer.SIZE - 1 - Integer.numberOfLeadingZeros(pageSize);
332     }
333 
334     private static int validateAndCalculateChunkSize(int pageSize, int maxOrder) {
335         if (maxOrder > 14) {
336             throw new IllegalArgumentException("maxOrder: " + maxOrder + " (expected: 0-14)");
337         }
338 
339         // Ensure the resulting chunkSize does not overflow.
340         int chunkSize = pageSize;
341         for (int i = maxOrder; i > 0; i --) {
342             if (chunkSize > MAX_CHUNK_SIZE / 2) {
343                 throw new IllegalArgumentException(String.format(
344                         "pageSize (%d) << maxOrder (%d) must not exceed %d", pageSize, maxOrder, MAX_CHUNK_SIZE));
345             }
346             chunkSize <<= 1;
347         }
348         return chunkSize;
349     }

351     @Override
352     protected ByteBuf newHeapBuffer(int initialCapacity, int maxCapacity) {
353         PoolThreadCache cache = threadCache.get();
354         PoolArena<byte[]> heapArena = cache.heapArena;
355 
356         final ByteBuf buf;
357         if (heapArena != null) {
358             buf = heapArena.allocate(cache, initialCapacity, maxCapacity);
359         } else {
360             buf = PlatformDependent.hasUnsafe() ?
361                     new UnpooledUnsafeHeapByteBuf(this, initialCapacity, maxCapacity) :
362                     new UnpooledHeapByteBuf(this, initialCapacity, maxCapacity);
363         }
364 
365         return toLeakAwareBuffer(buf);
366     }
367 
368     @Override
369     protected ByteBuf newDirectBuffer(int initialCapacity, int maxCapacity) {
370         PoolThreadCache cache = threadCache.get();
371         PoolArena<ByteBuffer> directArena = cache.directArena;
372 
373         final ByteBuf buf;
374         if (directArena != null) {
375             buf = directArena.allocate(cache, initialCapacity, maxCapacity);
376         } else {
377             buf = PlatformDependent.hasUnsafe() ?
378                     UnsafeByteBufUtil.newUnsafeDirectByteBuf(this, initialCapacity, maxCapacity) :
379                     new UnpooledDirectByteBuf(this, initialCapacity, maxCapacity);
380         }
381 
382         return toLeakAwareBuffer(buf);
383     }

482     private final class PoolThreadLocalCache extends FastThreadLocal<PoolThreadCache> {
483         private final boolean useCacheForAllThreads;
484 
485         PoolThreadLocalCache(boolean useCacheForAllThreads) {
486             this.useCacheForAllThreads = useCacheForAllThreads;
487         }
488 
489         @Override
490         protected synchronized PoolThreadCache initialValue() {
491             final PoolArena<byte[]> heapArena = leastUsedArena(heapArenas);
492             final PoolArena<ByteBuffer> directArena = leastUsedArena(directArenas);
493 
494             final Thread current = Thread.currentThread();
495             final EventExecutor executor = ThreadExecutorMap.currentExecutor();
496 
497             if (useCacheForAllThreads ||
498                     // If the current thread is a FastThreadLocalThread we will always use the cache
499                     current instanceof FastThreadLocalThread ||
500                     // The Thread is used by an EventExecutor, let's use the cache as the chances are good that we
501                     // will allocate a lot!
502                     executor != null) {
503                 final PoolThreadCache cache = new PoolThreadCache(
504                         heapArena, directArena, smallCacheSize, normalCacheSize,
505                         DEFAULT_MAX_CACHED_BUFFER_CAPACITY, DEFAULT_CACHE_TRIM_INTERVAL);
506 
507                 if (DEFAULT_CACHE_TRIM_INTERVAL_MILLIS > 0) {
508                     if (executor != null) {
509                         executor.scheduleAtFixedRate(trimTask, DEFAULT_CACHE_TRIM_INTERVAL_MILLIS,
510                                 DEFAULT_CACHE_TRIM_INTERVAL_MILLIS, TimeUnit.MILLISECONDS);
511                     }
512                 }
513                 return cache;
514             }
515             // No caching so just use 0 as sizes.
516             return new PoolThreadCache(heapArena, directArena, 0, 0, 0, 0);
517         }
518 
519         @Override
520         protected void onRemoval(PoolThreadCache threadCache) {
521             threadCache.free(false);
522         }
523 
524         private <T> PoolArena<T> leastUsedArena(PoolArena<T>[] arenas) {
525             if (arenas == null || arenas.length == 0) {
526                 return null;
527             }
528 
529             PoolArena<T> minArena = arenas[0];
530             for (int i = 1; i < arenas.length; i++) {
531                 PoolArena<T> arena = arenas[i];
532                 if (arena.numThreadCaches.get() < minArena.numThreadCaches.get()) {
533                     minArena = arena;
534                 }
535             }
536 
537             return minArena;
538         }
539     }

654     private static long usedMemory(PoolArena<?>[] arenas) {
655         if (arenas == null) {
656             return -1;
657         }
658         long used = 0;
659         for (PoolArena<?> arena : arenas) {
660             used += arena.numActiveBytes();
661             if (used < 0) {
662                 return Long.MAX_VALUE;
663             }
664         }
665         return used;
666     }

688     private static long pinnedMemory(PoolArena<?>[] arenas) {
689         if (arenas == null) {
690             return -1;
691         }
692         long used = 0;
693         for (PoolArena<?> arena : arenas) {
694             used += arena.numPinnedBytes();

723     /**
724      * Returns the status of the allocator (which contains all metrics) as string. Be aware this may be expensive
725      * and so should not called too frequently.
726      */
727     public String dumpStats() {
728         int heapArenasLen = heapArenas == null ? 0 : heapArenas.length;
729         StringBuilder buf = new StringBuilder(512)
730                 .append(heapArenasLen)
731                 .append(" heap arena(s):")
732                 .append(StringUtil.NEWLINE);
733         if (heapArenasLen > 0) {
734             for (PoolArena<byte[]> a: heapArenas) {
735                 buf.append(a);
736             }
737         }
738 
739         int directArenasLen = directArenas == null ? 0 : directArenas.length;
740 
741         buf.append(directArenasLen)
742            .append(" direct arena(s):")
743            .append(StringUtil.NEWLINE);
744         if (directArenasLen > 0) {
745             for (PoolArena<ByteBuffer> a: directArenas) {
746                 buf.append(a);
747             }
748         }
749 
750         return buf.toString();
751     }
