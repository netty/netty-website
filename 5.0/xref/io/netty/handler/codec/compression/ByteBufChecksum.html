16  package io.netty5.handler.codec.compression;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import io.netty5.buffer.ByteBuf;
21  import io.netty5.util.ByteProcessor;
22  
23  import java.nio.ByteBuffer;
24  import java.util.zip.Adler32;
25  import java.util.zip.CRC32;
26  import java.util.zip.Checksum;
27  
28  /**
29   * {@link Checksum} implementation which can directly act on a {@link ByteBuf}.
30   *
31   * Implementations may optimize access patterns depending on if the {@link ByteBuf} is backed by a
32   * byte array ({@link ByteBuf#hasArray()} is {@code true}) or not.
33   */
34  abstract class ByteBufChecksum implements Checksum {
35  
36      private final ByteProcessor updateProcessor = value -> {
37          update(value);
38          return true;
39      };
40  
41      static ByteBufChecksum wrapChecksum(Checksum checksum) {
42          requireNonNull(checksum, "checksum");
43          if (checksum instanceof ByteBufChecksum) {
44              return (ByteBufChecksum) checksum;
45          }
46          if (checksum instanceof Adler32) {
47              return new OptimizedByteBufChecksum<Adler32>((Adler32) checksum) {
48                  @Override
49                  public void update(ByteBuffer b) {
50                      checksum.update(b);
51                  }
52              };
53          }
54          if (checksum instanceof CRC32) {
55              return new OptimizedByteBufChecksum<CRC32>((CRC32) checksum) {
56                  @Override
57                  public void update(ByteBuffer b) {
58                      checksum.update(b);
59                  }
60              };
61          }
62          return new SlowByteBufChecksum<>(checksum);
63      }
64  
65      /**
66       * @see #update(byte[], int, int)
67       */
68      public void update(ByteBuf b, int off, int len) {
69          if (b.hasArray()) {
70              update(b.array(), b.arrayOffset() + off, len);
71          } else {
72              b.forEachByte(off, len, updateProcessor);
73          }
74      }
75  
76      private abstract static class OptimizedByteBufChecksum<T extends Checksum> extends SlowByteBufChecksum<T> {
77          OptimizedByteBufChecksum(T checksum) {
78              super(checksum);
79          }
80  
81          @Override
82          public void update(ByteBuf b, int off, int len) {
83              if (b.hasArray()) {
84                  update(b.array(), b.arrayOffset() + off, len);
85              } else {
86                  try {
87                      update(CompressionUtil.safeNioBuffer(b, off, len));
88                  } catch (Throwable cause) {
89                      throw new Error();
90                  }
91              }
92          }
93  
94          public abstract void update(ByteBuffer b);
95      }
96  
97      private static class SlowByteBufChecksum<T extends Checksum> extends ByteBufChecksum {
98  
99          protected final T checksum;
100 
101         SlowByteBufChecksum(T checksum) {
102             this.checksum = checksum;
103         }
104 
105         @Override
106         public void update(int b) {
107             checksum.update(b);
108         }
109 
110         @Override
111         public void update(byte[] b, int off, int len) {
112             checksum.update(b, off, len);
113         }
114 
115         @Override
116         public long getValue() {
117             return checksum.getValue();
118         }
119 
120         @Override
121         public void reset() {
122             checksum.reset();
123         }
124     }
125 }
