31  /**
32   * Decompress a {@link ByteBuf} using the inflate algorithm.
33   */
34  public final class ZlibDecompressor implements Decompressor {
35      private static final int FHCRC = 0x02;
36      private static final int FEXTRA = 0x04;
37      private static final int FNAME = 0x08;
38      private static final int FCOMMENT = 0x10;
39      private static final int FRESERVED = 0xE0;
40  
41      private Inflater inflater;
42      private final byte[] dictionary;
43  
44      // GZIP related
45      private final ByteBufChecksum crc;
46      private final boolean decompressConcatenated;
47  
48      /**
49       * Maximum allowed size of the decompression buffer.
50       */
51      protected final int maxAllocation;
52  
53      private enum GzipState {
54          HEADER_START,
55          HEADER_END,
56          FLG_READ,
57          XLEN_READ,
58          SKIP_FNAME,
59          SKIP_COMMENT,
60          PROCESS_FHCRC,
61          FOOTER_START,
62      }
63  
64      private GzipState gzipState = GzipState.HEADER_START;
65      private int flags = -1;
66      private int xlen = -1;
67  
68      private boolean finished;
69      private boolean closed;
70  
71      private boolean decideZlibOrNone;
72  
73      private ZlibDecompressor(ZlibWrapper wrapper, byte[] dictionary, boolean decompressConcatenated,
74                               int maxAllocation) {
75          this.maxAllocation = maxAllocation;
76          this.decompressConcatenated = decompressConcatenated;
77          switch (wrapper) {
78              case GZIP:
79                  inflater = new Inflater(true);
80                  crc = ByteBufChecksum.wrapChecksum(new CRC32());
81                  break;
82              case NONE:
83                  inflater = new Inflater(true);
84                  crc = null;
85                  break;
86              case ZLIB:
87                  inflater = new Inflater();
88                  crc = null;
89                  break;
90              case ZLIB_OR_NONE:
91                  // Postpone the decision until decode(...) is called.
92                  decideZlibOrNone = true;
93                  crc = null;
94                  break;
95              default:
96                  throw new IllegalArgumentException("Only GZIP or ZLIB is supported, but you used " + wrapper);
97          }
98          this.dictionary = dictionary;
99      }

196     @Override
197     public ByteBuf decompress(ByteBuf in, ByteBufAllocator allocator) throws DecompressionException {
198         if (closed) {
199             throw new DecompressionException("Decompressor closed");
200         }
201         if (finished) {
202             return Unpooled.EMPTY_BUFFER;
203         }
204 
205         int readableBytes = in.readableBytes();
206         if (readableBytes == 0) {
207             return null;
208         }
209 
210         if (decideZlibOrNone) {
211             // First two bytes are needed to decide if it's a ZLIB stream.
212             if (readableBytes < 2) {
213                 return null;
214             }
215 
216             boolean nowrap = !looksLikeZlib(in.getShort(in.readerIndex()));
217             inflater = new Inflater(nowrap);
218             decideZlibOrNone = false;
219         }
220 
221         if (crc != null) {
222             if (gzipState != GzipState.HEADER_END) {
223                 if (gzipState == GzipState.FOOTER_START) {
224                     if (!handleGzipFooter(in)) {
225                         // Either there was not enough data or the input is finished.
226                         return null;
227                     }
228                     // If we consumed the footer we will start with the header again.
229                     assert gzipState == GzipState.HEADER_START;
230                 }
231                 if (!readGZIPHeader(in)) {
232                     // There was not enough data readable to read the GZIP header.
233                     return null;
234                 }
235                 // Some bytes may have been consumed, and so we must re-set the number of readable bytes.
236                 readableBytes = in.readableBytes();
237                 if (readableBytes == 0) {
238                     return null;
239                 }
240             }
241         }
242 
243         if (inflater.needsInput()) {
244             if (in.hasArray()) {
245                 inflater.setInput(in.array(), in.arrayOffset() + in.readerIndex(), readableBytes);
246             } else {
247                 if (in.nioBufferCount() == 1) {
248                     inflater.setInput(in.internalNioBuffer(in.readerIndex(), readableBytes));
249                 } else {
250                     inflater.setInput(in.nioBuffer(in.readerIndex(), readableBytes));
251                 }
252             }
253         }
254 
255         ByteBuf decompressed = prepareDecompressBuffer(allocator, null, inflater.getRemaining() << 1);
256         try {
257             boolean readFooter = false;
258             while (!inflater.needsInput()) {
259                 int writerIndex = decompressed.writerIndex();
260                 int writable = decompressed.writableBytes();
261                 int outputLength;
262                 if (decompressed.hasArray()) {
263                     byte[] outArray = decompressed.array();
264                     int outIndex = decompressed.arrayOffset() + writerIndex;
265                     outputLength = inflater.inflate(outArray, outIndex, writable);
266                 } else if (decompressed.nioBufferCount() == 1) {
267                     ByteBuffer buffer = decompressed.internalNioBuffer(writerIndex, writable);
268                     outputLength = inflater.inflate(buffer);
269                 } else {
270                     throw new IllegalStateException(
271                             "Decompress buffer must have array or exactly 1 NIO buffer: " + decompressed);
272                 }
273                 if (outputLength > 0) {
274                     decompressed.writerIndex(writerIndex + outputLength);
275                     if (crc != null) {
276                         crc.update(decompressed, writerIndex, outputLength);
277                     }
278                 } else  if (inflater.needsDictionary()) {
279                     if (dictionary == null) {
280                         throw new DecompressionException(
281                                 "decompression failure, unable to set dictionary as non was specified");
282                     }
283                     inflater.setDictionary(dictionary);
284                 }
285 
286                 if (inflater.finished()) {
287                     if (crc == null) {
288                         finished = true; // Do not decode anymore.
289                     } else {
290                         readFooter = true;
291                     }
292                     break;
293                 } else {
294                     decompressed = prepareDecompressBuffer(allocator, decompressed, inflater.getRemaining() << 1);
295                 }
296             }
297 
298             in.skipBytes(readableBytes - inflater.getRemaining());
299 
300             if (readFooter) {
301                 gzipState = GzipState.FOOTER_START;
302                 handleGzipFooter(in);
303             }
304 
305             if (decompressed.isReadable()) {
306                 return decompressed;
307             } else {
308                 decompressed.release();
309                 return null;
310             }
311         } catch (DataFormatException e) {
312             decompressed.release();
313             throw new DecompressionException("decompression failure", e);
314         } catch (Throwable cause) {
315             decompressed.release();
316             throw cause;
317         }
318     }

320     private boolean handleGzipFooter(ByteBuf in) {
321         if (readGZIPFooter(in)) {
322             finished = !decompressConcatenated;
323 
324             if (!finished) {
325                 inflater.reset();
326                 crc.reset();
327                 gzipState = GzipState.HEADER_START;
328                 return true;
329             }
330         }
331         return false;
332     }

334     private boolean readGZIPHeader(ByteBuf in) {
335         switch (gzipState) {
336             case HEADER_START:
337                 if (in.readableBytes() < 10) {
338                     return false;
339                 }
340                 // read magic numbers
341                 int magic0 = in.readByte();
342                 int magic1 = in.readByte();
343 
344                 if (magic0 != 31) {
345                     throw new DecompressionException("Input is not in the GZIP format");
346                 }
347                 crc.update(magic0);
348                 crc.update(magic1);
349 
350                 int method = in.readUnsignedByte();
351                 if (method != Deflater.DEFLATED) {
352                     throw new DecompressionException("Unsupported compression method "
353                             + method + " in the GZIP header");
354                 }
355                 crc.update(method);
356 
357                 flags = in.readUnsignedByte();
358                 crc.update(flags);
359 
360                 if ((flags & FRESERVED) != 0) {
361                     throw new DecompressionException(
362                             "Reserved flags are set in the GZIP header");
363                 }
364 
365                 // mtime (int)
366                 crc.update(in, in.readerIndex(), 4);
367                 in.skipBytes(4);
368 
369                 crc.update(in.readUnsignedByte()); // extra flags
370                 crc.update(in.readUnsignedByte()); // operating system
371 
372                 gzipState = GzipState.FLG_READ;
373                 // fall through
374             case FLG_READ:
375                 if ((flags & FEXTRA) != 0) {
376                     if (in.readableBytes() < 2) {
377                         return false;
378                     }
379                     int xlen1 = in.readUnsignedByte();
380                     int xlen2 = in.readUnsignedByte();
381                     crc.update(xlen1);
382                     crc.update(xlen2);
383 
384                     xlen |= xlen1 << 8 | xlen2;
385                 }
386                 gzipState = GzipState.XLEN_READ;
387                 // fall through
388             case XLEN_READ:
389                 if (xlen != -1) {
390                     if (in.readableBytes() < xlen) {
391                         return false;
392                     }
393                     crc.update(in, in.readerIndex(), xlen);
394                     in.skipBytes(xlen);
395                 }
396                 gzipState = GzipState.SKIP_FNAME;
397                 // fall through
398             case SKIP_FNAME:
399                 if (!skipIfNeeded(in, FNAME)) {
400                     return false;
401                 }
402                 gzipState = GzipState.SKIP_COMMENT;
403                 // fall through
404             case SKIP_COMMENT:
405                 if (!skipIfNeeded(in, FCOMMENT)) {
406                     return false;
407                 }
408                 gzipState = GzipState.PROCESS_FHCRC;
409                 // fall through
410             case PROCESS_FHCRC:
411                 if ((flags & FHCRC) != 0) {
412                     if (!verifyCrc16(in)) {
413                         return false;
414                     }
415                 }
416                 crc.reset();
417                 gzipState = GzipState.HEADER_END;
418                 // fall through
419             case HEADER_END:
420                 return true;
421             default:
422                 throw new IllegalStateException();
423         }
424     }

426     /**
427      * Skip bytes in the input if needed until we find the end marker {@code 0x00}.
428      * @param   in the input
429      * @param   flagMask the mask that should be present in the {@code flags} when we need to skip bytes.
430      * @return  {@code true} if the operation is complete and we can move to the next state, {@code false} if
431      * we need to retry again once we have more readable bytes.
432      */
433     private boolean skipIfNeeded(ByteBuf in, int flagMask) {
434         if ((flags & flagMask) != 0) {
435             for (;;) {
436                 if (!in.isReadable()) {
437                     // We didnt find the end yet, need to retry again once more data is readable
438                     return false;
439                 }
440                 int b = in.readUnsignedByte();
441                 crc.update(b);
442                 if (b == 0x00) {
443                     break;
444                 }
445             }
446         }
447         // Skip is handled, we can move to the next processing state.
448         return true;
449     }

451     /**
452      * Read the GZIP footer.
453      *
454      * @param   in the input.
455      * @return  {@code true} if the footer could be read, {@code false} if the read could not be performed as
456      *          the input {@link ByteBuf} doesn't have enough readable bytes (8 bytes).
457      */
458     private boolean readGZIPFooter(ByteBuf in) {
459         if (in.readableBytes() < 8) {
460             return false;
461         }
462 
463         boolean enoughData = verifyCrc(in);
464         assert enoughData;
465 
466         // read ISIZE and verify
467         int dataLength = 0;
468         for (int i = 0; i < 4; ++i) {
469             dataLength |= in.readUnsignedByte() << i * 8;
470         }
471         int readLength = inflater.getTotalOut();
472         if (dataLength != readLength) {
473             throw new DecompressionException(
474                     "Number of bytes mismatch. Expected: " + dataLength + ", Got: " + readLength);
475         }
476         return true;
477     }

479     /**
480      * Verifies CRC.
481      *
482      * @param   in the input.
483      * @return  {@code true} if verification could be performed, {@code false} if verification could not be
484      * performed as the input {@link ByteBuf} doesn't have enough readable bytes (4 bytes).
485      */
486     private boolean verifyCrc(ByteBuf in) {
487         if (in.readableBytes() < 4) {
488             return false;
489         }
490         long crcValue = 0;
491         for (int i = 0; i < 4; ++i) {
492             crcValue |= (long) in.readUnsignedByte() << i * 8;
493         }
494         long readCrc = crc.getValue();
495         if (crcValue != readCrc) {
496             throw new DecompressionException(
497                     "CRC value mismatch. Expected: " + crcValue + ", Got: " + readCrc);
498         }
499         return true;
500     }
501 
502     private boolean verifyCrc16(ByteBuf in) {
503         if (in.readableBytes() < 2) {
504             return false;
505         }
506         long readCrc32 = crc.getValue();
507         long crc16Value = 0;
508         long readCrc16 = 0; // the two least significant bytes from the CRC32
509         for (int i = 0; i < 2; ++i) {
510             crc16Value |= (long) in.readUnsignedByte() << (i * 8);
511             readCrc16 |= ((readCrc32 >> (i * 8)) & 0xff) << (i * 8);
512         }
513 
514         if (crc16Value != readCrc16) {
515             throw new DecompressionException(
516                     "CRC16 value mismatch. Expected: " + crc16Value + ", Got: " + readCrc16);
517         }
518         return true;
519     }

521     /*
522      * Returns true if the cmf_flg parameter (think: first two bytes of a zlib stream)
523      * indicates that this is a zlib stream.
524      * <p>
525      * You can lookup the details in the ZLIB RFC:
526      * <a href="https://tools.ietf.org/html/rfc1950#section-2.2">RFC 1950</a>.
527      */
528     private static boolean looksLikeZlib(short cmf_flg) {
529         return (cmf_flg & 0x7800) == 0x7800 &&
530                 cmf_flg % 31 == 0;
531     }

533     /**
534      * Allocate or expand the decompression buffer, without exceeding the maximum allocation.
535      * Calls {@link #decompressionBufferExhausted(ByteBuf)} if the buffer is full and cannot be expanded further.
536      */
537     protected ByteBuf prepareDecompressBuffer(ByteBufAllocator allocator, ByteBuf buffer, int preferredSize) {
538         if (buffer == null) {
539             if (maxAllocation == 0) {
540                 return allocator.buffer(preferredSize);
541             }
542 
543             return allocator.buffer(Math.min(preferredSize, maxAllocation), maxAllocation);
544         }
545 
546         // this always expands the buffer if possible, even if the expansion is less than preferredSize
547         // we throw the exception only if the buffer could not be expanded at all
548         // this means that one final attempt to deserialize will always be made with the buffer at maxAllocation
549         if (buffer.ensureWritable(preferredSize, true) == 1) {
550             // buffer must be consumed so subclasses don't add it to output
551             // we therefore duplicate it when calling decompressionBufferExhausted() to guarantee non-interference
552             // but wait until after to consume it so the subclass can tell how much output is really in the buffer
553             decompressionBufferExhausted(buffer.duplicate());
554             buffer.skipBytes(buffer.readableBytes());
555             throw new DecompressionException(
556                     "Decompression buffer has reached maximum size: " + buffer.maxCapacity());
557         }
558 
559         return buffer;
560     }
561 
562     protected void decompressionBufferExhausted(ByteBuf buffer) {
563         finished = true;
564     }
