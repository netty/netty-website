48  /**
49   * Compresses a {@link ByteBuf} using the LZ4 format.
50   *
51   * See original <a href="https://github.com/Cyan4973/lz4">LZ4 Github project</a>
52   * and <a href="https://fastcompression.blogspot.ru/2011/05/lz4-explained.html">LZ4 block format</a>
53   * for full description.
54   *
55   * Since the original LZ4 block format does not contains size of compressed block and size of original data
56   * this encoder uses format like <a href="https://github.com/idelpivnitskiy/lz4-java">LZ4 Java</a> library
57   * written by Adrien Grand and approved by Yann Collet (author of original LZ4 library).
58   *
59   *  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *     * * * * * * * * * *
60   *  * Magic * Token *  Compressed *  Decompressed *  Checksum *  +  *  LZ4 compressed *
61   *  *       *       *    length   *     length    *           *     *      block      *
62   *  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *     * * * * * * * * * *
63   */
64  public final class Lz4Compressor implements Compressor {
65      static final int DEFAULT_MAX_ENCODE_SIZE = Integer.MAX_VALUE;
66  
67      private final int blockSize;
68  
69      /**
70       * Underlying compressor in use.
71       */
72      private final LZ4Compressor compressor;
73  
74      /**
75       * Underlying checksum calculator in use.
76       */
77      private final ByteBufChecksum checksum;
78  
79      /**
80       * Compression level of current LZ4 encoder (depends on {@link #blockSize}).
81       */
82      private final int compressionLevel;
83  
84      /**
85       * Maximum size for any buffer to write encoded (compressed) data into.
86       */
87      private final int maxEncodeSize;
88  
89      private enum State {
90          PROCESSING,
91          FINISHED,
92          CLOSED
93      }
94  
95      private State state = State.PROCESSING;
96  
97      /**
98       * Creates the fastest LZ4 compressor factory with default block size (64 KB)
99       * and xxhash hashing for Java, based on Yann Collet's work available at
100      * <a href="https://github.com/Cyan4973/xxHash">Github</a>.
101      */
102     public static Supplier<Lz4Compressor> newFactory() {
103         return newFactory(false);
104     }
105 
106     /**
107      * Creates a new LZ4 compressor factory with high or fast compression, default block size (64 KB)
108      * and xxhash hashing for Java, based on Yann Collet's work available at
109      * <a href="https://github.com/Cyan4973/xxHash">Github</a>.
110      *
111      * @param highCompressor  if {@code true} codec will use compressor which requires more memory
112      *                        and is slower but compresses more efficiently
113      * @return the factory.
114      */
115     public static Supplier<Lz4Compressor> newFactory(boolean highCompressor) {
116         return newFactory(LZ4Factory.fastestInstance(), highCompressor,
117                 DEFAULT_BLOCK_SIZE, new Lz4XXHash32(DEFAULT_SEED));
118     }

120     /**
121      * Creates a new customizable LZ4 compressor factory.
122      *
123      * @param factory         user customizable {@link LZ4Factory} instance
124      *                        which may be JNI bindings to the original C implementation, a pure Java implementation
125      *                        or a Java implementation that uses the {@link sun.misc.Unsafe}
126      * @param highCompressor  if {@code true} codec will use compressor which requires more memory
127      *                        and is slower but compresses more efficiently
128      * @param blockSize       the maximum number of bytes to try to compress at once,
129      *                        must be >= 64 and <= 32 M
130      * @param checksum        the {@link Checksum} instance to use to check data for integrity
131      * @return the factory.
132      */
133     public static Supplier<Lz4Compressor> newFactory(LZ4Factory factory, boolean highCompressor,
134                                                      int blockSize, Checksum checksum) {
135         return newFactory(factory, highCompressor, blockSize, checksum, DEFAULT_MAX_ENCODE_SIZE);
136     }

138     /**
139      * Creates a new customizable LZ4 compressor factory.
140      *
141      * @param factory         user customizable {@link LZ4Factory} instance
142      *                        which may be JNI bindings to the original C implementation, a pure Java implementation
143      *                        or a Java implementation that uses the {@link sun.misc.Unsafe}
144      * @param highCompressor  if {@code true} codec will use compressor which requires more memory
145      *                        and is slower but compresses more efficiently
146      * @param blockSize       the maximum number of bytes to try to compress at once,
147      *                        must be >= 64 and <= 32 M
148      * @param checksum        the {@link Checksum} instance to use to check data for integrity
149      * @param maxEncodeSize   the maximum size for an encode (compressed) buffer
150      * @return the factory.
151      */
152     public static Supplier<Lz4Compressor> newFactory(LZ4Factory factory, boolean highCompressor, int blockSize,
153                                                      Checksum checksum, int maxEncodeSize) {
154         requireNonNull(factory, "factory");
155         requireNonNull(checksum, "checksum");
156         ObjectUtil.checkPositive(blockSize, "blockSize");
157         ObjectUtil.checkPositive(maxEncodeSize, "maxEncodeSize");
158         return () -> new Lz4Compressor(factory, highCompressor, blockSize, checksum, maxEncodeSize);
159     }
160 
161     private Lz4Compressor(LZ4Factory factory, boolean highCompressor, int blockSize,
162                           Checksum checksum, int maxEncodeSize) {
163         compressor = highCompressor ? factory.highCompressor() : factory.fastCompressor();
164         this.checksum = ByteBufChecksum.wrapChecksum(checksum);
165 
166         compressionLevel = compressionLevel(blockSize);
167         this.blockSize = blockSize;
168         this.maxEncodeSize = maxEncodeSize;
169     }
170 
171     /**
172      * Calculates compression level on the basis of block size.
173      */
174     private static int compressionLevel(int blockSize) {
175         if (blockSize < MIN_BLOCK_SIZE || blockSize > MAX_BLOCK_SIZE) {
176             throw new IllegalArgumentException(String.format(
177                     "blockSize: %d (expected: %d-%d)", blockSize, MIN_BLOCK_SIZE, MAX_BLOCK_SIZE));
178         }
179         int compressionLevel = 32 - Integer.numberOfLeadingZeros(blockSize - 1); // ceil of log2
180         compressionLevel = Math.max(0, compressionLevel - COMPRESSION_LEVEL_BASE);
181         return compressionLevel;
182     }
183 
184     private ByteBuf allocateBuffer(ByteBufAllocator allocator, ByteBuf msg) {
185         int targetBufSize = 0;
186         int remaining = msg.readableBytes();
187 
188         // quick overflow check
189         if (remaining < 0) {
190             throw new EncoderException("too much data to allocate a buffer for compression");
191         }
192 
193         while (remaining > 0) {
194             int curSize = Math.min(blockSize, remaining);
195             remaining -= curSize;
196             // calculate the total compressed size of the current block (including header) and add to the total
197             targetBufSize += compressor.maxCompressedLength(curSize) + HEADER_LENGTH;
198         }
199 
200         // in addition to just the raw byte count, the headers (HEADER_LENGTH) per block (configured via
201         // #blockSize) will also add to the targetBufSize, and the combination of those would never wrap around
202         // again to be >= 0, this is a good check for the overflow case.
203         if (targetBufSize > maxEncodeSize || 0 > targetBufSize) {
204             throw new EncoderException(String.format("requested encode buffer size (%d bytes) exceeds the maximum " +
205                                                      "allowable size (%d bytes)", targetBufSize, maxEncodeSize));
206         }
207 
208         return allocator.buffer(targetBufSize);
209     }

211     @Override
212     public ByteBuf compress(ByteBuf input, ByteBufAllocator allocator) throws CompressionException {
213         switch (state) {
214             case CLOSED:
215                 throw new CompressionException("Compressor closed");
216             case FINISHED:
217                 return Unpooled.EMPTY_BUFFER;
218             case PROCESSING:
219                 if (!input.isReadable()) {
220                     return Unpooled.EMPTY_BUFFER;
221                 }
222 
223                 ByteBuf out = allocateBuffer(allocator, input);
224                 try {
225                     // We need to compress as long as we have input to read as we are limited by the blockSize that
226                     // is used.
227                     while (input.isReadable()) {
228                         compressData(input, out);
229                     }
230                 } catch (Throwable cause) {
231                     out.release();
232                     throw cause;
233                 }
234                 return out;
235             default:
236                 throw new IllegalStateException();
237         }
238     }
239 
240     @Override
241     public ByteBuf finish(ByteBufAllocator allocator) {
242         switch (state) {
243             case CLOSED:
244                 throw new CompressionException("Compressor closed");
245             case FINISHED:
246             case PROCESSING:
247                 state = State.FINISHED;
248 
249                 final ByteBuf footer = allocator.buffer(HEADER_LENGTH);
250                 footer.ensureWritable(HEADER_LENGTH);
251                 final int idx = footer.writerIndex();
252                 footer.setLong(idx, MAGIC_NUMBER);
253                 footer.setByte(idx + TOKEN_OFFSET, (byte) (BLOCK_TYPE_NON_COMPRESSED | compressionLevel));
254                 footer.setInt(idx + COMPRESSED_LENGTH_OFFSET, 0);
255                 footer.setInt(idx + DECOMPRESSED_LENGTH_OFFSET, 0);
256                 footer.setInt(idx + CHECKSUM_OFFSET, 0);
257 
258                 footer.writerIndex(idx + HEADER_LENGTH);
259                 return footer;
260             default:
261                 throw new IllegalStateException();
262         }
263     }
264 
265     @Override
266     public boolean isFinished() {
267         return state != State.PROCESSING;
268     }
269 
270     @Override
271     public boolean isClosed() {
272         return state == State.CLOSED;
273     }
274 
275     @Override
276     public void close() {
277         state = State.CLOSED;
278     }

280     /**
281      *
282      * Encodes the input buffer into {@link #blockSize} chunks in the output buffer.
283      */
284     private void compressData(ByteBuf in, ByteBuf out) {
285         int inReaderIndex = in.readerIndex();
286         int flushableBytes = Math.min(in.readableBytes(), blockSize);
287         assert flushableBytes > 0;
288         checksum.reset();
289         checksum.update(in, inReaderIndex, flushableBytes);
290         final int check = (int) checksum.getValue();
291 
292         final int bufSize = compressor.maxCompressedLength(flushableBytes) + HEADER_LENGTH;
293         out.ensureWritable(bufSize);
294         final int idx = out.writerIndex();
295         int compressedLength;
296         try {
297             ByteBuffer outNioBuffer = out.internalNioBuffer(idx + HEADER_LENGTH, out.writableBytes() - HEADER_LENGTH);
298             int pos = outNioBuffer.position();
299             // We always want to start at position 0 as we take care of reusing the buffer in the encode(...) loop.
300             compressor.compress(in.internalNioBuffer(inReaderIndex, flushableBytes), outNioBuffer);
301             compressedLength = outNioBuffer.position() - pos;
302         } catch (LZ4Exception e) {
303             throw new CompressionException(e);
304         }
305         final int blockType;
306         if (compressedLength >= flushableBytes) {
307             blockType = BLOCK_TYPE_NON_COMPRESSED;
308             compressedLength = flushableBytes;
309             out.setBytes(idx + HEADER_LENGTH, in, inReaderIndex, flushableBytes);
310         } else {
311             blockType = BLOCK_TYPE_COMPRESSED;
312         }
313 
314         out.setLong(idx, MAGIC_NUMBER);
315         out.setByte(idx + TOKEN_OFFSET, (byte) (blockType | compressionLevel));
316         out.setIntLE(idx + COMPRESSED_LENGTH_OFFSET, compressedLength);
317         out.setIntLE(idx + DECOMPRESSED_LENGTH_OFFSET, flushableBytes);
318         out.setIntLE(idx + CHECKSUM_OFFSET, check);
319         out.writerIndex(idx + HEADER_LENGTH + compressedLength);
320 
321         in.readerIndex(inReaderIndex + flushableBytes);
322     }
