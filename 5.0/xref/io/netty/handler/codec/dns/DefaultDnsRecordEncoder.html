16  package io.netty5.handler.codec.dns;
17  
18  import io.netty5.buffer.ByteBuf;
19  import io.netty5.channel.socket.InternetProtocolFamily;
20  import io.netty5.handler.codec.UnsupportedMessageTypeException;
21  import io.netty5.util.internal.StringUtil;
22  import io.netty5.util.internal.UnstableApi;
23  
24  /**
25   * The default {@link DnsRecordEncoder} implementation.
26   *
27   * @see DefaultDnsRecordDecoder
28   */
29  @UnstableApi
30  public class DefaultDnsRecordEncoder implements DnsRecordEncoder {
31      private static final int PREFIX_MASK = Byte.SIZE - 1;
32  
33      /**
34       * Creates a new instance.
35       */
36      protected DefaultDnsRecordEncoder() { }
37  
38      @Override
39      public final void encodeQuestion(DnsQuestion question, ByteBuf out) throws Exception {
40          encodeName(question.name(), out);
41          out.writeShort(question.type().intValue());
42          out.writeShort(question.dnsClass());
43      }
44  
45      @Override
46      public void encodeRecord(DnsRecord record, ByteBuf out) throws Exception {
47          if (record instanceof DnsQuestion) {
48              encodeQuestion((DnsQuestion) record, out);
49          } else if (record instanceof DnsPtrRecord) {
50              encodePtrRecord((DnsPtrRecord) record, out);
51          } else if (record instanceof DnsOptEcsRecord) {
52              encodeOptEcsRecord((DnsOptEcsRecord) record, out);
53          } else if (record instanceof DnsOptPseudoRecord) {
54              encodeOptPseudoRecord((DnsOptPseudoRecord) record, out);
55          } else if (record instanceof DnsRawRecord) {
56              encodeRawRecord((DnsRawRecord) record, out);
57          } else {
58              throw new UnsupportedMessageTypeException(StringUtil.simpleClassName(record));
59          }
60      }
61  
62      private void encodeRecord0(DnsRecord record, ByteBuf out) throws Exception {
63          encodeName(record.name(), out);
64          out.writeShort(record.type().intValue());
65          out.writeShort(record.dnsClass());
66          out.writeInt((int) record.timeToLive());
67      }
68  
69      private void encodePtrRecord(DnsPtrRecord record, ByteBuf out) throws Exception {
70          encodeRecord0(record, out);
71          encodeName(record.hostname(), out);
72      }
73  
74      private void encodeOptPseudoRecord(DnsOptPseudoRecord record, ByteBuf out) throws Exception {
75          encodeRecord0(record, out);
76          out.writeShort(0);
77      }
78  
79      private void encodeOptEcsRecord(DnsOptEcsRecord record, ByteBuf out) throws Exception {
80          encodeRecord0(record, out);
81  
82          int sourcePrefixLength = record.sourcePrefixLength();
83          int scopePrefixLength = record.scopePrefixLength();
84          int lowOrderBitsToPreserve = sourcePrefixLength & PREFIX_MASK;
85  
86          byte[] bytes = record.address();
87          int addressBits = bytes.length << 3;
88          if (addressBits < sourcePrefixLength || sourcePrefixLength < 0) {
89              throw new IllegalArgumentException(sourcePrefixLength + ": " +
90                      sourcePrefixLength + " (expected: 0 >= " + addressBits + ')');
91          }
92  
93          // See https://www.iana.org/assignments/address-family-numbers/address-family-numbers.xhtml
94          final short addressNumber = (short) (bytes.length == 4 ?
95                  InternetProtocolFamily.IPv4.addressNumber() : InternetProtocolFamily.IPv6.addressNumber());
96          int payloadLength = calculateEcsAddressLength(sourcePrefixLength, lowOrderBitsToPreserve);
97  
98          int fullPayloadLength = 2 + // OPTION-CODE
99                  2 + // OPTION-LENGTH
100                 2 + // FAMILY
101                 1 + // SOURCE PREFIX-LENGTH
102                 1 + // SCOPE PREFIX-LENGTH
103                 payloadLength; //  ADDRESS...
104 
105         out.writeShort(fullPayloadLength);
106         out.writeShort(8); // This is the defined type for ECS.
107 
108         out.writeShort(fullPayloadLength - 4); // Not include OPTION-CODE and OPTION-LENGTH
109         out.writeShort(addressNumber);
110         out.writeByte(sourcePrefixLength);
111         out.writeByte(scopePrefixLength); // Must be 0 in queries.
112 
113         if (lowOrderBitsToPreserve > 0) {
114             int bytesLength = payloadLength - 1;
115             out.writeBytes(bytes, 0, bytesLength);
116 
117             // Pad the leftover of the last byte with zeros.
118             out.writeByte(padWithZeros(bytes[bytesLength], lowOrderBitsToPreserve));
119         } else {
120             // The sourcePrefixLength align with Byte so just copy in the bytes directly.
121             out.writeBytes(bytes, 0, payloadLength);
122         }
123     }
124 
125     // Package-Private for testing
126     static int calculateEcsAddressLength(int sourcePrefixLength, int lowOrderBitsToPreserve) {
127         return (sourcePrefixLength >>> 3) + (lowOrderBitsToPreserve != 0 ? 1 : 0);
128     }
129 
130     private void encodeRawRecord(DnsRawRecord record, ByteBuf out) throws Exception {
131         encodeRecord0(record, out);
132 
133         ByteBuf content = record.content();
134         int contentLen = content.readableBytes();
135 
136         out.writeShort(contentLen);
137         out.writeBytes(content, content.readerIndex(), contentLen);
138     }
139 
140     protected void encodeName(String name, ByteBuf buf) throws Exception {
141         DnsCodecUtil.encodeDomainName(name, buf);
142     }
143 
144     private static byte padWithZeros(byte b, int lowOrderBitsToPreserve) {
145         switch (lowOrderBitsToPreserve) {
146         case 0:
147             return 0;
148         case 1:
149             return (byte) (0x80 & b);
150         case 2:
151             return (byte) (0xC0 & b);
152         case 3:
153             return (byte) (0xE0 & b);
154         case 4:
155             return (byte) (0xF0 & b);
156         case 5:
157             return (byte) (0xF8 & b);
158         case 6:
159             return (byte) (0xFC & b);
160         case 7:
161             return (byte) (0xFE & b);
162         case 8:
163             return b;
164         default:
165             throw new IllegalArgumentException("lowOrderBitsToPreserve: " + lowOrderBitsToPreserve);
166         }
167     }
168 }
