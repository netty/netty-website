16  package io.netty5.handler.codec.dns;
17  
18  import io.netty5.buffer.ByteBuf;
19  import io.netty5.channel.ChannelHandlerContext;
20  import io.netty5.handler.codec.LengthFieldBasedFrameDecoder;
21  import static java.util.Objects.requireNonNull;
22  import io.netty5.util.internal.UnstableApi;
23  
24  @UnstableApi
25  public final class TcpDnsQueryDecoder extends LengthFieldBasedFrameDecoder {
26      private final DnsRecordDecoder decoder;
27  
28      /**
29       * Creates a new decoder with {@linkplain DnsRecordDecoder#DEFAULT the default record decoder}.
30       */
31      public TcpDnsQueryDecoder() {
32          this(DnsRecordDecoder.DEFAULT, 65535);
33      }
34  
35      /**
36       * Creates a new decoder with the specified {@code decoder}.
37       */
38      public TcpDnsQueryDecoder(DnsRecordDecoder decoder, int maxFrameLength) {
39          super(maxFrameLength, 0, 2, 0, 2);
40          this.decoder = requireNonNull(decoder, "decoder");
41      }
42  
43      @Override
44      protected Object decode0(ChannelHandlerContext ctx, ByteBuf in) throws Exception {
45          ByteBuf frame = (ByteBuf) super.decode0(ctx, in);
46          if (frame == null) {
47              return null;
48          }
49  
50          return DnsMessageUtil.decodeDnsQuery(decoder, frame.slice(), new DnsMessageUtil.DnsQueryFactory() {
51              @Override
52              public DnsQuery newQuery(int id, DnsOpCode dnsOpCode) {
53                  return new DefaultDnsQuery(id, dnsOpCode);
54              }
55          });
56      }
57  }
