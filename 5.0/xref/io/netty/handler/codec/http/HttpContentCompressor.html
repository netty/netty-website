39  /**
40   * Compresses an {@link HttpMessage} and an {@link HttpContent} in {@code gzip} or
41   * {@code deflate} encoding while respecting the {@code "Accept-Encoding"} header.
42   * If there is no matching encoding, no compression is done.  For more
43   * information on how this handler modifies the message, please refer to
44   * {@link HttpContentEncoder}.
45   */
46  public class HttpContentCompressor extends HttpContentEncoder {

49      private final BrotliOptions brotliOptions;
50      private final GzipOptions gzipOptions;
51      private final DeflateOptions deflateOptions;
52      private final ZstdOptions zstdOptions;
53  
54      private final int compressionLevel;
55      private final int windowBits;
56      private final int memLevel;
57      private final int contentSizeThreshold;
58      private final Map<String, Supplier<? extends Compressor>> factories;

114      * @param windowBits
115      *        The base two logarithm of the size of the history buffer.  The
116      *        value should be in the range {@code 9} to {@code 15} inclusive.
117      *        Larger values result in better compression at the expense of
118      *        memory usage.  The default value is {@code 15}.
119      * @param memLevel
120      *        How much memory should be allocated for the internal compression
121      *        state.  {@code 1} uses minimum memory and {@code 9} uses maximum
122      *        memory.  Larger values result in better and faster compression
123      *        at the expense of memory usage.  The default value is {@code 8}
124      * @param contentSizeThreshold

154     /**
155      * Create a new {@link HttpContentCompressor} instance with specified
156      * {@link CompressionOptions}s
157      *
158      * @param contentSizeThreshold
159      *        The response body is compressed when the size of the response
160      *        body exceeds the threshold. The value should be a non negative
161      *        number. {@code 0} will enable compression for all responses.
162      * @param compressionOptions {@link CompressionOptions} or {@code null}
163      *        if the default should be used.
164      */
165     public HttpContentCompressor(int contentSizeThreshold, CompressionOptions... compressionOptions) {
166         this.contentSizeThreshold = ObjectUtil.checkPositiveOrZero(contentSizeThreshold, "contentSizeThreshold");
167         BrotliOptions brotliOptions = null;
168         GzipOptions gzipOptions = null;
169         DeflateOptions deflateOptions = null;
170         ZstdOptions zstdOptions = null;
171         if (compressionOptions == null || compressionOptions.length == 0) {
172             brotliOptions = Brotli.isAvailable() ? StandardCompressionOptions.brotli() : null;
173             gzipOptions = StandardCompressionOptions.gzip();
174             deflateOptions = StandardCompressionOptions.deflate();
175             zstdOptions = Zstd.isAvailable() ? StandardCompressionOptions.zstd() : null;
176         } else {
177             ObjectUtil.deepCheckNotNull("compressionOptions", compressionOptions);
178             for (CompressionOptions compressionOption : compressionOptions) {
179                 if (compressionOption instanceof BrotliOptions) {
180                     // if we have BrotliOptions, it means Brotli is available
181                     brotliOptions = (BrotliOptions) compressionOption;
182                 } else if (compressionOption instanceof GzipOptions) {
183                     gzipOptions = (GzipOptions) compressionOption;
184                 } else if (compressionOption instanceof DeflateOptions) {
185                     deflateOptions = (DeflateOptions) compressionOption;
186                 } else if (compressionOption instanceof ZstdOptions) {
187                     zstdOptions = (ZstdOptions) compressionOption;
188                 } else {
189                     throw new IllegalArgumentException("Unsupported " + CompressionOptions.class.getSimpleName() +
190                             ": " + compressionOption);
191                 }
192             }
193         }
194 
195         this.gzipOptions = gzipOptions;
196         this.deflateOptions = deflateOptions;
197         this.brotliOptions = brotliOptions;
198         this.zstdOptions = zstdOptions;
199 
200         this.factories = new HashMap<>();
201 
202         if (this.gzipOptions != null) {
203             this.factories.put("gzip", ZlibCompressor.newFactory(
204                     ZlibWrapper.GZIP, gzipOptions.compressionLevel()));
205         }
206         if (this.deflateOptions != null) {
207             this.factories.put("deflate", ZlibCompressor.newFactory(
208                     ZlibWrapper.ZLIB, deflateOptions.compressionLevel()));
209         }
210         if (this.brotliOptions != null) {
211             this.factories.put("br", BrotliCompressor.newFactory(brotliOptions.parameters()));
212         }
213         if (this.zstdOptions != null) {
214             this.factories.put("zstd", ZstdCompressor.newFactory(zstdOptions.compressionLevel(),
215                     zstdOptions.blockSize(), zstdOptions.maxEncodeSize()));
216         }
217 
218         this.compressionLevel = -1;
219         this.windowBits = -1;
220         this.memLevel = -1;
221         supportsCompressionOptions = true;
222     }

224     @Override
225     protected Result beginEncode(HttpResponse httpResponse, String acceptEncoding) {
226         if (this.contentSizeThreshold > 0) {
227             if (httpResponse instanceof HttpContent &&
228                     ((HttpContent<?>) httpResponse).payload().readableBytes() < contentSizeThreshold) {
229                 return null;
230             }
231         }
232 
233         String contentEncoding = httpResponse.headers().get(HttpHeaderNames.CONTENT_ENCODING);
234         if (contentEncoding != null) {
235             // Content-Encoding was set, either as something specific or as the IDENTITY encoding
236             // Therefore, we should NOT encode here
237             return null;
238         }
239 
240         if (supportsCompressionOptions) {
241             String targetContentEncoding = determineEncoding(acceptEncoding);
242             if (targetContentEncoding == null) {
243                 return null;
244             }
245 
246             Supplier<? extends Compressor> compressorFactory = factories.get(targetContentEncoding);
247 
248             if (compressorFactory == null) {
249                 throw new Error();
250             }
251 
252             return new Result(targetContentEncoding, compressorFactory.get());
253         } else {
254             ZlibWrapper wrapper = determineWrapper(acceptEncoding);
255             if (wrapper == null) {
256                 return null;
257             }
258 
259             String targetContentEncoding;
260             switch (wrapper) {
261                 case GZIP:
262                     targetContentEncoding = "gzip";
263                     break;
264                 case ZLIB:
265                     targetContentEncoding = "deflate";
266                     break;
267                 default:
268                     throw new Error();
269             }
270 
271             return new Result(
272                     targetContentEncoding, ZlibCompressor.newFactory(wrapper, compressionLevel).get());
273         }
274     }

276     @SuppressWarnings("FloatingPointEquality")
277     protected String determineEncoding(String acceptEncoding) {
278         float starQ = -1.0f;
279         float brQ = -1.0f;
280         float zstdQ = -1.0f;
281         float gzipQ = -1.0f;
282         float deflateQ = -1.0f;
283         for (String encoding : acceptEncoding.split(",")) {
284             float q = 1.0f;
285             int equalsPos = encoding.indexOf('=');
286             if (equalsPos != -1) {
287                 try {
288                     q = Float.parseFloat(encoding.substring(equalsPos + 1));
289                 } catch (NumberFormatException e) {
290                     // Ignore encoding
291                     q = 0.0f;
292                 }
293             }
294             if (encoding.contains("*")) {
295                 starQ = q;
296             } else if (encoding.contains("br") && q > brQ) {
297                 brQ = q;
298             } else if (encoding.contains("zstd") && q > zstdQ) {
299                 zstdQ = q;
300             } else if (encoding.contains("gzip") && q > gzipQ) {
301                 gzipQ = q;
302             } else if (encoding.contains("deflate") && q > deflateQ) {
303                 deflateQ = q;
304             }
305         }
306         if (brQ > 0.0f || zstdQ > 0.0f || gzipQ > 0.0f || deflateQ > 0.0f) {
307             if (brQ != -1.0f && brQ >= zstdQ && this.brotliOptions != null) {
308                 return "br";
309             } else if (zstdQ != -1.0f && zstdQ >= gzipQ && this.zstdOptions != null) {
310                 return "zstd";
311             } else if (gzipQ != -1.0f && gzipQ >= deflateQ && this.gzipOptions != null) {
312                 return "gzip";
313             } else if (deflateQ != -1.0f && this.deflateOptions != null) {
314                 return "deflate";
315             }
316         }
317         if (starQ > 0.0f) {
318             if (brQ == -1.0f && this.brotliOptions != null) {
319                 return "br";
320             }
321             if (zstdQ == -1.0f && this.zstdOptions != null) {
322                 return "zstd";
323             }
324             if (gzipQ == -1.0f && this.gzipOptions != null) {
325                 return "gzip";
326             }
327             if (deflateQ == -1.0f && this.deflateOptions != null) {
328                 return "deflate";
329             }
330         }
331         return null;
332     }

334     @Deprecated
335     @SuppressWarnings("FloatingPointEquality")
336     protected ZlibWrapper determineWrapper(String acceptEncoding) {
337         float starQ = -1.0f;
338         float gzipQ = -1.0f;
339         float deflateQ = -1.0f;
340         for (String encoding : acceptEncoding.split(",")) {
341             float q = 1.0f;
342             int equalsPos = encoding.indexOf('=');
343             if (equalsPos != -1) {
344                 try {
345                     q = Float.parseFloat(encoding.substring(equalsPos + 1));
346                 } catch (NumberFormatException e) {
347                     // Ignore encoding
348                     q = 0.0f;
349                 }
350             }
351             if (encoding.contains("*")) {
352                 starQ = q;
353             } else if (encoding.contains("gzip") && q > gzipQ) {
354                 gzipQ = q;
355             } else if (encoding.contains("deflate") && q > deflateQ) {
356                 deflateQ = q;
357             }
358         }
359         if (gzipQ > 0.0f || deflateQ > 0.0f) {
360             if (gzipQ >= deflateQ) {
361                 return ZlibWrapper.GZIP;
362             } else {
363                 return ZlibWrapper.ZLIB;
364             }
365         }
366         if (starQ > 0.0f) {
367             if (gzipQ == -1.0f) {
368                 return ZlibWrapper.GZIP;
369             }
370             if (deflateQ == -1.0f) {
371                 return ZlibWrapper.ZLIB;
372             }
373         }
374         return null;
375     }
