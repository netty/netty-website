15  package io.netty5.handler.codec.http;
16  
17  import io.netty5.buffer.api.Send;
18  import io.netty5.channel.ChannelHandlerContext;
19  import io.netty5.util.AsciiString;
20  import io.netty5.util.concurrent.Future;
21  
22  import java.util.Collection;
23  import java.util.LinkedHashSet;
24  import java.util.Set;
25  
26  import static io.netty5.handler.codec.http.HttpResponseStatus.SWITCHING_PROTOCOLS;
27  import static java.util.Objects.requireNonNull;
28  
29  /**
30   * Client-side handler for handling an HTTP upgrade handshake to another protocol. When the first
31   * HTTP request is sent, this handler will add all appropriate headers to perform an upgrade to the
32   * new protocol. If the upgrade fails (i.e. response is not 101 Switching Protocols), this handler
33   * simply removes itself from the pipeline. If the upgrade is successful, upgrades the pipeline to
34   * the new protocol.
35   */
36  public class HttpClientUpgradeHandler<C extends HttpContent<C>> extends HttpObjectAggregator<C> {
37  
38      /**
39       * User events that are fired to notify about upgrade status.
40       */
41      public enum UpgradeEvent {
42          /**
43           * The Upgrade request was sent to the server.
44           */
45          UPGRADE_ISSUED,
46  
47          /**
48           * The Upgrade to the new protocol was successful.
49           */
50          UPGRADE_SUCCESSFUL,
51  
52          /**
53           * The Upgrade was unsuccessful due to the server not issuing
54           * with a 101 Switching Protocols response.
55           */
56          UPGRADE_REJECTED
57      }
58  
59      /**
60       * The source codec that is used in the pipeline initially.
61       */
62      public interface SourceCodec {
63  
64          /**
65           * Removes or disables the encoder of this codec so that the {@link UpgradeCodec} can send an initial greeting
66           * (if any).
67           */
68          void prepareUpgradeFrom(ChannelHandlerContext ctx);
69  
70          /**
71           * Removes this codec (i.e. all associated handlers) from the pipeline.
72           */
73          void upgradeFrom(ChannelHandlerContext ctx);
74      }
75  
76      /**
77       * A codec that the source can be upgraded to.
78       */
79      public interface UpgradeCodec {
80          /**
81           * Returns the name of the protocol supported by this codec, as indicated by the {@code 'UPGRADE'} header.
82           */
83          CharSequence protocol();
84  
85          /**
86           * Sets any protocol-specific headers required to the upgrade request. Returns the names of
87           * all headers that were added. These headers will be used to populate the CONNECTION header.
88           */
89          Collection<CharSequence> setUpgradeHeaders(ChannelHandlerContext ctx, HttpRequest upgradeRequest);
90  
91          /**
92           * Performs an HTTP protocol upgrade from the source codec. This method is responsible for
93           * adding all handlers required for the new protocol.
94           *
95           * @param ctx the context for the current handler.
96           * @param upgradeResponse the 101 Switching Protocols response that indicates that the server
97           *            has switched to this protocol.
98           */
99          void upgradeTo(ChannelHandlerContext ctx, Send<FullHttpResponse> upgradeResponse) throws Exception;
100     }
101 
102     private final SourceCodec sourceCodec;
103     private final UpgradeCodec upgradeCodec;
104     private boolean upgradeRequested;
105 
106     /**
107      * Constructs the client upgrade handler.
108      *
109      * @param sourceCodec the codec that is being used initially.
110      * @param upgradeCodec the codec that the client would like to upgrade to.
111      * @param maxContentLength the maximum length of the aggregated content.
112      */
113     public HttpClientUpgradeHandler(SourceCodec sourceCodec, UpgradeCodec upgradeCodec,
114                                     int maxContentLength) {
115         super(maxContentLength);
116         requireNonNull(sourceCodec, "sourceCodec");
117         requireNonNull(upgradeCodec, "upgradeCodec");
118         this.sourceCodec = sourceCodec;
119         this.upgradeCodec = upgradeCodec;
120     }
121 
122     @Override
123     public Future<Void> write(ChannelHandlerContext ctx, Object msg) {
124         if (!(msg instanceof HttpRequest)) {
125             return ctx.write(msg);
126         }
127 
128         if (upgradeRequested) {
129             return ctx.newFailedFuture(new IllegalStateException(
130                     "Attempting to write HTTP request with upgrade in progress"));
131         }
132 
133         upgradeRequested = true;
134         setUpgradeRequestHeaders(ctx, (HttpRequest) msg);
135 
136         // Continue writing the request.
137         Future<Void> f = ctx.write(msg);
138 
139         // Notify that the upgrade request was issued.
140         ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_ISSUED);
141         // Now we wait for the next HTTP response to see if we switch protocols.
142         return f;
143     }
144 
145     @Override
146     protected void decode(final ChannelHandlerContext ctx, HttpObject msg)
147             throws Exception {
148         FullHttpResponse response = null;
149         try {
150             if (!upgradeRequested) {
151                 throw new IllegalStateException("Read HTTP response without requesting protocol switch");
152             }
153 
154             if (msg instanceof HttpResponse) {
155                 HttpResponse rep = (HttpResponse) msg;
156                 if (!SWITCHING_PROTOCOLS.equals(rep.status())) {
157                     // The server does not support the requested protocol, just remove this handler
158                     // and continue processing HTTP.
159                     // NOTE: not releasing the response since we're letting it propagate to the
160                     // next handler.
161                     ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_REJECTED);
162                     ctx.fireChannelRead(msg);
163                     removeThisHandler(ctx);
164                     return;
165                 }
166             }
167 
168             if (msg instanceof FullHttpResponse) {
169                 response = (FullHttpResponse) msg;
170 
171                 tryUpgrade(ctx, response);
172             } else {
173                 // Call the base class to handle the aggregation of the full request.
174                 super.decode(new DelegatingChannelHandlerContext(ctx) {
175                     @Override
176                     public ChannelHandlerContext fireChannelRead(Object msg) {
177                         FullHttpResponse response = (FullHttpResponse) msg;
178                         tryUpgrade(ctx, response);
179                         return this;
180                     }
181                 }, msg);
182             }
183 
184         } catch (Throwable t) {
185             if (response != null && response.isAccessible()) {
186                 response.close();
187             }
188             ctx.fireExceptionCaught(t);
189             removeThisHandler(ctx);
190         }
191     }
192 
193     private void tryUpgrade(ChannelHandlerContext ctx, FullHttpResponse response) {
194         try (response) {
195             CharSequence upgradeHeader = response.headers().get(HttpHeaderNames.UPGRADE);
196             if (upgradeHeader != null && !AsciiString.contentEqualsIgnoreCase(upgradeCodec.protocol(), upgradeHeader)) {
197                 throw new IllegalStateException(
198                         "Switching Protocols response with unexpected UPGRADE protocol: " + upgradeHeader);
199             }
200 
201             // Upgrade to the new protocol.
202             sourceCodec.prepareUpgradeFrom(ctx);
203             upgradeCodec.upgradeTo(ctx, response.send());
204 
205             // Notify that the upgrade to the new protocol completed successfully.
206             ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_SUCCESSFUL);
207 
208             // We guarantee UPGRADE_SUCCESSFUL event will be arrived at the next handler
209             // before http2 setting frame and http response.
210             sourceCodec.upgradeFrom(ctx);
211             removeThisHandler(ctx);
212         } catch (Throwable t) {
213             ctx.fireExceptionCaught(t);
214             removeThisHandler(ctx);
215         }
216     }
217 
218     private static void removeThisHandler(ChannelHandlerContext ctx) {
219         ctx.pipeline().remove(ctx.name());
220     }
221 
222     /**
223      * Adds all upgrade request headers necessary for an upgrade to the supported protocols.
224      */
225     private void setUpgradeRequestHeaders(ChannelHandlerContext ctx, HttpRequest request) {
226         // Set the UPGRADE header on the request.
227         request.headers().set(HttpHeaderNames.UPGRADE, upgradeCodec.protocol());
228 
229         // Add all protocol-specific headers to the request.
230         Set<CharSequence> connectionParts = new LinkedHashSet<>(2);
231         connectionParts.addAll(upgradeCodec.setUpgradeHeaders(ctx, request));
232 
233         // Set the CONNECTION header from the set of all protocol-specific headers that were added.
234         StringBuilder builder = new StringBuilder();
235         for (CharSequence part : connectionParts) {
236             builder.append(part);
237             builder.append(',');
238         }
239         builder.append(HttpHeaderValues.UPGRADE);
240         request.headers().add(HttpHeaderNames.CONNECTION, builder.toString());
241     }
