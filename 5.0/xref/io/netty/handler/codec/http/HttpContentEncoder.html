37  /**
38   * Encodes the content of the outbound {@link HttpResponse} and {@link HttpContent}.
39   * The original content is replaced with the new content encoded by the
40   * {@link EmbeddedChannel}, which is created by {@link #beginEncode(HttpResponse, String)}.
41   * Once encoding is finished, the value of the <tt>'Content-Encoding'</tt> header
42   * is set to the target content encoding, as returned by
43   * {@link #beginEncode(HttpResponse, String)}.
44   * Also, the <tt>'Content-Length'</tt> header is updated to the length of the
45   * encoded content.  If there is no supported or allowed encoding in the
46   * corresponding {@link HttpRequest}'s {@code "Accept-Encoding"} header,
47   * {@link #beginEncode(HttpResponse, String)} should return {@code null} so that
48   * no encoding occurs (i.e. pass-through).
49   * <p>
50   * Please note that this is an abstract class.  You have to extend this class
51   * and implement {@link #beginEncode(HttpResponse, String)} properly to make
52   * this class functional.  For example, refer to the source code of
53   * {@link HttpContentCompressor}.
54   * <p>
55   * This handler must be placed after {@link HttpObjectEncoder} in the pipeline
56   * so that this handler can intercept HTTP responses before {@link HttpObjectEncoder}
57   * converts them into {@link Buffer}s.
58   */
59  public abstract class HttpContentEncoder extends MessageToMessageCodec<HttpRequest, HttpObject> {
60  
61      private enum State {
62          PASS_THROUGH,
63          AWAIT_HEADERS,
64          AWAIT_CONTENT
65      }
66  
67      private static final CharSequence ZERO_LENGTH_HEAD = "HEAD";
68      private static final CharSequence ZERO_LENGTH_CONNECT = "CONNECT";
69      private static final int CONTINUE_CODE = HttpResponseStatus.CONTINUE.code();
70  
71      private final Queue<CharSequence> acceptEncodingQueue = new ArrayDeque<>();
72      private Compressor compressor;
73      private State state = State.AWAIT_HEADERS;
74  
75      @Override
76      public boolean acceptOutboundMessage(Object msg) throws Exception {
77          return msg instanceof HttpContent || msg instanceof HttpResponse;
78      }
79  
80      @Override
81      protected void decode(ChannelHandlerContext ctx, HttpRequest msg) throws Exception {
82          CharSequence acceptEncoding;
83          List<String> acceptEncodingHeaders = msg.headers().getAll(ACCEPT_ENCODING);
84          switch (acceptEncodingHeaders.size()) {
85          case 0:
86              acceptEncoding = HttpContentDecoder.IDENTITY;
87              break;
88          case 1:
89              acceptEncoding = acceptEncodingHeaders.get(0);
90              break;
91          default:
92              // Multiple message-header fields https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.2
93              acceptEncoding = StringUtil.join(",", acceptEncodingHeaders);
94              break;
95          }
96  
97          HttpMethod method = msg.method();
98          if (HttpMethod.HEAD.equals(method)) {
99              acceptEncoding = ZERO_LENGTH_HEAD;
100         } else if (HttpMethod.CONNECT.equals(method)) {
101             acceptEncoding = ZERO_LENGTH_CONNECT;
102         }
103 
104         acceptEncodingQueue.add(acceptEncoding);
105         ctx.fireChannelRead(ReferenceCountUtil.retain(msg));
106     }
107 
108     @Override
109     protected void encode(ChannelHandlerContext ctx, HttpObject msg, List<Object> out) throws Exception {
110         final boolean isFull = msg instanceof HttpResponse && msg instanceof LastHttpContent;
111         switch (state) {
112             case AWAIT_HEADERS: {
113                 ensureHeaders(msg);
114                 assert compressor == null;
115                 assert msg instanceof HttpResponse;
116 
117                 final HttpResponse res = (HttpResponse) msg;
118                 final int code = res.status().code();
119                 final CharSequence acceptEncoding;
120                 if (code == CONTINUE_CODE) {
121                     // We need to not poll the encoding when response with CONTINUE as another response will follow
122                     // for the issued request. See https://github.com/netty/netty/issues/4079
123                     acceptEncoding = null;
124                 } else {
125                     // Get the list of encodings accepted by the peer.
126                     acceptEncoding = acceptEncodingQueue.poll();
127                     if (acceptEncoding == null) {
128                         throw new IllegalStateException("cannot send more responses than requests");
129                     }
130                 }
131 
132                 /*
133                  * per rfc2616 4.3 Message Body
134                  * All 1xx (informational), 204 (no content), and 304 (not modified) responses MUST NOT include a
135                  * message-body. All other responses do include a message-body, although it MAY be of zero length.
136                  *
137                  * 9.4 HEAD
138                  * The HEAD method is identical to GET except that the server MUST NOT return a message-body
139                  * in the response.
140                  *
141                  * Also we should pass through HTTP/1.0 as transfer-encoding: chunked is not supported.
142                  *
143                  * See https://github.com/netty/netty/issues/5382
144                  */
145                 if (isPassthru(res.protocolVersion(), code, acceptEncoding)) {
146                     out.add(res);
147                     if (!isFull) {
148                         // Pass through all following contents.
149                         state = State.PASS_THROUGH;
150                     }
151                     break;
152                 }
153 
154                 // Pass through the full response with empty content and continue waiting for the next resp.
155                 if (isFull && ((LastHttpContent<?>) res).payload().readableBytes() == 0) {
156                     out.add(res);
157                     break;
158                 }
159 
160                 // Prepare to encode the content.
161                 assert acceptEncoding != null;
162                 final Result result = beginEncode(res, acceptEncoding.toString());
163 
164                 // If unable to encode, pass through.
165                 if (result == null) {
166                     out.add(res);
167                     if (!isFull) {
168                         // Pass through all following contents.
169                         state = State.PASS_THROUGH;
170                     }
171                     break;
172                 }
173 
174                 compressor = result.contentCompressor();
175 
176                 // Encode the content and remove or replace the existing headers
177                 // so that the message looks like a decoded message.
178                 res.headers().set(HttpHeaderNames.CONTENT_ENCODING, result.targetContentEncoding());
179 
180                 // Output the rewritten response.
181                 if (isFull) {
182                     // Convert full message into unfull one.
183                     HttpResponse newRes = new DefaultHttpResponse(res.protocolVersion(), res.status());
184                     newRes.headers().set(res.headers());
185                     out.add(newRes);
186 
187                     ensureContent(res);
188                     encodeFullResponse(ctx, newRes, (HttpContent<?>) res, out);
189                     break;
190                 } else {
191                     // Make the response chunked to simplify content transformation.
192                     res.headers().remove(HttpHeaderNames.CONTENT_LENGTH);
193                     res.headers().set(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
194 
195                     out.add(res);
196                     state = State.AWAIT_CONTENT;
197                     if (!(msg instanceof HttpContent)) {
198                         // only break out the switch statement if we have not content to process
199                         // See https://github.com/netty/netty/issues/2006
200                         break;
201                     }
202                     // Fall through to encode the content
203                 }
204             }
205             case AWAIT_CONTENT: {
206                 ensureContent(msg);
207                 if (encodeContent(ctx, (HttpContent<?>) msg, out)) {
208                     state = State.AWAIT_HEADERS;
209                 }
210                 break;
211             }
212             case PASS_THROUGH: {
213                 ensureContent(msg);
214                 out.add(msg);
215                 // Passed through all following contents of the current response.
216                 if (msg instanceof LastHttpContent) {
217                     state = State.AWAIT_HEADERS;
218                 }
219                 break;
220             }
221         }
222     }
223 
224     private void encodeFullResponse(ChannelHandlerContext ctx, HttpResponse newRes, HttpContent<?> content,
225                                     List<Object> out) {
226         int existingMessages = out.size();
227         encodeContent(ctx, content, out);
228 
229         if (HttpUtil.isContentLengthSet(newRes)) {
230             // adjust the content-length header
231             int messageSize = 0;
232             for (int i = existingMessages; i < out.size(); i++) {
233                 Object item = out.get(i);
234                 if (item instanceof HttpContent) {
235                     messageSize += ((HttpContent<?>) item).payload().readableBytes();
236                 }
237             }
238             HttpUtil.setContentLength(newRes, messageSize);
239         } else {
240             newRes.headers().set(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
241         }
242     }
243 
244     private static boolean isPassthru(HttpVersion version, int code, CharSequence httpMethod) {
245         return code < 200 || code == 204 || code == 304 ||
246                (httpMethod == ZERO_LENGTH_HEAD || (httpMethod == ZERO_LENGTH_CONNECT && code == 200)) ||
247                 version == HttpVersion.HTTP_1_0;
248     }
249 
250     private static void ensureHeaders(HttpObject msg) {
251         if (!(msg instanceof HttpResponse)) {
252             throw new IllegalStateException(
253                     "unexpected message type: " +
254                     msg.getClass().getName() + " (expected: " + HttpResponse.class.getSimpleName() + ')');
255         }
256     }
257 
258     private static void ensureContent(HttpObject msg) {
259         if (!(msg instanceof HttpContent)) {
260             throw new IllegalStateException(
261                     "unexpected message type: " +
262                     msg.getClass().getName() + " (expected: " + HttpContent.class.getSimpleName() + ')');
263         }
264     }
265 
266     private boolean encodeContent(ChannelHandlerContext ctx, HttpContent<?> c, List<Object> out) {
267         Buffer content = c.payload();
268 
269         encode(content, ctx.alloc(), ctx.bufferAllocator(), out);
270 
271         if (c instanceof LastHttpContent) {
272             finishEncode(ctx.alloc(), ctx.bufferAllocator(), out);
273             LastHttpContent<?> last = (LastHttpContent<?>) c;
274 
275             // Generate an additional chunk if the decoder produced
276             // the last product on closure,
277             HttpHeaders headers = last.trailingHeaders();
278             if (headers.isEmpty()) {
279                 out.add(new EmptyLastHttpContent(ctx.bufferAllocator()));
280             } else {
281                 out.add(new DefaultLastHttpContent(ctx.bufferAllocator().allocate(0), headers));
282             }
283             return true;
284         }
285         return false;
286     }

288     /**
289      * Prepare to encode the HTTP message content.
290      *
291      * @param httpResponse
292      *        the http response
293      * @param acceptEncoding
294      *        the value of the {@code "Accept-Encoding"} header
295      *
296      * @return the result of preparation, which is composed of the determined
297      *         target content encoding and a new {@link EmbeddedChannel} that
298      *         encodes the content into the target content encoding.
299      *         {@code null} if {@code acceptEncoding} is unsupported or rejected
300      *         and thus the content should be handled as-is (i.e. no encoding).
301      */
302     protected abstract Result beginEncode(HttpResponse httpResponse, String acceptEncoding) throws Exception;
303 
304     @Override
305     public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
306         cleanupSafely(ctx);
307         super.handlerRemoved(ctx);
308     }
309 
310     @Override
311     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
312         cleanupSafely(ctx);
313         super.channelInactive(ctx);
314     }
315 
316     private void cleanup() {
317         if (compressor != null) {
318             // Clean-up the previous encoder if not cleaned up correctly.
319             try {
320                 compressor.close();
321             } finally {
322                 compressor = null;
323             }
324         }
325     }
326 
327     private void cleanupSafely(ChannelHandlerContext ctx) {
328         try {
329             cleanup();
330         } catch (Throwable cause) {
331             // If cleanup throws any error we need to propagate it through the pipeline
332             // so we don't fail to propagate pipeline events.
333             ctx.fireExceptionCaught(cause);
334         }
335     }
336 
337     private void encode(Buffer in, ByteBufAllocator byteBufAllocator, BufferAllocator allocator, List<Object> out) {
338         Buffer compressed = extractOrCopy(allocator, compressor.compress(ByteBufAdaptor.intoByteBuf(in),
339                 byteBufAllocator));
340         if (compressed.readableBytes() == 0) {
341             compressed.close();
342             return;
343         }
344         out.add(new DefaultHttpContent(compressed));
345     }
346 
347     private void finishEncode(ByteBufAllocator byteBufAllocator, BufferAllocator allocator, List<Object> out) {
348         Buffer trailer = extractOrCopy(allocator, compressor.finish(byteBufAllocator));
349         if (trailer.readableBytes() == 0) {
350             trailer.close();
351             return;
352         }
353         out.add(new DefaultHttpContent(trailer));
354         compressor = null;
355     }

357     public static final class Result {
358         private final String targetContentEncoding;
359         private final Compressor contentCompressor;
360 
361         public Result(String targetContentEncoding, Compressor contentCompressor) {
362             requireNonNull(targetContentEncoding, "targetContentEncoding");
363             requireNonNull(contentCompressor, "contentCompressor");
364 
365             this.targetContentEncoding = targetContentEncoding;
366             this.contentCompressor = contentCompressor;
367         }
368 
369         public String targetContentEncoding() {
370             return targetContentEncoding;
371         }
372 
373         public Compressor contentCompressor() {
374             return contentCompressor;
375         }
