46      /**
47       * Determine if a uri is in origin-form according to
48       * <a href="https://tools.ietf.org/html/rfc7230#section-5.3">rfc7230, 5.3</a>.
49       */
50      public static boolean isOriginForm(URI uri) {
51          return uri.getScheme() == null && uri.getSchemeSpecificPart() == null &&
52                 uri.getHost() == null && uri.getAuthority() == null;
53      }

55      /**
56       * Determine if a uri is in asterisk-form according to
57       * <a href="https://tools.ietf.org/html/rfc7230#section-5.3">rfc7230, 5.3</a>.
58       */
59      public static boolean isAsteriskForm(URI uri) {
60          return "*".equals(uri.getPath()) &&
61                  uri.getScheme() == null && uri.getSchemeSpecificPart() == null &&
62                  uri.getHost() == null && uri.getAuthority() == null && uri.getQuery() == null &&
63                  uri.getFragment() == null;
64      }

66      /**
67       * Returns {@code true} if and only if the connection can remain open and
68       * thus 'kept alive'.  This methods respects the value of the.
69       *
70       * {@code "Connection"} header first and then the return value of
71       * {@link HttpVersion#isKeepAliveDefault()}.
72       */
73      public static boolean isKeepAlive(HttpMessage message) {
74          return !message.headers().containsValue(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE, true) &&
75                 (message.protocolVersion().isKeepAliveDefault() ||
76                  message.headers().containsValue(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE, true));
77      }

79      /**
80       * Sets the value of the {@code "Connection"} header depending on the
81       * protocol version of the specified message. This getMethod sets or removes
82       * the {@code "Connection"} header depending on what the default keep alive
83       * mode of the message's protocol version is, as specified by
84       * {@link HttpVersion#isKeepAliveDefault()}.
85       * <ul>
86       * <li>If the connection is kept alive by default:
87       *     <ul>
88       *     <li>set to {@code "close"} if {@code keepAlive} is {@code false}.</li>
89       *     <li>remove otherwise.</li>
90       *     </ul></li>
91       * <li>If the connection is closed by default:
92       *     <ul>
93       *     <li>set to {@code "keep-alive"} if {@code keepAlive} is {@code true}.</li>
94       *     <li>remove otherwise.</li>
95       *     </ul></li>
96       * </ul>
97       * @see #setKeepAlive(HttpHeaders, HttpVersion, boolean)
98       */
99      public static void setKeepAlive(HttpMessage message, boolean keepAlive) {
100         setKeepAlive(message.headers(), message.protocolVersion(), keepAlive);
101     }

103     /**
104      * Sets the value of the {@code "Connection"} header depending on the
105      * protocol version of the specified message. This getMethod sets or removes
106      * the {@code "Connection"} header depending on what the default keep alive
107      * mode of the message's protocol version is, as specified by
108      * {@link HttpVersion#isKeepAliveDefault()}.
109      * <ul>
110      * <li>If the connection is kept alive by default:
111      *     <ul>
112      *     <li>set to {@code "close"} if {@code keepAlive} is {@code false}.</li>
113      *     <li>remove otherwise.</li>
114      *     </ul></li>
115      * <li>If the connection is closed by default:
116      *     <ul>
117      *     <li>set to {@code "keep-alive"} if {@code keepAlive} is {@code true}.</li>
118      *     <li>remove otherwise.</li>
119      *     </ul></li>
120      * </ul>
121      */
122     public static void setKeepAlive(HttpHeaders h, HttpVersion httpVersion, boolean keepAlive) {
123         if (httpVersion.isKeepAliveDefault()) {
124             if (keepAlive) {
125                 h.remove(HttpHeaderNames.CONNECTION);
126             } else {
127                 h.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
128             }
129         } else {
130             if (keepAlive) {
131                 h.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
132             } else {
133                 h.remove(HttpHeaderNames.CONNECTION);
134             }
135         }
136     }

138     /**
139      * Returns the length of the content. Please note that this value is
140      * not retrieved from {@link HttpContent#payload()} but from the
141      * {@code "Content-Length"} header, and thus they are independent from each
142      * other.
143      *
144      * @return the content length
145      *
146      * @throws NumberFormatException
147      *         if the message does not have the {@code "Content-Length"} header
148      *         or its value is not a number
149      */
150     public static long getContentLength(HttpMessage message) {
151         String value = message.headers().get(HttpHeaderNames.CONTENT_LENGTH);
152         if (value != null) {
153             return Long.parseLong(value);
154         }
155 
156         // We know the content length if it's a Web Socket message even if
157         // Content-Length header is missing.
158         long webSocketContentLength = getWebSocketContentLength(message);
159         if (webSocketContentLength >= 0) {
160             return webSocketContentLength;
161         }
162 
163         // Otherwise we don't.
164         throw new NumberFormatException("header not found: " + HttpHeaderNames.CONTENT_LENGTH);
165     }

167     /**
168      * Returns the length of the content or the specified default value if the message does not have the {@code
169      * "Content-Length" header}. Please note that this value is not retrieved from {@link HttpContent#payload()} but
170      * from the {@code "Content-Length"} header, and thus they are independent from each other.
171      *
172      * @param message      the message
173      * @param defaultValue the default value
174      * @return the content length or the specified default value
175      * @throws NumberFormatException if the {@code "Content-Length"} header does not parse as a long
176      */
177     public static long getContentLength(HttpMessage message, long defaultValue) {
178         String value = message.headers().get(HttpHeaderNames.CONTENT_LENGTH);
179         if (value != null) {
180             return Long.parseLong(value);
181         }
182 
183         // We know the content length if it's a Web Socket message even if
184         // Content-Length header is missing.
185         long webSocketContentLength = getWebSocketContentLength(message);
186         if (webSocketContentLength >= 0) {
187             return webSocketContentLength;
188         }
189 
190         // Otherwise we don't.
191         return defaultValue;
192     }

194     /**
195      * Get an {@code int} representation of {@link #getContentLength(HttpMessage, long)}.
196      *
197      * @return the content length or {@code defaultValue} if this message does
198      *         not have the {@code "Content-Length"} header.
199      *
200      * @throws NumberFormatException if the {@code "Content-Length"} header does not parse as an int
201      */
202     public static int getContentLength(HttpMessage message, int defaultValue) {
203         return (int) Math.min(Integer.MAX_VALUE, getContentLength(message, (long) defaultValue));
204     }

206     /**
207      * Returns the content length of the specified web socket message. If the
208      * specified message is not a web socket message, {@code -1} is returned.
209      */
210     private static int getWebSocketContentLength(HttpMessage message) {
211         // WebSocket messages have constant content-lengths.
212         HttpHeaders h = message.headers();
213         if (message instanceof HttpRequest) {
214             HttpRequest req = (HttpRequest) message;
215             if (HttpMethod.GET.equals(req.method()) &&
216                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_KEY1) &&
217                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_KEY2)) {
218                 return 8;
219             }
220         } else if (message instanceof HttpResponse) {
221             HttpResponse res = (HttpResponse) message;
222             if (res.status().code() == 101 &&
223                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_ORIGIN) &&
224                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_LOCATION)) {
225                 return 16;
226             }
227         }
228 
229         // Not a web socket message
230         return -1;
231     }

244     /**
245      * Returns {@code true} if and only if the specified message contains an expect header and the only expectation
246      * present is the 100-continue expectation. Note that this method returns {@code false} if the expect header is
247      * not valid for the message (e.g., the message is a response, or the version on the message is HTTP/1.0).
248      *
249      * @param message the message
250      * @return {@code true} if and only if the expectation 100-continue is present and it is the only expectation
251      * present
252      */
253     public static boolean is100ContinueExpected(HttpMessage message) {
254         return isExpectHeaderValid(message)
255           // unquoted tokens in the expect header are case-insensitive, thus 100-continue is case insensitive
256           && message.headers().contains(HttpHeaderNames.EXPECT, HttpHeaderValues.CONTINUE, true);
257     }

259     /**
260      * Returns {@code true} if the specified message contains an expect header specifying an expectation that is not
261      * supported. Note that this method returns {@code false} if the expect header is not valid for the message
262      * (e.g., the message is a response, or the version on the message is HTTP/1.0).
263      *
264      * @param message the message
265      * @return {@code true} if and only if an expectation is present that is not supported
266      */
267     static boolean isUnsupportedExpectation(HttpMessage message) {
268         if (!isExpectHeaderValid(message)) {
269             return false;
270         }
271 
272         final String expectValue = message.headers().get(HttpHeaderNames.EXPECT);
273         return expectValue != null && !HttpHeaderValues.CONTINUE.toString().equalsIgnoreCase(expectValue);
274     }

276     private static boolean isExpectHeaderValid(final HttpMessage message) {
277         /*
278          * Expect: 100-continue is for requests only and it works only on HTTP/1.1 or later. Note further that RFC 7231
279          * section 5.1.1 says "A server that receives a 100-continue expectation in an HTTP/1.0 request MUST ignore
280          * that expectation."
281          */
282         return message instanceof HttpRequest &&
283                 message.protocolVersion().compareTo(HttpVersion.HTTP_1_1) >= 0;
284     }

311     /**
312      * Set the {@link HttpHeaderNames#TRANSFER_ENCODING} to either include {@link HttpHeaderValues#CHUNKED} if
313      * {@code chunked} is {@code true}, or remove {@link HttpHeaderValues#CHUNKED} if {@code chunked} is {@code false}.
314      *
315      * @param m The message which contains the headers to modify.
316      * @param chunked if {@code true} then include {@link HttpHeaderValues#CHUNKED} in the headers. otherwise remove
317      * {@link HttpHeaderValues#CHUNKED} from the headers.
318      */
319     public static void setTransferEncodingChunked(HttpMessage m, boolean chunked) {
320         if (chunked) {
321             m.headers().set(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
322             m.headers().remove(HttpHeaderNames.CONTENT_LENGTH);
323         } else {
324             List<String> encodings = m.headers().getAll(HttpHeaderNames.TRANSFER_ENCODING);
325             if (encodings.isEmpty()) {
326                 return;
327             }
328             List<CharSequence> values = new ArrayList<>(encodings);
329             values.removeIf(HttpHeaderValues.CHUNKED::contentEqualsIgnoreCase);
330             if (values.isEmpty()) {
331                 m.headers().remove(HttpHeaderNames.TRANSFER_ENCODING);
332             } else {
333                 m.headers().set(HttpHeaderNames.TRANSFER_ENCODING, values);
334             }
335         }
336     }

381     /**
382      * Fetch charset from Content-Type header value.
383      *
384      * @param contentTypeValue Content-Type header value to parse
385      * @param defaultCharset   result to use in case of empty, incorrect or doesn't contain required part header value
386      * @return the charset from message's Content-Type header or {@code defaultCharset}
387      * if charset is not presented or unparsable
388      */
389     public static Charset getCharset(CharSequence contentTypeValue, Charset defaultCharset) {
390         if (contentTypeValue != null) {
391             CharSequence charsetRaw = getCharsetAsSequence(contentTypeValue);
392             if (charsetRaw != null) {
393                 if (charsetRaw.length() > 2) { // at least contains 2 quotes(")
394                     if (charsetRaw.charAt(0) == '"' && charsetRaw.charAt(charsetRaw.length() - 1) == '"') {
395                         charsetRaw = charsetRaw.subSequence(1, charsetRaw.length() - 1);
396                     }
397                 }
398                 try {
399                     return Charset.forName(charsetRaw.toString());
400                 } catch (UnsupportedCharsetException | IllegalCharsetNameException ignored) {
401                     // just return the default charset
402                     return defaultCharset;
403                 }
404             } else {
405                 return defaultCharset;
406             }
407         } else {
408             return defaultCharset;
409         }
410     }

446     /**
447      * Fetch charset from Content-Type header value as a char sequence.
448      *
449      * A lot of sites/possibly clients have charset="CHARSET", for example charset="utf-8". Or "utf8" instead of "utf-8"
450      * This is not according to standard, but this method provide an ability to catch desired mistakes manually in code
451      *
452      * @param contentTypeValue Content-Type header value to parse
453      * @return the {@code CharSequence} with charset from message's Content-Type header
454      * or {@code null} if charset is not presented
455      * @throws NullPointerException in case if {@code contentTypeValue == null}
456      */
457     public static CharSequence getCharsetAsSequence(CharSequence contentTypeValue) {
458         requireNonNull(contentTypeValue, "contentTypeValue");
459 
460         int indexOfCharset = AsciiString.indexOfIgnoreCaseAscii(contentTypeValue, CHARSET_EQUALS, 0);
461         if (indexOfCharset == AsciiString.INDEX_NOT_FOUND) {
462             return null;
463         }
464 
465         int indexOfEncoding = indexOfCharset + CHARSET_EQUALS.length();
466         if (indexOfEncoding < contentTypeValue.length()) {
467             CharSequence charsetCandidate = contentTypeValue.subSequence(indexOfEncoding, contentTypeValue.length());
468             int indexOfSemicolon = AsciiString.indexOfIgnoreCaseAscii(charsetCandidate, SEMICOLON, 0);
469             if (indexOfSemicolon == AsciiString.INDEX_NOT_FOUND) {
470                 return charsetCandidate;
471             }
472 
473             return charsetCandidate.subSequence(0, indexOfSemicolon);
474         }
475 
476         return null;
477     }

540     /**
541      * Validates, and optionally extracts the content length from headers. This method is not intended for
542      * general use, but is here to be shared between HTTP/1 and HTTP/2 parsing.
543      *
544      * @param contentLengthFields the content-length header fields.
545      * @param isHttp10OrEarlier {@code true} if we are handling HTTP/1.0 or earlier
546      * @param allowDuplicateContentLengths {@code true}  if multiple, identical-value content lengths should be allowed.
547      * @return the normalized content length from the headers or {@code -1} if the fields were empty.
548      * @throws IllegalArgumentException if the content-length fields are not valid
549      */
550     @UnstableApi
551     public static long normalizeAndGetContentLength(
552             List<? extends CharSequence> contentLengthFields, boolean isHttp10OrEarlier,
553             boolean allowDuplicateContentLengths) {
554         if (contentLengthFields.isEmpty()) {
555             return -1;
556         }
557 
558         // Guard against multiple Content-Length headers as stated in
559         // https://tools.ietf.org/html/rfc7230#section-3.3.2:
560         //
561         // If a message is received that has multiple Content-Length header
562         //   fields with field-values consisting of the same decimal value, or a
563         //   single Content-Length header field with a field value containing a
564         //   list of identical decimal values (e.g., "Content-Length: 42, 42"),
565         //   indicating that duplicate Content-Length header fields have been
566         //   generated or combined by an upstream message processor, then the
567         //   recipient MUST either reject the message as invalid or replace the
568         //   duplicated field-values with a single valid Content-Length field
569         //   containing that decimal value prior to determining the message body
570         //   length or forwarding the message.
571         String firstField = contentLengthFields.get(0).toString();
572         boolean multipleContentLengths =
573                 contentLengthFields.size() > 1 || firstField.indexOf(COMMA) >= 0;
574 
575         if (multipleContentLengths && !isHttp10OrEarlier) {
576             if (allowDuplicateContentLengths) {
577                 // Find and enforce that all Content-Length values are the same
578                 String firstValue = null;
579                 for (CharSequence field : contentLengthFields) {
580                     String[] tokens = field.toString().split(COMMA_STRING, -1);
581                     for (String token : tokens) {
582                         String trimmed = token.trim();
583                         if (firstValue == null) {
584                             firstValue = trimmed;
585                         } else if (!trimmed.equals(firstValue)) {
586                             throw new IllegalArgumentException(
587                                     "Multiple Content-Length values found: " + contentLengthFields);
588                         }
589                     }
590                 }
591                 // Replace the duplicated field-values with a single valid Content-Length field
592                 firstField = firstValue;
593             } else {
594                 // Reject the message as invalid
595                 throw new IllegalArgumentException(
596                         "Multiple Content-Length values found: " + contentLengthFields);
597             }
598         }
599         // Ensure we not allow sign as part of the content-length:
600         // See https://github.com/squid-cache/squid/security/advisories/GHSA-qf3v-rc95-96j5
601         if (firstField.isEmpty() || !Character.isDigit(firstField.charAt(0))) {
602             // Reject the message as invalid
603             throw new IllegalArgumentException(
604                     "Content-Length value is not a number: " + firstField);
605         }
606         try {
607             final long value = Long.parseLong(firstField);
608             return checkPositiveOrZero(value, "Content-Length value");
609         } catch (NumberFormatException e) {
610             // Reject the message as invalid
611             throw new IllegalArgumentException(
612                     "Content-Length value is not a number: " + firstField, e);
613         }
614     }
615 }
