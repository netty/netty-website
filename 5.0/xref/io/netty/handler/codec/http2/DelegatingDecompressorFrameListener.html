41  /**
42   * An HTTP2 frame listener that will decompress data frames according to the {@code content-encoding} header for each
43   * stream. The decompression provided by this class will be applied to the data for the entire stream.
44   */
45  @UnstableApi
46  public class DelegatingDecompressorFrameListener extends Http2FrameListenerDecorator {
47  
48      private final Http2Connection connection;
49      private final boolean strict;
50      private boolean flowControllerInitialized;
51      private final Http2Connection.PropertyKey propertyKey;
52  
53      public DelegatingDecompressorFrameListener(Http2Connection connection, Http2FrameListener listener) {
54          this(connection, listener, true);
55      }
56  
57      public DelegatingDecompressorFrameListener(Http2Connection connection, Http2FrameListener listener,
58                      boolean strict) {
59          super(listener);
60          this.connection = connection;
61          this.strict = strict;
62  
63          propertyKey = connection.newKey();
64          connection.addListener(new Http2ConnectionAdapter() {
65              @Override
66              public void onStreamRemoved(Http2Stream stream) {
67                  final Http2Decompressor decompressor = decompressor(stream);
68                  if (decompressor != null) {
69                      cleanup(decompressor);
70                  }
71              }
72          });
73      }
74  
75      @Override
76      public int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding, boolean endOfStream)
77              throws Http2Exception {
78          final Http2Stream stream = connection.stream(streamId);
79          final Http2Decompressor decompressor = decompressor(stream);
80          if (decompressor == null) {
81              // The decompressor may be null if no compatible encoding type was found in this stream's headers
82              return listener.onDataRead(ctx, streamId, data, padding, endOfStream);
83          }
84  
85          final Decompressor decomp = decompressor.decompressor();
86          final int compressedBytes = data.readableBytes() + padding;
87          decompressor.incrementCompressedBytes(compressedBytes);
88          ByteBuf decompressed = null;
89          try {
90              Http2LocalFlowController flowController = connection.local().flowController();
91              decompressor.incrementDecompressedBytes(padding);
92  
93              for (;;) {
94                  if (decomp.isFinished()) {
95                      flowController.consumeBytes(stream,
96                              listener.onDataRead(ctx, streamId, Unpooled.EMPTY_BUFFER, padding, endOfStream));
97                      break;
98                  }
99                  int idx = data.readerIndex();
100                 decompressed = decomp.decompress(data, ctx.alloc());
101                 if (decompressed == null || idx == data.readerIndex()) {
102                     flowController.consumeBytes(stream,
103                             listener.onDataRead(ctx, streamId, Unpooled.EMPTY_BUFFER, padding, endOfStream));
104                     break;
105                 } else {
106                     // Immediately return the bytes back to the flow controller. ConsumedBytesConverter will convert
107                     // from the decompressed amount which the user knows about to the compressed amount which flow
108                     // control knows about.
109                     decompressor.incrementDecompressedBytes(decompressed.readableBytes());
110                     flowController.consumeBytes(stream,
111                             listener.onDataRead(ctx, streamId, decompressed, padding, false));
112                     decompressed.release();
113                     decompressed = null;
114                 }
115                 padding = 0; // Padding is only communicated once on the first iteration.
116             }
117             // We consume bytes each time we call the listener to ensure if multiple frames are decompressed
118             // that the bytes are accounted for immediately. Otherwise the user may see an inconsistent state of
119             // flow control.
120             return 0;
121         } catch (Http2Exception e) {
122             throw e;
123         } catch (Throwable t) {
124             throw streamError(stream.id(), INTERNAL_ERROR, t,
125                     "Decompressor error detected while delegating data read on streamId %d", stream.id());
126         } finally {
127             if (decompressed != null) {
128                 decompressed.release();
129             }
130         }
131     }

133     @Override
134     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
135                     boolean endStream) throws Http2Exception {
136         initDecompressor(ctx, streamId, headers, endStream);
137         listener.onHeadersRead(ctx, streamId, headers, padding, endStream);
138     }

140     @Override
141     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int streamDependency,
142                     short weight, boolean exclusive, int padding, boolean endStream) throws Http2Exception {
143         initDecompressor(ctx, streamId, headers, endStream);
144         listener.onHeadersRead(ctx, streamId, headers, streamDependency, weight, exclusive, padding, endStream);
145     }

147     /**
148      * Returns a new {@link EmbeddedChannel} that decodes the HTTP2 message content encoded in the specified
149      * {@code contentEncoding}.
150      *
151      * @param contentEncoding the value of the {@code content-encoding} header
152      * @return a new {@link Decompressor} if the specified encoding is supported. {@code null} otherwise
153      *         (alternatively, you can throw a {@link Http2Exception} to block unknown encoding).
154      * @throws Http2Exception If the specified encoding is not not supported and warrants an exception
155      */
156     protected Decompressor newContentDecompressor(final ChannelHandlerContext ctx, CharSequence contentEncoding)
157             throws Http2Exception {
158         if (GZIP.contentEqualsIgnoreCase(contentEncoding) || X_GZIP.contentEqualsIgnoreCase(contentEncoding)) {
159             return ZlibDecompressor.newFactory(ZlibWrapper.GZIP).get();
160         }
161         if (DEFLATE.contentEqualsIgnoreCase(contentEncoding) || X_DEFLATE.contentEqualsIgnoreCase(contentEncoding)) {
162             final ZlibWrapper wrapper = strict ? ZlibWrapper.ZLIB : ZlibWrapper.ZLIB_OR_NONE;
163             // To be strict, 'deflate' means ZLIB, but some servers were not implemented correctly.
164             return ZlibDecompressor.newFactory(wrapper).get();
165         }
166         if (Brotli.isAvailable() && BR.contentEqualsIgnoreCase(contentEncoding)) {
167             return BrotliDecompressor.newFactory().get();
168         }
169         // 'identity' or unsupported
170         return null;
171     }
172 
173     /**
174      * Returns the expected content encoding of the decoded content. This getMethod returns {@code "identity"} by
175      * default, which is the case for most decompressors.
176      *
177      * @param contentEncoding the value of the {@code content-encoding} header
178      * @return the expected content encoding of the new content.
179      * @throws Http2Exception if the {@code contentEncoding} is not supported and warrants an exception
180      */
181     protected CharSequence getTargetContentEncoding(@SuppressWarnings("UnusedParameters") CharSequence contentEncoding)
182                     throws Http2Exception {
183         return IDENTITY;
184     }
185 
186     /**
187      * Checks if a new decompressor object is needed for the stream identified by {@code streamId}.
188      * This method will modify the {@code content-encoding} header contained in {@code headers}.
189      *
190      * @param ctx The context
191      * @param streamId The identifier for the headers inside {@code headers}
192      * @param headers Object representing headers which have been read
193      * @param endOfStream Indicates if the stream has ended
194      * @throws Http2Exception If the {@code content-encoding} is not supported
195      */
196     private void initDecompressor(ChannelHandlerContext ctx, int streamId, Http2Headers headers, boolean endOfStream)
197             throws Http2Exception {
198         final Http2Stream stream = connection.stream(streamId);
199         if (stream == null) {
200             return;
201         }
202 
203         Http2Decompressor decompressor = decompressor(stream);
204         if (decompressor == null && !endOfStream) {
205             // Determine the content encoding.
206             CharSequence contentEncoding = headers.get(CONTENT_ENCODING);
207             if (contentEncoding == null) {
208                 contentEncoding = IDENTITY;
209             }
210             final Decompressor decomp = newContentDecompressor(ctx, contentEncoding);
211             if (decomp != null) {
212                 decompressor = new Http2Decompressor(decomp);
213                 stream.setProperty(propertyKey, decompressor);
214                 // Decode the content and remove or replace the existing headers
215                 // so that the message looks like a decoded message.
216                 CharSequence targetContentEncoding = getTargetContentEncoding(contentEncoding);
217                 if (IDENTITY.contentEqualsIgnoreCase(targetContentEncoding)) {
218                     headers.remove(CONTENT_ENCODING);
219                 } else {
220                     headers.set(CONTENT_ENCODING, targetContentEncoding);
221                 }
222             }
223         }
224 
225         if (decompressor != null) {
226             // The content length will be for the compressed data. Since we will decompress the data
227             // this content-length will not be correct. Instead of queuing messages or delaying sending
228             // header frames just remove the content-length header.
229             headers.remove(CONTENT_LENGTH);
230 
231             // The first time that we initialize a decompressor, decorate the local flow controller to
232             // properly convert consumed bytes.
233             if (!flowControllerInitialized) {
234                 flowControllerInitialized = true;
235                 connection.local().flowController(new ConsumedBytesConverter(connection.local().flowController()));
236             }
237         }
238     }
239 
240     Http2Decompressor decompressor(Http2Stream stream) {
241         return stream == null ? null : (Http2Decompressor) stream.getProperty(propertyKey);
242     }

244     /**
245      * Release remaining content from the {@link Decompressor}.
246      *
247      * @param decompressor The decompressor for {@code stream}
248      */
249     private static void cleanup(Http2Decompressor decompressor) {
250         decompressor.decompressor().close();
251     }

253     /**
254      * A decorator around the local flow controller that converts consumed bytes from uncompressed to compressed.
255      */
256     private final class ConsumedBytesConverter implements Http2LocalFlowController {
257         private final Http2LocalFlowController flowController;
258 
259         ConsumedBytesConverter(Http2LocalFlowController flowController) {
260             this.flowController = requireNonNull(flowController, "flowController");
261         }
262 
263         @Override
264         public Http2LocalFlowController frameWriter(Http2FrameWriter frameWriter) {
265             return flowController.frameWriter(frameWriter);
266         }
267 
268         @Override
269         public void channelHandlerContext(ChannelHandlerContext ctx) throws Http2Exception {
270             flowController.channelHandlerContext(ctx);
271         }
272 
273         @Override
274         public void initialWindowSize(int newWindowSize) throws Http2Exception {
275             flowController.initialWindowSize(newWindowSize);
276         }
277 
278         @Override
279         public int initialWindowSize() {
280             return flowController.initialWindowSize();
281         }
282 
283         @Override
284         public int windowSize(Http2Stream stream) {
285             return flowController.windowSize(stream);
286         }
287 
288         @Override
289         public void incrementWindowSize(Http2Stream stream, int delta) throws Http2Exception {
290             flowController.incrementWindowSize(stream, delta);
291         }
292 
293         @Override
294         public void receiveFlowControlledFrame(Http2Stream stream, ByteBuf data, int padding,
295                 boolean endOfStream) throws Http2Exception {
296             flowController.receiveFlowControlledFrame(stream, data, padding, endOfStream);
297         }
298 
299         @Override
300         public boolean consumeBytes(Http2Stream stream, int numBytes) throws Http2Exception {
301             Http2Decompressor decompressor = decompressor(stream);
302             if (decompressor != null) {
303                 // Convert the decompressed bytes to compressed (on the wire) bytes.
304                 numBytes = decompressor.consumeBytes(stream.id(), numBytes);
305             }
306             try {
307                 return flowController.consumeBytes(stream, numBytes);
308             } catch (Http2Exception e) {
309                 throw e;
310             } catch (Throwable t) {
311                 // The stream should be closed at this point. We have already changed our state tracking the compressed
312                 // bytes, and there is no guarantee we can recover if the underlying flow controller throws.
313                 throw streamError(stream.id(), INTERNAL_ERROR, t, "Error while returning bytes to flow control window");
314             }
315         }
316 
317         @Override
318         public int unconsumedBytes(Http2Stream stream) {
319             return flowController.unconsumedBytes(stream);
320         }
321 
322         @Override
323         public int initialWindowSize(Http2Stream stream) {
324             return flowController.initialWindowSize(stream);
325         }
326     }

328     /**
329      * Provides the state for stream {@code DATA} frame decompression.
330      */
331     private static final class Http2Decompressor {
332         private final Decompressor decompressor;
333         private int compressed;
334         private int decompressed;
335 
336         Http2Decompressor(Decompressor decompressor) {
337             this.decompressor = decompressor;
338         }
339 
340         /**
341          * Responsible for taking compressed bytes in and producing decompressed bytes.
342          */
343         Decompressor decompressor() {
344             return decompressor;
345         }
346 
347         /**
348          * Increment the number of bytes received prior to doing any decompression.
349          */
350         void incrementCompressedBytes(int delta) {
351             assert delta >= 0;
352             compressed += delta;
353         }
354 
355         /**
356          * Increment the number of bytes after the decompression process.
357          */
358         void incrementDecompressedBytes(int delta) {
359             assert delta >= 0;
360             decompressed += delta;
361         }
362 
363         /**
364          * Determines the ratio between {@code numBytes} and {@link Http2Decompressor#decompressed}.
365          * This ratio is used to decrement {@link Http2Decompressor#decompressed} and
366          * {@link Http2Decompressor#compressed}.
367          * @param streamId the stream ID
368          * @param decompressedBytes The number of post-decompressed bytes to return to flow control
369          * @return The number of pre-decompressed bytes that have been consumed.
370          */
371         int consumeBytes(int streamId, int decompressedBytes) throws Http2Exception {
372             checkPositiveOrZero(decompressedBytes, "decompressedBytes");
373             if (decompressed - decompressedBytes < 0) {
374                 throw streamError(streamId, INTERNAL_ERROR,
375                         "Attempting to return too many bytes for stream %d. decompressed: %d " +
376                                 "decompressedBytes: %d", streamId, decompressed, decompressedBytes);
377             }
378             double consumedRatio = decompressedBytes / (double) decompressed;
379             int consumedCompressed = Math.min(compressed, (int) Math.ceil(compressed * consumedRatio));
380             if (compressed - consumedCompressed < 0) {
381                 throw streamError(streamId, INTERNAL_ERROR,
382                         "overflow when converting decompressed bytes to compressed bytes for stream %d." +
383                                 "decompressedBytes: %d decompressed: %d compressed: %d consumedCompressed: %d",
384                         streamId, decompressedBytes, decompressed, compressed, consumedCompressed);
385             }
386             decompressed -= decompressedBytes;
387             compressed -= consumedCompressed;
388 
389             return consumedCompressed;
390         }
