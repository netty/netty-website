135     /**
136      * Handles the client-side (cleartext) upgrade from HTTP to HTTP/2.
137      * Reserves local stream 1 for the HTTP/2 response.
138      */
139     public void onHttpClientUpgrade() throws Http2Exception {
140         if (connection().isServer()) {
141             throw connectionError(PROTOCOL_ERROR, "Client-side HTTP upgrade requested for a server");
142         }
143         if (!prefaceSent()) {
144             // If the preface was not sent yet it most likely means the handler was not added to the pipeline before
145             // calling this method.
146             throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
147         }
148         if (decoder.prefaceReceived()) {
149             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
150         }
151 
152         // Create a local stream used for the HTTP cleartext upgrade.
153         connection().local().createStream(HTTP_UPGRADE_STREAM_ID, true);
154     }
155 
156     /**
157      * Handles the server-side (cleartext) upgrade from HTTP to HTTP/2.
158      * @param settings the settings for the remote endpoint.
159      */
160     public void onHttpServerUpgrade(Http2Settings settings) throws Http2Exception {
161         if (!connection().isServer()) {
162             throw connectionError(PROTOCOL_ERROR, "Server-side HTTP upgrade requested for a client");
163         }
164         if (!prefaceSent()) {
165             // If the preface was not sent yet it most likely means the handler was not added to the pipeline before
166             // calling this method.
167             throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
168         }
169         if (decoder.prefaceReceived()) {
170             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
171         }
172 
173         // Apply the settings but no ACK is necessary.
174         encoder.remoteSettings(settings);
175 
176         // Create a stream in the half-closed state.
177         connection().remote().createStream(HTTP_UPGRADE_STREAM_ID, true);
178     }

180     @Override
181     public void flush(ChannelHandlerContext ctx) {
182         try {
183             // Trigger pending writes in the remote flow controller.
184             encoder.flowController().writePendingBytes();
185             ctx.flush();
186         } catch (Http2Exception e) {
187             onError(ctx, true, e);
188         } catch (Throwable cause) {
189             onError(ctx, true, connectionError(INTERNAL_ERROR, cause, "Error flushing"));
190         }

216     private final class PrefaceDecoder extends BaseDecoder {
217         private ByteBuf clientPrefaceString;
218         private boolean prefaceSent;

220         PrefaceDecoder(ChannelHandlerContext ctx) throws Exception {
221             clientPrefaceString = clientPrefaceString(encoder.connection());
222             // This handler was just added to the context. In case it was handled after
223             // the connection became active, send the connection preface now.
224             sendPreface(ctx);

232         @Override
233         public void decode(ChannelHandlerContext ctx, ByteBuf in) throws Exception {
234             try {
235                 if (ctx.channel().isActive() && readClientPrefaceString(in) && verifyFirstFrameIsSettings(in)) {
236                     // After the preface is read, it is time to hand over control to the post initialized decoder.
237                     byteDecoder = new FrameDecoder();
238                     byteDecoder.decode(ctx, in);
239                 }
240             } catch (Throwable e) {
241                 onError(ctx, false, e);
242             }
243         }

246         public void channelActive(ChannelHandlerContext ctx) throws Exception {
247             // The channel just became active - send the connection preface to the remote endpoint.
248             sendPreface(ctx);
249         }

275         /**
276          * Decodes the client connection preface string from the input buffer.
277          *
278          * @return {@code true} if processing of the client preface string is complete. Since client preface strings can
279          *         only be received by servers, returns true immediately for client endpoints.
280          */
281         private boolean readClientPrefaceString(ByteBuf in) throws Http2Exception {
282             if (clientPrefaceString == null) {
283                 return true;
284             }
285 
286             int prefaceRemaining = clientPrefaceString.readableBytes();
287             int bytesRead = min(in.readableBytes(), prefaceRemaining);
288 
289             // If the input so far doesn't match the preface, break the connection.
290             if (bytesRead == 0 || !ByteBufUtil.equals(in, in.readerIndex(),
291                                                       clientPrefaceString, clientPrefaceString.readerIndex(),
292                                                       bytesRead)) {
293                 int maxSearch = 1024; // picked because 512 is too little, and 2048 too much
294                 int http1Index =
295                     ByteBufUtil.indexOf(HTTP_1_X_BUF, in.slice(in.readerIndex(), min(in.readableBytes(), maxSearch)));
296                 if (http1Index != -1) {
297                     String chunk = in.toString(in.readerIndex(), http1Index - in.readerIndex(), CharsetUtil.US_ASCII);
298                     throw connectionError(PROTOCOL_ERROR, "Unexpected HTTP/1.x request: %s", chunk);
299                 }
300                 String receivedBytes = hexDump(in, in.readerIndex(),
301                                                min(in.readableBytes(), clientPrefaceString.readableBytes()));
302                 throw connectionError(PROTOCOL_ERROR, "HTTP/2 client preface string missing or corrupt. " +
303                                                       "Hex dump for received bytes: %s", receivedBytes);
304             }
305             in.skipBytes(bytesRead);
306             clientPrefaceString.skipBytes(bytesRead);
307 
308             if (!clientPrefaceString.isReadable()) {
309                 // Entire preface has been read.
310                 clientPrefaceString.release();
311                 clientPrefaceString = null;
312                 return true;
313             }
314             return false;
315         }
316 
317         /**
318          * Peeks at that the next frame in the buffer and verifies that it is a non-ack {@code SETTINGS} frame.
319          *
320          * @param in the inbound buffer.
321          * @return {@code true} if the next frame is a non-ack {@code SETTINGS} frame, {@code false} if more
322          * data is required before we can determine the next frame type.
323          * @throws Http2Exception thrown if the next frame is NOT a non-ack {@code SETTINGS} frame.
324          */
325         private boolean verifyFirstFrameIsSettings(ByteBuf in) throws Http2Exception {
326             if (in.readableBytes() < 5) {
327                 // Need more data before we can see the frame type for the first frame.
328                 return false;
329             }
330 
331             short frameType = in.getUnsignedByte(in.readerIndex() + 3);
332             short flags = in.getUnsignedByte(in.readerIndex() + 4);
333             if (frameType != SETTINGS || (flags & Http2Flags.ACK) != 0) {
334                 throw connectionError(PROTOCOL_ERROR, "First received frame was not SETTINGS. " +
335                                                       "Hex dump for first 5 bytes: %s",
336                                       hexDump(in, in.readerIndex(), 5));
337             }
338             return true;
339         }

341         /**
342          * Sends the HTTP/2 connection preface upon establishment of the connection, if not already sent.
343          */
344         private void sendPreface(ChannelHandlerContext ctx) throws Exception {
345             if (prefaceSent || !ctx.channel().isActive()) {
346                 return;
347             }
348 
349             prefaceSent = true;
350 
351             final boolean isClient = !connection().isServer();
352             if (isClient) {
353                 // Clients must send the preface string as the first bytes on the connection.
354                 ctx.write(connectionPrefaceBuf()).addListener(ctx.channel(), ChannelFutureListeners.CLOSE_ON_FAILURE);
355             }
356 
357             // Both client and server must send their initial settings.
358             encoder.writeSettings(ctx, initialSettings)
359                    .addListener(ctx.channel(), ChannelFutureListeners.CLOSE_ON_FAILURE);
360 
361             if (isClient) {
362                 // If this handler is extended by the user and we directly fire the userEvent from this context then
363                 // the user will not see the event. We should fire the event starting with this handler so this class
364                 // (and extending classes) have a chance to process the event.
365                 userEventTriggered(ctx, Http2ConnectionPrefaceAndSettingsFrameWrittenEvent.INSTANCE);
366             }
367         }
368     }

418     @Override
419     public void channelWritabilityChanged(ChannelHandlerContext ctx) throws Exception {
420         // Writability is expected to change while we are writing. We cannot allow this event to trigger reentering
421         // the allocation and write loop. Reentering the event loop will lead to over or illegal allocation.
422         try {
423             if (ctx.channel().isWritable()) {
424                 flush(ctx);
425             }
426             encoder.flowController().channelWritabilityChanged();
427         } finally {
428             super.channelWritabilityChanged(ctx);

437     @Override
438     public Future<Void> close(ChannelHandlerContext ctx) {
439         if (decoupleCloseAndGoAway) {
440             return ctx.close();
441         }
442         // Avoid NotYetConnectedException and avoid sending before connection preface
443         if (!ctx.channel().isActive() || !prefaceSent()) {
444             return ctx.close();
445         }
446 
447         // If the user has already sent a GO_AWAY frame they may be attempting to do a graceful shutdown which requires
448         // sending multiple GO_AWAY frames. We should only send a GO_AWAY here if one has not already been sent. If
449         // a GO_AWAY has been sent we send a empty buffer just so we can wait to close until all other data has been
450         // flushed to the OS.
451         // https://github.com/netty/netty/issues/5307
452         Future<Void> f = connection().goAwaySent() ? ctx.write(EMPTY_BUFFER) : goAway(ctx, null);
453         ctx.flush();
454         Promise<Void> promise = ctx.newPromise();
455         doGracefulShutdown(ctx, f, promise);
456         return promise.asFuture();
457     }

459     private FutureListener<Object> newClosingChannelFutureListener(
460             ChannelHandlerContext ctx, Promise<Void> promise) {
461         long gracefulShutdownTimeoutMillis = this.gracefulShutdownTimeoutMillis;
462         return gracefulShutdownTimeoutMillis < 0 ?
463                 new ClosingChannelFutureListener(ctx, promise) :
464                 new ClosingChannelFutureListener(ctx, promise, gracefulShutdownTimeoutMillis, MILLISECONDS);
465     }
466 
467     private void doGracefulShutdown(ChannelHandlerContext ctx, Future<Void> future, final Promise<Void> promise) {
468         FutureListener<Object> listener = newClosingChannelFutureListener(ctx, promise);
469         if (isGracefulShutdownComplete()) {
470             // If there are no active streams, close immediately after the GO_AWAY write completes or the timeout
471             // elapsed.
472             future.addListener(listener);
473         } else {
474             // If there are active streams we should wait until they are all closed before closing the connection.
475 
476             // The ClosingChannelFutureListener will cascade promise completion. We need to always notify the
477             // new ClosingChannelFutureListener when the graceful close completes if the promise is not null.
478             if (closeListener == null) {
479                 closeListener = listener;
480             } else if (promise != null) {
481                 FutureListener<Object> oldCloseListener = closeListener;
482                 closeListener = future1 -> {
483                     try {
484                         oldCloseListener.operationComplete(future1);
485                     } finally {
486                         listener.operationComplete(future1);
487                     }
488                 };
489             }
490         }
491     }

493     @Override
494     public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
495         // Trigger flush after read on the assumption that flush is cheap if there is nothing to write and that
496         // for flow-control the read may release window that causes data to be written that can now be flushed.
497         try {
498             // First call channelReadComplete0(...) as this may produce more data that we want to flush
499             channelReadComplete0(ctx);
500         } finally {
501             flush(ctx);
502         }
503     }
504 
505     final void channelReadComplete0(ChannelHandlerContext ctx) {
506         // Discard bytes of the cumulation buffer if needed.
507         discardSomeReadBytes();
508 
509         // Ensure we never stall the HTTP/2 Channel. Flow-control is enforced by HTTP/2.
510         //
511         // See https://tools.ietf.org/html/rfc7540#section-5.2.2
512         if (!ctx.channel().config().isAutoRead()) {
513             ctx.read();
514         }
515 
516         ctx.fireChannelReadComplete();
517     }

583     /**
584      * Central handler for all exceptions caught during HTTP/2 processing.
585      */
586     @Override
587     public void onError(ChannelHandlerContext ctx, boolean outbound, Throwable cause) {
588         Http2Exception embedded = getEmbeddedHttp2Exception(cause);
589         if (isStreamError(embedded)) {
590             onStreamError(ctx, outbound, cause, (StreamException) embedded);
591         } else if (embedded instanceof CompositeStreamException) {
592             CompositeStreamException compositException = (CompositeStreamException) embedded;
593             for (StreamException streamException : compositException) {
594                 onStreamError(ctx, outbound, cause, streamException);
595             }
596         } else {
597             onConnectionError(ctx, outbound, cause, embedded);
598         }
599         ctx.flush();
600     }

604      * if the graceful shutdown has completed and the connection can be safely closed. This implementation just

611     /**
612      * Handler for a connection error. Sends a GO_AWAY frame to the remote endpoint. Once all
613      * streams are closed, the connection is shut down.
614      *
615      * @param ctx the channel context
616      * @param outbound {@code true} if the error was caused by an outbound operation.
617      * @param cause the exception that was caught
618      * @param http2Ex the {@link Http2Exception} that is embedded in the causality chain. This may
619      *            be {@code null} if it's an unknown exception.
620      */
621     protected void onConnectionError(ChannelHandlerContext ctx, boolean outbound,
622                                      Throwable cause, Http2Exception http2Ex) {
623         if (http2Ex == null) {
624             http2Ex = new Http2Exception(INTERNAL_ERROR, cause.getMessage(), cause);
625         }
626 
627         Promise<Void> promise = ctx.newPromise();
628         Future<Void> future = goAway(ctx, http2Ex);
629         if (http2Ex.shutdownHint() == Http2Exception.ShutdownHint.GRACEFUL_SHUTDOWN) {
630             doGracefulShutdown(ctx, future, promise);
631         } else {
632             future.addListener(newClosingChannelFutureListener(ctx, promise));
633         }
634     }

643      * @param http2Ex the {@link StreamException} that is embedded in the causality chain.
644      */
645     protected void onStreamError(ChannelHandlerContext ctx, boolean outbound,
646                                  @SuppressWarnings("unused") Throwable cause, StreamException http2Ex) {
647         final int streamId = http2Ex.streamId();
648         Http2Stream stream = connection().stream(streamId);
649 
650         //if this is caused by reading headers that are too large, send a header with status 431
651         if (http2Ex instanceof Http2Exception.HeaderListSizeException &&
652             ((Http2Exception.HeaderListSizeException) http2Ex).duringDecode() &&
653             connection().isServer()) {
654 
655             // NOTE We have to check to make sure that a stream exists before we send our reply.
656             // We likely always create the stream below as the stream isn't created until the
657             // header block is completely processed.
658 
659             // The case of a streamId referring to a stream which was already closed is handled
660             // by createStream and will land us in the catch block below
661             if (stream == null) {
662                 try {
663                     stream = encoder.connection().remote().createStream(streamId, true);
664                 } catch (Http2Exception e) {
665                     resetUnknownStream(ctx, streamId, http2Ex.error().code());
666                     return;
667                 }
668             }
669 
670             // ensure that we have not already sent headers on this stream
671             if (stream != null && !stream.isHeadersSent()) {
672                 try {
673                     handleServerHeaderDecodeSizeError(ctx, stream);
674                 } catch (Throwable cause2) {
675                     onError(ctx, outbound, connectionError(INTERNAL_ERROR, cause2, "Error DecodeSizeError"));
676                 }
677             }
678         }
679 
680         if (stream == null) {
681             if (!outbound || connection().local().mayHaveCreatedStream(streamId)) {
682                 resetUnknownStream(ctx, streamId, http2Ex.error().code());
683             }
684         } else {
685             resetStream(ctx, stream, http2Ex.error().code());
686         }
687     }

696     protected void handleServerHeaderDecodeSizeError(ChannelHandlerContext ctx, Http2Stream stream) {

704     /**
705      * Sends a {@code RST_STREAM} frame even if we don't know about the stream. This error condition is most likely
706      * triggered by the first frame of a stream being invalid. That is, there was an error reading the frame before
707      * we could create a new stream.
708      */
709     private Future<Void> resetUnknownStream(final ChannelHandlerContext ctx, int streamId, long errorCode) {
710         Future<Void> future = frameWriter().writeRstStream(ctx, streamId, errorCode);
711         if (future.isDone()) {
712             closeConnectionOnError(ctx, future);
713         } else {
714             future.addListener(ctx, this::closeConnectionOnError);
715         }
716         return future;
717     }

719     @Override
720     public Future<Void> resetStream(final ChannelHandlerContext ctx, int streamId, long errorCode) {
721         final Http2Stream stream = connection().stream(streamId);
722         if (stream == null) {
723             return resetUnknownStream(ctx, streamId, errorCode);
724         }
725 
726        return resetStream(ctx, stream, errorCode);
727     }
728 
729     private Future<Void> resetStream(final ChannelHandlerContext ctx, final Http2Stream stream,
730                                       long errorCode) {
731         if (stream.isResetSent()) {
732             // Don't write a RST_STREAM frame if we have already written one.
733             return ctx.newSucceededFuture();
734         }
735         // Synchronously set the resetSent flag to prevent any subsequent calls
736         // from resulting in multiple reset frames being sent.
737         //
738         // This needs to be done before we notify the promise as the promise may have a listener attached that
739         // call resetStream(...) again.
740         stream.resetSent();
741 
742         final Future<Void> future;
743         // If the remote peer is not aware of the steam, then we are not allowed to send a RST_STREAM
744         // https://tools.ietf.org/html/rfc7540#section-6.4.
745         if (stream.state() == IDLE ||
746             connection().local().created(stream) && !stream.isHeadersSent() && !stream.isPushPromiseSent()) {
747             future = ctx.newSucceededFuture();
748         } else {
749             future = frameWriter().writeRstStream(ctx, stream.id(), errorCode);
750         }
751         if (future.isDone()) {
752             processRstStreamWriteResult(ctx, stream, future);
753         } else {
754             future.addListener(future1 -> processRstStreamWriteResult(ctx, stream, future1));
755         }
756 
757         return future;
758     }

760     @Override
761     public Future<Void> goAway(final ChannelHandlerContext ctx, final int lastStreamId, final long errorCode,
762                                 final ByteBuf debugData) {
763         final Http2Connection connection = connection();
764         try {
765             if (!connection.goAwaySent(lastStreamId, errorCode, debugData)) {
766                 debugData.release();
767                 return ctx.newSucceededFuture();
768             }
769         } catch (Throwable cause) {
770             debugData.release();
771             return ctx.newFailedFuture(cause);
772         }
773 
774         // Need to retain before we write the buffer because if we do it after the refCnt could already be 0 and
775         // result in an IllegalRefCountException.
776         debugData.retain();
777         Future<Void> future = frameWriter().writeGoAway(ctx, lastStreamId, errorCode, debugData);
778 
779         if (future.isDone()) {
780             processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
781         } else {
782             future.addListener(future1 ->
783                     processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future1));
784         }
785 
786         return future;
787     }

789     /**
790      * Closes the connection if the graceful shutdown process has completed.
791      * @param future Represents the status that will be passed to the {@link #closeListener}.
792      */
793     private void checkCloseConnection(Future<?> future) {
794         // If this connection is closing and the graceful shutdown has completed, close the connection
795         // once this operation completes.
796         if (closeListener != null && isGracefulShutdownComplete()) {
797             FutureListener<Object> closeListener = this.closeListener;
798             // This method could be called multiple times
799             // and we don't want to notify the closeListener multiple times.
800             this.closeListener = null;
801             try {
802                 closeListener.operationComplete(future);
803             } catch (Exception e) {
804                 throw new IllegalStateException("Close listener threw an unexpected exception", e);
805             }
806         }
807     }

809     /**
810      * Close the remote endpoint with with a {@code GO_AWAY} frame. Does <strong>not</strong> flush
811      * immediately, this is the responsibility of the caller.
812      */
813     private Future<Void> goAway(ChannelHandlerContext ctx, Http2Exception cause) {
814         long errorCode = cause != null ? cause.error().code() : NO_ERROR.code();
815         int lastKnownStream;
816         if (cause != null && cause.shutdownHint() == Http2Exception.ShutdownHint.HARD_SHUTDOWN) {
817             // The hard shutdown could have been triggered during header processing, before updating
818             // lastStreamCreated(). Specifically, any connection errors encountered by Http2FrameReader or HPACK
819             // decoding will fail to update the last known stream. So we must be pessimistic.
820             // https://github.com/netty/netty/issues/10670
821             lastKnownStream = Integer.MAX_VALUE;
822         } else {
823             lastKnownStream = connection().remote().lastStreamCreated();
824         }
825         return goAway(ctx, lastKnownStream, errorCode, Http2CodecUtil.toByteBuf(ctx, cause));
826     }

828     @SuppressWarnings("unchecked")
829     private void processRstStreamWriteResult(ChannelHandlerContext ctx, Http2Stream stream, Future<?> future) {
830         if (future.isSuccess()) {
831             closeStream(stream, (Future<Void>) future);
832         } else {
833             // The connection will be closed and so no need to change the resetSent flag to false.
834             onConnectionError(ctx, true, future.cause(), null);
835         }
836     }
837 
838     private void closeConnectionOnError(ChannelHandlerContext ctx, Future<?> future) {
839         if (future.isFailed()) {
840             onConnectionError(ctx, true, future.cause(), null);
841         }
842     }

851     private static void processGoAwayWriteResult(final ChannelHandlerContext ctx, final int lastStreamId,
852                                                  final long errorCode, final ByteBuf debugData, Future<?> future) {
853         try {
854             if (future.isSuccess()) {
855                 if (errorCode != NO_ERROR.code()) {
856                     if (logger.isDebugEnabled()) {
857                         logger.debug("{} Sent GOAWAY: lastStreamId '{}', errorCode '{}', " +
858                                      "debugData '{}'. Forcing shutdown of the connection.",
859                                      ctx.channel(), lastStreamId, errorCode, debugData.toString(UTF_8), future.cause());
860                     }
861                     ctx.close();
862                 }
863             } else {
864                 if (logger.isDebugEnabled()) {
865                     logger.debug("{} Sending GOAWAY failed: lastStreamId '{}', errorCode '{}', " +
866                                  "debugData '{}'. Forcing shutdown of the connection.",
867                                  ctx.channel(), lastStreamId, errorCode, debugData.toString(UTF_8), future.cause());
868                 }
869                 ctx.close();
870             }
871         } finally {
872             // We're done with the debug data now.
873             debugData.release();
874         }
875     }

877     /**
878      * Closes the channel when the future completes.
879      */
880     private static final class ClosingChannelFutureListener implements FutureListener<Object> {
881         private final ChannelHandlerContext ctx;
882         private final Promise<Void> promise;
883         private final Future<?> timeoutTask;
884         private boolean closed;
885 
886         ClosingChannelFutureListener(ChannelHandlerContext ctx, Promise<Void> promise) {
887             this.ctx = ctx;
888             this.promise = promise;
889             timeoutTask = null;
890         }
891 
892         ClosingChannelFutureListener(final ChannelHandlerContext ctx, final Promise<Void> promise,
893                                      long timeout, TimeUnit unit) {
894             this.ctx = ctx;
895             this.promise = promise;
896             timeoutTask = ctx.executor().schedule(this::doClose, timeout, unit);
897         }
898 
899         @Override
900         public void operationComplete(Future<?> sentGoAwayFuture) {
901             if (timeoutTask != null) {
902                 timeoutTask.cancel();
903             }
904             doClose();
905         }
906 
907         private void doClose() {
908             // We need to guard against multiple calls as the timeout may trigger close() first and then it will be
909             // triggered again because of operationComplete(...) is called.
910             if (closed) {
911                 // This only happens if we also scheduled a timeout task.
912                 assert timeoutTask != null;
913                 return;
914             }
915             closed = true;
916             // Before trying to close we need to check if the handler still exists in the pipeline as it may
917             // have been removed already.
918             if (!ctx.isRemoved()) {
919                 if (promise == null) {
920                     ctx.close();
921                 } else {
922                     ctx.close().cascadeTo(promise);
923                 }
924             } else if (promise != null) {
925                 promise.setSuccess(null);
926             }
927         }
928     }
