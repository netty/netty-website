36  /**
37   * Implementation of a {@link Http2ConnectionEncoder} that dispatches all method call to another
38   * {@link Http2ConnectionEncoder}, until {@code SETTINGS_MAX_CONCURRENT_STREAMS} is reached.
39   * <p/>
40   * <p>When this limit is hit, instead of rejecting any new streams this implementation buffers newly
41   * created streams and their corresponding frames. Once an active stream gets closed or the maximum
42   * number of concurrent streams is increased, this encoder will automatically try to empty its
43   * buffer and create as many new streams as possible.
44   * <p/>
45   * <p>
46   * If a {@code GOAWAY} frame is received from the remote endpoint, all buffered writes for streams
47   * with an ID less than the specified {@code lastStreamId} will immediately fail with a
48   * {@link Http2GoAwayException}.
49   * <p/>
50   * <p>
51   * If the channel/encoder gets closed, all new and buffered writes will immediately fail with a
52   * {@link Http2ChannelClosedException}.
53   * </p>
54   * <p>This implementation makes the buffering mostly transparent and is expected to be used as a
55   * drop-in decorator of {@link DefaultHttp2ConnectionEncoder}.
56   * </p>
57   */
58  @UnstableApi
59  public class StreamBufferingEncoder extends DecoratingHttp2ConnectionEncoder {
60  
61      /**
62       * Thrown if buffered streams are terminated due to this encoder being closed.
63       */
64      public static final class Http2ChannelClosedException extends Http2Exception {
65          private static final long serialVersionUID = 4768543442094476971L;
66  
67          public Http2ChannelClosedException() {
68              super(Http2Error.REFUSED_STREAM, "Connection closed");
69          }
70      }
71  
72      private static final class GoAwayDetail {
73          private final int lastStreamId;
74          private final long errorCode;
75          private final byte[] debugData;
76  
77          GoAwayDetail(int lastStreamId, long errorCode, byte[] debugData) {
78              this.lastStreamId = lastStreamId;
79              this.errorCode = errorCode;
80              this.debugData = debugData.clone();
81          }
82      }
83  
84      /**
85       * Thrown by {@link StreamBufferingEncoder} if buffered streams are terminated due to
86       * receipt of a {@code GOAWAY}.
87       */
88      public static final class Http2GoAwayException extends Http2Exception {
89          private static final long serialVersionUID = 1326785622777291198L;
90          private final GoAwayDetail goAwayDetail;
91  
92          public Http2GoAwayException(int lastStreamId, long errorCode, byte[] debugData) {
93              this(new GoAwayDetail(lastStreamId, errorCode, debugData));
94          }
95  
96          Http2GoAwayException(GoAwayDetail goAwayDetail) {
97              super(Http2Error.STREAM_CLOSED);
98              this.goAwayDetail = goAwayDetail;
99          }
100 
101         public int lastStreamId() {
102             return goAwayDetail.lastStreamId;
103         }
104 
105         public long errorCode() {
106             return goAwayDetail.errorCode;
107         }
108 
109         public byte[] debugData() {
110             return goAwayDetail.debugData.clone();
111         }
112     }

114     /**
115      * Buffer for any streams and corresponding frames that could not be created due to the maximum
116      * concurrent stream limit being hit.
117      */
118     private final TreeMap<Integer, PendingStream> pendingStreams = new TreeMap<>();
119     private int maxConcurrentStreams;
120     private boolean closed;
121     private GoAwayDetail goAwayDetail;
122 
123     public StreamBufferingEncoder(Http2ConnectionEncoder delegate) {
124         this(delegate, SMALLEST_MAX_CONCURRENT_STREAMS);
125     }
126 
127     public StreamBufferingEncoder(Http2ConnectionEncoder delegate, int initialMaxConcurrentStreams) {
128         super(delegate);
129         maxConcurrentStreams = initialMaxConcurrentStreams;
130         connection().addListener(new Http2ConnectionAdapter() {
131 
132             @Override
133             public void onGoAwayReceived(int lastStreamId, long errorCode, ByteBuf debugData) {
134                 goAwayDetail = new GoAwayDetail(
135                     // Using getBytes(..., false) is safe here as GoAwayDetail(...) will clone the byte[].
136                     lastStreamId, errorCode,
137                     ByteBufUtil.getBytes(debugData, debugData.readerIndex(), debugData.readableBytes(), false));
138                 cancelGoAwayStreams(goAwayDetail);
139             }
140 
141             @Override
142             public void onStreamClosed(Http2Stream stream) {
143                 tryCreatePendingStreams();
144             }
145         });
146     }
147 
148     /**
149      * Indicates the number of streams that are currently buffered, awaiting creation.
150      */
151     public int numBufferedStreams() {
152         return pendingStreams.size();
153     }
154 
155     @Override
156     public Future<Void> writeHeaders(ChannelHandlerContext ctx, int streamId, Http2Headers headers,
157                                      int padding, boolean endStream) {
158         return writeHeaders(ctx, streamId, headers, 0, Http2CodecUtil.DEFAULT_PRIORITY_WEIGHT,
159                 false, padding, endStream);
160     }
161 
162     @Override
163     public Future<Void> writeHeaders(ChannelHandlerContext ctx, int streamId, Http2Headers headers,
164                                      int streamDependency, short weight, boolean exclusive,
165                                      int padding, boolean endOfStream) {
166         if (closed) {
167             return ctx.newFailedFuture(new Http2ChannelClosedException());
168         }
169         if (isExistingStream(streamId) || canCreateStream()) {
170             return super.writeHeaders(ctx, streamId, headers, streamDependency, weight,
171                     exclusive, padding, endOfStream);
172         }
173         if (goAwayDetail != null) {
174             return ctx.newFailedFuture(new Http2GoAwayException(goAwayDetail));
175         }
176         PendingStream pendingStream = pendingStreams.get(streamId);
177         if (pendingStream == null) {
178             pendingStream = new PendingStream(ctx, streamId);
179             pendingStreams.put(streamId, pendingStream);
180         }
181         Promise<Void> promise = ctx.newPromise();
182 
183         pendingStream.frames.add(new HeadersFrame(headers, streamDependency, weight, exclusive,
184                 padding, endOfStream, promise));
185         return promise.asFuture();
186     }
187 
188     @Override
189     public Future<Void> writeRstStream(ChannelHandlerContext ctx, int streamId, long errorCode) {
190         if (isExistingStream(streamId)) {
191             return super.writeRstStream(ctx, streamId, errorCode);
192         }
193         // Since the delegate doesn't know about any buffered streams we have to handle cancellation
194         // of the promises and releasing of the ByteBufs here.
195         PendingStream stream = pendingStreams.remove(streamId);
196         if (stream != null) {
197             // Sending a RST_STREAM to a buffered stream will succeed the promise of all frames
198             // associated with the stream, as sending a RST_STREAM means that someone "doesn't care"
199             // about the stream anymore and thus there is not point in failing the promises and invoking
200             // error handling routines.
201             stream.close(null);
202             return ctx.newSucceededFuture();
203         } else {
204             return ctx.newFailedFuture(connectionError(PROTOCOL_ERROR, "Stream does not exist %d", streamId));
205         }
206     }
207 
208     @Override
209     public Future<Void> writeData(ChannelHandlerContext ctx, int streamId, ByteBuf data,
210                                   int padding, boolean endOfStream) {
211         if (isExistingStream(streamId)) {
212             return super.writeData(ctx, streamId, data, padding, endOfStream);
213         }
214         PendingStream pendingStream = pendingStreams.get(streamId);
215         if (pendingStream != null) {
216             Promise<Void> promise = ctx.newPromise();
217             pendingStream.frames.add(new DataFrame(data, padding, endOfStream, promise));
218             return promise.asFuture();
219         } else {
220             ReferenceCountUtil.safeRelease(data);
221             return ctx.newFailedFuture(connectionError(PROTOCOL_ERROR, "Stream does not exist %d", streamId));
222         }
223     }
224 
225     @Override
226     public void remoteSettings(Http2Settings settings) throws Http2Exception {
227         // Need to let the delegate decoder handle the settings first, so that it sees the
228         // new setting before we attempt to create any new streams.
229         super.remoteSettings(settings);
230 
231         // Get the updated value for SETTINGS_MAX_CONCURRENT_STREAMS.
232         maxConcurrentStreams = connection().local().maxActiveStreams();
233 
234         // Try to create new streams up to the new threshold.
235         tryCreatePendingStreams();
236     }
237 
238     @Override
239     public void close() {
240         try {
241             if (!closed) {
242                 closed = true;
243 
244                 // Fail all buffered streams.
245                 Http2ChannelClosedException e = new Http2ChannelClosedException();
246                 while (!pendingStreams.isEmpty()) {
247                     PendingStream stream = pendingStreams.pollFirstEntry().getValue();
248                     stream.close(e);
249                 }
250             }
251         } finally {
252             super.close();
253         }
254     }
255 
256     private void tryCreatePendingStreams() {
257         while (!pendingStreams.isEmpty() && canCreateStream()) {
258             Map.Entry<Integer, PendingStream> entry = pendingStreams.pollFirstEntry();
259             PendingStream pendingStream = entry.getValue();
260             try {
261                 pendingStream.sendFrames();
262             } catch (Throwable t) {
263                 pendingStream.close(t);
264             }
265         }
266     }
267 
268     private void cancelGoAwayStreams(GoAwayDetail goAwayDetail) {
269         Iterator<PendingStream> iter = pendingStreams.values().iterator();
270         Exception e = new Http2GoAwayException(goAwayDetail);
271         while (iter.hasNext()) {
272             PendingStream stream = iter.next();
273             if (stream.streamId > goAwayDetail.lastStreamId) {
274                 iter.remove();
275                 stream.close(e);
276             }
277         }
278     }
279 
280     /**
281      * Determines whether or not we're allowed to create a new stream right now.
282      */
283     private boolean canCreateStream() {
284         return connection().local().numActiveStreams() < maxConcurrentStreams;
285     }
286 
287     private boolean isExistingStream(int streamId) {
288         return streamId <= connection().local().lastStreamCreated();
289     }
290 
291     private static final class PendingStream {
292         final ChannelHandlerContext ctx;
293         final int streamId;
294         final Queue<Frame> frames = new ArrayDeque<>(2);
295 
296         PendingStream(ChannelHandlerContext ctx, int streamId) {
297             this.ctx = ctx;
298             this.streamId = streamId;
299         }
300 
301         void sendFrames() {
302             for (Frame frame : frames) {
303                 frame.send(ctx, streamId);
304             }
305         }
306 
307         void close(Throwable t) {
308             for (Frame frame : frames) {
309                 frame.release(t);
310             }
311         }
312     }
313 
314     private abstract static class Frame {
315         final Promise<Void> promise;
316 
317         Frame(Promise<Void> promise) {
318             this.promise = promise;
319         }
320 
321         /**
322          * Release any resources (features, buffers, ...) associated with the frame.
323          */
324         void release(Throwable t) {
325             if (t == null) {
326                 promise.setSuccess(null);
327             } else {
328                 promise.setFailure(t);
329             }
330         }
331 
332         abstract void send(ChannelHandlerContext ctx, int streamId);
333     }
334 
335     private final class HeadersFrame extends Frame {
336         final Http2Headers headers;
337         final int streamDependency;
338         final short weight;
339         final boolean exclusive;
340         final int padding;
341         final boolean endOfStream;
342 
343         HeadersFrame(Http2Headers headers, int streamDependency, short weight, boolean exclusive,
344                      int padding, boolean endOfStream, Promise<Void> promise) {
345             super(promise);
346             this.headers = headers;
347             this.streamDependency = streamDependency;
348             this.weight = weight;
349             this.exclusive = exclusive;
350             this.padding = padding;
351             this.endOfStream = endOfStream;
352         }
353 
354         @Override
355         void send(ChannelHandlerContext ctx, int streamId) {
356             writeHeaders(ctx, streamId, headers, streamDependency, weight, exclusive, padding, endOfStream)
357                     .cascadeTo(promise);
358         }
359     }
360 
361     private final class DataFrame extends Frame {
362         final ByteBuf data;
363         final int padding;
364         final boolean endOfStream;
365 
366         DataFrame(ByteBuf data, int padding, boolean endOfStream, Promise<Void> promise) {
367             super(promise);
368             this.data = data;
369             this.padding = padding;
370             this.endOfStream = endOfStream;
371         }
372 
373         @Override
374         void release(Throwable t) {
375             super.release(t);
376             ReferenceCountUtil.safeRelease(data);
377         }
378 
379         @Override
380         void send(ChannelHandlerContext ctx, int streamId) {
381             writeData(ctx, streamId, data, padding, endOfStream).cascadeTo(promise);
382         }
383     }
