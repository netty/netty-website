38  /**
39   * This adapter provides just header/data events from the HTTP message flow defined
40   * in <a href="https://tools.ietf.org/html/rfc7540#section-8.1">[RFC 7540], Section 8.1</a>.
41   * <p>
42   * See {@link HttpToHttp2ConnectionHandler} to get translation from HTTP/1.x objects to HTTP/2 frames for writes.
43   */
44  @UnstableApi
45  public class InboundHttp2ToHttpAdapter extends Http2EventAdapter {
46      private static final ImmediateSendDetector DEFAULT_SEND_DETECTOR = new ImmediateSendDetector() {
47          @Override
48          public boolean mustSendImmediately(FullHttpMessage<?> msg) {
49              if (msg instanceof FullHttpResponse) {
50                  return ((FullHttpResponse) msg).status().codeClass() == HttpStatusClass.INFORMATIONAL;
51              }
52              if (msg instanceof FullHttpRequest) {
53                  return msg.headers().contains(HttpHeaderNames.EXPECT);
54              }
55              return false;
56          }
57  
58          @Override
59          public FullHttpMessage<?> copyIfNeeded(BufferAllocator allocator, FullHttpMessage<?> msg) {
60              if (msg instanceof FullHttpRequest) {
61                  final FullHttpRequest original = (FullHttpRequest) msg;
62                  FullHttpRequest copy = new DefaultFullHttpRequest(original.protocolVersion(), original.method(),
63                          original.uri(), allocator.allocate(0), original.headers(), original.trailingHeaders());
64                  copy.headers().remove(HttpHeaderNames.EXPECT);
65                  return copy;
66              }
67              return null;
68          }
69      };

71      private final int maxContentLength;
72      private final ImmediateSendDetector sendDetector;
73      private final Http2Connection.PropertyKey messageKey;
74      private final boolean propagateSettings;
75      protected final Http2Connection connection;
76      protected final boolean validateHttpHeaders;
77  
78      protected InboundHttp2ToHttpAdapter(Http2Connection connection, int maxContentLength,
79                                          boolean validateHttpHeaders, boolean propagateSettings) {
80  
81          requireNonNull(connection, "connection");
82          this.connection = connection;
83          this.maxContentLength = checkPositive(maxContentLength, "maxContentLength");
84          this.validateHttpHeaders = validateHttpHeaders;
85          this.propagateSettings = propagateSettings;
86          sendDetector = DEFAULT_SEND_DETECTOR;
87          messageKey = connection.newKey();
88      }
89  
90      /**
91       * The stream is out of scope for the HTTP message flow and will no longer be tracked
92       * @param stream The stream to remove associated state with
93       * @param release {@code true} to call release on the value if it is present. {@code false} to not call release.
94       */
95      protected final void removeMessage(Http2Stream stream, boolean release) {
96          FullHttpMessage<?> msg = stream.removeProperty(messageKey);
97          if (release && msg != null) {
98              msg.close();
99          }
100     }
101 
102     /**
103      * Get the {@link FullHttpMessage} associated with {@code stream}.
104      * @param stream The stream to get the associated state from
105      * @return The {@link FullHttpMessage} associated with {@code stream}.
106      */
107     protected final FullHttpMessage<?> getMessage(Http2Stream stream) {
108         return stream.getProperty(messageKey);
109     }
110 
111     /**
112      * Make {@code message} be the state associated with {@code stream}.
113      * @param stream The stream which {@code message} is associated with.
114      * @param message The message which contains the HTTP semantics.
115      */
116     protected final void putMessage(Http2Stream stream, FullHttpMessage<?> message) {
117         FullHttpMessage<?> previous = stream.setProperty(messageKey, message);
118         if (previous != message && previous != null) {
119             previous.close();
120         }
121     }

123     @Override
124     public void onStreamRemoved(Http2Stream stream) {
125         removeMessage(stream, true);
126     }

128     /**
129      * Set final headers and fire a channel read event
130      *
131      * @param ctx The context to fire the event on
132      * @param msg The message to send
133      * @param release {@code true} to call release on the value if it is present. {@code false} to not call release.
134      * @param stream the stream of the message which is being fired
135      */
136     protected void fireChannelRead(ChannelHandlerContext ctx, FullHttpMessage<?> msg, boolean release,
137                                    Http2Stream stream) {
138         removeMessage(stream, release);
139         HttpUtil.setContentLength(msg, msg.payload().readableBytes());
140         ctx.fireChannelRead(msg);
141     }

143     /**
144      * Create a new {@link FullHttpMessage} based upon the current connection parameters
145      *
146      * @param stream The stream to create a message for
147      * @param headers The headers associated with {@code stream}
148      * @param validateHttpHeaders
149      * <ul>
150      * <li>{@code true} to validate HTTP headers in the http-codec</li>
151      * <li>{@code false} not to validate HTTP headers in the http-codec</li>
152      * </ul>
153      * @param alloc The {@link BufferAllocator} to use to generate the content of the message
154      * @throws Http2Exception If there is an error when creating {@link FullHttpMessage} from
155      *                        {@link Http2Stream} and {@link Http2Headers}
156      */
157     protected FullHttpMessage<?> newMessage(Http2Stream stream, Http2Headers headers, boolean validateHttpHeaders,
158                                             BufferAllocator alloc) throws Http2Exception {
159         return connection.isServer() ? HttpConversionUtil.toFullHttpRequest(stream.id(), headers, alloc,
160                 validateHttpHeaders) : HttpConversionUtil.toFullHttpResponse(stream.id(), headers, alloc,
161                 validateHttpHeaders);
162     }
163 
164     /**
165      * Provides translation between HTTP/2 and HTTP header objects while ensuring the stream
166      * is in a valid state for additional headers.
167      *
168      * @param ctx The context for which this message has been received.
169      * Used to send informational header if detected.
170      * @param stream The stream the {@code headers} apply to
171      * @param headers The headers to process
172      * @param endOfStream {@code true} if the {@code stream} has received the end of stream flag
173      * @param allowAppend
174      * <ul>
175      * <li>{@code true} if headers will be appended if the stream already exists.</li>
176      * <li>if {@code false} and the stream already exists this method returns {@code null}.</li>
177      * </ul>
178      * @param appendToTrailer
179      * <ul>
180      * <li>{@code true} if a message {@code stream} already exists then the headers
181      * should be added to the trailing headers.</li>
182      * <li>{@code false} then appends will be done to the initial headers.</li>
183      * </ul>
184      * @return The object used to track the stream corresponding to {@code stream}. {@code null} if
185      *         {@code allowAppend} is {@code false} and the stream already exists.
186      * @throws Http2Exception If the stream id is not in the correct state to process the headers request
187      */
188     protected FullHttpMessage<?> processHeadersBegin(ChannelHandlerContext ctx, Http2Stream stream,
189                                                      Http2Headers headers, boolean endOfStream, boolean allowAppend,
190                                                      boolean appendToTrailer)
191             throws Http2Exception {
192         FullHttpMessage<?> msg = getMessage(stream);
193         boolean release = true;
194         if (msg == null) {
195             msg = newMessage(stream, headers, validateHttpHeaders, ctx.bufferAllocator());
196         } else if (allowAppend) {
197             release = false;
198             HttpConversionUtil.addHttp2ToHttpHeaders(stream.id(), headers, msg, appendToTrailer);
199         } else {
200             release = false;
201             msg = null;
202         }
203 
204         if (sendDetector.mustSendImmediately(msg)) {
205             // Copy the message (if necessary) before sending. The content is not expected to be copied (or used) in
206             // this operation but just in case it is used do the copy before sending and the resource may be released
207             final FullHttpMessage<?> copy = endOfStream ? null : sendDetector.copyIfNeeded(ctx.bufferAllocator(), msg);
208             fireChannelRead(ctx, msg, release, stream);
209             return copy;
210         }
211 
212         return msg;
213     }
214 
215     /**
216      * After HTTP/2 headers have been processed by {@link #processHeadersBegin} this method either
217      * sends the result up the pipeline or retains the message for future processing.
218      *
219      * @param ctx The context for which this message has been received
220      * @param stream The stream the {@code objAccumulator} corresponds to
221      * @param msg The object which represents all headers/data for corresponding to {@code stream}
222      * @param endOfStream {@code true} if this is the last event for the stream
223      */
224     private void processHeadersEnd(ChannelHandlerContext ctx, Http2Stream stream, FullHttpMessage<?> msg,
225                                    boolean endOfStream) {
226         if (endOfStream) {
227             // Release if the msg from the map is different from the object being forwarded up the pipeline.
228             fireChannelRead(ctx, msg, getMessage(stream) != msg, stream);
229         } else {
230             putMessage(stream, msg);
231         }
232     }

234     @Override
235     public int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding, boolean endOfStream)
236             throws Http2Exception {
237         Http2Stream stream = connection.stream(streamId);
238         FullHttpMessage<?> msg = getMessage(stream);
239         if (msg == null) {
240             throw connectionError(PROTOCOL_ERROR, "Data Frame received for unknown stream id %d", streamId);
241         }
242 
243         Buffer content = msg.payload();
244         final int dataReadableBytes = data.readableBytes();
245         if (content.readableBytes() > maxContentLength - dataReadableBytes) {
246             throw connectionError(INTERNAL_ERROR,
247                     "Content length exceeded max of %d for stream id %d", maxContentLength, streamId);
248         }
249 
250         content.ensureWritable(dataReadableBytes);
251         content.writeBytes(ByteBufAdaptor.extractOrCopy(ctx.bufferAllocator(), data.retain()));
252 
253         if (endOfStream) {
254             fireChannelRead(ctx, msg, false, stream);
255         }
256 
257         // All bytes have been processed.
258         return dataReadableBytes + padding;
259     }
260 
261     @Override
262     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
263                               boolean endOfStream) throws Http2Exception {
264         Http2Stream stream = connection.stream(streamId);
265         FullHttpMessage<?> msg = processHeadersBegin(ctx, stream, headers, endOfStream, true, true);
266         if (msg != null) {
267             processHeadersEnd(ctx, stream, msg, endOfStream);
268         }
269     }
270 
271     @Override
272     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int streamDependency,
273                               short weight, boolean exclusive, int padding, boolean endOfStream)
274             throws Http2Exception {
275         Http2Stream stream = connection.stream(streamId);
276         FullHttpMessage<?> msg = processHeadersBegin(ctx, stream, headers, endOfStream, true, true);
277         if (msg != null) {
278             // Add headers for dependency and weight.
279             // See https://github.com/netty/netty/issues/5866
280             if (streamDependency != Http2CodecUtil.CONNECTION_STREAM_ID) {
281                 msg.headers().setInt(HttpConversionUtil.ExtensionHeaderNames.STREAM_DEPENDENCY_ID.text(),
282                         streamDependency);
283             }
284             msg.headers().setShort(HttpConversionUtil.ExtensionHeaderNames.STREAM_WEIGHT.text(), weight);
285 
286             processHeadersEnd(ctx, stream, msg, endOfStream);
287         }
288     }

290     @Override
291     public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception {
292         Http2Stream stream = connection.stream(streamId);
293         FullHttpMessage<?> msg = getMessage(stream);
294         if (msg != null) {
295             onRstStreamRead(stream, msg);
296         }
297         ctx.fireExceptionCaught(Http2Exception.streamError(streamId, Http2Error.valueOf(errorCode),
298                 "HTTP/2 to HTTP layer caught stream reset"));
299     }

301     @Override
302     public void onPushPromiseRead(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
303                                   Http2Headers headers, int padding) throws Http2Exception {
304         // A push promise should not be allowed to add headers to an existing stream
305         Http2Stream promisedStream = connection.stream(promisedStreamId);
306         if (headers.status() == null) {
307             // A PUSH_PROMISE frame has no Http response status.
308             // https://tools.ietf.org/html/rfc7540#section-8.2.1
309             // Server push is semantically equivalent to a server responding to a
310             // request; however, in this case, that request is also sent by the
311             // server, as a PUSH_PROMISE frame.
312             headers.status(OK.codeAsText());
313         }
314         FullHttpMessage<?> msg = processHeadersBegin(ctx, promisedStream, headers, false, false, false);
315         if (msg == null) {
316             throw connectionError(PROTOCOL_ERROR, "Push Promise Frame received for pre-existing stream id %d",
317                     promisedStreamId);
318         }
319 
320         msg.headers().setInt(HttpConversionUtil.ExtensionHeaderNames.STREAM_PROMISE_ID.text(), streamId);
321         msg.headers().setShort(HttpConversionUtil.ExtensionHeaderNames.STREAM_WEIGHT.text(),
322                 Http2CodecUtil.DEFAULT_PRIORITY_WEIGHT);
323 
324         processHeadersEnd(ctx, promisedStream, msg, false);
325     }

327     @Override
328     public void onSettingsRead(ChannelHandlerContext ctx, Http2Settings settings) throws Http2Exception {
329         if (propagateSettings) {
330             // Provide an interface for non-listeners to capture settings
331             ctx.fireChannelRead(settings);
332         }
333     }

342     /**
343      * Allows messages to be sent up the pipeline before the next phase in the
344      * HTTP message flow is detected.
345      */
346     private interface ImmediateSendDetector {
347         /**
348          * Determine if the response should be sent immediately, or wait for the end of the stream
349          *
350          * @param msg The response to test
351          * @return {@code true} if the message should be sent immediately
352          *         {@code false) if we should wait for the end of the stream
353          */
354         boolean mustSendImmediately(FullHttpMessage<?> msg);
355 
356         /**
357          * Determine if a copy must be made after an immediate send happens.
358          * <p>
359          * An example of this use case is if a request is received
360          * with a 'Expect: 100-continue' header. The message will be sent immediately,
361          * and the data will be queued and sent at the end of the stream.
362          *
363          * @param allocator The {@link BufferAllocator} that can be used to allocate
364          * @param msg The message which has just been sent due to {@link #mustSendImmediately(FullHttpMessage)}
365          * @return A modified copy of the {@code msg} or {@code null} if a copy is not needed.
366          */
367         FullHttpMessage<?> copyIfNeeded(BufferAllocator allocator, FullHttpMessage<?> msg);
368     }
