98      private static void windowUpdateFrameWriteComplete(Channel streamChannel, Future<?> future) {
99          Throwable cause = future.cause();
100         if (cause != null) {
101             Throwable unwrappedCause;
102             // Unwrap if needed
103             if (cause instanceof Http2FrameStreamException && (unwrappedCause = cause.getCause()) != null) {
104                 cause = unwrappedCause;
105             }
106 
107             // Notify the child-channel and close it.
108             streamChannel.pipeline().fireExceptionCaught(cause);
109             streamChannel.unsafe().close(streamChannel.newPromise());
110         }
111     }

189     private void incrementPendingOutboundBytes(long size, boolean invokeLater) {
190         if (size == 0) {
191             return;
192         }
193 
194         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, size);
195         if (newWriteBufferSize > config().getWriteBufferHighWaterMark()) {
196             setUnwritable(invokeLater);
197         }
198     }
199 
200     private void decrementPendingOutboundBytes(long size, boolean invokeLater) {
201         if (size == 0) {
202             return;
203         }
204 
205         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, -size);
206         // Once the totalPendingSize dropped below the low water-mark we can mark the child channel
207         // as writable again. Before doing so we also need to ensure the parent channel is writable to
208         // prevent excessive buffering in the parent outbound buffer. If the parent is not writable
209         // we will mark the child channel as writable once the parent becomes writable by calling
210         // trySetWritable() later.
211         if (newWriteBufferSize < config().getWriteBufferLowWaterMark() && parent().isWritable()) {
212             setWritable(invokeLater);
213         }
214     }
215 
216     final void trySetWritable() {
217         // The parent is writable again but the child channel itself may still not be writable.
218         // Lets try to set the child channel writable to match the state of the parent channel
219         // if (and only if) the totalPendingSize is smaller then the low water-mark.
220         // If this is not the case we will try again later once we drop under it.
221         if (totalPendingSize < config().getWriteBufferLowWaterMark()) {
222             setWritable(false);
223         }
224     }
225 
226     private void setWritable(boolean invokeLater) {
227         for (;;) {
228             final int oldValue = unwritable;
229             final int newValue = oldValue & ~1;
230             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
231                 if (oldValue != 0 && newValue == 0) {
232                     fireChannelWritabilityChanged(invokeLater);
233                 }
234                 break;
235             }
236         }
237     }
238 
239     private void setUnwritable(boolean invokeLater) {
240         for (;;) {
241             final int oldValue = unwritable;
242             final int newValue = oldValue | 1;
243             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
244                 if (oldValue == 0) {
245                     fireChannelWritabilityChanged(invokeLater);
246                 }
247                 break;
248             }
249         }
250     }
251 
252     private void fireChannelWritabilityChanged(boolean invokeLater) {
253         final ChannelPipeline pipeline = pipeline();
254         if (invokeLater) {
255             Runnable task = fireChannelWritabilityChangedTask;
256             if (task == null) {
257                 fireChannelWritabilityChangedTask = task = pipeline::fireChannelWritabilityChanged;
258             }
259             executor().execute(task);
260         } else {
261             pipeline.fireChannelWritabilityChanged();
262         }
263     }

340     @Override
341     public long bytesBeforeUnwritable() {
342         long bytes = config().getWriteBufferHighWaterMark() - totalPendingSize;
343         // If bytes is negative we know we are not writable, but if bytes is non-negative we have to check
344         // writability. Note that totalPendingSize and isWritable() use different volatile variables that are not
345         // synchronized together. totalPendingSize will be updated before isWritable().
346         if (bytes > 0) {
347             return isWritable() ? bytes : 0;
348         }
349         return 0;
350     }
351 
352     @Override
353     public long bytesBeforeWritable() {
354         long bytes = totalPendingSize - config().getWriteBufferLowWaterMark();
355         // If bytes is negative we know we are writable, but if bytes is non-negative we have to check writability.
356         // Note that totalPendingSize and isWritable() use different volatile variables that are not synchronized
357         // together. totalPendingSize will be updated before isWritable().
358         if (bytes > 0) {
359             return isWritable() ? 0 : bytes;
360         }
361         return 0;
362     }

398     /**
399      * Receive a read message. This does not notify handlers unless a read is in progress on the
400      * channel.
401      */
402     void fireChildRead(Http2Frame frame) {
403         assert executor().inEventLoop();
404         if (!isActive()) {
405             ReferenceCountUtil.release(frame);
406         } else if (readStatus != ReadStatus.IDLE) {
407             // If a read is in progress or has been requested, there cannot be anything in the queue,
408             // otherwise we would have drained it from the queue and processed it during the read cycle.
409             assert inboundBuffer == null || inboundBuffer.isEmpty();
410             final RecvBufferAllocator.Handle allocHandle = unsafe.recvBufAllocHandle();
411             unsafe.doRead0(frame, allocHandle);
412             // We currently don't need to check for readEOS because the parent channel and child channel are limited
413             // to the same EventLoop thread. There are a limited number of frame types that may come after EOS is
414             // read (unknown, reset) and the trade off is less conditionals for the hot path (headers/data) at the
415             // cost of additional readComplete notifications on the rare path.
416             if (allocHandle.continueReading()) {
417                 maybeAddChannelToReadCompletePendingQueue();
418             } else {
419                 unsafe.notifyReadComplete(allocHandle, true);
420             }
421         } else {
422             if (inboundBuffer == null) {
423                 inboundBuffer = new ArrayDeque<>(4);
424             }
425             inboundBuffer.add(frame);
426         }
427     }
428 
429     void fireChildReadComplete() {
430         assert executor().inEventLoop();
431         assert readStatus != ReadStatus.IDLE || !readCompletePending;
432         unsafe.notifyReadComplete(unsafe.recvBufAllocHandle(), false);
433     }

505         @Override
506         public void close(final Promise<Void> promise) {
507             if (!promise.setUncancellable()) {
508                 return;
509             }
510             if (closeInitiated) {
511                 if (closePromise.isDone()) {
512                     // Closed already.
513                     promise.setSuccess(null);
514                 } else  {
515                     // This means close() was called before so we just register a listener and return
516                     closeFuture().addListener(promise, (p, future) -> p.setSuccess(null));
517                 }
518                 return;
519             }
520             closeInitiated = true;
521             // Just set to false as removing from an underlying queue would even be more expensive.
522             readCompletePending = false;
523 
524             final boolean wasActive = isActive();
525 
526             // There is no need to update the local window as once the stream is closed all the pending bytes will be
527             // given back to the connection window by the controller itself.
528 
529             // Only ever send a reset frame if the connection is still alive and if the stream was created before
530             // as otherwise we may send a RST on a stream in an invalid state and cause a connection error.
531             if (parent().isActive() && !readEOS && isStreamIdValid(stream.id())) {
532                 Http2StreamFrame resetFrame = new DefaultHttp2ResetFrame(Http2Error.CANCEL).stream(stream());
533                 write(resetFrame, newPromise());
534                 flush();
535             }
536 
537             if (inboundBuffer != null) {
538                 for (;;) {
539                     Object msg = inboundBuffer.poll();
540                     if (msg == null) {
541                         break;
542                     }
543                     ReferenceCountUtil.release(msg);
544                 }
545                 inboundBuffer = null;
546             }
547 
548             // The promise should be notified before we call fireChannelInactive().
549             outboundClosed = true;
550             closePromise.setSuccess(null);
551             promise.setSuccess(null);
552 
553             fireChannelInactiveAndDeregister(newPromise(), wasActive);
554         }
555 
556         @Override
557         public void closeForcibly() {
558             close(newPromise());
559         }

564         }

566         private void fireChannelInactiveAndDeregister(Promise<Void> promise,
567                                                       final boolean fireChannelInactive) {
568             if (!promise.setUncancellable()) {
569                 return;
570             }
571 
572             if (!registered) {
573                 promise.setSuccess(null);
574                 return;
575             }
576 
577             // As a user may call deregister() from within any method while doing processing in the ChannelPipeline,
578             // we need to ensure we do the actual deregister operation later. This is necessary to preserve the
579             // behavior of the AbstractChannel, which always invokes channelUnregistered and channelInactive
580             // events 'later' to ensure the current events in the handler are completed before these events.
581             //
582             // See:
583             // https://github.com/netty/netty/issues/4435
584             invokeLater(()-> {
585                 if (fireChannelInactive) {
586                     pipeline.fireChannelInactive();
587                 }
588                 // The user can fire `deregister` events multiple times but we only want to fire the pipeline
589                 // event if the channel was actually registered.
590                 if (registered) {
591                     registered = false;
592                     pipeline.fireChannelUnregistered();
593                 }
594                 safeSetSuccess(promise);
595             });
596         }
597 
598         private void safeSetSuccess(Promise<Void> promise) {
599             if (!promise.trySuccess(null)) {
600                 logger.warn("Failed to mark a promise as success because it is done already: {}", promise);
601             }
602         }

604         private void invokeLater(Runnable task) {
605             try {
606                 // This method is used by outbound operation implementations to trigger an inbound event later.
607                 // They do not trigger an inbound event immediately because an outbound operation might have been
608                 // triggered by another inbound event handler method.  If fired immediately, the call stack
609                 // will look like this for example:
610                 //
611                 //   handlerA.inboundBufferUpdated() - (1) an inbound handler method closes a connection.
612                 //   -> handlerA.ctx.close()
613                 //     -> channel.unsafe.close()
614                 //       -> handlerA.channelInactive() - (2) another inbound handler method called while in (1) yet
615                 //
616                 // which means the execution of two inbound handler methods of the same handler overlap undesirably.
617                 executor().execute(task);
618             } catch (RejectedExecutionException e) {
619                 logger.warn("Can't invoke task later as EventLoop rejected it", e);
620             }
621         }

623         @Override
624         public void beginRead() {
625             if (!isActive()) {
626                 return;
627             }
628             updateLocalWindowIfNeeded();
629 
630             switch (readStatus) {
631                 case IDLE:
632                     readStatus = ReadStatus.IN_PROGRESS;
633                     doBeginRead();
634                     break;
635                 case IN_PROGRESS:
636                     readStatus = ReadStatus.REQUESTED;
637                     break;
638                 default:
639                     break;
640             }
641         }
642 
643         private Object pollQueuedMessage() {
644             return inboundBuffer == null ? null : inboundBuffer.poll();
645         }
646 
647         void doBeginRead() {
648             // Process messages until there are none left (or the user stopped requesting) and also handle EOS.
649             while (readStatus != ReadStatus.IDLE) {
650                 Object message = pollQueuedMessage();
651                 if (message == null) {
652                     if (readEOS) {
653                         unsafe.closeForcibly();
654                     }
655                     // We need to double check that there is nothing left to flush such as a
656                     // window update frame.
657                     flush();
658                     break;
659                 }
660                 final RecvBufferAllocator.Handle allocHandle = recvBufAllocHandle();
661                 allocHandle.reset(config());
662                 boolean continueReading = false;
663                 do {
664                     doRead0((Http2Frame) message, allocHandle);
665                 } while ((readEOS || (continueReading = allocHandle.continueReading()))
666                         && (message = pollQueuedMessage()) != null);
667 
668                 if (continueReading && isParentReadInProgress() && !readEOS) {
669                     // Currently the parent and child channel are on the same EventLoop thread. If the parent is
670                     // currently reading it is possible that more frames will be delivered to this child channel. In
671                     // the case that this child channel still wants to read we delay the channelReadComplete on this
672                     // child channel until the parent is done reading.
673                     maybeAddChannelToReadCompletePendingQueue();
674                 } else {
675                     notifyReadComplete(allocHandle, true);
676                 }
677             }
678         }
679 
680         void readEOS() {
681             readEOS = true;
682         }

684         private void updateLocalWindowIfNeeded() {
685             if (flowControlledBytes != 0) {
686                 int bytes = flowControlledBytes;
687                 flowControlledBytes = 0;
688                 Future<Void> future = write0(parentContext(), new DefaultHttp2WindowUpdateFrame(bytes).stream(stream));
689                 // window update frames are commonly swallowed by the Http2FrameCodec and the promise is synchronously
690                 // completed but the flow controller _may_ have generated a wire level WINDOW_UPDATE. Therefore we need,
691                 // to assume there was a write done that needs to be flushed or we risk flow control starvation.
692                 writeDoneAndNoFlush = true;
693                 // Add a listener which will notify and teardown the stream
694                 // when a window update fails if needed or check the result of the future directly if it was completed
695                 // already.
696                 // See https://github.com/netty/netty/issues/9663
697                 if (future.isDone()) {
698                     windowUpdateFrameWriteComplete(AbstractHttp2StreamChannel.this, future);
699                 } else {
700                     future.addListener(AbstractHttp2StreamChannel.this,
701                                        AbstractHttp2StreamChannel::windowUpdateFrameWriteComplete);
702                 }
703             }
704         }
705 
706         void notifyReadComplete(RecvBufferAllocator.Handle allocHandle, boolean forceReadComplete) {
707             if (!readCompletePending && !forceReadComplete) {
708                 return;
709             }
710             // Set to false just in case we added the channel multiple times before.
711             readCompletePending = false;
712 
713             if (readStatus == ReadStatus.REQUESTED) {
714                 readStatus = ReadStatus.IN_PROGRESS;
715             } else {
716                 readStatus = ReadStatus.IDLE;
717             }
718 
719             allocHandle.readComplete();
720             pipeline().fireChannelReadComplete();
721             if (config().isAutoRead()) {
722                 read();
723             }
724 
725             // Reading data may result in frames being written (e.g. WINDOW_UPDATE, RST, etc..). If the parent
726             // channel is not currently reading we need to force a flush at the child channel, because we cannot
727             // rely upon flush occurring in channelReadComplete on the parent channel.
728             flush();
729             if (readEOS) {
730                 unsafe.closeForcibly();
731             }
732         }

734         void doRead0(Http2Frame frame, RecvBufferAllocator.Handle allocHandle) {
735             final int bytes;
736             if (frame instanceof Http2DataFrame) {
737                 bytes = ((Http2DataFrame) frame).initialFlowControlledBytes();
738 
739                 // It is important that we increment the flowControlledBytes before we call fireChannelRead(...)
740                 // as it may cause a read() that will call updateLocalWindowIfNeeded() and we need to ensure
741                 // in this case that we accounted for it.
742                 //
743                 // See https://github.com/netty/netty/issues/9663
744                 flowControlledBytes += bytes;
745             } else {
746                 bytes = MIN_HTTP2_FRAME_SIZE;
747             }
748             // Update before firing event through the pipeline to be consistent with other Channel implementation.
749             allocHandle.attemptedBytesRead(bytes);
750             allocHandle.lastBytesRead(bytes);
751             allocHandle.incMessagesRead(1);
752 
753             pipeline().fireChannelRead(frame);
754         }

756         @Override
757         public void write(Object msg, Promise<Void> promise) {
758             // After this point its not possible to cancel a write anymore.
759             if (!promise.setUncancellable()) {
760                 ReferenceCountUtil.release(msg);
761                 return;
762             }
763 
764             if (!isActive() ||
765                     // Once the outbound side was closed we should not allow header / data frames
766                     outboundClosed && (msg instanceof Http2HeadersFrame || msg instanceof Http2DataFrame)) {
767                 ReferenceCountUtil.release(msg);
768                 promise.setFailure(new ClosedChannelException());
769                 return;
770             }
771 
772             try {
773                 if (msg instanceof Http2StreamFrame) {
774                     Http2StreamFrame frame = validateStreamFrame((Http2StreamFrame) msg).stream(stream());
775                     writeHttp2StreamFrame(frame, promise);
776                 } else {
777                     String msgStr = msg.toString();
778                     ReferenceCountUtil.release(msg);
779                     promise.setFailure(new IllegalArgumentException(
780                             "Message must be an " + StringUtil.simpleClassName(Http2StreamFrame.class) +
781                                     ": " + msgStr));
782                 }
783             } catch (Throwable t) {
784                 promise.tryFailure(t);
785             }
786         }
787 
788         private void writeHttp2StreamFrame(Http2StreamFrame frame, Promise<Void> promise) {
789             if (!firstFrameWritten && !isStreamIdValid(stream().id()) && !(frame instanceof Http2HeadersFrame)) {
790                 ReferenceCountUtil.release(frame);
791                 promise.setFailure(
792                     new IllegalArgumentException("The first frame must be a headers frame. Was: "
793                         + frame.name()));
794                 return;
795             }
796 
797             final boolean firstWrite;
798             if (firstFrameWritten) {
799                 firstWrite = false;
800             } else {
801                 firstWrite = firstFrameWritten = true;
802             }
803 
804             Future<Void> f = write0(parentContext(), frame);
805             if (f.isDone()) {
806                 if (firstWrite) {
807                     firstWriteComplete(f, promise);
808                 } else {
809                     writeComplete(f, promise);
810                 }
811             } else {
812                 final long bytes = FlowControlledFrameSizeEstimator.HANDLE_INSTANCE.size(frame);
813                 incrementPendingOutboundBytes(bytes, false);
814                 f.addListener(future ->  {
815                     if (firstWrite) {
816                         firstWriteComplete(future, promise);
817                     } else {
818                         writeComplete(future, promise);
819                     }
820                     decrementPendingOutboundBytes(bytes, false);
821                 });
822                 writeDoneAndNoFlush = true;
823             }
824         }
825 
826         private void firstWriteComplete(Future<?> future, Promise<Void> promise) {
827             Throwable cause = future.cause();
828             if (cause == null) {
829                 promise.setSuccess(null);
830             } else {
831                 // If the first write fails there is not much we can do, just close
832                 closeForcibly();
833                 promise.setFailure(wrapStreamClosedError(cause));
834             }
835         }
836 
837         private void writeComplete(Future<?> future, Promise<Void> promise) {
838             Throwable cause = future.cause();
839             if (cause == null) {
840                 promise.setSuccess(null);
841             } else {
842                 Throwable error = wrapStreamClosedError(cause);
843                 // To make it more consistent with AbstractChannel we handle all IOExceptions here.
844                 if (error instanceof IOException) {
845                     if (config.isAutoClose()) {
846                         // Close channel if needed.
847                         closeForcibly();
848                     } else {
849                         // TODO: Once Http2StreamChannel extends DuplexChannel we should call shutdownOutput(...)
850                         outboundClosed = true;
851                     }
852                 }
853                 promise.setFailure(error);
854             }
855         }
856 
857         private Throwable wrapStreamClosedError(Throwable cause) {
858             // If the error was caused by STREAM_CLOSED we should use a ClosedChannelException to better
859             // mimic other transports and make it easier to reason about what exceptions to expect.
860             if (cause instanceof Http2Exception && ((Http2Exception) cause).error() == Http2Error.STREAM_CLOSED) {
861                 return new ClosedChannelException().initCause(cause);
862             }
863             return cause;
864         }
865 
866         private Http2StreamFrame validateStreamFrame(Http2StreamFrame frame) {
867             if (frame.stream() != null && frame.stream() != stream) {
868                 String msgString = frame.toString();
869                 ReferenceCountUtil.release(frame);
870                 throw new IllegalArgumentException(
871                         "Stream " + frame.stream() + " must not be set on the frame: " + msgString);
872             }
873             return frame;
874         }

876         @Override
877         public void flush() {
878             // If we are currently in the parent channel's read loop we should just ignore the flush.
879             // We will ensure we trigger ctx.flush() after we processed all Channels later on and
880             // so aggregate the flushes. This is done as ctx.flush() is expensive when as it may trigger an
881             // write(...) or writev(...) operation on the socket.
882             if (!writeDoneAndNoFlush || isParentReadInProgress()) {
883                 // There is nothing to flush so this is a NOOP.
884                 return;
885             }
886             // We need to set this to false before we call flush0(...) as FutureListener may produce more data
887             // that are explicit flushed.
888             writeDoneAndNoFlush = false;
889             flush0(parentContext());
890         }

894             // Always return null as we not use the ChannelOutboundBuffer and not even support it.
895             return null;
896         }
