40  /**
41   * An HTTP/2 handler that creates child channels for each stream. This handler must be used in combination
42   * with {@link Http2FrameCodec}.
43   *
44   * <p>When a new stream is created, a new {@link Channel} is created for it. Applications send and
45   * receive {@link Http2StreamFrame}s on the created channel. {@link ByteBuf}s cannot be processed by the channel;
46   * all writes that reach the head of the pipeline must be an instance of {@link Http2StreamFrame}. Writes that reach
47   * the head of the pipeline are processed directly by this handler and cannot be intercepted.
48   *
49   * <p>The child channel will be notified of user events that impact the stream, such as {@link
50   * Http2GoAwayFrame} and {@link Http2ResetFrame}, as soon as they occur. Although {@code
51   * Http2GoAwayFrame} and {@code Http2ResetFrame} signify that the remote is ignoring further
52   * communication, closing of the channel is delayed until any inbound queue is drained with {@link
53   * Channel#read()}, which follows the default behavior of channels in Netty. Applications are
54   * free to close the channel in response to such events if they don't have use for any queued
55   * messages. Any connection level events like {@link Http2SettingsFrame} and {@link Http2GoAwayFrame}
56   * will be processed internally and also propagated down the pipeline for other handlers to act on.
57   *
58   * <p>Outbound streams are supported via the {@link Http2StreamChannelBootstrap}.
59   *
60   * <p>{@link ChannelConfig#setMaxMessagesPerRead(int)} and {@link ChannelConfig#setAutoRead(boolean)} are supported.
61   *
62   * <h3>Reference Counting</h3>
63   *
64   * Some {@link Http2StreamFrame}s implement the {@link ReferenceCounted} interface, as they carry
65   * reference counted objects (e.g. {@link ByteBuf}s). The multiplex codec will call {@link ReferenceCounted#retain()}
66   * before propagating a reference counted object through the pipeline, and thus an application handler needs to release
67   * such an object after having consumed it. For more information on reference counting take a look at
68   * https://netty.io/wiki/reference-counted-objects.html
69   *
70   * <h3>Channel Events</h3>
71   *
72   * A child channel becomes active as soon as it is registered to an {@link EventLoop}. Therefore, an active channel
73   * does not map to an active HTTP/2 stream immediately. Only once a {@link Http2HeadersFrame} has been successfully sent
74   * or received, does the channel map to an active HTTP/2 stream. In case it is not possible to open a new HTTP/2 stream
75   * (i.e. due to the maximum number of active streams being exceeded), the child channel receives an exception
76   * indicating the cause and is closed immediately thereafter.
77   *
78   * <h3>Writability and Flow Control</h3>
79   *
80   * A child channel observes outbound/remote flow control via the channel's writability. A channel only becomes writable
81   * when it maps to an active HTTP/2 stream . A child channel does not know about the connection-level flow control
82   * window. {@link ChannelHandler}s are free to ignore the channel's writability, in which case the excessive writes will
83   * be buffered by the parent channel. It's important to note that only {@link Http2DataFrame}s are subject to
84   * HTTP/2 flow control.
85   */
86  @UnstableApi
87  public final class Http2MultiplexHandler extends Http2ChannelDuplexHandler {

89      static final FutureContextListener<Channel, Void> CHILD_CHANNEL_REGISTRATION_LISTENER =
90              Http2MultiplexHandler::registerDone;

94      private final Queue<AbstractHttp2StreamChannel> readCompletePendingQueue =
95              new MaxCapacityQueue<AbstractHttp2StreamChannel>(new ArrayDeque<AbstractHttp2StreamChannel>(8),
96                      // Choose 100 which is what is used most of the times as default.
97                      Http2CodecUtil.SMALLEST_MAX_CONCURRENT_STREAMS);

128     static void registerDone(Channel childChannel, Future<?> future) {
129         // Handle any errors that occurred on the local thread while registering. Even though
130         // failures can happen after this point, they will be handled by the channel by closing the
131         // childChannel.
132         if (future.isFailed()) {
133             if (childChannel.isRegistered()) {
134                 childChannel.close();
135             } else {
136                 childChannel.unsafe().closeForcibly();
137             }
138         }
139     }

142     protected void handlerAdded0(ChannelHandlerContext ctx) {
143         if (ctx.executor() != ctx.channel().executor()) {
144             throw new IllegalStateException("EventExecutor must be EventLoop of Channel");
145         }
146         this.ctx = ctx;
147     }

154     @Override
155     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
156         parentReadInProgress = true;
157         if (msg instanceof Http2StreamFrame) {
158             if (msg instanceof Http2WindowUpdateFrame) {
159                 // We dont want to propagate update frames to the user
160                 return;
161             }
162             Http2StreamFrame streamFrame = (Http2StreamFrame) msg;
163             DefaultHttp2FrameStream s =
164                     (DefaultHttp2FrameStream) streamFrame.stream();
165 
166             AbstractHttp2StreamChannel channel = (AbstractHttp2StreamChannel) s.attachment;
167             if (msg instanceof Http2ResetFrame) {
168                 // Reset frames needs to be propagated via user events as these are not flow-controlled and so
169                 // must not be controlled by suppressing channel.read() on the child channel.
170                 channel.pipeline().fireUserEventTriggered(msg);
171 
172                 // RST frames will also trigger closing of the streams which then will call
173                 // AbstractHttp2StreamChannel.streamClosed()
174             } else {
175                 channel.fireChildRead(streamFrame);
176             }
177             return;
178         }
179 
180         if (msg instanceof Http2GoAwayFrame) {
181             // goaway frames will also trigger closing of the streams which then will call
182             // AbstractHttp2StreamChannel.streamClosed()
183             onHttp2GoAwayFrame(ctx, (Http2GoAwayFrame) msg);
184         }
185 
186         // Send everything down the pipeline
187         ctx.fireChannelRead(msg);
188     }

190     @Override
191     public void channelWritabilityChanged(final ChannelHandlerContext ctx) throws Exception {
192         if (ctx.channel().isWritable()) {
193             // While the writability state may change during iterating of the streams we just set all of the streams
194             // to writable to not affect fairness. These will be "limited" by their own watermarks in any case.
195             forEachActiveStream(AbstractHttp2StreamChannel.WRITABLE_VISITOR);
196         }
197 
198         ctx.fireChannelWritabilityChanged();
199     }

201     @Override
202     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
203         if (evt instanceof Http2FrameStreamEvent) {
204             Http2FrameStreamEvent event = (Http2FrameStreamEvent) evt;
205             DefaultHttp2FrameStream stream = (DefaultHttp2FrameStream) event.stream();
206             if (event.type() == Http2FrameStreamEvent.Type.State) {
207                 switch (stream.state()) {
208                     case HALF_CLOSED_LOCAL:
209                         if (stream.id() != Http2CodecUtil.HTTP_UPGRADE_STREAM_ID) {
210                             // Ignore everything which was not caused by an upgrade
211                             break;
212                         }
213                         // fall-through
214                     case HALF_CLOSED_REMOTE:
215                         // fall-through
216                     case OPEN:
217                         if (stream.attachment != null) {
218                             // ignore if child channel was already created.
219                             break;
220                         }
221                         final AbstractHttp2StreamChannel ch;
222                         // We need to handle upgrades special when on the client side.
223                         if (stream.id() == Http2CodecUtil.HTTP_UPGRADE_STREAM_ID && !isServer(ctx)) {
224                             // We must have an upgrade handler or else we can't handle the stream
225                             if (upgradeStreamHandler == null) {
226                                 throw connectionError(INTERNAL_ERROR,
227                                         "Client is misconfigured for upgrade requests");
228                             }
229                             ch = new Http2MultiplexHandlerStreamChannel(stream, upgradeStreamHandler);
230                             ch.closeOutbound();
231                         } else {
232                             ch = new Http2MultiplexHandlerStreamChannel(stream, inboundStreamHandler);
233                         }
234                         Future<Void> future = ch.register();
235                         if (future.isDone()) {
236                             registerDone(ch, future);
237                         } else {
238                             future.addListener(ch, CHILD_CHANNEL_REGISTRATION_LISTENER);
239                         }
240                         break;
241                     case CLOSED:
242                         AbstractHttp2StreamChannel channel = (AbstractHttp2StreamChannel) stream.attachment;
243                         if (channel != null) {
244                             channel.streamClosed();
245                         }
246                         break;
247                     default:
248                         // ignore for now
249                         break;
250                 }
251             }
252             return;
253         }
254         ctx.fireUserEventTriggered(evt);
255     }

262     @Override
263     public void exceptionCaught(ChannelHandlerContext ctx, final Throwable cause) throws Exception {
264         if (cause instanceof Http2FrameStreamException) {
265             Http2FrameStreamException exception = (Http2FrameStreamException) cause;
266             Http2FrameStream stream = exception.stream();
267             AbstractHttp2StreamChannel childChannel = (AbstractHttp2StreamChannel)
268                     ((DefaultHttp2FrameStream) stream).attachment;
269             try {
270                 childChannel.pipeline().fireExceptionCaught(cause.getCause());
271             } finally {
272                 childChannel.unsafe().closeForcibly();
273             }
274             return;
275         }
276         if (cause.getCause() instanceof SSLException) {
277             forEachActiveStream(new Http2FrameStreamVisitor() {
278                 @Override
279                 public boolean visit(Http2FrameStream stream) {
280                     AbstractHttp2StreamChannel childChannel = (AbstractHttp2StreamChannel)
281                             ((DefaultHttp2FrameStream) stream).attachment;
282                     childChannel.pipeline().fireExceptionCaught(cause);
283                     return true;
284                 }
285             });
286         }
287         ctx.fireExceptionCaught(cause);
288     }

290     private static boolean isServer(ChannelHandlerContext ctx) {
291         return ctx.channel().parent() instanceof ServerChannel;
292     }

294     private void onHttp2GoAwayFrame(ChannelHandlerContext ctx, final Http2GoAwayFrame goAwayFrame) {
295         if (goAwayFrame.lastStreamId() == Integer.MAX_VALUE) {
296             // None of the streams can have an id greater than Integer.MAX_VALUE
297             return;
298         }
299         // Notify which streams were not processed by the remote peer and are safe to retry on another connection:
300         try {
301             final boolean server = isServer(ctx);
302             forEachActiveStream(stream -> {
303                 final int streamId = stream.id();
304                 if (streamId > goAwayFrame.lastStreamId() && Http2CodecUtil.isStreamIdValid(streamId, server)) {
305                     final AbstractHttp2StreamChannel childChannel = (AbstractHttp2StreamChannel)
306                             ((DefaultHttp2FrameStream) stream).attachment;
307                     childChannel.pipeline().fireUserEventTriggered(goAwayFrame.retainedDuplicate());
308                 }
309                 return true;
310             });
311         } catch (Http2Exception e) {
312             ctx.fireExceptionCaught(e);
313             ctx.close();
314         }
315     }

317     /**
318      * Notifies any child streams of the read completion.
319      */
320     @Override
321     public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
322         processPendingReadCompleteQueue();
323         ctx.fireChannelReadComplete();
324     }
325 
326     private void processPendingReadCompleteQueue() {
327         parentReadInProgress = true;
328         // If we have many child channel we can optimize for the case when multiple call flush() in
329         // channelReadComplete(...) callbacks and only do it once as otherwise we will end-up with multiple
330         // write calls on the socket which is expensive.
331         AbstractHttp2StreamChannel childChannel = readCompletePendingQueue.poll();
332         if (childChannel != null) {
333             try {
334                 do {
335                     childChannel.fireChildReadComplete();
336                     childChannel = readCompletePendingQueue.poll();
337                 } while (childChannel != null);
338             } finally {
339                 parentReadInProgress = false;
340                 readCompletePendingQueue.clear();
341                 ctx.flush();
342             }
343         } else {
344             parentReadInProgress = false;
345         }
346     }

348     private final class Http2MultiplexHandlerStreamChannel extends AbstractHttp2StreamChannel {
349 
350         Http2MultiplexHandlerStreamChannel(DefaultHttp2FrameStream stream, ChannelHandler inboundHandler) {
351             super(stream, ++idCount, inboundHandler);
352         }
353 
354         @Override
355         protected boolean isParentReadInProgress() {
356             return parentReadInProgress;
357         }
358 
359         @Override
360         protected void addChannelToReadCompletePendingQueue() {
361             // If there is no space left in the queue, just keep on processing everything that is already
362             // stored there and try again.
363             while (!readCompletePendingQueue.offer(this)) {
364                 processPendingReadCompleteQueue();
365             }
366         }
367 
368         @Override
369         protected ChannelHandlerContext parentContext() {
370             return ctx;
371         }
372     }
