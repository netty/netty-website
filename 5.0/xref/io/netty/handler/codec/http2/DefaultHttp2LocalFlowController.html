38  /**
39   * Basic implementation of {@link Http2LocalFlowController}.
40   * <p>
41   * This class is <strong>NOT</strong> thread safe. The assumption is all methods must be invoked from a single thread.
42   * Typically this thread is the event loop thread for the {@link ChannelHandlerContext} managed by this class.
43   */
44  @UnstableApi
45  public class DefaultHttp2LocalFlowController implements Http2LocalFlowController {
46      /**
47       * The default ratio of window size to initial window size below which a {@code WINDOW_UPDATE}
48       * is sent to expand the window.
49       */
50      public static final float DEFAULT_WINDOW_UPDATE_RATIO = 0.5f;
51  
52      private final Http2Connection connection;
53      private final Http2Connection.PropertyKey stateKey;
54      private Http2FrameWriter frameWriter;
55      private ChannelHandlerContext ctx;
56      private float windowUpdateRatio;
57      private int initialWindowSize = DEFAULT_WINDOW_SIZE;
58  
59      public DefaultHttp2LocalFlowController(Http2Connection connection) {
60          this(connection, DEFAULT_WINDOW_UPDATE_RATIO, false);
61      }
62  
63      /**
64       * Constructs a controller with the given settings.
65       *
66       * @param connection the connection state.
67       * @param windowUpdateRatio the window percentage below which to send a {@code WINDOW_UPDATE}.
68       * @param autoRefillConnectionWindow if {@code true}, effectively disables the connection window
69       * in the flow control algorithm as they will always refill automatically without requiring the
70       * application to consume the bytes. When enabled, the maximum bytes you must be prepared to
71       * queue is proportional to {@code maximum number of concurrent streams * the initial window
72       * size per stream}
73       * (<a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_MAX_CONCURRENT_STREAMS</a>
74       * <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_INITIAL_WINDOW_SIZE</a>).
75       */
76      public DefaultHttp2LocalFlowController(Http2Connection connection,
77                                             float windowUpdateRatio,
78                                             boolean autoRefillConnectionWindow) {
79          this.connection = requireNonNull(connection, "connection");
80          windowUpdateRatio(windowUpdateRatio);
81  
82          // Add a flow state for the connection.
83          stateKey = connection.newKey();
84          FlowState connectionState = autoRefillConnectionWindow ?
85                  new AutoRefillState(connection.connectionStream(), initialWindowSize) :
86                  new DefaultState(connection.connectionStream(), initialWindowSize);
87          connection.connectionStream().setProperty(stateKey, connectionState);
88  
89          // Register for notification of new streams.
90          connection.addListener(new Http2ConnectionAdapter() {
91              @Override
92              public void onStreamAdded(Http2Stream stream) {
93                  // Unconditionally used the reduced flow control state because it requires no object allocation
94                  // and the DefaultFlowState will be allocated in onStreamActive.
95                  stream.setProperty(stateKey, REDUCED_FLOW_STATE);
96              }
97  
98              @Override
99              public void onStreamActive(Http2Stream stream) {
100                 // Need to be sure the stream's initial window is adjusted for SETTINGS
101                 // frames which may have been exchanged while it was in IDLE
102                 stream.setProperty(stateKey, new DefaultState(stream, initialWindowSize));
103             }
104 
105             @Override
106             public void onStreamClosed(Http2Stream stream) {
107                 try {
108                     // When a stream is closed, consume any remaining bytes so that they
109                     // are restored to the connection window.
110                     FlowState state = state(stream);
111                     int unconsumedBytes = state.unconsumedBytes();
112                     if (ctx != null && unconsumedBytes > 0) {
113                         if (consumeAllBytes(state, unconsumedBytes)) {
114                             // As the user has no real control on when this callback is used we should better
115                             // call flush() if we produced any window update to ensure we not stale.
116                             ctx.flush();
117                         }
118                     }
119                 } catch (Http2Exception e) {
120                     PlatformDependent.throwException(e);
121                 } finally {
122                     // Unconditionally reduce the amount of memory required for flow control because there is no
123                     // object allocation costs associated with doing so and the stream will not have any more
124                     // local flow control state to keep track of anymore.
125                     stream.setProperty(stateKey, REDUCED_FLOW_STATE);
126                 }
127             }
128         });
129     }

142     @Override
143     public void initialWindowSize(int newWindowSize) throws Http2Exception {
144         assert ctx == null || ctx.executor().inEventLoop();
145         int delta = newWindowSize - initialWindowSize;
146         initialWindowSize = newWindowSize;
147 
148         WindowUpdateVisitor visitor = new WindowUpdateVisitor(delta);
149         connection.forEachActiveStream(visitor);
150         visitor.throwIfError();
151     }

155         return initialWindowSize;

160         return state(stream).windowSize();
161     }

165         return state(stream).initialWindowSize();

168     @Override
169     public void incrementWindowSize(Http2Stream stream, int delta) throws Http2Exception {
170         assert ctx != null && ctx.executor().inEventLoop();
171         FlowState state = state(stream);
172         // Just add the delta to the stream-specific initial window size so that the next time the window
173         // expands it will grow to the new initial size.
174         state.incrementInitialStreamWindow(delta);
175         state.writeWindowUpdateIfNeeded();
176     }
177 
178     @Override
179     public boolean consumeBytes(Http2Stream stream, int numBytes) throws Http2Exception {
180         assert ctx != null && ctx.executor().inEventLoop();
181         checkPositiveOrZero(numBytes, "numBytes");
182         if (numBytes == 0) {
183             return false;
184         }
185 
186         // Streams automatically consume all remaining bytes when they are closed, so just ignore
187         // if already closed.
188         if (stream != null && !isClosed(stream)) {
189             if (stream.id() == CONNECTION_STREAM_ID) {
190                 throw new UnsupportedOperationException("Returning bytes for the connection window is not supported");
191             }
192 
193             return consumeAllBytes(state(stream), numBytes);
194         }
195         return false;
196     }
197 
198     private boolean consumeAllBytes(FlowState state, int numBytes) throws Http2Exception {
199         return connectionState().consumeBytes(numBytes) | state.consumeBytes(numBytes);
200     }
201 
202     @Override
203     public int unconsumedBytes(Http2Stream stream) {
204         return state(stream).unconsumedBytes();
205     }

207     private static void checkValidRatio(float ratio) {
208         if (Double.compare(ratio, 0.0) <= 0 || Double.compare(ratio, 1.0) >= 0) {
209             throw new IllegalArgumentException("Invalid ratio: " + ratio);
210         }
211     }

213     /**
214      * The window update ratio is used to determine when a window update must be sent. If the ratio
215      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
216      * be sent. This is the global window update ratio that will be used for new streams.
217      * @param ratio the ratio to use when checking if a {@code WINDOW_UPDATE} is determined necessary for new streams.
218      * @throws IllegalArgumentException If the ratio is out of bounds (0, 1).
219      */
220     public void windowUpdateRatio(float ratio) {
221         assert ctx == null || ctx.executor().inEventLoop();
222         checkValidRatio(ratio);
223         windowUpdateRatio = ratio;
224     }
225 
226     /**
227      * The window update ratio is used to determine when a window update must be sent. If the ratio
228      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
229      * be sent. This is the global window update ratio that will be used for new streams.
230      */
231     public float windowUpdateRatio() {
232         return windowUpdateRatio;
233     }
234 
235     /**
236      * The window update ratio is used to determine when a window update must be sent. If the ratio
237      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
238      * be sent. This window update ratio will only be applied to {@code streamId}.
239      * <p>
240      * Note it is the responsibly of the caller to ensure that the the
241      * initial {@code SETTINGS} frame is sent before this is called. It would
242      * be considered a {@link Http2Error#PROTOCOL_ERROR} if a {@code WINDOW_UPDATE}
243      * was generated by this method before the initial {@code SETTINGS} frame is sent.
244      * @param stream the stream for which {@code ratio} applies to.
245      * @param ratio the ratio to use when checking if a {@code WINDOW_UPDATE} is determined necessary.
246      * @throws Http2Exception If a protocol-error occurs while generating {@code WINDOW_UPDATE} frames
247      */
248     public void windowUpdateRatio(Http2Stream stream, float ratio) throws Http2Exception {
249         assert ctx != null && ctx.executor().inEventLoop();
250         checkValidRatio(ratio);
251         FlowState state = state(stream);
252         state.windowUpdateRatio(ratio);
253         state.writeWindowUpdateIfNeeded();
254     }
255 
256     /**
257      * The window update ratio is used to determine when a window update must be sent. If the ratio
258      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
259      * be sent. This window update ratio will only be applied to {@code streamId}.
260      * @throws Http2Exception If no stream corresponding to {@code stream} could be found.
261      */
262     public float windowUpdateRatio(Http2Stream stream) throws Http2Exception {
263         return state(stream).windowUpdateRatio();
264     }

266     @Override
267     public void receiveFlowControlledFrame(Http2Stream stream, ByteBuf data, int padding,
268             boolean endOfStream) throws Http2Exception {
269         assert ctx != null && ctx.executor().inEventLoop();
270         int dataLength = data.readableBytes() + padding;
271 
272         // Apply the connection-level flow control
273         FlowState connectionState = connectionState();
274         connectionState.receiveFlowControlledFrame(dataLength);
275 
276         if (stream != null && !isClosed(stream)) {
277             // Apply the stream-level flow control
278             FlowState state = state(stream);
279             state.endOfStream(endOfStream);
280             state.receiveFlowControlledFrame(dataLength);
281         } else if (dataLength > 0) {
282             // Immediately consume the bytes for the connection window.
283             connectionState.consumeBytes(dataLength);
284         }
285     }

287     private FlowState connectionState() {
288         return connection.connectionStream().getProperty(stateKey);
289     }

291     private FlowState state(Http2Stream stream) {
292         return stream.getProperty(stateKey);

295     private static boolean isClosed(Http2Stream stream) {
296         return stream.state() == Http2Stream.State.CLOSED;

300      * Flow control state that does autorefill of the flow control window when the data is
301      * received.
302      */
303     private final class AutoRefillState extends DefaultState {
304         AutoRefillState(Http2Stream stream, int initialWindowSize) {
305             super(stream, initialWindowSize);
306         }
307 
308         @Override
309         public void receiveFlowControlledFrame(int dataLength) throws Http2Exception {
310             super.receiveFlowControlledFrame(dataLength);
311             // Need to call the super to consume the bytes, since this.consumeBytes does nothing.
312             super.consumeBytes(dataLength);
313         }
314 
315         @Override
316         public boolean consumeBytes(int numBytes) throws Http2Exception {
317             // Do nothing, since the bytes are already consumed upon receiving the data.
318             return false;
319         }

322     /**
323      * Flow control window state for an individual stream.
324      */
325     private class DefaultState implements FlowState {
326         private final Http2Stream stream;
327 
328         /**
329          * The actual flow control window that is decremented as soon as {@code DATA} arrives.
330          */
331         private int window;
332 
333         /**
334          * A view of {@link #window} that is used to determine when to send {@code WINDOW_UPDATE}
335          * frames. Decrementing this window for received {@code DATA} frames is delayed until the
336          * application has indicated that the data has been fully processed. This prevents sending
337          * a {@code WINDOW_UPDATE} until the number of processed bytes drops below the threshold.
338          */
339         private int processedWindow;
340 
341         /**
342          * This is what is used to determine how many bytes need to be returned relative to {@link #processedWindow}.
343          * Each stream has their own initial window size.
344          */
345         private int initialStreamWindowSize;
346 
347         /**
348          * This is used to determine when {@link #processedWindow} is sufficiently far away from
349          * {@link #initialStreamWindowSize} such that a {@code WINDOW_UPDATE} should be sent.
350          * Each stream has their own window update ratio.
351          */
352         private float streamWindowUpdateRatio;
353 
354         private int lowerBound;
355         private boolean endOfStream;
356 
357         DefaultState(Http2Stream stream, int initialWindowSize) {
358             this.stream = stream;
359             window(initialWindowSize);
360             streamWindowUpdateRatio = windowUpdateRatio;
361         }
362 
363         @Override
364         public void window(int initialWindowSize) {
365             assert ctx == null || ctx.executor().inEventLoop();
366             window = processedWindow = initialStreamWindowSize = initialWindowSize;
367         }
368 
369         @Override
370         public int windowSize() {
371             return window;
372         }
373 
374         @Override
375         public int initialWindowSize() {
376             return initialStreamWindowSize;
377         }
378 
379         @Override
380         public void endOfStream(boolean endOfStream) {
381             this.endOfStream = endOfStream;
382         }
383 
384         @Override
385         public float windowUpdateRatio() {
386             return streamWindowUpdateRatio;
387         }
388 
389         @Override
390         public void windowUpdateRatio(float ratio) {
391             assert ctx == null || ctx.executor().inEventLoop();
392             streamWindowUpdateRatio = ratio;
393         }
394 
395         @Override
396         public void incrementInitialStreamWindow(int delta) {
397             // Clip the delta so that the resulting initialStreamWindowSize falls within the allowed range.
398             int newValue = (int) min(MAX_INITIAL_WINDOW_SIZE,
399                     max(MIN_INITIAL_WINDOW_SIZE, initialStreamWindowSize + (long) delta));
400             delta = newValue - initialStreamWindowSize;
401 
402             initialStreamWindowSize += delta;
403         }
404 
405         @Override
406         public void incrementFlowControlWindows(int delta) throws Http2Exception {
407             if (delta > 0 && window > MAX_INITIAL_WINDOW_SIZE - delta) {
408                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
409                         "Flow control window overflowed for stream: %d", stream.id());
410             }
411 
412             window += delta;
413             processedWindow += delta;
414             lowerBound = min(delta, 0);
415         }
416 
417         @Override
418         public void receiveFlowControlledFrame(int dataLength) throws Http2Exception {
419             assert dataLength >= 0;
420 
421             // Apply the delta. Even if we throw an exception we want to have taken this delta into account.
422             window -= dataLength;
423 
424             // Window size can become negative if we sent a SETTINGS frame that reduces the
425             // size of the transfer window after the peer has written data frames.
426             // The value is bounded by the length that SETTINGS frame decrease the window.
427             // This difference is stored for the connection when writing the SETTINGS frame
428             // and is cleared once we send a WINDOW_UPDATE frame.
429             if (window < lowerBound) {
430                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
431                         "Flow control window exceeded for stream: %d", stream.id());
432             }
433         }
434 
435         private void returnProcessedBytes(int delta) throws Http2Exception {
436             if (processedWindow - delta < window) {
437                 throw streamError(stream.id(), INTERNAL_ERROR,
438                         "Attempting to return too many bytes for stream %d", stream.id());
439             }
440             processedWindow -= delta;
441         }
442 
443         @Override
444         public boolean consumeBytes(int numBytes) throws Http2Exception {
445             // Return the bytes processed and update the window.
446             returnProcessedBytes(numBytes);
447             return writeWindowUpdateIfNeeded();
448         }
449 
450         @Override
451         public int unconsumedBytes() {
452             return processedWindow - window;
453         }
454 
455         @Override
456         public boolean writeWindowUpdateIfNeeded() throws Http2Exception {
457             if (endOfStream || initialStreamWindowSize <= 0 ||
458                     // If the stream is already closed there is no need to try to write a window update for it.
459                     isClosed(stream)) {
460                 return false;
461             }
462 
463             int threshold = (int) (initialStreamWindowSize * streamWindowUpdateRatio);
464             if (processedWindow <= threshold) {
465                 writeWindowUpdate();
466                 return true;
467             }
468             return false;
469         }
470 
471         /**
472          * Called to perform a window update for this stream (or connection). Updates the window size back
473          * to the size of the initial window and sends a window update frame to the remote endpoint.
474          */
475         private void writeWindowUpdate() throws Http2Exception {
476             // Expand the window for this stream back to the size of the initial window.
477             int deltaWindowSize = initialStreamWindowSize - processedWindow;
478             try {
479                 incrementFlowControlWindows(deltaWindowSize);
480             } catch (Throwable t) {
481                 throw connectionError(INTERNAL_ERROR, t,
482                         "Attempting to return too many bytes for stream %d", stream.id());
483             }
484 
485             // Send a window update for the stream/connection.
486             frameWriter.writeWindowUpdate(ctx, stream.id(), deltaWindowSize);
487         }
488     }

559     /**
560      * An abstraction which provides specific extensions used by local flow control.
561      */
562     private interface FlowState {
563 
564         int windowSize();
565 
566         int initialWindowSize();
567 
568         void window(int initialWindowSize);
569 
570         /**
571          * Increment the initial window size for this stream.
572          * @param delta The amount to increase the initial window size by.
573          */
574         void incrementInitialStreamWindow(int delta);
575 
576         /**
577          * Updates the flow control window for this stream if it is appropriate.
578          *
579          * @return true if {@code WINDOW_UPDATE} was written, false otherwise.
580          */
581         boolean writeWindowUpdateIfNeeded() throws Http2Exception;
582 
583         /**
584          * Indicates that the application has consumed {@code numBytes} from the connection or stream and is
585          * ready to receive more data.
586          *
587          * @param numBytes the number of bytes to be returned to the flow control window.
588          * @return true if {@code WINDOW_UPDATE} was written, false otherwise.
589          * @throws Http2Exception If the number of bytes is too great for the current window,
590          * or an internal error occurs.
591          */
592         boolean consumeBytes(int numBytes) throws Http2Exception;
593 
594         int unconsumedBytes();
595 
596         float windowUpdateRatio();
597 
598         void windowUpdateRatio(float ratio);
599 
600         /**
601          * A flow control event has occurred and we should decrement the amount of available bytes for this stream.
602          * @param dataLength The amount of data to for which this stream is no longer eligible to use for flow control.
603          * @throws Http2Exception If too much data is used relative to how much is available.
604          */
605         void receiveFlowControlledFrame(int dataLength) throws Http2Exception;
606 
607         /**
608          * Increment the windows which are used to determine many bytes have been processed.
609          * @param delta The amount to increment the window by.
610          * @throws Http2Exception if integer overflow occurs on the window.
611          */
612         void incrementFlowControlWindows(int delta) throws Http2Exception;
613 
614         void endOfStream(boolean endOfStream);
615     }

617     /**
618      * Provides a means to iterate over all active streams and increment the flow control windows.
619      */
620     private final class WindowUpdateVisitor implements Http2StreamVisitor {
621         private CompositeStreamException compositeException;
622         private final int delta;
623 
624         WindowUpdateVisitor(int delta) {
625             this.delta = delta;
626         }
627 
628         @Override
629         public boolean visit(Http2Stream stream) throws Http2Exception {
630             try {
631                 // Increment flow control window first so state will be consistent if overflow is detected.
632                 FlowState state = state(stream);
633                 state.incrementFlowControlWindows(delta);
634                 state.incrementInitialStreamWindow(delta);
635             } catch (StreamException e) {
636                 if (compositeException == null) {
637                     compositeException = new CompositeStreamException(e.error(), 4);
638                 }
639                 compositeException.add(e);
640             }
641             return true;
642         }
643 
644         public void throwIfError() throws CompositeStreamException {
645             if (compositeException != null) {
646                 throw compositeException;
647             }
648         }
649     }
