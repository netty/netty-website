179     /**
180      * Apply HTTP/2 rules while translating status code to {@link HttpResponseStatus}
181      *
182      * @param status The status from an HTTP/2 frame
183      * @return The HTTP/1.x status
184      * @throws Http2Exception If there is a problem translating from HTTP/2 to HTTP/1.x
185      */
186     public static HttpResponseStatus parseStatus(CharSequence status) throws Http2Exception {
187         HttpResponseStatus result;
188         try {
189             result = parseLine(status);
190             if (result == HttpResponseStatus.SWITCHING_PROTOCOLS) {
191                 throw connectionError(PROTOCOL_ERROR, "Invalid HTTP/2 status code '%d'", result.code());
192             }
193         } catch (Http2Exception e) {
194             throw e;
195         } catch (Throwable t) {
196             throw connectionError(PROTOCOL_ERROR, t,
197                             "Unrecognized HTTP status code '%s' encountered in translation to HTTP/1.x", status);
198         }
199         return result;
200     }

220     /**
221      * Create a new object to contain the response data
222      *
223      * @param streamId The stream associated with the response
224      * @param http2Headers The initial set of HTTP/2 headers to create the response with
225      * @param content {@link Buffer} content to put in {@link FullHttpResponse}
226      * @param validateHttpHeaders <ul>
227      *        <li>{@code true} to validate HTTP headers in the http-codec</li>
228      *        <li>{@code false} not to validate HTTP headers in the http-codec</li>
229      *        </ul>
230      * @return A new response object which represents headers/data
231      * @throws Http2Exception see {@link #addHttp2ToHttpHeaders(int, Http2Headers, FullHttpMessage, boolean)}
232      */
233     public static FullHttpResponse toFullHttpResponse(int streamId, Http2Headers http2Headers, Buffer content,
234                                                       boolean validateHttpHeaders)
235                     throws Http2Exception {
236         HttpResponseStatus status = parseStatus(http2Headers.status());
237         // HTTP/2 does not define a way to carry the version or reason phrase that is included in an
238         // HTTP/1.1 status line.
239         FullHttpResponse msg = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, status, content,
240                                                            validateHttpHeaders);
241         try {
242             addHttp2ToHttpHeaders(streamId, http2Headers, msg, false);
243         } catch (Http2Exception e) {
244             msg.close();
245             throw e;
246         } catch (Throwable t) {
247             msg.close();
248             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
249         }
250         return msg;
251     }

271     private static CharSequence extractPath(CharSequence method, Http2Headers headers) {
272         if (HttpMethod.CONNECT.asciiName().contentEqualsIgnoreCase(method)) {
273             // See https://tools.ietf.org/html/rfc7231#section-4.3.6
274             return requireNonNull(headers.authority(),
275                     "authority header cannot be null in the conversion to HTTP/1.x");
276         } else {
277             return requireNonNull(headers.path(),
278                     "path header cannot be null in conversion to HTTP/1.x");
279         }
280     }

284      *
285      * @param streamId The stream associated with the request
286      * @param http2Headers The initial set of HTTP/2 headers to create the request with
287      * @param content {@link Buffer} content to put in {@link FullHttpRequest}
288      * @param validateHttpHeaders <ul>

293      * @throws Http2Exception see {@link #addHttp2ToHttpHeaders(int, Http2Headers, FullHttpMessage, boolean)}

309         } catch (Throwable t) {

311             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
312         }

393     /**
394      * Translate and add HTTP/2 headers to HTTP/1.x headers.
395      *
396      * @param streamId The stream associated with {@code sourceHeaders}.
397      * @param inputHeaders The HTTP/2 headers to convert.
398      * @param outputHeaders The object which will contain the resulting HTTP/1.x headers..
399      * @param httpVersion What HTTP/1.x version {@code outputHeaders} should be treated as when doing the conversion.
400      * @param isTrailer {@code true} if {@code outputHeaders} should be treated as trailing headers.
401      * {@code false} otherwise.
402      * @param isRequest {@code true} if the {@code outputHeaders} will be used in a request message.
403      * {@code false} for response message.
404      * @throws Http2Exception If not all HTTP/2 headers can be translated to HTTP/1.x.
405      */
406     public static void addHttp2ToHttpHeaders(int streamId, Http2Headers inputHeaders, HttpHeaders outputHeaders,
407             HttpVersion httpVersion, boolean isTrailer, boolean isRequest) throws Http2Exception {
408         Http2ToHttpHeaderTranslator translator = new Http2ToHttpHeaderTranslator(streamId, outputHeaders, isRequest);
409         try {
410             translator.translateHeaders(inputHeaders);
411         } catch (Http2Exception ex) {
412             throw ex;
413         } catch (Throwable t) {
414             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
415         }
416 
417         outputHeaders.remove(HttpHeaderNames.TRANSFER_ENCODING);
418         outputHeaders.remove(HttpHeaderNames.TRAILER);
419         if (!isTrailer) {
420             outputHeaders.setInt(ExtensionHeaderNames.STREAM_ID.text(), streamId);
421             HttpUtil.setKeepAlive(outputHeaders, httpVersion, true);
422         }
423     }

425     /**
426      * Converts the given HTTP/1.x headers into HTTP/2 headers.
427      * The following headers are only used if they can not be found in from the {@code HOST} header or the
428      * {@code Request-Line} as defined by <a href="https://tools.ietf.org/html/rfc7230">rfc7230</a>
429      * <ul>
430      * <li>{@link ExtensionHeaderNames#SCHEME}</li>
431      * </ul>
432      * {@link ExtensionHeaderNames#PATH} is ignored and instead extracted from the {@code Request-Line}.
433      */
434     public static Http2Headers toHttp2Headers(HttpMessage in, boolean validateHeaders) {
435         HttpHeaders inHeaders = in.headers();
436         final Http2Headers out = new DefaultHttp2Headers(validateHeaders, inHeaders.size());
437         if (in instanceof HttpRequest) {
438             HttpRequest request = (HttpRequest) in;
439             URI requestTargetUri = URI.create(request.uri());
440             out.path(toHttp2Path(requestTargetUri));
441             out.method(request.method().asciiName());
442             setHttp2Scheme(inHeaders, requestTargetUri, out);
443 
444             if (!isOriginForm(requestTargetUri) && !isAsteriskForm(requestTargetUri)) {
445                 // Attempt to take from HOST header before taking from the request-line
446                 String host = inHeaders.getAsString(HttpHeaderNames.HOST);
447                 setHttp2Authority(host == null || host.isEmpty() ? requestTargetUri.getAuthority() : host, out);
448             }
449         } else if (in instanceof HttpResponse) {
450             HttpResponse response = (HttpResponse) in;
451             out.status(response.status().codeAsText());
452         }
453 
454         // Add the HTTP headers which have not been consumed above
455         toHttp2Headers(inHeaders, out);
456         return out;
457     }

459     public static Http2Headers toHttp2Headers(HttpHeaders inHeaders, boolean validateHeaders) {
460         if (inHeaders.isEmpty()) {
461             return EmptyHttp2Headers.INSTANCE;
462         }
463 
464         final Http2Headers out = new DefaultHttp2Headers(validateHeaders, inHeaders.size());
465         toHttp2Headers(inHeaders, out);
466         return out;
467     }

469     private static CharSequenceMap<AsciiString> toLowercaseMap(Iterator<? extends CharSequence> valuesIter,
470                                                                int arraySizeHint) {
471         UnsupportedValueConverter<AsciiString> valueConverter = UnsupportedValueConverter.instance();
472         CharSequenceMap<AsciiString> result = new CharSequenceMap<>(true, valueConverter, arraySizeHint);
473 
474         while (valuesIter.hasNext()) {
475             AsciiString lowerCased = AsciiString.of(valuesIter.next()).toLowerCase();
476             int index = lowerCased.forEachByte(FIND_COMMA);
477             if (index != -1) {
478                 int start = 0;
479                 do {
480                     result.add(lowerCased.subSequence(start, index, false).trim(), EMPTY_STRING);
481                     start = index + 1;
482                 } while (start < lowerCased.length() &&
483                          (index = lowerCased.forEachByte(start, lowerCased.length() - start, FIND_COMMA)) != -1);
484                 result.add(lowerCased.subSequence(start, lowerCased.length(), false).trim(), EMPTY_STRING);
485             } else {
486                 result.add(lowerCased.trim(), EMPTY_STRING);
487             }
488         }
489         return result;
490     }

492     /**
493      * Filter the {@link HttpHeaderNames#TE} header according to the
494      * <a href="https://tools.ietf.org/html/rfc7540#section-8.1.2.2">special rules in the HTTP/2 RFC</a>.
495      * @param entry An entry whose name is {@link HttpHeaderNames#TE}.
496      * @param out the resulting HTTP/2 headers.
497      */
498     private static void toHttp2HeadersFilterTE(Entry<CharSequence, CharSequence> entry,
499                                                Http2Headers out) {
500         if (indexOf(entry.getValue(), ',', 0) == -1) {
501             if (contentEqualsIgnoreCase(trim(entry.getValue()), TRAILERS)) {
502                 out.add(TE, TRAILERS);
503             }
504         } else {
505             List<CharSequence> teValues = unescapeCsvFields(entry.getValue());
506             for (CharSequence teValue : teValues) {
507                 if (contentEqualsIgnoreCase(trim(teValue), TRAILERS)) {
508                     out.add(TE, TRAILERS);
509                     break;
510                 }
511             }
512         }
513     }

515     public static void toHttp2Headers(HttpHeaders inHeaders, Http2Headers out) {
516         Iterator<Entry<CharSequence, CharSequence>> iter = inHeaders.iteratorCharSequence();
517         // Choose 8 as a default size because it is unlikely we will see more than 4 Connection headers values, but
518         // still allowing for "enough" space in the map to reduce the chance of hash code collision.
519         CharSequenceMap<AsciiString> connectionBlacklist =
520             toLowercaseMap(inHeaders.valueCharSequenceIterator(CONNECTION), 8);
521         while (iter.hasNext()) {
522             Entry<CharSequence, CharSequence> entry = iter.next();
523             final AsciiString aName = AsciiString.of(entry.getKey()).toLowerCase();
524             if (!HTTP_TO_HTTP2_HEADER_BLACKLIST.contains(aName) && !connectionBlacklist.contains(aName)) {
525                 // https://tools.ietf.org/html/rfc7540#section-8.1.2.2 makes a special exception for TE
526                 if (aName.contentEqualsIgnoreCase(TE)) {
527                     toHttp2HeadersFilterTE(entry, out);
528                 } else if (aName.contentEqualsIgnoreCase(COOKIE)) {
529                     AsciiString value = AsciiString.of(entry.getValue());
530                     // split up cookies to allow for better compression
531                     // https://tools.ietf.org/html/rfc7540#section-8.1.2.5
532                     int index = value.forEachByte(FIND_SEMI_COLON);
533                     if (index != -1) {
534                         int start = 0;
535                         do {
536                             out.add(COOKIE, value.subSequence(start, index, false));
537                             // skip 2 characters "; " (see https://tools.ietf.org/html/rfc6265#section-4.2.1)
538                             start = index + 2;
539                         } while (start < value.length() &&
540                                  (index = value.forEachByte(start, value.length() - start, FIND_SEMI_COLON)) != -1);
541                         if (start >= value.length()) {
542                             throw new IllegalArgumentException("cookie value is of unexpected format: " + value);
543                         }
544                         out.add(COOKIE, value.subSequence(start, value.length(), false));
545                     } else {
546                         out.add(COOKIE, value);
547                     }
548                 } else {
549                     out.add(aName, entry.getValue());
550                 }
551             }
552         }
553     }

555     /**
556      * Generate an HTTP/2 {code :path} from a URI in accordance with
557      * <a href="https://tools.ietf.org/html/rfc7230#section-5.3">rfc7230, 5.3</a>.
558      */
559     private static AsciiString toHttp2Path(URI uri) {
560         StringBuilder pathBuilder = new StringBuilder(length(uri.getRawPath()) +
561                 length(uri.getRawQuery()) + length(uri.getRawFragment()) + 2);
562         if (!isNullOrEmpty(uri.getRawPath())) {
563             pathBuilder.append(uri.getRawPath());
564         }
565         if (!isNullOrEmpty(uri.getRawQuery())) {
566             pathBuilder.append('?');
567             pathBuilder.append(uri.getRawQuery());
568         }
569         if (!isNullOrEmpty(uri.getRawFragment())) {
570             pathBuilder.append('#');
571             pathBuilder.append(uri.getRawFragment());
572         }
573         String path = pathBuilder.toString();
574         return path.isEmpty() ? EMPTY_REQUEST_PATH : new AsciiString(path);
575     }

577     // package-private for testing only
578     static void setHttp2Authority(String authority, Http2Headers out) {
579         // The authority MUST NOT include the deprecated "userinfo" subcomponent
580         if (authority != null) {
581             if (authority.isEmpty()) {
582                 out.authority(EMPTY_STRING);
583             } else {
584                 int start = authority.indexOf('@') + 1;
585                 int length = authority.length() - start;
586                 if (length == 0) {
587                     throw new IllegalArgumentException("authority: " + authority);
588                 }
589                 out.authority(new AsciiString(authority, start, length));
590             }
591         }
592     }

594     private static void setHttp2Scheme(HttpHeaders in, URI uri, Http2Headers out) {
595         String value = uri.getScheme();
596         if (value != null) {
597             out.scheme(new AsciiString(value));
598             return;
599         }
600 
601         // Consume the Scheme extension header if present
602         CharSequence cValue = in.get(ExtensionHeaderNames.SCHEME.text());
603         if (cValue != null) {
604             out.scheme(AsciiString.of(cValue));
605             return;
606         }
607 
608         if (uri.getPort() == HTTPS.port()) {
609             out.scheme(HTTPS.name());
610         } else if (uri.getPort() == HTTP.port()) {
611             out.scheme(HTTP.name());
612         } else {
613             throw new IllegalArgumentException(":scheme must be specified. " +
614                     "see https://tools.ietf.org/html/rfc7540#section-8.1.2.3");
615         }
616     }

618     /**
619      * Utility which translates HTTP/2 headers to HTTP/1 headers.
620      */
621     private static final class Http2ToHttpHeaderTranslator {
622         /**
623          * Translations from HTTP/2 header name to the HTTP/1.x equivalent.
624          */
625         private static final CharSequenceMap<AsciiString>
626             REQUEST_HEADER_TRANSLATIONS = new CharSequenceMap<>();
627         private static final CharSequenceMap<AsciiString>
628             RESPONSE_HEADER_TRANSLATIONS = new CharSequenceMap<>();
629         static {
630             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.AUTHORITY.value(),
631                             HttpHeaderNames.HOST);
632             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.SCHEME.value(),
633                             ExtensionHeaderNames.SCHEME.text());
634             REQUEST_HEADER_TRANSLATIONS.add(RESPONSE_HEADER_TRANSLATIONS);
635             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.PATH.value(),
636                             ExtensionHeaderNames.PATH.text());
637         }
638 
639         private final int streamId;
640         private final HttpHeaders output;
641         private final CharSequenceMap<AsciiString> translations;
642 
643         /**
644          * Create a new instance
645          *
646          * @param output The HTTP/1.x headers object to store the results of the translation
647          * @param request if {@code true}, translates headers using the request translation map. Otherwise uses the
648          *        response translation map.
649          */
650         Http2ToHttpHeaderTranslator(int streamId, HttpHeaders output, boolean request) {
651             this.streamId = streamId;
652             this.output = output;
653             translations = request ? REQUEST_HEADER_TRANSLATIONS : RESPONSE_HEADER_TRANSLATIONS;
654         }
655 
656         void translateHeaders(Iterable<Entry<CharSequence, CharSequence>> inputHeaders) throws Http2Exception {
657             // lazily created as needed
658             StringBuilder cookies = null;
659 
660             for (Entry<CharSequence, CharSequence> entry : inputHeaders) {
661                 final CharSequence name = entry.getKey();
662                 final CharSequence value = entry.getValue();
663                 AsciiString translatedName = translations.get(name);
664                 if (translatedName != null) {
665                     output.add(translatedName, AsciiString.of(value));
666                 } else if (!Http2Headers.PseudoHeaderName.isPseudoHeader(name)) {
667                     // https://tools.ietf.org/html/rfc7540#section-8.1.2.3
668                     // All headers that start with ':' are only valid in HTTP/2 context
669                     if (name.length() == 0 || name.charAt(0) == ':') {
670                         throw streamError(streamId, PROTOCOL_ERROR,
671                                 "Invalid HTTP/2 header '%s' encountered in translation to HTTP/1.x", name);
672                     }
673                     if (COOKIE.equals(name)) {
674                         // combine the cookie values into 1 header entry.
675                         // https://tools.ietf.org/html/rfc7540#section-8.1.2.5
676                         if (cookies == null) {
677                             cookies = StringUtil.threadLocalStringBuilder();
678                         } else if (cookies.length() > 0) {
679                             cookies.append("; ");
680                         }
681                         cookies.append(value);
682                     } else {
683                         output.add(name, value);
684                     }
685                 }
686             }
687             if (cookies != null) {
688                 output.add(COOKIE, cookies.toString());
689             }
690         }
691     }
