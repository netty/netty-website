41  /**
42   * {@link ChannelHandler} which decodes bytes in a stream-like fashion from one {@link Buffer} to an
43   * other Message type.
44   *
45   * For example here is an implementation which reads all readable bytes from
46   * the input {@link Buffer}, creates a new {@link Buffer} and forward it to the next {@link ChannelHandler}
47   * in the {@link ChannelPipeline}.
48   *
49   * <pre>
50   *     public class SquareDecoder extends {@link ByteToMessageDecoderForBuffer} {
51   *         {@code @Override}
52   *         public void decode({@link ChannelHandlerContext} ctx, {@link Buffer} in)
53   *                 throws {@link Exception} {
54   *             ctx.fireChannelRead(in.readBytes(in.readableBytes()));
55   *         }
56   *     }
57   * </pre>
58   *
59   * <h3>Frame detection</h3>
60   * <p>
61   * Generally frame detection should be handled earlier in the pipeline by adding a
62   * {@link DelimiterBasedFrameDecoder}, {@link FixedLengthFrameDecoder}, {@link LengthFieldBasedFrameDecoder},
63   * or {@link LineBasedFrameDecoder}.
64   * <p>
65   * If a custom frame decoder is required, then one needs to be careful when implementing
66   * one with {@link ByteToMessageDecoderForBuffer}. Ensure there are enough bytes in the buffer for a
67   * complete frame by checking {@link Buffer#readableBytes()}. If there are not enough bytes
68   * for a complete frame, return without modifying the reader index to allow more bytes to arrive.
69   * <p>
70   * To check for complete frames without modifying the reader index, use methods like {@link Buffer#getInt(int)}.
71   * One <strong>MUST</strong> use the reader index when using methods like {@link Buffer#getInt(int)}.
72   * For example calling <tt>in.getInt(0)</tt> is assuming the frame starts at the beginning of the buffer, which
73   * is not always the case. Use <tt>in.getInt(in.readerIndex())</tt> instead.
74   * <h3>Pitfalls</h3>
75   * <p>
76   * Be aware that sub-classes of {@link ByteToMessageDecoderForBuffer} <strong>MUST NOT</strong>
77   * annotated with {@link @Sharable}.
78   */
79  public abstract class ByteToMessageDecoderForBuffer extends ChannelHandlerAdapter {

82       * Cumulate {@link Buffer}s by merge them into one {@link Buffer}'s, using memory copies.

86      /**
87       * Cumulate {@link Buffer}s by add them to a {@link CompositeBuffer} and so do no memory copy whenever possible.
88       * Be aware that {@link CompositeBuffer} use a more complex indexing implementation so depending on your use-case
89       * and the decoder implementation this may be slower then just use the {@link #MERGE_CUMULATOR}.
90       */
91      public static final Cumulator COMPOSITE_CUMULATOR = new CompositeBufferCumulator();

166     @Override
167     public final void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
168         Buffer buf = cumulation;
169         if (buf != null) {
170             // Directly set this to null so we are sure we not access it in any other method here anymore.
171             cumulation = null;
172             numReads = 0;
173             int readable = buf.readableBytes();
174             if (readable > 0) {
175                 ctx.fireChannelRead(buf);
176                 ctx.fireChannelReadComplete();
177             } else {
178                 buf.close();
179             }
180         }
181         handlerRemoved0(context);
182     }
183 
184     /**
185      * Gets called after the {@link ByteToMessageDecoderForBuffer} was removed from the actual context and it doesn't
186      * handle events anymore.
187      */
188     protected void handlerRemoved0(ChannelHandlerContext ctx) throws Exception { }

190     @Override
191     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
192         if (msg instanceof Buffer) {
193             try {
194                 Buffer data = (Buffer) msg;
195                 first = cumulation == null;
196                 if (first) {
197                     if (data.readOnly()) {
198                         cumulation = CompositeBuffer.compose(ctx.bufferAllocator(), data.copy().send());
199                         data.close();
200                     } else {
201                         cumulation = data;
202                     }
203                 } else {
204                     cumulation = cumulator.cumulate(ctx.bufferAllocator(), cumulation, data);
205                 }
206                 assert context.ctx == ctx || ctx == context;
207 
208                 callDecode(context, cumulation);
209             } catch (DecoderException e) {
210                 throw e;
211             } catch (Exception e) {
212                 throw new DecoderException(e);
213             } finally {
214                 if (cumulation != null && cumulation.readableBytes() == 0) {
215                     numReads = 0;
216                     if (cumulation.isAccessible()) {
217                         cumulation.close();
218                     }
219                     cumulation = null;
220                 } else if (++numReads >= discardAfterReads) {
221                     // We did enough reads already try to discard some bytes so we not risk to see a OOME.
222                     // See https://github.com/netty/netty/issues/4275
223                     numReads = 0;
224                     discardSomeReadBytes();
225                 }
226 
227                 firedChannelRead |= context.fireChannelReadCallCount() > 0;
228                 context.reset();
229             }
230         } else {
231             ctx.fireChannelRead(msg);
232         }
233     }
234 
235     @Override
236     public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
237         numReads = 0;
238         discardSomeReadBytes();
239         if (!firedChannelRead && !ctx.channel().config().isAutoRead()) {
240             ctx.read();
241         }
242         firedChannelRead = false;
243         ctx.fireChannelReadComplete();
244     }
245 
246     protected final void discardSomeReadBytes() {
247         if (cumulation != null && !first) {
248             // discard some bytes if possible to make more room in the buffer.
249             cumulation.compact();
250         }
251     }

253     @Override
254     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
255         assert context.ctx == ctx || ctx == context;
256         channelInputClosed(context, true);
257     }

259     @Override
260     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
261         ctx.fireUserEventTriggered(evt);
262         if (evt instanceof ChannelInputShutdownEvent) {
263             // The decodeLast method is invoked when a channelInactive event is encountered.
264             // This method is responsible for ending requests in some situations and must be called
265             // when the input has been shutdown.
266             assert context.ctx == ctx || ctx == context;
267             channelInputClosed(context, false);
268         }
269     }

271     private void channelInputClosed(ByteToMessageDecoderContext ctx, boolean callChannelInactive) {
272         try {
273             channelInputClosed(ctx);
274         } catch (DecoderException e) {
275             throw e;
276         } catch (Exception e) {
277             throw new DecoderException(e);
278         } finally {
279             if (cumulation != null) {
280                 cumulation.close();
281                 cumulation = null;
282             }
283             if (ctx.fireChannelReadCallCount() > 0) {
284                 ctx.reset();
285                 // Something was read, call fireChannelReadComplete()
286                 ctx.fireChannelReadComplete();
287             }
288             if (callChannelInactive) {
289                 ctx.fireChannelInactive();
290             }
291         }
292     }

294     /**
295      * Called when the input of the channel was closed which may be because it changed to inactive or because of
296      * {@link ChannelInputShutdownEvent}.
297      */
298     void channelInputClosed(ByteToMessageDecoderContext ctx) throws Exception {
299         if (cumulation != null) {
300             callDecode(ctx, cumulation);
301             // If callDecode(...) removed the handle from the pipeline we should not call decodeLast(...) as this would
302             // be unexpected.
303             if (!ctx.isRemoved()) {
304                 // Use Unpooled.EMPTY_BUFFER if cumulation become null after calling callDecode(...).
305                 // See https://github.com/netty/netty/issues/10802.
306                 Buffer buffer = cumulation == null ? ctx.bufferAllocator().allocate(0) : cumulation;
307                 decodeLast(ctx, buffer);
308             }
309         } else {
310             decodeLast(ctx, ctx.bufferAllocator().allocate(0));
311         }
312     }

314     /**
315      * Called once data should be decoded from the given {@link Buffer}. This method will call
316      * {@link #decode(ChannelHandlerContext, Buffer)} as long as decoding should take place.
317      *
318      * @param ctx the {@link ChannelHandlerContext} which this {@link ByteToMessageDecoderForBuffer} belongs to
319      * @param in  the {@link Buffer} from which to read data
320      */
321     void callDecode(ByteToMessageDecoderContext ctx, Buffer in) {
322         try {
323             while (in.readableBytes() > 0 && !ctx.isRemoved()) {
324 
325                 int oldInputLength = in.readableBytes();
326                 int numReadCalled = ctx.fireChannelReadCallCount();
327                 decodeRemovalReentryProtection(ctx, in);
328 
329                 // Check if this handler was removed before continuing the loop.
330                 // If it was removed, it is not safe to continue to operate on the buffer.
331                 //
332                 // See https://github.com/netty/netty/issues/1664
333                 if (ctx.isRemoved()) {
334                     break;
335                 }
336 
337                 if (numReadCalled == ctx.fireChannelReadCallCount()) {
338                     if (oldInputLength == in.readableBytes()) {
339                         break;
340                     } else {
341                         continue;
342                     }
343                 }
344 
345                 if (oldInputLength == in.readableBytes()) {
346                     throw new DecoderException(
347                             StringUtil.simpleClassName(getClass()) +
348                                     ".decode() did not read anything but decoded a message.");
349                 }
350 
351                 if (isSingleDecode()) {
352                     break;
353                 }
354             }
355         } catch (DecoderException e) {
356             throw e;
357         } catch (Exception cause) {
358             throw new DecoderException(cause);
359         }
360     }

362     /**
363      * Decode the from one {@link Buffer} to another. This method will be called till either the input
364      * {@link Buffer} has nothing to read when return from this method or till nothing was read from the input
365      * {@link Buffer}.
366      *
367      * @param ctx the {@link ChannelHandlerContext} which this {@link ByteToMessageDecoderForBuffer} belongs to
368      * @param in  the {@link Buffer} from which to read data
369      * @throws Exception is thrown if an error occurs
370      */
371     protected abstract void decode(ChannelHandlerContext ctx, Buffer in) throws Exception;
372 
373     /**
374      * Decode the from one {@link Buffer} to an other. This method will be called till either the input
375      * {@link Buffer} has nothing to read when return from this method or till nothing was read from the input
376      * {@link Buffer}.
377      *
378      * @param ctx the {@link ChannelHandlerContext} which this {@link ByteToMessageDecoderForBuffer} belongs to
379      * @param in the {@link Buffer} from which to read data
380      * @throws Exception is thrown if an error occurs
381      */
382     final void decodeRemovalReentryProtection(ChannelHandlerContext ctx, Buffer in)
383             throws Exception {
384         decode(ctx, in);
385     }
386 
387     /**
388      * Is called one last time when the {@link ChannelHandlerContext} goes in-active. Which means the
389      * {@link #channelInactive(ChannelHandlerContext)} was triggered.
390      *
391      * By default this will just call {@link #decode(ChannelHandlerContext, Buffer)} but sub-classes may
392      * override this for some special cleanup operation.
393      */
394     protected void decodeLast(ChannelHandlerContext ctx, Buffer in) throws Exception {
395         if (in.readableBytes() > 0) {
396             // Only call decode() if there is something left in the buffer to decode.
397             // See https://github.com/netty/netty/issues/4386
398             decodeRemovalReentryProtection(ctx, in);
399         }
400     }

402     private static Buffer expandCumulationAndWrite(BufferAllocator alloc, Buffer oldCumulation, Buffer in) {
403         final int newSize = safeFindNextPositivePowerOfTwo(oldCumulation.readableBytes() + in.readableBytes());
404         Buffer newCumulation = oldCumulation.readOnly() ? alloc.allocate(newSize) :
405                 oldCumulation.ensureWritable(newSize);
406         try {
407             if (newCumulation != oldCumulation) {
408                 newCumulation.writeBytes(oldCumulation);
409             }
410             newCumulation.writeBytes(in);
411             return newCumulation;
412         } finally {
413             if (newCumulation != oldCumulation) {
414                 oldCumulation.close();
415             }
416         }
417     }

426          * call {@link Buffer#close()} if a {@link Buffer} is fully consumed.
427          */
428         Buffer cumulate(BufferAllocator alloc, Buffer cumulation, Buffer in);
429     }

628     private static final class CompositeBufferCumulator implements Cumulator {
629         @Override
630         public Buffer cumulate(BufferAllocator alloc, Buffer cumulation, Buffer in) {
631             if (cumulation.readableBytes() == 0) {
632                 cumulation.close();
633                 return in;
634             }
635             CompositeBuffer composite;
636             try (in) {
637                 if (CompositeBuffer.isComposite(cumulation)) {
638                     CompositeBuffer tmp = (CompositeBuffer) cumulation;
639                     // Since we are extending the composite buffer below, we have to make sure there is no space to
640                     // write in the existing cumulation.
641                     if (tmp.writerOffset() < tmp.capacity()) {
642                         composite = tmp.split();
643                         tmp.close();
644                     } else {
645                         composite = tmp;
646                     }
647                 } else {
648                     composite = CompositeBuffer.compose(alloc, cumulation.send());
649                 }
650                 composite.extendWith((in.readOnly() ? in.copy() : in).send());
651                 return composite;
652             }
653         }
654 
655         @Override
656         public String toString() {
657             return "CompositeBufferCumulator";
658         }
659     }

661     private static final class MergeCumulator implements Cumulator {
662         @Override
663         public Buffer cumulate(BufferAllocator alloc, Buffer cumulation, Buffer in) {
664             if (cumulation.readableBytes() == 0) {
665                 // If cumulation is empty and input buffer is contiguous, use it directly
666                 cumulation.close();
667                 return in;
668             }
669             // We must close input Buffer in all cases as otherwise it may produce a leak if writeBytes(...) throw
670             // for whatever close (for example because of OutOfMemoryError)
671             try (in) {
672                 final int required = in.readableBytes();
673                 if (required > cumulation.writableBytes() || cumulation.readOnly()) {
674                     return expandCumulationAndWrite(alloc, cumulation, in);
675                 }
676                 cumulation.writeBytes(in);
677                 return cumulation;
678             }
679         }
680 
681         @Override
682         public String toString() {
683             return "MergeCumulator";
684         }
685     }
