30  import static io.netty5.handler.ssl.SslUtils.toSSLHandshakeException;
31  import static io.netty5.handler.ssl.JdkApplicationProtocolNegotiator.ProtocolSelectionListener;
32  import static io.netty5.handler.ssl.JdkApplicationProtocolNegotiator.ProtocolSelector;

34  class JdkAlpnSslEngine extends JdkSslEngine {
35      private final ProtocolSelectionListener selectionListener;
36      private final AlpnSelector alpnSelector;
37  
38      final class AlpnSelector implements BiFunction<SSLEngine, List<String>, String> {
39          private final ProtocolSelector selector;
40          private boolean called;
41  
42          AlpnSelector(ProtocolSelector selector) {
43              this.selector = selector;
44          }
45  
46          @Override
47          public String apply(SSLEngine sslEngine, List<String> strings) {
48              assert !called;
49              called = true;
50  
51              try {
52                  String selected = selector.select(strings);
53                  return selected == null ? StringUtil.EMPTY_STRING : selected;
54              } catch (Exception cause) {
55                  // Returning null means we want to fail the handshake.
56                  //
57                  // See https://download.java.net/java/jdk9/docs/api/javax/net/ssl/
58                  // SSLEngine.html#setHandshakeApplicationProtocolSelector-java.util.function.BiFunction-
59                  return null;
60              }
61          }
62  
63          void checkUnsupported() {
64              if (called) {
65                  // ALPN message was received by peer and so apply(...) was called.
66                  // See:
67                  // https://hg.openjdk.java.net/jdk9/dev/jdk/file/65464a307408/src/
68                  // java.base/share/classes/sun/security/ssl/ServerHandshaker.java#l933
69                  return;
70              }
71              String protocol = getApplicationProtocol();
72              assert protocol != null;
73  
74              if (protocol.isEmpty()) {
75                  // ALPN is not supported
76                  selector.unsupported();
77              }
78          }
79      }
80  
81      JdkAlpnSslEngine(SSLEngine engine,
82                       @SuppressWarnings("deprecation") JdkApplicationProtocolNegotiator applicationNegotiator,
83                       boolean isServer, BiConsumer<SSLEngine, AlpnSelector> setHandshakeApplicationProtocolSelector,
84                       BiConsumer<SSLEngine, List<String>> setApplicationProtocols) {
85          super(engine);
86          if (isServer) {
87              selectionListener = null;
88              alpnSelector = new AlpnSelector(applicationNegotiator.protocolSelectorFactory().
89                      newSelector(this, new LinkedHashSet<>(applicationNegotiator.protocols())));
90              setHandshakeApplicationProtocolSelector.accept(engine, alpnSelector);
91          } else {
92              selectionListener = applicationNegotiator.protocolListenerFactory()
93                      .newListener(this, applicationNegotiator.protocols());
94              alpnSelector = null;
95              setApplicationProtocols.accept(engine, applicationNegotiator.protocols());
96          }
97      }
98  
99      JdkAlpnSslEngine(SSLEngine engine,
100                      @SuppressWarnings("deprecation") JdkApplicationProtocolNegotiator applicationNegotiator,
101                      boolean isServer) {
102        this(engine, applicationNegotiator, isServer,
103                new BiConsumer<SSLEngine, AlpnSelector>() {
104                    @Override
105                    public void accept(SSLEngine e, AlpnSelector s) {
106                        JdkAlpnSslUtils.setHandshakeApplicationProtocolSelector(e, s);
107                    }
108                },
109                new BiConsumer<SSLEngine, List<String>>() {
110                    @Override
111                    public void accept(SSLEngine e, List<String> p) {
112                        JdkAlpnSslUtils.setApplicationProtocols(e, p);
113                    }
114                });
115     }
116 
117     private SSLEngineResult verifyProtocolSelection(SSLEngineResult result) throws SSLException {
118         if (result.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.FINISHED) {
119             if (alpnSelector == null) {
120                 // This means we are using client-side and
121                 try {
122                     String protocol = getApplicationProtocol();
123                     assert protocol != null;
124                     if (protocol.isEmpty()) {
125                         // If empty the server did not announce ALPN:
126                         // See:
127                         // https://hg.openjdk.java.net/jdk9/dev/jdk/file/65464a307408/src/java.base/
128                         // share/classes/sun/security/ssl/ClientHandshaker.java#l741
129                         selectionListener.unsupported();
130                     } else {
131                         selectionListener.selected(protocol);
132                     }
133                 } catch (Throwable e) {
134                     throw toSSLHandshakeException(e);
135                 }
136             } else {
137                 assert selectionListener == null;
138                 alpnSelector.checkUnsupported();
139             }
140         }
141         return result;
142     }

144     @Override
145     public SSLEngineResult wrap(ByteBuffer src, ByteBuffer dst) throws SSLException {
146         return verifyProtocolSelection(super.wrap(src, dst));
147     }
148 
149     @Override
150     public SSLEngineResult wrap(ByteBuffer[] srcs, ByteBuffer dst) throws SSLException {
151         return verifyProtocolSelection(super.wrap(srcs, dst));
152     }
153 
154     @Override
155     public SSLEngineResult wrap(ByteBuffer[] srcs, int offset, int len, ByteBuffer dst) throws SSLException {
156         return verifyProtocolSelection(super.wrap(srcs, offset, len, dst));
157     }
158 
159     @Override
160     public SSLEngineResult unwrap(ByteBuffer src, ByteBuffer dst) throws SSLException {
161         return verifyProtocolSelection(super.unwrap(src, dst));
162     }
163 
164     @Override
165     public SSLEngineResult unwrap(ByteBuffer src, ByteBuffer[] dsts) throws SSLException {
166         return verifyProtocolSelection(super.unwrap(src, dsts));
167     }

169     @Override
170     public SSLEngineResult unwrap(ByteBuffer src, ByteBuffer[] dst, int offset, int len) throws SSLException {
171         return verifyProtocolSelection(super.unwrap(src, dst, offset, len));
172     }

174     @Override
175     void setNegotiatedApplicationProtocol(String applicationProtocol) {
176         // Do nothing as this is handled internally by the Java8u251+ implementation of SSLEngine.
177     }
178 
179     @Override
180     public String getNegotiatedApplicationProtocol() {
181         String protocol = getApplicationProtocol();
182         if (protocol != null) {
183             return protocol.isEmpty() ? null : protocol;
184         }
185         return null;
186     }

188     // These methods will override the methods defined by Java 8u251 and later. As we may compile with an earlier
189     // java8 version we don't use @Override annotations here.
190     public String getApplicationProtocol() {
191         return JdkAlpnSslUtils.getApplicationProtocol(getWrappedEngine());
192     }
193 
194     public String getHandshakeApplicationProtocol() {
195         return JdkAlpnSslUtils.getHandshakeApplicationProtocol(getWrappedEngine());
196     }
197 
198     public void setHandshakeApplicationProtocolSelector(BiFunction<SSLEngine, List<String>, String> selector) {
199         JdkAlpnSslUtils.setHandshakeApplicationProtocolSelector(getWrappedEngine(), selector);
200     }
201 
202     public BiFunction<SSLEngine, List<String>, String> getHandshakeApplicationProtocolSelector() {
203         return JdkAlpnSslUtils.getHandshakeApplicationProtocolSelector(getWrappedEngine());
204     }
