76  /**
77   * An implementation of {@link SslContext} which works with libraries that support the
78   * <a href="https://www.openssl.org/">OpenSsl</a> C library API.
79   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
80   *
81   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
82   * which depends upon the instance of this class is released. Otherwise if any method of
83   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
84   */
85  public abstract class ReferenceCountedOpenSslContext extends SslContext implements ReferenceCounted {

207     ReferenceCountedOpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
208                                    OpenSslApplicationProtocolNegotiator apn, int mode, Certificate[] keyCertChain,
209                                    ClientAuth clientAuth, String[] protocols, boolean startTls, boolean enableOcsp,
210                                    boolean leakDetection, Map.Entry<SslContextOption<?>, Object>... ctxOptions)
211             throws SSLException {
212         super(startTls);
213 
214         OpenSsl.ensureAvailability();
215 
216         if (enableOcsp && !OpenSsl.isOcspSupported()) {
217             throw new IllegalStateException("OCSP is not supported.");
218         }
219 
220         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
221             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
222         }
223 
224         boolean tlsFalseStart = false;
225         boolean useTasks = USE_TASKS;
226         OpenSslPrivateKeyMethod privateKeyMethod = null;
227         OpenSslAsyncPrivateKeyMethod asyncPrivateKeyMethod = null;
228         OpenSslCertificateCompressionConfig certCompressionConfig = null;
229 
230         if (ctxOptions != null) {
231             for (Map.Entry<SslContextOption<?>, Object> ctxOpt : ctxOptions) {
232                 SslContextOption<?> option = ctxOpt.getKey();
233 
234                 if (option == OpenSslContextOption.TLS_FALSE_START) {
235                     tlsFalseStart = (Boolean) ctxOpt.getValue();
236                 } else if (option == OpenSslContextOption.USE_TASKS) {
237                     useTasks = (Boolean) ctxOpt.getValue();
238                 } else if (option == OpenSslContextOption.PRIVATE_KEY_METHOD) {
239                     privateKeyMethod = (OpenSslPrivateKeyMethod) ctxOpt.getValue();
240                 } else if (option == OpenSslContextOption.ASYNC_PRIVATE_KEY_METHOD) {
241                     asyncPrivateKeyMethod = (OpenSslAsyncPrivateKeyMethod) ctxOpt.getValue();
242                 } else if (option == OpenSslContextOption.CERTIFICATE_COMPRESSION_ALGORITHMS) {
243                     certCompressionConfig = (OpenSslCertificateCompressionConfig) ctxOpt.getValue();
244                 } else {
245                     logger.debug("Skipping unsupported " + SslContextOption.class.getSimpleName()
246                             + ": " + ctxOpt.getKey());
247                 }
248             }
249         }
250         if (privateKeyMethod != null && asyncPrivateKeyMethod != null) {
251             throw new IllegalArgumentException("You can either only use "
252                     + OpenSslAsyncPrivateKeyMethod.class.getSimpleName() + " or "
253                     + OpenSslPrivateKeyMethod.class.getSimpleName());
254         }
255 
256         this.tlsFalseStart = tlsFalseStart;
257 
258         leak = leakDetection ? leakDetector.track(this) : null;
259         this.mode = mode;
260         this.clientAuth = isServer() ? requireNonNull(clientAuth, "clientAuth") : ClientAuth.NONE;
261         this.protocols = protocols;
262         this.enableOcsp = enableOcsp;
263 
264         this.keyCertChain = keyCertChain == null ? null : keyCertChain.clone();
265 
266         String[] suites = requireNonNull(cipherFilter, "cipherFilter").filterCipherSuites(
267                 ciphers, DEFAULT_CIPHERS, availableJavaCipherSuites());
268         // Filter out duplicates.
269         LinkedHashSet<String> suitesSet = new LinkedHashSet<>(suites.length);
270         Collections.addAll(suitesSet, suites);
271         unmodifiableCiphers = new ArrayList<>(suitesSet);
272 
273         this.apn = requireNonNull(apn, "apn");
274 
275         // Create a new SSL_CTX and configure it.
276         boolean success = false;
277         try {
278             boolean tlsv13Supported = OpenSsl.isTlsv13Supported();
279 
280             try {
281                 int protocolOpts = SSL.SSL_PROTOCOL_SSLV3 | SSL.SSL_PROTOCOL_TLSV1 |
282                         SSL.SSL_PROTOCOL_TLSV1_1 | SSL.SSL_PROTOCOL_TLSV1_2;
283                 if (tlsv13Supported) {
284                     protocolOpts |= SSL.SSL_PROTOCOL_TLSV1_3;
285                 }
286                 ctx = SSLContext.make(protocolOpts, mode);
287             } catch (Exception e) {
288                 throw new SSLException("failed to create an SSL_CTX", e);
289             }
290 
291             StringBuilder cipherBuilder = new StringBuilder();
292             StringBuilder cipherTLSv13Builder = new StringBuilder();
293 
294             /* List the ciphers that are permitted to negotiate. */
295             try {
296                 if (unmodifiableCiphers.isEmpty()) {
297                     // Set non TLSv1.3 ciphers.
298                     SSLContext.setCipherSuite(ctx, StringUtil.EMPTY_STRING, false);
299                     if (tlsv13Supported) {
300                         // Set TLSv1.3 ciphers.
301                         SSLContext.setCipherSuite(ctx, StringUtil.EMPTY_STRING, true);
302                     }
303                 } else {
304                     CipherSuiteConverter.convertToCipherStrings(
305                             unmodifiableCiphers, cipherBuilder, cipherTLSv13Builder, OpenSsl.isBoringSSL());
306 
307                     // Set non TLSv1.3 ciphers.
308                     SSLContext.setCipherSuite(ctx, cipherBuilder.toString(), false);
309                     if (tlsv13Supported) {
310                         // Set TLSv1.3 ciphers.
311                         SSLContext.setCipherSuite(ctx,
312                                 OpenSsl.checkTls13Ciphers(logger, cipherTLSv13Builder.toString()), true);
313                     }
314                 }
315             } catch (SSLException e) {
316                 throw e;
317             } catch (Exception e) {
318                 throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
319             }
320 
321             int options = SSLContext.getOptions(ctx) |
322                     SSL.SSL_OP_NO_SSLv2 |
323                     SSL.SSL_OP_NO_SSLv3 |
324                     // Disable TLSv1 and TLSv1.1 by default as these are not considered secure anymore
325                     // and the JDK is doing the same:
326                     // https://www.oracle.com/java/technologies/javase/8u291-relnotes.html
327                     SSL.SSL_OP_NO_TLSv1 |
328                     SSL.SSL_OP_NO_TLSv1_1 |
329 
330                     SSL.SSL_OP_CIPHER_SERVER_PREFERENCE |
331 
332                     // We do not support compression at the moment so we should explicitly disable it.
333                     SSL.SSL_OP_NO_COMPRESSION |
334 
335                     // Disable ticket support by default to be more inline with SSLEngineImpl of the JDK.
336                     // This also let SSLSession.getId() work the same way for the JDK implementation and the
337                     // OpenSSLEngine. If tickets are supported SSLSession.getId() will only return an ID on the
338                     // server-side if it could make use of tickets.
339                     SSL.SSL_OP_NO_TICKET;
340 
341             if (cipherBuilder.length() == 0) {
342                 // No ciphers that are compatible with SSLv2 / SSLv3 / TLSv1 / TLSv1.1 / TLSv1.2
343                 options |= SSL.SSL_OP_NO_SSLv2 | SSL.SSL_OP_NO_SSLv3 | SSL.SSL_OP_NO_TLSv1
344                         | SSL.SSL_OP_NO_TLSv1_1 | SSL.SSL_OP_NO_TLSv1_2;
345             }
346 
347             SSLContext.setOptions(ctx, options);
348 
349             // We need to enable SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER as the memory address may change between
350             // calling OpenSSLEngine.wrap(...).
351             // See https://github.com/netty/netty-tcnative/issues/100
352             SSLContext.setMode(ctx, SSLContext.getMode(ctx) | SSL.SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
353 
354             if (DH_KEY_LENGTH != null) {
355                 SSLContext.setTmpDHLength(ctx, DH_KEY_LENGTH);
356             }
357 
358             List<String> nextProtoList = apn.protocols();
359             /* Set next protocols for next protocol negotiation extension, if specified */
360             if (!nextProtoList.isEmpty()) {
361                 String[] appProtocols = nextProtoList.toArray(EmptyArrays.EMPTY_STRINGS);
362                 int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
363 
364                 switch (apn.protocol()) {
365                     case NPN:
366                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
367                         break;
368                     case ALPN:
369                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
370                         break;
371                     case NPN_AND_ALPN:
372                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
373                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
374                         break;
375                     default:
376                         throw new Error();
377                 }
378             }
379 
380             if (enableOcsp) {
381                 SSLContext.enableOcsp(ctx, isClient());
382             }
383 
384             SSLContext.setUseTasks(ctx, useTasks);
385             if (privateKeyMethod != null) {
386                 SSLContext.setPrivateKeyMethod(ctx, new PrivateKeyMethod(engineMap, privateKeyMethod));
387             }
388             if (asyncPrivateKeyMethod != null) {
389                 SSLContext.setPrivateKeyMethod(ctx, new AsyncPrivateKeyMethod(engineMap, asyncPrivateKeyMethod));
390             }
391             if (certCompressionConfig != null) {
392                 for (OpenSslCertificateCompressionConfig.AlgorithmConfig configPair : certCompressionConfig) {
393                     final CertificateCompressionAlgo algo = new CompressionAlgorithm(engineMap, configPair.algorithm());
394                     switch (configPair.mode()) {
395                         case Decompress:
396                             SSLContext.addCertificateCompressionAlgorithm(
397                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_DECOMPRESS, algo);
398                             break;
399                         case Compress:
400                             SSLContext.addCertificateCompressionAlgorithm(
401                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_COMPRESS, algo);
402                             break;
403                         case Both:
404                             SSLContext.addCertificateCompressionAlgorithm(
405                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_BOTH, algo);
406                             break;
407                         default:
408                             throw new IllegalStateException();
409                     }
410                 }
411             }
412             // Set the curves.
413             SSLContext.setCurvesList(ctx, OpenSsl.NAMED_GROUPS);
414             success = true;
415         } finally {
416             if (!success) {
417                 release();
418             }
419         }
420     }

422     private static int opensslSelectorFailureBehavior(ApplicationProtocolConfig.SelectorFailureBehavior behavior) {
423         switch (behavior) {
424             case NO_ADVERTISE:
425                 return SSL.SSL_SELECTOR_FAILURE_NO_ADVERTISE;
426             case CHOOSE_MY_LAST_PROTOCOL:
427                 return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
428             default:
429                 throw new Error();
430         }
431     }

616     //            producing a segfault.
617     private void destroy() {
618         Lock writerLock = ctxLock.writeLock();
619         writerLock.lock();
620         try {
621             if (ctx != 0) {
622                 if (enableOcsp) {
623                     SSLContext.disableOcsp(ctx);
624                 }
625 
626                 SSLContext.free(ctx);
627                 ctx = 0;
628 
629                 OpenSslSessionContext context = sessionContext();
630                 if (context != null) {
631                     context.destroy();
632                 }
633             }
634         } finally {
635             writerLock.unlock();
636         }
637     }

665     /**
666      * Translate a {@link ApplicationProtocolConfig} object to a
667      * {@link OpenSslApplicationProtocolNegotiator} object.
668      *
669      * @param config The configuration which defines the translation
670      * @return The results of the translation
671      */
672     @SuppressWarnings("deprecation")
673     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
674         if (config == null) {
675             return NONE_PROTOCOL_NEGOTIATOR;
676         }
677 
678         switch (config.protocol()) {
679             case NONE:
680                 return NONE_PROTOCOL_NEGOTIATOR;
681             case ALPN:
682             case NPN:
683             case NPN_AND_ALPN:
684                 switch (config.selectedListenerFailureBehavior()) {
685                     case CHOOSE_MY_LAST_PROTOCOL:
686                     case ACCEPT:
687                         switch (config.selectorFailureBehavior()) {
688                             case CHOOSE_MY_LAST_PROTOCOL:
689                             case NO_ADVERTISE:
690                                 return new OpenSslDefaultApplicationProtocolNegotiator(
691                                         config);
692                             default:
693                                 throw new UnsupportedOperationException(
694                                         new StringBuilder("OpenSSL provider does not support ")
695                                                 .append(config.selectorFailureBehavior())
696                                                 .append(" behavior").toString());
697                         }
698                     default:
699                         throw new UnsupportedOperationException(
700                                 new StringBuilder("OpenSSL provider does not support ")
701                                         .append(config.selectedListenerFailureBehavior())
702                                         .append(" behavior").toString());
703                 }
704             default:
705                 throw new Error();
706         }
707     }

752     abstract static class AbstractCertificateVerifier extends CertificateVerifier {
753         private final OpenSslEngineMap engineMap;
754 
755         AbstractCertificateVerifier(OpenSslEngineMap engineMap) {
756             this.engineMap = engineMap;
757         }
758 
759         @Override
760         public final int verify(long ssl, byte[][] chain, String auth) {
761             final ReferenceCountedOpenSslEngine engine = engineMap.get(ssl);
762             if (engine == null) {
763                 // May be null if it was destroyed in the meantime.
764                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
765             }
766             X509Certificate[] peerCerts = certificates(chain);
767             try {
768                 verify(engine, peerCerts, auth);
769                 return CertificateVerifier.X509_V_OK;
770             } catch (Throwable cause) {
771                 logger.debug("verification of certificate failed", cause);
772                 engine.initHandshakeException(cause);
773 
774                 // Try to extract the correct error code that should be used.
775                 if (cause instanceof OpenSslCertificateException) {
776                     // This will never return a negative error code as its validated when constructing the
777                     // OpenSslCertificateException.
778                     return ((OpenSslCertificateException) cause).errorCode();
779                 }
780                 if (cause instanceof CertificateExpiredException) {
781                     return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
782                 }
783                 if (cause instanceof CertificateNotYetValidException) {
784                     return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
785                 }
786                 if (cause instanceof CertificateRevokedException) {
787                     return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
788                 }
789 
790                 // The X509TrustManagerImpl uses a Validator which wraps a CertPathValidatorException into
791                 // an CertificateException. So we need to handle the wrapped CertPathValidatorException to be
792                 // able to send the correct alert.
793                 Throwable wrapped = cause.getCause();
794                 while (wrapped != null) {
795                     if (wrapped instanceof CertPathValidatorException) {
796                         CertPathValidatorException ex = (CertPathValidatorException) wrapped;
797                         CertPathValidatorException.Reason reason = ex.getReason();
798                         if (reason == CertPathValidatorException.BasicReason.EXPIRED) {
799                             return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
800                         }
801                         if (reason == CertPathValidatorException.BasicReason.NOT_YET_VALID) {
802                             return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
803                         }
804                         if (reason == CertPathValidatorException.BasicReason.REVOKED) {
805                             return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
806                         }
807                     }
808                     wrapped = wrapped.getCause();
809                 }
810 
811                 // Could not detect a specific error code to use, so fallback to a default code.
812                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
813             }
814         }
815 
816         abstract void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts,
817                              String auth) throws Exception;
818     }

839     static void setKeyMaterial(long ctx, X509Certificate[] keyCertChain, PrivateKey key, String keyPassword)
840             throws SSLException {
841          /* Load the certificate file and private key. */
842         long keyBio = 0;
843         long keyCertChainBio = 0;
844         long keyCertChainBio2 = 0;
845         PemEncoded encoded = null;
846         try {
847             // Only encode one time
848             encoded = PemX509Certificate.toPEM(ByteBufAllocator.DEFAULT, true, keyCertChain);
849             keyCertChainBio = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
850             keyCertChainBio2 = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
851 
852             if (key != null) {
853                 keyBio = toBIO(ByteBufAllocator.DEFAULT, key);
854             }
855 
856             SSLContext.setCertificateBio(
857                     ctx, keyCertChainBio, keyBio,
858                     keyPassword == null ? StringUtil.EMPTY_STRING : keyPassword);
859             // We may have more then one cert in the chain so add all of them now.
860             SSLContext.setCertificateChainBio(ctx, keyCertChainBio2, true);
861         } catch (SSLException e) {
862             throw e;
863         } catch (Exception e) {
864             throw new SSLException("failed to set certificate and key", e);
865         } finally {
866             freeBio(keyBio);
867             freeBio(keyCertChainBio);
868             freeBio(keyCertChainBio2);
869             if (encoded != null) {
870                 encoded.release();
871             }
872         }
873     }

875     static void freeBio(long bio) {
876         if (bio != 0) {
877             SSL.freeBIO(bio);
878         }

881     /**
882      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
883      * or {@code 0} if the {@code key} is {@code null}. The BIO contains the content of the {@code key}.
884      */
885     static long toBIO(ByteBufAllocator allocator, PrivateKey key) throws Exception {
886         if (key == null) {
887             return 0;
888         }
889 
890         PemEncoded pem = PemPrivateKey.toPEM(allocator, true, key);
891         try {
892             return toBIO(allocator, pem.retain());
893         } finally {
894             pem.release();
895         }
896     }

898     /**
899      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
900      * or {@code 0} if the {@code certChain} is {@code null}. The BIO contains the content of the {@code certChain}.
901      */
902     static long toBIO(ByteBufAllocator allocator, X509Certificate... certChain) throws Exception {
903         if (certChain == null) {
904             return 0;
905         }
906 
907         checkNonEmpty(certChain, "certChain");
908 
909         PemEncoded pem = PemX509Certificate.toPEM(allocator, true, certChain);
910         try {
911             return toBIO(allocator, pem.retain());
912         } finally {
913             pem.release();
914         }
915     }

917     static long toBIO(ByteBufAllocator allocator, PemEncoded pem) throws Exception {
918         try {
919             // We can turn direct buffers straight into BIOs. No need to
920             // make a yet another copy.
921             ByteBuf content = pem.content();
922 
923             if (content.isDirect()) {
924                 return newBIO(content.retainedSlice());
925             }
926 
927             ByteBuf buffer = allocator.directBuffer(content.readableBytes());
928             try {
929                 buffer.writeBytes(content, content.readerIndex(), content.readableBytes());
930                 return newBIO(buffer.retainedSlice());
931             } finally {
932                 try {
933                     // If the contents of the ByteBuf is sensitive (e.g. a PrivateKey) we
934                     // need to zero out the bytes of the copy before we're releasing it.
935                     if (pem.isSensitive()) {
936                         SslUtils.zeroout(buffer);
937                     }
938                 } finally {
939                     buffer.release();
940                 }
941             }
942         } finally {
943             pem.release();
944         }
945     }

947     private static long newBIO(ByteBuf buffer) throws Exception {
948         try {
949             long bio = SSL.newMemBIO();
950             int readable = buffer.readableBytes();
951             if (SSL.bioWrite(bio, OpenSsl.memoryAddress(buffer) + buffer.readerIndex(), readable) != readable) {
952                 SSL.freeBIO(bio);
953                 throw new IllegalStateException("Could not write data to memory BIO");
954             }
955             return bio;
956         } finally {
957             buffer.release();
958         }
959     }

961     /**
962      * Returns the {@link OpenSslKeyMaterialProvider} that should be used for OpenSSL. Depending on the given
963      * {@link KeyManagerFactory} this may cache the {@link OpenSslKeyMaterial} for better performance if it can
964      * ensure that the same material is always returned for the same alias.
965      */
966     static OpenSslKeyMaterialProvider providerFor(KeyManagerFactory factory, String password) {
967         if (factory instanceof OpenSslX509KeyManagerFactory) {
968             return ((OpenSslX509KeyManagerFactory) factory).newProvider();
969         }
970 
971         if (factory instanceof OpenSslCachingX509KeyManagerFactory) {
972             // The user explicit used OpenSslCachingX509KeyManagerFactory which signals us that its fine to cache.
973             return ((OpenSslCachingX509KeyManagerFactory) factory).newProvider(password);
974         }
975         // We can not be sure if the material may change at runtime so we will not cache it.
976         return new OpenSslKeyMaterialProvider(chooseX509KeyManager(factory.getKeyManagers()), password);
977     }

1053         private static final class ResultCallbackListener implements FutureListener<byte[]> {
1054             private final ReferenceCountedOpenSslEngine engine;
1055             private final long ssl;
1056             private final ResultCallback<byte[]> resultCallback;
1057 
1058             ResultCallbackListener(ReferenceCountedOpenSslEngine engine, long ssl,
1059                                    ResultCallback<byte[]> resultCallback) {
1060                 this.engine = engine;
1061                 this.ssl = ssl;
1062                 this.resultCallback = resultCallback;
1063             }
1064 
1065             @Override
1066             public void operationComplete(Future<? extends byte[]> future) {
1067                 Throwable cause = future.cause();
1068                 if (cause == null) {
1069                     try {
1070                         byte[] result = verifyResult(future.getNow());
1071                         resultCallback.onSuccess(ssl, result);
1072                         return;
1073                     } catch (SignatureException e) {
1074                         cause = e;
1075                         engine.initHandshakeException(e);
1076                     }
1077                 }
1078                 resultCallback.onError(ssl, cause);
1079             }
1080         }
