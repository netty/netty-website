40  /**
41   * A client-side {@link SslContext} which uses OpenSSL's SSL/TLS implementation.
42   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
43   *
44   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
45   * which depends upon the instance of this class is released. Otherwise if any method of
46   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
47   */
48  public final class ReferenceCountedOpenSslClientContext extends ReferenceCountedOpenSslContext {
49  
50      private static final Set<String> SUPPORTED_KEY_TYPES = Collections.unmodifiableSet(new LinkedHashSet<>(
51              Arrays.asList(OpenSslKeyMaterialManager.KEY_TYPE_RSA,
52                            OpenSslKeyMaterialManager.KEY_TYPE_DH_RSA,
53                            OpenSslKeyMaterialManager.KEY_TYPE_EC,
54                            OpenSslKeyMaterialManager.KEY_TYPE_EC_RSA,
55                            OpenSslKeyMaterialManager.KEY_TYPE_EC_EC)));
56  
57      private final OpenSslSessionContext sessionContext;
58  
59      ReferenceCountedOpenSslClientContext(X509Certificate[] trustCertCollection, TrustManagerFactory trustManagerFactory,
60                                           X509Certificate[] keyCertChain, PrivateKey key, String keyPassword,
61                                           KeyManagerFactory keyManagerFactory, Iterable<String> ciphers,
62                                           CipherSuiteFilter cipherFilter, ApplicationProtocolConfig apn,
63                                           String[] protocols, long sessionCacheSize, long sessionTimeout,
64                                           boolean enableOcsp, String keyStore,
65                                           Map.Entry<SslContextOption<?>, Object>... options) throws SSLException {
66          super(ciphers, cipherFilter, toNegotiator(apn), SSL.SSL_MODE_CLIENT, keyCertChain,
67                ClientAuth.NONE, protocols, false, enableOcsp, true, options);
68          boolean success = false;
69          try {
70              sessionContext = newSessionContext(this, ctx, engineMap, trustCertCollection, trustManagerFactory,
71                                                 keyCertChain, key, keyPassword, keyManagerFactory, keyStore,
72                                                 sessionCacheSize, sessionTimeout);
73              success = true;
74          } finally {
75              if (!success) {
76                  release();
77              }
78          }
79      }
80  
81      @Override
82      public OpenSslSessionContext sessionContext() {
83          return sessionContext;
84      }

86      static OpenSslSessionContext newSessionContext(ReferenceCountedOpenSslContext thiz, long ctx,
87                                                     OpenSslEngineMap engineMap,
88                                                     X509Certificate[] trustCertCollection,
89                                                     TrustManagerFactory trustManagerFactory,
90                                                     X509Certificate[] keyCertChain, PrivateKey key,
91                                                     String keyPassword, KeyManagerFactory keyManagerFactory,
92                                                     String keyStore, long sessionCacheSize, long sessionTimeout)
93              throws SSLException {
94          if (key == null && keyCertChain != null || key != null && keyCertChain == null) {
95              throw new IllegalArgumentException(
96                      "Either both keyCertChain and key needs to be null or none of them");
97          }
98          OpenSslKeyMaterialProvider keyMaterialProvider = null;
99          try {
100             try {
101                 if (!OpenSsl.supportsKeyManagerFactory()) {
102                     if (keyManagerFactory != null) {
103                         throw new IllegalArgumentException(
104                                 "KeyManagerFactory not supported");
105                     }
106                     if (keyCertChain != null/* && key != null*/) {
107                         setKeyMaterial(ctx, keyCertChain, key, keyPassword);
108                     }
109                 } else {
110                     // javadocs state that keyManagerFactory has precedent over keyCertChain
111                     if (keyManagerFactory == null && keyCertChain != null) {
112                         char[] keyPasswordChars = keyStorePassword(keyPassword);
113                         KeyStore ks = buildKeyStore(keyCertChain, key, keyPasswordChars, keyStore);
114                         if (ks.aliases().hasMoreElements()) {
115                             keyManagerFactory = new OpenSslX509KeyManagerFactory();
116                         } else {
117                             keyManagerFactory = new OpenSslCachingX509KeyManagerFactory(
118                                     KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm()));
119                         }
120                         keyManagerFactory.init(ks, keyPasswordChars);
121                         keyMaterialProvider = providerFor(keyManagerFactory, keyPassword);
122                     } else if (keyManagerFactory != null) {
123                         keyMaterialProvider = providerFor(keyManagerFactory, keyPassword);
124                     }
125 
126                     if (keyMaterialProvider != null) {
127                         OpenSslKeyMaterialManager materialManager = new OpenSslKeyMaterialManager(keyMaterialProvider);
128                         SSLContext.setCertificateCallback(ctx, new OpenSslClientCertificateCallback(
129                                 engineMap, materialManager));
130                     }
131                 }
132             } catch (Exception e) {
133                 throw new SSLException("failed to set certificate and key", e);
134             }
135 
136             // On the client side we always need to use SSL_CVERIFY_OPTIONAL (which will translate to SSL_VERIFY_PEER)
137             // to ensure that when the TrustManager throws we will produce the correct alert back to the server.
138             //
139             // See:
140             //   - https://www.openssl.org/docs/man1.0.2/man3/SSL_CTX_set_verify.html
141             //   - https://github.com/netty/netty/issues/8942
142             SSLContext.setVerify(ctx, SSL.SSL_CVERIFY_OPTIONAL, VERIFY_DEPTH);
143 
144             try {
145                 if (trustCertCollection != null) {
146                     trustManagerFactory = buildTrustManagerFactory(trustCertCollection, trustManagerFactory, keyStore);
147                 } else if (trustManagerFactory == null) {
148                     trustManagerFactory = TrustManagerFactory.getInstance(
149                             TrustManagerFactory.getDefaultAlgorithm());
150                     trustManagerFactory.init((KeyStore) null);
151                 }
152                 final X509TrustManager manager = chooseTrustManager(trustManagerFactory.getTrustManagers());
153 
154                 // IMPORTANT: The callbacks set for verification must be static to prevent memory leak as
155                 //            otherwise the context can never be collected. This is because the JNI code holds
156                 //            a global reference to the callbacks.
157                 //
158                 //            See https://github.com/netty/netty/issues/5372
159 
160                 // Use this to prevent an error when running on java < 7
161                 if (useExtendedTrustManager(manager)) {
162                     SSLContext.setCertVerifyCallback(ctx,
163                             new ExtendedTrustManagerVerifyCallback(engineMap, (X509ExtendedTrustManager) manager));
164                 } else {
165                     SSLContext.setCertVerifyCallback(ctx, new TrustManagerVerifyCallback(engineMap, manager));
166                 }
167             } catch (Exception e) {
168                 if (keyMaterialProvider != null) {
169                     keyMaterialProvider.destroy();
170                 }
171                 throw new SSLException("unable to setup trustmanager", e);
172             }
173             OpenSslClientSessionContext context = new OpenSslClientSessionContext(thiz, keyMaterialProvider);
174             context.setSessionCacheEnabled(CLIENT_ENABLE_SESSION_CACHE);
175             if (sessionCacheSize > 0) {
176                 context.setSessionCacheSize((int) Math.min(sessionCacheSize, Integer.MAX_VALUE));
177             }
178             if (sessionTimeout > 0) {
179                 context.setSessionTimeout((int) Math.min(sessionTimeout, Integer.MAX_VALUE));
180             }
181 
182             if (CLIENT_ENABLE_SESSION_TICKET) {
183                 context.setTicketKeys();
184             }
185 
186             keyMaterialProvider = null;
187             return context;
188         } finally {
189             if (keyMaterialProvider != null) {
190                 keyMaterialProvider.destroy();
191             }
192         }
193     }

201     private static final class TrustManagerVerifyCallback extends AbstractCertificateVerifier {
202         private final X509TrustManager manager;
203 
204         TrustManagerVerifyCallback(OpenSslEngineMap engineMap, X509TrustManager manager) {
205             super(engineMap);
206             this.manager = manager;
207         }
208 
209         @Override
210         void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts, String auth)
211                 throws Exception {
212             manager.checkServerTrusted(peerCerts, auth);
213         }
214     }

231     private static final class OpenSslClientCertificateCallback implements CertificateCallback {
232         private final OpenSslEngineMap engineMap;
233         private final OpenSslKeyMaterialManager keyManagerHolder;
234 
235         OpenSslClientCertificateCallback(OpenSslEngineMap engineMap, OpenSslKeyMaterialManager keyManagerHolder) {
236             this.engineMap = engineMap;
237             this.keyManagerHolder = keyManagerHolder;
238         }
239 
240         @Override
241         public void handle(long ssl, byte[] keyTypeBytes, byte[][] asn1DerEncodedPrincipals) throws Exception {
242             final ReferenceCountedOpenSslEngine engine = engineMap.get(ssl);
243             // May be null if it was destroyed in the meantime.
244             if (engine == null) {
245                 return;
246             }
247             try {
248                 final Set<String> keyTypesSet = supportedClientKeyTypes(keyTypeBytes);
249                 final String[] keyTypes = keyTypesSet.toArray(new String[0]);
250                 final X500Principal[] issuers;
251                 if (asn1DerEncodedPrincipals == null) {
252                     issuers = null;
253                 } else {
254                     issuers = new X500Principal[asn1DerEncodedPrincipals.length];
255                     for (int i = 0; i < asn1DerEncodedPrincipals.length; i++) {
256                         issuers[i] = new X500Principal(asn1DerEncodedPrincipals[i]);
257                     }
258                 }
259                 keyManagerHolder.setKeyMaterialClientSide(engine, keyTypes, issuers);
260             } catch (Throwable cause) {
261                 engine.initHandshakeException(cause);
262                 if (cause instanceof Exception) {
263                     throw (Exception) cause;
264                 }
265                 throw new SSLException(cause);
266             }
267         }
268 
269         /**
270          * Gets the supported key types for client certificates.
271          *
272          * @param clientCertificateTypes {@code ClientCertificateType} values provided by the server.
273          *        See https://www.ietf.org/assignments/tls-parameters/tls-parameters.xml.
274          * @return supported key types that can be used in {@code X509KeyManager.chooseClientAlias} and
275          *         {@code X509ExtendedKeyManager.chooseEngineClientAlias}.
276          */
277         private static Set<String> supportedClientKeyTypes(byte[] clientCertificateTypes) {
278             if (clientCertificateTypes == null) {
279                 // Try all of the supported key types.
280                 return SUPPORTED_KEY_TYPES;
281             }
282             Set<String> result = new HashSet<>(clientCertificateTypes.length);
283             for (byte keyTypeCode : clientCertificateTypes) {
284                 String keyType = clientKeyType(keyTypeCode);
285                 if (keyType == null) {
286                     // Unsupported client key type -- ignore
287                     continue;
288                 }
289                 result.add(keyType);
290             }
291             return result;
292         }
293 
294         private static String clientKeyType(byte clientCertificateType) {
295             // See also https://www.ietf.org/assignments/tls-parameters/tls-parameters.xml
296             switch (clientCertificateType) {
297                 case CertificateCallback.TLS_CT_RSA_SIGN:
298                     return OpenSslKeyMaterialManager.KEY_TYPE_RSA; // RFC rsa_sign
299                 case CertificateCallback.TLS_CT_RSA_FIXED_DH:
300                     return OpenSslKeyMaterialManager.KEY_TYPE_DH_RSA; // RFC rsa_fixed_dh
301                 case CertificateCallback.TLS_CT_ECDSA_SIGN:
302                     return OpenSslKeyMaterialManager.KEY_TYPE_EC; // RFC ecdsa_sign
303                 case CertificateCallback.TLS_CT_RSA_FIXED_ECDH:
304                     return OpenSslKeyMaterialManager.KEY_TYPE_EC_RSA; // RFC rsa_fixed_ecdh
305                 case CertificateCallback.TLS_CT_ECDSA_FIXED_ECDH:
306                     return OpenSslKeyMaterialManager.KEY_TYPE_EC_EC; // RFC ecdsa_fixed_ecdh
307                 default:
308                     return null;
309             }
310         }
