73      protected PoolArena(PooledBufferAllocator parent, MemoryManager manager, AllocationType allocationType,
74                          int pageSize, int pageShifts, int chunkSize, int cacheAlignment) {
75          super(pageSize, pageShifts, chunkSize, cacheAlignment);
76          this.parent = parent;
77          this.manager = manager;
78          this.allocationType = allocationType;
79          directMemoryCacheAlignment = cacheAlignment;
80  
81          numSmallSubpagePools = nSubpages;
82          smallSubpagePools = newSubpagePoolArray(numSmallSubpagePools);
83  
84          q100 = new PoolChunkList(this, null, 100, Integer.MAX_VALUE, chunkSize);
85          q075 = new PoolChunkList(this, q100, 75, 100, chunkSize);
86          q050 = new PoolChunkList(this, q075, 50, 100, chunkSize);
87          q025 = new PoolChunkList(this, q050, 25, 75, chunkSize);
88          q000 = new PoolChunkList(this, q025, 1, 50, chunkSize);
89          qInit = new PoolChunkList(this, q000, Integer.MIN_VALUE, 25, chunkSize);
90  
91          q100.prevList(q075);
92          q075.prevList(q050);
93          q050.prevList(q025);
94          q025.prevList(q000);
95          q000.prevList(null);
96          qInit.prevList(qInit);
97  
98          chunkListMetrics = List.of(qInit, q000, q025, q050, q075, q100);
99      }

101     private static PoolSubpage newSubpagePoolHead() {
102         PoolSubpage head = new PoolSubpage();
103         head.prev = head;
104         head.next = head;
105         return head;
106     }

108     private static PoolSubpage[] newSubpagePoolArray(int size) {
109         return new PoolSubpage[size];

112     UntetheredMemory allocate(PoolThreadCache cache, int size) {
113         final int sizeIdx = size2SizeIdx(size);
114 
115         if (sizeIdx <= smallMaxSizeIdx) {
116             return tcacheAllocateSmall(cache, size, sizeIdx);
117         } else if (sizeIdx < nSizes) {
118             return tcacheAllocateNormal(cache, size, sizeIdx);
119         } else {
120             int normCapacity = directMemoryCacheAlignment > 0
121                     ? normalizeSize(size) : size;
122             // Huge allocations are never served via the cache so just call allocateHuge
123             return allocateHuge(normCapacity);
124         }
125     }
126 
127     private UntetheredMemory tcacheAllocateSmall(PoolThreadCache cache, final int size, final int sizeIdx) {
128         UntetheredMemory memory = cache.allocateSmall(size, sizeIdx);
129         if (memory != null) {
130             // was able to allocate out of the cache so move on
131             return memory;
132         }
133 
134         /*
135          * Synchronize on the head. This is needed as {@link PoolChunk#allocateSubpage(int)} and
136          * {@link PoolChunk#free(long)} may modify the doubly linked list as well.
137          */
138         PoolSubpage head = findSubpagePoolHead(sizeIdx);
139         final boolean needsNormalAllocation;
140         synchronized (head) {
141             final PoolSubpage s = head.next;
142             needsNormalAllocation = s == head;
143             if (!needsNormalAllocation) {
144                 assert s.doNotDestroy && s.elemSize == sizeIdx2size(sizeIdx) :
145                         "doNotDestroy=" + s.doNotDestroy + ", elemSize=" + s.elemSize + ", sizeIdx=" + sizeIdx;
146                 long handle = s.allocate();
147                 assert handle >= 0;
148                 memory = s.chunk.allocateBufferWithSubpage(handle, size, cache);
149             }
150         }
151 
152         if (needsNormalAllocation) {
153             synchronized (this) {
154                 memory = allocateNormal(size, sizeIdx, cache);
155             }
156         }
157 
158         incSmallAllocation();
159         return memory;
160     }
161 
162     private UntetheredMemory tcacheAllocateNormal(
163             PoolThreadCache cache, int size, int sizeIdx) {
164         UntetheredMemory memory = cache.allocateNormal(this, size, sizeIdx);
165         if (memory != null) {
166             // was able to allocate out of the cache so move on
167             return memory;
168         }
169         synchronized (this) {
170             memory = allocateNormal(size, sizeIdx, cache);
171             allocationsNormal++;
172         }
173         return memory;
174     }
175 
176     // Method must be called inside synchronized(this) { ... } block
177     private UntetheredMemory allocateNormal(int size, int sizeIdx, PoolThreadCache threadCache) {
178         UntetheredMemory memory = q050.allocate(size, sizeIdx, threadCache);
179         if (memory != null) {
180             return memory;
181         }
182         memory = q025.allocate(size, sizeIdx, threadCache);
183         if (memory != null) {
184             return memory;
185         }
186         memory = q000.allocate(size, sizeIdx, threadCache);
187         if (memory != null) {
188             return memory;
189         }
190         memory = qInit.allocate(size, sizeIdx, threadCache);
191         if (memory != null) {
192             return memory;
193         }
194         memory = q075.allocate(size, sizeIdx, threadCache);
195         if (memory != null) {
196             return memory;
197         }
198 
199         // Add a new chunk.
200         PoolChunk c = newChunk(pageSize, nPSizes, pageShifts, chunkSize);
201         memory = c.allocate(size, sizeIdx, threadCache);
202         assert memory != null;
203         qInit.add(c);
204         return memory;
205     }

207     private void incSmallAllocation() {
208         allocationsSmall.increment();
209     }

211     private UntetheredMemory allocateHuge(int size) {
212         activeBytesHuge.add(size);
213         allocationsHuge.increment();
214         return new UnpooledUntetheredMemory(parent, manager, allocationType, size);
215     }

217     void free(PoolChunk chunk, long handle, int normCapacity, PoolThreadCache cache) {
218         SizeClass sizeClass = sizeClass(handle);
219         if (cache != null && cache.add(this, chunk, handle, normCapacity, sizeClass)) {
220             // cached so not free it.
221             return;
222         }
223         freeChunk(chunk, handle, normCapacity, sizeClass);
224     }
225 
226     private static SizeClass sizeClass(long handle) {
227         return isSubpage(handle) ? SizeClass.Small : SizeClass.Normal;
228     }
229 
230     void freeChunk(PoolChunk chunk, long handle, int normCapacity, SizeClass sizeClass) {
231         final boolean destroyChunk;
232         synchronized (this) {
233             if (sizeClass == SizeClass.Normal) {
234                 ++deallocationsNormal;
235             } else if (sizeClass == SizeClass.Small) {
236                 ++deallocationsSmall;
237             } else {
238                 throw new AssertionError("Unexpected size class: " + sizeClass);
239             }
240             destroyChunk = !chunk.parent.free(chunk, handle, normCapacity);
241         }
242         if (destroyChunk) {
243             // destroyChunk not need to be called while holding the synchronized lock.
244             chunk.destroy();
245         }
246     }

248     PoolSubpage findSubpagePoolHead(int sizeIdx) {
249         PoolSubpage head = (PoolSubpage) SUBPAGE_ARRAY.getVolatile(smallSubpagePools, sizeIdx);
250         if (head == null) {
251             head = newSubpagePoolHead();
252             if (!SUBPAGE_ARRAY.compareAndSet(smallSubpagePools, sizeIdx, null, head)) {
253                 // We lost the race. Read the winning value.
254                 head = (PoolSubpage) SUBPAGE_ARRAY.getVolatile(smallSubpagePools, sizeIdx);
255             }
256         }
257         return head;
258     }

285     private static List<PoolSubpageMetric> subPageMetricList(PoolSubpage[] pages) {
286         List<PoolSubpageMetric> metrics = new ArrayList<>();
287         for (int i = 0, len = pages.length; i < len; i++) {
288             PoolSubpage head = (PoolSubpage) SUBPAGE_ARRAY.getVolatile(pages, i);
289             if (head == null || head.next == head) {
290                 continue;
291             }
292             PoolSubpage s = head.next;
293             do {
294                 metrics.add(s);
295                 s = s.next;
296             } while (s != head);
297         }
298         return metrics;
299     }

352         long val = allocationsSmall.longValue() + allocationsHuge.longValue()
353                 - deallocationsHuge.longValue();
354         synchronized (this) {

356         }

379     @Override
380     public long numActiveBytes() {
381         long val = activeBytesHuge.longValue();
382         synchronized (this) {
383             for (int i = 0; i < chunkListMetrics.size(); i++) {
384                 for (PoolChunkMetric m: chunkListMetrics.get(i)) {
385                     val += m.chunkSize();
386                 }
387             }
388         }
389         return max(0, val);
390     }

443     private static void appendPoolSubPages(StringBuilder buf, PoolSubpage[] subpages) {
444         for (int i = 0; i < subpages.length; i ++) {
445             PoolSubpage head = (PoolSubpage) SUBPAGE_ARRAY.getVolatile(subpages, i);
446             if (head == null || head.next == head) {
447                 continue;
448             }
449 
450             buf.append(StringUtil.NEWLINE)
451                     .append(i)
452                     .append(": ");
453             PoolSubpage s = head.next;
454             do {
455                 buf.append(s);
456                 s = s.next;
457             } while (s != head);
458         }
459     }

461     public void close() {
462         for (int i = 0, len = smallSubpagePools.length; i < len; i++) {
463             PoolSubpage page = (PoolSubpage) SUBPAGE_ARRAY.getVolatile(smallSubpagePools, i);
464             if (page != null) {
465                 page.destroy();
466                 SUBPAGE_ARRAY.setVolatile(smallSubpagePools, i, null);
467             }
468         }
469         for (PoolChunkList list : new PoolChunkList[] {qInit, q000, q025, q050, q100}) {
470             list.destroy();
471         }
472     }
