59      private static final InternalLogger logger = InternalLoggerFactory.getInstance(SocketSslSessionReuseTest.class);
60  
61      private static final File CERT_FILE;
62      private static final File KEY_FILE;
63  
64      static {
65          SelfSignedCertificate ssc;
66          try {
67              ssc = new SelfSignedCertificate();
68          } catch (CertificateException e) {
69              throw new Error(e);
70          }
71          CERT_FILE = ssc.certificate();
72          KEY_FILE = ssc.privateKey();
73      }
74  
75      public static Collection<Object[]> data() throws Exception {
76          return Collections.singletonList(new Object[] {
77            SslContextBuilder.forServer(CERT_FILE, KEY_FILE).sslProvider(SslProvider.JDK).build(),
78            SslContextBuilder.forClient().trustManager(CERT_FILE).sslProvider(SslProvider.JDK).build()
79          });
80      }

82      @ParameterizedTest(name = "{index}: serverEngine = {0}, clientEngine = {1}")
83      @MethodSource("data")
84      @Timeout(value = 30000, unit = TimeUnit.MILLISECONDS)
85      public void testSslSessionReuse(SslContext serverCtx, SslContext clientCtx, TestInfo testInfo) throws Throwable {
86          run(testInfo, (sb, cb) -> testSslSessionReuse(sb, cb, serverCtx, clientCtx));
87      }
88  
89      public void testSslSessionReuse(ServerBootstrap sb, Bootstrap cb,
90                                      SslContext serverCtx, SslContext clientCtx) throws Throwable {
91          final ReadAndDiscardHandler sh = new ReadAndDiscardHandler(true, true);
92          final ReadAndDiscardHandler ch = new ReadAndDiscardHandler(false, true);
93          final String[] protocols = { "TLSv1", "TLSv1.1", "TLSv1.2" };
94  
95          sb.childHandler(new ChannelInitializer<SocketChannel>() {
96              @Override
97              protected void initChannel(SocketChannel sch) throws Exception {
98                  SSLEngine engine = serverCtx.newEngine(sch.alloc());
99                  engine.setUseClientMode(false);
100                 engine.setEnabledProtocols(protocols);
101 
102                 sch.pipeline().addLast(new SslHandler(engine));
103                 sch.pipeline().addLast(sh);
104             }
105         });
106         final Channel sc = sb.bind().get();
107 
108         cb.handler(new ChannelInitializer<SocketChannel>() {
109             @Override
110             protected void initChannel(SocketChannel sch) throws Exception {
111                 InetSocketAddress serverAddr = (InetSocketAddress) sc.localAddress();
112                 SSLEngine engine = clientCtx.newEngine(sch.alloc(), serverAddr.getHostString(), serverAddr.getPort());
113                 engine.setUseClientMode(true);
114                 engine.setEnabledProtocols(protocols);
115 
116                 sch.pipeline().addLast(new SslHandler(engine));
117                 sch.pipeline().addLast(ch);
118             }
119         });
120 
121         try {
122             SSLSessionContext clientSessionCtx = clientCtx.sessionContext();
123             ByteBuf msg = Unpooled.wrappedBuffer(new byte[] { 0xa, 0xb, 0xc, 0xd }, 0, 4);
124             Channel cc = cb.connect(sc.localAddress()).get();
125             cc.writeAndFlush(msg).sync();
126             cc.closeFuture().sync();
127             rethrowHandlerExceptions(sh, ch);
128             Set<String> sessions = sessionIdSet(clientSessionCtx.getIds());
129 
130             msg = Unpooled.wrappedBuffer(new byte[] { 0xa, 0xb, 0xc, 0xd }, 0, 4);
131             cc = cb.connect(sc.localAddress()).get();
132             cc.writeAndFlush(msg).sync();
133             cc.closeFuture().sync();
134             assertEquals(sessions, sessionIdSet(clientSessionCtx.getIds()), "Expected no new sessions");
135             rethrowHandlerExceptions(sh, ch);
136         } finally {
137             sc.close().awaitUninterruptibly();
138         }
139     }
140 
141     private static void rethrowHandlerExceptions(ReadAndDiscardHandler sh, ReadAndDiscardHandler ch) throws Throwable {
142         if (sh.exception.get() != null && !(sh.exception.get() instanceof IOException)) {
143             throw sh.exception.get();
144         }
145         if (ch.exception.get() != null && !(ch.exception.get() instanceof IOException)) {
146             throw ch.exception.get();
147         }
148         if (sh.exception.get() != null) {
149             throw sh.exception.get();
150         }
151         if (ch.exception.get() != null) {
152             throw ch.exception.get();
153         }
154     }
155 
156     private static Set<String> sessionIdSet(Enumeration<byte[]> sessionIds) {
157         Set<String> idSet = new HashSet<>();
158         byte[] id;
159         while (sessionIds.hasMoreElements()) {
160             id = sessionIds.nextElement();
161             idSet.add(ByteBufUtil.hexDump(Unpooled.wrappedBuffer(id)));
162         }
163         return idSet;
164     }
165 
166     @Sharable
167     private static class ReadAndDiscardHandler extends SimpleChannelInboundHandler<ByteBuf> {
168         final AtomicReference<Throwable> exception = new AtomicReference<>();
169         private final boolean server;
170         private final boolean autoRead;
171 
172         ReadAndDiscardHandler(boolean server, boolean autoRead) {
173             this.server = server;
174             this.autoRead = autoRead;
175         }
176 
177         @Override
178         public void messageReceived(ChannelHandlerContext ctx, ByteBuf in) throws Exception {
179             byte[] actual = new byte[in.readableBytes()];
180             in.readBytes(actual);
181             ctx.close();
182         }
183 
184         @Override
185         public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
186             try {
187                 ctx.flush();
188             } finally {
189                 if (!autoRead) {
190                     ctx.read();
191                 }
192             }
193         }
194 
195         @Override
196         public void exceptionCaught(ChannelHandlerContext ctx,
197                 Throwable cause) throws Exception {
198             if (logger.isWarnEnabled()) {
199                 logger.warn(
200                         "Unexpected exception from the " +
201                         (server? "server" : "client") + " side", cause);
202             }
203 
204             exception.compareAndSet(null, cause);
205             ctx.close();
206         }
207     }
