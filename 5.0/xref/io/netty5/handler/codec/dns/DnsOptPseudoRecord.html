16  package io.netty5.handler.codec.dns;
17  
18  import io.netty5.util.internal.UnstableApi;
19  
20  /**
21   * An <a href="https://tools.ietf.org/html/rfc6891#section-6.1">OPT RR</a> record.
22   * <p>
23   * This is used for <a href="https://tools.ietf.org/html/rfc6891#section-6.1.3">Extension
24   * Mechanisms for DNS (EDNS(0))</a>.
25   */
26  @UnstableApi
27  public interface DnsOptPseudoRecord extends DnsRecord {
28  
29      /**
30       * Returns the {@code EXTENDED-RCODE} which is encoded into {@link DnsOptPseudoRecord#timeToLive()}.
31       */
32      int extendedRcode();
33  
34      /**
35       * Returns the {@code VERSION} which is encoded into {@link DnsOptPseudoRecord#timeToLive()}.
36       */
37      int version();
38  
39      /**
40       * Returns the {@code flags} which includes {@code DO} and {@code Z} which is encoded
41       * into {@link DnsOptPseudoRecord#timeToLive()}.
42       */
43      int flags();
44  }
