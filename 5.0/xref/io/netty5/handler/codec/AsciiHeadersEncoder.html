17  package io.netty5.handler.codec;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.Map.Entry;
22  
23  import io.netty5.buffer.ByteBuf;
24  import io.netty5.buffer.ByteBufUtil;
25  import io.netty5.util.AsciiString;
26  import io.netty5.util.CharsetUtil;
27  
28  public final class AsciiHeadersEncoder {
29  
30      /**
31       * The separator characters to insert between a header name and a header value.
32       */
33      public enum SeparatorType {
34          /**
35           * {@code ':'}
36           */
37          COLON,
38          /**
39           * {@code ': '}
40           */
41          COLON_SPACE,
42      }
43  
44      /**
45       * The newline characters to insert between header entries.
46       */
47      public enum NewlineType {
48          /**
49           * {@code '\n'}
50           */
51          LF,
52          /**
53           * {@code '\r\n'}
54           */
55          CRLF
56      }
57  
58      private final ByteBuf buf;
59      private final SeparatorType separatorType;
60      private final NewlineType newlineType;
61  
62      public AsciiHeadersEncoder(ByteBuf buf) {
63          this(buf, SeparatorType.COLON_SPACE, NewlineType.CRLF);
64      }
65  
66      public AsciiHeadersEncoder(ByteBuf buf, SeparatorType separatorType, NewlineType newlineType) {
67          requireNonNull(buf, "buf");
68          requireNonNull(separatorType, "separatorType");
69          requireNonNull(newlineType, "newlineType");
70  
71          this.buf = buf;
72          this.separatorType = separatorType;
73          this.newlineType = newlineType;
74      }
75  
76      public void encode(Entry<CharSequence, CharSequence> entry) {
77          final CharSequence name = entry.getKey();
78          final CharSequence value = entry.getValue();
79          final ByteBuf buf = this.buf;
80          final int nameLen = name.length();
81          final int valueLen = value.length();
82          final int entryLen = nameLen + valueLen + 4;
83          int offset = buf.writerIndex();
84          buf.ensureWritable(entryLen);
85          writeAscii(buf, offset, name);
86          offset += nameLen;
87  
88          switch (separatorType) {
89              case COLON:
90                  buf.setByte(offset ++, ':');
91                  break;
92              case COLON_SPACE:
93                  buf.setByte(offset ++, ':');
94                  buf.setByte(offset ++, ' ');
95                  break;
96              default:
97                  throw new Error();
98          }
99  
100         writeAscii(buf, offset, value);
101         offset += valueLen;
102 
103         switch (newlineType) {
104             case LF:
105                 buf.setByte(offset ++, '\n');
106                 break;
107             case CRLF:
108                 buf.setByte(offset ++, '\r');
109                 buf.setByte(offset ++, '\n');
110                 break;
111             default:
112                 throw new Error();
113         }
114 
115         buf.writerIndex(offset);
116     }
117 
118     private static void writeAscii(ByteBuf buf, int offset, CharSequence value) {
119         if (value instanceof AsciiString) {
120             ByteBufUtil.copy((AsciiString) value, 0, buf, offset, value.length());
121         } else {
122             buf.setCharSequence(offset, value, CharsetUtil.US_ASCII);
123         }
124     }
125 }
