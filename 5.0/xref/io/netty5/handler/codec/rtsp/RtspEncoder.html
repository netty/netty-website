16  package io.netty5.handler.codec.rtsp;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty5.buffer.api.Buffer;
20  import io.netty5.handler.codec.UnsupportedMessageTypeException;
21  import io.netty5.handler.codec.http.HttpContent;
22  import io.netty5.handler.codec.http.HttpMessage;
23  import io.netty5.handler.codec.http.HttpObjectEncoder;
24  import io.netty5.handler.codec.http.HttpRequest;
25  import io.netty5.handler.codec.http.HttpResponse;
26  import io.netty5.util.internal.StringUtil;
27  
28  import static io.netty5.handler.codec.http.HttpConstants.CR;
29  import static io.netty5.handler.codec.http.HttpConstants.LF;
30  import static io.netty5.handler.codec.http.HttpConstants.SP;
31  import static io.netty5.util.CharsetUtil.US_ASCII;
32  import static io.netty5.util.CharsetUtil.UTF_8;
33  
34  /**
35   * Encodes an RTSP message represented in {@link HttpMessage} or an {@link HttpContent} into
36   * a {@link ByteBuf}.
37   */
38  public class RtspEncoder extends HttpObjectEncoder<HttpMessage> {
39      private static final short CRLF_SHORT = (CR << 8) | LF;
40  
41      @Override
42      public boolean acceptOutboundMessage(final Object msg)
43             throws Exception {
44          return super.acceptOutboundMessage(msg) && ((msg instanceof HttpRequest) || (msg instanceof HttpResponse));
45      }
46  
47      @Override
48      protected void encodeInitialLine(final Buffer buf, final HttpMessage message) throws Exception {
49          if (message instanceof HttpRequest) {
50              HttpRequest request = (HttpRequest) message;
51              buf.writeCharSequence(request.method().asciiName(), US_ASCII);
52              buf.writeByte(SP);
53              buf.writeCharSequence(request.uri(), UTF_8);
54              buf.writeByte(SP);
55              buf.writeCharSequence(request.protocolVersion().toString(), US_ASCII);
56              buf.writeShort(CRLF_SHORT);
57          } else if (message instanceof HttpResponse) {
58              HttpResponse response = (HttpResponse) message;
59              buf.writeCharSequence(response.protocolVersion().toString(), US_ASCII);
60              buf.writeByte(SP);
61              buf.writeCharSequence(response.status().codeAsText(), US_ASCII);
62              buf.writeByte(SP);
63              buf.writeCharSequence(response.status().reasonPhrase(), US_ASCII);
64              buf.writeShort(CRLF_SHORT);
65          } else {
66              throw new UnsupportedMessageTypeException("Unsupported type " + StringUtil.simpleClassName(message));
67          }
68      }
69  }
