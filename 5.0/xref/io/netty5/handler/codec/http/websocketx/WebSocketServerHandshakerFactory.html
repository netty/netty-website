16  package io.netty5.handler.codec.http.websocketx;
17  
18  import io.netty5.channel.Channel;
19  
20  import io.netty5.handler.codec.http.DefaultFullHttpResponse;
21  import io.netty5.handler.codec.http.HttpHeaderNames;
22  import io.netty5.handler.codec.http.HttpRequest;
23  import io.netty5.handler.codec.http.HttpResponse;
24  import io.netty5.handler.codec.http.HttpResponseStatus;
25  import io.netty5.handler.codec.http.HttpUtil;
26  import io.netty5.handler.codec.http.HttpVersion;
27  import io.netty5.util.concurrent.Future;
28  
29  import java.util.Objects;

31  /**
32   * Auto-detects the version of the Web Socket protocol in use and creates a new proper
33   * {@link WebSocketServerHandshaker}.
34   */
35  public class WebSocketServerHandshakerFactory {
36  
37      private final String webSocketURL;
38  
39      private final String subprotocols;
40  
41      private final WebSocketDecoderConfig decoderConfig;
42  
43      /**
44       * Constructor specifying the destination web socket location
45       *
46       * @param webSocketURL
47       *            URL for web socket communications. e.g "ws://myhost.com/mypath".
48       *            Subsequent web socket frames will be sent to this URL.
49       * @param subprotocols
50       *            CSV of supported protocols. Null if sub protocols not supported.
51       * @param allowExtensions
52       *            Allow extensions to be used in the reserved bits of the web socket frame
53       */
54      public WebSocketServerHandshakerFactory(
55              String webSocketURL, String subprotocols, boolean allowExtensions) {
56          this(webSocketURL, subprotocols, allowExtensions, 65536);
57      }
58  
59      /**
60       * Constructor specifying the destination web socket location
61       *
62       * @param webSocketURL
63       *            URL for web socket communications. e.g "ws://myhost.com/mypath".
64       *            Subsequent web socket frames will be sent to this URL.
65       * @param subprotocols
66       *            CSV of supported protocols. Null if sub protocols not supported.
67       * @param allowExtensions
68       *            Allow extensions to be used in the reserved bits of the web socket frame
69       * @param maxFramePayloadLength
70       *            Maximum allowable frame payload length. Setting this value to your application's
71       *            requirement may reduce denial of service attacks using long data frames.
72       */
73      public WebSocketServerHandshakerFactory(
74              String webSocketURL, String subprotocols, boolean allowExtensions,
75              int maxFramePayloadLength) {
76          this(webSocketURL, subprotocols, allowExtensions, maxFramePayloadLength, false);
77      }
78  
79      /**
80       * Constructor specifying the destination web socket location
81       *
82       * @param webSocketURL
83       *            URL for web socket communications. e.g "ws://myhost.com/mypath".
84       *            Subsequent web socket frames will be sent to this URL.
85       * @param subprotocols
86       *            CSV of supported protocols. Null if sub protocols not supported.
87       * @param allowExtensions
88       *            Allow extensions to be used in the reserved bits of the web socket frame
89       * @param maxFramePayloadLength
90       *            Maximum allowable frame payload length. Setting this value to your application's
91       *            requirement may reduce denial of service attacks using long data frames.
92       * @param allowMaskMismatch
93       *            When set to true, frames which are not masked properly according to the standard will still be
94       *            accepted.
95       */
96      public WebSocketServerHandshakerFactory(
97              String webSocketURL, String subprotocols, boolean allowExtensions,
98              int maxFramePayloadLength, boolean allowMaskMismatch) {
99          this(webSocketURL, subprotocols, WebSocketDecoderConfig.newBuilder()
100             .allowExtensions(allowExtensions)
101             .maxFramePayloadLength(maxFramePayloadLength)
102             .allowMaskMismatch(allowMaskMismatch)
103             .build());
104     }
105 
106     /**
107      * Constructor specifying the destination web socket location
108      *
109      * @param webSocketURL
110      *            URL for web socket communications. e.g "ws://myhost.com/mypath".
111      *            Subsequent web socket frames will be sent to this URL.
112      * @param subprotocols
113      *            CSV of supported protocols. Null if sub protocols not supported.
114      * @param decoderConfig
115      *            Frames decoder options.
116      */
117     public WebSocketServerHandshakerFactory(
118             String webSocketURL, String subprotocols, WebSocketDecoderConfig decoderConfig) {
119         this.webSocketURL = webSocketURL;
120         this.subprotocols = subprotocols;
121         this.decoderConfig = Objects.requireNonNull(decoderConfig, "decoderConfig");
122     }
123 
124     /**
125      * Instances a new handshaker
126      *
127      * @return A new WebSocketServerHandshaker for the requested web socket version. Null if web
128      *         socket version is not supported.
129      */
130     public WebSocketServerHandshaker newHandshaker(HttpRequest req) {
131 
132         CharSequence version = req.headers().get(HttpHeaderNames.SEC_WEBSOCKET_VERSION);
133         if (version != null && version.equals(WebSocketVersion.V13.toHttpHeaderValue())) {
134             // Version 13 of the wire protocol - RFC 6455 (version 17 of the draft hybi specification).
135             return new WebSocketServerHandshaker13(webSocketURL, subprotocols, decoderConfig);
136         }
137         return null;
138     }
139 
140     /**
141      * @deprecated use {@link #sendUnsupportedVersionResponse(Channel)}
142      */
143     @Deprecated
144     public static void sendUnsupportedWebSocketVersionResponse(Channel channel) {
145         sendUnsupportedVersionResponse(channel);
146     }
147 
148     /**
149      * Return that we need cannot not support the web socket version
150      */
151     public static Future<Void> sendUnsupportedVersionResponse(Channel channel) {
152         HttpResponse res = new DefaultFullHttpResponse(
153                 HttpVersion.HTTP_1_1,
154                 HttpResponseStatus.UPGRADE_REQUIRED, channel.bufferAllocator().allocate(0));
155         res.headers().set(HttpHeaderNames.SEC_WEBSOCKET_VERSION, WebSocketVersion.V13.toHttpHeaderValue());
156         HttpUtil.setContentLength(res, 0);
157         return channel.writeAndFlush(res);
158     }
159 }
