16  package io.netty5.handler.adaptor;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty5.buffer.api.Buffer;
20  import io.netty5.buffer.api.adaptor.ByteBufAdaptor;
21  import io.netty5.buffer.api.adaptor.ByteBufBuffer;
22  import io.netty5.channel.ChannelHandler;
23  import io.netty5.channel.ChannelHandler.Sharable;
24  import io.netty5.channel.ChannelHandlerContext;
25  import io.netty5.util.concurrent.Future;
26  
27  /**
28   * {@link ChannelHandler} that converts {@link ByteBuf} messages into {@link Buffer} messages, and vice versa,
29   * depending on configuration.
30   * <p>
31   * This class is useful as an intermediate handler that allows {@link ByteBuf}-based
32   * {@linkplain ChannelHandler handlers} and {@link Buffer}-based {@linkplain ChannelHandler handlers} to work together
33   * in the same {@linkplain io.netty5.channel.ChannelPipeline pipeline}.
34   * <p>
35   * It is, however, recommended that all handlers eventually be converted to use the {@link Buffer} API, as that is more
36   * future-proof.
37   * <p>
38   * Instances of this handler are {@link Sharable} and can be added to multiple pipelines.
39   * This is safe because the instances are immutable and thread-safe.
40   *
41   * @deprecated This handler will be moved out of Netty core and into a contrib repository, before Netty 5.0.0.Final
42   * is released.
43   */
44  @Deprecated
45  @Sharable
46  public final class BufferConversionHandler implements ChannelHandler {
47      private final Conversion onRead;
48      private final Conversion onWrite;
49      private final Conversion onUserEvent;
50  
51      /**
52       * Create a conversion handler where incoming reads are passed through the given conversion,
53       * and outgoing writes are passed through its {@linkplain Conversion#invert() inverse}.
54       *
55       * @param conversion The conversion to apply to all {@linkplain #channelRead(ChannelHandlerContext, Object) read}
56       *                  and {@linkplain #write(ChannelHandlerContext, Object) write} messages.
57       */
58      public BufferConversionHandler(Conversion conversion) {
59          this(conversion, conversion.invert(), Conversion.NONE);
60      }
61  
62      /**
63       * Create a conversion handler where incoming reads, and outgoing writes, are passed through their configured
64       * conversions.
65       *
66       * @param onRead The conversion to apply to all incoming
67       *              {@linkplain #channelRead(ChannelHandlerContext, Object) read} messages.
68       * @param onWrite The conversion to apply to all outgoing {@linkplain #write(ChannelHandlerContext, Object)}
69       *               messages.
70       */
71      public BufferConversionHandler(Conversion onRead, Conversion onWrite) {
72          this(onRead, onWrite, Conversion.NONE);
73      }
74  
75      /**
76       * Create a conversion handler where incoming reads, outgoing writes, and user events, are passed through their
77       * configured conversions.
78       *
79       * @param onRead The conversion to apply to all incoming
80       *              {@linkplain #channelRead(ChannelHandlerContext, Object) read} messages.
81       * @param onWrite The conversion to apply to all outgoing {@linkplain #write(ChannelHandlerContext, Object)}
82       *               messages.
83       * @param onUserEvent The conversion to apply to all incoming
84       *                    {@linkplain #userEventTriggered(ChannelHandlerContext, Object) user events}.
85       */
86      public BufferConversionHandler(Conversion onRead, Conversion onWrite, Conversion onUserEvent) {
87          this.onRead = onRead;
88          this.onWrite = onWrite;
89          this.onUserEvent = onUserEvent;
90      }
91  
92      /**
93       * Create a conversion handler to insert after {@link Buffer}-reading handles, and before {@link ByteBuf}-reading
94       * handles.
95       * For writes, the handler will convert in the opposite direction.
96       *
97       * @return A conversion handler.
98       */
99      public static BufferConversionHandler bufferToByteBuf() {
100         return LazyBufferToByteBuf.HANDLER;
101     }
102 
103     /**
104      * Create a conversion handler to insert after {@link ByteBuf}-reading handles, and before {@link Buffer}-reading
105      * handles.
106      * For writes, the handler will convert in the opposite direction.
107      *
108      * @return A conversion handler.
109      */
110     public static BufferConversionHandler byteBufToBuffer() {
111         return LazyByteBufToBuffer.HANDLER;
112     }
113 
114     @Override
115     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
116         ctx.fireChannelRead(onRead.convert(msg));
117     }
118 
119     @Override
120     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
121         ctx.fireUserEventTriggered(onUserEvent.convert(evt));
122     }
123 
124     @Override
125     public Future<Void> write(ChannelHandlerContext ctx, Object msg) {
126         return ctx.write(onWrite.convert(msg));
127     }
128 
129     /**
130      * The particular conversion operation to apply.
131      * See the individual operations for their specific behaviour.
132      */
133     public enum Conversion {
134         /**
135          * Convert {@link ByteBuf} instances to {@link Buffer} instances.
136          * <p>
137          * Messages that are either already of a {@link Buffer} type, or of an unknown type, will pass through
138          * unchanged.
139          */
140         BYTEBUF_TO_BUFFER {
141             @Override
142             public Object convert(Object msg) {
143                 if (msg instanceof ByteBuf) {
144                     ByteBuf buf = (ByteBuf) msg;
145                     return ByteBufBuffer.wrap(buf);
146                 }
147                 return msg;
148             }
149 
150             @Override
151             public Conversion invert() {
152                 return BUFFER_TO_BYTEBUF;
153             }
154         },
155         /**
156          * Convert {@link Buffer} instances to {@link ByteBuf} instances.
157          * <p>
158          * Messages that are either already of a {@link ByteBuf} type, or of an unknown type, will pass through
159          * unchanged.
160          */
161         BUFFER_TO_BYTEBUF {
162             @Override
163             public Object convert(Object msg) {
164                 if (msg instanceof Buffer) {
165                     Buffer buf = (Buffer) msg;
166                     return ByteBufAdaptor.intoByteBuf(buf);
167                 }
168                 return msg;
169             }
170 
171             @Override
172             public Conversion invert() {
173                 return BYTEBUF_TO_BUFFER;
174             }
175         },
176         /**
177          * Convert any {@link Buffer}s into {@link ByteBuf}s, and any {@link ByteBuf}s into {@link Buffer}s.
178          * <p>
179          * Messages of unknown types are passed through unchanged.
180          */
181         BOTH {
182             @Override
183             public Object convert(Object msg) {
184                 if (msg instanceof ByteBuf) {
185                     ByteBuf buf = (ByteBuf) msg;
186                     return ByteBufBuffer.wrap(buf);
187                 }
188                 if (msg instanceof Buffer) {
189                     Buffer buf = (Buffer) msg;
190                     return ByteBufAdaptor.intoByteBuf(buf);
191                 }
192                 return msg;
193             }
194 
195             @Override
196             public Conversion invert() {
197                 return this;
198             }
199         },
200         /**
201          * Do not convert anything, but let the messages pass through unchanged.
202          */
203         NONE {
204             @Override
205             public Object convert(Object msg) {
206                 return msg;
207             }
208 
209             @Override
210             public Conversion invert() {
211                 return this;
212             }
213         };
214 
215         /**
216          * Apply this conversion to the given message, if applicable.
217          *
218          * @param msg The message to maybe be converted.
219          * @return The result of the conversion.
220          */
221         public abstract Object convert(Object msg);
222 
223         /**
224          * Return a {@link Conversion} that is the inverse of this one.
225          *
226          * @return A conversion that converts buffers in the opposite direction.
227          */
228         public abstract Conversion invert();
229     }
230 
231     static final class LazyBufferToByteBuf {
232         static final BufferConversionHandler HANDLER =
233                 new BufferConversionHandler(Conversion.BUFFER_TO_BYTEBUF);
234 
235         private LazyBufferToByteBuf() {
236         }
237     }
238 
239     static final class LazyByteBufToBuffer {
240         static final BufferConversionHandler HANDLER =
241                 new BufferConversionHandler(Conversion.BYTEBUF_TO_BUFFER);
242 
243         private LazyByteBufToBuffer() {
244         }
245     }
