32  import io.netty5.util.internal.logging.InternalLoggerFactory;

34  import java.net.InetSocketAddress;
35  import java.util.concurrent.TimeUnit;
36  
37  import static java.util.Objects.requireNonNull;
38  
39  abstract class DnsQueryContext implements FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>> {
40  
41      private static final InternalLogger logger = InternalLoggerFactory.getInstance(DnsQueryContext.class);
42  
43      private final DnsNameResolver parent;
44      private final Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise;
45      private final int id;
46      private final DnsQuestion question;
47      private final DnsRecord[] additionals;
48      private final DnsRecord optResource;
49      private final InetSocketAddress nameServerAddr;
50  
51      private final boolean recursionDesired;
52      private volatile Future<?> timeoutFuture;
53  
54      DnsQueryContext(DnsNameResolver parent,
55                      InetSocketAddress nameServerAddr,
56                      DnsQuestion question,
57                      DnsRecord[] additionals,
58                      Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise) {
59  
60          this.parent = requireNonNull(parent, "parent");
61          this.nameServerAddr = requireNonNull(nameServerAddr, "nameServerAddr");
62          this.question = requireNonNull(question, "question");
63          this.additionals = requireNonNull(additionals, "additionals");
64          this.promise = requireNonNull(promise, "promise");
65          recursionDesired = parent.isRecursionDesired();
66          id = parent.queryContextManager.add(this);
67  
68          // Ensure we remove the id from the QueryContextManager once the query completes.
69          promise.asFuture().addListener(this);
70  
71          if (parent.isOptResourceEnabled() &&
72                  // Only add the extra OPT record if there is not already one. This is required as only one is allowed
73                  // as per RFC:
74                  //  - https://datatracker.ietf.org/doc/html/rfc6891#section-6.1.1
75                  !hasOptRecord(additionals)) {
76              optResource = new AbstractDnsOptPseudoRrRecord(parent.maxPayloadSize(), 0, 0) {
77                  // We may want to remove this in the future and let the user just specify the opt record in the query.
78              };
79          } else {
80              optResource = null;
81          }
82      }
83  
84      private static boolean hasOptRecord(DnsRecord[] additionals) {
85          if (additionals != null && additionals.length > 0) {
86              for (DnsRecord additional: additionals) {
87                  if (additional.type() == DnsRecordType.OPT) {
88                      return true;
89                  }
90              }
91          }
92          return false;
93      }
94  
95      InetSocketAddress nameServerAddr() {
96          return nameServerAddr;
97      }
98  
99      DnsQuestion question() {
100         return question;
101     }
102 
103     DnsNameResolver parent() {
104         return parent;
105     }
106 
107     protected abstract DnsQuery newQuery(int id);
108     protected abstract Channel channel();
109     protected abstract String protocol();
110 
111     void query(boolean flush, Promise<Void> writePromise) {
112         final DnsQuestion question = question();
113         final InetSocketAddress nameServerAddr = nameServerAddr();
114         final DnsQuery query = newQuery(id);
115 
116         query.setRecursionDesired(recursionDesired);
117 
118         query.addRecord(DnsSection.QUESTION, question);
119 
120         for (DnsRecord record: additionals) {
121             query.addRecord(DnsSection.ADDITIONAL, record);
122         }
123 
124         if (optResource != null) {
125             query.addRecord(DnsSection.ADDITIONAL, optResource);
126         }
127 
128         if (logger.isDebugEnabled()) {
129             logger.debug("{} WRITE: {}, [{}: {}], {}", channel(), protocol(), id, nameServerAddr, question);
130         }
131 
132         sendQuery(query, flush, writePromise);
133     }
134 
135     private void sendQuery(final DnsQuery query, final boolean flush, final Promise<Void> writePromise) {
136         if (parent.channelReadyPromise.isSuccess()) {
137             writeQuery(query, flush, writePromise);
138         } else if (parent.channelReadyPromise.isFailed()) {
139             failQuery(query, parent.channelReadyPromise.cause(), writePromise);
140         } else {
141             parent.channelReadyPromise.asFuture().addListener(future -> {
142                 if (future.isSuccess()) {
143                     // If the query is done in a late fashion (as the channel was not ready yet) we always flush
144                     // to ensure we did not race with a previous flush() that was done when the Channel was not
145                     // ready yet.
146                     writeQuery(query, true, writePromise);
147                 } else {
148                     failQuery(query, future.cause(), writePromise);
149                 }
150             });
151         }
152     }
153 
154     private void failQuery(DnsQuery query, Throwable cause, Promise<Void> writePromise) {
155         try {
156             promise.tryFailure(cause);
157             writePromise.setFailure(cause);
158         } finally {
159             ReferenceCountUtil.release(query);
160         }
161     }
162 
163     private void writeQuery(final DnsQuery query, final boolean flush, final Promise<Void> writePromise) {
164         final Future<Void> writeFuture = flush ? channel().writeAndFlush(query) :
165                 channel().write(query);
166         if (writeFuture.isDone()) {
167             onQueryWriteCompletion(writeFuture, writePromise);
168         } else {
169             writeFuture.addListener(future ->
170                     onQueryWriteCompletion(future, writePromise));
171         }
172     }
173 
174     private void onQueryWriteCompletion(Future<?> writeFuture, Promise<Void> writePromise) {
175         if (writeFuture.isFailed()) {
176             writePromise.setFailure(writeFuture.cause());
177             tryFailure("failed to send a query via " + protocol(), writeFuture.cause(), false);
178             return;
179         }
180         writePromise.setSuccess(null);
181         // Schedule a query timeout task if necessary.
182         final long queryTimeoutMillis = parent.queryTimeoutMillis();
183         if (queryTimeoutMillis > 0) {
184             timeoutFuture = parent.ch.executor().schedule(() -> {
185                 if (promise.isDone()) {
186                     // Received a response before the query times out.
187                     return;
188                 }
189 
190                 tryFailure("query via " + protocol() + " timed out after " +
191                         queryTimeoutMillis + " milliseconds", null, true);
192                 }, queryTimeoutMillis, TimeUnit.MILLISECONDS);
193         }
194     }
195 
196     /**
197      * Takes ownership of passed envelope
198      */
199     void finish(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope) {
200         final DnsResponse res = envelope.content();
201         if (res.count(DnsSection.QUESTION) != 1) {
202             logger.warn("Received a DNS response with invalid number of questions: {}", envelope);
203         } else if (!question().equals(res.recordAt(DnsSection.QUESTION))) {
204             logger.warn("Received a mismatching DNS response: {}", envelope);
205         } else if (trySuccess(envelope)) {
206             return; // Ownership transferred, don't release
207         }
208         ReferenceCountUtil.release(envelope);
209     }
210 
211     @SuppressWarnings("unchecked")
212     private boolean trySuccess(AddressedEnvelope<? extends DnsResponse, InetSocketAddress> envelope) {
213         return promise.trySuccess((AddressedEnvelope<DnsResponse, InetSocketAddress>) envelope);
214     }
215 
216     boolean tryFailure(String message, Throwable cause, boolean timeout) {
217         if (promise.isDone()) {
218             return false;
219         }
220         final InetSocketAddress nameServerAddr = nameServerAddr();
221 
222         final StringBuilder buf = new StringBuilder(message.length() + 64);
223         buf.append('[')
224            .append(nameServerAddr)
225            .append("] ")
226            .append(message)
227            .append(" (no stack trace available)");
228 
229         final DnsNameResolverException e;
230         if (timeout) {
231             // This was caused by an timeout so use DnsNameResolverTimeoutException to allow the user to
232             // handle it special (like retry the query).
233             e = new DnsNameResolverTimeoutException(nameServerAddr, question(), buf.toString());
234         } else {
235             e = new DnsNameResolverException(nameServerAddr, question(), buf.toString(), cause);
236         }
237         return promise.tryFailure(e);
238     }
239 
240     @Override
241     public void operationComplete(Future<? extends AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
242         // Cancel the timeout task.
243         final Future<?> timeoutFuture = this.timeoutFuture;
244         if (timeoutFuture != null) {
245             this.timeoutFuture = null;
246             timeoutFuture.cancel();
247         }
248 
249         // Remove the id from the manager as soon as the query completes. This may be because of success, failure or
250         // cancellation
251         parent.queryContextManager.remove(nameServerAddr, id);
252     }
