16  package io.netty5.channel.epoll;
17  
18  import io.netty5.channel.Channel;
19  import io.netty5.channel.ChannelConfig;
20  import io.netty5.channel.ChannelMetadata;
21  import io.netty5.channel.ChannelOutboundBuffer;
22  import io.netty5.channel.ChannelPipeline;
23  import io.netty5.channel.ChannelShutdownDirection;
24  import io.netty5.channel.EventLoop;
25  import io.netty5.channel.EventLoopGroup;
26  import io.netty5.channel.ServerChannel;
27  import io.netty5.channel.unix.UnixChannel;
28  
29  import java.net.SocketAddress;
30  
31  public abstract class AbstractEpollServerChannel
32          <P extends UnixChannel, L extends SocketAddress, R extends SocketAddress>
33          extends AbstractEpollChannel<P, L, R> implements ServerChannel {
34      private static final ChannelMetadata METADATA = new ChannelMetadata(false, 16);
35  
36      private final EventLoopGroup childEventLoopGroup;
37      // Will hold the remote address after accept(...) was successful.
38      // We need 24 bytes for the address as maximum + 1 byte for storing the length.
39      // So use 26 bytes as it's a power of two.
40      private final byte[] acceptedAddress = new byte[26];
41  
42      AbstractEpollServerChannel(EventLoop eventLoop, EventLoopGroup childEventLoopGroup,
43                                 Class<? extends Channel> childChannelType, int fd) {
44          this(eventLoop, childEventLoopGroup, childChannelType, new LinuxSocket(fd), false);
45      }
46  
47      AbstractEpollServerChannel(EventLoop eventLoop, EventLoopGroup childEventLoopGroup,
48                                 Class<? extends Channel> childChannelType, LinuxSocket fd) {
49          this(eventLoop, childEventLoopGroup, childChannelType, fd, isSoErrorZero(fd));
50      }
51  
52      AbstractEpollServerChannel(EventLoop eventLoop, EventLoopGroup childEventLoopGroup,
53                                 Class<? extends Channel> childChannelType, LinuxSocket fd, boolean active) {
54          super(null, eventLoop, fd, active);
55          this.childEventLoopGroup = validateEventLoopGroup(childEventLoopGroup, "childEventLoopGroup", childChannelType);
56      }
57  
58      @Override
59      public final EventLoopGroup childEventLoopGroup() {
60          return childEventLoopGroup;
61      }
62  
63      @Override
64      public final ChannelMetadata metadata() {
65          return METADATA;
66      }
67  
68      @Override
69      protected final R remoteAddress0() {
70          return null;
71      }
72  
73      @Override
74      public abstract EpollServerChannelConfig config();
75  
76      @Override
77      protected final void doWrite(ChannelOutboundBuffer in) {
78          throw new UnsupportedOperationException();
79      }
80  
81      @Override
82      protected final Object filterOutboundMessage(Object msg) {
83          throw new UnsupportedOperationException();
84      }
85  
86      abstract Channel newChildChannel(int fd, byte[] remote, int offset, int len) throws Exception;
87  
88      @Override
89      final void epollInReady() {
90          assert executor().inEventLoop();
91          final ChannelConfig config = config();
92          if (shouldBreakEpollInReady(config)) {
93              clearEpollIn0();
94              return;
95          }
96          final EpollRecvBufferAllocatorHandle allocHandle = recvBufAllocHandle();
97  
98          final ChannelPipeline pipeline = pipeline();
99          allocHandle.reset(config);
100         allocHandle.attemptedBytesRead(1);
101         epollInBefore();
102 
103         Throwable exception = null;
104         try {
105             try {
106                 do {
107                     // lastBytesRead represents the fd. We use lastBytesRead because it must be set so that the
108                     // EpollRecvBufferAllocatorHandle knows if it should try to read again or not when autoRead is
109                     // enabled.
110                     allocHandle.lastBytesRead(socket.accept(acceptedAddress));
111                     if (allocHandle.lastBytesRead() == -1) {
112                         // this means everything was handled for now
113                         break;
114                     }
115                     allocHandle.incMessagesRead(1);
116 
117                     readPending = false;
118                     pipeline.fireChannelRead(newChildChannel(allocHandle.lastBytesRead(), acceptedAddress, 1,
119                                                              acceptedAddress[0]));
120                 } while (allocHandle.continueReading() && !isShutdown(ChannelShutdownDirection.Inbound));
121             } catch (Throwable t) {
122                 exception = t;
123             }
124             allocHandle.readComplete();
125             pipeline.fireChannelReadComplete();
126 
127             if (exception != null) {
128                 pipeline.fireChannelExceptionCaught(exception);
129             }
130             readIfIsAutoRead();
131         } finally {
132             epollInFinally(config);
133         }
134     }
135 
136     @Override
137     protected final void doShutdown(ChannelShutdownDirection direction) {
138         throw new UnsupportedOperationException();
139     }
140 
141     @Override
142     public final boolean isShutdown(ChannelShutdownDirection direction) {
143         return !isActive();
144     }
145 
146     @Override
147     protected final boolean doFinishConnect(R requestedRemoteAddress) {
148         // Connect not supported by ServerChannel implementations
149         throw new UnsupportedOperationException();
150     }
151 
152     @Override
153     protected final boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) {
154         throw new UnsupportedOperationException();
155     }
156 }
