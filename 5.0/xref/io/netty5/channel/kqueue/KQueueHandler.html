43  /**
44   * {@link IoHandler} which uses kqueue under the covers. Only works on BSD!
45   */
46  @UnstableApi
47  public final class KQueueHandler implements IoHandler {
48      private static final InternalLogger logger = InternalLoggerFactory.getInstance(KQueueHandler.class);
49      private static final AtomicIntegerFieldUpdater<KQueueHandler> WAKEN_UP_UPDATER =
50              AtomicIntegerFieldUpdater.newUpdater(KQueueHandler.class, "wakenUp");
51      private static final int KQUEUE_WAKE_UP_IDENT = 0;
52  
53      static {
54          // Ensure JNI is initialized by the time this class is loaded by this time!
55          // We use unix-common methods in this class which are backed by JNI methods.
56          KQueue.ensureAvailability();
57      }
58  
59      private final boolean allowGrowing;
60      private final FileDescriptor kqueueFd;
61      private final KQueueEventArray changeList;
62      private final KQueueEventArray eventList;
63      private final SelectStrategy selectStrategy;
64      private final IovArray iovArray = new IovArray();
65      private final IntSupplier selectNowSupplier = this::kqueueWaitNow;
66      private final IntObjectMap<AbstractKQueueChannel> channels = new IntObjectHashMap<AbstractKQueueChannel>(4096);
67  
68      private volatile int wakenUp;
69  
70      private static AbstractKQueueChannel cast(Channel channel) {
71          if (channel instanceof AbstractKQueueChannel) {
72              return (AbstractKQueueChannel) channel;
73          }
74          throw new IllegalArgumentException("Channel of type " + StringUtil.simpleClassName(channel) + " not supported");
75      }
76  
77      private KQueueHandler() {
78          this(0, DefaultSelectStrategyFactory.INSTANCE.newSelectStrategy());
79      }
80  
81      private KQueueHandler(int maxEvents, SelectStrategy strategy) {
82          selectStrategy = requireNonNull(strategy, "strategy");
83          this.kqueueFd = Native.newKQueue();
84          if (maxEvents == 0) {
85              allowGrowing = true;
86              maxEvents = 4096;
87          } else {
88              allowGrowing = false;
89          }
90          changeList = new KQueueEventArray(maxEvents);
91          eventList = new KQueueEventArray(maxEvents);
92          int result = Native.keventAddUserEvent(kqueueFd.intValue(), KQUEUE_WAKE_UP_IDENT);
93          if (result < 0) {
94              destroy();
95              throw new IllegalStateException("kevent failed to add user event with errno: " + (-result));
96          }
97      }

109     public static IoHandlerFactory newFactory(final int maxEvents,
110                                               final SelectStrategyFactory selectStrategyFactory) {
111         checkPositiveOrZero(maxEvents, "maxEvents");
112         requireNonNull(selectStrategyFactory, "selectStrategyFactory");
113         return () -> new KQueueHandler(maxEvents, selectStrategyFactory.newSelectStrategy());
114     }

116     @Override
117     public void register(Channel channel) {
118         final AbstractKQueueChannel kQueueChannel = cast(channel);
119         final int id = kQueueChannel.fd().intValue();
120         AbstractKQueueChannel old = channels.put(id, kQueueChannel);
121         // We either expect to have no Channel in the map with the same FD or that the FD of the old Channel is already
122         // closed.
123         assert old == null || !old.isOpen();
124 
125         kQueueChannel.register0(new KQueueRegistration() {
126             @Override
127             public void evSet(short filter, short flags, int fflags) {
128                 KQueueHandler.this.evSet(kQueueChannel, filter, flags, fflags);
129             }
130 
131             @Override
132             public IovArray cleanArray() {
133                 return KQueueHandler.this.cleanArray();
134             }
135         });
136     }
137 
138     @Override
139     public void deregister(Channel channel) throws Exception {
140         AbstractKQueueChannel kQueueChannel = cast(channel);
141         int fd = kQueueChannel.fd().intValue();
142 
143         AbstractKQueueChannel old = channels.remove(fd);
144         if (old != null && old != kQueueChannel) {
145             // The Channel mapping was already replaced due FD reuse, put back the stored Channel.
146             channels.put(fd, old);
147 
148             // If we found another Channel in the map that is mapped to the same FD the given Channel MUST be closed.
149             assert !kQueueChannel.isOpen();
150         } else if (kQueueChannel.isOpen()) {
151             // Remove the filters. This is only needed if it's still open as otherwise it will be automatically
152             // removed once the file-descriptor is closed.
153             //
154             // See also https://www.freebsd.org/cgi/man.cgi?query=kqueue&sektion=2
155             kQueueChannel.unregisterFilters();
156         }
157 
158         kQueueChannel.deregister0();
159     }
160 
161     private void evSet(AbstractKQueueChannel ch, short filter, short flags, int fflags) {
162         changeList.evSet(ch, filter, flags, fflags);
163     }
164 
165     private IovArray cleanArray() {
166         iovArray.clear();
167         return iovArray;
168     }
169 
170     @Override
171     public void wakeup(boolean inEventLoop) {
172         if (!inEventLoop && WAKEN_UP_UPDATER.compareAndSet(this, 0, 1)) {
173             wakeup();
174         }
175     }
176 
177     private void wakeup() {
178         Native.keventTriggerUserEvent(kqueueFd.intValue(), KQUEUE_WAKE_UP_IDENT);
179         // Note that the result may return an error (e.g. errno = EBADF after the event loop has been shutdown).
180         // So it is not very practical to assert the return value is always >= 0.
181     }
182 
183     private int kqueueWait(IoExecutionContext context, boolean oldWakeup) throws IOException {
184         // If a task was submitted when wakenUp value was 1, the task didn't get a chance to produce wakeup event.
185         // So we need to check task queue again before calling kqueueWait. If we don't, the task might be pended
186         // until kqueueWait was timed out. It might be pended until idle timeout if IdleStateHandler existed
187         // in pipeline.
188         if (oldWakeup && !context.canBlock()) {
189             return kqueueWaitNow();
190         }
191 
192         long totalDelay = context.delayNanos(System.nanoTime());
193         int delaySeconds = (int) min(totalDelay / 1000000000L, Integer.MAX_VALUE);
194         return kqueueWait(delaySeconds, (int) min(totalDelay - delaySeconds * 1000000000L, Integer.MAX_VALUE));
195     }
196 
197     private int kqueueWaitNow() throws IOException {
198         return kqueueWait(0, 0);
199     }
200 
201     private int kqueueWait(int timeoutSec, int timeoutNs) throws IOException {
202         int numEvents = Native.keventWait(kqueueFd.intValue(), changeList, eventList, timeoutSec, timeoutNs);
203         changeList.clear();
204         return numEvents;
205     }
206 
207     private void processReady(int ready) {
208         for (int i = 0; i < ready; ++i) {
209             final short filter = eventList.filter(i);
210             final short flags = eventList.flags(i);
211             final int fd = eventList.fd(i);
212             if (filter == Native.EVFILT_USER || (flags & Native.EV_ERROR) != 0) {
213                 // EV_ERROR is returned if the FD is closed synchronously (which removes from kqueue) and then
214                 // we later attempt to delete the filters from kqueue.
215                 assert filter != Native.EVFILT_USER ||
216                         (filter == Native.EVFILT_USER && fd == KQUEUE_WAKE_UP_IDENT);
217                 continue;
218             }
219 
220             AbstractKQueueChannel channel = channels.get(fd);
221             if (channel == null) {
222                 // This may happen if the channel has already been closed, and it will be removed from kqueue anyways.
223                 // We also handle EV_ERROR above to skip this even early if it is a result of a referencing a closed and
224                 // thus removed from kqueue FD.
225                 logger.warn("events[{}]=[{}, {}] had no channel!", i, fd, filter);
226                 continue;
227             }
228 
229             AbstractKQueueUnsafe unsafe = (AbstractKQueueUnsafe) channel.unsafe();
230             // First check for EPOLLOUT as we may need to fail the connect Promise before try
231             // to read from the file descriptor.
232             if (filter == Native.EVFILT_WRITE) {
233                 unsafe.writeReady();
234             } else if (filter == Native.EVFILT_READ) {
235                 // Check READ before EOF to ensure all data is read before shutting down the input.
236                 unsafe.readReady(eventList.data(i));
237             } else if (filter == Native.EVFILT_SOCK && (eventList.fflags(i) & Native.NOTE_RDHUP) != 0) {
238                 unsafe.readEOF();
239             }
240 
241             // Check if EV_EOF was set, this will notify us for connection-reset in which case
242             // we may close the channel directly or try to read more data depending on the state of the
243             // Channel and also depending on the AbstractKQueueChannel subtype.
244             if ((flags & Native.EV_EOF) != 0) {
245                 unsafe.readEOF();
246             }
247         }
248     }

250     @Override
251     public int run(IoExecutionContext context) {
252         int handled = 0;
253         try {
254             int strategy = selectStrategy.calculateStrategy(selectNowSupplier, !context.canBlock());
255             switch (strategy) {
256                 case SelectStrategy.CONTINUE:
257                     return 0;
258 
259                 case SelectStrategy.BUSY_WAIT:
260                     // fall-through to SELECT since the busy-wait is not supported with kqueue
261 
262                 case SelectStrategy.SELECT:
263                     strategy = kqueueWait(context, WAKEN_UP_UPDATER.getAndSet(this, 0) == 1);
264 
265                     // 'wakenUp.compareAndSet(false, true)' is always evaluated
266                     // before calling 'selector.wakeup()' to reduce the wake-up
267                     // overhead. (Selector.wakeup() is an expensive operation.)
268                     //
269                     // However, there is a race condition in this approach.
270                     // The race condition is triggered when 'wakenUp' is set to
271                     // true too early.
272                     //
273                     // 'wakenUp' is set to true too early if:
274                     // 1) Selector is waken up between 'wakenUp.set(false)' and
275                     //    'selector.select(...)'. (BAD)
276                     // 2) Selector is waken up between 'selector.select(...)' and
277                     //    'if (wakenUp.get()) { ... }'. (OK)
278                     //
279                     // In the first case, 'wakenUp' is set to true and the
280                     // following 'selector.select(...)' will wake up immediately.
281                     // Until 'wakenUp' is set to false again in the next round,
282                     // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
283                     // any attempt to wake up the Selector will fail, too, causing
284                     // the following 'selector.select(...)' call to block
285                     // unnecessarily.
286                     //
287                     // To fix this problem, we wake up the selector again if wakenUp
288                     // is true immediately after selector.select(...).
289                     // It is inefficient in that it wakes up the selector for both
290                     // the first case (BAD - wake-up required) and the second case
291                     // (OK - no wake-up required).
292 
293                     if (wakenUp == 1) {
294                         wakeup();
295                     }
296                     // fallthrough
297                 default:
298             }
299 
300             if (strategy > 0) {
301                 handled = strategy;
302                 processReady(strategy);
303             }
304             if (allowGrowing && strategy == eventList.capacity()) {
305                 //increase the size of the array as we needed the whole space for the events
306                 eventList.realloc(false);
307             }
308         } catch (Error e) {
309             throw e;
310         } catch (Throwable t) {
311             handleLoopException(t);
312         }
313         return handled;
314     }

316     @Override
317     public void destroy() {
318         try {
319             try {
320                 kqueueFd.close();
321             } catch (IOException e) {
322                 logger.warn("Failed to close the kqueue fd.", e);
323             }
324         } finally {
325             // Cleanup all native memory!
326             changeList.free();
327             eventList.free();
328         }
329     }
330 
331     @Override
332     public void prepareToDestroy() {
333         try {
334             kqueueWaitNow();
335         } catch (IOException e) {
336             // ignore on close
337         }
338 
339         // Using the intermediate collection to prevent ConcurrentModificationException.
340         // In the `close()` method, the channel is deleted from `channels` map.
341         AbstractKQueueChannel[] localChannels = channels.values().toArray(new AbstractKQueueChannel[0]);
342 
343         for (AbstractKQueueChannel ch: localChannels) {
344             ch.unsafe().close(ch.newPromise());
345         }
346     }
347 
348     private static void handleLoopException(Throwable t) {
349         logger.warn("Unexpected exception in the selector loop.", t);
350 
351         // Prevent possible consecutive immediate failures that lead to
352         // excessive CPU consumption.
353         try {
354             Thread.sleep(1000);
355         } catch (InterruptedException e) {
356             // Ignore.
357         }
358     }
