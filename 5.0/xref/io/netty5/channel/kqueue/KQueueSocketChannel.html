16  package io.netty5.channel.kqueue;
17  
18  import io.netty5.buffer.api.Buffer;
19  import io.netty5.channel.ChannelOutboundBuffer;
20  import io.netty5.channel.EventLoop;
21  import io.netty5.channel.socket.InternetProtocolFamily;
22  import io.netty5.channel.socket.SocketChannel;
23  import io.netty5.channel.unix.IovArray;
24  import io.netty5.util.concurrent.Future;
25  import io.netty5.util.concurrent.GlobalEventExecutor;
26  import io.netty5.util.internal.UnstableApi;
27  
28  import java.net.InetSocketAddress;
29  import java.net.SocketAddress;
30  import java.util.concurrent.Executor;
31  
32  @UnstableApi
33  public final class KQueueSocketChannel
34          extends AbstractKQueueStreamChannel<KQueueServerSocketChannel, InetSocketAddress, InetSocketAddress>
35          implements SocketChannel {
36      private final KQueueSocketChannelConfig config;
37  
38      public KQueueSocketChannel(EventLoop eventLoop) {
39          this(eventLoop, null);
40      }
41  
42      public KQueueSocketChannel(EventLoop eventLoop, InternetProtocolFamily protocol) {
43          super(null, eventLoop, BsdSocket.newSocketStream(protocol), false);
44          config = new KQueueSocketChannelConfig(this);
45      }
46  
47      public KQueueSocketChannel(EventLoop eventLoop, int fd) {
48          super(eventLoop, new BsdSocket(fd));
49          config = new KQueueSocketChannelConfig(this);
50      }
51  
52      KQueueSocketChannel(KQueueServerSocketChannel parent, EventLoop eventLoop,
53                          BsdSocket fd, InetSocketAddress remoteAddress) {
54          super(parent, eventLoop, fd, remoteAddress);
55          config = new KQueueSocketChannelConfig(this);
56      }
57  
58      @Override
59      public KQueueSocketChannelConfig config() {
60          return config;
61      }
62  
63      @Override
64      protected boolean doConnect0(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
65          if (config.isTcpFastOpenConnect()) {
66              ChannelOutboundBuffer outbound = outboundBuffer();
67              outbound.addFlush();
68              Object curr;
69              if ((curr = outbound.current()) instanceof Buffer) {
70                  Buffer initialData = (Buffer) curr;
71                  // Don't bother with TCP FastOpen if we don't have any initial data to send anyway.
72                  if (initialData.readableBytes() > 0) {
73                      IovArray iov = new IovArray();
74                      try {
75                          initialData.forEachReadable(0, iov);
76                          int bytesSent = socket.connectx(
77                                  (InetSocketAddress) localAddress, (InetSocketAddress) remoteAddress, iov, true);
78                          writeFilter(true);
79                          outbound.removeBytes(Math.abs(bytesSent));
80                          // The `connectx` method returns a negative number if connection is in-progress.
81                          // So we should return `true` to indicate that connection was established, if it's positive.
82                          return bytesSent > 0;
83                      } finally {
84                          iov.release();
85                      }
86                  }
87              }
88          }
89          return super.doConnect0(remoteAddress, localAddress);
90      }
91  
92      @Override
93      protected Future<Executor> prepareToClose() {
94          try {
95              // Check isOpen() first as otherwise it will throw a RuntimeException
96              // when call getSoLinger() as the fd is not valid anymore.
97              if (isOpen() && config().getSoLinger() > 0) {
98                  // We need to cancel this key of the channel so we may not end up in a eventloop spin
99                  // because we try to read or write until the actual close happens which may be later due
100                 // SO_LINGER handling.
101                 // See https://github.com/netty/netty/issues/4449
102                 return executor().deregisterForIo(this).map(v -> GlobalEventExecutor.INSTANCE);
103             }
104         } catch (Throwable ignore) {
105             // Ignore the error as the underlying channel may be closed in the meantime and so
106             // getSoLinger() may produce an exception. In this case we just return null.
107             // See https://github.com/netty/netty/issues/4449
108         }
109         return null;
110     }
111 }
