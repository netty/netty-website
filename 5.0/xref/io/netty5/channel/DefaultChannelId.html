35  /**
36   * The default {@link ChannelId} implementation.
37   */
38  public final class DefaultChannelId implements ChannelId {
39  
40      private static final long serialVersionUID = 3884076183504074063L;
41  
42      private static final InternalLogger logger = InternalLoggerFactory.getInstance(DefaultChannelId.class);
43      private static final byte[] MACHINE_ID;
44      private static final int PROCESS_ID_LEN = 4;
45      private static final int PROCESS_ID;
46      private static final int SEQUENCE_LEN = 4;
47      private static final int TIMESTAMP_LEN = 8;
48      private static final int RANDOM_LEN = 4;
49  
50      private static final AtomicInteger nextSequence = new AtomicInteger();
51  
52      /**
53       * Returns a new {@link DefaultChannelId} instance.
54       */
55      public static DefaultChannelId newInstance() {
56          return new DefaultChannelId();
57      }
58  
59      static {
60          int processId = -1;
61          String customProcessId = SystemPropertyUtil.get("io.netty5.processId");
62          if (customProcessId != null) {
63              try {
64                  processId = Integer.parseInt(customProcessId);
65              } catch (NumberFormatException e) {
66                  // Malformed input.
67              }
68  
69              if (processId < 0) {
70                  processId = -1;
71                  logger.warn("-Dio.netty5.processId: {} (malformed)", customProcessId);
72              } else if (logger.isDebugEnabled()) {
73                  logger.debug("-Dio.netty5.processId: {} (user-set)", processId);
74              }
75          }
76  
77          if (processId < 0) {
78              processId = defaultProcessId();
79              if (logger.isDebugEnabled()) {
80                  logger.debug("-Dio.netty5.processId: {} (auto-detected)", processId);
81              }
82          }
83  
84          PROCESS_ID = processId;
85  
86          byte[] machineId = null;
87          String customMachineId = SystemPropertyUtil.get("io.netty5.machineId");
88          if (customMachineId != null) {
89              try {
90                  machineId = parseMAC(customMachineId);
91              } catch (Exception e) {
92                  logger.warn("-Dio.netty5.machineId: {} (malformed)", customMachineId, e);
93              }
94              if (machineId != null) {
95                  logger.debug("-Dio.netty5.machineId: {} (user-set)", customMachineId);
96              }
97          }
98  
99          if (machineId == null) {
100             machineId = defaultMachineId();
101             if (logger.isDebugEnabled()) {
102                 logger.debug("-Dio.netty5.machineId: {} (auto-detected)", MacAddressUtil.formatAddress(machineId));
103             }
104         }
105 
106         MACHINE_ID = machineId;
107     }
108 
109     private static int defaultProcessId() {
110         ClassLoader loader = null;
111         String value;
112         try {
113             loader = PlatformDependent.getClassLoader(DefaultChannelId.class);
114             // Invoke java.lang.management.ManagementFactory.getRuntimeMXBean().getName()
115             Class<?> mgmtFactoryType = Class.forName("java.lang.management.ManagementFactory", true, loader);
116             Class<?> runtimeMxBeanType = Class.forName("java.lang.management.RuntimeMXBean", true, loader);
117 
118             Method getRuntimeMXBean = mgmtFactoryType.getMethod("getRuntimeMXBean", EmptyArrays.EMPTY_CLASSES);
119             Object bean = getRuntimeMXBean.invoke(null, EmptyArrays.EMPTY_OBJECTS);
120             Method getName = runtimeMxBeanType.getMethod("getName", EmptyArrays.EMPTY_CLASSES);
121             value = (String) getName.invoke(bean, EmptyArrays.EMPTY_OBJECTS);
122         } catch (Throwable t) {
123             logger.debug("Could not invoke ManagementFactory.getRuntimeMXBean().getName(); Android?", t);
124             try {
125                 // Invoke android.os.Process.myPid()
126                 Class<?> processType = Class.forName("android.os.Process", true, loader);
127                 Method myPid = processType.getMethod("myPid", EmptyArrays.EMPTY_CLASSES);
128                 value = myPid.invoke(null, EmptyArrays.EMPTY_OBJECTS).toString();
129             } catch (Throwable t2) {
130                 logger.debug("Could not invoke Process.myPid(); not Android?", t2);
131                 value = "";
132             }
133         }
134 
135         int atIndex = value.indexOf('@');
136         if (atIndex >= 0) {
137             value = value.substring(0, atIndex);
138         }
139 
140         int pid;
141         try {
142             pid = Integer.parseInt(value);
143         } catch (NumberFormatException e) {
144             // value did not contain an integer.
145             pid = -1;
146         }
147 
148         if (pid < 0) {
149             pid = ThreadLocalRandom.current().nextInt();
150             logger.warn("Failed to find the current process ID from '{}'; using a random value: {}",  value, pid);
151         }
152 
153         return pid;
154     }

156     private final byte[] data;
157     private final int hashCode;
158 
159     private transient String shortValue;
160     private transient String longValue;
161 
162     private DefaultChannelId() {
163         data = new byte[MACHINE_ID.length + PROCESS_ID_LEN + SEQUENCE_LEN + TIMESTAMP_LEN + RANDOM_LEN];
164         int i = 0;
165 
166         // machineId
167         System.arraycopy(MACHINE_ID, 0, data, i, MACHINE_ID.length);
168         i += MACHINE_ID.length;
169 
170         // processId
171         i = writeInt(i, PROCESS_ID);
172 
173         // sequence
174         i = writeInt(i, nextSequence.getAndIncrement());
175 
176         // timestamp (kind of)
177         i = writeLong(i, Long.reverse(System.nanoTime()) ^ System.currentTimeMillis());
178 
179         // random
180         int random = ThreadLocalRandom.current().nextInt();
181         i = writeInt(i, random);
182         assert i == data.length;
183 
184         hashCode = Arrays.hashCode(data);
185     }
186 
187     private int writeInt(int i, int value) {
188         data[i ++] = (byte) (value >>> 24);
189         data[i ++] = (byte) (value >>> 16);
190         data[i ++] = (byte) (value >>> 8);
191         data[i ++] = (byte) value;
192         return i;
193     }
194 
195     private int writeLong(int i, long value) {
196         data[i ++] = (byte) (value >>> 56);
197         data[i ++] = (byte) (value >>> 48);
198         data[i ++] = (byte) (value >>> 40);
199         data[i ++] = (byte) (value >>> 32);
200         data[i ++] = (byte) (value >>> 24);
201         data[i ++] = (byte) (value >>> 16);
202         data[i ++] = (byte) (value >>> 8);
203         data[i ++] = (byte) value;
204         return i;
205     }

207     @Override
208     public String asShortText() {
209         String shortValue = this.shortValue;
210         if (shortValue == null) {
211             this.shortValue = shortValue = ByteBufUtil.hexDump(data, data.length - RANDOM_LEN, RANDOM_LEN);
212         }
213         return shortValue;
214     }
215 
216     @Override
217     public String asLongText() {
218         String longValue = this.longValue;
219         if (longValue == null) {
220             this.longValue = longValue = newLongValue();
221         }
222         return longValue;
223     }
224 
225     private String newLongValue() {
226         StringBuilder buf = new StringBuilder(2 * data.length + 5);
227         int i = 0;
228         i = appendHexDumpField(buf, i, MACHINE_ID.length);
229         i = appendHexDumpField(buf, i, PROCESS_ID_LEN);
230         i = appendHexDumpField(buf, i, SEQUENCE_LEN);
231         i = appendHexDumpField(buf, i, TIMESTAMP_LEN);
232         i = appendHexDumpField(buf, i, RANDOM_LEN);
233         assert i == data.length;
234         return buf.substring(0, buf.length() - 1);
235     }
236 
237     private int appendHexDumpField(StringBuilder buf, int i, int length) {
238         buf.append(ByteBufUtil.hexDump(data, i, length));
239         buf.append('-');
240         i += length;
241         return i;
242     }

244     @Override
245     public int hashCode() {
246         return hashCode;
247     }
248 
249     @Override
250     public int compareTo(final ChannelId o) {
251         if (this == o) {
252             // short circuit
253             return 0;
254         }
255         if (o instanceof DefaultChannelId) {
256             // lexicographic comparison
257             final byte[] otherData = ((DefaultChannelId) o).data;
258             int len1 = data.length;
259             int len2 = otherData.length;
260             int len = Math.min(len1, len2);
261 
262             for (int k = 0; k < len; k++) {
263                 byte x = data[k];
264                 byte y = otherData[k];
265                 if (x != y) {
266                     // treat these as unsigned bytes for comparison
267                     return (x & 0xff) - (y & 0xff);
268                 }
269             }
270             return len1 - len2;
271         }
272 
273         return asLongText().compareTo(o.asLongText());
274     }
275 
276     @Override
277     public boolean equals(Object obj) {
278         if (this == obj) {
279             return true;
280         }
281         if (!(obj instanceof DefaultChannelId)) {
282             return false;
283         }
284         DefaultChannelId other = (DefaultChannelId) obj;
285         return hashCode == other.hashCode && Arrays.equals(data, other.data);
286     }
287 
288     @Override
289     public String toString() {
290         return asShortText();
291     }
