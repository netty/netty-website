16  package io.netty.example.discard;
17  
18  import io.netty.bootstrap.Bootstrap;
19  import io.netty.channel.ChannelFuture;
20  import io.netty.channel.ChannelInitializer;
21  import io.netty.channel.ChannelPipeline;
22  import io.netty.channel.EventLoopGroup;
23  import io.netty.channel.nio.NioEventLoopGroup;
24  import io.netty.channel.socket.SocketChannel;
25  import io.netty.channel.socket.nio.NioSocketChannel;
26  import io.netty.example.util.ServerUtil;
27  import io.netty.handler.ssl.SslContext;
28  
29  /**
30   * Keeps sending random data to the specified address.
31   */
32  public final class DiscardClient {
33  
34      static final String HOST = System.getProperty("host", "127.0.0.1");
35      static final int PORT = Integer.parseInt(System.getProperty("port", "8009"));
36      static final int SIZE = Integer.parseInt(System.getProperty("size", "256"));
37  
38      public static void main(String[] args) throws Exception {
39          // Configure SSL.
40          final SslContext sslCtx = ServerUtil.buildSslContext();
41  
42          EventLoopGroup group = new NioEventLoopGroup();
43          try {
44              Bootstrap b = new Bootstrap();
45              b.group(group)
46               .channel(NioSocketChannel.class)
47               .handler(new ChannelInitializer<SocketChannel>() {
48                   @Override
49                   protected void initChannel(SocketChannel ch) throws Exception {
50                       ChannelPipeline p = ch.pipeline();
51                       if (sslCtx != null) {
52                           p.addLast(sslCtx.newHandler(ch.alloc(), HOST, PORT));
53                       }
54                       p.addLast(new DiscardClientHandler());
55                   }
56               });
57  
58              // Make the connection attempt.
59              ChannelFuture f = b.connect(HOST, PORT).sync();
60  
61              // Wait until the connection is closed.
62              f.channel().closeFuture().sync();
63          } finally {
64              group.shutdownGracefully();
65          }
66      }
67  }
