31  /**
32   * Triggers an {@link IdleStateEvent} when a {@link Channel} has not performed
33   * read, write, or both operation for a while.
34   *
35   * <h3>Supported idle states</h3>
36   * <table border="1">
37   * <tr>
38   * <th>Property</th><th>Meaning</th>
39   * </tr>
40   * <tr>
41   * <td>{@code readerIdleTime}</td>
42   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#READER_IDLE}
43   *     will be triggered when no read was performed for the specified period of
44   *     time.  Specify {@code 0} to disable.</td>
45   * </tr>
46   * <tr>
47   * <td>{@code writerIdleTime}</td>
48   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#WRITER_IDLE}
49   *     will be triggered when no write was performed for the specified period of
50   *     time.  Specify {@code 0} to disable.</td>
51   * </tr>
52   * <tr>
53   * <td>{@code allIdleTime}</td>
54   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#ALL_IDLE}
55   *     will be triggered when neither read nor write was performed for the
56   *     specified period of time.  Specify {@code 0} to disable.</td>
57   * </tr>
58   * </table>
59   *
60   * <pre>
61   * // An example that sends a ping message when there is no outbound traffic
62   * // for 30 seconds.  The connection is closed when there is no inbound traffic
63   * // for 60 seconds.
64   *
65   * public class MyChannelInitializer extends {@link ChannelInitializer}&lt;{@link Channel}&gt; {
66   *     {@code @Override}
67   *     public void initChannel({@link Channel} channel) {
68   *         channel.pipeline().addLast("idleStateHandler", new {@link IdleStateHandler}(60, 30, 0));
69   *         channel.pipeline().addLast("myHandler", new MyHandler());
70   *     }
71   * }
72   *
73   * // Handler should handle the {@link IdleStateEvent} triggered by {@link IdleStateHandler}.
74   * public class MyHandler extends {@link ChannelDuplexHandler} {
75   *     {@code @Override}
76   *     public void userEventTriggered({@link ChannelHandlerContext} ctx, {@link Object} evt) throws {@link Exception} {
77   *         if (evt instanceof {@link IdleStateEvent}) {
78   *             {@link IdleStateEvent} e = ({@link IdleStateEvent}) evt;
79   *             if (e.state() == {@link IdleState}.READER_IDLE) {
80   *                 ctx.close();
81   *             } else if (e.state() == {@link IdleState}.WRITER_IDLE) {
82   *                 ctx.writeAndFlush(new PingMessage());
83   *             }
84   *         }
85   *     }
86   * }
87   *
88   * {@link ServerBootstrap} bootstrap = ...;
89   * ...
90   * bootstrap.childHandler(new MyChannelInitializer());
91   * ...
92   * </pre>
93   *
94   * @see ReadTimeoutHandler
95   * @see WriteTimeoutHandler
96   */
97  public class IdleStateHandler extends ChannelDuplexHandler {
98      private static final long MIN_TIMEOUT_NANOS = TimeUnit.MILLISECONDS.toNanos(1);

100     // Not create a new ChannelFutureListener per write operation to reduce GC pressure.
101     private final ChannelFutureListener writeListener = new ChannelFutureListener() {
102         @Override
103         public void operationComplete(ChannelFuture future) throws Exception {
104             lastWriteTime = System.nanoTime();
105             firstWriterIdleEvent = firstAllIdleEvent = true;
106         }
107     };

111     private final long allIdleTimeNanos;

114     volatile long lastReadTime;

124     private volatile int state; // 0 - none, 1 - initialized, 2 - destroyed

152     /**
153      * Creates a new instance firing {@link IdleStateEvent}s.
154      *
155      * @param readerIdleTime
156      *        an {@link IdleStateEvent} whose state is {@link IdleState#READER_IDLE}
157      *        will be triggered when no read was performed for the specified
158      *        period of time.  Specify {@code 0} to disable.
159      * @param writerIdleTime
160      *        an {@link IdleStateEvent} whose state is {@link IdleState#WRITER_IDLE}
161      *        will be triggered when no write was performed for the specified
162      *        period of time.  Specify {@code 0} to disable.
163      * @param allIdleTime
164      *        an {@link IdleStateEvent} whose state is {@link IdleState#ALL_IDLE}
165      *        will be triggered when neither read nor write was performed for
166      *        the specified period of time.  Specify {@code 0} to disable.
167      * @param unit
168      *        the {@link TimeUnit} of {@code readerIdleTime},
169      *        {@code writeIdleTime}, and {@code allIdleTime}
170      */
171     public IdleStateHandler(
172             long readerIdleTime, long writerIdleTime, long allIdleTime,
173             TimeUnit unit) {
174         if (unit == null) {
175             throw new NullPointerException("unit");
176         }
177 
178         if (readerIdleTime <= 0) {
179             readerIdleTimeNanos = 0;
180         } else {
181             readerIdleTimeNanos = Math.max(unit.toNanos(readerIdleTime), MIN_TIMEOUT_NANOS);
182         }
183         if (writerIdleTime <= 0) {
184             writerIdleTimeNanos = 0;
185         } else {
186             writerIdleTimeNanos = Math.max(unit.toNanos(writerIdleTime), MIN_TIMEOUT_NANOS);
187         }
188         if (allIdleTime <= 0) {
189             allIdleTimeNanos = 0;
190         } else {
191             allIdleTimeNanos = Math.max(unit.toNanos(allIdleTime), MIN_TIMEOUT_NANOS);
192         }
193     }

219     @Override
220     public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
221         if (ctx.channel().isActive() && ctx.channel().isRegistered()) {
222             // channelActvie() event has been fired already, which means this.channelActive() will
223             // not be invoked. We have to initialize here instead.
224             initialize(ctx);
225         } else {
226             // channelActive() event has not been fired yet.  this.channelActive() will be invoked
227             // and initialization will occur there.
228         }
229     }
230 
231     @Override
232     public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
233         destroy();
234     }
235 
236     @Override
237     public void channelRegistered(ChannelHandlerContext ctx) throws Exception {
238         // Initialize early if channel is active already.
239         if (ctx.channel().isActive()) {
240             initialize(ctx);
241         }
242         super.channelRegistered(ctx);
243     }
244 
245     @Override
246     public void channelActive(ChannelHandlerContext ctx) throws Exception {
247         // This method will be invoked only if this handler was added
248         // before channelActive() event is fired.  If a user adds this handler
249         // after the channelActive() event, initialize() will be called by beforeAdd().
250         initialize(ctx);
251         super.channelActive(ctx);
252     }

254     @Override
255     public void channelInactive(ChannelHandlerContext ctx) throws Exception {

257         super.channelInactive(ctx);
258     }

260     @Override
261     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
262         if (readerIdleTimeNanos > 0 || allIdleTimeNanos > 0) {
263             reading = true;
264             firstReaderIdleEvent = firstAllIdleEvent = true;
265         }
266         ctx.fireChannelRead(msg);
267     }
268 
269     @Override
270     public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
271         if (readerIdleTimeNanos > 0 || allIdleTimeNanos > 0) {
272             lastReadTime = System.nanoTime();
273             reading = false;
274         }
275         ctx.fireChannelReadComplete();
276     }
277 
278     @Override
279     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
280         // Allow writing with void promise if handler is only configured for read timeout events.
281         if (writerIdleTimeNanos > 0 || allIdleTimeNanos > 0) {
282             ChannelPromise unvoid = promise.unvoid();
283             unvoid.addListener(writeListener);
284             ctx.write(msg, unvoid);
285         } else {
286             ctx.write(msg, promise);
287         }
288     }

290     private void initialize(ChannelHandlerContext ctx) {
291         // Avoid the case where destroy() is called before scheduling timeouts.
292         // See: https://github.com/netty/netty/issues/143
293         switch (state) {
294         case 1:
295         case 2:
296             return;
297         }
298 
299         state = 1;
300 
301         EventExecutor loop = ctx.executor();
302 
303         lastReadTime = lastWriteTime = System.nanoTime();
304         if (readerIdleTimeNanos > 0) {
305             readerIdleTimeout = loop.schedule(
306                     new ReaderIdleTimeoutTask(ctx),
307                     readerIdleTimeNanos, TimeUnit.NANOSECONDS);
308         }
309         if (writerIdleTimeNanos > 0) {
310             writerIdleTimeout = loop.schedule(
311                     new WriterIdleTimeoutTask(ctx),
312                     writerIdleTimeNanos, TimeUnit.NANOSECONDS);
313         }
314         if (allIdleTimeNanos > 0) {
315             allIdleTimeout = loop.schedule(
316                     new AllIdleTimeoutTask(ctx),
317                     allIdleTimeNanos, TimeUnit.NANOSECONDS);
318         }
319     }
320 
321     private void destroy() {
322         state = 2;
323 
324         if (readerIdleTimeout != null) {
325             readerIdleTimeout.cancel(false);
326             readerIdleTimeout = null;
327         }
328         if (writerIdleTimeout != null) {
329             writerIdleTimeout.cancel(false);
330             writerIdleTimeout = null;
331         }
332         if (allIdleTimeout != null) {
333             allIdleTimeout.cancel(false);
334             allIdleTimeout = null;
335         }
336     }

346     /**
347      * Returns a {@link IdleStateEvent}.
348      */
349     protected IdleStateEvent newIdleStateEvent(IdleState state, boolean first) {
350         switch (state) {
351             case ALL_IDLE:
352                 return first ? IdleStateEvent.FIRST_ALL_IDLE_STATE_EVENT : IdleStateEvent.ALL_IDLE_STATE_EVENT;
353             case READER_IDLE:
354                 return first ? IdleStateEvent.FIRST_READER_IDLE_STATE_EVENT : IdleStateEvent.READER_IDLE_STATE_EVENT;
355             case WRITER_IDLE:
356                 return first ? IdleStateEvent.FIRST_WRITER_IDLE_STATE_EVENT : IdleStateEvent.WRITER_IDLE_STATE_EVENT;
357             default:
358                 throw new Error();
359         }
360     }

362     private final class ReaderIdleTimeoutTask implements Runnable {
363 
364         private final ChannelHandlerContext ctx;
365 
366         ReaderIdleTimeoutTask(ChannelHandlerContext ctx) {
367             this.ctx = ctx;
368         }
369 
370         @Override
371         public void run() {
372             if (!ctx.channel().isOpen()) {
373                 return;
374             }
375 
376             long nextDelay = readerIdleTimeNanos;
377             if (!reading) {
378                 nextDelay -= System.nanoTime() - lastReadTime;
379             }
380 
381             if (nextDelay <= 0) {
382                 // Reader is idle - set a new timeout and notify the callback.
383                 readerIdleTimeout =
384                     ctx.executor().schedule(this, readerIdleTimeNanos, TimeUnit.NANOSECONDS);
385                 try {
386                     IdleStateEvent event = newIdleStateEvent(IdleState.READER_IDLE, firstReaderIdleEvent);
387                     if (firstReaderIdleEvent) {
388                         firstReaderIdleEvent = false;
389                     }
390 
391                     channelIdle(ctx, event);
392                 } catch (Throwable t) {
393                     ctx.fireExceptionCaught(t);
394                 }
395             } else {
396                 // Read occurred before the timeout - set a new timeout with shorter delay.
397                 readerIdleTimeout = ctx.executor().schedule(this, nextDelay, TimeUnit.NANOSECONDS);
398             }
399         }
400     }

402     private final class WriterIdleTimeoutTask implements Runnable {
403 
404         private final ChannelHandlerContext ctx;
405 
406         WriterIdleTimeoutTask(ChannelHandlerContext ctx) {
407             this.ctx = ctx;
408         }
409 
410         @Override
411         public void run() {
412             if (!ctx.channel().isOpen()) {
413                 return;
414             }
415 
416             long lastWriteTime = IdleStateHandler.this.lastWriteTime;
417             long nextDelay = writerIdleTimeNanos - (System.nanoTime() - lastWriteTime);
418             if (nextDelay <= 0) {
419                 // Writer is idle - set a new timeout and notify the callback.
420                 writerIdleTimeout = ctx.executor().schedule(
421                         this, writerIdleTimeNanos, TimeUnit.NANOSECONDS);
422                 try {
423                     IdleStateEvent event = newIdleStateEvent(IdleState.WRITER_IDLE, firstWriterIdleEvent);
424                     if (firstWriterIdleEvent) {
425                         firstWriterIdleEvent = false;
426                     }
427 
428                     channelIdle(ctx, event);
429                 } catch (Throwable t) {
430                     ctx.fireExceptionCaught(t);
431                 }
432             } else {
433                 // Write occurred before the timeout - set a new timeout with shorter delay.
434                 writerIdleTimeout = ctx.executor().schedule(this, nextDelay, TimeUnit.NANOSECONDS);
435             }
436         }
437     }

439     private final class AllIdleTimeoutTask implements Runnable {
440 
441         private final ChannelHandlerContext ctx;
442 
443         AllIdleTimeoutTask(ChannelHandlerContext ctx) {
444             this.ctx = ctx;
445         }
446 
447         @Override
448         public void run() {
449             if (!ctx.channel().isOpen()) {
450                 return;
451             }
452 
453             long nextDelay = allIdleTimeNanos;
454             if (!reading) {
455                 nextDelay -= System.nanoTime() - Math.max(lastReadTime, lastWriteTime);
456             }
457             if (nextDelay <= 0) {
458                 // Both reader and writer are idle - set a new timeout and
459                 // notify the callback.
460                 allIdleTimeout = ctx.executor().schedule(
461                         this, allIdleTimeNanos, TimeUnit.NANOSECONDS);
462                 try {
463                     IdleStateEvent event = newIdleStateEvent(IdleState.ALL_IDLE, firstAllIdleEvent);
464                     if (firstAllIdleEvent) {
465                         firstAllIdleEvent = false;
466                     }
467 
468                     channelIdle(ctx, event);
469                 } catch (Throwable t) {
470                     ctx.fireExceptionCaught(t);
471                 }
472             } else {
473                 // Either read or write occurred before the timeout - set a new
474                 // timeout with shorter delay.
475                 allIdleTimeout = ctx.executor().schedule(this, nextDelay, TimeUnit.NANOSECONDS);
476             }
477         }
