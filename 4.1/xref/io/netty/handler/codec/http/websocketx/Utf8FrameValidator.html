16  package io.netty.handler.codec.http.websocketx;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.buffer.Unpooled;
20  import io.netty.channel.ChannelFutureListener;
21  import io.netty.channel.ChannelHandlerContext;
22  import io.netty.channel.ChannelInboundHandlerAdapter;
23  import io.netty.handler.codec.CorruptedFrameException;
24  
25  /**
26   *
27   */
28  public class Utf8FrameValidator extends ChannelInboundHandlerAdapter {
29  
30      private int fragmentedFramesCount;
31      private Utf8Validator utf8Validator;
32  
33      @Override
34      public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
35          if (msg instanceof WebSocketFrame) {
36              WebSocketFrame frame = (WebSocketFrame) msg;
37  
38              // Processing for possible fragmented messages for text and binary
39              // frames
40              if (((WebSocketFrame) msg).isFinalFragment()) {
41                  // Final frame of the sequence. Apparently ping frames are
42                  // allowed in the middle of a fragmented message
43                  if (!(frame instanceof PingWebSocketFrame)) {
44                      fragmentedFramesCount = 0;
45  
46                      // Check text for UTF8 correctness
47                      if ((frame instanceof TextWebSocketFrame) ||
48                              (utf8Validator != null && utf8Validator.isChecking())) {
49                          // Check UTF-8 correctness for this payload
50                          checkUTF8String(ctx, frame.content());
51  
52                          // This does a second check to make sure UTF-8
53                          // correctness for entire text message
54                          utf8Validator.finish();
55                      }
56                  }
57              } else {
58                  // Not final frame so we can expect more frames in the
59                  // fragmented sequence
60                  if (fragmentedFramesCount == 0) {
61                      // First text or binary frame for a fragmented set
62                      if (frame instanceof TextWebSocketFrame) {
63                          checkUTF8String(ctx, frame.content());
64                      }
65                  } else {
66                      // Subsequent frames - only check if init frame is text
67                      if (utf8Validator != null && utf8Validator.isChecking()) {
68                          checkUTF8String(ctx, frame.content());
69                      }
70                  }
71  
72                  // Increment counter
73                  fragmentedFramesCount++;
74              }
75          }
76  
77          super.channelRead(ctx, msg);
78      }
79  
80      private void checkUTF8String(ChannelHandlerContext ctx, ByteBuf buffer) {
81          try {
82              if (utf8Validator == null) {
83                  utf8Validator = new Utf8Validator();
84              }
85              utf8Validator.check(buffer);
86          } catch (CorruptedFrameException ex) {
87              if (ctx.channel().isActive()) {
88                  ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(ChannelFutureListener.CLOSE);
89              }
90          }
91      }
92  
93  }
