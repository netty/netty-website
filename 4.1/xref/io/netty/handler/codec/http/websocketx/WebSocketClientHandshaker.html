216     /**
217      * Begins the opening handshake
218      *
219      * @param channel
220      *            Channel
221      * @param promise
222      *            the {@link ChannelPromise} to be notified when the opening handshake is sent
223      */
224     public final ChannelFuture handshake(Channel channel, final ChannelPromise promise) {
225         FullHttpRequest request =  newHandshakeRequest();
226 
227         HttpResponseDecoder decoder = channel.pipeline().get(HttpResponseDecoder.class);
228         if (decoder == null) {
229             HttpClientCodec codec = channel.pipeline().get(HttpClientCodec.class);
230             if (codec == null) {
231                promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
232                        "a HttpResponseDecoder or HttpClientCodec"));
233                return promise;
234             }
235         }
236 
237         channel.writeAndFlush(request).addListener(new ChannelFutureListener() {
238             @Override
239             public void operationComplete(ChannelFuture future) {
240                 if (future.isSuccess()) {
241                     ChannelPipeline p = future.channel().pipeline();
242                     ChannelHandlerContext ctx = p.context(HttpRequestEncoder.class);
243                     if (ctx == null) {
244                         ctx = p.context(HttpClientCodec.class);
245                     }
246                     if (ctx == null) {
247                         promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
248                                 "a HttpRequestEncoder or HttpClientCodec"));
249                         return;
250                     }
251                     p.addAfter(ctx.name(), "ws-encoder", newWebSocketEncoder());
252 
253                     promise.setSuccess();
254                 } else {
255                     promise.setFailure(future.cause());
256                 }
257             }
258         });
259         return promise;
260     }

267     /**
268      * Validates and finishes the opening handshake initiated by {@link #handshake}}.
269      *
270      * @param channel
271      *            Channel
272      * @param response
273      *            HTTP response containing the closing handshake details
274      */
275     public final void finishHandshake(Channel channel, FullHttpResponse response) {
276         verify(response);
277 
278         // Verify the subprotocol that we received from the server.
279         // This must be one of our expected subprotocols - or null/empty if we didn't want to speak a subprotocol
280         String receivedProtocol = response.headers().get(HttpHeaderNames.SEC_WEBSOCKET_PROTOCOL);
281         receivedProtocol = receivedProtocol != null ? receivedProtocol.trim() : null;
282         String expectedProtocol = expectedSubprotocol != null ? expectedSubprotocol : "";
283         boolean protocolValid = false;
284 
285         if (expectedProtocol.isEmpty() && receivedProtocol == null) {
286             // No subprotocol required and none received
287             protocolValid = true;
288             setActualSubprotocol(expectedSubprotocol); // null or "" - we echo what the user requested
289         } else if (!expectedProtocol.isEmpty() && receivedProtocol != null && !receivedProtocol.isEmpty()) {
290             // We require a subprotocol and received one -> verify it
291             for (String protocol : expectedProtocol.split(",")) {
292                 if (protocol.trim().equals(receivedProtocol)) {
293                     protocolValid = true;
294                     setActualSubprotocol(receivedProtocol);
295                     break;
296                 }
297             }
298         } // else mixed cases - which are all errors
299 
300         if (!protocolValid) {
301             throw new WebSocketHandshakeException(String.format(
302                     "Invalid subprotocol. Actual: %s. Expected one of: %s",
303                     receivedProtocol, expectedSubprotocol));
304         }
305 
306         setHandshakeComplete();
307 
308         final ChannelPipeline p = channel.pipeline();
309         // Remove decompressor from pipeline if its in use
310         HttpContentDecompressor decompressor = p.get(HttpContentDecompressor.class);
311         if (decompressor != null) {
312             p.remove(decompressor);
313         }
314 
315         // Remove aggregator if present before
316         HttpObjectAggregator aggregator = p.get(HttpObjectAggregator.class);
317         if (aggregator != null) {
318             p.remove(aggregator);
319         }
320 
321         ChannelHandlerContext ctx = p.context(HttpResponseDecoder.class);
322         if (ctx == null) {
323             ctx = p.context(HttpClientCodec.class);
324             if (ctx == null) {
325                 throw new IllegalStateException("ChannelPipeline does not contain " +
326                         "a HttpRequestEncoder or HttpClientCodec");
327             }
328             final HttpClientCodec codec =  (HttpClientCodec) ctx.handler();
329             // Remove the encoder part of the codec as the user may start writing frames after this method returns.
330             codec.removeOutboundHandler();
331 
332             p.addAfter(ctx.name(), "ws-decoder", newWebsocketDecoder());
333 
334             // Delay the removal of the decoder so the user can setup the pipeline if needed to handle
335             // WebSocketFrame messages.
336             // See https://github.com/netty/netty/issues/4533
337             channel.eventLoop().execute(new Runnable() {
338                 @Override
339                 public void run() {
340                     p.remove(codec);
341                 }
342             });
343         } else {
344             if (p.get(HttpRequestEncoder.class) != null) {
345                 // Remove the encoder part of the codec as the user may start writing frames after this method returns.
346                 p.remove(HttpRequestEncoder.class);
347             }
348             final ChannelHandlerContext context = ctx;
349             p.addAfter(context.name(), "ws-decoder", newWebsocketDecoder());
350 
351             // Delay the removal of the decoder so the user can setup the pipeline if needed to handle
352             // WebSocketFrame messages.
353             // See https://github.com/netty/netty/issues/4533
354             channel.eventLoop().execute(new Runnable() {
355                 @Override
356                 public void run() {
357                     p.remove(context.handler());
358                 }
359             });
360         }
361     }

377     /**
378      * Process the opening handshake initiated by {@link #handshake}}.
379      *
380      * @param channel
381      *            Channel
382      * @param response
383      *            HTTP response containing the closing handshake details
384      * @param promise
385      *            the {@link ChannelPromise} to notify once the handshake completes.
386      * @return future
387      *            the {@link ChannelFuture}which is notified once the handshake completes.
388      */
389     public final ChannelFuture processHandshake(final Channel channel, HttpResponse response,
390                                                 final ChannelPromise promise) {
391         if (response instanceof FullHttpResponse) {
392             try {
393                 finishHandshake(channel, (FullHttpResponse) response);
394                 promise.setSuccess();
395             } catch (Throwable cause) {
396                 promise.setFailure(cause);
397             }
398         } else {
399             ChannelPipeline p = channel.pipeline();
400             ChannelHandlerContext ctx = p.context(HttpResponseDecoder.class);
401             if (ctx == null) {
402                 ctx = p.context(HttpClientCodec.class);
403                 if (ctx == null) {
404                     return promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
405                             "a HttpResponseDecoder or HttpClientCodec"));
406                 }
407             }
408             // Add aggregator and ensure we feed the HttpResponse so it is aggregated. A limit of 8192 should be more
409             // then enough for the websockets handshake payload.
410             //
411             // TODO: Make handshake work without HttpObjectAggregator at all.
412             String aggregatorName = "httpAggregator";
413             p.addAfter(ctx.name(), aggregatorName, new HttpObjectAggregator(8192));
414             p.addAfter(aggregatorName, "handshaker", new SimpleChannelInboundHandler<FullHttpResponse>() {
415                 @Override
416                 protected void channelRead0(ChannelHandlerContext ctx, FullHttpResponse msg) throws Exception {
417                     // Remove ourself and do the actual handshake
418                     ctx.pipeline().remove(this);
419                     try {
420                         finishHandshake(channel, msg);
421                         promise.setSuccess();
422                     } catch (Throwable cause) {
423                         promise.setFailure(cause);
424                     }
425                 }
426 
427                 @Override
428                 public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
429                     // Remove ourself and fail the handshake promise.
430                     ctx.pipeline().remove(this);
431                     promise.setFailure(cause);
432                 }
433 
434                 @Override
435                 public void channelInactive(ChannelHandlerContext ctx) throws Exception {
436                     // Fail promise if Channel was closed
437                     promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
438                     ctx.fireChannelInactive();
439                 }
440             });
441             try {
442                 ctx.fireChannelRead(ReferenceCountUtil.retain(response));
443             } catch (Throwable cause) {
444                 promise.setFailure(cause);
445             }
446         }
447         return promise;
448     }

499     private void applyForceCloseTimeout(final Channel channel, ChannelFuture flushFuture) {
500         final long forceCloseTimeoutMillis = this.forceCloseTimeoutMillis;
501         final WebSocketClientHandshaker handshaker = this;
502         if (forceCloseTimeoutMillis <= 0 || !channel.isActive() || forceCloseInit != 0) {
503             return;
504         }
505 
506         flushFuture.addListener(new ChannelFutureListener() {
507             @Override
508             public void operationComplete(ChannelFuture future) throws Exception {
509                 // If flush operation failed, there is no reason to expect
510                 // a server to receive CloseFrame. Thus this should be handled
511                 // by the application separately.
512                 // Also, close might be called twice from different threads.
513                 if (future.isSuccess() && channel.isActive() &&
514                         FORCE_CLOSE_INIT_UPDATER.compareAndSet(handshaker, 0, 1)) {
515                     final Future<?> forceCloseFuture = channel.eventLoop().schedule(new Runnable() {
516                         @Override
517                         public void run() {
518                             if (channel.isActive()) {
519                                 channel.close();
520                                 forceCloseComplete = true;
521                             }
522                         }
523                     }, forceCloseTimeoutMillis, TimeUnit.MILLISECONDS);
524 
525                     channel.closeFuture().addListener(new ChannelFutureListener() {
526                         @Override
527                         public void operationComplete(ChannelFuture future) throws Exception {
528                             forceCloseFuture.cancel(false);
529                         }
530                     });
531                 }
532             }
533         });
534     }

549     static CharSequence websocketHostValue(URI wsURL) {
550         int port = wsURL.getPort();
551         if (port == -1) {
552             return wsURL.getHost();
553         }
554         String host = wsURL.getHost();
555         if (port == HttpScheme.HTTP.port()) {
556             return HttpScheme.HTTP.name().contentEquals(wsURL.getScheme())
557                     || WebSocketScheme.WS.name().contentEquals(wsURL.getScheme()) ?
558                     host : NetUtil.toSocketAddressString(host, port);
559         }
560         if (port == HttpScheme.HTTPS.port()) {
561             return HttpScheme.HTTPS.name().contentEquals(wsURL.getScheme())
562                     || WebSocketScheme.WSS.name().contentEquals(wsURL.getScheme()) ?
563                     host : NetUtil.toSocketAddressString(host, port);
564         }
565 
566         // if the port is not standard (80/443) its needed to add the port to the header.
567         // See http://tools.ietf.org/html/rfc6454#section-6.2
568         return NetUtil.toSocketAddressString(host, port);
569     }

571     static CharSequence websocketOriginValue(URI wsURL) {
572         String scheme = wsURL.getScheme();
573         final String schemePrefix;
574         int port = wsURL.getPort();
575         final int defaultPort;
576         if (WebSocketScheme.WSS.name().contentEquals(scheme)
577             || HttpScheme.HTTPS.name().contentEquals(scheme)
578             || (scheme == null && port == WebSocketScheme.WSS.port())) {
579 
580             schemePrefix = HTTPS_SCHEME_PREFIX;
581             defaultPort = WebSocketScheme.WSS.port();
582         } else {
583             schemePrefix = HTTP_SCHEME_PREFIX;
584             defaultPort = WebSocketScheme.WS.port();
585         }
586 
587         // Convert uri-host to lower case (by RFC 6454, chapter 4 "Origin of a URI")
588         String host = wsURL.getHost().toLowerCase(Locale.US);
589 
590         if (port != defaultPort && port != -1) {
591             // if the port is not standard (80/443) its needed to add the port to the header.
592             // See http://tools.ietf.org/html/rfc6454#section-6.2
593             return schemePrefix + NetUtil.toSocketAddressString(host, port);
594         }
595         return schemePrefix + host;
596     }
