35  /**
36   * Decodes {@link ByteBuf}s into {@link HttpMessage}s and
37   * {@link HttpContent}s.
38   *
39   * <h3>Parameters that prevents excessive memory consumption</h3>
40   * <table border="1">
41   * <tr>
42   * <th>Name</th><th>Default value</th><th>Meaning</th>
43   * </tr>
44   * <tr>
45   * <td>{@code maxInitialLineLength}</td>
46   * <td>{@value #DEFAULT_MAX_INITIAL_LINE_LENGTH}</td>
47   * <td>The maximum length of the initial line
48   *     (e.g. {@code "GET / HTTP/1.0"} or {@code "HTTP/1.0 200 OK"})
49   *     If the length of the initial line exceeds this value, a
50   *     {@link TooLongHttpLineException} will be raised.</td>
51   * </tr>
52   * <tr>
53   * <td>{@code maxHeaderSize}</td>
54   * <td>{@value #DEFAULT_MAX_HEADER_SIZE}</td>
55   * <td>The maximum length of all headers.  If the sum of the length of each
56   *     header exceeds this value, a {@link TooLongHttpHeaderException} will be raised.</td>
57   * </tr>
58   * <tr>
59   * <td>{@code maxChunkSize}</td>
60   * <td>{@value #DEFAULT_MAX_CHUNK_SIZE}</td>
61   * <td>The maximum length of the content or each chunk.  If the content length
62   *     (or the length of each chunk) exceeds this value, the content or chunk
63   *     will be split into multiple {@link HttpContent}s whose length is
64   *     {@code maxChunkSize} at maximum.</td>
65   * </tr>
66   * </table>
67   *
68   * <h3>Parameters that control parsing behavior</h3>
69   * <table border="1">
70   * <tr>
71   * <th>Name</th><th>Default value</th><th>Meaning</th>
72   * </tr>
73   * <tr>
74   * <td>{@code allowDuplicateContentLengths}</td>
75   * <td>{@value #DEFAULT_ALLOW_DUPLICATE_CONTENT_LENGTHS}</td>
76   * <td>When set to {@code false}, will reject any messages that contain multiple Content-Length header fields.
77   *     When set to {@code true}, will allow multiple Content-Length headers only if they are all the same decimal value.
78   *     The duplicated field-values will be replaced with a single valid Content-Length field.
79   *     See <a href="https://tools.ietf.org/html/rfc7230#section-3.3.2">RFC 7230, Section 3.3.2</a>.</td>
80   * </tr>
81   * <tr>
82   * <td>{@code allowPartialChunks}</td>
83   * <td>{@value #DEFAULT_ALLOW_PARTIAL_CHUNKS}</td>
84   * <td>If the length of a chunk exceeds the {@link ByteBuf}s readable bytes and {@code allowPartialChunks}
85   *     is set to {@code true}, the chunk will be split into multiple {@link HttpContent}s.
86   *     Otherwise, if the chunk size does not exceed {@code maxChunkSize} and {@code allowPartialChunks}
87   *     is set to {@code false}, the {@link ByteBuf} is not decoded into an {@link HttpContent} until
88   *     the readable bytes are greater or equal to the chunk size.</td>
89   * </tr>
90   * </table>
91   *
92   * <h3>Chunked Content</h3>
93   *
94   * If the content of an HTTP message is greater than {@code maxChunkSize} or
95   * the transfer encoding of the HTTP message is 'chunked', this decoder
96   * generates one {@link HttpMessage} instance and its following
97   * {@link HttpContent}s per single HTTP message to avoid excessive memory
98   * consumption. For example, the following HTTP message:
99   * <pre>
100  * GET / HTTP/1.1
101  * Transfer-Encoding: chunked
102  *
103  * 1a
104  * abcdefghijklmnopqrstuvwxyz
105  * 10
106  * 1234567890abcdef
107  * 0
108  * Content-MD5: ...
109  * <i>[blank line]</i>
110  * </pre>
111  * triggers {@link HttpRequestDecoder} to generate 3 objects:
112  * <ol>
113  * <li>An {@link HttpRequest},</li>
114  * <li>The first {@link HttpContent} whose content is {@code 'abcdefghijklmnopqrstuvwxyz'},</li>
115  * <li>The second {@link LastHttpContent} whose content is {@code '1234567890abcdef'}, which marks
116  * the end of the content.</li>
117  * </ol>
118  *
119  * If you prefer not to handle {@link HttpContent}s by yourself for your
120  * convenience, insert {@link HttpObjectAggregator} after this decoder in the
121  * {@link ChannelPipeline}.  However, please note that your server might not
122  * be as memory efficient as without the aggregator.
123  *
124  * <h3>Extensibility</h3>
125  *
126  * Please note that this decoder is designed to be extended to implement
127  * a protocol derived from HTTP, such as
128  * <a href="https://en.wikipedia.org/wiki/Real_Time_Streaming_Protocol">RTSP</a> and
129  * <a href="https://en.wikipedia.org/wiki/Internet_Content_Adaptation_Protocol">ICAP</a>.
130  * To implement the decoder of such a derived protocol, extend this class and
131  * implement all abstract methods properly.
132  *
133  * <h3>Header Validation</h3>
134  *
135  * It is recommended to always enable header validation.
136  * <p>
137  * Without header validation, your system can become vulnerable to
138  * <a href="https://cwe.mitre.org/data/definitions/113.html">
139  *     CWE-113: Improper Neutralization of CRLF Sequences in HTTP Headers ('HTTP Response Splitting')
140  * </a>.
141  * <p>
142  * This recommendation stands even when both peers in the HTTP exchange are trusted,
143  * as it helps with defence-in-depth.
144  */

334     @Override
335     protected void decode(ChannelHandlerContext ctx, ByteBuf buffer, List<Object> out) throws Exception {
336         if (resetRequested.get()) {
337             resetNow();
338         }
339 
340         switch (currentState) {
341         case SKIP_CONTROL_CHARS:
342             // Fall-through
343         case READ_INITIAL: try {
344             ByteBuf line = lineParser.parse(buffer);
345             if (line == null) {
346                 return;
347             }
348             final String[] initialLine = splitInitialLine(line);
349             assert initialLine.length == 3 : "initialLine::length must be 3";
350 
351             message = createMessage(initialLine);
352             currentState = State.READ_HEADER;
353             // fall-through
354         } catch (Exception e) {
355             out.add(invalidMessage(buffer, e));
356             return;
357         }
358         case READ_HEADER: try {
359             State nextState = readHeaders(buffer);
360             if (nextState == null) {
361                 return;
362             }
363             currentState = nextState;
364             switch (nextState) {
365             case SKIP_CONTROL_CHARS:
366                 // fast-path
367                 // No content is expected.
368                 out.add(message);
369                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
370                 resetNow();
371                 return;
372             case READ_CHUNK_SIZE:
373                 if (!chunkedSupported) {
374                     throw new IllegalArgumentException("Chunked messages not supported");
375                 }
376                 // Chunked encoding - generate HttpMessage first.  HttpChunks will follow.
377                 out.add(message);
378                 return;
379             default:
380                 /*
381                  * RFC 7230, 3.3.3 (https://tools.ietf.org/html/rfc7230#section-3.3.3) states that if a
382                  * request does not have either a transfer-encoding or a content-length header then the message body
383                  * length is 0. However, for a response the body length is the number of octets received prior to the
384                  * server closing the connection. So we treat this as variable length chunked encoding.
385                  */
386                 long contentLength = contentLength();
387                 if (contentLength == 0 || contentLength == -1 && isDecodingRequest()) {
388                     out.add(message);
389                     out.add(LastHttpContent.EMPTY_LAST_CONTENT);
390                     resetNow();
391                     return;
392                 }
393 
394                 assert nextState == State.READ_FIXED_LENGTH_CONTENT ||
395                         nextState == State.READ_VARIABLE_LENGTH_CONTENT;
396 
397                 out.add(message);
398 
399                 if (nextState == State.READ_FIXED_LENGTH_CONTENT) {
400                     // chunkSize will be decreased as the READ_FIXED_LENGTH_CONTENT state reads data chunk by chunk.
401                     chunkSize = contentLength;
402                 }
403 
404                 // We return here, this forces decode to be called again where we will decode the content
405                 return;
406             }
407         } catch (Exception e) {
408             out.add(invalidMessage(buffer, e));
409             return;
410         }
411         case READ_VARIABLE_LENGTH_CONTENT: {
412             // Keep reading data as a chunk until the end of connection is reached.
413             int toRead = Math.min(buffer.readableBytes(), maxChunkSize);
414             if (toRead > 0) {
415                 ByteBuf content = buffer.readRetainedSlice(toRead);
416                 out.add(new DefaultHttpContent(content));
417             }
418             return;
419         }
420         case READ_FIXED_LENGTH_CONTENT: {
421             int readLimit = buffer.readableBytes();
422 
423             // Check if the buffer is readable first as we use the readable byte count
424             // to create the HttpChunk. This is needed as otherwise we may end up with
425             // create an HttpChunk instance that contains an empty buffer and so is
426             // handled like it is the last HttpChunk.
427             //
428             // See https://github.com/netty/netty/issues/433
429             if (readLimit == 0) {
430                 return;
431             }
432 
433             int toRead = Math.min(readLimit, maxChunkSize);
434             if (toRead > chunkSize) {
435                 toRead = (int) chunkSize;
436             }
437             ByteBuf content = buffer.readRetainedSlice(toRead);
438             chunkSize -= toRead;
439 
440             if (chunkSize == 0) {
441                 // Read all content.
442                 out.add(new DefaultLastHttpContent(content, trailersFactory));
443                 resetNow();
444             } else {
445                 out.add(new DefaultHttpContent(content));
446             }
447             return;
448         }
449         /*
450          * everything else after this point takes care of reading chunked content. basically, read chunk size,
451          * read chunk, read and ignore the CRLF and repeat until 0
452          */
453         case READ_CHUNK_SIZE: try {
454             ByteBuf line = lineParser.parse(buffer);
455             if (line == null) {
456                 return;
457             }
458             int chunkSize = getChunkSize(line.array(), line.arrayOffset() + line.readerIndex(), line.readableBytes());
459             this.chunkSize = chunkSize;
460             if (chunkSize == 0) {
461                 currentState = State.READ_CHUNK_FOOTER;
462                 return;
463             }
464             currentState = State.READ_CHUNKED_CONTENT;
465             // fall-through
466         } catch (Exception e) {
467             out.add(invalidChunk(buffer, e));
468             return;
469         }
470         case READ_CHUNKED_CONTENT: {
471             assert chunkSize <= Integer.MAX_VALUE;
472             int toRead = Math.min((int) chunkSize, maxChunkSize);
473             if (!allowPartialChunks && buffer.readableBytes() < toRead) {
474                 return;
475             }
476             toRead = Math.min(toRead, buffer.readableBytes());
477             if (toRead == 0) {
478                 return;
479             }
480             HttpContent chunk = new DefaultHttpContent(buffer.readRetainedSlice(toRead));
481             chunkSize -= toRead;
482 
483             out.add(chunk);
484 
485             if (chunkSize != 0) {
486                 return;
487             }
488             currentState = State.READ_CHUNK_DELIMITER;
489             // fall-through
490         }
491         case READ_CHUNK_DELIMITER: {
492             final int wIdx = buffer.writerIndex();
493             int rIdx = buffer.readerIndex();
494             while (wIdx > rIdx) {
495                 byte next = buffer.getByte(rIdx++);
496                 if (next == HttpConstants.LF) {
497                     currentState = State.READ_CHUNK_SIZE;
498                     break;
499                 }
500             }
501             buffer.readerIndex(rIdx);
502             return;
503         }
504         case READ_CHUNK_FOOTER: try {
505             LastHttpContent trailer = readTrailingHeaders(buffer);
506             if (trailer == null) {
507                 return;
508             }
509             out.add(trailer);
510             resetNow();
511             return;
512         } catch (Exception e) {
513             out.add(invalidChunk(buffer, e));
514             return;
515         }
516         case BAD_MESSAGE: {
517             // Keep discarding until disconnection.
518             buffer.skipBytes(buffer.readableBytes());
519             break;
520         }
521         case UPGRADED: {
522             int readableBytes = buffer.readableBytes();
523             if (readableBytes > 0) {
524                 // Keep on consuming as otherwise we may trigger an DecoderException,
525                 // other handler will replace this codec with the upgraded protocol codec to
526                 // take the traffic over at some point then.
527                 // See https://github.com/netty/netty/issues/2173
528                 out.add(buffer.readBytes(readableBytes));
529             }
530             break;
531         }
532         default:
533             break;
534         }
535     }

537     @Override
538     protected void decodeLast(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
539         super.decodeLast(ctx, in, out);
540 
541         if (resetRequested.get()) {
542             // If a reset was requested by decodeLast() we need to do it now otherwise we may produce a
543             // LastHttpContent while there was already one.
544             resetNow();
545         }
546         // Handle the last unfinished message.
547         if (message != null) {
548             boolean chunked = HttpUtil.isTransferEncodingChunked(message);
549             if (currentState == State.READ_VARIABLE_LENGTH_CONTENT && !in.isReadable() && !chunked) {
550                 // End of connection.
551                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
552                 resetNow();
553                 return;
554             }
555 
556             if (currentState == State.READ_HEADER) {
557                 // If we are still in the state of reading headers we need to create a new invalid message that
558                 // signals that the connection was closed before we received the headers.
559                 out.add(invalidMessage(Unpooled.EMPTY_BUFFER,
560                         new PrematureChannelClosureException("Connection closed before received headers")));
561                 resetNow();
562                 return;
563             }
564 
565             // Check if the closure of the connection signifies the end of the content.
566             boolean prematureClosure;
567             if (isDecodingRequest() || chunked) {
568                 // The last request did not wait for a response.
569                 prematureClosure = true;
570             } else {
571                 // Compare the length of the received content and the 'Content-Length' header.
572                 // If the 'Content-Length' header is absent, the length of the content is determined by the end of the
573                 // connection, so it is perfectly fine.
574                 prematureClosure = contentLength() > 0;
575             }
576 
577             if (!prematureClosure) {
578                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
579             }
580             resetNow();
581         }
582     }

584     @Override
585     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
586         if (evt instanceof HttpExpectationFailedEvent) {
587             switch (currentState) {
588             case READ_FIXED_LENGTH_CONTENT:
589             case READ_VARIABLE_LENGTH_CONTENT:
590             case READ_CHUNK_SIZE:
591                 reset();
592                 break;
593             default:
594                 break;
595             }
596         }
597         super.userEventTriggered(ctx, evt);
598     }

600     protected boolean isContentAlwaysEmpty(HttpMessage msg) {
601         if (msg instanceof HttpResponse) {
602             HttpResponse res = (HttpResponse) msg;
603             final HttpResponseStatus status = res.status();
604             final int code = status.code();
605             final HttpStatusClass statusClass = status.codeClass();
606 
607             // Correctly handle return codes of 1xx.
608             //
609             // See:
610             //     - https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html Section 4.4
611             //     - https://github.com/netty/netty/issues/222
612             if (statusClass == HttpStatusClass.INFORMATIONAL) {
613                 // One exception: Hixie 76 websocket handshake response
614                 return !(code == 101 && !res.headers().contains(HttpHeaderNames.SEC_WEBSOCKET_ACCEPT)
615                          && res.headers().contains(HttpHeaderNames.UPGRADE, HttpHeaderValues.WEBSOCKET, true));
616             }
617 
618             switch (code) {
619             case 204: case 304:
620                 return true;
621             default:
622                 return false;
623             }
624         }
625         return false;
626     }

628     /**
629      * Returns true if the server switched to a different protocol than HTTP/1.0 or HTTP/1.1, e.g. HTTP/2 or Websocket.
630      * Returns false if the upgrade happened in a different layer, e.g. upgrade from HTTP/1.1 to HTTP/1.1 over TLS.
631      */
632     protected boolean isSwitchingToNonHttp1Protocol(HttpResponse msg) {
633         if (msg.status().code() != HttpResponseStatus.SWITCHING_PROTOCOLS.code()) {
634             return false;
635         }
636         String newProtocol = msg.headers().get(HttpHeaderNames.UPGRADE);
637         return newProtocol == null ||
638                 !newProtocol.contains(HttpVersion.HTTP_1_0.text()) &&
639                 !newProtocol.contains(HttpVersion.HTTP_1_1.text());
640     }

650     private void resetNow() {
651         HttpMessage message = this.message;
652         this.message = null;
653         name = null;
654         value = null;
655         contentLength = Long.MIN_VALUE;
656         lineParser.reset();
657         headerParser.reset();
658         trailer = null;
659         if (!isDecodingRequest()) {
660             HttpResponse res = (HttpResponse) message;
661             if (res != null && isSwitchingToNonHttp1Protocol(res)) {
662                 currentState = State.UPGRADED;
663                 return;
664             }
665         }
666 
667         resetRequested.lazySet(false);
668         currentState = State.SKIP_CONTROL_CHARS;
669     }

702     private State readHeaders(ByteBuf buffer) {
703         final HttpMessage message = this.message;
704         final HttpHeaders headers = message.headers();
705 
706         final HeaderParser headerParser = this.headerParser;
707 
708         ByteBuf line = headerParser.parse(buffer);
709         if (line == null) {
710             return null;
711         }
712         int lineLength = line.readableBytes();
713         while (lineLength > 0) {
714             final byte[] lineContent = line.array();
715             final int startLine = line.arrayOffset() + line.readerIndex();
716             final byte firstChar = lineContent[startLine];
717             if (name != null && (firstChar == ' ' || firstChar == '\t')) {
718                 //please do not make one line from below code
719                 //as it breaks +XX:OptimizeStringConcat optimization
720                 String trimmedLine = langAsciiString(lineContent, startLine, lineLength).trim();
721                 String valueStr = value;
722                 value = valueStr + ' ' + trimmedLine;
723             } else {
724                 if (name != null) {
725                     headers.add(name, value);
726                 }
727                 splitHeader(lineContent, startLine, lineLength);
728             }
729 
730             line = headerParser.parse(buffer);
731             if (line == null) {
732                 return null;
733             }
734             lineLength = line.readableBytes();
735         }
736 
737         // Add the last header.
738         if (name != null) {
739             headers.add(name, value);
740         }
741 
742         // reset name and value fields
743         name = null;
744         value = null;
745 
746         // Done parsing initial line and headers. Set decoder result.
747         HttpMessageDecoderResult decoderResult = new HttpMessageDecoderResult(lineParser.size, headerParser.size);
748         message.setDecoderResult(decoderResult);
749 
750         List<String> contentLengthFields = headers.getAll(HttpHeaderNames.CONTENT_LENGTH);
751         if (!contentLengthFields.isEmpty()) {
752             HttpVersion version = message.protocolVersion();
753             boolean isHttp10OrEarlier = version.majorVersion() < 1 || (version.majorVersion() == 1
754                     && version.minorVersion() == 0);
755             // Guard against multiple Content-Length headers as stated in
756             // https://tools.ietf.org/html/rfc7230#section-3.3.2:
757             contentLength = HttpUtil.normalizeAndGetContentLength(contentLengthFields,
758                     isHttp10OrEarlier, allowDuplicateContentLengths);
759             if (contentLength != -1) {
760                 String lengthValue = contentLengthFields.get(0).trim();
761                 if (contentLengthFields.size() > 1 || // don't unnecessarily re-order headers
762                         !lengthValue.equals(Long.toString(contentLength))) {
763                     headers.set(HttpHeaderNames.CONTENT_LENGTH, contentLength);
764                 }
765             }
766         }
767 
768         if (isContentAlwaysEmpty(message)) {
769             HttpUtil.setTransferEncodingChunked(message, false);
770             return State.SKIP_CONTROL_CHARS;
771         } else if (HttpUtil.isTransferEncodingChunked(message)) {
772             if (!contentLengthFields.isEmpty() && message.protocolVersion() == HttpVersion.HTTP_1_1) {
773                 handleTransferEncodingChunkedWithContentLength(message);
774             }
775             return State.READ_CHUNK_SIZE;
776         } else if (contentLength() >= 0) {
777             return State.READ_FIXED_LENGTH_CONTENT;
778         } else {
779             return State.READ_VARIABLE_LENGTH_CONTENT;
780         }
781     }

783     /**
784      * Invoked when a message with both a "Transfer-Encoding: chunked" and a "Content-Length" header field is detected.
785      * The default behavior is to <i>remove</i> the Content-Length field, but this method could be overridden
786      * to change the behavior (to, e.g., throw an exception and produce an invalid message).
787      * <p>
788      * See: https://tools.ietf.org/html/rfc7230#section-3.3.3
789      * <pre>
790      *     If a message is received with both a Transfer-Encoding and a
791      *     Content-Length header field, the Transfer-Encoding overrides the
792      *     Content-Length.  Such a message might indicate an attempt to
793      *     perform request smuggling (Section 9.5) or response splitting
794      *     (Section 9.4) and ought to be handled as an error.  A sender MUST
795      *     remove the received Content-Length field prior to forwarding such
796      *     a message downstream.
797      * </pre>
798      * Also see:
799      * https://github.com/apache/tomcat/blob/b693d7c1981fa7f51e58bc8c8e72e3fe80b7b773/
800      * java/org/apache/coyote/http11/Http11Processor.java#L747-L755
801      * https://github.com/nginx/nginx/blob/0ad4393e30c119d250415cb769e3d8bc8dce5186/
802      * src/http/ngx_http_request.c#L1946-L1953
803      */
804     protected void handleTransferEncodingChunkedWithContentLength(HttpMessage message) {
805         message.headers().remove(HttpHeaderNames.CONTENT_LENGTH);
806         contentLength = Long.MIN_VALUE;
807     }

816     private LastHttpContent readTrailingHeaders(ByteBuf buffer) {
817         final HeaderParser headerParser = this.headerParser;
818         ByteBuf line = headerParser.parse(buffer);
819         if (line == null) {
820             return null;
821         }
822         LastHttpContent trailer = this.trailer;
823         int lineLength = line.readableBytes();
824         if (lineLength == 0 && trailer == null) {
825             // We have received the empty line which signals the trailer is complete and did not parse any trailers
826             // before. Just return an empty last content to reduce allocations.
827             return LastHttpContent.EMPTY_LAST_CONTENT;
828         }
829 
830         CharSequence lastHeader = null;
831         if (trailer == null) {
832             trailer = this.trailer = new DefaultLastHttpContent(Unpooled.EMPTY_BUFFER, trailersFactory);
833         }
834         while (lineLength > 0) {
835             final byte[] lineContent = line.array();
836             final int startLine = line.arrayOffset() + line.readerIndex();
837             final byte firstChar = lineContent[startLine];
838             if (lastHeader != null && (firstChar == ' ' || firstChar == '\t')) {
839                 List<String> current = trailer.trailingHeaders().getAll(lastHeader);
840                 if (!current.isEmpty()) {
841                     int lastPos = current.size() - 1;
842                     //please do not make one line from below code
843                     //as it breaks +XX:OptimizeStringConcat optimization
844                     String lineTrimmed = langAsciiString(lineContent, startLine, line.readableBytes()).trim();
845                     String currentLastPos = current.get(lastPos);
846                     current.set(lastPos, currentLastPos + lineTrimmed);
847                 }
848             } else {
849                 splitHeader(lineContent, startLine, lineLength);
850                 AsciiString headerName = name;
851                 if (!HttpHeaderNames.CONTENT_LENGTH.contentEqualsIgnoreCase(headerName) &&
852                         !HttpHeaderNames.TRANSFER_ENCODING.contentEqualsIgnoreCase(headerName) &&
853                         !HttpHeaderNames.TRAILER.contentEqualsIgnoreCase(headerName)) {
854                     trailer.trailingHeaders().add(headerName, value);
855                 }
856                 lastHeader = name;
857                 // reset name and value fields
858                 name = null;
859                 value = null;
860             }
861             line = headerParser.parse(buffer);
862             if (line == null) {
863                 return null;
864             }
865             lineLength = line.readableBytes();
866         }
867 
868         this.trailer = null;
869         return trailer;
870     }

888     private static int getChunkSize(byte[] hex, int start, int length) {
889         // trim the leading bytes if white spaces, if any
890         final int skipped = skipWhiteSpaces(hex, start, length);
891         if (skipped == length) {
892             // empty case
893             throw new NumberFormatException();
894         }
895         start += skipped;
896         length -= skipped;
897         int result = 0;
898         for (int i = 0; i < length; i++) {
899             final int digit = StringUtil.decodeHexNibble(hex[start + i]);
900             if (digit == -1) {
901                 // uncommon path
902                 final byte b = hex[start + i];
903                 if (b == ';' || isControlOrWhitespaceAsciiChar(b)) {
904                     if (i == 0) {
905                         // empty case
906                         throw new NumberFormatException("Empty chunk size");
907                     }
908                     return result;
909                 }
910                 // non-hex char fail-fast path
911                 throw new NumberFormatException("Invalid character in chunk size");
912             }
913             result *= 16;
914             result += digit;
915             if (result < 0) {
916                 throw new NumberFormatException("Chunk size overflow: " + result);
917             }
918         }
919         return result;
920     }

922     private String[] splitInitialLine(ByteBuf asciiBuffer) {
923         final byte[] asciiBytes = asciiBuffer.array();
924 
925         final int arrayOffset = asciiBuffer.arrayOffset();
926 
927         final int startContent = arrayOffset + asciiBuffer.readerIndex();
928 
929         final int end = startContent + asciiBuffer.readableBytes();
930 
931         final int aStart = findNonSPLenient(asciiBytes, startContent, end);
932         final int aEnd = findSPLenient(asciiBytes, aStart, end);
933 
934         final int bStart = findNonSPLenient(asciiBytes, aEnd, end);
935         final int bEnd = findSPLenient(asciiBytes, bStart, end);
936 
937         final int cStart = findNonSPLenient(asciiBytes, bEnd, end);
938         final int cEnd = findEndOfString(asciiBytes, Math.max(cStart - 1, startContent), end);
939 
940         return new String[]{
941                 splitFirstWordInitialLine(asciiBytes, aStart, aEnd - aStart),
942                 splitSecondWordInitialLine(asciiBytes, bStart, bEnd - bStart),
943                 cStart < cEnd ? splitThirdWordInitialLine(asciiBytes, cStart, cEnd - cStart) : StringUtil.EMPTY_STRING};
944     }

961     private static String langAsciiString(final byte[] asciiContent, int start, int length) {
962         if (length == 0) {
963             return StringUtil.EMPTY_STRING;
964         }
965         // DON'T REMOVE: it helps JIT to use a simpler intrinsic stub for System::arrayCopy based on the call-site
966         if (start == 0) {
967             if (length == asciiContent.length) {
968                 return new String(asciiContent, 0, 0, asciiContent.length);
969             }
970             return new String(asciiContent, 0, 0, length);
971         }
972         return new String(asciiContent, 0, start, length);

975     private void splitHeader(byte[] line, int start, int length) {
976         final int end = start + length;
977         int nameEnd;
978         final int nameStart = findNonWhitespace(line, start, end);
979         // hoist this load out of the loop, because it won't change!
980         final boolean isDecodingRequest = isDecodingRequest();
981         for (nameEnd = nameStart; nameEnd < end; nameEnd ++) {
982             byte ch = line[nameEnd];
983             // https://tools.ietf.org/html/rfc7230#section-3.2.4
984             //
985             // No whitespace is allowed between the header field-name and colon. In
986             // the past, differences in the handling of such whitespace have led to
987             // security vulnerabilities in request routing and response handling. A
988             // server MUST reject any received request message that contains
989             // whitespace between a header field-name and colon with a response code
990             // of 400 (Bad Request). A proxy MUST remove any such whitespace from a
991             // response message before forwarding the message downstream.
992             if (ch == ':' ||
993                     // In case of decoding a request we will just continue processing and header validation
994                     // is done in the DefaultHttpHeaders implementation.
995                     //
996                     // In the case of decoding a response we will "skip" the whitespace.
997                     (!isDecodingRequest && isOWS(ch))) {
998                 break;
999             }
1000         }
1001 
1002         if (nameEnd == end) {
1003             // There was no colon present at all.
1004             throw new IllegalArgumentException("No colon found");
1005         }
1006         int colonEnd;
1007         for (colonEnd = nameEnd; colonEnd < end; colonEnd ++) {
1008             if (line[colonEnd] == ':') {
1009                 colonEnd ++;
1010                 break;
1011             }
1012         }
1013         name = splitHeaderName(line, nameStart, nameEnd - nameStart);
1014         final int valueStart = findNonWhitespace(line, colonEnd, end);
1015         if (valueStart == end) {
1016             value = StringUtil.EMPTY_STRING;
1017         } else {
1018             final int valueEnd = findEndOfString(line, start, end);
1019             // no need to make uses of the ByteBuf's toString ASCII method here, and risk to get JIT confused
1020             value = langAsciiString(line, valueStart, valueEnd - valueStart);
1021         }
1022     }

1028     private static int findNonSPLenient(byte[] sb, int offset, int end) {
1029         for (int result = offset; result < end; ++result) {
1030             byte c = sb[result];
1031             // See https://tools.ietf.org/html/rfc7230#section-3.5
1032             if (isSPLenient(c)) {
1033                 continue;
1034             }
1035             if (isWhitespace(c)) {
1036                 // Any other whitespace delimiter is invalid
1037                 throw new IllegalArgumentException("Invalid separator");
1038             }
1039             return result;
1040         }
1041         return end;
1042     }

1044     private static int findSPLenient(byte[] sb, int offset, int end) {
1045         for (int result = offset; result < end; ++result) {
1046             if (isSPLenient(sb[result])) {
1047                 return result;
1048             }
1049         }
1050         return end;
1051     }

1054     private static final boolean[] LATIN_WHITESPACE;

1056     static {
1057         // See https://tools.ietf.org/html/rfc7230#section-3.5
1058         SP_LENIENT_BYTES = new boolean[256];
1059         SP_LENIENT_BYTES[128 + ' '] = true;
1060         SP_LENIENT_BYTES[128 + 0x09] = true;
1061         SP_LENIENT_BYTES[128 + 0x0B] = true;
1062         SP_LENIENT_BYTES[128 + 0x0C] = true;
1063         SP_LENIENT_BYTES[128 + 0x0D] = true;
1064         // TO SAVE PERFORMING Character::isWhitespace ceremony
1065         LATIN_WHITESPACE = new boolean[256];
1066         for (byte b = Byte.MIN_VALUE; b < Byte.MAX_VALUE; b++) {
1067             LATIN_WHITESPACE[128 + b] = Character.isWhitespace(b);
1068         }
1069     }

1071     private static boolean isSPLenient(byte c) {
1072         // See https://tools.ietf.org/html/rfc7230#section-3.5
1073         return SP_LENIENT_BYTES[c + 128];
1074     }

1076     private static boolean isWhitespace(byte b) {
1077         return LATIN_WHITESPACE[b + 128];
1078     }

1080     private static int findNonWhitespace(byte[] sb, int offset, int end) {
1081         for (int result = offset; result < end; ++result) {
1082             byte c = sb[result];
1083             if (!isWhitespace(c)) {
1084                 return result;
1085             } else if (!isOWS(c)) {
1086                 // Only OWS is supported for whitespace
1087                 throw new IllegalArgumentException("Invalid separator, only a single space or horizontal tab allowed," +
1088                         " but received a '" + c + "' (0x" + Integer.toHexString(c) + ")");
1089             }
1090         }
1091         return end;
1092     }
1093 
1094     private static int findEndOfString(byte[] sb, int start, int end) {
1095         for (int result = end - 1; result > start; --result) {
1096             if (!isWhitespace(sb[result])) {
1097                 return result + 1;
1098             }
1099         }
1100         return 0;
1101     }

1107     private static class HeaderParser {
1108         protected final ByteBuf seq;
1109         protected final int maxLength;
1110         int size;
1111 
1112         HeaderParser(ByteBuf seq, int maxLength) {
1113             this.seq = seq;
1114             this.maxLength = maxLength;
1115         }
1116 
1117         public ByteBuf parse(ByteBuf buffer) {
1118             final int readableBytes = buffer.readableBytes();
1119             final int readerIndex = buffer.readerIndex();
1120             final int maxBodySize = maxLength - size;
1121             assert maxBodySize >= 0;
1122             // adding 2 to account for both CR (if present) and LF
1123             // don't remove 2L: it's key to cover maxLength = Integer.MAX_VALUE
1124             final long maxBodySizeWithCRLF = maxBodySize + 2L;
1125             final int toProcess = (int) Math.min(maxBodySizeWithCRLF, readableBytes);
1126             final int toIndexExclusive = readerIndex + toProcess;
1127             assert toIndexExclusive >= readerIndex;
1128             final int indexOfLf = buffer.indexOf(readerIndex, toIndexExclusive, HttpConstants.LF);
1129             if (indexOfLf == -1) {
1130                 if (readableBytes > maxBodySize) {
1131                     // TODO: Respond with Bad Request and discard the traffic
1132                     //    or close the connection.
1133                     //       No need to notify the upstream handlers - just log.
1134                     //       If decoding a response, just throw an exception.
1135                     throw newException(maxLength);
1136                 }
1137                 return null;
1138             }
1139             final int endOfSeqIncluded;
1140             if (indexOfLf > readerIndex && buffer.getByte(indexOfLf - 1) == HttpConstants.CR) {
1141                 // Drop CR if we had a CRLF pair
1142                 endOfSeqIncluded = indexOfLf - 1;
1143             } else {
1144                 endOfSeqIncluded = indexOfLf;
1145             }
1146             final int newSize = endOfSeqIncluded - readerIndex;
1147             if (newSize == 0) {
1148                 seq.clear();
1149                 buffer.readerIndex(indexOfLf + 1);
1150                 return seq;
1151             }
1152             int size = this.size + newSize;
1153             if (size > maxLength) {
1154                 throw newException(maxLength);
1155             }
1156             this.size = size;
1157             seq.clear();
1158             seq.writeBytes(buffer, readerIndex, newSize);
1159             buffer.readerIndex(indexOfLf + 1);
1160             return seq;
1161         }
1162 
1163         public void reset() {
1164             size = 0;
1165         }
1166 
1167         protected TooLongFrameException newException(int maxLength) {
1168             return new TooLongHttpHeaderException("HTTP header is larger than " + maxLength + " bytes.");
1169         }
1170     }

1175             super(seq, maxLength);

1179         public ByteBuf parse(ByteBuf buffer) {
1180             // Suppress a warning because HeaderParser.reset() is supposed to be called
1181             reset();
1182             final int readableBytes = buffer.readableBytes();
1183             if (readableBytes == 0) {
1184                 return null;
1185             }

1188                 return null;
1189             }

1193         private boolean skipControlChars(ByteBuf buffer, int readableBytes, int readerIndex) {
1194             assert currentState == State.SKIP_CONTROL_CHARS;
1195             final int maxToSkip = Math.min(maxLength, readableBytes);
1196             final int firstNonControlIndex = buffer.forEachByte(readerIndex, maxToSkip, SKIP_CONTROL_CHARS_BYTES);
1197             if (firstNonControlIndex == -1) {
1198                 buffer.skipBytes(maxToSkip);
1199                 if (readableBytes > maxLength) {
1200                     throw newException(maxLength);
1201                 }
1202                 return true;
1203             }
1204             // from now on we don't care about control chars
1205             buffer.readerIndex(firstNonControlIndex);
1206             currentState = State.READ_INITIAL;
1207             return false;
1208         }

1211         protected TooLongFrameException newException(int maxLength) {
1212             return new TooLongHttpLineException("An HTTP line is larger than " + maxLength + " bytes.");
1213         }
