18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelPipeline;
20  import io.netty.util.AsciiString;

22  /**
23   * Decodes {@link ByteBuf}s into {@link HttpRequest}s and {@link HttpContent}s.
24   *
25   * <h3>Parameters that prevents excessive memory consumption</h3>
26   * <table border="1">
27   * <tr>
28   * <th>Name</th><th>Meaning</th>
29   * </tr>
30   * <tr>
31   * <td>{@code maxInitialLineLength}</td>
32   * <td>The maximum length of the initial line (e.g. {@code "GET / HTTP/1.0"})
33   *     If the length of the initial line exceeds this value, a
34   *     {@link TooLongHttpLineException} will be raised.</td>
35   * </tr>
36   * <tr>
37   * <td>{@code maxHeaderSize}</td>
38   * <td>The maximum length of all headers.  If the sum of the length of each
39   *     header exceeds this value, a {@link TooLongHttpHeaderException} will be raised.</td>
40   * </tr>
41   * <tr>
42   * <td>{@code maxChunkSize}</td>
43   * <td>The maximum length of the content or each chunk.  If the content length
44   *     exceeds this value, the transfer encoding of the decoded request will be
45   *     converted to 'chunked' and the content will be split into multiple
46   *     {@link HttpContent}s.  If the transfer encoding of the HTTP request is
47   *     'chunked' already, each chunk will be split into smaller chunks if the
48   *     length of the chunk exceeds this value.  If you prefer not to handle
49   *     {@link HttpContent}s in your handler, insert {@link HttpObjectAggregator}
50   *     after this decoder in the {@link ChannelPipeline}.</td>
51   * </tr>
52   * </table>
53   *
54   * <h3>Parameters that control parsing behavior</h3>
55   * <table border="1">
56   * <tr>
57   * <th>Name</th><th>Default value</th><th>Meaning</th>
58   * </tr>
59   * <tr>
60   * <td>{@code allowDuplicateContentLengths}</td>
61   * <td>{@value #DEFAULT_ALLOW_DUPLICATE_CONTENT_LENGTHS}</td>
62   * <td>When set to {@code false}, will reject any messages that contain multiple Content-Length header fields.
63   *     When set to {@code true}, will allow multiple Content-Length headers only if they are all the same decimal value.
64   *     The duplicated field-values will be replaced with a single valid Content-Length field.
65   *     See <a href="https://tools.ietf.org/html/rfc7230#section-3.3.2">RFC 7230, Section 3.3.2</a>.</td>
66   * </tr>
67   * <tr>
68   * <td>{@code allowPartialChunks}</td>
69   * <td>{@value #DEFAULT_ALLOW_PARTIAL_CHUNKS}</td>
70   * <td>If the length of a chunk exceeds the {@link ByteBuf}s readable bytes and {@code allowPartialChunks}
71   *     is set to {@code true}, the chunk will be split into multiple {@link HttpContent}s.
72   *     Otherwise, if the chunk size does not exceed {@code maxChunkSize} and {@code allowPartialChunks}
73   *     is set to {@code false}, the {@link ByteBuf} is not decoded into an {@link HttpContent} until
74   *     the readable bytes are greater or equal to the chunk size.</td>
75   * </tr>
76   * </table>
77   */
78  public class HttpRequestDecoder extends HttpObjectDecoder {
79  
80      private static final AsciiString Host = AsciiString.cached("Host");
81      private static final AsciiString Connection = AsciiString.cached("Connection");
82      private static final AsciiString ContentType = AsciiString.cached("Content-Type");
83      private static final AsciiString ContentLength = AsciiString.cached("Content-Length");
84  
85      private static final int GET_AS_INT = 'G' | 'E' << 8 | 'T' << 16;
86      private static final int POST_AS_INT = 'P' | 'O' << 8 | 'S' << 16 | 'T' << 24;
87      private static final long HTTP_1_1_AS_LONG = 'H' | 'T' << 8 | 'T' << 16 | 'P' << 24 | (long) '/' << 32 |
88              (long) '1' << 40 | (long) '.' << 48 | (long) '1' << 56;
89  
90      private static final long HTTP_1_0_AS_LONG = 'H' | 'T' << 8 | 'T' << 16 | 'P' << 24 | (long) '/' << 32 |
91              (long) '1' << 40 | (long) '.' << 48 | (long) '0' << 56;
92  
93      private static final int HOST_AS_INT = 'H' | 'o' << 8 | 's' << 16 | 't' << 24;
94  
95      private static final long CONNECTION_AS_LONG_0 = 'C' | 'o' << 8 | 'n' << 16 | 'n' << 24 |
96              (long) 'e' << 32 | (long) 'c' << 40 | (long) 't' << 48 | (long) 'i' << 56;
97  
98      private static final short CONNECTION_AS_SHORT_1 = 'o' | 'n' << 8;
99  
100     private static final long CONTENT_AS_LONG = 'C' | 'o' << 8 | 'n' << 16 | 't' << 24 |
101             (long) 'e' << 32 | (long) 'n' << 40 | (long) 't' << 48 | (long) '-' << 56;
102 
103     private static final int TYPE_AS_INT = 'T' | 'y' << 8 | 'p' << 16 | 'e' << 24;
104 
105     private static final long LENGTH_AS_LONG = 'L' | 'e' << 8 | 'n' << 16 | 'g' << 24 |
106             (long) 't' << 32 | (long) 'h' << 40;
107 
108     /**

110      * {@code maxInitialLineLength (4096)}, {@code maxHeaderSize (8192)}, and
111      * {@code maxChunkSize (8192)}.
112      */

114     }

144             int maxInitialLineLength, int maxHeaderSize, int maxChunkSize, boolean validateHeaders,
145             int initialBufferSize, boolean allowDuplicateContentLengths, boolean allowPartialChunks) {

150     @Override
151     protected HttpMessage createMessage(String[] initialLine) throws Exception {
152         return new DefaultHttpRequest(
153                 HttpVersion.valueOf(initialLine[2]),
154                 HttpMethod.valueOf(initialLine[0]), initialLine[1], validateHeaders);
155     }
156 
157     @Override
158     protected AsciiString splitHeaderName(final byte[] sb, final int start, final int length) {
159         final byte firstChar = sb[start];
160         if (firstChar == 'H' && length == 4) {
161             if (isHost(sb, start)) {
162                 return Host;
163             }
164         } else if (firstChar == 'C') {
165             if (length == 10) {
166                 if (isConnection(sb, start)) {
167                     return Connection;
168                 }
169             } else if (length == 12) {
170                 if (isContentType(sb, start)) {
171                     return ContentType;
172                 }
173             } else if (length == 14) {
174                 if (isContentLength(sb, start)) {
175                     return ContentLength;
176                 }
177             }
178         }
179         return super.splitHeaderName(sb, start, length);
180     }
181 
182     private static boolean isHost(byte[] sb, int start) {
183         final int maybeHost = sb[start] |
184                 sb[start + 1] << 8 |
185                 sb[start + 2] << 16 |
186                 sb[start + 3] << 24;
187         return maybeHost == HOST_AS_INT;
188     }
189 
190     private static boolean isConnection(byte[] sb, int start) {
191         final long maybeConnecti = sb[start] |
192                 sb[start + 1] << 8 |
193                 sb[start + 2] << 16 |
194                 sb[start + 3] << 24 |
195                 (long) sb[start + 4] << 32 |
196                 (long) sb[start + 5] << 40 |
197                 (long) sb[start + 6] << 48 |
198                 (long) sb[start + 7] << 56;
199         if (maybeConnecti != CONNECTION_AS_LONG_0) {
200             return false;
201         }
202         final short maybeOn = (short) (sb[start + 8] | sb[start + 9] << 8);
203         return maybeOn == CONNECTION_AS_SHORT_1;
204     }
205 
206     private static boolean isContentType(byte[] sb, int start) {
207         final long maybeContent = sb[start] |
208                 sb[start + 1] << 8 |
209                 sb[start + 2] << 16 |
210                 sb[start + 3] << 24 |
211                 (long) sb[start + 4] << 32 |
212                 (long) sb[start + 5] << 40 |
213                 (long) sb[start + 6] << 48 |
214                 (long) sb[start + 7] << 56;
215         if (maybeContent != CONTENT_AS_LONG) {
216             return false;
217         }
218         final int maybeType = sb[start + 8] |
219                 sb[start + 9] << 8 |
220                 sb[start + 10] << 16 |
221                 sb[start + 11] << 24;
222         return maybeType == TYPE_AS_INT;
223     }
224 
225     private static boolean isContentLength(byte[] sb, int start) {
226         final long maybeContent = sb[start] |
227                 sb[start + 1] << 8 |
228                 sb[start + 2] << 16 |
229                 sb[start + 3] << 24 |
230                 (long) sb[start + 4] << 32 |
231                 (long) sb[start + 5] << 40 |
232                 (long) sb[start + 6] << 48 |
233                 (long) sb[start + 7] << 56;
234         if (maybeContent != CONTENT_AS_LONG) {
235             return false;
236         }
237         final long maybeLength = sb[start + 8] |
238                 sb[start + 9] << 8 |
239                 sb[start + 10] << 16 |
240                 sb[start + 11] << 24 |
241                 (long) sb[start + 12] << 32 |
242                 (long) sb[start + 13] << 40;
243         return maybeLength == LENGTH_AS_LONG;
244     }
245 
246     private static boolean isGetMethod(final byte[] sb, int start) {
247         final int maybeGet = sb[start] |
248                 sb[start + 1] << 8 |
249                 sb[start + 2] << 16;
250         return maybeGet == GET_AS_INT;
251     }
252 
253     private static boolean isPostMethod(final byte[] sb, int start) {
254         final int maybePost = sb[start] |
255                 sb[start + 1] << 8 |
256                 sb[start + 2] << 16 |
257                 sb[start + 3] << 24;
258         return maybePost == POST_AS_INT;
259     }
260 
261     @Override
262     protected String splitFirstWordInitialLine(final byte[] sb, final int start, final int length) {
263         if (length == 3) {
264             if (isGetMethod(sb, start)) {
265                 return HttpMethod.GET.name();
266             }
267         } else if (length == 4) {
268             if (isPostMethod(sb, start)) {
269                 return HttpMethod.POST.name();
270             }
271         }
272         return super.splitFirstWordInitialLine(sb, start, length);
273     }
274 
275     @Override
276     protected String splitThirdWordInitialLine(final byte[] sb, final int start, final int length) {
277         if (length == 8) {
278             final long maybeHttp1_x = sb[start] |
279                     sb[start + 1] << 8 |
280                     sb[start + 2] << 16 |
281                     sb[start + 3] << 24 |
282                     (long) sb[start + 4] << 32 |
283                     (long) sb[start + 5] << 40 |
284                     (long) sb[start + 6] << 48 |
285                     (long) sb[start + 7] << 56;
286             if (maybeHttp1_x == HTTP_1_1_AS_LONG) {
287                 return HttpVersion.HTTP_1_1_STRING;
288             } else if (maybeHttp1_x == HTTP_1_0_AS_LONG) {
289                 return HttpVersion.HTTP_1_0_STRING;
290             }
291         }
292         return super.splitThirdWordInitialLine(sb, start, length);
293     }
294 
295     @Override
296     protected HttpMessage createInvalidMessage() {
297         return new DefaultFullHttpRequest(HttpVersion.HTTP_1_0, HttpMethod.GET, "/bad-request", validateHeaders);
298     }
299 
300     @Override
301     protected boolean isDecodingRequest() {
302         return true;
303     }
304 
305     @Override
306     protected boolean isContentAlwaysEmpty(final HttpMessage msg) {
307         // fast-path to save expensive O(n) checks; users can override createMessage
308         // and extends DefaultHttpRequest making implementing HttpResponse:
309         // this is why we cannot use instanceof DefaultHttpRequest here :(
310         if (msg.getClass() == DefaultHttpRequest.class) {
311             return false;
312         }
313         return super.isContentAlwaysEmpty(msg);
314     }
