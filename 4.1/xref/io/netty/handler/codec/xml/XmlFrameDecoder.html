18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandlerContext;
20  import io.netty.handler.codec.ByteToMessageDecoder;
21  import io.netty.handler.codec.CorruptedFrameException;
22  import io.netty.handler.codec.TooLongFrameException;

24  import java.util.List;

26  /**
27   * A frame decoder for single separate XML based message streams.
28   * <p/>
29   * A couple examples will better help illustrate
30   * what this decoder actually does.
31   * <p/>
32   * Given an input array of bytes split over 3 frames like this:
33   * <pre>
34   * +-----+-----+-----------+
35   * | &lt;an | Xml | Element/&gt; |
36   * +-----+-----+-----------+
37   * </pre>
38   * <p/>
39   * this decoder would output a single frame:
40   * <p/>
41   * <pre>
42   * +-----------------+
43   * | &lt;anXmlElement/&gt; |
44   * +-----------------+
45   * </pre>
46   *
47   * Given an input array of bytes split over 5 frames like this:
48   * <pre>
49   * +-----+-----+-----------+-----+----------------------------------+
50   * | &lt;an | Xml | Element/&gt; | &lt;ro | ot&gt;&lt;child&gt;content&lt;/child&gt;&lt;/root&gt; |
51   * +-----+-----+-----------+-----+----------------------------------+
52   * </pre>
53   * <p/>
54   * this decoder would output two frames:
55   * <p/>
56   * <pre>
57   * +-----------------+-------------------------------------+
58   * | &lt;anXmlElement/&gt; | &lt;root&gt;&lt;child&gt;content&lt;/child&gt;&lt;/root&gt; |
59   * +-----------------+-------------------------------------+
60   * </pre>
61   *
62   * <p/>
63   * The byte stream is expected to be in UTF-8 character encoding or ASCII. The current implementation
64   * uses direct {@code byte} to {@code char} cast and then compares that {@code char} to a few low range
65   * ASCII characters like {@code '<'}, {@code '>'} or {@code '/'}. UTF-8 is not using low range [0..0x7F]
66   * byte values for multibyte codepoint representations therefore fully supported by this implementation.
67   * <p/>
68   * Please note that this decoder is not suitable for
69   * xml streaming protocols such as
70   * <a href="http://xmpp.org/rfcs/rfc6120.html">XMPP</a>,
71   * where an initial xml element opens the stream and only
72   * gets closed at the end of the session, although this class
73   * could probably allow for such type of message flow with
74   * minor modifications.
75   */
76  public class XmlFrameDecoder extends ByteToMessageDecoder {
77  
78      private final int maxFrameLength;
79  
80      public XmlFrameDecoder(int maxFrameLength) {
81          if (maxFrameLength < 1) {
82              throw new IllegalArgumentException("maxFrameLength must be a positive int");
83          }
84          this.maxFrameLength = maxFrameLength;
85      }
86  
87      @Override
88      protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
89          boolean openingBracketFound = false;
90          boolean atLeastOneXmlElementFound = false;
91          boolean inCDATASection = false;
92          long openBracketsCount = 0;
93          int length = 0;
94          int leadingWhiteSpaceCount = 0;
95          final int bufferLength = in.writerIndex();
96  
97          if (bufferLength > maxFrameLength) {
98              // bufferLength exceeded maxFrameLength; dropping frame
99              in.skipBytes(in.readableBytes());
100             fail(bufferLength);
101             return;
102         }
103 
104         for (int i = in.readerIndex(); i < bufferLength; i++) {
105             final byte readByte = in.getByte(i);
106             if (!openingBracketFound && Character.isWhitespace(readByte)) {
107                 // xml has not started and whitespace char found
108                 leadingWhiteSpaceCount++;
109             } else if (!openingBracketFound && readByte != '<') {
110                 // garbage found before xml start
111                 fail(ctx);
112                 in.skipBytes(in.readableBytes());
113                 return;
114             } else if (!inCDATASection && readByte == '<') {
115                 openingBracketFound = true;
116 
117                 if (i < bufferLength - 1) {
118                     final byte peekAheadByte = in.getByte(i + 1);
119                     if (peekAheadByte == '/') {
120                         // found </, we must check if it is enclosed
121                         int peekFurtherAheadIndex = i + 2;
122                         while (peekFurtherAheadIndex <= bufferLength - 1) {
123                             //if we have </ and enclosing > we can decrement openBracketsCount
124                             if (in.getByte(peekFurtherAheadIndex) == '>') {
125                                 openBracketsCount--;
126                                 break;
127                             }
128                             peekFurtherAheadIndex++;
129                         }
130                     } else if (isValidStartCharForXmlElement(peekAheadByte)) {
131                         atLeastOneXmlElementFound = true;
132                         // char after < is a valid xml element start char,
133                         // incrementing openBracketsCount
134                         openBracketsCount++;
135                     } else if (peekAheadByte == '!') {
136                         if (isCommentBlockStart(in, i)) {
137                             // <!-- comment --> start found
138                             openBracketsCount++;
139                         } else if (isCDATABlockStart(in, i)) {
140                             // <![CDATA[ start found
141                             openBracketsCount++;
142                             inCDATASection = true;
143                         }
144                     } else if (peekAheadByte == '?') {
145                         // <?xml ?> start found
146                         openBracketsCount++;
147                     }
148                 }
149             } else if (!inCDATASection && readByte == '/') {
150                 if (i < bufferLength - 1 && in.getByte(i + 1) == '>') {
151                     // found />, decrementing openBracketsCount
152                     openBracketsCount--;
153                 }
154             } else if (readByte == '>') {
155                 length = i + 1;
156 
157                 if (i - 1 > -1) {
158                     final byte peekBehindByte = in.getByte(i - 1);
159 
160                     if (!inCDATASection) {
161                         if (peekBehindByte == '?') {
162                             // an <?xml ?> tag was closed
163                             openBracketsCount--;
164                         } else if (peekBehindByte == '-' && i - 2 > -1 && in.getByte(i - 2) == '-') {
165                             // a <!-- comment --> was closed
166                             openBracketsCount--;
167                         }
168                     } else if (peekBehindByte == ']' && i - 2 > -1 && in.getByte(i - 2) == ']') {
169                         // a <![CDATA[...]]> block was closed
170                         openBracketsCount--;
171                         inCDATASection = false;
172                     }
173                 }
174 
175                 if (atLeastOneXmlElementFound && openBracketsCount == 0) {
176                     // xml is balanced, bailing out
177                     break;
178                 }
179             }
180         }
181 
182         final int readerIndex = in.readerIndex();
183         int xmlElementLength = length - readerIndex;
184 
185         if (openBracketsCount == 0 && xmlElementLength > 0) {
186             if (readerIndex + xmlElementLength >= bufferLength) {
187                 xmlElementLength = in.readableBytes();
188             }
189             final ByteBuf frame =
190                     extractFrame(in, readerIndex + leadingWhiteSpaceCount, xmlElementLength - leadingWhiteSpaceCount);
191             in.skipBytes(xmlElementLength);
192             out.add(frame);
193         }
194     }
195 
196     private void fail(long frameLength) {
197         if (frameLength > 0) {
198             throw new TooLongFrameException(
199                             "frame length exceeds " + maxFrameLength + ": " + frameLength + " - discarded");
200         } else {
201             throw new TooLongFrameException(
202                             "frame length exceeds " + maxFrameLength + " - discarding");
203         }
204     }
205 
206     private static void fail(ChannelHandlerContext ctx) {
207         ctx.fireExceptionCaught(new CorruptedFrameException("frame contains content before the xml starts"));
208     }
209 
210     private static ByteBuf extractFrame(ByteBuf buffer, int index, int length) {
211         return buffer.copy(index, length);
212     }
213 
214     /**
215      * Asks whether the given byte is a valid
216      * start char for an xml element name.
217      * <p/>
218      * Please refer to the
219      * <a href="http://www.w3.org/TR/2004/REC-xml11-20040204/#NT-NameStartChar">NameStartChar</a>
220      * formal definition in the W3C XML spec for further info.
221      *
222      * @param b the input char
223      * @return true if the char is a valid start char
224      */
225     private static boolean isValidStartCharForXmlElement(final byte b) {
226         return b >= 'a' && b <= 'z' || b >= 'A' && b <= 'Z' || b == ':' || b == '_';
227     }
228 
229     private static boolean isCommentBlockStart(final ByteBuf in, final int i) {
230         return i < in.writerIndex() - 3
231                 && in.getByte(i + 2) == '-'
232                 && in.getByte(i + 3) == '-';
233     }
234 
235     private static boolean isCDATABlockStart(final ByteBuf in, final int i) {
236         return i < in.writerIndex() - 8
237                 && in.getByte(i + 2) == '['
238                 && in.getByte(i + 3) == 'C'
239                 && in.getByte(i + 4) == 'D'
240                 && in.getByte(i + 5) == 'A'
241                 && in.getByte(i + 6) == 'T'
242                 && in.getByte(i + 7) == 'A'
243                 && in.getByte(i + 8) == '[';
244     }
