16  package io.netty.handler.codec.dns;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandler;
20  import io.netty.channel.ChannelHandlerContext;
21  import io.netty.channel.socket.DatagramPacket;
22  import io.netty.handler.codec.MessageToMessageEncoder;
23  import io.netty.util.CharsetUtil;
24  import io.netty.util.internal.StringUtil;
25  
26  import java.nio.charset.Charset;
27  import java.util.List;
28  
29  /**
30   * DnsQueryEncoder accepts {@link DnsQuery} and encodes to {@link ByteBuf}. This
31   * class also contains methods for encoding parts of DnsQuery's such as the
32   * header and questions.
33   */
34  @ChannelHandler.Sharable
35  public class DnsQueryEncoder extends MessageToMessageEncoder<DnsQuery> {
36  
37      @Override
38      protected void encode(ChannelHandlerContext ctx, DnsQuery query, List<Object> out) throws Exception {
39          ByteBuf buf = ctx.alloc().buffer();
40          encodeHeader(query.header(), buf);
41          List<DnsQuestion> questions = query.questions();
42          for (DnsQuestion question : questions) {
43              encodeQuestion(question, CharsetUtil.UTF_8, buf);
44          }
45          out.add(new DatagramPacket(buf, query.recipient(), null));
46      }
47  
48      /**
49       * Encodes the information in a {@link DnsHeader} and writes it to the
50       * specified {@link ByteBuf}. The header is always 12 bytes long.
51       *
52       * @param header
53       *            the query header being encoded
54       * @param buf
55       *            the buffer the encoded data should be written to
56       */
57      private static void encodeHeader(DnsHeader header, ByteBuf buf) {
58          buf.writeShort(header.id());
59          int flags = 0;
60          flags |= header.type() << 15;
61          flags |= header.opcode() << 14;
62          flags |= header.isRecursionDesired() ? 1 << 8 : 0;
63          buf.writeShort(flags);
64          buf.writeShort(header.questionCount());
65          buf.writeShort(header.answerCount()); // Must be 0
66          buf.writeShort(header.authorityResourceCount()); // Must be 0
67          buf.writeShort(header.additionalResourceCount()); // Must be 0
68      }
69  
70      /**
71       * Encodes the information in a {@link DnsQuestion} and writes it to the
72       * specified {@link ByteBuf}.
73       *
74       * @param question
75       *            the question being encoded
76       * @param charset
77       *            charset names are encoded in
78       * @param buf
79       *            the buffer the encoded data should be written to
80       */
81      private static void encodeQuestion(DnsQuestion question, Charset charset, ByteBuf buf) {
82          String[] parts = StringUtil.split(question.name(), '.');
83          for (String part: parts) {
84              buf.writeByte(part.length());
85              buf.writeBytes(part.getBytes(charset));
86          }
87          buf.writeByte(0); // marks end of name field
88          buf.writeShort(question.type());
89          buf.writeShort(question.dnsClass());
90      }
91  }
