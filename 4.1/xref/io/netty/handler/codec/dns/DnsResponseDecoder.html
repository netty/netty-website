16  package io.netty.handler.codec.dns;

18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandler;
20  import io.netty.channel.ChannelHandlerContext;
21  import io.netty.channel.socket.DatagramPacket;
22  import io.netty.handler.codec.MessageToMessageDecoder;
23  import io.netty.util.CharsetUtil;
24  
25  import java.util.List;

27  /**
28   * DnsResponseDecoder accepts {@link io.netty.channel.socket.DatagramPacket} and encodes to
29   * {@link DnsResponse}. This class also contains methods for decoding parts of
30   * DnsResponses such as questions and resource records.
31   */
32  @ChannelHandler.Sharable
33  public class DnsResponseDecoder extends MessageToMessageDecoder<DatagramPacket> {
34  
35      @Override
36      protected void decode(ChannelHandlerContext ctx, DatagramPacket packet, List<Object> out) throws Exception {
37          ByteBuf buf = packet.content();
38  
39          int id = buf.readUnsignedShort();
40  
41          DnsResponse response = new DnsResponse(id, packet.sender());
42          DnsResponseHeader header = response.header();
43          int flags = buf.readUnsignedShort();
44          header.setType(flags >> 15);
45          header.setOpcode(flags >> 11 & 0xf);
46          header.setRecursionDesired((flags >> 8 & 1) == 1);
47          header.setAuthoritativeAnswer((flags >> 10 & 1) == 1);
48          header.setTruncated((flags >> 9 & 1) == 1);
49          header.setRecursionAvailable((flags >> 7 & 1) == 1);
50          header.setZ(flags >> 4 & 0x7);
51          header.setResponseCode(DnsResponseCode.valueOf(flags & 0xf));
52  
53          int questions = buf.readUnsignedShort();
54          int answers = buf.readUnsignedShort();
55          int authorities = buf.readUnsignedShort();
56          int additionals = buf.readUnsignedShort();
57  
58          for (int i = 0; i < questions; i++) {
59              response.addQuestion(decodeQuestion(buf));
60          }
61          if (header.responseCode() != DnsResponseCode.NOERROR) {
62              // response code for error
63              out.add(response);
64              return;
65          }
66          for (int i = 0; i < answers; i++) {
67              response.addAnswer(decodeResource(buf));
68          }
69          for (int i = 0; i < authorities; i++) {
70              response.addAuthorityResource(decodeResource(buf));
71          }
72          for (int i = 0; i < additionals; i++) {
73              response.addAdditionalResource(decodeResource(buf));
74          }
75          out.add(response);
76      }
77  
78      /**
79       * Retrieves a domain name given a buffer containing a DNS packet. If the
80       * name contains a pointer, the position of the buffer will be set to
81       * directly after the pointer's index after the name has been read.
82       *
83       * @param buf
84       *            the byte buffer containing the DNS packet
85       * @return the domain name for an entry
86       */
87      private static String readName(ByteBuf buf) {
88          int position = -1;
89          StringBuilder name = new StringBuilder();
90          for (int len = buf.readUnsignedByte(); buf.isReadable() && len != 0; len = buf.readUnsignedByte()) {
91              boolean pointer = (len & 0xc0) == 0xc0;
92              if (pointer) {
93                  if (position == -1) {
94                      position = buf.readerIndex() + 1;
95                  }
96                  buf.readerIndex((len & 0x3f) << 8 | buf.readUnsignedByte());
97              } else {
98                  name.append(buf.toString(buf.readerIndex(), len, CharsetUtil.UTF_8)).append('.');
99                  buf.skipBytes(len);
100             }
101         }
102         if (position != -1) {
103             buf.readerIndex(position);
104         }
105         if (name.length() == 0) {
106             return null;
107         }
108         return name.substring(0, name.length() - 1);
109     }
110 
111     /**
112      * Decodes a question, given a DNS packet in a byte buffer.
113      *
114      * @param buf
115      *            the byte buffer containing the DNS packet
116      * @return a decoded {@link DnsQuestion}
117      */
118     private static DnsQuestion decodeQuestion(ByteBuf buf) {
119         String name = readName(buf);
120         int type = buf.readUnsignedShort();
121         int qClass = buf.readUnsignedShort();
122         return new DnsQuestion(name, type, qClass);
123     }
124 
125     /**
126      * Decodes a resource record, given a DNS packet in a byte buffer.
127      *
128      * @param buf
129      *            the byte buffer containing the DNS packet
130      * @return a {@link DnsResource} record containing response data
131      */
132     private static DnsResource decodeResource(ByteBuf buf) {
133         String name = readName(buf);
134         int type = buf.readUnsignedShort();
135         int aClass = buf.readUnsignedShort();
136         long ttl = buf.readUnsignedInt();
137         int len = buf.readUnsignedShort();
138 
139         int readerIndex = buf.readerIndex();
140         ByteBuf payload = buf.duplicate().setIndex(readerIndex, readerIndex + len).retain();
141         buf.readerIndex(readerIndex + len);
142         return new DnsResource(name, type, aClass, ttl, payload);
143     }
