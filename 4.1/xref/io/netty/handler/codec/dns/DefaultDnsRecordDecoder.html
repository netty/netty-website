23  /**
24   * The default {@link DnsRecordDecoder} implementation.
25   *
26   * @see DefaultDnsRecordEncoder
27   */
28  @UnstableApi
29  public class DefaultDnsRecordDecoder implements DnsRecordDecoder {
30  
31      static final String ROOT = ".";
32  
33      /**
34       * Creates a new instance.
35       */
36      protected DefaultDnsRecordDecoder() { }
37  
38      @Override
39      public final DnsQuestion decodeQuestion(ByteBuf in) throws Exception {
40          String name = decodeName(in);
41          DnsRecordType type = DnsRecordType.valueOf(in.readUnsignedShort());
42          int qClass = in.readUnsignedShort();
43          return new DefaultDnsQuestion(name, type, qClass);
44      }
45  
46      @Override
47      public final <T extends DnsRecord> T decodeRecord(ByteBuf in) throws Exception {
48          final int startOffset = in.readerIndex();
49          final String name = decodeName(in);
50  
51          final int endOffset = in.writerIndex();
52          if (endOffset - startOffset < 10) {
53              // Not enough data
54              in.readerIndex(startOffset);
55              return null;
56          }
57  
58          final DnsRecordType type = DnsRecordType.valueOf(in.readUnsignedShort());
59          final int aClass = in.readUnsignedShort();
60          final long ttl = in.readUnsignedInt();
61          final int length = in.readUnsignedShort();
62          final int offset = in.readerIndex();
63  
64          if (endOffset - offset < length) {
65              // Not enough data
66              in.readerIndex(startOffset);
67              return null;
68          }
69  
70          @SuppressWarnings("unchecked")
71          T record = (T) decodeRecord(name, type, aClass, ttl, in, offset, length);
72          in.readerIndex(offset + length);
73          return record;
74      }
75  
76      /**
77       * Decodes a record from the information decoded so far by {@link #decodeRecord(ByteBuf)}.
78       *
79       * @param name the domain name of the record
80       * @param type the type of the record
81       * @param dnsClass the class of the record
82       * @param timeToLive the TTL of the record
83       * @param in the {@link ByteBuf} that contains the RDATA
84       * @param offset the start offset of the RDATA in {@code in}
85       * @param length the length of the RDATA
86       *
87       * @return a {@link DnsRawRecord}. Override this method to decode RDATA and return other record implementation.
88       */
89      protected DnsRecord decodeRecord(
90              String name, DnsRecordType type, int dnsClass, long timeToLive,
91              ByteBuf in, int offset, int length) throws Exception {
92  
93          if (type == DnsRecordType.PTR) {
94              return new DefaultDnsPtrRecord(name, dnsClass, timeToLive, decodeName0(in.slice(offset, length)));
95          }
96          return new DefaultDnsRawRecord(
97                  name, type, dnsClass, timeToLive, in.retainedSlice(offset, length));
98      }

100     /**
101      * Retrieves a domain name given a buffer containing a DNS packet. If the
102      * name contains a pointer, the position of the buffer will be set to
103      * directly after the pointer's index after the name has been read.
104      *
105      * @param in the byte buffer containing the DNS packet
106      * @return the domain name for an entry
107      */
108     protected String decodeName0(ByteBuf in) {
109         return decodeName(in);
110     }

112     /**
113      * Retrieves a domain name given a buffer containing a DNS packet. If the
114      * name contains a pointer, the position of the buffer will be set to
115      * directly after the pointer's index after the name has been read.
116      *
117      * @param in the byte buffer containing the DNS packet
118      * @return the domain name for an entry
119      */
120     public static String decodeName(ByteBuf in) {
121         int position = -1;
122         int checked = 0;
123         final int end = in.writerIndex();
124         final int readable = in.readableBytes();
125 
126         // Looking at the spec we should always have at least enough readable bytes to read a byte here but it seems
127         // some servers do not respect this for empty names. So just workaround this and return an empty name in this
128         // case.
129         //
130         // See:
131         // - https://github.com/netty/netty/issues/5014
132         // - https://www.ietf.org/rfc/rfc1035.txt , Section 3.1
133         if (readable == 0) {
134             return ROOT;
135         }
136 
137         final StringBuilder name = new StringBuilder(readable << 1);
138         while (in.isReadable()) {
139             final int len = in.readUnsignedByte();
140             final boolean pointer = (len & 0xc0) == 0xc0;
141             if (pointer) {
142                 if (position == -1) {
143                     position = in.readerIndex() + 1;
144                 }
145 
146                 if (!in.isReadable()) {
147                     throw new CorruptedFrameException("truncated pointer in a name");
148                 }
149 
150                 final int next = (len & 0x3f) << 8 | in.readUnsignedByte();
151                 if (next >= end) {
152                     throw new CorruptedFrameException("name has an out-of-range pointer");
153                 }
154                 in.readerIndex(next);
155 
156                 // check for loops
157                 checked += 2;
158                 if (checked >= end) {
159                     throw new CorruptedFrameException("name contains a loop.");
160                 }
161             } else if (len != 0) {
162                 if (!in.isReadable(len)) {
163                     throw new CorruptedFrameException("truncated label in a name");
164                 }
165                 name.append(in.toString(in.readerIndex(), len, CharsetUtil.UTF_8)).append('.');
166                 in.skipBytes(len);
167             } else { // len == 0
168                 break;
169             }
170         }
171 
172         if (position != -1) {
173             in.readerIndex(position);
174         }
175 
176         if (name.length() == 0) {
177             return ROOT;
178         }
179 
180         if (name.charAt(name.length() - 1) != '.') {
181             name.append('.');
182         }
183 
184         return name.toString();
185     }
