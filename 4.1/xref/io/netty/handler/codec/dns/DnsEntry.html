311     private final String name;
312     private final int type;
313     private final int dnsClass;

315     // only allow to extend from same package
316     DnsEntry(String name, int type, int dnsClass) {
317         if (name == null) {
318             throw new NullPointerException("name");
319         }
320         if (name.isEmpty()) {
321             throw new IllegalArgumentException("name must not be left blank.");
322         }
323         if ((type & 0xffff) != type) {
324             throw new IllegalArgumentException("type must be an unsigned short.");
325         }
326         if (dnsClass < 1 || dnsClass > 4 && dnsClass != CLASS_NONE && dnsClass != CLASS_ANY) {
327             throw new IllegalArgumentException("an invalid class has been supplied.");
328         }
329         this.name = name;
330         this.type = type;
331         this.dnsClass = dnsClass;
332     }
