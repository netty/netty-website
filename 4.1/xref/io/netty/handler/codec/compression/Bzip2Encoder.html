32  /**
33   * Compresses a {@link ByteBuf} using the Bzip2 algorithm.
34   *
35   * See <a href="http://en.wikipedia.org/wiki/Bzip2">Bzip2</a>.
36   */
37  public class Bzip2Encoder extends MessageToByteEncoder<ByteBuf> {
38      /**
39       * Current state of stream.
40       */
41      private enum State {
42          INIT,
43          INIT_BLOCK,
44          WRITE_DATA,
45          CLOSE_BLOCK
46      }
47  
48      private State currentState = State.INIT;
49  
50      /**
51       * A writer that provides bit-level writes.
52       */
53      private final Bzip2BitWriter writer = new Bzip2BitWriter();
54  
55      /**
56       * The declared maximum block size of the stream (before final run-length decoding).
57       */
58      private final int streamBlockSize;
59  
60      /**
61       * The merged CRC of all blocks compressed so far.
62       */
63      private int streamCRC;
64  
65      /**
66       * The compressor for the current block.
67       */
68      private Bzip2BlockCompressor blockCompressor;
69  
70      /**
71       * (@code true} if the compressed stream has been finished, otherwise {@code false}.
72       */
73      private volatile boolean finished;
74  
75      /**
76       * Used to interact with its {@link ChannelPipeline} and other handlers.
77       */
78      private volatile ChannelHandlerContext ctx;
79  
80      /**
81       * Creates a new bzip2 encoder with the maximum (900,000 byte) block size.
82       */
83      public Bzip2Encoder() {
84          this(MAX_BLOCK_SIZE);
85      }
86  
87      /**
88       * Creates a new bzip2 encoder with the specified {@code blockSizeMultiplier}.
89       * @param blockSizeMultiplier
90       *        The Bzip2 block size as a multiple of 100,000 bytes (minimum {@code 1}, maximum {@code 9}).
91       *        Larger block sizes require more memory for both compression and decompression,
92       *        but give better compression ratios. {@code 9} will usually be the best value to use.
93       */
94      public Bzip2Encoder(final int blockSizeMultiplier) {
95          if (blockSizeMultiplier < MIN_BLOCK_SIZE || blockSizeMultiplier > MAX_BLOCK_SIZE) {
96              throw new IllegalArgumentException(
97                      "blockSizeMultiplier: " + blockSizeMultiplier + " (expected: 1-9)");
98          }
99          streamBlockSize = blockSizeMultiplier * BASE_BLOCK_SIZE;
100     }
101 
102     @Override
103     protected void encode(ChannelHandlerContext ctx, ByteBuf in, ByteBuf out) throws Exception {
104         if (finished) {
105             out.writeBytes(in);
106             return;
107         }
108 
109         for (;;) {
110             switch (currentState) {
111                 case INIT:
112                     out.ensureWritable(4);
113                     out.writeMedium(MAGIC_NUMBER);
114                     out.writeByte('0' + streamBlockSize / BASE_BLOCK_SIZE);
115                     currentState = State.INIT_BLOCK;
116                 case INIT_BLOCK:
117                     blockCompressor = new Bzip2BlockCompressor(writer, streamBlockSize);
118                     currentState = State.WRITE_DATA;
119                 case WRITE_DATA:
120                     if (!in.isReadable()) {
121                         return;
122                     }
123                     Bzip2BlockCompressor blockCompressor = this.blockCompressor;
124                     final int length = Math.min(in.readableBytes(), blockCompressor.availableSize());
125                     final int bytesWritten = blockCompressor.write(in, in.readerIndex(), length);
126                     in.skipBytes(bytesWritten);
127                     if (!blockCompressor.isFull()) {
128                         if (in.isReadable()) {
129                             break;
130                         } else {
131                             return;
132                         }
133                     }
134                     currentState = State.CLOSE_BLOCK;
135                 case CLOSE_BLOCK:
136                     closeBlock(out);
137                     currentState = State.INIT_BLOCK;
138                     break;
139                 default:
140                     throw new IllegalStateException();
141             }
142         }
143     }
144 
145     /**
146      * Close current block and update {@link #streamCRC}.
147      */
148     private void closeBlock(ByteBuf out) {
149         final Bzip2BlockCompressor blockCompressor = this.blockCompressor;
150         if (!blockCompressor.isEmpty()) {
151             blockCompressor.close(out);
152             final int blockCRC = blockCompressor.crc();
153             streamCRC = (streamCRC << 1 | streamCRC >>> 31) ^ blockCRC;
154         }
155     }
156 
157     /**
158      * Returns {@code true} if and only if the end of the compressed stream has been reached.
159      */
160     public boolean isClosed() {
161         return finished;
162     }
163 
164     /**
165      * Close this {@link Bzip2Encoder} and so finish the encoding.
166      *
167      * The returned {@link ChannelFuture} will be notified once the operation completes.
168      */
169     public ChannelFuture close() {
170         return close(ctx().newPromise());
171     }
172 
173     /**
174      * Close this {@link Bzip2Encoder} and so finish the encoding.
175      * The given {@link ChannelFuture} will be notified once the operation
176      * completes and will also be returned.
177      */
178     public ChannelFuture close(final ChannelPromise promise) {
179         ChannelHandlerContext ctx = ctx();
180         EventExecutor executor = ctx.executor();
181         if (executor.inEventLoop()) {
182             return finishEncode(ctx, promise);
183         } else {
184             executor.execute(new Runnable() {
185                 @Override
186                 public void run() {
187                     ChannelFuture f = finishEncode(ctx(), promise);
188                     f.addListener(new ChannelPromiseNotifier(promise));
189                 }
190             });
191             return promise;
192         }
193     }
194 
195     @Override
196     public void close(final ChannelHandlerContext ctx, final ChannelPromise promise) throws Exception {
197         ChannelFuture f = finishEncode(ctx, ctx.newPromise());
198         f.addListener(new ChannelFutureListener() {
199             @Override
200             public void operationComplete(ChannelFuture f) throws Exception {
201                 ctx.close(promise);
202             }
203         });
204 
205         if (!f.isDone()) {
206             // Ensure the channel is closed even if the write operation completes in time.
207             ctx.executor().schedule(new Runnable() {
208                 @Override
209                 public void run() {
210                     ctx.close(promise);
211                 }
212             }, 10, TimeUnit.SECONDS); // FIXME: Magic number
213         }
214     }
215 
216     private ChannelFuture finishEncode(final ChannelHandlerContext ctx, ChannelPromise promise) {
217         if (finished) {
218             promise.setSuccess();
219             return promise;
220         }
221         finished = true;
222 
223         final ByteBuf footer = ctx.alloc().buffer();
224         closeBlock(footer);
225 
226         final int streamCRC = this.streamCRC;
227         final Bzip2BitWriter writer = this.writer;
228         try {
229             writer.writeBits(footer, 24, END_OF_STREAM_MAGIC_1);
230             writer.writeBits(footer, 24, END_OF_STREAM_MAGIC_2);
231             writer.writeInt(footer, streamCRC);
232             writer.flush(footer);
233         } finally {
234             blockCompressor = null;
235         }
236         return ctx.writeAndFlush(footer, promise);
237     }
238 
239     private ChannelHandlerContext ctx() {
240         ChannelHandlerContext ctx = this.ctx;
241         if (ctx == null) {
242             throw new IllegalStateException("not added to a pipeline");
243         }
244         return ctx;
245     }
246 
247     @Override
248     public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
249         this.ctx = ctx;
250     }
