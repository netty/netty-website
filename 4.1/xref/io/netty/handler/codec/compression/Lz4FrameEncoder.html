42  /**
43   * Compresses a {@link ByteBuf} using the LZ4 format.
44   *
45   * See original <a href="http://code.google.com/p/lz4/">LZ4 website</a>
46   * and <a href="http://fastcompression.blogspot.ru/2011/05/lz4-explained.html">LZ4 block format</a>
47   * for full description.
48   *
49   * Since the original LZ4 block format does not contains size of compressed block and size of original data
50   * this encoder uses format like <a href="https://github.com/idelpivnitskiy/lz4-java">LZ4 Java</a> library
51   * written by Adrien Grand and approved by Yann Collet (author of original LZ4 library).
52   *
53   *  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *     * * * * * * * * * *
54   *  * Magic * Token *  Compressed *  Decompressed *  Checksum *  +  *  LZ4 compressed *
55   *  *       *       *    length   *     length    *           *     *      block      *
56   *  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *     * * * * * * * * * *
57   */

136         /**
137          * Creates a new customizable LZ4 encoder.
138          *
139          * @param factory         user customizable {@link LZ4Factory} instance
140          *                        which may be JNI bindings to the original C implementation, a pure Java implementation
141          *                        or a Java implementation that uses the {@link sun.misc.Unsafe}
142          * @param highCompressor  if {@code true} codec will use compressor which requires more memory
143          *                        and is slower but compresses more efficiently
144          * @param blockSize       the maximum number of bytes to try to compress at once,
145          *                        must be >= 64 and <= 32 M
146          * @param checksum        the {@link Checksum} instance to use to check data for integrity
147          * @param maxEncodeSize   the maximum size for an encode (compressed) buffer
148          */
149     public Lz4FrameEncoder(LZ4Factory factory, boolean highCompressor, int blockSize,
150                            Checksum checksum, int maxEncodeSize) {
151         if (factory == null) {
152             throw new NullPointerException("factory");
153         }
154         if (checksum == null) {
155             throw new NullPointerException("checksum");
156         }
157 
158         compressor = highCompressor ? factory.highCompressor() : factory.fastCompressor();
159         this.checksum = ByteBufChecksum.wrapChecksum(checksum);
160 
161         compressionLevel = compressionLevel(blockSize);
162         this.blockSize = blockSize;
163         this.maxEncodeSize = ObjectUtil.checkPositive(maxEncodeSize, "maxEncodeSize");
164         finished = false;
165     }

167     /**
168      * Calculates compression level on the basis of block size.
169      */
170     private static int compressionLevel(int blockSize) {
171         if (blockSize < MIN_BLOCK_SIZE || blockSize > MAX_BLOCK_SIZE) {
172             throw new IllegalArgumentException(String.format(
173                     "blockSize: %d (expected: %d-%d)", blockSize, MIN_BLOCK_SIZE, MAX_BLOCK_SIZE));
174         }
175         int compressionLevel = 32 - Integer.numberOfLeadingZeros(blockSize - 1); // ceil of log2
176         compressionLevel = Math.max(0, compressionLevel - COMPRESSION_LEVEL_BASE);
177         return compressionLevel;
178     }

185     private ByteBuf allocateBuffer(ChannelHandlerContext ctx, ByteBuf msg, boolean preferDirect,
186                                    boolean allowEmptyReturn) {
187         int targetBufSize = 0;
188         int remaining = msg.readableBytes() + buffer.readableBytes();
189 
190         // quick overflow check
191         if (remaining < 0) {
192             throw new EncoderException("too much data to allocate a buffer for compression");
193         }
194 
195         while (remaining > 0) {
196             int curSize = Math.min(blockSize, remaining);
197             remaining -= curSize;
198             // calculate the total compressed size of the current block (including header) and add to the total
199             targetBufSize += compressor.maxCompressedLength(curSize) + HEADER_LENGTH;
200         }
201 
202         // in addition to just the raw byte count, the headers (HEADER_LENGTH) per block (configured via
203         // #blockSize) will also add to the targetBufSize, and the combination of those would never wrap around
204         // again to be >= 0, this is a good check for the overflow case.
205         if (targetBufSize > maxEncodeSize || 0 > targetBufSize) {
206             throw new EncoderException(String.format("requested encode buffer size (%d bytes) exceeds the maximum " +
207                                                      "allowable size (%d bytes)", targetBufSize, maxEncodeSize));
208         }
209 
210         if (allowEmptyReturn && targetBufSize < blockSize) {
211             return Unpooled.EMPTY_BUFFER;
212         }
213 
214         if (preferDirect) {
215             return ctx.alloc().ioBuffer(targetBufSize, targetBufSize);
216         } else {
217             return ctx.alloc().heapBuffer(targetBufSize, targetBufSize);
218         }
219     }

221     /**
222      * {@inheritDoc}
223      *
224      * Encodes the input buffer into {@link #blockSize} chunks in the output buffer. Data is only compressed and
225      * written once we hit the {@link #blockSize}; else, it is copied into the backing {@link #buffer} to await
226      * more data.
227      */
228     @Override
229     protected void encode(ChannelHandlerContext ctx, ByteBuf in, ByteBuf out) throws Exception {
230         if (finished) {
231             out.writeBytes(in);
232             return;
233         }
234 
235         final ByteBuf buffer = this.buffer;
236         int length;
237         while ((length = in.readableBytes()) > 0) {
238             final int nextChunkSize = Math.min(length, buffer.writableBytes());
239             in.readBytes(buffer, nextChunkSize);
240 
241             if (!buffer.isWritable()) {
242                 flushBufferedData(out);
243             }
244         }
245     }
246 
247     private void flushBufferedData(ByteBuf out) {
248         int flushableBytes = buffer.readableBytes();
249         if (flushableBytes == 0) {
250             return;
251         }
252         checksum.reset();
253         checksum.update(buffer, buffer.readerIndex(), flushableBytes);
254         final int check = (int) checksum.getValue();
255 
256         final int bufSize = compressor.maxCompressedLength(flushableBytes) + HEADER_LENGTH;
257         out.ensureWritable(bufSize);
258         final int idx = out.writerIndex();
259         int compressedLength;
260         try {
261             ByteBuffer outNioBuffer = out.internalNioBuffer(idx + HEADER_LENGTH, out.writableBytes() - HEADER_LENGTH);
262             int pos = outNioBuffer.position();
263             // We always want to start at position 0 as we take care of reusing the buffer in the encode(...) loop.
264             compressor.compress(buffer.internalNioBuffer(0, flushableBytes), outNioBuffer);
265             compressedLength = outNioBuffer.position() - pos;
266         } catch (LZ4Exception e) {
267             throw new CompressionException(e);
268         }
269         final int blockType;
270         if (compressedLength >= flushableBytes) {
271             blockType = BLOCK_TYPE_NON_COMPRESSED;
272             compressedLength = flushableBytes;
273             out.setBytes(idx + HEADER_LENGTH, buffer, 0, flushableBytes);
274         } else {
275             blockType = BLOCK_TYPE_COMPRESSED;
276         }
277 
278         out.setLong(idx, MAGIC_NUMBER);
279         out.setByte(idx + TOKEN_OFFSET, (byte) (blockType | compressionLevel));
280         out.setIntLE(idx + COMPRESSED_LENGTH_OFFSET, compressedLength);
281         out.setIntLE(idx + DECOMPRESSED_LENGTH_OFFSET, flushableBytes);
282         out.setIntLE(idx + CHECKSUM_OFFSET, check);
283         out.writerIndex(idx + HEADER_LENGTH + compressedLength);
284         buffer.clear();
285     }

287     @Override
288     public void flush(final ChannelHandlerContext ctx) throws Exception {
289         if (buffer != null && buffer.isReadable()) {
290             final ByteBuf buf = allocateBuffer(ctx, Unpooled.EMPTY_BUFFER, isPreferDirect(), false);
291             flushBufferedData(buf);
292             ctx.write(buf);
293         }
294         ctx.flush();
295     }

297     private ChannelFuture finishEncode(final ChannelHandlerContext ctx, ChannelPromise promise) {
298         if (finished) {
299             promise.setSuccess();
300             return promise;
301         }
302         finished = true;
303 
304         try {
305             final ByteBuf footer = ctx.alloc().heapBuffer(
306                     compressor.maxCompressedLength(buffer.readableBytes()) + HEADER_LENGTH);
307             flushBufferedData(footer);
308 
309             final int idx = footer.writerIndex();
310             footer.setLong(idx, MAGIC_NUMBER);
311             footer.setByte(idx + TOKEN_OFFSET, (byte) (BLOCK_TYPE_NON_COMPRESSED | compressionLevel));
312             footer.setInt(idx + COMPRESSED_LENGTH_OFFSET, 0);
313             footer.setInt(idx + DECOMPRESSED_LENGTH_OFFSET, 0);
314             footer.setInt(idx + CHECKSUM_OFFSET, 0);
315 
316             footer.writerIndex(idx + HEADER_LENGTH);
317 
318             return ctx.writeAndFlush(footer, promise);
319         } finally {
320             cleanup();
321         }
322     }

324     private void cleanup() {
325         compressor = null;
326         checksum = null;
327         if (buffer != null) {
328             buffer.release();
329             buffer = null;
330         }

349     /**
350      * Close this {@link Lz4FrameEncoder} and so finish the encoding.
351      * The given {@link ChannelFuture} will be notified once the operation
352      * completes and will also be returned.
353      */
354     public ChannelFuture close(final ChannelPromise promise) {
355         ChannelHandlerContext ctx = ctx();
356         EventExecutor executor = ctx.executor();
357         if (executor.inEventLoop()) {
358             return finishEncode(ctx, promise);
359         } else {
360             executor.execute(new Runnable() {
361                 @Override
362                 public void run() {
363                     ChannelFuture f = finishEncode(ctx(), promise);
364                     f.addListener(new ChannelPromiseNotifier(promise));
365                 }
366             });
367             return promise;
368         }
369     }

371     @Override
372     public void close(final ChannelHandlerContext ctx, final ChannelPromise promise) throws Exception {
373         ChannelFuture f = finishEncode(ctx, ctx.newPromise());
374         f.addListener(new ChannelFutureListener() {
375             @Override
376             public void operationComplete(ChannelFuture f) throws Exception {
377                 ctx.close(promise);
378             }
379         });
380 
381         if (!f.isDone()) {
382             // Ensure the channel is closed even if the write operation completes in time.
383             ctx.executor().schedule(new Runnable() {
384                 @Override
385                 public void run() {
386                     ctx.close(promise);
387                 }
388             }, 10, TimeUnit.SECONDS); // FIXME: Magic number
389         }
390     }
