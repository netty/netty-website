16  package io.netty.handler.codec.compression;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandlerContext;
20  import io.netty.handler.codec.ByteToMessageDecoder;
21  
22  import java.util.List;
23  import java.util.zip.Adler32;
24  import java.util.zip.Checksum;
25  
26  import static io.netty.handler.codec.compression.FastLz.*;
27  
28  /**
29   * Uncompresses a {@link ByteBuf} encoded by {@link FastLzFrameEncoder} using the FastLZ algorithm.
30   *
31   * See <a href="https://github.com/netty/netty/issues/2750">FastLZ format</a>.
32   */
33  public class FastLzFrameDecoder extends ByteToMessageDecoder {
34      /**
35       * Current state of decompression.
36       */
37      private enum State {
38          INIT_BLOCK,
39          INIT_BLOCK_PARAMS,
40          DECOMPRESS_DATA,
41          CORRUPTED
42      }
43  
44      private State currentState = State.INIT_BLOCK;
45  
46      /**
47       * Underlying checksum calculator in use.
48       */
49      private final Checksum checksum;
50  
51      /**
52       * Length of current received chunk of data.
53       */
54      private int chunkLength;
55  
56      /**
57       * Original of current received chunk of data.
58       * It is equal to {@link #chunkLength} for non compressed chunks.
59       */
60      private int originalLength;
61  
62      /**
63       * Indicates is this chunk compressed or not.
64       */
65      private boolean isCompressed;
66  
67      /**
68       * Indicates is this chunk has checksum or not.
69       */
70      private boolean hasChecksum;
71  
72      /**
73       * Chechsum value of current received chunk of data which has checksum.
74       */
75      private int currentChecksum;
76  
77      /**
78       * Creates the fastest FastLZ decoder without checksum calculation.
79       */
80      public FastLzFrameDecoder() {
81          this(false);
82      }
83  
84      /**
85       * Creates a FastLZ decoder with calculation of checksums as specified.
86       *
87       * @param validateChecksums
88       *        If true, the checksum field will be validated against the actual
89       *        uncompressed data, and if the checksums do not match, a suitable
90       *        {@link DecompressionException} will be thrown.
91       *        Note, that in this case decoder will use {@link java.util.zip.Adler32}
92       *        as a default checksum calculator.
93       */
94      public FastLzFrameDecoder(boolean validateChecksums) {
95          this(validateChecksums ? new Adler32() : null);
96      }
97  
98      /**
99       * Creates a FastLZ decoder with specified checksum calculator.
100      *
101      * @param checksum
102      *        the {@link Checksum} instance to use to check data for integrity.
103      *        You may set {@code null} if you do not want to validate checksum of each block.
104      */
105     public FastLzFrameDecoder(Checksum checksum) {
106         this.checksum = checksum;
107     }
108 
109     @Override
110     protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
111         for (;;) {
112             try {
113                 switch (currentState) {
114                     case INIT_BLOCK:
115                         if (in.readableBytes() < 4) {
116                             return;
117                         }
118 
119                         final int magic = in.readUnsignedMedium();
120                         if (magic != MAGIC_NUMBER) {
121                             throw new DecompressionException("unexpected block identifier");
122                         }
123 
124                         final byte options = in.readByte();
125                         isCompressed = (options & 0x01) == BLOCK_TYPE_COMPRESSED;
126                         hasChecksum = (options & 0x10) == BLOCK_WITH_CHECKSUM;
127 
128                         currentState = State.INIT_BLOCK_PARAMS;
129                     case INIT_BLOCK_PARAMS:
130                         if (in.readableBytes() < 2 + (isCompressed ? 2 : 0) + (hasChecksum ? 4 : 0)) {
131                             return;
132                         }
133                         currentChecksum = hasChecksum ? in.readInt() : 0;
134                         chunkLength = in.readUnsignedShort();
135                         originalLength = isCompressed ? in.readUnsignedShort() : chunkLength;
136 
137                         currentState = State.DECOMPRESS_DATA;
138                     case DECOMPRESS_DATA:
139                         final int chunkLength = this.chunkLength;
140                         if (in.readableBytes() < chunkLength) {
141                             return;
142                         }
143 
144                         final int idx = in.readerIndex();
145                         final int originalLength = this.originalLength;
146 
147                         ByteBuf uncompressed = ctx.alloc().heapBuffer(originalLength, originalLength);
148                         final byte[] output = uncompressed.array();
149                         final int outputPtr = uncompressed.arrayOffset() + uncompressed.writerIndex();
150 
151                         boolean success = false;
152                         try {
153                             if (isCompressed) {
154                                 final byte[] input;
155                                 final int inputPtr;
156                                 if (in.hasArray()) {
157                                     input = in.array();
158                                     inputPtr = in.arrayOffset() + idx;
159                                 } else {
160                                     input = new byte[chunkLength];
161                                     in.getBytes(idx, input);
162                                     inputPtr = 0;
163                                 }
164 
165                                 final int decompressedBytes = decompress(input, inputPtr, chunkLength,
166                                         output, outputPtr, originalLength);
167                                 if (originalLength != decompressedBytes) {
168                                     throw new DecompressionException(String.format(
169                                             "stream corrupted: originalLength(%d) and actual length(%d) mismatch",
170                                             originalLength, decompressedBytes));
171                                 }
172                             } else {
173                                 in.getBytes(idx, output, outputPtr, chunkLength);
174                             }
175 
176                             final Checksum checksum = this.checksum;
177                             if (hasChecksum && checksum != null) {
178                                 checksum.reset();
179                                 checksum.update(output, outputPtr, originalLength);
180                                 final int checksumResult = (int) checksum.getValue();
181                                 if (checksumResult != currentChecksum) {
182                                     throw new DecompressionException(String.format(
183                                             "stream corrupted: mismatching checksum: %d (expected: %d)",
184                                             checksumResult, currentChecksum));
185                                 }
186                             }
187                             uncompressed.writerIndex(uncompressed.writerIndex() + originalLength);
188                             out.add(uncompressed);
189                             in.skipBytes(chunkLength);
190 
191                             currentState = State.INIT_BLOCK;
192                             success = true;
193                         } finally {
194                             if (!success) {
195                                 uncompressed.release();
196                             }
197                         }
198                         break;
199                     case CORRUPTED:
200                         in.skipBytes(in.readableBytes());
201                         return;
202                     default:
203                         throw new IllegalStateException();
204                 }
205             } catch (Exception e) {
206                 currentState = State.CORRUPTED;
207                 throw e;
208             }
209         }
210     }
