72      private HAProxyMessage(
73              HAProxyProtocolVersion protocolVersion, HAProxyCommand command, HAProxyProxiedProtocol proxiedProtocol,
74              String sourceAddress, String destinationAddress, int sourcePort, int destinationPort) {

76          if (proxiedProtocol == null) {

95      /**
96       * Decodes a version 2, binary proxy protocol header.
97       *
98       * @param header                     a version 2 proxy protocol header
99       * @return                           {@link HAProxyMessage} instance
100      * @throws HAProxyProtocolException  if any portion of the header is invalid
101      */
102     static HAProxyMessage decodeHeader(ByteBuf header) {
103         if (header == null) {
104             throw new NullPointerException("header");
105         }
106 
107         if (header.readableBytes() < 16) {
108             throw new HAProxyProtocolException(
109                     "incomplete header: " + header.readableBytes() + " bytes (expected: 16+ bytes)");
110         }
111 
112         // Per spec, the 13th byte is the protocol version and command byte
113         header.skipBytes(12);
114         final byte verCmdByte = header.readByte();
115 
116         HAProxyProtocolVersion ver;
117         try {
118             ver = HAProxyProtocolVersion.valueOf(verCmdByte);
119         } catch (IllegalArgumentException e) {
120             throw new HAProxyProtocolException(e);
121         }
122 
123         if (ver != HAProxyProtocolVersion.V2) {
124             throw new HAProxyProtocolException("version 1 unsupported: 0x" + Integer.toHexString(verCmdByte));
125         }
126 
127         HAProxyCommand cmd;
128         try {
129             cmd = HAProxyCommand.valueOf(verCmdByte);
130         } catch (IllegalArgumentException e) {
131             throw new HAProxyProtocolException(e);
132         }
133 
134         if (cmd == HAProxyCommand.LOCAL) {
135             return V2_LOCAL_MSG;
136         }
137 
138         // Per spec, the 14th byte is the protocol and address family byte
139         HAProxyProxiedProtocol protAndFam;
140         try {
141             protAndFam = HAProxyProxiedProtocol.valueOf(header.readByte());
142         } catch (IllegalArgumentException e) {
143             throw new HAProxyProtocolException(e);
144         }
145 
146         if (protAndFam == HAProxyProxiedProtocol.UNKNOWN) {
147             return V2_UNKNOWN_MSG;
148         }
149 
150         int addressInfoLen = header.readUnsignedShort();
151 
152         String srcAddress;
153         String dstAddress;
154         int addressLen;
155         int srcPort = 0;
156         int dstPort = 0;
157 
158         AddressFamily addressFamily = protAndFam.addressFamily();
159 
160         if (addressFamily == AddressFamily.AF_UNIX) {
161             // unix sockets require 216 bytes for address information
162             if (addressInfoLen < 216 || header.readableBytes() < 216) {
163                 throw new HAProxyProtocolException(
164                     "incomplete UNIX socket address information: " +
165                             Math.min(addressInfoLen, header.readableBytes()) + " bytes (expected: 216+ bytes)");
166             }
167             int startIdx = header.readerIndex();
168             int addressEnd = header.forEachByte(startIdx, 108, ByteProcessor.FIND_NUL);
169             if (addressEnd == -1) {
170                 addressLen = 108;
171             } else {
172                 addressLen = addressEnd - startIdx;
173             }
174             srcAddress = header.toString(startIdx, addressLen, CharsetUtil.US_ASCII);
175 
176             startIdx += 108;
177 
178             addressEnd = header.forEachByte(startIdx, 108, ByteProcessor.FIND_NUL);
179             if (addressEnd == -1) {
180                 addressLen = 108;
181             } else {
182                 addressLen = addressEnd - startIdx;
183             }
184             dstAddress = header.toString(startIdx, addressLen, CharsetUtil.US_ASCII);
185         } else {
186             if (addressFamily == AddressFamily.AF_IPv4) {
187                 // IPv4 requires 12 bytes for address information
188                 if (addressInfoLen < 12 || header.readableBytes() < 12) {
189                     throw new HAProxyProtocolException(
190                         "incomplete IPv4 address information: " +
191                                 Math.min(addressInfoLen, header.readableBytes()) + " bytes (expected: 12+ bytes)");
192                 }
193                 addressLen = 4;
194             } else if (addressFamily == AddressFamily.AF_IPv6) {
195                 // IPv6 requires 36 bytes for address information
196                 if (addressInfoLen < 36 || header.readableBytes() < 36) {
197                     throw new HAProxyProtocolException(
198                         "incomplete IPv6 address information: " +
199                                 Math.min(addressInfoLen, header.readableBytes()) + " bytes (expected: 36+ bytes)");
200                 }
201                 addressLen = 16;
202             } else {
203                 throw new HAProxyProtocolException(
204                     "unable to parse address information (unkown address family: " + addressFamily + ')');
205             }
206 
207             // Per spec, the src address begins at the 17th byte
208             srcAddress = ipBytestoString(header, addressLen);
209             dstAddress = ipBytestoString(header, addressLen);
210             srcPort = header.readUnsignedShort();
211             dstPort = header.readUnsignedShort();
212         }
213 
214         return new HAProxyMessage(ver, cmd, protAndFam, srcAddress, dstAddress, srcPort, dstPort);
215     }

217     /**
218      * Decodes a version 1, human-readable proxy protocol header.
219      *
220      * @param header                     a version 1 proxy protocol header
221      * @return                           {@link HAProxyMessage} instance
222      * @throws HAProxyProtocolException  if any portion of the header is invalid
223      */
224     static HAProxyMessage decodeHeader(String header) {
225         if (header == null) {
226             throw new HAProxyProtocolException("header");
227         }
228 
229         String[] parts = header.split(" ");
230         int numParts = parts.length;
231 
232         if (numParts < 2) {
233             throw new HAProxyProtocolException(
234                     "invalid header: " + header + " (expected: 'PROXY' and proxied protocol values)");
235         }
236 
237         if (!"PROXY".equals(parts[0])) {
238             throw new HAProxyProtocolException("unknown identifier: " + parts[0]);
239         }
240 
241         HAProxyProxiedProtocol protAndFam;
242         try {
243             protAndFam = HAProxyProxiedProtocol.valueOf(parts[1]);
244         } catch (IllegalArgumentException e) {
245             throw new HAProxyProtocolException(e);
246         }
247 
248         if (protAndFam != HAProxyProxiedProtocol.TCP4 &&
249                 protAndFam != HAProxyProxiedProtocol.TCP6 &&
250                 protAndFam != HAProxyProxiedProtocol.UNKNOWN) {
251             throw new HAProxyProtocolException("unsupported v1 proxied protocol: " + parts[1]);
252         }
253 
254         if (protAndFam == HAProxyProxiedProtocol.UNKNOWN) {
255             return V1_UNKNOWN_MSG;
256         }
257 
258         if (numParts != 6) {
259             throw new HAProxyProtocolException("invalid TCP4/6 header: " + header + " (expected: 6 parts)");
260         }
261 
262         return new HAProxyMessage(
263                 HAProxyProtocolVersion.V1, HAProxyCommand.PROXY,
264                 protAndFam, parts[2], parts[3], parts[4], parts[5]);
265     }

267     /**
268      * Convert ip address bytes to string representation
269      *
270      * @param header     buffer containing ip address bytes
271      * @param addressLen number of bytes to read (4 bytes for IPv4, 16 bytes for IPv6)
272      * @return           string representation of the ip address
273      */
274     private static String ipBytestoString(ByteBuf header, int addressLen) {
275         StringBuilder sb = new StringBuilder();
276         if (addressLen == 4) {
277             sb.append(header.readByte() & 0xff);
278             sb.append('.');
279             sb.append(header.readByte() & 0xff);
280             sb.append('.');
281             sb.append(header.readByte() & 0xff);
282             sb.append('.');
283             sb.append(header.readByte() & 0xff);
284         } else {
285             sb.append(Integer.toHexString(header.readUnsignedShort()));
286             sb.append(':');
287             sb.append(Integer.toHexString(header.readUnsignedShort()));
288             sb.append(':');
289             sb.append(Integer.toHexString(header.readUnsignedShort()));
290             sb.append(':');
291             sb.append(Integer.toHexString(header.readUnsignedShort()));
292             sb.append(':');
293             sb.append(Integer.toHexString(header.readUnsignedShort()));
294             sb.append(':');
295             sb.append(Integer.toHexString(header.readUnsignedShort()));
296             sb.append(':');
297             sb.append(Integer.toHexString(header.readUnsignedShort()));
298             sb.append(':');
299             sb.append(Integer.toHexString(header.readUnsignedShort()));
300         }
301         return sb.toString();
302     }

304     /**
305      * Convert port to integer
306      *
307      * @param value                      the port
308      * @return                           port as an integer
309      * @throws HAProxyProtocolException  if port is not a valid integer
310      */
311     private static int portStringToInt(String value) {
312         int port;
313         try {
314             port = Integer.parseInt(value);
315         } catch (NumberFormatException e) {
316             throw new HAProxyProtocolException("invalid port: " + value, e);
317         }
318 
319         if (port <= 0 || port > 65535) {
320             throw new HAProxyProtocolException("invalid port: " + value + " (expected: 1 ~ 65535)");
321         }
322 
323         return port;
324     }

326     /**
327      * Validate an address (IPv4, IPv6, Unix Socket)
328      *
329      * @param address                    human-readable address
330      * @param addrFamily                 the {@link AddressFamily} to check the address against
331      * @throws HAProxyProtocolException  if the address is invalid
332      */
333     private static void checkAddress(String address, AddressFamily addrFamily) {
334         if (addrFamily == null) {
335             throw new NullPointerException("addrFamily");
336         }
337 
338         switch (addrFamily) {
339             case AF_UNSPEC:
340                 if (address != null) {
341                     throw new HAProxyProtocolException("unable to validate an AF_UNSPEC address: " + address);
342                 }
343                 return;
344             case AF_UNIX:
345                 return;
346         }
347 
348         if (address == null) {
349             throw new NullPointerException("address");
350         }
351 
352         switch (addrFamily) {
353             case AF_IPv4:
354                 if (!NetUtil.isValidIpV4Address(address)) {
355                     throw new HAProxyProtocolException("invalid IPv4 address: " + address);
356                 }
357                 break;
358             case AF_IPv6:
359                 if (!NetUtil.isValidIpV6Address(address)) {
360                     throw new HAProxyProtocolException("invalid IPv6 address: " + address);
361                 }
362                 break;
363             default:
364                 throw new Error();
365         }
366     }
