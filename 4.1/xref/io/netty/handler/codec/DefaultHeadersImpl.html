17  import io.netty.util.HashingStrategy;

19  /**
20   * A concrete implementation of {@link DefaultHeaders} that allows for direct instantiation.
21   * @param <K> the type of the header name.
22   * @param <V> the type of the header value.
23   */
24  public final class DefaultHeadersImpl<K, V> extends DefaultHeaders<K, V, DefaultHeadersImpl<K, V>> {
25      public DefaultHeadersImpl(HashingStrategy<K> nameHashingStrategy,
26              ValueConverter<V> valueConverter, NameValidator<K> nameValidator) {
27          super(nameHashingStrategy, valueConverter, nameValidator);
28      }
29  }
