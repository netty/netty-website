31  /**
32   * Encodes Mqtt messages into bytes following the protocol specification v3.1
33   * as described here <a href="http://public.dhe.ibm.com/software/dw/webservices/ws-mqtt/mqtt-v3r1.html">MQTTV3.1</a>
34   */
35  @ChannelHandler.Sharable
36  public final class MqttEncoder extends MessageToMessageEncoder<MqttMessage> {
37  
38      public static final MqttEncoder INSTANCE = new MqttEncoder();
39  
40      private MqttEncoder() { }
41  
42      @Override
43      protected void encode(ChannelHandlerContext ctx, MqttMessage msg, List<Object> out) throws Exception {
44          out.add(doEncode(ctx.alloc(), msg));
45      }

47      /**
48       * This is the main encoding method.
49       * It's only visible for testing.
50       *
51       * @param byteBufAllocator Allocates ByteBuf
52       * @param message MQTT message to encode
53       * @return ByteBuf with encoded bytes
54       */
55      static ByteBuf doEncode(ByteBufAllocator byteBufAllocator, MqttMessage message) {
56  
57          switch (message.fixedHeader().messageType()) {
58              case CONNECT:
59                  return encodeConnectMessage(byteBufAllocator, (MqttConnectMessage) message);
60  
61              case CONNACK:
62                  return encodeConnAckMessage(byteBufAllocator, (MqttConnAckMessage) message);
63  
64              case PUBLISH:
65                  return encodePublishMessage(byteBufAllocator, (MqttPublishMessage) message);
66  
67              case SUBSCRIBE:
68                  return encodeSubscribeMessage(byteBufAllocator, (MqttSubscribeMessage) message);
69  
70              case UNSUBSCRIBE:
71                  return encodeUnsubscribeMessage(byteBufAllocator, (MqttUnsubscribeMessage) message);
72  
73              case SUBACK:
74                  return encodeSubAckMessage(byteBufAllocator, (MqttSubAckMessage) message);
75  
76              case UNSUBACK:
77              case PUBACK:
78              case PUBREC:
79              case PUBREL:
80              case PUBCOMP:
81                  return encodeMessageWithOnlySingleByteFixedHeaderAndMessageId(byteBufAllocator, message);
82  
83              case PINGREQ:
84              case PINGRESP:
85              case DISCONNECT:
86                  return encodeMessageWithOnlySingleByteFixedHeader(byteBufAllocator, message);
87  
88              default:
89                  throw new IllegalArgumentException(
90                          "Unknown message type: " + message.fixedHeader().messageType().value());
91          }
92      }

94      private static ByteBuf encodeConnectMessage(
95              ByteBufAllocator byteBufAllocator,
96              MqttConnectMessage message) {
97          int payloadBufferSize = 0;
98  
99          MqttFixedHeader mqttFixedHeader = message.fixedHeader();
100         MqttConnectVariableHeader variableHeader = message.variableHeader();
101         MqttConnectPayload payload = message.payload();
102         MqttVersion mqttVersion = MqttVersion.fromProtocolNameAndLevel(variableHeader.name(),
103                 (byte) variableHeader.version());
104 
105         // Client id
106         String clientIdentifier = payload.clientIdentifier();
107         if (!isValidClientId(mqttVersion, clientIdentifier)) {
108             throw new MqttIdentifierRejectedException("invalid clientIdentifier: " + clientIdentifier);
109         }
110         byte[] clientIdentifierBytes = encodeStringUtf8(clientIdentifier);
111         payloadBufferSize += 2 + clientIdentifierBytes.length;
112 
113         // Will topic and message
114         String willTopic = payload.willTopic();
115         byte[] willTopicBytes = willTopic != null ? encodeStringUtf8(willTopic) : EmptyArrays.EMPTY_BYTES;
116         byte[] willMessage = payload.willMessageInBytes();
117         byte[] willMessageBytes = willMessage != null ? willMessage : EmptyArrays.EMPTY_BYTES;
118         if (variableHeader.isWillFlag()) {
119             payloadBufferSize += 2 + willTopicBytes.length;
120             payloadBufferSize += 2 + willMessageBytes.length;
121         }
122 
123         String userName = payload.userName();
124         byte[] userNameBytes = userName != null ? encodeStringUtf8(userName) : EmptyArrays.EMPTY_BYTES;
125         if (variableHeader.hasUserName()) {
126             payloadBufferSize += 2 + userNameBytes.length;
127         }
128 
129         byte[] password = payload.passwordInBytes();
130         byte[] passwordBytes = password != null ? password : EmptyArrays.EMPTY_BYTES;
131         if (variableHeader.hasPassword()) {
132             payloadBufferSize += 2 + passwordBytes.length;
133         }
134 
135         // Fixed header
136         byte[] protocolNameBytes = mqttVersion.protocolNameBytes();
137         int variableHeaderBufferSize = 2 + protocolNameBytes.length + 4;
138         int variablePartSize = variableHeaderBufferSize + payloadBufferSize;
139         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variablePartSize);
140         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variablePartSize);
141         buf.writeByte(getFixedHeaderByte1(mqttFixedHeader));
142         writeVariableLengthInt(buf, variablePartSize);
143 
144         buf.writeShort(protocolNameBytes.length);
145         buf.writeBytes(protocolNameBytes);
146 
147         buf.writeByte(variableHeader.version());
148         buf.writeByte(getConnVariableHeaderFlag(variableHeader));
149         buf.writeShort(variableHeader.keepAliveTimeSeconds());
150 
151         // Payload
152         buf.writeShort(clientIdentifierBytes.length);
153         buf.writeBytes(clientIdentifierBytes, 0, clientIdentifierBytes.length);
154         if (variableHeader.isWillFlag()) {
155             buf.writeShort(willTopicBytes.length);
156             buf.writeBytes(willTopicBytes, 0, willTopicBytes.length);
157             buf.writeShort(willMessageBytes.length);
158             buf.writeBytes(willMessageBytes, 0, willMessageBytes.length);
159         }
160         if (variableHeader.hasUserName()) {
161             buf.writeShort(userNameBytes.length);
162             buf.writeBytes(userNameBytes, 0, userNameBytes.length);
163         }
164         if (variableHeader.hasPassword()) {
165             buf.writeShort(passwordBytes.length);
166             buf.writeBytes(passwordBytes, 0, passwordBytes.length);
167         }
168         return buf;
169     }

171     private static int getConnVariableHeaderFlag(MqttConnectVariableHeader variableHeader) {
172         int flagByte = 0;
173         if (variableHeader.hasUserName()) {
174             flagByte |= 0x80;
175         }
176         if (variableHeader.hasPassword()) {
177             flagByte |= 0x40;
178         }
179         if (variableHeader.isWillRetain()) {
180             flagByte |= 0x20;
181         }
182         flagByte |= (variableHeader.willQos() & 0x03) << 3;
183         if (variableHeader.isWillFlag()) {
184             flagByte |= 0x04;
185         }
186         if (variableHeader.isCleanSession()) {
187             flagByte |= 0x02;
188         }
189         return flagByte;
190     }

192     private static ByteBuf encodeConnAckMessage(
193             ByteBufAllocator byteBufAllocator,
194             MqttConnAckMessage message) {
195         ByteBuf buf = byteBufAllocator.buffer(4);
196         buf.writeByte(getFixedHeaderByte1(message.fixedHeader()));
197         buf.writeByte(2);
198         buf.writeByte(message.variableHeader().isSessionPresent() ? 0x01 : 0x00);
199         buf.writeByte(message.variableHeader().connectReturnCode().byteValue());
200 
201         return buf;
202     }

204     private static ByteBuf encodeSubscribeMessage(
205             ByteBufAllocator byteBufAllocator,
206             MqttSubscribeMessage message) {
207         int variableHeaderBufferSize = 2;
208         int payloadBufferSize = 0;
209 
210         MqttFixedHeader mqttFixedHeader = message.fixedHeader();
211         MqttMessageIdVariableHeader variableHeader = message.variableHeader();
212         MqttSubscribePayload payload = message.payload();
213 
214         for (MqttTopicSubscription topic : payload.topicSubscriptions()) {
215             String topicName = topic.topicName();
216             byte[] topicNameBytes = encodeStringUtf8(topicName);
217             payloadBufferSize += 2 + topicNameBytes.length;
218             payloadBufferSize += 1;
219         }
220 
221         int variablePartSize = variableHeaderBufferSize + payloadBufferSize;
222         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variablePartSize);
223 
224         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variablePartSize);
225         buf.writeByte(getFixedHeaderByte1(mqttFixedHeader));
226         writeVariableLengthInt(buf, variablePartSize);
227 
228         // Variable Header
229         int messageId = variableHeader.messageId();
230         buf.writeShort(messageId);
231 
232         // Payload
233         for (MqttTopicSubscription topic : payload.topicSubscriptions()) {
234             String topicName = topic.topicName();
235             byte[] topicNameBytes = encodeStringUtf8(topicName);
236             buf.writeShort(topicNameBytes.length);
237             buf.writeBytes(topicNameBytes, 0, topicNameBytes.length);
238             buf.writeByte(topic.qualityOfService().value());
239         }
240 
241         return buf;
242     }

245             ByteBufAllocator byteBufAllocator,
246             MqttUnsubscribeMessage message) {
247         int variableHeaderBufferSize = 2;
248         int payloadBufferSize = 0;
249 
250         MqttFixedHeader mqttFixedHeader = message.fixedHeader();
251         MqttMessageIdVariableHeader variableHeader = message.variableHeader();
252         MqttUnsubscribePayload payload = message.payload();
253 
254         for (String topicName : payload.topics()) {
255             byte[] topicNameBytes = encodeStringUtf8(topicName);
256             payloadBufferSize += 2 + topicNameBytes.length;
257         }
258 
259         int variablePartSize = variableHeaderBufferSize + payloadBufferSize;
260         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variablePartSize);
261 
262         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variablePartSize);
263         buf.writeByte(getFixedHeaderByte1(mqttFixedHeader));
264         writeVariableLengthInt(buf, variablePartSize);
265 
266         // Variable Header
267         int messageId = variableHeader.messageId();
268         buf.writeShort(messageId);
269 
270         // Payload
271         for (String topicName : payload.topics()) {
272             byte[] topicNameBytes = encodeStringUtf8(topicName);
273             buf.writeShort(topicNameBytes.length);
274             buf.writeBytes(topicNameBytes, 0, topicNameBytes.length);
275         }

280     private static ByteBuf encodeSubAckMessage(
281             ByteBufAllocator byteBufAllocator,
282             MqttSubAckMessage message) {
283         int variableHeaderBufferSize = 2;
284         int payloadBufferSize = message.payload().grantedQoSLevels().size();
285         int variablePartSize = variableHeaderBufferSize + payloadBufferSize;
286         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variablePartSize);
287         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variablePartSize);
288         buf.writeByte(getFixedHeaderByte1(message.fixedHeader()));
289         writeVariableLengthInt(buf, variablePartSize);
290         buf.writeShort(message.variableHeader().messageId());
291         for (int qos : message.payload().grantedQoSLevels()) {
292             buf.writeByte(qos);
293         }
294 
295         return buf;
296     }

298     private static ByteBuf encodePublishMessage(
299             ByteBufAllocator byteBufAllocator,
300             MqttPublishMessage message) {
301         MqttFixedHeader mqttFixedHeader = message.fixedHeader();
302         MqttPublishVariableHeader variableHeader = message.variableHeader();
303         ByteBuf payload = message.payload().duplicate();
304 
305         String topicName = variableHeader.topicName();
306         byte[] topicNameBytes = encodeStringUtf8(topicName);
307 
308         int variableHeaderBufferSize = 2 + topicNameBytes.length +
309                 (mqttFixedHeader.qosLevel().value() > 0 ? 2 : 0);
310         int payloadBufferSize = payload.readableBytes();
311         int variablePartSize = variableHeaderBufferSize + payloadBufferSize;
312         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variablePartSize);
313 
314         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variablePartSize);
315         buf.writeByte(getFixedHeaderByte1(mqttFixedHeader));
316         writeVariableLengthInt(buf, variablePartSize);
317         buf.writeShort(topicNameBytes.length);
318         buf.writeBytes(topicNameBytes);
319         if (mqttFixedHeader.qosLevel().value() > 0) {
320             buf.writeShort(variableHeader.messageId());
321         }
322         buf.writeBytes(payload);
323 
324         return buf;
325     }

327     private static ByteBuf encodeMessageWithOnlySingleByteFixedHeaderAndMessageId(
328             ByteBufAllocator byteBufAllocator,
329             MqttMessage message) {
330         MqttFixedHeader mqttFixedHeader = message.fixedHeader();
331         MqttMessageIdVariableHeader variableHeader = (MqttMessageIdVariableHeader) message.variableHeader();
332         int msgId = variableHeader.messageId();
333 
334         int variableHeaderBufferSize = 2; // variable part only has a message id
335         int fixedHeaderBufferSize = 1 + getVariableLengthInt(variableHeaderBufferSize);
336         ByteBuf buf = byteBufAllocator.buffer(fixedHeaderBufferSize + variableHeaderBufferSize);
337         buf.writeByte(getFixedHeaderByte1(mqttFixedHeader));
338         writeVariableLengthInt(buf, variableHeaderBufferSize);
339         buf.writeShort(msgId);
340 
341         return buf;
342     }

344     private static ByteBuf encodeMessageWithOnlySingleByteFixedHeader(
345             ByteBufAllocator byteBufAllocator,
346             MqttMessage message) {
347         MqttFixedHeader mqttFixedHeader = message.fixedHeader();

352         return buf;

355     private static int getFixedHeaderByte1(MqttFixedHeader header) {
356         int ret = 0;
357         ret |= header.messageType().value() << 4;
358         if (header.isDup()) {
359             ret |= 0x08;
360         }
361         ret |= header.qosLevel().value() << 1;
362         if (header.isRetain()) {
363             ret |= 0x01;
364         }
365         return ret;
366     }

368     private static void writeVariableLengthInt(ByteBuf buf, int num) {
369         do {
370             int digit = num % 128;
371             num /= 128;
372             if (num > 0) {
373                 digit |= 0x80;
374             }
375             buf.writeByte(digit);
376         } while (num > 0);
377     }
378 
379     private static int getVariableLengthInt(int num) {
380         int count = 0;
381         do {
382             num /= 128;
383             count++;
384         } while (num > 0);
385         return count;
386     }
