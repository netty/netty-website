16  package io.netty.handler.codec.socksx.v4;

18  import io.netty.buffer.ByteBuf;
19  import io.netty.util.CharsetUtil;
20  import io.netty.util.NetUtil;

22  import java.net.IDN;
23  
24  /**
25   * An socksv4a cmd request.
26   *
27   * @see Socks4Response
28   * @see Socks4CmdRequestDecoder
29   */
30  public final class Socks4CmdRequest extends Socks4Request {
31      private final String userId;
32      private final Socks4CmdType cmdType;
33      private final String host;
34      private final int port;
35  
36      private static final byte[] IPv4_DOMAIN_MARKER = {0x00, 0x00, 0x00, 0x01};
37  
38      public Socks4CmdRequest(String userId, Socks4CmdType cmdType, String host, int port) {
39              if (userId == null) {
40                  throw new NullPointerException("username");
41              }
42              if (cmdType == null) {
43                  throw new NullPointerException("cmdType");
44              }
45              if (host == null) {
46                  throw new NullPointerException("host");
47              }
48              if (port <= 0 || port >= 65536) {
49                  throw new IllegalArgumentException(port + " is not in bounds 0 < x < 65536");
50              }
51              this.userId = userId;
52              this.cmdType = cmdType;
53              this.host = IDN.toASCII(host);
54              this.port = port;
55          }
56  
57      public Socks4CmdRequest(Socks4CmdType cmdType, String host, int port) {
58          this("", cmdType, host, port);
59      }
60  
61      /**
62       * Returns the {@link Socks4CmdType} of this {@link Socks4Request}
63       *
64       * @return The {@link Socks4CmdType} of this {@link Socks4Request}
65       */
66      public Socks4CmdType cmdType() {
67          return cmdType;
68      }
69  
70      /**
71       * Returns host that is used as a parameter in {@link Socks4CmdType}
72       *
73       * @return host that is used as a parameter in {@link Socks4CmdType}
74       */
75      public String host() {
76          return IDN.toUnicode(host);
77      }
78  
79      /**
80       * Returns userId that is used as a parameter in {@link Socks4CmdType}
81       *
82       * @return userId that is used as a parameter in {@link Socks4CmdType}
83       */
84      public String userId() {
85          return userId;
86      }
87  
88      /**
89       * Returns port that is used as a parameter in {@link Socks4CmdType}
90       *
91       * @return port that is used as a parameter in {@link Socks4CmdType}
92       */
93      public int port() {
94          return port;
95      }
96  
97      @Override
98      void encodeAsByteBuf(ByteBuf byteBuf) {
99          byteBuf.writeByte(protocolVersion().byteValue());
100         byteBuf.writeByte(cmdType.byteValue());
101         byteBuf.writeShort(port);
102         if (NetUtil.isValidIpV4Address(host)) {
103             byteBuf.writeBytes(NetUtil.createByteArrayFromIpAddressString(host));
104             byteBuf.writeBytes(userId.getBytes());
105             byteBuf.writeZero(1);
106         } else {
107             byteBuf.writeBytes(IPv4_DOMAIN_MARKER);
108             byteBuf.writeBytes(userId.getBytes());
109             byteBuf.writeZero(1);
110             byteBuf.writeBytes(host.getBytes(CharsetUtil.US_ASCII));
111             byteBuf.writeZero(1);
112         }
113     }
