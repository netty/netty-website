128     @Override
129     public ChannelFuture writeData(ChannelHandlerContext ctx, int streamId, ByteBuf data,
130             int padding, boolean endStream, ChannelPromise promise) {
131         final SimpleChannelPromiseAggregator promiseAggregator =
132                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
133         final DataFrameHeader header = new DataFrameHeader(ctx, streamId);
134         boolean needToReleaseHeaders = true;
135         boolean needToReleaseData = true;
136         try {
137             verifyStreamId(streamId, STREAM_ID);
138             verifyPadding(padding);
139 
140             boolean lastFrame;
141             int remainingData = data.readableBytes();
142             do {
143                 // Determine how much data and padding to write in this frame. Put all padding at the end.
144                 int frameDataBytes = min(remainingData, maxFrameSize);
145                 int framePaddingBytes = min(padding, max(0, (maxFrameSize - 1) - frameDataBytes));
146 
147                 // Decrement the remaining counters.
148                 padding -= framePaddingBytes;
149                 remainingData -= frameDataBytes;
150 
151                 // Determine whether or not this is the last frame to be sent.
152                 lastFrame = remainingData == 0 && padding == 0;
153 
154                 // Only the last frame is not retained. Until then, the outer finally must release.
155                 ByteBuf frameHeader = header.slice(frameDataBytes, framePaddingBytes, lastFrame && endStream);
156                 needToReleaseHeaders = !lastFrame;
157                 ctx.write(lastFrame ? frameHeader : frameHeader.retain(), promiseAggregator.newPromise());
158 
159                 // Write the frame data.
160                 ByteBuf frameData = data.readSlice(frameDataBytes);
161                 // Only the last frame is not retained. Until then, the outer finally must release.
162                 needToReleaseData = !lastFrame;
163                 ctx.write(lastFrame ? frameData : frameData.retain(), promiseAggregator.newPromise());
164 
165                 // Write the frame padding.
166                 if (framePaddingBytes > 0) {
167                     ctx.write(ZERO_BUFFER.slice(0, framePaddingBytes), promiseAggregator.newPromise());
168                 }
169             } while (!lastFrame);
170         } catch (Throwable t) {
171             if (needToReleaseHeaders) {
172                 header.release();
173             }
174             if (needToReleaseData) {
175                 data.release();
176             }
177             promiseAggregator.setFailure(t);
178         }
179         return promiseAggregator.doneAllocatingPromises();
180     }

287     @Override
288     public ChannelFuture writePushPromise(ChannelHandlerContext ctx, int streamId,
289             int promisedStreamId, Http2Headers headers, int padding, ChannelPromise promise) {
290         ByteBuf headerBlock = null;
291         SimpleChannelPromiseAggregator promiseAggregator =
292                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
293         try {
294             verifyStreamId(streamId, STREAM_ID);
295             verifyStreamId(promisedStreamId, "Promised Stream ID");
296             verifyPadding(padding);
297 
298             // Encode the entire header block into an intermediate buffer.
299             headerBlock = ctx.alloc().buffer();
300             headersEncoder.encodeHeaders(headers, headerBlock);
301 
302             // Read the first fragment (possibly everything).
303             Http2Flags flags = new Http2Flags().paddingPresent(padding > 0);
304             // INT_FIELD_LENGTH is for the length of the promisedStreamId
305             int nonFragmentLength = INT_FIELD_LENGTH + padding + flags.getPaddingPresenceFieldLength();
306             int maxFragmentLength = maxFrameSize - nonFragmentLength;
307             ByteBuf fragment = headerBlock.readRetainedSlice(min(headerBlock.readableBytes(), maxFragmentLength));
308 
309             flags.endOfHeaders(!headerBlock.isReadable());
310 
311             int payloadLength = fragment.readableBytes() + nonFragmentLength;
312             ByteBuf buf = ctx.alloc().buffer(PUSH_PROMISE_FRAME_HEADER_LENGTH);
313             writeFrameHeaderInternal(buf, payloadLength, PUSH_PROMISE, flags, streamId);
314             writePaddingLength(buf, padding);
315 
316             // Write out the promised stream ID.
317             buf.writeInt(promisedStreamId);
318             ctx.write(buf, promiseAggregator.newPromise());
319 
320             // Write the first fragment.
321             ctx.write(fragment, promiseAggregator.newPromise());
322 
323             if (padding > 0) { // Write out the padding, if any.
324                 ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
325             }
326 
327             if (!flags.endOfHeaders()) {
328                 writeContinuationFrames(ctx, streamId, headerBlock, padding, promiseAggregator);
329             }
330         } catch (Throwable t) {
331             promiseAggregator.setFailure(t);
332         } finally {
333             if (headerBlock != null) {
334                 headerBlock.release();
335             }
336         }
337         return promiseAggregator.doneAllocatingPromises();
338     }

407     private ChannelFuture writeHeadersInternal(ChannelHandlerContext ctx,
408             int streamId, Http2Headers headers, int padding, boolean endStream,
409             boolean hasPriority, int streamDependency, short weight, boolean exclusive, ChannelPromise promise) {
410         ByteBuf headerBlock = null;
411         SimpleChannelPromiseAggregator promiseAggregator =
412                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
413         try {
414             verifyStreamId(streamId, STREAM_ID);
415             if (hasPriority) {
416                 verifyStreamOrConnectionId(streamDependency, STREAM_DEPENDENCY);
417                 verifyPadding(padding);
418                 verifyWeight(weight);
419             }
420 
421             // Encode the entire header block.
422             headerBlock = ctx.alloc().buffer();
423             headersEncoder.encodeHeaders(headers, headerBlock);
424 
425             Http2Flags flags =
426                     new Http2Flags().endOfStream(endStream).priorityPresent(hasPriority).paddingPresent(padding > 0);
427 
428             // Read the first fragment (possibly everything).
429             int nonFragmentBytes = padding + flags.getNumPriorityBytes() + flags.getPaddingPresenceFieldLength();
430             int maxFragmentLength = maxFrameSize - nonFragmentBytes;
431             ByteBuf fragment = headerBlock.readRetainedSlice(min(headerBlock.readableBytes(), maxFragmentLength));
432 
433             // Set the end of headers flag for the first frame.
434             flags.endOfHeaders(!headerBlock.isReadable());
435 
436             int payloadLength = fragment.readableBytes() + nonFragmentBytes;
437             ByteBuf buf = ctx.alloc().buffer(HEADERS_FRAME_HEADER_LENGTH);
438             writeFrameHeaderInternal(buf, payloadLength, HEADERS, flags, streamId);
439             writePaddingLength(buf, padding);
440 
441             if (hasPriority) {
442                 long word1 = exclusive ? 0x80000000L | streamDependency : streamDependency;
443                 writeUnsignedInt(word1, buf);
444 
445                 // Adjust the weight so that it fits into a single byte on the wire.
446                 buf.writeByte(weight - 1);
447             }
448             ctx.write(buf, promiseAggregator.newPromise());
449 
450             // Write the first fragment.
451             ctx.write(fragment, promiseAggregator.newPromise());
452 
453             if (padding > 0) { // Write out the padding, if any.
454                 ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
455             }
456 
457             if (!flags.endOfHeaders()) {
458                 writeContinuationFrames(ctx, streamId, headerBlock, padding, promiseAggregator);
459             }
460         } catch (Throwable t) {
461             promiseAggregator.setFailure(t);
462         } finally {
463             if (headerBlock != null) {
464                 headerBlock.release();
465             }
466         }
467         return promiseAggregator.doneAllocatingPromises();
468     }

470     /**
471      * Writes as many continuation frames as needed until {@code padding} and {@code headerBlock} are consumed.
472      */
473     private ChannelFuture writeContinuationFrames(ChannelHandlerContext ctx, int streamId,
474             ByteBuf headerBlock, int padding, SimpleChannelPromiseAggregator promiseAggregator) {
475         Http2Flags flags = new Http2Flags().paddingPresent(padding > 0);
476         int nonFragmentLength = padding + flags.getPaddingPresenceFieldLength();
477         int maxFragmentLength = maxFrameSize - nonFragmentLength;
478         // TODO: same padding is applied to all frames, is this desired?
479         if (maxFragmentLength <= 0) {
480             return promiseAggregator.setFailure(new IllegalArgumentException(
481                     "Padding [" + padding + "] is too large for max frame size [" + maxFrameSize + "]"));
482         }
483 
484         if (headerBlock.isReadable()) {
485             // The frame header (and padding) only changes on the last frame, so allocate it once and re-use
486             int fragmentReadableBytes = min(headerBlock.readableBytes(), maxFragmentLength);
487             int payloadLength = fragmentReadableBytes + nonFragmentLength;
488             ByteBuf buf = ctx.alloc().buffer(CONTINUATION_FRAME_HEADER_LENGTH);
489             writeFrameHeaderInternal(buf, payloadLength, CONTINUATION, flags, streamId);
490             writePaddingLength(buf, padding);
491 
492             do {
493                 fragmentReadableBytes = min(headerBlock.readableBytes(), maxFragmentLength);
494                 ByteBuf fragment = headerBlock.readRetainedSlice(fragmentReadableBytes);
495 
496                 payloadLength = fragmentReadableBytes + nonFragmentLength;
497                 if (headerBlock.isReadable()) {
498                     ctx.write(buf.retain(), promiseAggregator.newPromise());
499                 } else {
500                     // The frame header is different for the last frame, so re-allocate and release the old buffer
501                     flags = flags.endOfHeaders(true);
502                     buf.release();
503                     buf = ctx.alloc().buffer(CONTINUATION_FRAME_HEADER_LENGTH);
504                     writeFrameHeaderInternal(buf, payloadLength, CONTINUATION, flags, streamId);
505                     writePaddingLength(buf, padding);
506                     ctx.write(buf, promiseAggregator.newPromise());
507                 }
508 
509                 ctx.write(fragment, promiseAggregator.newPromise());
510 
511                 // Write out the padding, if any.
512                 if (padding > 0) {
513                     ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
514                 }
515             } while(headerBlock.isReadable());
516         }
517         return promiseAggregator;
518     }

520     private static void writePaddingLength(ByteBuf buf, int paddingLength) {
521         if (paddingLength > 0) {
522             // It is assumed that the padding length has been bounds checked before this
523             buf.writeByte(paddingLength);
524         }
525     }

569     /**
570      * Utility class that manages the creation of frame header buffers for {@code DATA} frames. Attempts
571      * to reuse the same buffer repeatedly when splitting data into multiple frames.
572      */
573     private static final class DataFrameHeader {
574         private final int streamId;
575         private final ByteBuf buffer;
576         private final Http2Flags flags = new Http2Flags();
577         private int prevData;
578         private int prevPadding;
579         private ByteBuf frameHeader;
580 
581         DataFrameHeader(ChannelHandlerContext ctx, int streamId) {
582             // All padding will be put at the end, so in the worst case we need 3 headers:
583             // a repeated no-padding frame of maxFrameSize, a frame that has part data and part
584             // padding, and a frame that has the remainder of the padding.
585             buffer = ctx.alloc().buffer(3 * DATA_FRAME_HEADER_LENGTH);
586             this.streamId = streamId;
587         }
588 
589         /**
590          * Gets the frame header buffer configured for the current frame.
591          */
592         ByteBuf slice(int data, int padding, boolean endOfStream) {
593             // Since we're reusing the current frame header whenever possible, check if anything changed
594             // that requires a new header.
595             if (data != prevData || padding != prevPadding
596                     || endOfStream != flags.endOfStream() || frameHeader == null) {
