57  /**
58   * An HPACK encoder.
59   *
60   * <p>Implementation note:  This class is security sensitive, and depends on users correctly identifying their headers
61   * as security sensitive or not.  If a header is considered not sensitive, methods names "insensitive" are used which
62   * are fast, but don't provide any security guarantees.
63   */
64  final class HpackEncoder {
65      static final int HUFF_CODE_THRESHOLD = 512;
66      // a linked hash map of header fields
67      private final HeaderEntry[] headerFields;
68      private final HeaderEntry head = new HeaderEntry(-1, AsciiString.EMPTY_STRING,
69              AsciiString.EMPTY_STRING, Integer.MAX_VALUE, null);
70      private final HpackHuffmanEncoder hpackHuffmanEncoder = new HpackHuffmanEncoder();
71      private final byte hashMask;
72      private final boolean ignoreMaxHeaderListSize;
73      private final int huffCodeThreshold;
74      private long size;
75      private long maxHeaderTableSize;
76      private long maxHeaderListSize;

82          this(false);

86       * Creates a new encoder.

89          this(ignoreMaxHeaderListSize, 16, HUFF_CODE_THRESHOLD);

92      /**
93       * Creates a new encoder.
94       */
95      HpackEncoder(boolean ignoreMaxHeaderListSize, int arraySizeHint, int huffCodeThreshold) {
96          this.ignoreMaxHeaderListSize = ignoreMaxHeaderListSize;
97          maxHeaderTableSize = DEFAULT_HEADER_TABLE_SIZE;
98          maxHeaderListSize = MAX_HEADER_LIST_SIZE;
99          // Enforce a bound of [2, 128] because hashMask is a byte. The max possible value of hashMask is one less
100         // than the length of this array, and we want the mask to be > 0.
101         headerFields = new HeaderEntry[findNextPositivePowerOfTwo(max(2, min(arraySizeHint, 128)))];
102         hashMask = (byte) (headerFields.length - 1);
103         head.before = head.after = head;
104         this.huffCodeThreshold = huffCodeThreshold;
105     }

107     /**
108      * Encode the header field into the header block.
109      *
110      * <strong>The given {@link CharSequence}s must be immutable!</strong>
111      */
112     public void encodeHeaders(int streamId, ByteBuf out, Http2Headers headers, SensitivityDetector sensitivityDetector)
113             throws Http2Exception {
114         if (ignoreMaxHeaderListSize) {
115             encodeHeadersIgnoreMaxHeaderListSize(out, headers, sensitivityDetector);
116         } else {
117             encodeHeadersEnforceMaxHeaderListSize(streamId, out, headers, sensitivityDetector);
118         }
119     }

121     private void encodeHeadersEnforceMaxHeaderListSize(int streamId, ByteBuf out, Http2Headers headers,
122                                                        SensitivityDetector sensitivityDetector)
123             throws Http2Exception {
124         long headerSize = 0;
125         // To ensure we stay consistent with our peer check the size is valid before we potentially modify HPACK state.
126         for (Map.Entry<CharSequence, CharSequence> header : headers) {
127             CharSequence name = header.getKey();
128             CharSequence value = header.getValue();
129             // OK to increment now and check for bounds after because this value is limited to unsigned int and will not
130             // overflow.
131             headerSize += HpackHeaderField.sizeOf(name, value);
132             if (headerSize > maxHeaderListSize) {
133                 headerListSizeExceeded(streamId, maxHeaderListSize, false);
134             }
135         }
136         encodeHeadersIgnoreMaxHeaderListSize(out, headers, sensitivityDetector);
137     }

139     private void encodeHeadersIgnoreMaxHeaderListSize(ByteBuf out, Http2Headers headers,
140                                                       SensitivityDetector sensitivityDetector) throws Http2Exception {
141         for (Map.Entry<CharSequence, CharSequence> header : headers) {
142             CharSequence name = header.getKey();
143             CharSequence value = header.getValue();
144             encodeHeader(out, name, value, sensitivityDetector.isSensitive(name, value),
145                          HpackHeaderField.sizeOf(name, value));
146         }
147     }

149     /**
150      * Encode the header field into the header block.
151      *
152      * <strong>The given {@link CharSequence}s must be immutable!</strong>
153      */
154     private void encodeHeader(ByteBuf out, CharSequence name, CharSequence value, boolean sensitive, long headerSize) {
155         // If the header value is sensitive then it must never be indexed
156         if (sensitive) {
157             int nameIndex = getNameIndex(name);
158             encodeLiteral(out, name, value, IndexType.NEVER, nameIndex);
159             return;
160         }
161 
162         // If the peer will only use the static table
163         if (maxHeaderTableSize == 0) {
164             int staticTableIndex = HpackStaticTable.getIndexInsensitive(name, value);
165             if (staticTableIndex == -1) {
166                 int nameIndex = HpackStaticTable.getIndex(name);
167                 encodeLiteral(out, name, value, IndexType.NONE, nameIndex);
168             } else {
169                 encodeInteger(out, 0x80, 7, staticTableIndex);
170             }
171             return;
172         }
173 
174         // If the headerSize is greater than the max table size then it must be encoded literally
175         if (headerSize > maxHeaderTableSize) {
176             int nameIndex = getNameIndex(name);
177             encodeLiteral(out, name, value, IndexType.NONE, nameIndex);
178             return;
179         }
180 
181         HeaderEntry headerField = getEntryInsensitive(name, value);
182         if (headerField != null) {
183             int index = getIndex(headerField.index) + HpackStaticTable.length;
184             // Section 6.1. Indexed Header Field Representation
185             encodeInteger(out, 0x80, 7, index);
186         } else {
187             int staticTableIndex = HpackStaticTable.getIndexInsensitive(name, value);
188             if (staticTableIndex != -1) {
189                 // Section 6.1. Indexed Header Field Representation
190                 encodeInteger(out, 0x80, 7, staticTableIndex);
191             } else {
192                 ensureCapacity(headerSize);
193                 encodeLiteral(out, name, value, IndexType.INCREMENTAL, getNameIndex(name));
194                 add(name, value, headerSize);
195             }
196         }
197     }

199     /**
200      * Set the maximum table size.
201      */
202     public void setMaxHeaderTableSize(ByteBuf out, long maxHeaderTableSize) throws Http2Exception {
203         if (maxHeaderTableSize < MIN_HEADER_TABLE_SIZE || maxHeaderTableSize > MAX_HEADER_TABLE_SIZE) {
204             throw connectionError(PROTOCOL_ERROR, "Header Table Size must be >= %d and <= %d but was %d",
205                     MIN_HEADER_TABLE_SIZE, MAX_HEADER_TABLE_SIZE, maxHeaderTableSize);
206         }
207         if (this.maxHeaderTableSize == maxHeaderTableSize) {
208             return;
209         }
210         this.maxHeaderTableSize = maxHeaderTableSize;
211         ensureCapacity(0);
212         // Casting to integer is safe as we verified the maxHeaderTableSize is a valid unsigned int.
213         encodeInteger(out, 0x20, 5, maxHeaderTableSize);
214     }

236      * Encode integer according to <a href="https://tools.ietf.org/html/rfc7541#section-5.1">Section 5.1</a>.

239         encodeInteger(out, mask, n, (long) i);
240     }

242     /**
243      * Encode integer according to <a href="https://tools.ietf.org/html/rfc7541#section-5.1">Section 5.1</a>.
244      */
245     private static void encodeInteger(ByteBuf out, int mask, int n, long i) {
246         assert n >= 0 && n <= 8 : "N: " + n;
247         int nbits = 0xFF >>> (8 - n);
248         if (i < nbits) {
249             out.writeByte((int) (mask | i));
250         } else {
251             out.writeByte(mask | nbits);
252             long length = i - nbits;
253             for (; (length & ~0x7F) != 0; length >>>= 7) {
254                 out.writeByte((int) ((length & 0x7F) | 0x80));
255             }
256             out.writeByte((int) length);
257         }
258     }

260     /**
261      * Encode string literal according to Section 5.2.
262      */
263     private void encodeStringLiteral(ByteBuf out, CharSequence string) {
264         int huffmanLength;
265         if (string.length() >= huffCodeThreshold
266                 && (huffmanLength = hpackHuffmanEncoder.getEncodedLength(string)) < string.length()) {
267             encodeInteger(out, 0x80, 7, huffmanLength);
268             hpackHuffmanEncoder.encode(out, string);
269         } else {
270             encodeInteger(out, 0x00, 7, string.length());
271             if (string instanceof AsciiString) {
272                 // Fast-path
273                 AsciiString asciiString = (AsciiString) string;
274                 out.writeBytes(asciiString.array(), asciiString.arrayOffset(), asciiString.length());
275             } else {
276                 // Only ASCII is allowed in http2 headers, so its fine to use this.
277                 // https://tools.ietf.org/html/rfc7540#section-8.1.2
278                 out.writeCharSequence(string, CharsetUtil.ISO_8859_1);
279             }
280         }
281     }

283     /**
284      * Encode literal header field according to Section 6.2.
285      */
286     private void encodeLiteral(ByteBuf out, CharSequence name, CharSequence value, IndexType indexType,
287                                int nameIndex) {
288         boolean nameIndexValid = nameIndex != -1;
289         switch (indexType) {
290             case INCREMENTAL:
291                 encodeInteger(out, 0x40, 6, nameIndexValid ? nameIndex : 0);
292                 break;
293             case NONE:
294                 encodeInteger(out, 0x00, 4, nameIndexValid ? nameIndex : 0);
295                 break;
296             case NEVER:
297                 encodeInteger(out, 0x10, 4, nameIndexValid ? nameIndex : 0);
298                 break;
299             default:
300                 throw new Error("should not reach here");
301         }
302         if (!nameIndexValid) {
303             encodeStringLiteral(out, name);
304         }
305         encodeStringLiteral(out, value);
306     }

308     private int getNameIndex(CharSequence name) {
309         int index = HpackStaticTable.getIndex(name);
310         if (index == -1) {
311             index = getIndex(name);
312             if (index >= 0) {
313                 index += HpackStaticTable.length;
314             }
315         }
316         return index;
317     }

319     /**
320      * Ensure that the dynamic table has enough room to hold 'headerSize' more bytes. Removes the
321      * oldest entry from the dynamic table until sufficient space is available.
322      */
323     private void ensureCapacity(long headerSize) {
324         while (maxHeaderTableSize - size < headerSize) {
325             int index = length();
326             if (index == 0) {
327                 break;
328             }
329             remove();
330         }
331     }

358     /**
359      * Returns the header entry with the lowest index value for the header field. Returns null if
360      * header field is not in the dynamic table.
361      */
362     private HeaderEntry getEntryInsensitive(CharSequence name, CharSequence value) {
363         if (length() == 0 || name == null || value == null) {
364             return null;
365         }
366         int h = AsciiString.hashCode(name);
367         int i = index(h);
368         for (HeaderEntry e = headerFields[i]; e != null; e = e.next) {
369             // Check the value before then name, as it is more likely the value will be different incase there is no
370             // match.
371             if (e.hash == h && equalsVariableTime(value, e.value) && equalsVariableTime(name, e.name)) {
372                 return e;
373             }
374         }
375         return null;
376     }

378     /**
379      * Returns the lowest index value for the header field name in the dynamic table. Returns -1 if
380      * the header field name is not in the dynamic table.
381      */
382     private int getIndex(CharSequence name) {
383         if (length() == 0 || name == null) {
384             return -1;
385         }
386         int h = AsciiString.hashCode(name);
387         int i = index(h);
388         for (HeaderEntry e = headerFields[i]; e != null; e = e.next) {
389             if (e.hash == h && equalsConstantTime(name, e.name) != 0) {
390                 return getIndex(e.index);
391             }
392         }
393         return -1;
394     }

396     /**
397      * Compute the index into the dynamic table given the index in the header entry.
398      */
399     private int getIndex(int index) {
400         return index == -1 ? -1 : index - head.before.index + 1;
401     }

403     /**
404      * Add the header field to the dynamic table. Entries are evicted from the dynamic table until
405      * the size of the table and the new header field is less than the table's maxHeaderTableSize. If the size
406      * of the new entry is larger than the table's maxHeaderTableSize, the dynamic table will be cleared.
407      */
408     private void add(CharSequence name, CharSequence value, long headerSize) {
409         // Clear the table if the header field size is larger than the maxHeaderTableSize.
410         if (headerSize > maxHeaderTableSize) {
411             clear();
412             return;
413         }
414 
415         // Evict oldest entries until we have enough maxHeaderTableSize.
416         while (maxHeaderTableSize - size < headerSize) {
417             remove();
418         }
419 
420         int h = AsciiString.hashCode(name);
421         int i = index(h);
422         HeaderEntry old = headerFields[i];
423         HeaderEntry e = new HeaderEntry(h, name, value, head.before.index - 1, old);
424         headerFields[i] = e;
425         e.addBefore(head);
426         size += headerSize;
427     }

429     /**
430      * Remove and return the oldest header field from the dynamic table.
431      */
432     private HpackHeaderField remove() {
433         if (size == 0) {
434             return null;
435         }
436         HeaderEntry eldest = head.after;
437         int h = eldest.hash;
438         int i = index(h);
439         HeaderEntry prev = headerFields[i];
440         HeaderEntry e = prev;
441         while (e != null) {
442             HeaderEntry next = e.next;
443             if (e == eldest) {
444                 if (prev == eldest) {
445                     headerFields[i] = next;
446                 } else {
447                     prev.next = next;
448                 }
449                 eldest.remove();
450                 size -= eldest.size();
451                 return eldest;
452             }
453             prev = e;
454             e = next;
455         }
456         return null;
457     }

459     /**
460      * Remove all entries from the dynamic table.
461      */
462     private void clear() {
463         Arrays.fill(headerFields, null);
464         head.before = head.after = head;
465         size = 0;
466     }

475     /**
476      * A linked hash map HpackHeaderField entry.
477      */
478     private static final class HeaderEntry extends HpackHeaderField {
479         // These fields comprise the doubly linked list used for iteration.
480         HeaderEntry before, after;
481 
482         // These fields comprise the chained list for header fields with the same hash.
483         HeaderEntry next;
484         int hash;
485 
486         // This is used to compute the index in the dynamic table.
487         int index;
488 
489         /**
490          * Creates new entry.
491          */
492         HeaderEntry(int hash, CharSequence name, CharSequence value, int index, HeaderEntry next) {
493             super(name, value);
494             this.index = index;
495             this.hash = hash;
496             this.next = next;
497         }
498 
499         /**
500          * Removes this entry from the linked list.
501          */
502         private void remove() {
503             before.after = after;
504             after.before = before;
505             before = null; // null references to prevent nepotism in generational GC.
506             after = null;
507             next = null;
508         }
509 
510         /**
511          * Inserts this entry before the specified existing entry in the list.
512          */
513         private void addBefore(HeaderEntry existingEntry) {
514             after = existingEntry;
515             before = existingEntry.before;
516             before.after = this;
517             after.before = this;
518         }
519     }
