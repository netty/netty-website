80      @Override
81      public void remoteSettings(Http2Settings settings) throws Http2Exception {
82          Boolean pushEnabled = settings.pushEnabled();
83          Http2FrameWriter.Configuration config = configuration();
84          Http2HeadersEncoder.Configuration outboundHeaderConfig = config.headersConfiguration();
85          Http2FrameSizePolicy outboundFrameSizePolicy = config.frameSizePolicy();
86          if (pushEnabled != null) {
87              if (!connection.isServer() && pushEnabled) {
88                  throw connectionError(PROTOCOL_ERROR,
89                      "Client received a value of ENABLE_PUSH specified to other than 0");
90              }
91              connection.remote().allowPushTo(pushEnabled);
92          }
93  
94          Long maxConcurrentStreams = settings.maxConcurrentStreams();
95          if (maxConcurrentStreams != null) {
96              connection.local().maxActiveStreams((int) min(maxConcurrentStreams, MAX_VALUE));
97          }
98  
99          Long headerTableSize = settings.headerTableSize();
100         if (headerTableSize != null) {
101             outboundHeaderConfig.maxHeaderTableSize((int) min(headerTableSize, MAX_VALUE));
102         }
103 
104         Long maxHeaderListSize = settings.maxHeaderListSize();
105         if (maxHeaderListSize != null) {
106             outboundHeaderConfig.maxHeaderListSize(maxHeaderListSize);
107         }
108 
109         Integer maxFrameSize = settings.maxFrameSize();
110         if (maxFrameSize != null) {
111             outboundFrameSizePolicy.maxFrameSize(maxFrameSize);
112         }
113 
114         Integer initialWindowSize = settings.initialWindowSize();
115         if (initialWindowSize != null) {
116             flowController().initialWindowSize(initialWindowSize);
117         }
118     }

120     @Override
121     public ChannelFuture writeData(final ChannelHandlerContext ctx, final int streamId, ByteBuf data, int padding,
122             final boolean endOfStream, ChannelPromise promise) {
123         final Http2Stream stream;
124         try {
125             stream = requireStream(streamId);
126 
127             // Verify that the stream is in the appropriate state for sending DATA frames.
128             switch (stream.state()) {
129                 case OPEN:
130                 case HALF_CLOSED_REMOTE:
131                     // Allowed sending DATA frames in these states.
132                     break;
133                 default:
134                     throw new IllegalStateException("Stream " + stream.id() + " in unexpected state " + stream.state());
135             }
136         } catch (Throwable e) {
137             data.release();
138             return promise.setFailure(e);
139         }
140 
141         // Hand control of the frame to the flow controller.
142         flowController().addFlowControlled(stream,
143                 new FlowControlledData(stream, data, padding, endOfStream, promise));
144         return promise;
145     }

153     private static boolean validateHeadersSentState(Http2Stream stream, Http2Headers headers, boolean isServer,
154                                                     boolean endOfStream) {
155         boolean isInformational = isServer && HttpStatusClass.valueOf(headers.status()) == INFORMATIONAL;
156         if ((isInformational || !endOfStream) && stream.isHeadersSent() || stream.isTrailersSent()) {
157             throw new IllegalStateException("Stream " + stream.id() + " sent too many headers EOS: " + endOfStream);
158         }
159         return isInformational;
160     }

170     /**
171      * Write headers via {@link Http2FrameWriter}. If {@code hasPriority} is {@code false} it will ignore the
172      * {@code streamDependency}, {@code weight} and {@code exclusive} parameters.
173      */
174     private static ChannelFuture sendHeaders(Http2FrameWriter frameWriter, ChannelHandlerContext ctx, int streamId,
175                                        Http2Headers headers, final boolean hasPriority,
176                                        int streamDependency, final short weight,
177                                        boolean exclusive, final int padding,
178                                        boolean endOfStream, ChannelPromise promise) {
179         if (hasPriority) {
180             return frameWriter.writeHeaders(ctx, streamId, headers, streamDependency,
181                     weight, exclusive, padding, endOfStream, promise);
182         }
183         return frameWriter.writeHeaders(ctx, streamId, headers, padding, endOfStream, promise);
184     }

186     private ChannelFuture writeHeaders0(final ChannelHandlerContext ctx, final int streamId,
187                                         final Http2Headers headers, final boolean hasPriority,
188                                         final int streamDependency, final short weight,
189                                         final boolean exclusive, final int padding,
190                                         final boolean endOfStream, ChannelPromise promise) {
191         try {
192             Http2Stream stream = connection.stream(streamId);
193             if (stream == null) {
194                 try {
195                     // We don't create the stream in a `halfClosed` state because if this is an initial
196                     // HEADERS frame we don't want the connection state to signify that the HEADERS have
197                     // been sent until after they have been encoded and placed in the outbound buffer.
198                     // Therefore, we let the `LifeCycleManager` will take care of transitioning the state
199                     // as appropriate.
200                     stream = connection.local().createStream(streamId, /*endOfStream*/ false);
201                 } catch (Http2Exception cause) {
202                     if (connection.remote().mayHaveCreatedStream(streamId)) {
203                         promise.tryFailure(new IllegalStateException("Stream no longer exists: " + streamId, cause));
204                         return promise;
205                     }
206                     throw cause;
207                 }
208             } else {
209                 switch (stream.state()) {
210                     case RESERVED_LOCAL:
211                         stream.open(endOfStream);
212                         break;
213                     case OPEN:
214                     case HALF_CLOSED_REMOTE:
215                         // Allowed sending headers in these states.
216                         break;
217                     default:
218                         throw new IllegalStateException("Stream " + stream.id() + " in unexpected state " +
219                                                         stream.state());
220                 }
221             }
222 
223             // Trailing headers must go through flow control if there are other frames queued in flow control
224             // for this stream.
225             Http2RemoteFlowController flowController = flowController();
226             if (!endOfStream || !flowController.hasFlowControlled(stream)) {
227                 // The behavior here should mirror that in FlowControlledHeaders
228 
229                 promise = promise.unvoid();
230                 boolean isInformational = validateHeadersSentState(stream, headers, connection.isServer(), endOfStream);
231 
232                 ChannelFuture future = sendHeaders(frameWriter, ctx, streamId, headers, hasPriority, streamDependency,
233                         weight, exclusive, padding, endOfStream, promise);
234 
235                 // Writing headers may fail during the encode state if they violate HPACK limits.
236                 Throwable failureCause = future.cause();
237                 if (failureCause == null) {
238                     // Synchronously set the headersSent flag to ensure that we do not subsequently write
239                     // other headers containing pseudo-header fields.
240                     //
241                     // This just sets internal stream state which is used elsewhere in the codec and doesn't
242                     // necessarily mean the write will complete successfully.
243                     stream.headersSent(isInformational);
244 
245                     if (!future.isSuccess()) {
246                         // Either the future is not done or failed in the meantime.
247                         notifyLifecycleManagerOnError(future, ctx);
248                     }
249                 } else {
250                     lifecycleManager.onError(ctx, true, failureCause);
251                 }
252 
253                 if (endOfStream) {
254                     // Must handle calling onError before calling closeStreamLocal, otherwise the error handler will
255                     // incorrectly think the stream no longer exists and so may not send RST_STREAM or perform similar
256                     // appropriate action.
257                     lifecycleManager.closeStreamLocal(stream, future);
258                 }
259 
260                 return future;
261             } else {
262                 // Pass headers to the flow-controller so it can maintain their sequence relative to DATA frames.
263                 flowController.addFlowControlled(stream,
264                         new FlowControlledHeaders(stream, headers, hasPriority, streamDependency,
265                                 weight, exclusive, padding, true, promise));
266                 return promise;
267             }
268         } catch (Throwable t) {
269             lifecycleManager.onError(ctx, true, t);
270             promise.tryFailure(t);
271             return promise;
272         }
273     }

304     @Override
305     public ChannelFuture writeSettingsAck(ChannelHandlerContext ctx, ChannelPromise promise) {
306         if (outstandingRemoteSettingsQueue == null) {
307             return frameWriter.writeSettingsAck(ctx, promise);
308         }
309         Http2Settings settings = outstandingRemoteSettingsQueue.poll();
310         if (settings == null) {
311             return promise.setFailure(new Http2Exception(INTERNAL_ERROR, "attempted to write a SETTINGS ACK with no " +
312                     " pending SETTINGS"));
313         }
314         SimpleChannelPromiseAggregator aggregator = new SimpleChannelPromiseAggregator(promise, ctx.channel(),
315                 ctx.executor());
316         // Acknowledge receipt of the settings. We should do this before we process the settings to ensure our
317         // remote peer applies these settings before any subsequent frames that we may send which depend upon
318         // these new settings. See https://github.com/netty/netty/issues/6520.
319         frameWriter.writeSettingsAck(ctx, aggregator.newPromise());
320 
321         // We create a "new promise" to make sure that status from both the write and the application are taken into
322         // account independently.
323         ChannelPromise applySettingsPromise = aggregator.newPromise();
324         try {
325             remoteSettings(settings);
326             applySettingsPromise.setSuccess();
327         } catch (Throwable e) {
328             applySettingsPromise.setFailure(e);
329             lifecycleManager.onError(ctx, true, e);
330         }
331         return aggregator.doneAllocatingPromises();
332     }

340     public ChannelFuture writePushPromise(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
341             Http2Headers headers, int padding, ChannelPromise promise) {
342         try {
343             if (connection.goAwayReceived()) {
344                 throw connectionError(PROTOCOL_ERROR, "Sending PUSH_PROMISE after GO_AWAY received.");
345             }
346 
347             Http2Stream stream = requireStream(streamId);
348             // Reserve the promised stream.
349             connection.local().reservePushStream(promisedStreamId, stream);
350 
351             promise = promise.unvoid();
352             ChannelFuture future = frameWriter.writePushPromise(ctx, streamId, promisedStreamId, headers, padding,
353                                                                 promise);
354             // Writing headers may fail during the encode state if they violate HPACK limits.
355             Throwable failureCause = future.cause();
356             if (failureCause == null) {
357                 // This just sets internal stream state which is used elsewhere in the codec and doesn't
358                 // necessarily mean the write will complete successfully.
359                 stream.pushPromiseSent();
360 
361                 if (!future.isSuccess()) {
362                     // Either the future is not done or failed in the meantime.
363                     notifyLifecycleManagerOnError(future, ctx);
364                 }
365             } else {
366                 lifecycleManager.onError(ctx, true, failureCause);
367             }
368             return future;
369         } catch (Throwable t) {
370             lifecycleManager.onError(ctx, true, t);
371             promise.tryFailure(t);

410     private Http2Stream requireStream(int streamId) {
411         Http2Stream stream = connection.stream(streamId);
412         if (stream == null) {
413             final String message;
414             if (connection.streamMayHaveExisted(streamId)) {
415                 message = "Stream no longer exists: " + streamId;
416             } else {
417                 message = "Stream does not exist: " + streamId;
418             }
419             throw new IllegalArgumentException(message);
420         }
421         return stream;

432     /**
433      * Wrap a DATA frame so it can be written subject to flow-control. Note that this implementation assumes it
434      * only writes padding once for the entire payload as opposed to writing it once per-frame. This makes the
435      * {@link #size} calculation deterministic thereby greatly simplifying the implementation.
436      * <p>
437      * If frame-splitting is required to fit within max-frame-size and flow-control constraints we ensure that
438      * the passed promise is not completed until last frame write.
439      * </p>
440      */
441     private final class FlowControlledData extends FlowControlledBase {
442         private final CoalescingBufferQueue queue;
443         private int dataSize;
444 
445         FlowControlledData(Http2Stream stream, ByteBuf buf, int padding, boolean endOfStream,
446                                    ChannelPromise promise) {
447             super(stream, padding, endOfStream, promise);
448             queue = new CoalescingBufferQueue(promise.channel());
449             queue.add(buf, promise);
450             dataSize = queue.readableBytes();
451         }
452 
453         @Override
454         public int size() {
455             return dataSize + padding;
456         }
457 
458         @Override
459         public void error(ChannelHandlerContext ctx, Throwable cause) {
460             queue.releaseAndFailAll(cause);
461             // Don't update dataSize because we need to ensure the size() method returns a consistent size even after
462             // error so we don't invalidate flow control when returning bytes to flow control.
463             //
464             // That said we will set dataSize and padding to 0 in the write(...) method if we cleared the queue
465             // because of an error.
466             lifecycleManager.onError(ctx, true, cause);
467         }
468 
469         @Override
470         public void write(ChannelHandlerContext ctx, int allowedBytes) {
471             int queuedData = queue.readableBytes();
472             if (!endOfStream) {
473                 if (queuedData == 0) {
474                     if (queue.isEmpty()) {
475                         // When the queue is empty it means we did clear it because of an error(...) call
476                         // (as otherwise we will have at least 1 entry in there), which will happen either when called
477                         // explicit or when the write itself fails. In this case just set dataSize and padding to 0
478                         // which will signal back that the whole frame was consumed.
479                         //
480                         // See https://github.com/netty/netty/issues/8707.
481                         padding = dataSize = 0;
482                     } else {
483                         // There's no need to write any data frames because there are only empty data frames in the
484                         // queue and it is not end of stream yet. Just complete their promises by getting the buffer
485                         // corresponding to 0 bytes and writing it to the channel (to preserve notification order).
486                         ChannelPromise writePromise = ctx.newPromise().addListener(this);
487                         ctx.write(queue.remove(0, writePromise), writePromise);
488                     }
489                     return;
490                 }
491 
492                 if (allowedBytes == 0) {
493                     return;
494                 }
495             }
496 
497             // Determine how much data to write.
498             int writableData = min(queuedData, allowedBytes);
499             ChannelPromise writePromise = ctx.newPromise().addListener(this);
500             ByteBuf toWrite = queue.remove(writableData, writePromise);
501             dataSize = queue.readableBytes();
502 
503             // Determine how much padding to write.
504             int writablePadding = min(allowedBytes - writableData, padding);
505             padding -= writablePadding;
506 
507             // Write the frame(s).
508             frameWriter().writeData(ctx, stream.id(), toWrite, writablePadding,
509                     endOfStream && size() == 0, writePromise);
510         }
511 
512         @Override
513         public boolean merge(ChannelHandlerContext ctx, Http2RemoteFlowController.FlowControlled next) {
514             FlowControlledData nextData;
515             if (FlowControlledData.class != next.getClass() ||
516                 MAX_VALUE - (nextData = (FlowControlledData) next).size() < size()) {
517                 return false;
518             }
519             nextData.queue.copyTo(queue);
520             dataSize = queue.readableBytes();
521             // Given that we're merging data into a frame it doesn't really make sense to accumulate padding.
522             padding = Math.max(padding, nextData.padding);
523             endOfStream = nextData.endOfStream;
524             return true;
525         }
526     }

540     /**
541      * Wrap headers so they can be written subject to flow-control. While headers do not have cost against the
542      * flow-control window their order with respect to other frames must be maintained, hence if a DATA frame is
543      * blocked on flow-control a HEADER frame must wait until this frame has been written.
544      */
545     private final class FlowControlledHeaders extends FlowControlledBase {
546         private final Http2Headers headers;
547         private final boolean hasPriorty;
548         private final int streamDependency;
549         private final short weight;
550         private final boolean exclusive;
551 
552         FlowControlledHeaders(Http2Stream stream, Http2Headers headers, boolean hasPriority,
553                               int streamDependency, short weight, boolean exclusive,
554                               int padding, boolean endOfStream, ChannelPromise promise) {
555             super(stream, padding, endOfStream, promise.unvoid());
556             this.headers = headers;
557             this.hasPriorty = hasPriority;
558             this.streamDependency = streamDependency;
559             this.weight = weight;
560             this.exclusive = exclusive;
561         }
562 
563         @Override
564         public int size() {
565             return 0;
566         }
567 
568         @Override
569         public void error(ChannelHandlerContext ctx, Throwable cause) {
570             if (ctx != null) {
571                 lifecycleManager.onError(ctx, true, cause);
572             }
573             promise.tryFailure(cause);
574         }
575 
576         @Override
577         public void write(ChannelHandlerContext ctx, int allowedBytes) {
578             boolean isInformational = validateHeadersSentState(stream, headers, connection.isServer(), endOfStream);
579             // The code is currently requiring adding this listener before writing, in order to call onError() before
580             // closeStreamLocal().
581             promise.addListener(this);
582 
583             ChannelFuture f = sendHeaders(frameWriter, ctx, stream.id(), headers, hasPriorty, streamDependency,
584                     weight, exclusive, padding, endOfStream, promise);
585             // Writing headers may fail during the encode state if they violate HPACK limits.
586             Throwable failureCause = f.cause();
587             if (failureCause == null) {
588                 // This just sets internal stream state which is used elsewhere in the codec and doesn't
589                 // necessarily mean the write will complete successfully.
590                 stream.headersSent(isInformational);
591             }
592         }
593 
594         @Override
595         public boolean merge(ChannelHandlerContext ctx, Http2RemoteFlowController.FlowControlled next) {
596             return false;
597         }
598     }

600     /**
601      * Common base type for payloads to deliver via flow-control.
602      */
603     public abstract class FlowControlledBase implements Http2RemoteFlowController.FlowControlled,
604             ChannelFutureListener {
605         protected final Http2Stream stream;
606         protected ChannelPromise promise;
607         protected boolean endOfStream;
608         protected int padding;
609 
610         FlowControlledBase(final Http2Stream stream, int padding, boolean endOfStream,
611                 final ChannelPromise promise) {
612             checkPositiveOrZero(padding, "padding");
613             this.padding = padding;
614             this.endOfStream = endOfStream;
615             this.stream = stream;
616             this.promise = promise;
617         }
618 
619         @Override
620         public void writeComplete() {
621             if (endOfStream) {
622                 lifecycleManager.closeStreamLocal(stream, promise);
623             }
624         }
625 
626         @Override
627         public void operationComplete(ChannelFuture future) throws Exception {
628             if (!future.isSuccess()) {
629                 error(flowController().channelHandlerContext(), future.cause());
630             }
631         }
