26  /**
27   * A {@link StreamByteDistributor} that implements the HTTP/2 priority tree algorithm for allocating
28   * bytes for all streams in the connection.
29   */
30  public final class PriorityStreamByteDistributor implements StreamByteDistributor {
31      private final Http2Connection connection;
32      private final Http2Connection.PropertyKey stateKey;
33      private final WriteVisitor writeVisitor = new WriteVisitor();
34  
35      public PriorityStreamByteDistributor(Http2Connection connection) {
36          this.connection = checkNotNull(connection, "connection");
37  
38          // Add a state for the connection.
39          stateKey = connection.newKey();
40          connection.connectionStream().setProperty(stateKey,
41                  new PriorityState(connection.connectionStream()));
42  
43          // Register for notification of new streams.
44          connection.addListener(new Http2ConnectionAdapter() {
45              @Override
46              public void onStreamAdded(Http2Stream stream) {
47                  stream.setProperty(stateKey, new PriorityState(stream));
48              }
49  
50              @Override
51              public void onStreamClosed(Http2Stream stream) {
52                  state(stream).close();
53              }
54  
55              @Override
56              public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {
57                  Http2Stream parent = stream.parent();
58                  if (parent != null) {
59                      long delta = state(stream).unallocatedStreamableBytesForTree();
60                      if (delta != 0) {
61                          state(parent).unallocatedStreamableBytesForTreeChanged(delta);
62                      }
63                  }
64              }
65  
66              @Override
67              public void onPriorityTreeParentChanging(Http2Stream stream, Http2Stream newParent) {
68                  Http2Stream parent = stream.parent();
69                  if (parent != null) {
70                      long delta = state(stream).unallocatedStreamableBytesForTree();
71                      if (delta != 0) {
72                          state(parent).unallocatedStreamableBytesForTreeChanged(-delta);
73                      }
74                  }
75              }
76          });
77      }
78  
79      @Override
80      public void updateStreamableBytes(StreamState streamState) {
81          state(streamState.stream()).updateStreamableBytes(streamState.streamableBytes(),
82                  streamState.hasFrame());
83      }
84  
85      @Override
86      public boolean distribute(int maxBytes, Writer writer) throws Http2Exception {
87          checkNotNull(writer, "writer");
88          if (maxBytes > 0) {
89              allocateBytesForTree(connection.connectionStream(), maxBytes);
90          }
91  
92          // Need to write even if maxBytes == 0 in order to handle the case of empty frames.
93          writeVisitor.writeAllocatedBytes(writer);
94  
95          return state(connection.connectionStream()).unallocatedStreamableBytesForTree() > 0;
96      }

112     /**
113      * This will allocate bytes by stream weight and priority for the entire tree rooted at {@code
114      * parent}, but does not write any bytes. The connection window is generally distributed amongst
115      * siblings according to their weight, however we need to ensure that the entire connection
116      * window is used (assuming streams have >= connection window bytes to send) and we may need
117      * some sort of rounding to accomplish this.
118      *
119      * @param parent The parent of the tree.
120      * @param connectionWindowSize The connection window this is available for use at this point in
121      * the tree.
122      * @return The number of bytes actually allocated.
123      */
124     private int allocateBytesForTree(Http2Stream parent, int connectionWindowSize) {
125         PriorityState state = state(parent);
126         if (state.unallocatedStreamableBytesForTree() <= 0) {
127             return 0;
128         }
129         // If the number of streamable bytes for this tree will fit in the connection window
130         // then there is no need to prioritize the bytes...everyone sends what they have
131         if (state.unallocatedStreamableBytesForTree() <= connectionWindowSize) {
132             SimpleChildFeeder childFeeder = new SimpleChildFeeder(connectionWindowSize);
133             forEachChild(parent, childFeeder);
134             return childFeeder.bytesAllocated;
135         }
136 
137         ChildFeeder childFeeder = new ChildFeeder(parent, connectionWindowSize);
138         // Iterate once over all children of this parent and try to feed all the children.
139         forEachChild(parent, childFeeder);
140 
141         // Now feed any remaining children that are still hungry until the connection
142         // window collapses.
143         childFeeder.feedHungryChildren();
144 
145         return childFeeder.bytesAllocated;
146     }
147 
148     private void forEachChild(Http2Stream parent, Http2StreamVisitor childFeeder) {
149         try {
150             parent.forEachChild(childFeeder);
151         } catch (Http2Exception e) {
152             // Should never happen since the feeder doesn't throw.
153             throw new IllegalStateException(e);
154         }
155     }
156 
157     private PriorityState state(Http2Stream stream) {
158         return checkNotNull(stream, "stream").getProperty(stateKey);
159     }

161     /**
162      * A {@link Http2StreamVisitor} that performs the HTTP/2 priority algorithm to distribute the
163      * available connection window appropriately to the children of a given stream.
164      */
165     private final class ChildFeeder implements Http2StreamVisitor {
166         final int maxSize;
167         int totalWeight;
168         int connectionWindow;
169         int nextTotalWeight;
170         int nextConnectionWindow;
171         int bytesAllocated;
172         Http2Stream[] stillHungry;
173         int nextTail;
174 
175         ChildFeeder(Http2Stream parent, int connectionWindow) {
176             maxSize = parent.numChildren();
177             totalWeight = parent.totalChildWeights();
178             this.connectionWindow = connectionWindow;
179             this.nextConnectionWindow = connectionWindow;
180         }
181 
182         @Override
183         public boolean visit(Http2Stream child) {
184             // In order to make progress toward the connection window due to possible rounding errors, we make sure
185             // that each stream (with data to send) is given at least 1 byte toward the connection window.
186             int connectionWindowChunk =
187                     max(1, (int) (connectionWindow * (child.weight() / (double) totalWeight)));
188             int bytesForTree = min(nextConnectionWindow, connectionWindowChunk);
189 
190             PriorityState state = state(child);
191             int bytesForChild = min(state.unallocatedStreamableBytes(), bytesForTree);
192 
193             // Allocate the bytes to this child.
194             if (bytesForChild > 0) {
195                 state.allocate(bytesForChild);
196                 bytesAllocated += bytesForChild;
197                 nextConnectionWindow -= bytesForChild;
198                 bytesForTree -= bytesForChild;
199             }
200 
201             // Allocate any remaining bytes to the children of this stream.
202             if (bytesForTree > 0) {
203                 int childBytesAllocated = allocateBytesForTree(child, bytesForTree);
204                 bytesAllocated += childBytesAllocated;
205                 nextConnectionWindow -= childBytesAllocated;
206             }
207 
208             if (nextConnectionWindow > 0) {
209                 // If this subtree still wants to send then it should be re-considered to take bytes that are unused by
210                 // sibling nodes. This is needed because we don't yet know if all the peers will be able to use all of
211                 // their "fair share" of the connection window, and if they don't use it then we should divide their
212                 // unused shared up for the peers who still want to send.
213                 if (state.unallocatedStreamableBytesForTree() > 0) {
214                     stillHungry(child);
215                 }
216                 return true;
217             }
218 
219             return false;
220         }
221 
222         void feedHungryChildren() {
223             if (stillHungry == null) {
224                 // There are no hungry children to feed.
225                 return;
226             }
227 
228             totalWeight = nextTotalWeight;
229             connectionWindow = nextConnectionWindow;
230 
231             // Loop until there are not bytes left to stream or the connection window has collapsed.
232             for (int tail = nextTail; tail > 0 && connectionWindow > 0;) {
233                 nextTotalWeight = 0;
234                 nextTail = 0;
235 
236                 // Iterate over the children that are currently still hungry.
237                 for (int head = 0; head < tail && nextConnectionWindow > 0; ++head) {
238                     if (!visit(stillHungry[head])) {
239                         // The connection window has collapsed, break out of the loop.
240                         break;
241                     }
242                 }
243                 connectionWindow = nextConnectionWindow;
244                 totalWeight = nextTotalWeight;
245                 tail = nextTail;
246             }
247         }
248 
249         /**
250          * Indicates that the given child is still hungry (i.e. still has streamable bytes that can
251          * fit within the current connection window).
252          */
253         void stillHungry(Http2Stream child) {
254             ensureSpaceIsAllocated(nextTail);
255             stillHungry[nextTail++] = child;
256             nextTotalWeight += child.weight();
257         }
258 
259         /**
260          * Ensures that the {@link #stillHungry} array is properly sized to hold the given index.
261          */
262         void ensureSpaceIsAllocated(int index) {
263             if (stillHungry == null) {
264                 // Initial size is 1/4 the number of children. Clipping the minimum at 2, which will over allocate if
265                 // maxSize == 1 but if this was true we shouldn't need to re-allocate because the 1 child should get
266                 // all of the available connection window.
267                 stillHungry = new Http2Stream[max(2, maxSize >>> 2)];
268             } else if (index == stillHungry.length) {
269                 // Grow the array by a factor of 2.
270                 stillHungry = Arrays.copyOf(stillHungry, min(maxSize, stillHungry.length << 1));
271             }
272         }
273     }

275     /**
276      * A simplified version of {@link ChildFeeder} that is only used when all streamable bytes fit
277      * within the available connection window.
278      */
279     private final class SimpleChildFeeder implements Http2StreamVisitor {
280         int bytesAllocated;
281         int connectionWindow;
282 
283         SimpleChildFeeder(int connectionWindow) {
284             this.connectionWindow = connectionWindow;
285         }
286 
287         @Override
288         public boolean visit(Http2Stream child) {
289             PriorityState childState = state(child);
290             int bytesForChild = childState.unallocatedStreamableBytes();
291 
292             if (bytesForChild > 0 || childState.hasFrame()) {
293                 childState.allocate(bytesForChild);
294                 bytesAllocated += bytesForChild;
295                 connectionWindow -= bytesForChild;
296             }
297             int childBytesAllocated = allocateBytesForTree(child, connectionWindow);
298             bytesAllocated += childBytesAllocated;
299             connectionWindow -= childBytesAllocated;
300             return true;
301         }
302     }

304     /**
305      * The remote flow control state for a single stream.
306      */
307     private final class PriorityState {
308         final Http2Stream stream;
309         boolean hasFrame;
310         int streamableBytes;
311         int allocated;
312         long unallocatedStreamableBytesForTree;
313 
314         PriorityState(Http2Stream stream) {
315             this.stream = stream;
316         }
317 
318         /**
319          * Recursively increments the {@link #unallocatedStreamableBytesForTree()} for this branch in
320          * the priority tree starting at the current node.
321          */
322         void unallocatedStreamableBytesForTreeChanged(long delta) {
323             unallocatedStreamableBytesForTree += delta;
324             if (!stream.isRoot()) {
325                 state(stream.parent()).unallocatedStreamableBytesForTreeChanged(delta);
326             }
327         }
328 
329         void allocate(int bytes) {
330             allocated += bytes;
331 
332             if (bytes != 0) {
333                 // Also artificially reduce the streamable bytes for this tree to give the appearance
334                 // that the data has been written. This will be restored before the allocated bytes are
335                 // actually written.
336                 unallocatedStreamableBytesForTreeChanged(-bytes);
337             }
338         }
339 
340         /**
341          * Reset the number of bytes that have been allocated to this stream by the priority
342          * algorithm.
343          */
344         void resetAllocated() {
345             allocate(-allocated);
346         }
347 
348         void updateStreamableBytes(int newStreamableBytes, boolean hasFrame) {
349             this.hasFrame = hasFrame;
350 
351             int delta = newStreamableBytes - streamableBytes;
352             if (delta != 0) {
353                 streamableBytes = newStreamableBytes;
354 
355                 // Update this branch of the priority tree if the streamable bytes have changed for this node.
356                 unallocatedStreamableBytesForTreeChanged(delta);
357             }
358         }
359 
360         void close() {
361             // Unallocate all bytes.
362             resetAllocated();
363 
364             // Clear the streamable bytes.
365             updateStreamableBytes(0, false);
366         }
367 
368         boolean hasFrame() {
369             return hasFrame;
370         }
371 
372         int unallocatedStreamableBytes() {
373             return streamableBytes - allocated;
374         }
375 
376         long unallocatedStreamableBytesForTree() {
377             return unallocatedStreamableBytesForTree;
378         }
379     }

381     /**
382      * A connection stream visitor that delegates to the user provided visitor.
383      */
384     private final class WriteVisitor implements Http2StreamVisitor {
385         private boolean iterating;
386         private Writer writer;
387 
388         void writeAllocatedBytes(Writer writer) throws Http2Exception {
389             if (iterating) {
390                 throw connectionError(INTERNAL_ERROR, "byte distribution re-entry error");
391             }
392             this.writer = writer;
393             try {
394                 iterating = true;
395                 connection.forEachActiveStream(this);
396             } finally {
397                 iterating = false;
398             }
399         }
400 
401         @Override
402         public boolean visit(Http2Stream stream) throws Http2Exception {
403             PriorityState state = state(stream);
404             int allocated = state.allocated;
405 
406             // Unallocate all bytes for this stream.
407             state.resetAllocated();
408 
409             try {
410                 // Write the allocated bytes.
411                 writer.write(stream, allocated);
412             } catch (Throwable t) { // catch Throwable in case any unchecked re-throw tricks are used.
413                 // Stop calling the visitor and close the connection as exceptions from the writer are not supported.
414                 // If we don't close the connection there is risk that our internal state may be corrupted.
415                 throw connectionError(INTERNAL_ERROR, t, "byte distribution write error");
416             }
417 
418             // We have to iterate across all streams to ensure that we reset the allocated bytes.
419             return true;
420         }
421     }
