15  package io.netty.handler.codec.http2;
16  
17  import io.netty.util.internal.UnstableApi;
18  
19  import static io.netty.handler.codec.http2.Http2Error.PROTOCOL_ERROR;
20  
21  /**
22   * This exception is thrown when there are no more stream IDs available for the current connection
23   */
24  @UnstableApi
25  public class Http2NoMoreStreamIdsException extends Http2Exception {
26      private static final long serialVersionUID = -7756236161274851110L;
27      private static final String ERROR_MESSAGE = "No more streams can be created on this connection";
28  
29      public Http2NoMoreStreamIdsException() {
30          super(PROTOCOL_ERROR, ERROR_MESSAGE, ShutdownHint.GRACEFUL_SHUTDOWN);
31      }
32  
33      public Http2NoMoreStreamIdsException(Throwable cause) {
34          super(PROTOCOL_ERROR, ERROR_MESSAGE, cause, ShutdownHint.GRACEFUL_SHUTDOWN);
35      }
36  }
