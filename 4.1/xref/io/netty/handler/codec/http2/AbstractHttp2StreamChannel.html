184     private void incrementPendingOutboundBytes(long size, boolean invokeLater) {
185         if (size == 0) {
186             return;
187         }
188 
189         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, size);
190         if (newWriteBufferSize > config().getWriteBufferHighWaterMark()) {
191             setUnwritable(invokeLater);
192         }
193     }
194 
195     private void decrementPendingOutboundBytes(long size, boolean invokeLater) {
196         if (size == 0) {
197             return;
198         }
199 
200         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, -size);
201         // Once the totalPendingSize dropped below the low water-mark we can mark the child channel
202         // as writable again. Before doing so we also need to ensure the parent channel is writable to
203         // prevent excessive buffering in the parent outbound buffer. If the parent is not writable
204         // we will mark the child channel as writable once the parent becomes writable by calling
205         // trySetWritable() later.
206         if (newWriteBufferSize < config().getWriteBufferLowWaterMark() && parent().isWritable()) {
207             setWritable(invokeLater);
208         }
209     }
210 
211     final void trySetWritable() {
212         // The parent is writable again but the child channel itself may still not be writable.
213         // Lets try to set the child channel writable to match the state of the parent channel
214         // if (and only if) the totalPendingSize is smaller then the low water-mark.
215         // If this is not the case we will try again later once we drop under it.
216         if (totalPendingSize < config().getWriteBufferLowWaterMark()) {
217             setWritable(false);
218         }
219     }
220 
221     private void setWritable(boolean invokeLater) {
222         for (;;) {
223             final int oldValue = unwritable;
224             final int newValue = oldValue & ~1;
225             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
226                 if (oldValue != 0 && newValue == 0) {
227                     fireChannelWritabilityChanged(invokeLater);
228                 }
229                 break;
230             }
231         }
232     }
233 
234     private void setUnwritable(boolean invokeLater) {
235         for (;;) {
236             final int oldValue = unwritable;
237             final int newValue = oldValue | 1;
238             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
239                 if (oldValue == 0 && newValue != 0) {
240                     fireChannelWritabilityChanged(invokeLater);
241                 }
242                 break;
243             }
244         }
245     }
246 
247     private void fireChannelWritabilityChanged(boolean invokeLater) {
248         final ChannelPipeline pipeline = pipeline();
249         if (invokeLater) {
250             Runnable task = fireChannelWritabilityChangedTask;
251             if (task == null) {
252                 fireChannelWritabilityChangedTask = task = new Runnable() {
253                     @Override
254                     public void run() {
255                         pipeline.fireChannelWritabilityChanged();
256                     }
257                 };
258             }
259             eventLoop().execute(task);
260         } else {
261             pipeline.fireChannelWritabilityChanged();
262         }
263     }

340     @Override
341     public long bytesBeforeUnwritable() {
342         long bytes = config().getWriteBufferHighWaterMark() - totalPendingSize;
343         // If bytes is negative we know we are not writable, but if bytes is non-negative we have to check
344         // writability. Note that totalPendingSize and isWritable() use different volatile variables that are not
345         // synchronized together. totalPendingSize will be updated before isWritable().
346         if (bytes > 0) {
347             return isWritable() ? bytes : 0;
348         }
349         return 0;
350     }

352     @Override
353     public long bytesBeforeWritable() {
354         long bytes = totalPendingSize - config().getWriteBufferLowWaterMark();
355         // If bytes is negative we know we are writable, but if bytes is non-negative we have to check writability.
356         // Note that totalPendingSize and isWritable() use different volatile variables that are not synchronized
357         // together. totalPendingSize will be updated before isWritable().
358         if (bytes > 0) {
359             return isWritable() ? 0 : bytes;
360         }
361         return 0;
362     }

520     /**
521      * Receive a read message. This does not notify handlers unless a read is in progress on the
522      * channel.
523      */
524     void fireChildRead(Http2Frame frame) {
525         assert eventLoop().inEventLoop();
526         if (!isActive()) {
527             ReferenceCountUtil.release(frame);
528         } else if (readStatus != ReadStatus.IDLE) {
529             // If a read is in progress or has been requested, there cannot be anything in the queue,
530             // otherwise we would have drained it from the queue and processed it during the read cycle.
531             assert inboundBuffer == null || inboundBuffer.isEmpty();
532             final RecvByteBufAllocator.Handle allocHandle = unsafe.recvBufAllocHandle();
533             flowControlledBytes += unsafe.doRead0(frame, allocHandle);
534             // We currently don't need to check for readEOS because the parent channel and child channel are limited
535             // to the same EventLoop thread. There are a limited number of frame types that may come after EOS is
536             // read (unknown, reset) and the trade off is less conditionals for the hot path (headers/data) at the
537             // cost of additional readComplete notifications on the rare path.
538             if (allocHandle.continueReading()) {
539                 if (!readCompletePending) {
540                     readCompletePending = true;
541                     addChannelToReadCompletePendingQueue();
542                 }
543             } else {
544                 unsafe.notifyReadComplete(allocHandle, true);
545             }
546         } else {
547             if (inboundBuffer == null) {
548                 inboundBuffer = new ArrayDeque<Object>(4);
549             }
550             inboundBuffer.add(frame);
551         }
552     }
553 
554     void fireChildReadComplete() {
555         assert eventLoop().inEventLoop();
556         assert readStatus != ReadStatus.IDLE || !readCompletePending;
557         unsafe.notifyReadComplete(unsafe.recvBufAllocHandle(), false);
558     }

630         @Override
631         public void close(final ChannelPromise promise) {
632             if (!promise.setUncancellable()) {
633                 return;
634             }
635             if (closeInitiated) {
636                 if (closePromise.isDone()) {
637                     // Closed already.
638                     promise.setSuccess();
639                 } else if (!(promise instanceof VoidChannelPromise)) { // Only needed if no VoidChannelPromise.
640                     // This means close() was called before so we just register a listener and return
641                     closePromise.addListener(new ChannelFutureListener() {
642                         @Override
643                         public void operationComplete(ChannelFuture future) {
644                             promise.setSuccess();
645                         }
646                     });
647                 }
648                 return;
649             }
650             closeInitiated = true;
651             // Just set to false as removing from an underlying queue would even be more expensive.
652             readCompletePending = false;
653 
654             final boolean wasActive = isActive();
655 
656             updateLocalWindowIfNeeded();
657 
658             // Only ever send a reset frame if the connection is still alive and if the stream was created before
659             // as otherwise we may send a RST on a stream in an invalid state and cause a connection error.
660             if (parent().isActive() && !readEOS && Http2CodecUtil.isStreamIdValid(stream.id())) {
661                 Http2StreamFrame resetFrame = new DefaultHttp2ResetFrame(Http2Error.CANCEL).stream(stream());
662                 write(resetFrame, unsafe().voidPromise());
663                 flush();
664             }
665 
666             if (inboundBuffer != null) {
667                 for (;;) {
668                     Object msg = inboundBuffer.poll();
669                     if (msg == null) {
670                         break;
671                     }
672                     ReferenceCountUtil.release(msg);
673                 }
674                 inboundBuffer = null;
675             }
676 
677             // The promise should be notified before we call fireChannelInactive().
678             outboundClosed = true;
679             closePromise.setSuccess();
680             promise.setSuccess();
681 
682             fireChannelInactiveAndDeregister(voidPromise(), wasActive);
683         }

687             close(unsafe().voidPromise());
688         }

692             fireChannelInactiveAndDeregister(promise, false);
693         }

695         private void fireChannelInactiveAndDeregister(final ChannelPromise promise,
696                                                       final boolean fireChannelInactive) {
697             if (!promise.setUncancellable()) {
698                 return;
699             }
700 
701             if (!registered) {
702                 promise.setSuccess();
703                 return;
704             }
705 
706             // As a user may call deregister() from within any method while doing processing in the ChannelPipeline,
707             // we need to ensure we do the actual deregister operation later. This is necessary to preserve the
708             // behavior of the AbstractChannel, which always invokes channelUnregistered and channelInactive
709             // events 'later' to ensure the current events in the handler are completed before these events.
710             //
711             // See:
712             // https://github.com/netty/netty/issues/4435
713             invokeLater(new Runnable() {
714                 @Override
715                 public void run() {
716                     if (fireChannelInactive) {
717                         pipeline.fireChannelInactive();
718                     }
719                     // The user can fire `deregister` events multiple times but we only want to fire the pipeline
720                     // event if the channel was actually registered.
721                     if (registered) {
722                         registered = false;
723                         pipeline.fireChannelUnregistered();
724                     }
725                     safeSetSuccess(promise);
726                 }
727             });
728         }

730         private void safeSetSuccess(ChannelPromise promise) {
731             if (!(promise instanceof VoidChannelPromise) && !promise.trySuccess()) {
732                 logger.warn("Failed to mark a promise as success because it is done already: {}", promise);
733             }
734         }

736         private void invokeLater(Runnable task) {
737             try {
738                 // This method is used by outbound operation implementations to trigger an inbound event later.
739                 // They do not trigger an inbound event immediately because an outbound operation might have been
740                 // triggered by another inbound event handler method.  If fired immediately, the call stack
741                 // will look like this for example:
742                 //
743                 //   handlerA.inboundBufferUpdated() - (1) an inbound handler method closes a connection.
744                 //   -> handlerA.ctx.close()
745                 //     -> channel.unsafe.close()
746                 //       -> handlerA.channelInactive() - (2) another inbound handler method called while in (1) yet
747                 //
748                 // which means the execution of two inbound handler methods of the same handler overlap undesirably.
749                 eventLoop().execute(task);
750             } catch (RejectedExecutionException e) {
751                 logger.warn("Can't invoke task later as EventLoop rejected it", e);
752             }
753         }

755         @Override
756         public void beginRead() {
757             if (!isActive()) {
758                 return;
759             }
760             switch (readStatus) {
761                 case IDLE:
762                     readStatus = ReadStatus.IN_PROGRESS;
763                     doBeginRead();
764                     break;
765                 case IN_PROGRESS:
766                     readStatus = ReadStatus.REQUESTED;
767                     break;
768                 default:
769                     break;
770             }
771         }

773         void doBeginRead() {
774             Object message;
775             if (inboundBuffer == null || (message = inboundBuffer.poll()) == null) {
776                 if (readEOS) {
777                     unsafe.closeForcibly();
778                 }
779             } else {
780                 final RecvByteBufAllocator.Handle allocHandle = recvBufAllocHandle();
781                 allocHandle.reset(config());
782                 boolean continueReading = false;
783                 do {
784                     flowControlledBytes += doRead0((Http2Frame) message, allocHandle);
785                 } while ((readEOS || (continueReading = allocHandle.continueReading())) &&
786                         (message = inboundBuffer.poll()) != null);
787 
788                 if (continueReading && isParentReadInProgress() && !readEOS) {
789                     // Currently the parent and child channel are on the same EventLoop thread. If the parent is
790                     // currently reading it is possile that more frames will be delivered to this child channel. In
791                     // the case that this child channel still wants to read we delay the channelReadComplete on this
792                     // child channel until the parent is done reading.
793                     if (!readCompletePending) {
794                         readCompletePending = true;
795                         addChannelToReadCompletePendingQueue();
796                     }
797                 } else {
798                     notifyReadComplete(allocHandle, true);
799                 }
800             }
801         }

808             if (flowControlledBytes != 0) {
809                 int bytes = flowControlledBytes;
810                 flowControlledBytes = 0;

816         void notifyReadComplete(RecvByteBufAllocator.Handle allocHandle, boolean forceReadComplete) {
817             if (!readCompletePending && !forceReadComplete) {
818                 return;
819             }
820             // Set to false just in case we added the channel multiple times before.
821             readCompletePending = false;
822 
823             if (readStatus == ReadStatus.REQUESTED) {
824                 readStatus = ReadStatus.IN_PROGRESS;
825             } else {
826                 readStatus = ReadStatus.IDLE;
827             }
828 
829             updateLocalWindowIfNeeded();
830 
831             allocHandle.readComplete();
832             pipeline().fireChannelReadComplete();
833             // Reading data may result in frames being written (e.g. WINDOW_UPDATE, RST, etc..). If the parent
834             // channel is not currently reading we need to force a flush at the child channel, because we cannot
835             // rely upon flush occurring in channelReadComplete on the parent channel.
836             flush();
837             if (readEOS) {
838                 unsafe.closeForcibly();
839             }
840         }

859         @Override
860         public void write(Object msg, final ChannelPromise promise) {
861             // After this point its not possible to cancel a write anymore.
862             if (!promise.setUncancellable()) {
863                 ReferenceCountUtil.release(msg);
864                 return;
865             }
866 
867             if (!isActive() ||
868                     // Once the outbound side was closed we should not allow header / data frames
869                     outboundClosed && (msg instanceof Http2HeadersFrame || msg instanceof Http2DataFrame)) {
870                 ReferenceCountUtil.release(msg);
871                 promise.setFailure(new ClosedChannelException());
872                 return;
873             }
874 
875             try {
876                 if (msg instanceof Http2StreamFrame) {
877                     Http2StreamFrame frame = validateStreamFrame((Http2StreamFrame) msg).stream(stream());
878                     if (!firstFrameWritten && !isStreamIdValid(stream().id())) {
879                         if (!(frame instanceof Http2HeadersFrame)) {
880                             ReferenceCountUtil.release(frame);
881                             promise.setFailure(
882                                     new IllegalArgumentException("The first frame must be a headers frame. Was: "
883                                             + frame.name()));
884                             return;
885                         }
886                         firstFrameWritten = true;
887                         ChannelFuture f = write0(parentContext(), frame);
888                         if (f.isDone()) {
889                             firstWriteComplete(f, promise);
890                         } else {
891                             final long bytes = FlowControlledFrameSizeEstimator.HANDLE_INSTANCE.size(msg);
892                             incrementPendingOutboundBytes(bytes, false);
893                             f.addListener(new ChannelFutureListener() {
894                                 @Override
895                                 public void operationComplete(ChannelFuture future) {
896                                     firstWriteComplete(future, promise);
897                                     decrementPendingOutboundBytes(bytes, false);
898                                 }
899                             });
900                             writeDoneAndNoFlush = true;
901                         }
902                         return;
903                     }
904                 } else {
905                     String msgStr = msg.toString();
906                     ReferenceCountUtil.release(msg);
907                     promise.setFailure(new IllegalArgumentException(
908                             "Message must be an " + StringUtil.simpleClassName(Http2StreamFrame.class) +
909                                     ": " + msgStr));
910                     return;
911                 }
912 
913                 ChannelFuture f = write0(parentContext(), msg);
914                 if (f.isDone()) {
915                     writeComplete(f, promise);
916                 } else {
917                     final long bytes = FlowControlledFrameSizeEstimator.HANDLE_INSTANCE.size(msg);
918                     incrementPendingOutboundBytes(bytes, false);
919                     f.addListener(new ChannelFutureListener() {
920                         @Override
921                         public void operationComplete(ChannelFuture future) {
922                             writeComplete(future, promise);
923                             decrementPendingOutboundBytes(bytes, false);
924                         }
925                     });
926                     writeDoneAndNoFlush = true;
927                 }
928             } catch (Throwable t) {
929                 promise.tryFailure(t);
930             }
931         }

933         private void firstWriteComplete(ChannelFuture future, ChannelPromise promise) {
934             Throwable cause = future.cause();
935             if (cause == null) {
936                 promise.setSuccess();
937             } else {
938                 // If the first write fails there is not much we can do, just close
939                 closeForcibly();
940                 promise.setFailure(wrapStreamClosedError(cause));
941             }
942         }

944         private void writeComplete(ChannelFuture future, ChannelPromise promise) {
945             Throwable cause = future.cause();
946             if (cause == null) {
947                 promise.setSuccess();
948             } else {
949                 Throwable error = wrapStreamClosedError(cause);
950                 // To make it more consistent with AbstractChannel we handle all IOExceptions here.
951                 if (error instanceof IOException) {
952                     if (config.isAutoClose()) {
953                         // Close channel if needed.
954                         closeForcibly();
955                     } else {
956                         // TODO: Once Http2StreamChannel extends DuplexChannel we should call shutdownOutput(...)
957                         outboundClosed = true;
958                     }
959                 }
960                 promise.setFailure(error);
961             }
962         }

964         private Throwable wrapStreamClosedError(Throwable cause) {
965             // If the error was caused by STREAM_CLOSED we should use a ClosedChannelException to better
966             // mimic other transports and make it easier to reason about what exceptions to expect.
967             if (cause instanceof Http2Exception && ((Http2Exception) cause).error() == Http2Error.STREAM_CLOSED) {
968                 return new ClosedChannelException().initCause(cause);
969             }
970             return cause;
971         }

983         @Override
984         public void flush() {
985             // If we are currently in the parent channel's read loop we should just ignore the flush.
986             // We will ensure we trigger ctx.flush() after we processed all Channels later on and
987             // so aggregate the flushes. This is done as ctx.flush() is expensive when as it may trigger an
988             // write(...) or writev(...) operation on the socket.
989             if (!writeDoneAndNoFlush || isParentReadInProgress()) {
990                 // There is nothing to flush so this is a NOOP.
991                 return;
992             }
993             try {
994                 flush0(parentContext());
995             } finally {
996                 writeDoneAndNoFlush = false;
997             }
998         }
