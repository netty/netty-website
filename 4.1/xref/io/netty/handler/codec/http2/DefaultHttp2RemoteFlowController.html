33  /**
34   * Basic implementation of {@link Http2RemoteFlowController}.
35   * <p>
36   * This class is <strong>NOT</strong> thread safe. The assumption is all methods must be invoked from a single thread.
37   * Typically this thread is the event loop thread for the {@link ChannelHandlerContext} managed by this class.
38   */
39  public class DefaultHttp2RemoteFlowController implements Http2RemoteFlowController {
40      private static final int MIN_WRITABLE_CHUNK = 32 * 1024;
41      private final Http2StreamVisitor WRITE_ALLOCATED_BYTES = new Http2StreamVisitor() {
42          @Override
43          public boolean visit(Http2Stream stream) {
44              int written = state(stream).writeAllocatedBytes();
45              if (written != -1 && listener != null) {
46                  listener.streamWritten(stream, written);
47              }
48              return true;
49          }
50      };
51      private final Http2Connection connection;
52      private final Http2Connection.PropertyKey stateKey;
53      private int initialWindowSize = DEFAULT_WINDOW_SIZE;
54      private ChannelHandlerContext ctx;
55      private Listener listener;
56  
57      public DefaultHttp2RemoteFlowController(Http2Connection connection) {
58          this.connection = checkNotNull(connection, "connection");
59  
60          // Add a flow state for the connection.
61          stateKey = connection.newKey();
62          connection.connectionStream().setProperty(stateKey,
63                  new DefaultState(connection.connectionStream(), initialWindowSize));
64  
65          // Register for notification of new streams.
66          connection.addListener(new Http2ConnectionAdapter() {
67              @Override
68              public void onStreamAdded(Http2Stream stream) {
69                  // If the stream state is not open then the stream is not yet eligible for flow controlled frames and
70                  // only requires the ReducedFlowState. Otherwise the full amount of memory is required.
71                  stream.setProperty(stateKey, stream.state() == IDLE ?
72                          new ReducedState(stream) :
73                          new DefaultState(stream, 0));
74              }
75  
76              @Override
77              public void onStreamActive(Http2Stream stream) {
78                  // If the object was previously created, but later activated then we have to ensure
79                  // the full state is allocated and the proper initialWindowSize is used.
80                  AbstractState state = state(stream);
81                  if (state.getClass() == DefaultState.class) {
82                      state.window(initialWindowSize);
83                  } else {
84                      stream.setProperty(stateKey, new DefaultState(state, initialWindowSize));
85                  }
86              }
87  
88              @Override
89              public void onStreamClosed(Http2Stream stream) {
90                  // Any pending frames can never be written, cancel and
91                  // write errors for any pending frames.
92                  AbstractState state = state(stream);
93                  state.cancel();
94  
95                  // If the stream is now eligible for removal, but will persist in the priority tree then we can
96                  // decrease the amount of memory required for this stream because no flow controlled frames can
97                  // be exchanged on this stream
98                  if (stream.prioritizableForTree() != 0) {
99                      stream.setProperty(stateKey, new ReducedState(state));
100                 }
101             }
102 
103             @Override
104             public void onStreamHalfClosed(Http2Stream stream) {
105                 if (State.HALF_CLOSED_LOCAL.equals(stream.state())) {
106                     /**
107                      * When this method is called there should not be any
108                      * pending frames left if the API is used correctly. However,
109                      * it is possible that a erroneous application can sneak
110                      * in a frame even after having already written a frame with the
111                      * END_STREAM flag set, as the stream state might not transition
112                      * immediately to HALF_CLOSED_LOCAL / CLOSED due to flow control
113                      * delaying the write.
114                      *
115                      * This is to cancel any such illegal writes.
116                      */
117                     state(stream).cancel();
118                 }
119             }
120 
121             @Override
122             public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {
123                 Http2Stream parent = stream.parent();
124                 if (parent != null) {
125                     int delta = state(stream).streamableBytesForTree();
126                     if (delta != 0) {
127                         state(parent).incrementStreamableBytesForTree(delta);
128                     }
129                 }
130             }
131 
132             @Override
133             public void onPriorityTreeParentChanging(Http2Stream stream, Http2Stream newParent) {
134                 Http2Stream parent = stream.parent();
135                 if (parent != null) {
136                     int delta = -state(stream).streamableBytesForTree();
137                     if (delta != 0) {
138                         state(parent).incrementStreamableBytesForTree(delta);
139                     }
140                 }
141             }
142         });
143     }

147      * <p>

150     @Override
151     public void channelHandlerContext(ChannelHandlerContext ctx) throws Http2Exception {
152         this.ctx = ctx;
153 
154         // Don't worry about cleaning up queued frames here if ctx is null. It is expected that all streams will be
155         // closed and the queue cleanup will occur when the stream state transitions occur.
156 
157         // If any frames have been queued up, we should send them now that we have a channel context.
158         if (ctx != null && ctx.channel().isWritable()) {
159             writePendingBytes();
160         }
161     }

168     @Override
169     public void initialWindowSize(int newWindowSize) throws Http2Exception {
170         assert ctx == null || ctx.executor().inEventLoop();
171         if (newWindowSize < 0) {
172             throw new IllegalArgumentException("Invalid initial window size: " + newWindowSize);
173         }
174 
175         final int delta = newWindowSize - initialWindowSize;
176         initialWindowSize = newWindowSize;
177         connection.forEachActiveStream(new Http2StreamVisitor() {
178             @Override
179             public boolean visit(Http2Stream stream) throws Http2Exception {
180                 // Verify that the maximum value is not exceeded by this change.
181                 state(stream).incrementStreamWindow(delta);
182                 return true;
183             }
184         });
185 
186         if (delta > 0) {
187             // The window size increased, send any pending frames for all streams.
188             writePendingBytes();
189         }
190     }

208     public void incrementWindowSize(Http2Stream stream, int delta) throws Http2Exception {
209         assert ctx == null || ctx.executor().inEventLoop();
210         if (stream.id() == CONNECTION_STREAM_ID) {
211             // Update the connection window
212             connectionState().incrementStreamWindow(delta);
213         } else {
214             // Update the stream window
215             AbstractState state = state(stream);
216             state.incrementStreamWindow(delta);
217         }

230     @Override
231     public void addFlowControlled(Http2Stream stream, FlowControlled frame) {
232         // The context can be null assuming the frame will be queued and send later when the context is set.
233         assert ctx == null || ctx.executor().inEventLoop();
234         checkNotNull(frame, "frame");
235         final AbstractState state;
236         try {
237             state = state(stream);
238             state.enqueueFrame(frame);
239         } catch (Throwable t) {
240             frame.error(ctx, t);
241             return;
242         }
243     }

268     private int minUsableChannelBytes() {
269         // The current allocation algorithm values "fairness" and doesn't give any consideration to "goodput". It
270         // is possible that 1 byte will be allocated to many streams. In an effort to try to make "goodput"
271         // reasonable with the current allocation algorithm we have this "cheap" check up front to ensure there is
272         // an "adequate" amount of connection window before allocation is attempted. This is not foolproof as if the
273         // number of streams is >= this minimal number then we may still have the issue, but the idea is to narrow the
274         // circumstances in which this can happen without rewriting the allocation algorithm.
275         return Math.max(ctx.channel().config().getWriteBufferLowWaterMark(), MIN_WRITABLE_CHUNK);
276     }
277 
278     private int maxUsableChannelBytes() {
279         if (ctx == null) {
280             return 0;
281         }
282 
283         // If the channel isWritable, allow at least minUseableChannelBytes.
284         int channelWritableBytes = (int) Math.min(Integer.MAX_VALUE, ctx.channel().bytesBeforeUnwritable());
285         int useableBytes = channelWritableBytes > 0 ? max(channelWritableBytes, minUsableChannelBytes()) : 0;
286 
287         // Clip the usable bytes by the connection window.
288         return min(connectionState().windowSize(), useableBytes);
289     }
290 
291     private int writableBytes(int requestedBytes) {
292         return Math.min(requestedBytes, maxUsableChannelBytes());
293     }

295     /**
296      * Writes as many pending bytes as possible, according to stream priority.
297      */
298     @Override
299     public void writePendingBytes() throws Http2Exception {
300         Http2Stream connectionStream = connection.connectionStream();
301         int connectionWindowSize = writableBytes(state(connectionStream).windowSize());
302 
303         if (connectionWindowSize > 0) {
304             // Allocate the bytes for the connection window to the streams, but do not write.
305             allocateBytesForTree(connectionStream, connectionWindowSize);
306         }
307 
308         // Now write all of the allocated bytes, must write as there may be empty frames with
309         // EOS = true
310         connection.forEachActiveStream(WRITE_ALLOCATED_BYTES);
311     }

313     /**
314      * This will allocate bytes by stream weight and priority for the entire tree rooted at {@code parent}, but does not
315      * write any bytes. The connection window is generally distributed amongst siblings according to their weight,
316      * however we need to ensure that the entire connection window is used (assuming streams have >= connection window
317      * bytes to send) and we may need some sort of rounding to accomplish this.
318      *
319      * @param parent The parent of the tree.
320      * @param connectionWindowSize The connection window this is available for use at this point in the tree.
321      * @return An object summarizing the write and allocation results.
322      */
323     int allocateBytesForTree(Http2Stream parent, int connectionWindowSize) throws Http2Exception {
324         AbstractState state = state(parent);
325         if (state.streamableBytesForTree() <= 0) {
326             return 0;
327         }
328         // If the number of streamable bytes for this tree will fit in the connection window
329         // then there is no need to prioritize the bytes...everyone sends what they have
330         if (state.streamableBytesForTree() <= connectionWindowSize) {
331             SimpleChildFeeder childFeeder = new SimpleChildFeeder(connectionWindowSize);
332             parent.forEachChild(childFeeder);
333             return childFeeder.bytesAllocated;
334         }
335 
336         ChildFeeder childFeeder = new ChildFeeder(parent, connectionWindowSize);
337         // Iterate once over all children of this parent and try to feed all the children.
338         parent.forEachChild(childFeeder);
339 
340         // Now feed any remaining children that are still hungry until the connection
341         // window collapses.
342         childFeeder.feedHungryChildren();
343 
344         return childFeeder.bytesAllocated;
345     }

347     /**
348      * A {@link Http2StreamVisitor} that performs the HTTP/2 priority algorithm to distribute the available connection
349      * window appropriately to the children of a given stream.
350      */
351     private final class ChildFeeder implements Http2StreamVisitor {
352         final int maxSize;
353         int totalWeight;
354         int connectionWindow;
355         int nextTotalWeight;
356         int nextConnectionWindow;
357         int bytesAllocated;
358         Http2Stream[] stillHungry;
359         int nextTail;
360 
361         ChildFeeder(Http2Stream parent, int connectionWindow) {
362             maxSize = parent.numChildren();
363             totalWeight = parent.totalChildWeights();
364             this.connectionWindow = connectionWindow;
365             this.nextConnectionWindow = connectionWindow;
366         }
367 
368         @Override
369         public boolean visit(Http2Stream child) throws Http2Exception {
370             // In order to make progress toward the connection window due to possible rounding errors, we make sure
371             // that each stream (with data to send) is given at least 1 byte toward the connection window.
372             int connectionWindowChunk = max(1, (int) (connectionWindow * (child.weight() / (double) totalWeight)));
373             int bytesForTree = min(nextConnectionWindow, connectionWindowChunk);
374 
375             AbstractState state = state(child);
376             int bytesForChild = min(state.streamableBytes(), bytesForTree);
377 
378             // Allocate the bytes to this child.
379             if (bytesForChild > 0) {
380                 state.allocate(bytesForChild);
381                 bytesAllocated += bytesForChild;
382                 nextConnectionWindow -= bytesForChild;
383                 bytesForTree -= bytesForChild;
384 
385                 // If this subtree still wants to send then re-insert into children list and re-consider for next
386                 // iteration. This is needed because we don't yet know if all the peers will be able to use
387                 // all of their "fair share" of the connection window, and if they don't use it then we should
388                 // divide their unused shared up for the peers who still want to send.
389                 if (nextConnectionWindow > 0 && state.streamableBytesForTree() > 0) {
390                     stillHungry(child);
391                     nextTotalWeight += child.weight();
392                 }
393             }
394 
395             // Allocate any remaining bytes to the children of this stream.
396             if (bytesForTree > 0) {
397                 int childBytesAllocated = allocateBytesForTree(child, bytesForTree);
398                 bytesAllocated += childBytesAllocated;
399                 nextConnectionWindow -= childBytesAllocated;
400             }
401 
402             return nextConnectionWindow > 0;
403         }
404 
405         void feedHungryChildren() throws Http2Exception {
406             if (stillHungry == null) {
407                 // There are no hungry children to feed.
408                 return;
409             }
410 
411             totalWeight = nextTotalWeight;
412             connectionWindow = nextConnectionWindow;
413 
414             // Loop until there are not bytes left to stream or the connection window has collapsed.
415             for (int tail = nextTail; tail > 0 && connectionWindow > 0;) {
416                 nextTotalWeight = 0;
417                 nextTail = 0;
418 
419                 // Iterate over the children that are currently still hungry.
420                 for (int head = 0; head < tail && nextConnectionWindow > 0; ++head) {
421                     if (!visit(stillHungry[head])) {
422                         // The connection window has collapsed, break out of the loop.
423                         break;
424                     }
425                 }
426                 connectionWindow = nextConnectionWindow;
427                 totalWeight = nextTotalWeight;
428                 tail = nextTail;
429             }
430         }
431 
432         /**
433          * Indicates that the given child is still hungry (i.e. still has streamable bytes that can
434          * fit within the current connection window).
435          */
436         void stillHungry(Http2Stream child) {
437             ensureSpaceIsAllocated(nextTail);
438             stillHungry[nextTail++] = child;
439         }
440 
441         /**
442          * Ensures that the {@link #stillHungry} array is properly sized to hold the given index.
443          */
444         void ensureSpaceIsAllocated(int index) {
445             if (stillHungry == null) {
446                 // Initial size is 1/4 the number of children. Clipping the minimum at 2, which will over allocate if
447                 // maxSize == 1 but if this was true we shouldn't need to re-allocate because the 1 child should get
448                 // all of the available connection window.
449                 stillHungry = new Http2Stream[max(2, maxSize >>> 2)];
450             } else if (index == stillHungry.length) {
451                 // Grow the array by a factor of 2.
452                 stillHungry = Arrays.copyOf(stillHungry, min(maxSize, stillHungry.length << 1));
453             }
454         }
455     }

457     /**
458      * A simplified version of {@link ChildFeeder} that is only used when all streamable bytes fit within the
459      * available connection window.
460      */
461     private final class SimpleChildFeeder implements Http2StreamVisitor {
462         int bytesAllocated;
463         int connectionWindow;
464 
465         SimpleChildFeeder(int connectionWindow) {
466             this.connectionWindow = connectionWindow;
467         }
468 
469         @Override
470         public boolean visit(Http2Stream child) throws Http2Exception {
471             AbstractState childState = state(child);
472             int bytesForChild = childState.streamableBytes();
473 
474             if (bytesForChild > 0 || childState.hasFrame()) {
475                 childState.allocate(bytesForChild);
476                 bytesAllocated += bytesForChild;
477                 connectionWindow -= bytesForChild;
478             }
479             int childBytesAllocated = allocateBytesForTree(child, connectionWindow);
480             bytesAllocated += childBytesAllocated;
481             connectionWindow -= childBytesAllocated;
482             return true;
483         }
484     }

486     /**
487      * The remote flow control state for a single stream.
488      */
489     private final class DefaultState extends AbstractState {
490         private final Deque<FlowControlled> pendingWriteQueue;
491         private int window;
492         private int pendingBytes;
493         private int allocated;
494         // Set to true while a frame is being written, false otherwise.
495         private boolean writing;
496         // Set to true if cancel() was called.
497         private boolean cancelled;
498 
499         DefaultState(Http2Stream stream, int initialWindowSize) {
500             super(stream);
501             window(initialWindowSize);
502             pendingWriteQueue = new ArrayDeque<FlowControlled>(2);
503         }
504 
505         DefaultState(AbstractState existingState, int initialWindowSize) {
506             super(existingState);
507             window(initialWindowSize);
508             pendingWriteQueue = new ArrayDeque<FlowControlled>(2);
509         }
510 
511         @Override
512         int windowSize() {
513             return window;
514         }
515 
516         @Override
517         int initialWindowSize() {
518             return initialWindowSize;
519         }
520 
521         @Override
522         void window(int initialWindowSize) {
523             window = initialWindowSize;
524         }
525 
526         @Override
527         void allocate(int bytes) {
528             allocated += bytes;
529             // Also artificially reduce the streamable bytes for this tree to give the appearance
530             // that the data has been written. This will be restored before the allocated bytes are
531             // actually written.
532             incrementStreamableBytesForTree(-bytes);
533         }
534 
535         @Override
536         int writeAllocatedBytes() {
537             int numBytes = allocated;
538 
539             // Restore the number of streamable bytes to this branch.
540             incrementStreamableBytesForTree(allocated);
541             resetAllocated();
542 
543             // Perform the write.
544             return writeBytes(numBytes);
545         }
546 
547         /**
548          * Reset the number of bytes that have been allocated to this stream by the priority algorithm.
549          */
550         private void resetAllocated() {
551             allocated = 0;
552         }
553 
554         @Override
555         int incrementStreamWindow(int delta) throws Http2Exception {
556             if (delta > 0 && Integer.MAX_VALUE - delta < window) {
557                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
558                         "Window size overflow for stream: %d", stream.id());
559             }
560             int previouslyStreamable = streamableBytes();
561             window += delta;
562 
563             // Update this branch of the priority tree if the streamable bytes have changed for this node.
564             int streamableDelta = streamableBytes() - previouslyStreamable;
565             if (streamableDelta != 0) {
566                 incrementStreamableBytesForTree(streamableDelta);
567             }
568             return window;
569         }
570 
571         @Override
572         int writableWindow() {
573             return min(window, connectionWindowSize());
574         }
575 
576         @Override
577         int streamableBytes() {
578             return max(0, min(pendingBytes - allocated, window));
579         }
580 
581         @Override
582         int streamableBytesForTree() {
583             return streamableBytesForTree;
584         }
585 
586         @Override
587         void enqueueFrame(FlowControlled frame) {
588             incrementPendingBytes(frame.size());
589             FlowControlled last = pendingWriteQueue.peekLast();
590             if (last == null || !last.merge(ctx, frame)) {
591                 pendingWriteQueue.offer(frame);
592             }
593         }
594 
595         @Override
596         boolean hasFrame() {
597             return !pendingWriteQueue.isEmpty();
598         }
599 
600         /**
601          * Returns the the head of the pending queue, or {@code null} if empty.
602          */
603         private FlowControlled peek() {
604             return pendingWriteQueue.peek();
605         }
606 
607         @Override
608         void cancel() {
609             cancel(null);
610         }
611 
612         /**
613          * Clears the pending queue and writes errors for each remaining frame.
614          * @param cause the {@link Throwable} that caused this method to be invoked.
615          */
616         private void cancel(Throwable cause) {
617             cancelled = true;
618             // Ensure that the queue can't be modified while we are writing.
619             if (writing) {
620                 return;
621             }
622             for (;;) {
623                 FlowControlled frame = pendingWriteQueue.poll();
624                 if (frame == null) {
625                     break;
626                 }
627                 writeError(frame, streamError(stream.id(), INTERNAL_ERROR, cause,
628                                               "Stream closed before write could take place"));
629             }
630         }
631 
632         @Override
633         int writeBytes(int bytes) {
634             if (!hasFrame()) {
635                 return -1;
636             }
637             // Check if the first frame is a "writable" frame to get the "-1" return status out of the way
638             FlowControlled frame = peek();
639             int maxBytes = min(bytes, writableWindow());
640             if (maxBytes <= 0 && frame.size() != 0) {
641                 // The frame had data and all of it was written.
642                 return -1;
643             }
644             int originalBytes = bytes;
645             bytes -= write(frame, maxBytes);
646 
647             // Write the remainder of frames that we are allowed to
648             while (hasFrame()) {
649                 frame = peek();
650                 maxBytes = min(bytes, writableWindow());
651                 if (maxBytes <= 0 && frame.size() != 0) {
652                     // The frame had data and all of it was written.
653                     break;
654                 }
655                 bytes -= write(frame, maxBytes);
656             }
657             return originalBytes - bytes;
658         }
659 
660         /**
661          * Writes the frame and decrements the stream and connection window sizes. If the frame is in the pending
662          * queue, the written bytes are removed from this branch of the priority tree.
663          */
664         private int write(FlowControlled frame, int allowedBytes) {
665             int before = frame.size();
666             int writtenBytes;
667             // In case an exception is thrown we want to remember it and pass it to cancel(Throwable).
668             Throwable cause = null;
669             try {
670                 assert !writing;
671 
672                 // Write the portion of the frame.
673                 writing = true;
674                 frame.write(ctx, max(0, allowedBytes));
675                 if (!cancelled && frame.size() == 0) {
676                     // This frame has been fully written, remove this frame and notify it. Since we remove this frame
677                     // first, we're guaranteed that its error method will not be called when we call cancel.
678                     pendingWriteQueue.remove();
679                     frame.writeComplete();
680                 }
681             } catch (Throwable t) {
682                 // Mark the state as cancelled, we'll clear the pending queue via cancel() below.
683                 cancelled = true;
684                 cause = t;
685             } finally {
686                 writing = false;
687                 // Make sure we always decrement the flow control windows
688                 // by the bytes written.
689                 writtenBytes = before - frame.size();
690                 decrementFlowControlWindow(writtenBytes);
691                 decrementPendingBytes(writtenBytes);
692                 // If a cancellation occurred while writing, call cancel again to
693                 // clear and error all of the pending writes.
694                 if (cancelled) {
695                     cancel(cause);
696                 }
697             }
698             return writtenBytes;
699         }
700 
701         /**
702          * Increments the number of pending bytes for this node. If there was any change to the number of bytes that
703          * fit into the stream window, then {@link #incrementStreamableBytesForTree} is called to recursively update
704          * this branch of the priority tree.
705          */
706         private void incrementPendingBytes(int numBytes) {
707             int previouslyStreamable = streamableBytes();
708             pendingBytes += numBytes;
709 
710             int delta = streamableBytes() - previouslyStreamable;
711             if (delta != 0) {
712                 incrementStreamableBytesForTree(delta);
713             }
714         }
715 
716         /**
717          * If this frame is in the pending queue, decrements the number of pending bytes for the stream.
718          */
719         private void decrementPendingBytes(int bytes) {
720             incrementPendingBytes(-bytes);
721         }

723         /**
724          * Decrement the per stream and connection flow control window by {@code bytes}.
725          */
726         private void decrementFlowControlWindow(int bytes) {
727             try {
728                 int negativeBytes = -bytes;
729                 connectionState().incrementStreamWindow(negativeBytes);
730                 incrementStreamWindow(negativeBytes);
731             } catch (Http2Exception e) {
732                 // Should never get here since we're decrementing.
733                 throw new IllegalStateException("Invalid window state when writing frame: " + e.getMessage(), e);
734             }
735         }

738          * Discards this {@link FlowControlled}, writing an error. If this frame is in the pending queue,
739          * the unwritten bytes are removed from this branch of the priority tree.
740          */
741         private void writeError(FlowControlled frame, Http2Exception cause) {
742             assert ctx != null;
743             decrementPendingBytes(frame.size());
744             frame.error(ctx, cause);
745         }
