32  /**
33   * Provides the default implementation for processing inbound frame events and delegates to a
34   * {@link Http2FrameListener}
35   * <p>
36   * This class will read HTTP/2 frames and delegate the events to a {@link Http2FrameListener}
37   * <p>
38   * This interface enforces inbound flow control functionality through
39   * {@link Http2LocalFlowController}
40   */

103     @Override
104     public Http2Settings localSettings() {
105         Http2Settings settings = new Http2Settings();
106         Http2FrameReader.Configuration config = frameReader.configuration();
107         Http2HeaderTable headerTable = config.headerTable();
108         Http2FrameSizePolicy frameSizePolicy = config.frameSizePolicy();
109         settings.initialWindowSize(flowController().initialWindowSize());
110         settings.maxConcurrentStreams(connection.remote().maxActiveStreams());
111         settings.headerTableSize(headerTable.maxHeaderTableSize());
112         settings.maxFrameSize(frameSizePolicy.maxFrameSize());
113         settings.maxHeaderListSize(headerTable.maxHeaderListSize());
114         if (!connection.isServer()) {
115             // Only set the pushEnabled flag if this is a client endpoint.
116             settings.pushEnabled(connection.local().allowPushTo());
117         }
118         return settings;
119     }

121     @Override
122     public void localSettings(Http2Settings settings) throws Http2Exception {
123         Boolean pushEnabled = settings.pushEnabled();
124         Http2FrameReader.Configuration config = frameReader.configuration();
125         Http2HeaderTable inboundHeaderTable = config.headerTable();
126         Http2FrameSizePolicy inboundFrameSizePolicy = config.frameSizePolicy();
127         if (pushEnabled != null) {
128             if (connection.isServer()) {
129                 throw connectionError(PROTOCOL_ERROR, "Server sending SETTINGS frame with ENABLE_PUSH specified");
130             }
131             connection.local().allowPushTo(pushEnabled);
132         }
133 
134         Long maxConcurrentStreams = settings.maxConcurrentStreams();
135         if (maxConcurrentStreams != null) {
136             int value = (int) Math.min(maxConcurrentStreams, Integer.MAX_VALUE);
137             connection.remote().maxActiveStreams(value);
138         }
139 
140         Long headerTableSize = settings.headerTableSize();
141         if (headerTableSize != null) {
142             inboundHeaderTable.maxHeaderTableSize((int) Math.min(headerTableSize, Integer.MAX_VALUE));
143         }
144 
145         Integer maxHeaderListSize = settings.maxHeaderListSize();
146         if (maxHeaderListSize != null) {
147             inboundHeaderTable.maxHeaderListSize(maxHeaderListSize);
148         }
149 
150         Integer maxFrameSize = settings.maxFrameSize();
151         if (maxFrameSize != null) {
152             inboundFrameSizePolicy.maxFrameSize(maxFrameSize);
153         }
154 
155         Integer initialWindowSize = settings.initialWindowSize();
156         if (initialWindowSize != null) {
157             flowController().initialWindowSize(initialWindowSize);
158         }
159     }

184     private final class FrameReadListener implements Http2FrameListener {
185         @Override
186         public int onDataRead(final ChannelHandlerContext ctx, int streamId, ByteBuf data,
187                 int padding, boolean endOfStream) throws Http2Exception {
188             Http2Stream stream = connection.stream(streamId);
189             Http2LocalFlowController flowController = flowController();
190             int bytesToReturn = data.readableBytes() + padding;
191 
192             if (stream == null || stream.isResetSent() || streamCreatedAfterGoAwaySent(streamId)) {
193                 // Ignoring this frame. We still need to count the frame towards the connection flow control
194                 // window, but we immediately mark all bytes as consumed.
195                 flowController.receiveFlowControlledFrame(ctx, stream, data, padding, endOfStream);
196                 flowController.consumeBytes(ctx, stream, bytesToReturn);
197 
198                 // Verify that the stream may have existed after we apply flow control.
199                 verifyStreamMayHaveExisted(streamId);
200 
201                 // All bytes have been consumed.
202                 return bytesToReturn;
203             }
204 
205             Http2Exception error = null;
206             switch (stream.state()) {
207                 case OPEN:
208                 case HALF_CLOSED_LOCAL:
209                     break;
210                 case HALF_CLOSED_REMOTE:
211                 case CLOSED:
212                     error = streamError(stream.id(), STREAM_CLOSED, "Stream %d in unexpected state: %s",
213                         stream.id(), stream.state());
214                     break;
215                 default:
216                     error = streamError(stream.id(), PROTOCOL_ERROR,
217                         "Stream %d in unexpected state: %s", stream.id(), stream.state());
218                     break;
219             }
220 
221             int unconsumedBytes = unconsumedBytes(stream);
222             try {
223                 flowController.receiveFlowControlledFrame(ctx, stream, data, padding, endOfStream);
224                 // Update the unconsumed bytes after flow control is applied.
225                 unconsumedBytes = unconsumedBytes(stream);
226 
227                 // If the stream is in an invalid state to receive the frame, throw the error.
228                 if (error != null) {
229                     throw error;
230                 }
231 
232                 // Call back the application and retrieve the number of bytes that have been
233                 // immediately processed.
234                 bytesToReturn = listener.onDataRead(ctx, streamId, data, padding, endOfStream);
235                 return bytesToReturn;
236             } catch (Http2Exception e) {
237                 // If an exception happened during delivery, the listener may have returned part
238                 // of the bytes before the error occurred. If that's the case, subtract that from
239                 // the total processed bytes so that we don't return too many bytes.
240                 int delta = unconsumedBytes - unconsumedBytes(stream);
241                 bytesToReturn -= delta;
242                 throw e;
243             } catch (RuntimeException e) {
244                 // If an exception happened during delivery, the listener may have returned part
245                 // of the bytes before the error occurred. If that's the case, subtract that from
246                 // the total processed bytes so that we don't return too many bytes.
247                 int delta = unconsumedBytes - unconsumedBytes(stream);
248                 bytesToReturn -= delta;
249                 throw e;
250             } finally {
251                 // If appropriate, return the processed bytes to the flow controller.
252                 if (bytesToReturn > 0) {
253                     flowController.consumeBytes(ctx, stream, bytesToReturn);
254                 }
255 
256                 if (endOfStream) {
257                     lifecycleManager.closeStreamRemote(stream, ctx.newSucceededFuture());
258                 }
259             }
260         }

268         @Override
269         public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int streamDependency,
270                 short weight, boolean exclusive, int padding, boolean endOfStream) throws Http2Exception {
271             Http2Stream stream = connection.stream(streamId);
272             boolean allowHalfClosedRemote = false;
273             if (stream == null && !connection.streamMayHaveExisted(streamId)) {
274                 stream = connection.remote().createStream(streamId, endOfStream);
275                 // Allow the state to be HALF_CLOSE_REMOTE if we're creating it in that state.
276                 allowHalfClosedRemote = stream.state() == HALF_CLOSED_REMOTE;
277             }
278 
279             if (stream == null || stream.isResetSent() || streamCreatedAfterGoAwaySent(streamId)) {
280                 // Ignore this frame.
281                 return;
282             }
283 
284             switch (stream.state()) {
285                 case RESERVED_REMOTE:
286                     stream.open(endOfStream);
287                     break;
288                 case OPEN:
289                 case HALF_CLOSED_LOCAL:
290                     // Allowed to receive headers in these states.
291                     break;
292                 case HALF_CLOSED_REMOTE:
293                     if (!allowHalfClosedRemote) {
294                         throw streamError(stream.id(), STREAM_CLOSED, "Stream %d in unexpected state: %s",
295                                 stream.id(), stream.state());
296                     }
297                     break;
298                 case CLOSED:
299                     throw streamError(stream.id(), STREAM_CLOSED, "Stream %d in unexpected state: %s",
300                             stream.id(), stream.state());
301                 default:
302                     // Connection error.
303                     throw connectionError(PROTOCOL_ERROR, "Stream %d in unexpected state: %s", stream.id(),
304                             stream.state());
305             }
306 
307             try {
308                 // This call will create a stream for streamDependency if necessary.
309                 // For this reason it must be done before notifying the listener.
310                 stream.setPriority(streamDependency, weight, exclusive);
311             } catch (ClosedStreamCreationException ignored) {
312                 // It is possible that either the stream for this frame or the parent stream is closed.
313                 // In this case we should ignore the exception and allow the frame to be sent.
314             }
315 
316             listener.onHeadersRead(ctx, streamId, headers, streamDependency, weight, exclusive, padding, endOfStream);
317 
318             // If the headers completes this stream, close it.
319             if (endOfStream) {
320                 lifecycleManager.closeStreamRemote(stream, ctx.newSucceededFuture());
321             }
322         }

324         @Override
325         public void onPriorityRead(ChannelHandlerContext ctx, int streamId, int streamDependency, short weight,
326                 boolean exclusive) throws Http2Exception {
327             Http2Stream stream = connection.stream(streamId);
328 
329             try {
330                 if (stream == null) {
331                     if (connection.streamMayHaveExisted(streamId)) {
332                         // Ignore this frame.
333                         return;
334                     }
335 
336                     // PRIORITY frames always identify a stream. This means that if a PRIORITY frame is the
337                     // first frame to be received for a stream that we must create the stream.
338                     stream = connection.remote().createIdleStream(streamId);
339                 } else if (streamCreatedAfterGoAwaySent(streamId)) {
340                     // Ignore this frame.
341                     return;
342                 }
343 
344                 // This call will create a stream for streamDependency if necessary.
345                 // For this reason it must be done before notifying the listener.
346                 stream.setPriority(streamDependency, weight, exclusive);
347             } catch (ClosedStreamCreationException ignored) {
348                 // It is possible that either the stream for this frame or the parent stream is closed.
349                 // In this case we should ignore the exception and allow the frame to be sent.
350             }
351 
352             listener.onPriorityRead(ctx, streamId, streamDependency, weight, exclusive);
353         }

355         @Override
356         public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception {
357             Http2Stream stream = connection.stream(streamId);
358             if (stream == null) {
359                 verifyStreamMayHaveExisted(streamId);
360                 return;
361             }
362 
363             switch(stream.state()) {
364             case IDLE:
365                 throw connectionError(PROTOCOL_ERROR, "RST_STREAM received for IDLE stream %d", streamId);
366             case CLOSED:
367                 return; // RST_STREAM frames must be ignored for closed streams.
368             default:
369                 break;
370             }
371 
372             listener.onRstStreamRead(ctx, streamId, errorCode);
373 
374             lifecycleManager.closeStream(stream, ctx.newSucceededFuture());
375         }

389         /**
390          * Applies settings sent from the local endpoint.
391          */
392         private void applyLocalSettings(Http2Settings settings) throws Http2Exception {
393             Boolean pushEnabled = settings.pushEnabled();
394             final Http2FrameReader.Configuration config = frameReader.configuration();
395             final Http2HeaderTable headerTable = config.headerTable();
396             final Http2FrameSizePolicy frameSizePolicy = config.frameSizePolicy();
397             if (pushEnabled != null) {
398                 if (connection.isServer()) {
399                     throw connectionError(PROTOCOL_ERROR, "Server sending SETTINGS frame with ENABLE_PUSH specified");
400                 }
401                 connection.local().allowPushTo(pushEnabled);
402             }
403 
404             Long maxConcurrentStreams = settings.maxConcurrentStreams();
405             if (maxConcurrentStreams != null) {
406                 int value = (int) Math.min(maxConcurrentStreams, Integer.MAX_VALUE);
407                 connection.remote().maxActiveStreams(value);
408             }
409 
410             Long headerTableSize = settings.headerTableSize();
411             if (headerTableSize != null) {
412                 headerTable.maxHeaderTableSize((int) Math.min(headerTableSize, Integer.MAX_VALUE));
413             }
414 
415             Integer maxHeaderListSize = settings.maxHeaderListSize();
416             if (maxHeaderListSize != null) {
417                 headerTable.maxHeaderListSize(maxHeaderListSize);
418             }
419 
420             Integer maxFrameSize = settings.maxFrameSize();
421             if (maxFrameSize != null) {
422                 frameSizePolicy.maxFrameSize(maxFrameSize);
423             }
424 
425             Integer initialWindowSize = settings.initialWindowSize();
426             if (initialWindowSize != null) {
427                 flowController().initialWindowSize(initialWindowSize);
428             }
429         }

455         @Override
456         public void onPushPromiseRead(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
457                 Http2Headers headers, int padding) throws Http2Exception {
458             Http2Stream parentStream = connection.stream(streamId);
459 
460             if (streamCreatedAfterGoAwaySent(streamId)) {
461                 return;
462             }
463 
464             if (parentStream == null) {
465                 throw connectionError(PROTOCOL_ERROR, "Stream does not exist %d", streamId);
466             }
467 
468             switch (parentStream.state()) {
469               case OPEN:
470               case HALF_CLOSED_LOCAL:
471                   // Allowed to receive push promise in these states.
472                   break;
473               default:
474                   // Connection error.
475                   throw connectionError(PROTOCOL_ERROR,
476                       "Stream %d in unexpected state for receiving push promise: %s",
477                       parentStream.id(), parentStream.state());
478             }
479 
480             if (!requestVerifier.isAuthoritative(ctx, headers)) {
481                 throw streamError(promisedStreamId, PROTOCOL_ERROR,
482                         "Promised request on stream %d for promised stream %d is not authoritative",
483                         streamId, promisedStreamId);
484             }
485             if (!requestVerifier.isCacheable(headers)) {
486                 throw streamError(promisedStreamId, PROTOCOL_ERROR,
487                         "Promised request on stream %d for promised stream %d is not known to be cacheable",
488                         streamId, promisedStreamId);
489             }
490             if (!requestVerifier.isSafe(headers)) {
491                 throw streamError(promisedStreamId, PROTOCOL_ERROR,
492                         "Promised request on stream %d for promised stream %d is not known to be safe",
493                         streamId, promisedStreamId);
494             }
495 
496             // Reserve the push stream based with a priority based on the current stream's priority.
497             connection.remote().reservePushStream(promisedStreamId, parentStream);
498 
499             listener.onPushPromiseRead(ctx, streamId, promisedStreamId, headers, padding);
500         }

530         /**
531          * Helper method for determining whether or not to ignore inbound frames. A stream is considered to be created
532          * after a {@code GOAWAY} is sent if the following conditions hold:
533          * <p/>
534          * <ul>
535          *     <li>A {@code GOAWAY} must have been sent by the local endpoint</li>
536          *     <li>The {@code streamId} must identify a legitimate stream id for the remote endpoint to be creating</li>
537          *     <li>{@code streamId} is greater than the Last Known Stream ID which was sent by the local endpoint
538          *     in the last {@code GOAWAY} frame</li>
539          * </ul>
540          * <p/>
541          */
542         private boolean streamCreatedAfterGoAwaySent(int streamId) {
543             Http2Connection.Endpoint<?> remote = connection.remote();
544             return connection.goAwaySent() && remote.isValidStreamId(streamId) &&
545                     streamId > remote.lastStreamKnownByPeer();
546         }

548         private void verifyStreamMayHaveExisted(int streamId) throws Http2Exception {
549             if (!connection.streamMayHaveExisted(streamId)) {
550                 throw connectionError(PROTOCOL_ERROR, "Stream does not exist %d", streamId);
551             }
552         }

557          * Verifies that the HTTP/2 connection preface has been received from the remote endpoint.
558          * It is possible that the current call to
559          * {@link Http2FrameReader#readFrame(ChannelHandlerContext, ByteBuf, Http2FrameListener)} will have multiple
