40  /**
41   * An HTTP/2 handler that creates child channels for each stream. Creating outgoing streams is not
42   * yet supported. Server-side HTTP to HTTP/2 upgrade is supported in conjunction with {@link
43   * Http2ServerUpgradeCodec}; the necessary HTTP-to-HTTP/2 conversion is performed automatically.
44   *
45   * <p><em>This API is very immature.</em> The Http2Connection-based API is currently preferred over
46   * this API. This API is targeted to eventually replace or reduce the need for the
47   * Http2Connection-based API.
48   *
49   * <p>This handler notifies the pipeline of channel events, such as {@link Http2GoAwayFrame}. It
50   * is also capable of writing such messages. Directly writing {@link Http2StreamFrame}s for this
51   * handler is unsupported.
52   *
53   * <h3>Child Channels</h3>
54   *
55   * <p>When a new stream is created, a new {@link Channel} is created for it. Applications send and
56   * receive {@link Http2StreamFrame}s on the created channel. The {@link Http2StreamFrame#stream} is
57   * expected to be {@code null}, but the channel can use the field for its own bookkeeping. {@link
58   * ByteBuf}s cannot be processed by the channel; all writes that reach the head of the pipeline must
59   * be an instance of {@link Http2StreamFrame}.  Writes that reach the head of the pipeline are
60   * processed directly by this handler and cannot be intercepted.
61   *
62   * <p>The child channel will be notified of user events that impact the stream, such as {@link
63   * Http2GoAwayFrame} and {@link Http2ResetFrame}, as soon as they occur. Although {@code
64   * Http2GoAwayFrame} and {@code Http2ResetFrame} signify that the remote is ignoring further
65   * communication, closing of the channel is delayed until any inbound queue is drained with {@link
66   * Channel#read()}, which follows the default behavior of channels in Netty. Applications are
67   * free to close the channel in response to such events if they don't have use for any queued
68   * messages.
69   *
70   * <p>{@link ChannelConfig#setMaxMessagesPerRead(int)} and {@link
71   * ChannelConfig#setAutoRead(boolean)} are supported.
72   */
73  @UnstableApi

110     Http2MultiplexCodec(boolean server, ChannelHandler streamHandler,
111             EventLoopGroup streamGroup, Http2FrameWriter frameWriter) {
112         if (!streamHandler.getClass().isAnnotationPresent(Sharable.class)) {
113             throw new IllegalArgumentException("streamHandler must be Sharable");
114         }
115         this.streamHandler = streamHandler;
116         this.streamGroup = streamGroup;
117         Http2Connection connection = new DefaultHttp2Connection(server);
118         frameWriter = new Http2OutboundFrameLogger(frameWriter, HTTP2_FRAME_LOGGER);
119         Http2ConnectionEncoder encoder = new DefaultHttp2ConnectionEncoder(connection, frameWriter);
120         Http2FrameReader reader = new Http2InboundFrameLogger(new DefaultHttp2FrameReader(), HTTP2_FRAME_LOGGER);
121         Http2ConnectionDecoder decoder = new DefaultHttp2ConnectionDecoder(connection, encoder, reader);
122         decoder.frameListener(new FrameListener());
123         http2Handler = new InternalHttp2ConnectionHandler(decoder, encoder, new Http2Settings());
124         http2Handler.connection().addListener(new ConnectionListener());
125         streamInfoKey = http2Handler.connection().newKey();
126     }

150     /**
151      * Handles the cleartext HTTP upgrade event. If an upgrade occurred, sends a simple response via
152      * HTTP/2 on stream 1 (the stream specifically reserved for cleartext HTTP upgrade).
153      */
154     @Override
155     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
156         if (!(evt instanceof UpgradeEvent)) {
157             super.userEventTriggered(ctx, evt);
158             return;
159         }
160 
161         UpgradeEvent upgrade = (UpgradeEvent) evt;
162         ctx.fireUserEventTriggered(upgrade.retain());
163         try {
164             Http2Stream stream = http2Handler.connection().stream(Http2CodecUtil.HTTP_UPGRADE_STREAM_ID);
165             // TODO: improve handler/stream lifecycle so that stream isn't active before handler added.
166             // The stream was already made active, but ctx may have been null so it wasn't initialized.
167             // https://github.com/netty/netty/issues/4942
168             new ConnectionListener().onStreamActive(stream);
169             upgrade.upgradeRequest().headers().setInt(
170                 HttpConversionUtil.ExtensionHeaderNames.STREAM_ID.text(), Http2CodecUtil.HTTP_UPGRADE_STREAM_ID);
171             new InboundHttpToHttp2Adapter(http2Handler.connection(), http2Handler.decoder().frameListener())
172                 .channelRead(ctx, upgrade.upgradeRequest().retain());
173         } finally {
174             upgrade.release();
175         }
176     }

190     void flushFromStreamChannel() {
191         EventExecutor executor = ctx.executor();
192         if (executor.inEventLoop()) {
193             flush(ctx);
194         } else {
195             Runnable task = flushTask;
196             if (task == null) {
197                 task = flushTask = new Runnable() {
198                     @Override
199                     public void run() {
200                         flush(ctx);
201                     }
202                 };
203             }
204             executor.execute(task);
205         }
206     }

208     void writeFromStreamChannel(final Object msg, final boolean flush) {
209         final ChannelPromise promise = ctx.newPromise();
210         EventExecutor executor = ctx.executor();
211         if (executor.inEventLoop()) {
212             writeFromStreamChannel0(msg, flush, promise);
213         } else {
214             try {
215                 executor.execute(new OneTimeTask() {
216                     @Override
217                     public void run() {
218                         writeFromStreamChannel0(msg, flush, promise);
219                     }
220                 });
221             } catch (Throwable cause) {
222                 promise.setFailure(cause);
223             }
224         }
225     }

227     private void writeFromStreamChannel0(Object msg, boolean flush, ChannelPromise promise) {
228         try {
229             write(ctx, msg, promise);
230         } catch (Throwable cause) {
231             promise.tryFailure(cause);
232         }
233         if (flush) {
234             flush(ctx);
235         }
236     }

238     /**
239      * Processes all {@link Http2Frame}s. {@link Http2StreamFrame}s may only originate in child
240      * streams.
241      */
242     @Override
243     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) {
244         if (!(msg instanceof Http2Frame)) {
245             ctx.write(msg, promise);
246             return;
247         }
248         try {
249             if (msg instanceof Http2StreamFrame) {
250                 Object streamObject = ((Http2StreamFrame) msg).stream();
251                 int streamId = ((Http2StreamChannel) streamObject).stream.id();
252                 if (msg instanceof Http2DataFrame) {
253                     Http2DataFrame frame = (Http2DataFrame) msg;
254                     http2Handler.encoder().writeData(http2HandlerCtx, streamId, frame.content().retain(),
255                         frame.padding(), frame.isEndStream(), promise);
256                 } else if (msg instanceof Http2HeadersFrame) {
257                     Http2HeadersFrame frame = (Http2HeadersFrame) msg;
258                     http2Handler.encoder().writeHeaders(
259                         http2HandlerCtx, streamId, frame.headers(), frame.padding(), frame.isEndStream(), promise);
260                 } else if (msg instanceof Http2ResetFrame) {
261                     Http2ResetFrame frame = (Http2ResetFrame) msg;
262                     http2Handler.resetStream(http2HandlerCtx, streamId, frame.errorCode(), promise);
263                 } else {
264                     throw new UnsupportedMessageTypeException(msg);
265                 }
266             } else if (msg instanceof Http2GoAwayFrame) {
267                 Http2GoAwayFrame frame = (Http2GoAwayFrame) msg;
268                 int lastStreamId = http2Handler.connection().remote().lastStreamCreated()
269                     + frame.extraStreamIds() * 2;
270                 http2Handler.goAway(
271                     http2HandlerCtx, lastStreamId, frame.errorCode(), frame.content().retain(), promise);
272             } else {
273                 throw new UnsupportedMessageTypeException(msg);
274             }
275         } finally {
276             ReferenceCountUtil.release(msg);
277         }
278     }

280     ChannelFuture createStreamChannel(ChannelHandlerContext ctx, Http2Stream stream,
281             ChannelHandler handler) {
282         EventLoopGroup group = streamGroup != null ? streamGroup : ctx.channel().eventLoop();
283         Http2StreamChannel channel = new Http2StreamChannel(stream);
284         channel.pipeline().addLast(handler);
285         ChannelFuture future = group.register(channel);
286         // Handle any errors that occurred on the local thread while registering. Even though
287         // failures can happen after this point, they will be handled by the channel by closing the
288         // channel.
289         if (future.cause() != null) {
290             if (channel.isRegistered()) {
291                 channel.close();
292             } else {
293                 channel.unsafe().closeForcibly();
294             }
295         }
296         return future;
297     }

299     /**
300      * Notifies any child streams of the read completion.
301      */
302     @Override
303     public void channelReadComplete(ChannelHandlerContext ctx) {
304         for (int i = 0; i < streamsToFireChildReadComplete.size(); i++) {
305             final StreamInfo streamInfo = streamsToFireChildReadComplete.get(i);
306             // Clear early in case fireChildReadComplete() causes it to need to be re-processed
307             streamInfo.inStreamsToFireChildReadComplete = false;
308             streamInfo.childChannel.fireChildReadComplete();
309         }
310         streamsToFireChildReadComplete.clear();
311     }
312 
313     void fireChildReadAndRegister(StreamInfo streamInfo, Object msg) {
314         // Can't use childChannel.fireChannelRead() as it would fire independent of whether
315         // channel.read() had been called.
316         streamInfo.childChannel.fireChildRead(msg);
317         if (!streamInfo.inStreamsToFireChildReadComplete) {
318             streamsToFireChildReadComplete.add(streamInfo);
319             streamInfo.inStreamsToFireChildReadComplete = true;
320         }
321     }

323     final class ConnectionListener extends Http2ConnectionAdapter {
324         @Override
325         public void onStreamActive(Http2Stream stream) {
326             if (ctx == null) {
327                 // UPGRADE stream is active before handlerAdded().
328                 return;
329             }
330             // If it is an outgoing stream, then we already created the channel.
331             // TODO: support outgoing streams. https://github.com/netty/netty/issues/4913
332             if (stream.getProperty(streamInfoKey) != null) {
333                 return;
334             }
335             ChannelFuture future = createStreamChannel(ctx, stream, streamHandler);
336             stream.setProperty(streamInfoKey, new StreamInfo((Http2StreamChannel) future.channel()));
337         }
338 
339         @Override
340         public void onStreamClosed(Http2Stream stream) {
341             final StreamInfo streamInfo = stream.getProperty(streamInfoKey);
342             if (streamInfo != null) {
343                 final EventLoop eventLoop = streamInfo.childChannel.eventLoop();
344                 if (eventLoop.inEventLoop()) {
345                     onStreamClosed0(streamInfo);
346                 } else {
347                     eventLoop.execute(new OneTimeTask() {
348                         @Override
349                         public void run() {
350                             onStreamClosed0(streamInfo);
351                         }
352                     });
353                 }
354             }
355         }
356 
357         private void onStreamClosed0(StreamInfo streamInfo) {
358             streamInfo.childChannel.onStreamClosedFired = true;
359             streamInfo.childChannel.fireChildRead(AbstractHttp2StreamChannel.CLOSE_MESSAGE);
360         }
361 
362         @Override
363         public void onGoAwayReceived(final int lastStreamId, long errorCode, ByteBuf debugData) {
364             final Http2GoAwayFrame goAway = new DefaultHttp2GoAwayFrame(errorCode, debugData);
365             try {
366                 http2Handler.connection().forEachActiveStream(new Http2StreamVisitor() {
367                     @Override
368                     public boolean visit(Http2Stream stream) {
369                         if (stream.id() > lastStreamId
370                                 && http2Handler.connection().local().isValidStreamId(stream.id())) {
371                             StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
372                             // TODO: Can we force a user interaction pattern that doesn't require us to duplicate()?
373                             // https://github.com/netty/netty/issues/4943
374                             streamInfo.childChannel.pipeline().fireUserEventTriggered(goAway.retainedDuplicate());
375                         }
376                         return true;
377                     }
378                 });
379             } catch (final Throwable t) {
380                 EventExecutor executor = ctx.executor();
381                 if (executor.inEventLoop()) {
382                     exceptionCaught(ctx, t);
383                 } else {
384                     executor.execute(new OneTimeTask() {
385                         @Override
386                         public void run() {
387                             exceptionCaught(ctx, t);
388                         }
389                     });
390                 }
391             }
392             ctx.fireUserEventTriggered(goAway.retainedDuplicate());
393         }
394     }

402         @Override
403         protected void onStreamError(ChannelHandlerContext ctx, Throwable cause,
404                 Http2Exception.StreamException http2Ex) {
405             try {
406                 Http2Stream stream = http2Handler.connection().stream(http2Ex.streamId());
407                 if (stream == null) {
408                     return;
409                 }
410                 StreamInfo streamInfo = stream.getProperty(streamInfoKey);
411                 if (streamInfo == null) {
412                     return;
413                 }
414                 streamInfo.childChannel.pipeline().fireExceptionCaught(http2Ex);
415             } finally {
416                 super.onStreamError(ctx, cause, http2Ex);
417             }
418         }

421     class FrameListener extends Http2FrameAdapter {
422         @Override
423         public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode)
424                 throws Http2Exception {
425             Http2Stream stream = http2Handler.connection().stream(streamId);
426             StreamInfo streamInfo = stream.getProperty(streamInfoKey);
427             // Use a user event in order to circumvent read queue.

434                 exclusive, int padding, boolean endStream) throws Http2Exception {
435             onHeadersRead(ctx, streamId, headers, padding, endStream);
436         }
437 
438         @Override
439         public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers,
440                 int padding, boolean endOfStream) throws Http2Exception {
441             Http2Stream stream = http2Handler.connection().stream(streamId);
442             StreamInfo streamInfo = stream.getProperty(streamInfoKey);
443             fireChildReadAndRegister(streamInfo, new DefaultHttp2HeadersFrame(headers, endOfStream, padding));
444         }
445 
446         @Override
447         public int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding,
448                 boolean endOfStream) throws Http2Exception {
449             Http2Stream stream = http2Handler.connection().stream(streamId);
450             StreamInfo streamInfo = stream.getProperty(streamInfoKey);
451             fireChildReadAndRegister(streamInfo, new DefaultHttp2DataFrame(data.retain(), endOfStream, padding));
452             // We return the bytes in bytesConsumed() once the stream channel consumed the bytes.
453             return 0;

469     // This class uses ctx.invoker().invoke* instead of ctx.* to send to the ctx's handler instead
470     // of the 'next' handler.
471     final class Http2StreamChannel extends AbstractHttp2StreamChannel {
472         private final Http2Stream stream;
473         boolean onStreamClosedFired;
474 
475         Http2StreamChannel(Http2Stream stream) {
476             super(ctx.channel());
477             this.stream = stream;
478         }
479 
480         @Override
481         protected void doClose() throws Exception {
482             if (!onStreamClosedFired) {
483                 Http2StreamFrame resetFrame = new DefaultHttp2ResetFrame(Http2Error.CANCEL).setStream(this);
484                 writeFromStreamChannel(resetFrame, true);
485             }
486             super.doClose();
487         }
488 
489         @Override
490         protected void doWrite(Object msg) {
491             if (!(msg instanceof Http2StreamFrame)) {
492                 ReferenceCountUtil.release(msg);
493                 throw new IllegalArgumentException("Message must be an Http2StreamFrame: " + msg);
494             }
495             Http2StreamFrame frame = (Http2StreamFrame) msg;
496             if (frame.stream() != null) {
497                 ReferenceCountUtil.release(frame);
498                 throw new IllegalArgumentException("Stream must be null on the frame");
499             }
500             frame.setStream(this);
501 
502             writeFromStreamChannel(msg, false);
503         }
504 
505         @Override
506         protected void doWriteComplete() {
507             flushFromStreamChannel();
508         }
509 
510         @Override
511         protected EventExecutor preferredEventExecutor() {
512             return ctx.executor();
513         }
514 
515         @Override
516         protected void bytesConsumed(final int bytes) {
517             EventExecutor executor = ctx.executor();
518             if (executor.inEventLoop()) {
519                 bytesConsumed0(bytes);
520             } else {
521                 executor.execute(new OneTimeTask() {
522                     @Override
523                     public void run() {
524                         bytesConsumed0(bytes);
525                     }
526                 });
527             }
528         }
529 
530         private void bytesConsumed0(int bytes) {
531             try {
532                 http2Handler.connection().local().flowController().consumeBytes(stream, bytes);
533             } catch (Throwable t) {
534                 exceptionCaught(ctx, t);
535             }
536         }
537     }
