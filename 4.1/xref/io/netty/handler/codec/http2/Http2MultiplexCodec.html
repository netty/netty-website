53  /**
54   * An HTTP/2 handler that creates child channels for each stream.
55   *
56   * <p>When a new stream is created, a new {@link Channel} is created for it. Applications send and
57   * receive {@link Http2StreamFrame}s on the created channel. {@link ByteBuf}s cannot be processed by the channel;
58   * all writes that reach the head of the pipeline must be an instance of {@link Http2StreamFrame}. Writes that reach
59   * the head of the pipeline are processed directly by this handler and cannot be intercepted.
60   *
61   * <p>The child channel will be notified of user events that impact the stream, such as {@link
62   * Http2GoAwayFrame} and {@link Http2ResetFrame}, as soon as they occur. Although {@code
63   * Http2GoAwayFrame} and {@code Http2ResetFrame} signify that the remote is ignoring further
64   * communication, closing of the channel is delayed until any inbound queue is drained with {@link
65   * Channel#read()}, which follows the default behavior of channels in Netty. Applications are
66   * free to close the channel in response to such events if they don't have use for any queued
67   * messages.
68   *
69   * <p>Outbound streams are supported via the {@link Http2StreamChannelBootstrap}.
70   *
71   * <p>{@link ChannelConfig#setMaxMessagesPerRead(int)} and {@link ChannelConfig#setAutoRead(boolean)} are supported.
72   */
73  @UnstableApi
74  public final class Http2MultiplexCodec extends ChannelDuplexHandler {

87      /**
88       * Construct a new handler whose child channels run in a different event loop.
89       *
90       * @param server {@code true} this is a server
91       * @param bootstrap bootstrap used to instantiate child channels for remotely-created streams.
92       */
93      public Http2MultiplexCodec(boolean server, Http2StreamChannelBootstrap bootstrap) {
94          if (bootstrap.parentChannel() != null) {
95              throw new IllegalStateException("The parent channel must not be set on the bootstrap.");
96          }
97          this.server = server;
98          this.bootstrap = new Http2StreamChannelBootstrap(bootstrap);
99      }

107     @Override
108     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
109         if (!(cause instanceof StreamException)) {
110             ctx.fireExceptionCaught(cause);
111             return;
112         }
113 
114         StreamException streamEx = (StreamException) cause;
115         try {
116             Http2StreamChannel childChannel = childChannels.get(streamEx.streamId());
117             if (childChannel != null) {
118                 childChannel.pipeline().fireExceptionCaught(streamEx);
119             } else {
120                 logger.warn(format("Exception caught for unknown HTTP/2 stream '%d'", streamEx.streamId()),
121                             streamEx);
122             }
123         } finally {
124             onStreamClosed(streamEx.streamId());
125         }
126     }

134     @Override
135     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
136         if (!(msg instanceof Http2Frame)) {
137             ctx.fireChannelRead(msg);
138             return;
139         }
140         if (msg instanceof Http2StreamFrame) {
141             Http2StreamFrame frame = (Http2StreamFrame) msg;
142             int streamId = frame.streamId();
143             Http2StreamChannel childChannel = childChannels.get(streamId);
144             if (childChannel == null) {
145                 // TODO: Combine with DefaultHttp2ConnectionDecoder.shouldIgnoreHeadersOrDataFrame logic.
146                 ReferenceCountUtil.release(msg);
147                 throw new StreamException(streamId, STREAM_CLOSED, format("Received %s frame for an unknown stream %d",
148                                                                           frame.name(), streamId));
149             }
150             fireChildReadAndRegister(childChannel, frame);
151         } else if (msg instanceof Http2GoAwayFrame) {
152             Http2GoAwayFrame goAwayFrame = (Http2GoAwayFrame) msg;
153             for (PrimitiveEntry<Http2StreamChannel> entry : childChannels.entries()) {
154                 Http2StreamChannel childChannel = entry.value();
155                 int streamId = entry.key();
156                 if (streamId > goAwayFrame.lastStreamId() && isOutboundStream(server, streamId)) {
157                     childChannel.pipeline().fireUserEventTriggered(goAwayFrame.retainedDuplicate());
158                 }
159             }
160             goAwayFrame.release();
161         } else {
162             // It's safe to release, as UnsupportedMessageTypeException just calls msg.getClass()
163             ReferenceCountUtil.release(msg);
164             throw new UnsupportedMessageTypeException(msg);
165         }
166     }
167 
168     private void fireChildReadAndRegister(Http2StreamChannel childChannel, Http2StreamFrame frame) {
169         // Can't use childChannel.fireChannelRead() as it would fire independent of whether
170         // channel.read() had been called.
171         childChannel.fireChildRead(frame);
172         if (!childChannel.inStreamsToFireChildReadComplete) {
173             channelsToFireChildReadComplete.add(childChannel);
174             childChannel.inStreamsToFireChildReadComplete = true;
175         }
176     }

178     @Override
179     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
180         if (evt instanceof Http2StreamActiveEvent) {
181             Http2StreamActiveEvent activeEvent = (Http2StreamActiveEvent) evt;
182             onStreamActive(activeEvent.streamId(), activeEvent.headers());
183         } else if (evt instanceof Http2StreamClosedEvent) {
184             onStreamClosed(((Http2StreamClosedEvent) evt).streamId());
185         } else {
186             ctx.fireUserEventTriggered(evt);
187         }
188     }

190     private void onStreamActive(int streamId, Http2HeadersFrame headersFrame) {
191         final Http2StreamChannel childChannel;
192         if (isOutboundStream(server, streamId)) {
193             if (!(headersFrame instanceof ChannelCarryingHeadersFrame)) {
194                 throw new IllegalArgumentException("needs to be wrapped");
195             }
196             childChannel = ((ChannelCarryingHeadersFrame) headersFrame).channel();
197             childChannel.streamId(streamId);
198         } else {
199             ChannelFuture future = bootstrap.connect(streamId);
200             childChannel = (Http2StreamChannel) future.channel();
201         }
202 
203         Http2StreamChannel existing = childChannels.put(streamId, childChannel);
204         assert existing == null;
205     }
206 
207     private void onStreamClosed(int streamId) {
208         final Http2StreamChannel childChannel = childChannels.remove(streamId);
209         if (childChannel != null) {
210             final EventLoop eventLoop = childChannel.eventLoop();
211             if (eventLoop.inEventLoop()) {
212                 onStreamClosed0(childChannel);
213             } else {
214                 eventLoop.execute(new Runnable() {
215                     @Override
216                     public void run() {
217                         onStreamClosed0(childChannel);
218                     }
219                 });
220             }
221         }
222     }
223 
224     private void onStreamClosed0(Http2StreamChannel childChannel) {
225         assert childChannel.eventLoop().inEventLoop();
226 
227         childChannel.onStreamClosedFired = true;
228         childChannel.fireChildRead(CLOSE_MESSAGE);
229     }

231     void flushFromStreamChannel() {
232         EventExecutor executor = ctx.executor();
233         if (executor.inEventLoop()) {
234             flush(ctx);
235         } else {
236             Runnable task = flushTask;
237             if (task == null) {
238                 task = flushTask = new Runnable() {
239                     @Override
240                     public void run() {
241                         flush(ctx);
242                     }
243                 };
244             }
245             executor.execute(task);
246         }
247     }

249     void writeFromStreamChannel(Object msg, boolean flush) {
250         writeFromStreamChannel(msg, ctx.newPromise(), flush);
251     }
252 
253     void writeFromStreamChannel(final Object msg, final ChannelPromise promise, final boolean flush) {
254         EventExecutor executor = ctx.executor();
255         if (executor.inEventLoop()) {
256             writeFromStreamChannel0(msg, flush, promise);
257         } else {
258             try {
259                 executor.execute(new Runnable() {
260                     @Override
261                     public void run() {
262                         writeFromStreamChannel0(msg, flush, promise);
263                     }
264                 });
265             } catch (Throwable cause) {
266                 promise.setFailure(cause);
267             }
268         }
269     }
270 
271     private void writeFromStreamChannel0(Object msg, boolean flush, ChannelPromise promise) {
272         try {
273             write(ctx, msg, promise);
274         } catch (Throwable cause) {
275             promise.tryFailure(cause);
276         }
277         if (flush) {
278             flush(ctx);
279         }
280     }

283      * Notifies any child streams of the read completion.
284      */
285     @Override
286     public void channelReadComplete(ChannelHandlerContext ctx) {
287         for (int i = 0; i < channelsToFireChildReadComplete.size(); i++) {
288             Http2StreamChannel childChannel = channelsToFireChildReadComplete.get(i);
289             // Clear early in case fireChildReadComplete() causes it to need to be re-processed
290             childChannel.inStreamsToFireChildReadComplete = false;
291             childChannel.fireChildReadComplete();
292         }
293         channelsToFireChildReadComplete.clear();
294     }

296     ChannelFuture createStreamChannel(Channel parentChannel, EventLoopGroup group, ChannelHandler handler,
297                                               Map<ChannelOption<?>, Object> options,
298                                               Map<AttributeKey<?>, Object> attrs,
299                                               int streamId) {
300         final Http2StreamChannel channel = new Http2StreamChannel(parentChannel);
301         if (isStreamIdValid(streamId)) {
302             assert !isOutboundStream(server, streamId);
303             assert ctx.channel().eventLoop().inEventLoop();
304             channel.streamId(streamId);
305         }
306         channel.pipeline().addLast(handler);
307 
308         initOpts(channel, options);
309         initAttrs(channel, attrs);
310 
311         ChannelFuture future = group.register(channel);
312         // Handle any errors that occurred on the local thread while registering. Even though
313         // failures can happen after this point, they will be handled by the channel by closing the
314         // channel.
315         if (future.cause() != null) {
316             if (channel.isRegistered()) {
317                 channel.close();
318             } else {
319                 channel.unsafe().closeForcibly();
320             }
321         }
322         return future;
323     }

325     @SuppressWarnings("unchecked")
326     private static void initOpts(Channel channel, Map<ChannelOption<?>, Object> opts) {
327         if (opts != null) {
328             for (Entry<ChannelOption<?>, Object> e: opts.entrySet()) {
329                 try {
330                     if (!channel.config().setOption((ChannelOption<Object>) e.getKey(), e.getValue())) {
331                         logger.warn("Unknown channel option: " + e);
332                     }
333                 } catch (Throwable t) {
334                     logger.warn("Failed to set a channel option: " + channel, t);
335                 }
336             }
337         }

349     final class Http2StreamChannel extends AbstractHttp2StreamChannel implements ChannelFutureListener {
350         boolean onStreamClosedFired;
351 
352         /**
353          * {@code true} if stream is in {@link Http2MultiplexCodec#channelsToFireChildReadComplete}.
354          */
355         boolean inStreamsToFireChildReadComplete;
356 
357         Http2StreamChannel(Channel parentChannel) {
358             super(parentChannel);
359         }
360 
361         @Override
362         protected void doClose() throws Exception {
363             if (!onStreamClosedFired && isStreamIdValid(streamId())) {
364                 Http2StreamFrame resetFrame = new DefaultHttp2ResetFrame(Http2Error.CANCEL).streamId(streamId());
365                 writeFromStreamChannel(resetFrame, true);
366             }
367             super.doClose();
368         }
369 
370         @Override
371         protected void doWrite(Object msg) {
372 
373             if (msg instanceof Http2StreamFrame) {
374                 Http2StreamFrame frame = (Http2StreamFrame) msg;
375                 ChannelPromise promise = ctx.newPromise();
376                 if (isStreamIdValid(frame.streamId())) {
377                     ReferenceCountUtil.release(frame);
378                     throw new IllegalArgumentException("Stream id must not be set on the frame. Was: "
379                         + frame.streamId());
380                 }
381                 if (!isStreamIdValid(streamId())) {
382                     if (!(frame instanceof Http2HeadersFrame)) {
383                         ReferenceCountUtil.release(frame);
384                         throw new IllegalArgumentException("The first frame must be a headers frame. Was: "
385                             + frame.name());
386                     }
387                     frame = new ChannelCarryingHeadersFrame((Http2HeadersFrame) frame, this);
388                     // Handle errors on stream creation
389                     promise.addListener(this);
390                 } else {
391                     frame.streamId(streamId());
392                 }
393                 writeFromStreamChannel(frame, promise, false);
394             } else if (msg instanceof Http2GoAwayFrame) {
395                 ChannelPromise promise = ctx.newPromise();
396                 promise.addListener(this);
397                 writeFromStreamChannel(msg, promise, false);
398             } else {
399                 ReferenceCountUtil.release(msg);
400                 throw new IllegalArgumentException("Message must be an Http2GoAwayFrame or Http2StreamFrame: " + msg);
401             }
402         }
403 
404         @Override
405         protected void doWriteComplete() {
406             flushFromStreamChannel();
407         }
408 
409         @Override
410         protected EventExecutor preferredEventExecutor() {
411             return ctx.executor();
412         }
413 
414         @Override
415         protected void bytesConsumed(final int bytes) {
416             ctx.write(new DefaultHttp2WindowUpdateFrame(bytes).streamId(streamId()));
417         }
418 
419         @Override
420         public void operationComplete(ChannelFuture future) throws Exception {
421             Throwable cause = future.cause();
422             if (cause != null) {
423                 pipeline().fireExceptionCaught(cause);
424                 close();
425             }
426         }
427     }
