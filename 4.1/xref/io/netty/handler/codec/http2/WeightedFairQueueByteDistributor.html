30  /**
31   * A {@link StreamByteDistributor} that is sensitive to stream priority and uses
32   * <a href="https://en.wikipedia.org/wiki/Weighted_fair_queueing">Weighted Fair Queueing</a> approach for distributing
33   * bytes.
34   * <p>
35   * Inspiration for this distributor was taken from Linux's
36   * <a href="https://git.kernel.org/cgit/linux/kernel/git/stable/linux-stable.git/tree/Documentation/scheduler
37   * /sched-design-CFS.txt">Completely Fair Scheduler</a>
38   * to model the distribution of bytes to simulate an "ideal multi-tasking CPU", but in this case we are simulating
39   * an "ideal multi-tasking NIC".
40   * <p>
41   * Each write operation will use the {@link #allocationQuantum(int)} to know how many more bytes should be allocated
42   * relative to the next stream which wants to write. This is to balance fairness while also considering goodput.
43   */
44  @UnstableApi
45  public final class WeightedFairQueueByteDistributor implements StreamByteDistributor {
46      private final Http2Connection.PropertyKey stateKey;
47      private final State connectionState;
48      /**
49       * The minimum number of bytes that we will attempt to allocate to a stream. This is to
50       * help improve goodput on a per-stream basis.
51       */
52      private int allocationQuantum = 1024;
53  
54      public WeightedFairQueueByteDistributor(Http2Connection connection) {
55          stateKey = connection.newKey();
56          Http2Stream connectionStream = connection.connectionStream();
57          connectionStream.setProperty(stateKey, connectionState = new State(connectionStream, 16));
58  
59          // Register for notification of new streams.
60          connection.addListener(new Http2ConnectionAdapter() {
61              @Override
62              public void onStreamAdded(Http2Stream stream) {
63                  stream.setProperty(stateKey, new State(stream));
64              }
65  
66              @Override
67              public void onWeightChanged(Http2Stream stream, short oldWeight) {
68                  Http2Stream parent;
69                  if (state(stream).activeCountForTree != 0 && (parent = stream.parent()) != null) {
70                      state(parent).totalQueuedWeights += stream.weight() - oldWeight;
71                  }
72              }
73  
74              @Override
75              public void onStreamClosed(Http2Stream stream) {
76                  state(stream).close();
77              }
78  
79              @Override
80              public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {
81                  Http2Stream parent = stream.parent();
82                  if (parent != null) {
83                      State state = state(stream);
84                      if (state.activeCountForTree != 0) {
85                          State pState = state(parent);
86                          pState.offerAndInitializePseudoTime(state);
87                          pState.activeCountChangeForTree(state.activeCountForTree);
88                      }
89                  }
90              }
91  
92              @Override
93              public void onPriorityTreeParentChanging(Http2Stream stream, Http2Stream newParent) {
94                  Http2Stream parent = stream.parent();
95                  if (parent != null) {
96                      State state = state(stream);
97                      if (state.activeCountForTree != 0) {
98                          State pState = state(parent);
99                          pState.remove(state);
100                         pState.activeCountChangeForTree(-state.activeCountForTree);
101                     }
102                 }
103             }
104         });
105     }

108     public void updateStreamableBytes(StreamState state) {
109         state(state.stream()).updateStreamableBytes(streamableBytes(state),
110                                                     state.hasFrame() && state.windowSize() >= 0);

113     @Override
114     public boolean distribute(int maxBytes, Writer writer) throws Http2Exception {
115         // As long as there is some active frame we should write at least 1 time.
116         if (connectionState.activeCountForTree == 0) {
117             return false;
118         }
119 
120         // The goal is to write until we write all the allocated bytes or are no longer making progress.
121         // We still attempt to write even after the number of allocated bytes has been exhausted to allow empty frames
122         // to be sent. Making progress means the active streams rooted at the connection stream has changed.
123         int oldIsActiveCountForTree;
124         do {
125             oldIsActiveCountForTree = connectionState.activeCountForTree;
126             // connectionState will never be active, so go right to its children.
127             maxBytes -= distributeToChildren(maxBytes, writer, connectionState);
128         } while (connectionState.activeCountForTree != 0 &&
129                 (maxBytes > 0 || oldIsActiveCountForTree != connectionState.activeCountForTree));
130 
131         return connectionState.activeCountForTree != 0;
132     }

134     /**
135      * Sets the amount of bytes that will be allocated to each stream. Defaults to 1KiB.
136      * @param allocationQuantum the amount of bytes that will be allocated to each stream. Must be &gt; 0.
137      */
138     public void allocationQuantum(int allocationQuantum) {
139         if (allocationQuantum <= 0) {
140             throw new IllegalArgumentException("allocationQuantum must be > 0");
141         }
142         this.allocationQuantum = allocationQuantum;
143     }

145     private int distribute(int maxBytes, Writer writer, State state) throws Http2Exception {
146         if (state.isActive()) {
147             int nsent = min(maxBytes, state.streamableBytes);
148             state.write(nsent, writer);
149             if (nsent == 0 && maxBytes != 0) {
150                 // If a stream sends zero bytes, then we gave it a chance to write empty frames and it is now
151                 // considered inactive until the next call to updateStreamableBytes. This allows descendant streams to
152                 // be allocated bytes when the parent stream can't utilize them. This may be as a result of the
153                 // stream's flow control window being 0.
154                 state.updateStreamableBytes(state.streamableBytes, false);
155             }
156             return nsent;
157         }
158 
159         return distributeToChildren(maxBytes, writer, state);
160     }
161 
162     /**
163      * It is a pre-condition that {@code state.poll()} returns a non-{@code null} value. This is a result of the way
164      * the allocation algorithm is structured and can be explained in the following cases:
165      * <h3>For the recursive case</h3>
166      * If a stream has no children (in the allocation tree) than that node must be active or it will not be in the
167      * allocation tree. If a node is active then it will not delegate to children and recursion ends.
168      * <h3>For the initial case</h3>
169      * We check connectionState.activeCountForTree == 0 before any allocation is done. So if the connection stream
170      * has no active children we don't get into this method.
171      */
172     private int distributeToChildren(int maxBytes, Writer writer, State state) throws Http2Exception {
173         long oldTotalQueuedWeights = state.totalQueuedWeights;
174         State childState = state.poll();
175         State nextChildState = state.peek();
176         childState.setDistributing();
177         try {
178             assert nextChildState == null || nextChildState.pseudoTimeToWrite >= childState.pseudoTimeToWrite :
179                 "nextChildState[" + nextChildState.stream.id() + "].pseudoTime(" + nextChildState.pseudoTimeToWrite +
180                 ") < " + " childState[" + childState.stream.id() + "].pseudoTime(" + childState.pseudoTimeToWrite + ")";
181             int nsent = distribute(nextChildState == null ? maxBytes :
182                             min(maxBytes, (int) min((nextChildState.pseudoTimeToWrite - childState.pseudoTimeToWrite) *
183                                                 childState.stream.weight() / oldTotalQueuedWeights + allocationQuantum,
184                                                 Integer.MAX_VALUE)
185                                ),
186                                writer,
187                                childState);
188             state.pseudoTime += nsent;
189             childState.updatePseudoTime(state, nsent, oldTotalQueuedWeights);
190             return nsent;
191         } finally {
192             childState.unsetDistributing();
193             // Do in finally to ensure the internal flags is not corrupted if an exception is thrown.
194             // The offer operation is delayed until we unroll up the recursive stack, so we don't have to remove from
195             // the priority queue due to a write operation.
196             if (childState.activeCountForTree != 0) {
197                 state.offer(childState);
198             }
199         }
200     }

213     /**
214      * The remote flow control state for a single stream.
215      */
216     private final class State implements PriorityQueueNode<State> {
217         private static final int STATE_IS_ACTIVE = 0x1;
218         private static final int STATE_IS_DISTRIBUTING = 0x2;
219         final Http2Stream stream;
220         private final Queue<State> queue;
221         int streamableBytes;
222         /**
223          * Count of nodes rooted at this sub tree with {@link #isActive()} equal to {@code true}.
224          */
225         int activeCountForTree;
226         private int priorityQueueIndex = INDEX_NOT_IN_QUEUE;
227         /**
228          * An estimate of when this node should be given the opportunity to write data.
229          */
230         long pseudoTimeToWrite;
231         /**
232          * A pseudo time maintained for immediate children to base their {@link #pseudoTimeToWrite} off of.
233          */
234         long pseudoTime;
235         long totalQueuedWeights;
236         private byte flags;
237 
238         State(Http2Stream stream) {
239             this(stream, 0);
240         }
241 
242         State(Http2Stream stream, int initialSize) {
243             this.stream = stream;
244             queue = new PriorityQueue<State>(initialSize);
245         }
246 
247         void write(int numBytes, Writer writer) throws Http2Exception {
248             try {
249                 writer.write(stream, numBytes);
250             } catch (Throwable t) {
251                 throw connectionError(INTERNAL_ERROR, t, "byte distribution write error");
252             }
253         }
254 
255         void activeCountChangeForTree(int increment) {
256             assert activeCountForTree + increment >= 0;
257             activeCountForTree += increment;
258             if (!stream.isRoot()) {
259                 State pState = state(stream.parent());
260                 assert activeCountForTree != increment ||
261                        priorityQueueIndex == INDEX_NOT_IN_QUEUE ||
262                        pState.queue.contains(this) :
263                      "State[" + stream.id() + "].activeCountForTree changed from 0 to " + increment + " is in a queue" +
264                      ", but not in parent[ " + pState.stream.id() + "]'s queue";
265                 if (activeCountForTree == 0) {
266                     pState.remove(this);
267                 } else if (activeCountForTree == increment && !isDistributing()) {
268                     // If frame count was 0 but is now not, and this node is not already in a queue (assumed to be
269                     // pState's queue) then enqueue it. If this State object is being processed the pseudoTime for this
270                     // node should not be adjusted, and the node will be added back to the queue/tree structure after it
271                     // is done being processed. This may happen if the activeCountForTree == 0 (a node which can't
272                     // stream anything and is blocked) is at/near root of the tree, and is poped off the queue during
273                     // processing, and then put back on the queue because a child changes position in the priority tree
274                     // (or is closed because it is not blocked and finished writing all data).
275                     pState.offerAndInitializePseudoTime(this);
276                 }
277                 pState.activeCountChangeForTree(increment);
278             }
279         }
280 
281         void updateStreamableBytes(int newStreamableBytes, boolean isActive) {
282             if (isActive() != isActive) {
283                 if (isActive) {
284                     activeCountChangeForTree(1);
285                     setActive();
286                 } else {
287                     activeCountChangeForTree(-1);
288                     unsetActive();
289                 }
290             }
291 
292             streamableBytes = newStreamableBytes;
293         }
294 
295         /**
296          * Assumes the parents {@link #totalQueuedWeights} includes this node's weight.
297          */
298         void updatePseudoTime(State parentState, int nsent, long totalQueuedWeights) {
299             assert stream.id() != CONNECTION_STREAM_ID && nsent >= 0;
300             // If the current pseudoTimeToSend is greater than parentState.pseudoTime then we previously over accounted
301             // and should use parentState.pseudoTime.
302             pseudoTimeToWrite = min(pseudoTimeToWrite, parentState.pseudoTime) +
303                                 nsent * totalQueuedWeights / stream.weight();
304         }
305 
306         /**
307          * The concept of pseudoTime can be influenced by priority tree manipulations or if a stream goes from "active"
308          * to "non-active". This method accounts for that by initializing the {@link #pseudoTimeToWrite} for
309          * {@code state} to {@link #pseudoTime} of this node and then calls {@link #offer(State)}.
310          */
311         void offerAndInitializePseudoTime(State state) {
312             state.pseudoTimeToWrite = pseudoTime;
313             offer(state);
314         }
315 
316         void offer(State state) {
317             queue.offer(state);
318             totalQueuedWeights += state.stream.weight();
319         }
320 
321         /**
322          * Must only be called if the queue is non-empty!
323          */
324         State poll() {
325             State state = queue.poll();
326             // This method is only ever called if the queue is non-empty.
327             totalQueuedWeights -= state.stream.weight();
328             return state;
329         }
330 
331         void remove(State state) {
332             if (queue.remove(state)) {
333                 totalQueuedWeights -= state.stream.weight();
334             }
335         }
336 
337         State peek() {
338             return queue.peek();
339         }
340 
341         void close() {
342             updateStreamableBytes(0, false);
343         }

345         boolean isActive() {
346             return (flags & STATE_IS_ACTIVE) != 0;
347         }
348 
349         private void setActive() {
350             flags |= STATE_IS_ACTIVE;
351         }
352 
353         private void unsetActive() {
354             flags &= ~STATE_IS_ACTIVE;
355         }
356 
357         boolean isDistributing() {
358             return (flags & STATE_IS_DISTRIBUTING) != 0;
359         }
360 
361         void setDistributing() {
362             flags |= STATE_IS_DISTRIBUTING;
363         }
364 
365         void unsetDistributing() {
366             flags &= ~STATE_IS_DISTRIBUTING;
367         }

384         @Override
385         public String toString() {
386             // Use activeCountForTree as a rough estimate for how many nodes are in this subtree.
387             StringBuilder sb = new StringBuilder(256 * (activeCountForTree > 0 ? activeCountForTree : 1));
388             toString(sb);
389             return sb.toString();
390         }
391 
392         private void toString(StringBuilder sb) {
393             sb.append("{stream ").append(stream.id())
394                     .append(" streamableBytes ").append(streamableBytes)
395                     .append(" activeCountForTree ").append(activeCountForTree)
396                     .append(" priorityQueueIndex ").append(priorityQueueIndex)
397                     .append(" pseudoTimeToWrite ").append(pseudoTimeToWrite)
398                     .append(" pseudoTime ").append(pseudoTime)
399                     .append(" flags ").append(flags)
400                     .append(" queue.size() ").append(queue.size()).append("} [");
401 
402             if (!queue.isEmpty()) {
403                 for (State s : queue) {
404                     s.toString(sb);
405                     sb.append(", ");
406                 }
407                 // Remove the last ", "
408                 sb.setLength(sb.length() - 2);
409             }
410             sb.append(']');
411         }
