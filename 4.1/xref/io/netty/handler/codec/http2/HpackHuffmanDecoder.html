42  final class HpackHuffmanDecoder {
43  
44      private static final Http2Exception EOS_DECODED = ThrowableUtil.unknownStackTrace(
45              Http2Exception.newStatic(COMPRESSION_ERROR, "HPACK - EOS Decoded",
46                      Http2Exception.ShutdownHint.HARD_SHUTDOWN), HpackHuffmanDecoder.class, "decode(..)");
47      private static final Http2Exception INVALID_PADDING = ThrowableUtil.unknownStackTrace(
48              Http2Exception.newStatic(COMPRESSION_ERROR, "HPACK - Invalid Padding",
49                      Http2Exception.ShutdownHint.HARD_SHUTDOWN), HpackHuffmanDecoder.class, "decode(..)");

51      private static final Node ROOT = buildTree(HpackUtil.HUFFMAN_CODES, HpackUtil.HUFFMAN_CODE_LENGTHS);
52  
53      private final DecoderProcessor processor;
54  
55      HpackHuffmanDecoder(int initialCapacity) {
56          processor = new DecoderProcessor(initialCapacity);
57      }
58  
59      /**
60       * Decompresses the given Huffman coded string literal.
61       *
62       * @param buf the string literal to be decoded
63       * @return the output stream for the compressed data
64       * @throws Http2Exception EOS Decoded
65       */
66      public AsciiString decode(ByteBuf buf, int length) throws Http2Exception {
67          processor.reset();
68          buf.forEachByte(buf.readerIndex(), length, processor);
69          buf.skipBytes(length);
70          return processor.end();
71      }

73      private static final class Node {
74  
75          private final int symbol;      // terminal nodes have a symbol
76          private final int bits;        // number of bits matched by the node
77          private final Node[] children; // internal nodes have children
78  
79          /**
80           * Construct an internal node
81           */
82          Node() {
83              symbol = 0;
84              bits = 8;
85              children = new Node[256];
86          }
87  
88          /**
89           * Construct a terminal node
90           *
91           * @param symbol the symbol the node represents
92           * @param bits the number of bits matched by this node
93           */
94          Node(int symbol, int bits) {
95              assert bits > 0 && bits <= 8;
96              this.symbol = symbol;
97              this.bits = bits;
98              children = null;
99          }
100 
101         private boolean isTerminal() {
102             return children == null;
103         }
104     }

106     private static Node buildTree(int[] codes, byte[] lengths) {
107         Node root = new Node();
108         for (int i = 0; i < codes.length; i++) {
109             insert(root, i, codes[i], lengths[i]);
110         }
111         return root;
112     }
113 
114     private static void insert(Node root, int symbol, int code, byte length) {
115         // traverse tree using the most significant bytes of code
116         Node current = root;
117         while (length > 8) {
118             if (current.isTerminal()) {
119                 throw new IllegalStateException("invalid Huffman code: prefix not unique");
120             }
121             length -= 8;
122             int i = (code >>> length) & 0xFF;
123             if (current.children[i] == null) {
124                 current.children[i] = new Node();
125             }
126             current = current.children[i];
127         }
128 
129         Node terminal = new Node(symbol, length);
130         int shift = 8 - length;
131         int start = (code << shift) & 0xFF;
132         int end = 1 << shift;
133         for (int i = start; i < start + end; i++) {
134             current.children[i] = terminal;
135         }
136     }

138     private static final class DecoderProcessor implements ByteProcessor {
139         private final int initialCapacity;
140         private byte[] bytes;
141         private int index;
142         private Node node;
143         private int current;
144         private int currentBits;
145         private int symbolBits;
146 
147         DecoderProcessor(int initialCapacity) {
148             this.initialCapacity = ObjectUtil.checkPositive(initialCapacity, "initialCapacity");
149         }
150 
151         void reset() {
152             node = ROOT;
153             current = 0;
154             currentBits = 0;
155             symbolBits = 0;
156             bytes = new byte[initialCapacity];
157             index = 0;
158         }
159 
160         /*
161          * The idea here is to consume whole bytes at a time rather than individual bits. node
162          * represents the Huffman tree, with all bit patterns denormalized as 256 children. Each
163          * child represents the last 8 bits of the huffman code. The parents of each child each
164          * represent the successive 8 bit chunks that lead up to the last most part. 8 bit bytes
165          * from buf are used to traverse these tree until a terminal node is found.
166          *
167          * current is a bit buffer. The low order bits represent how much of the huffman code has
168          * not been used to traverse the tree. Thus, the high order bits are just garbage.
169          * currentBits represents how many of the low order bits of current are actually valid.
170          * currentBits will vary between 0 and 15.
171          *
172          * symbolBits is the number of bits of the symbol being decoded, *including* all those of
173          * the parent nodes. symbolBits tells how far down the tree we are. For example, when
174          * decoding the invalid sequence {0xff, 0xff}, currentBits will be 0, but symbolBits will be
175          * 16. This is used to know if buf ended early (before consuming a whole symbol) or if
176          * there is too much padding.
177          */
178         @Override
179         public boolean process(byte value) throws Http2Exception {
180             current = (current << 8) | (value & 0xFF);
181             currentBits += 8;
182             symbolBits += 8;
183             // While there are unconsumed bits in current, keep consuming symbols.
184             do {
185                 node = node.children[(current >>> (currentBits - 8)) & 0xFF];
186                 currentBits -= node.bits;
187                 if (node.isTerminal()) {
188                     if (node.symbol == HpackUtil.HUFFMAN_EOS) {
189                         throw EOS_DECODED;
190                     }
191                     append(node.symbol);
192                     node = ROOT;
193                     // Upon consuming a whole symbol, reset the symbol bits to the number of bits
194                     // left over in the byte.
195                     symbolBits = currentBits;
196                 }
197             } while (currentBits >= 8);
198             return true;
199         }
200 
201         AsciiString end() throws Http2Exception {
202             /*
203              * We have consumed all the bytes in buf, but haven't consumed all the symbols. We may be on
204              * a partial symbol, so consume until there is nothing left. This will loop at most 2 times.
205              */
206             while (currentBits > 0) {
207                 node = node.children[(current << (8 - currentBits)) & 0xFF];
208                 if (node.isTerminal() && node.bits <= currentBits) {
209                     if (node.symbol == HpackUtil.HUFFMAN_EOS) {
210                         throw EOS_DECODED;
211                     }
212                     currentBits -= node.bits;
213                     append(node.symbol);
214                     node = ROOT;
215                     symbolBits = currentBits;
216                 } else {
217                     break;
218                 }
219             }
220 
221             // Section 5.2. String Literal Representation
222             // A padding strictly longer than 7 bits MUST be treated as a decoding error.
223             // Padding not corresponding to the most significant bits of the code
224             // for the EOS symbol (0xFF) MUST be treated as a decoding error.
225             int mask = (1 << symbolBits) - 1;
226             if (symbolBits > 7 || (current & mask) != mask) {
227                 throw INVALID_PADDING;
228             }
229 
230             return new AsciiString(bytes, 0, index, false);
231         }
232 
233         private void append(int i) {
234             if (bytes.length == index) {
235                 // Choose an expanding strategy depending on how big the buffer already is.
236                 // 1024 was choosen as a good guess and we may be able to investigate more if there are better choices.
237                 // See also https://github.com/netty/netty/issues/6846
238                 final int newLength = bytes.length >= 1024 ? bytes.length + initialCapacity : bytes.length << 1;
239                 byte[] newBytes = new byte[newLength];
240                 System.arraycopy(bytes, 0, newBytes, 0, bytes.length);
241                 bytes = newBytes;
242             }
243             bytes[index++] = (byte) i;
244         }
245     }
