41  /**
42   * <p> {@link PcapWriteHandler} captures {@link ByteBuf} from {@link SocketChannel} / {@link ServerChannel}
43   * or {@link DatagramPacket} and writes it into Pcap {@link OutputStream}. </p>
44   *
45   * <p>
46   * Things to keep in mind when using {@link PcapWriteHandler} with TCP:
47   *
48   *    <ul>
49   *        <li> Whenever {@link ChannelInboundHandlerAdapter#channelActive(ChannelHandlerContext)} is called,
50   *        a fake TCP 3-way handshake (SYN, SYN+ACK, ACK) is simulated as new connection in Pcap. </li>
51   *
52   *        <li> Whenever {@link ChannelInboundHandlerAdapter#handlerRemoved(ChannelHandlerContext)} is called,
53   *        a fake TCP 3-way handshake (FIN+ACK, FIN+ACK, ACK) is simulated as connection shutdown in Pcap.  </li>
54   *
55   *        <li> Whenever {@link ChannelInboundHandlerAdapter#exceptionCaught(ChannelHandlerContext, Throwable)}
56   *        is called, a fake TCP RST is sent to simulate connection Reset in Pcap. </li>
57   *
58   *        <li> ACK is sent each time data is send / received. </li>
59   *
60   *        <li> Zero Length Data Packets can cause TCP Double ACK error in Wireshark. To tackle this,
61   *        set {@code captureZeroByte} to {@code false}. </li>
62   *    </ul>
63   * </p>
64   */
65  public final class PcapWriteHandler extends ChannelDuplexHandler implements Closeable {
66  
67      /**
68       * Logger for logging events
69       */
70      private final InternalLogger logger = InternalLoggerFactory.getInstance(PcapWriteHandler.class);
71  
72      /**
73       * {@link PcapWriter} Instance
74       */
75      private PcapWriter pCapWriter;
76  
77      /**
78       * {@link OutputStream} where we'll write Pcap data.
79       */
80      private final OutputStream outputStream;
81  
82      /**
83       * {@code true} if we want to capture packets with zero bytes else {@code false}.
84       */
85      private final boolean captureZeroByte;
86  
87      /**
88       * {@code true} if we want to write Pcap Global Header on initialization of
89       * {@link PcapWriter} else {@code false}.
90       */
91      private final boolean writePcapGlobalHeader;
92  
93      /**
94       * TCP Sender Segment Number.
95       * It'll start with 1 and keep incrementing with number of bytes read/sent.
96       */
97      private int sendSegmentNumber = 1;
98  
99      /**
100      * TCP Receiver Segment Number.
101      * It'll start with 1 and keep incrementing with number of bytes read/sent.
102      */
103     private int receiveSegmentNumber = 1;
104 
105     /**
106      * Source Address
107      */
108     private InetSocketAddress srcAddr;
109 
110     /**
111      * Destination Address
112      */
113     private InetSocketAddress dstAddr;
114 
115     /**
116      * Set to {@code true} if {@link #close()} is called and we should stop writing Pcap.
117      */
118     private boolean isClosed;
119 
120     /**
121      * Create new {@link PcapWriteHandler} Instance.
122      * {@code captureZeroByte} is set to {@code false} and
123      * {@code writePcapGlobalHeader} is set to {@code true}.
124      *
125      * @param outputStream OutputStream where Pcap data will be written. Call {@link #close()} to close this
126      *                     OutputStream.
127      * @throws NullPointerException If {@link OutputStream} is {@code null} then we'll throw an
128      *                              {@link NullPointerException}
129      */
130     public PcapWriteHandler(OutputStream outputStream) {
131         this(outputStream, false, true);
132     }
133 
134     /**
135      * Create new {@link PcapWriteHandler} Instance
136      *
137      * @param outputStream          OutputStream where Pcap data will be written. Call {@link #close()} to close this
138      *                              OutputStream.
139      * @param captureZeroByte       Set to {@code true} to enable capturing packets with empty (0 bytes) payload.
140      *                              Otherwise, if set to {@code false}, empty packets will be filtered out.
141      * @param writePcapGlobalHeader Set to {@code true} to write Pcap Global Header on initialization.
142      *                              Otherwise, if set to {@code false}, Pcap Global Header will not be written
143      *                              on initialization. This could when writing Pcap data on a existing file where
144      *                              Pcap Global Header is already present.
145      * @throws NullPointerException If {@link OutputStream} is {@code null} then we'll throw an
146      *                              {@link NullPointerException}
147      */
148     public PcapWriteHandler(OutputStream outputStream, boolean captureZeroByte, boolean writePcapGlobalHeader) {
149         this.outputStream = ObjectUtil.checkNotNull(outputStream, "OutputStream");
150         this.captureZeroByte = captureZeroByte;
151         this.writePcapGlobalHeader = writePcapGlobalHeader;
152     }
153 
154     @Override
155     public void channelActive(ChannelHandlerContext ctx) throws Exception {
156 
157         ByteBufAllocator byteBufAllocator = ctx.alloc();
158 
159         /*
160          * If `writePcapGlobalHeader` is `true`, we'll write Pcap Global Header.
161          */
162         if (writePcapGlobalHeader) {
163 
164             ByteBuf byteBuf = byteBufAllocator.buffer();
165             try {
166                 this.pCapWriter = new PcapWriter(this.outputStream, byteBuf);
167             } catch (IOException ex) {
168                 ctx.channel().close();
169                 ctx.fireExceptionCaught(ex);
170                 logger.error("Caught Exception While Initializing PcapWriter, Closing Channel.", ex);
171             } finally {
172                 byteBuf.release();
173             }
174         } else {
175             this.pCapWriter = new PcapWriter(this.outputStream);
176         }
177 
178         // If Channel belongs to `SocketChannel` then we're handling TCP.
179         if (ctx.channel() instanceof SocketChannel) {
180 
181             // Capture correct `localAddress` and `remoteAddress`
182             if (ctx.channel().parent() instanceof ServerSocketChannel) {
183                 srcAddr = (InetSocketAddress) ctx.channel().remoteAddress();
184                 dstAddr = (InetSocketAddress) ctx.channel().localAddress();
185             } else {
186                 srcAddr = (InetSocketAddress) ctx.channel().localAddress();
187                 dstAddr = (InetSocketAddress) ctx.channel().remoteAddress();
188             }
189 
190             logger.debug("Initiating Fake TCP 3-Way Handshake");
191 
192             ByteBuf tcpBuf = byteBufAllocator.buffer();
193 
194             try {
195                 // Write SYN with Normal Source and Destination Address
196                 TCPPacket.writePacket(tcpBuf, null, 0, 0, srcAddr.getPort(), dstAddr.getPort(), TCPPacket.TCPFlag.SYN);
197                 completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
198 
199                 // Write SYN+ACK with Reversed Source and Destination Address
200                 TCPPacket.writePacket(tcpBuf, null, 0, 1, dstAddr.getPort(), srcAddr.getPort(), TCPPacket.TCPFlag.SYN,
201                         TCPPacket.TCPFlag.ACK);
202                 completeTCPWrite(dstAddr, srcAddr, tcpBuf, byteBufAllocator, ctx);
203 
204                 // Write ACK with Normal Source and Destination Address
205                 TCPPacket.writePacket(tcpBuf, null, 1, 1, srcAddr.getPort(), dstAddr.getPort(), TCPPacket.TCPFlag.ACK);
206                 completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
207             } finally {
208                 tcpBuf.release();
209             }
210 
211             logger.debug("Finished Fake TCP 3-Way Handshake");
212         } else if (ctx.channel() instanceof DatagramChannel) {
213             DatagramChannel datagramChannel = (DatagramChannel) ctx.channel();
214 
215             // If `DatagramChannel` is connected then we can get
216             // `localAddress` and `remoteAddress` from Channel.
217             if (datagramChannel.isConnected()) {
218                 srcAddr = (InetSocketAddress) ctx.channel().localAddress();
219                 dstAddr = (InetSocketAddress) ctx.channel().remoteAddress();
220             }
221         }
222 
223         super.channelActive(ctx);
224     }
225 
226     @Override
227     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
228         if (!isClosed) {
229             if (ctx.channel() instanceof SocketChannel) {
230                 handleTCP(ctx, msg, false);
231             } else if (ctx.channel() instanceof DatagramChannel) {
232                 handleUDP(ctx, msg);
233             } else {
234                 logger.debug("Discarding Pcap Write for Unknown Channel Type: {}", ctx.channel());
235             }
236         }
237         super.channelRead(ctx, msg);
238     }
239 
240     @Override
241     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
242         if (!isClosed) {
243             if (ctx.channel() instanceof SocketChannel) {
244                 handleTCP(ctx, msg, true);
245             } else if (ctx.channel() instanceof DatagramChannel) {
246                 handleUDP(ctx, msg);
247             } else {
248                 logger.debug("Discarding Pcap Write for Unknown Channel Type: {}", ctx.channel());
249             }
250         }
251         super.write(ctx, msg, promise);
252     }

254     /**
255      * Handle TCP L4
256      *
257      * @param ctx              {@link ChannelHandlerContext} for {@link ByteBuf} allocation and
258      *                         {@code fireExceptionCaught}
259      * @param msg              {@link Object} must be {@link ByteBuf} else it'll be discarded
260      * @param isWriteOperation Set {@code true} if we have to process packet when packets are being sent out
261      *                         else set {@code false}
262      */
263     private void handleTCP(ChannelHandlerContext ctx, Object msg, boolean isWriteOperation) {
264         if (msg instanceof ByteBuf) {
265 
266             // If bytes are 0 and `captureZeroByte` is false, we won't capture this.
267             if (((ByteBuf) msg).readableBytes() == 0 && !captureZeroByte) {
268                 logger.debug("Discarding Zero Byte TCP Packet. isWriteOperation {}", isWriteOperation);
269                 return;
270             }
271 
272             ByteBufAllocator byteBufAllocator = ctx.alloc();
273             ByteBuf packet = ((ByteBuf) msg).duplicate();
274             ByteBuf tcpBuf = byteBufAllocator.buffer();
275             int bytes = packet.readableBytes();
276 
277             try {
278                 if (isWriteOperation) {
279                     TCPPacket.writePacket(tcpBuf, packet, sendSegmentNumber, receiveSegmentNumber, srcAddr.getPort(),
280                             dstAddr.getPort(), TCPPacket.TCPFlag.ACK);
281                     completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
282                     logTCP(true, bytes, sendSegmentNumber, receiveSegmentNumber, srcAddr, dstAddr, false);
283 
284                     sendSegmentNumber += bytes;
285 
286                     TCPPacket.writePacket(tcpBuf, null, receiveSegmentNumber, sendSegmentNumber, dstAddr.getPort(),
287                             srcAddr.getPort(), TCPPacket.TCPFlag.ACK);
288                     completeTCPWrite(dstAddr, srcAddr, tcpBuf, byteBufAllocator, ctx);
289                     logTCP(true, bytes, sendSegmentNumber, receiveSegmentNumber, dstAddr, srcAddr, true);
290                 } else {
291                     TCPPacket.writePacket(tcpBuf, packet, receiveSegmentNumber, sendSegmentNumber, dstAddr.getPort(),
292                             srcAddr.getPort(), TCPPacket.TCPFlag.ACK);
293                     completeTCPWrite(dstAddr, srcAddr, tcpBuf, byteBufAllocator, ctx);
294                     logTCP(false, bytes, receiveSegmentNumber, sendSegmentNumber, dstAddr, srcAddr, false);
295 
296                     receiveSegmentNumber += bytes;
297 
298                     TCPPacket.writePacket(tcpBuf, null, sendSegmentNumber, receiveSegmentNumber, srcAddr.getPort(),
299                             dstAddr.getPort(), TCPPacket.TCPFlag.ACK);
300                     completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
301                     logTCP(false, bytes, sendSegmentNumber, receiveSegmentNumber, srcAddr, dstAddr, true);
302                 }
303             } finally {
304                 tcpBuf.release();
305             }
306         } else {
307             logger.debug("Discarding Pcap Write for TCP Object: {}", msg);
308         }
309     }

311     /**
312      * Write TCP/IP L3 and L2 here.
313      *
314      * @param srcAddr          {@link InetSocketAddress} Source Address of this Packet
315      * @param dstAddr          {@link InetSocketAddress} Destination Address of this Packet
316      * @param tcpBuf           {@link ByteBuf} containing TCP L4 Data
317      * @param byteBufAllocator {@link ByteBufAllocator} for allocating bytes for TCP/IP L3 and L2 data.
318      * @param ctx              {@link ChannelHandlerContext} for {@code fireExceptionCaught}
319      */
320     private void completeTCPWrite(InetSocketAddress srcAddr, InetSocketAddress dstAddr, ByteBuf tcpBuf,
321                                   ByteBufAllocator byteBufAllocator, ChannelHandlerContext ctx) {
322 
323         ByteBuf ipBuf = byteBufAllocator.buffer();
324         ByteBuf ethernetBuf = byteBufAllocator.buffer();
325         ByteBuf pcap = byteBufAllocator.buffer();
326 
327         try {
328             if (srcAddr.getAddress() instanceof Inet4Address && dstAddr.getAddress() instanceof Inet4Address) {
329                 IPPacket.writeTCPv4(ipBuf, tcpBuf,
330                         NetUtil.ipv4AddressToInt((Inet4Address) srcAddr.getAddress()),
331                         NetUtil.ipv4AddressToInt((Inet4Address) dstAddr.getAddress()));
332 
333                 EthernetPacket.writeIPv4(ethernetBuf, ipBuf);
334             } else if (srcAddr.getAddress() instanceof Inet6Address && dstAddr.getAddress() instanceof Inet6Address) {
335                 IPPacket.writeTCPv6(ipBuf, tcpBuf,
336                         srcAddr.getAddress().getAddress(),
337                         dstAddr.getAddress().getAddress());
338 
339                 EthernetPacket.writeIPv6(ethernetBuf, ipBuf);
340             } else {
341                 logger.error("Source and Destination IP Address versions are not same. Source Address: {}, " +
342                         "Destination Address: {}", srcAddr.getAddress(), dstAddr.getAddress());
343                 return;
344             }
345 
346             // Write Packet into Pcap
347             pCapWriter.writePacket(pcap, ethernetBuf);
348         } catch (IOException ex) {
349             logger.error("Caught Exception While Writing Packet into Pcap", ex);
350             ctx.fireExceptionCaught(ex);
351         } finally {
352             ipBuf.release();
353             ethernetBuf.release();
354             pcap.release();
355         }
356     }

358     /**
359      * Logger for TCP
360      */
361     private void logTCP(boolean isWriteOperation, int bytes, int sendSegmentNumber, int receiveSegmentNumber,
362                         InetSocketAddress srcAddr, InetSocketAddress dstAddr, boolean ackOnly) {
363         // If `ackOnly` is `true` when we don't need to write any data so we'll not
364         // log number of bytes being written and mark the operation as "TCP ACK".
365         if (logger.isDebugEnabled()) {
366             if (ackOnly) {
367                 logger.debug("Writing TCP ACK, isWriteOperation {}, Segment Number {}, Ack Number {}, Src Addr {}, "
368                         + "Dst Addr {}", isWriteOperation, sendSegmentNumber, receiveSegmentNumber, dstAddr, srcAddr);
369             } else {
370                 logger.debug("Writing TCP Data of {} Bytes, isWriteOperation {}, Segment Number {}, Ack Number {}, " +
371                                 "Src Addr {}, Dst Addr {}", bytes, isWriteOperation, sendSegmentNumber,
372                         receiveSegmentNumber, srcAddr, dstAddr);
373             }
374         }
375     }

377     /**
378      * Handle UDP l4
379      *
380      * @param ctx {@link ChannelHandlerContext} for {@code localAddress} / {@code remoteAddress},
381      *            {@link ByteBuf} allocation and {@code fireExceptionCaught}
382      * @param msg {@link DatagramPacket} or {@link DatagramChannel}
383      */
384     private void handleUDP(ChannelHandlerContext ctx, Object msg) {
385         ByteBuf udpBuf = ctx.alloc().buffer();
386 
387         try {
388             if (msg instanceof DatagramPacket) {
389 
390                 // If bytes are 0 and `captureZeroByte` is false, we won't capture this.
391                 if (((DatagramPacket) msg).content().readableBytes() == 0 && !captureZeroByte) {
392                     logger.debug("Discarding Zero Byte UDP Packet");
393                     return;
394                 }
395 
396                 DatagramPacket datagramPacket = ((DatagramPacket) msg).duplicate();
397                 InetSocketAddress srcAddr = datagramPacket.sender();
398                 InetSocketAddress dstAddr = datagramPacket.recipient();
399 
400                 // If `datagramPacket.sender()` is `null` then DatagramPacket is initialized
401                 // `sender` (local) address. In this case, we'll get source address from Channel.
402                 if (srcAddr == null) {
403                     srcAddr = (InetSocketAddress) ctx.channel().localAddress();
404                 }
405 
406                 logger.debug("Writing UDP Data of {} Bytes, Src Addr {}, Dst Addr {}",
407                         datagramPacket.content().readableBytes(), srcAddr, dstAddr);
408 
409                 UDPPacket.writePacket(udpBuf, datagramPacket.content(), srcAddr.getPort(), dstAddr.getPort());
410                 completeUDPWrite(srcAddr, dstAddr, udpBuf, ctx.alloc(), ctx);
411             } else if (msg instanceof ByteBuf && ((DatagramChannel) ctx.channel()).isConnected()) {
412 
413                 // If bytes are 0 and `captureZeroByte` is false, we won't capture this.
414                 if (((ByteBuf) msg).readableBytes() == 0 && !captureZeroByte) {
415                     logger.debug("Discarding Zero Byte UDP Packet");
416                     return;
417                 }
418 
419                 ByteBuf byteBuf = ((ByteBuf) msg).duplicate();
420 
421                 logger.debug("Writing UDP Data of {} Bytes, Src Addr {}, Dst Addr {}",
422                         byteBuf.readableBytes(), srcAddr, dstAddr);
423 
424                 UDPPacket.writePacket(udpBuf, byteBuf, srcAddr.getPort(), dstAddr.getPort());
425                 completeUDPWrite(srcAddr, dstAddr, udpBuf, ctx.alloc(), ctx);
426             } else {
427                 logger.debug("Discarding Pcap Write for UDP Object: {}", msg);
428             }
429         } finally {
430             udpBuf.release();
431         }
432     }

435      * Write UDP/IP L3 and L2 here.
436      *
437      * @param srcAddr          {@link InetSocketAddress} Source Address of this Packet

481     @Override
482     public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
483 
484         // If `isTCP` is true, then we'll simulate a `FIN` flow.
485         if (ctx.channel() instanceof SocketChannel) {
486             logger.debug("Starting Fake TCP FIN+ACK Flow to close connection");
487 
488             ByteBufAllocator byteBufAllocator = ctx.alloc();
489             ByteBuf tcpBuf = byteBufAllocator.buffer();
490 
491             try {
492                 // Write FIN+ACK with Normal Source and Destination Address
493                 TCPPacket.writePacket(tcpBuf, null, sendSegmentNumber, receiveSegmentNumber, srcAddr.getPort(),
494                         dstAddr.getPort(), TCPPacket.TCPFlag.FIN, TCPPacket.TCPFlag.ACK);
495                 completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
496 
497                 // Write FIN+ACK with Reversed Source and Destination Address
498                 TCPPacket.writePacket(tcpBuf, null, receiveSegmentNumber, sendSegmentNumber, dstAddr.getPort(),
499                         srcAddr.getPort(), TCPPacket.TCPFlag.FIN, TCPPacket.TCPFlag.ACK);
500                 completeTCPWrite(dstAddr, srcAddr, tcpBuf, byteBufAllocator, ctx);
501 
502                 // Write ACK with Normal Source and Destination Address
503                 TCPPacket.writePacket(tcpBuf, null, sendSegmentNumber + 1, receiveSegmentNumber + 1,
504                         srcAddr.getPort(), dstAddr.getPort(), TCPPacket.TCPFlag.ACK);
505                 completeTCPWrite(srcAddr, dstAddr, tcpBuf, byteBufAllocator, ctx);
506             } finally {
507                 tcpBuf.release();
508             }
509 
510             logger.debug("Finished Fake TCP FIN+ACK Flow to close connection");
511         }
512 
513         close();
514         super.handlerRemoved(ctx);
515     }
516 
517     @Override
518     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
519 
520         if (ctx.channel() instanceof SocketChannel) {
521             ByteBuf tcpBuf = ctx.alloc().buffer();
522 
523             try {
524                 // Write RST with Normal Source and Destination Address
525                 TCPPacket.writePacket(tcpBuf, null, sendSegmentNumber, receiveSegmentNumber, srcAddr.getPort(),
526                         dstAddr.getPort(), TCPPacket.TCPFlag.RST, TCPPacket.TCPFlag.ACK);
527                 completeTCPWrite(srcAddr, dstAddr, tcpBuf, ctx.alloc(), ctx);
528             } finally {
529                 tcpBuf.release();
530             }
531 
532             logger.debug("Sent Fake TCP RST to close connection");
533         }
534 
535         close();
536         ctx.fireExceptionCaught(cause);
537     }

539     /**
540      * <p> Close {@code PcapWriter} and {@link OutputStream}. </p>
541      * <p> Note: Calling this method does not close {@link PcapWriteHandler}.
542      * Only Pcap Writes are closed. </p>
543      *
544      * @throws IOException If {@link OutputStream#close()} throws an exception
545      */
546     @Override
547     public void close() throws IOException {
548         if (isClosed) {
549             logger.debug("PcapWriterHandler is already closed");
550         } else {
551             isClosed = true;
552             pCapWriter.close();
553             logger.debug("PcapWriterHandler is now closed");
554         }
555     }
