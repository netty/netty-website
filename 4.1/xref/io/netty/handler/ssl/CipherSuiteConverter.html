32  /**
33   * Converts a Java cipher suite string to an OpenSSL cipher suite string and vice versa.
34   *
35   * @see <a href="https://en.wikipedia.org/wiki/Cipher_suite">Wikipedia page about cipher suite</a>
36   */
37  final class CipherSuiteConverter {
38  
39      private static final InternalLogger logger = InternalLoggerFactory.getInstance(CipherSuiteConverter.class);
40  
41      /**
42       * A_B_WITH_C_D, where:
43       *
44       * A - TLS or SSL (protocol)
45       * B - handshake algorithm (key exchange and authentication algorithms to be precise)
46       * C - bulk cipher
47       * D - HMAC algorithm
48       *
49       * This regular expression assumes that:
50       *
51       * 1) A is always TLS or SSL, and
52       * 2) D is always a single word.
53       */
54      private static final Pattern JAVA_CIPHERSUITE_PATTERN =
55              Pattern.compile("^(?:TLS|SSL)_((?:(?!_WITH_).)+)_WITH_(.*)_(.*)$");
56  
57      /**
58       * A-B-C, where:
59       *
60       * A - handshake algorithm (key exchange and authentication algorithms to be precise)
61       * B - bulk cipher
62       * C - HMAC algorithm
63       *
64       * This regular expression assumes that:
65       *
66       * 1) A has some deterministic pattern as shown below, and
67       * 2) C is always a single word
68       */
69      private static final Pattern OPENSSL_CIPHERSUITE_PATTERN =
70              // Be very careful not to break the indentation while editing.
71              Pattern.compile(
72                      "^(?:(" + // BEGIN handshake algorithm
73                          "(?:(?:EXP-)?" +
74                              "(?:" +
75                                  "(?:DHE|EDH|ECDH|ECDHE|SRP|RSA)-(?:DSS|RSA|ECDSA|PSK)|" +
76                                  "(?:ADH|AECDH|KRB5|PSK|SRP)" +
77                              ')' +
78                          ")|" +
79                          "EXP" +
80                      ")-)?" +  // END handshake algorithm
81                      "(.*)-(.*)$");
82  
83      private static final Pattern JAVA_AES_CBC_PATTERN = Pattern.compile("^(AES)_([0-9]+)_CBC$");
84      private static final Pattern JAVA_AES_PATTERN = Pattern.compile("^(AES)_([0-9]+)_(.*)$");
85      private static final Pattern OPENSSL_AES_CBC_PATTERN = Pattern.compile("^(AES)([0-9]+)$");
86      private static final Pattern OPENSSL_AES_PATTERN = Pattern.compile("^(AES)([0-9]+)-(.*)$");

88      /**
89       * Java-to-OpenSSL cipher suite conversion map
90       * Note that the Java cipher suite has the protocol prefix (TLS_, SSL_)
91       */
92      private static final ConcurrentMap<String, String> j2o = PlatformDependent.newConcurrentHashMap();
93  
94      /**
95       * OpenSSL-to-Java cipher suite conversion map.
96       * Note that one OpenSSL cipher suite can be converted to more than one Java cipher suites because
97       * a Java cipher suite has the protocol name prefix (TLS_, SSL_)
98       */
99      private static final ConcurrentMap<String, Map<String, String>> o2j = PlatformDependent.newConcurrentHashMap();
100 
101     private static final Map<String, String> j2oTls13;
102     private static final Map<String, Map<String, String>> o2jTls13;
103 
104     static {
105         Map<String, String> j2oTls13Map = new HashMap<String, String>();
106         j2oTls13Map.put("TLS_AES_128_GCM_SHA256", "AEAD-AES128-GCM-SHA256");
107         j2oTls13Map.put("TLS_AES_256_GCM_SHA384", "AEAD-AES256-GCM-SHA384");
108         j2oTls13Map.put("TLS_CHACHA20_POLY1305_SHA256", "AEAD-CHACHA20-POLY1305-SHA256");
109         j2oTls13 = Collections.unmodifiableMap(j2oTls13Map);
110 
111         Map<String, Map<String, String>> o2jTls13Map = new HashMap<String, Map<String, String>>();
112         o2jTls13Map.put("TLS_AES_128_GCM_SHA256", singletonMap("TLS", "TLS_AES_128_GCM_SHA256"));
113         o2jTls13Map.put("TLS_AES_256_GCM_SHA384", singletonMap("TLS", "TLS_AES_256_GCM_SHA384"));
114         o2jTls13Map.put("TLS_CHACHA20_POLY1305_SHA256", singletonMap("TLS", "TLS_CHACHA20_POLY1305_SHA256"));
115         o2jTls13Map.put("AEAD-AES128-GCM-SHA256", singletonMap("TLS", "TLS_AES_128_GCM_SHA256"));
116         o2jTls13Map.put("AEAD-AES256-GCM-SHA384", singletonMap("TLS", "TLS_AES_256_GCM_SHA384"));
117         o2jTls13Map.put("AEAD-CHACHA20-POLY1305-SHA256", singletonMap("TLS", "TLS_CHACHA20_POLY1305_SHA256"));
118         o2jTls13 = Collections.unmodifiableMap(o2jTls13Map);
119     }

138      */
139     static boolean isO2JCached(String key, String protocol, String value) {
140         Map<String, String> p2j = o2j.get(key);
141         if (p2j == null) {
142             return false;
143         } else {
144             return value.equals(p2j.get(protocol));
145         }

148     /**
149      * Converts the specified Java cipher suite to its corresponding OpenSSL cipher suite name.
150      *
151      * @return {@code null} if the conversion has failed
152      */
153     static String toOpenSsl(String javaCipherSuite, boolean boringSSL) {
154         String converted = j2o.get(javaCipherSuite);
155         if (converted != null) {
156             return converted;
157         }
158         return cacheFromJava(javaCipherSuite, boringSSL);
159     }
160 
161     private static String cacheFromJava(String javaCipherSuite, boolean boringSSL) {
162         String converted = j2oTls13.get(javaCipherSuite);
163         if (converted != null) {
164             return boringSSL ? converted : javaCipherSuite;
165         }
166 
167         String openSslCipherSuite = toOpenSslUncached(javaCipherSuite, boringSSL);
168         if (openSslCipherSuite == null) {
169             return null;
170         }
171 
172         // Cache the mapping.
173         j2o.putIfAbsent(javaCipherSuite, openSslCipherSuite);
174 
175         // Cache the reverse mapping after stripping the protocol prefix (TLS_ or SSL_)
176         final String javaCipherSuiteSuffix = javaCipherSuite.substring(4);
177         Map<String, String> p2j = new HashMap<String, String>(4);
178         p2j.put("", javaCipherSuiteSuffix);
179         p2j.put("SSL", "SSL_" + javaCipherSuiteSuffix);
180         p2j.put("TLS", "TLS_" + javaCipherSuiteSuffix);
181         o2j.put(openSslCipherSuite, p2j);
182 
183         logger.debug("Cipher suite mapping: {} => {}", javaCipherSuite, openSslCipherSuite);
184 
185         return openSslCipherSuite;
186     }

188     static String toOpenSslUncached(String javaCipherSuite, boolean boringSSL) {
189         String converted = j2oTls13.get(javaCipherSuite);
190         if (converted != null) {
191             return boringSSL ? converted : javaCipherSuite;
192         }
193 
194         Matcher m = JAVA_CIPHERSUITE_PATTERN.matcher(javaCipherSuite);
195         if (!m.matches()) {
196             return null;
197         }
198 
199         String handshakeAlgo = toOpenSslHandshakeAlgo(m.group(1));
200         String bulkCipher = toOpenSslBulkCipher(m.group(2));
201         String hmacAlgo = toOpenSslHmacAlgo(m.group(3));
202         if (handshakeAlgo.isEmpty()) {
203             return bulkCipher + '-' + hmacAlgo;
204         } else if (bulkCipher.contains("CHACHA20")) {
205             return handshakeAlgo + '-' + bulkCipher;
206         } else {
207             return handshakeAlgo + '-' + bulkCipher + '-' + hmacAlgo;
208         }
209     }

211     private static String toOpenSslHandshakeAlgo(String handshakeAlgo) {
212         final boolean export = handshakeAlgo.endsWith("_EXPORT");
213         if (export) {
214             handshakeAlgo = handshakeAlgo.substring(0, handshakeAlgo.length() - 7);
215         }
216 
217         if ("RSA".equals(handshakeAlgo)) {
218             handshakeAlgo = "";
219         } else if (handshakeAlgo.endsWith("_anon")) {
220             handshakeAlgo = 'A' + handshakeAlgo.substring(0, handshakeAlgo.length() - 5);
221         }
222 
223         if (export) {
224             if (handshakeAlgo.isEmpty()) {
225                 handshakeAlgo = "EXP";
226             } else {
227                 handshakeAlgo = "EXP-" + handshakeAlgo;
228             }
229         }
230 
231         return handshakeAlgo.replace('_', '-');
232     }
233 
234     private static String toOpenSslBulkCipher(String bulkCipher) {
235         if (bulkCipher.startsWith("AES_")) {
236             Matcher m = JAVA_AES_CBC_PATTERN.matcher(bulkCipher);
237             if (m.matches()) {
238                 return m.replaceFirst("$1$2");
239             }
240 
241             m = JAVA_AES_PATTERN.matcher(bulkCipher);
242             if (m.matches()) {
243                 return m.replaceFirst("$1$2-$3");
244             }
245         }
246 
247         if ("3DES_EDE_CBC".equals(bulkCipher)) {
248             return "DES-CBC3";
249         }
250 
251         if ("RC4_128".equals(bulkCipher) || "RC4_40".equals(bulkCipher)) {
252             return "RC4";
253         }
254 
255         if ("DES40_CBC".equals(bulkCipher) || "DES_CBC_40".equals(bulkCipher)) {
256             return "DES-CBC";
257         }
258 
259         if ("RC2_CBC_40".equals(bulkCipher)) {
260             return "RC2-CBC";
261         }
262 
263         return bulkCipher.replace('_', '-');
264     }

266     private static String toOpenSslHmacAlgo(String hmacAlgo) {
267         // Java and OpenSSL use the same algorithm names for:
268         //
269         //   * SHA
270         //   * SHA256
271         //   * MD5
272         //
273         return hmacAlgo;
274     }

276     /**
277      * Convert from OpenSSL cipher suite name convention to java cipher suite name convention.
278      * @param openSslCipherSuite An OpenSSL cipher suite name.
279      * @param protocol The cryptographic protocol (i.e. SSL, TLS, ...).
280      * @return The translated cipher suite name according to java conventions. This will not be {@code null}.
281      */
282     static String toJava(String openSslCipherSuite, String protocol) {
283         Map<String, String> p2j = o2j.get(openSslCipherSuite);
284         if (p2j == null) {
285             p2j = cacheFromOpenSsl(openSslCipherSuite);
286             // This may happen if this method is queried when OpenSSL doesn't yet have a cipher setup. It will return
287             // "(NONE)" in this case.
288             if (p2j == null) {
289                 return null;
290             }
291         }
292 
293         String javaCipherSuite = p2j.get(protocol);
294         if (javaCipherSuite == null) {
295             String cipher = p2j.get("");
296             if (cipher == null) {
297                 return null;
298             }
299             javaCipherSuite = protocol + '_' + cipher;
300         }
301 
302         return javaCipherSuite;
303     }

305     private static Map<String, String> cacheFromOpenSsl(String openSslCipherSuite) {
306         Map<String, String> converted = o2jTls13.get(openSslCipherSuite);
307         if (converted != null) {
308             return converted;
309         }
310 
311         String javaCipherSuiteSuffix = toJavaUncached0(openSslCipherSuite, false);
312         if (javaCipherSuiteSuffix == null) {
313             return null;
314         }
315 
316         final String javaCipherSuiteSsl = "SSL_" + javaCipherSuiteSuffix;
317         final String javaCipherSuiteTls = "TLS_" + javaCipherSuiteSuffix;
318 
319         // Cache the mapping.
320         final Map<String, String> p2j = new HashMap<String, String>(4);
321         p2j.put("", javaCipherSuiteSuffix);
322         p2j.put("SSL", javaCipherSuiteSsl);
323         p2j.put("TLS", javaCipherSuiteTls);
324         o2j.putIfAbsent(openSslCipherSuite, p2j);
325 
326         // Cache the reverse mapping after adding the protocol prefix (TLS_ or SSL_)
327         j2o.putIfAbsent(javaCipherSuiteTls, openSslCipherSuite);
328         j2o.putIfAbsent(javaCipherSuiteSsl, openSslCipherSuite);
329 
330         logger.debug("Cipher suite mapping: {} => {}", javaCipherSuiteTls, openSslCipherSuite);
331         logger.debug("Cipher suite mapping: {} => {}", javaCipherSuiteSsl, openSslCipherSuite);
332 
333         return p2j;
334     }

336     static String toJavaUncached(String openSslCipherSuite) {
337         return toJavaUncached0(openSslCipherSuite, true);
338     }

340     private static String toJavaUncached0(String openSslCipherSuite, boolean checkTls13) {
341         if (checkTls13) {
342             Map<String, String> converted = o2jTls13.get(openSslCipherSuite);
343             if (converted != null) {
344                 return converted.get("TLS");
345             }
346         }
347 
348         Matcher m = OPENSSL_CIPHERSUITE_PATTERN.matcher(openSslCipherSuite);
349         if (!m.matches()) {
350             return null;
351         }
352 
353         String handshakeAlgo = m.group(1);
354         final boolean export;
355         if (handshakeAlgo == null) {
356             handshakeAlgo = "";
357             export = false;
358         } else if (handshakeAlgo.startsWith("EXP-")) {
359             handshakeAlgo = handshakeAlgo.substring(4);
360             export = true;
361         } else if ("EXP".equals(handshakeAlgo)) {
362             handshakeAlgo = "";
363             export = true;
364         } else {
365             export = false;
366         }
367 
368         handshakeAlgo = toJavaHandshakeAlgo(handshakeAlgo, export);
369         String bulkCipher = toJavaBulkCipher(m.group(2), export);
370         String hmacAlgo = toJavaHmacAlgo(m.group(3));
371 
372         String javaCipherSuite = handshakeAlgo + "_WITH_" + bulkCipher + '_' + hmacAlgo;
373         // For historical reasons the CHACHA20 ciphers do not follow OpenSSL's custom naming convention and omits the
374         // HMAC algorithm portion of the name. There is currently no way to derive this information because it is
375         // omitted from the OpenSSL cipher name, but they currently all use SHA256 for HMAC [1].
376         // [1] https://www.openssl.org/docs/man1.1.0/apps/ciphers.html
377         return bulkCipher.contains("CHACHA20") ? javaCipherSuite + "_SHA256" : javaCipherSuite;
378     }

380     private static String toJavaHandshakeAlgo(String handshakeAlgo, boolean export) {
381         if (handshakeAlgo.isEmpty()) {
382             handshakeAlgo = "RSA";
383         } else if ("ADH".equals(handshakeAlgo)) {
384             handshakeAlgo = "DH_anon";
385         } else if ("AECDH".equals(handshakeAlgo)) {
386             handshakeAlgo = "ECDH_anon";
387         }
388 
389         handshakeAlgo = handshakeAlgo.replace('-', '_');
390         if (export) {
391             return handshakeAlgo + "_EXPORT";
392         } else {
393             return handshakeAlgo;
394         }
395     }

397     private static String toJavaBulkCipher(String bulkCipher, boolean export) {
398         if (bulkCipher.startsWith("AES")) {
399             Matcher m = OPENSSL_AES_CBC_PATTERN.matcher(bulkCipher);
400             if (m.matches()) {
401                 return m.replaceFirst("$1_$2_CBC");
402             }
403 
404             m = OPENSSL_AES_PATTERN.matcher(bulkCipher);
405             if (m.matches()) {
406                 return m.replaceFirst("$1_$2_$3");
407             }
408         }
409 
410         if ("DES-CBC3".equals(bulkCipher)) {
411             return "3DES_EDE_CBC";
412         }
413 
414         if ("RC4".equals(bulkCipher)) {
415             if (export) {
416                 return "RC4_40";
417             } else {
418                 return "RC4_128";
419             }
420         }
421 
422         if ("DES-CBC".equals(bulkCipher)) {
423             if (export) {
424                 return "DES_CBC_40";
425             } else {
426                 return "DES_CBC";
427             }
428         }
429 
430         if ("RC2-CBC".equals(bulkCipher)) {
431             if (export) {
432                 return "RC2_CBC_40";
433             } else {
434                 return "RC2_CBC";
435             }
436         }
437 
438         return bulkCipher.replace('-', '_');
439     }

443         //
444         //   * SHA
445         //   * SHA256
446         //   * MD5
447         //
448         return hmacAlgo;

451     /**
452      * Convert the given ciphers if needed to OpenSSL format and append them to the correct {@link StringBuilder}
453      * depending on if its a TLSv1.3 cipher or not. If this methods returns without throwing an exception its
454      * guaranteed that at least one of the {@link StringBuilder}s contain some ciphers that can be used to configure
455      * OpenSSL.
456      */
457     static void convertToCipherStrings(Iterable<String> cipherSuites, StringBuilder cipherBuilder,
458                                        StringBuilder cipherTLSv13Builder, boolean boringSSL) {
459         for (String c: cipherSuites) {
460             if (c == null) {
461                 break;
462             }
463 
464             String converted = toOpenSsl(c, boringSSL);
465             if (converted == null) {
466                 converted = c;
467             }
468 
469             if (!OpenSsl.isCipherSuiteAvailable(converted)) {
470                 throw new IllegalArgumentException("unsupported cipher suite: " + c + '(' + converted + ')');
471             }
472 
473             if (SslUtils.isTLSv13Cipher(converted) || SslUtils.isTLSv13Cipher(c)) {
474                 cipherTLSv13Builder.append(converted);
475                 cipherTLSv13Builder.append(':');
476             } else {
477                 cipherBuilder.append(converted);
478                 cipherBuilder.append(':');
479             }
480         }
481 
482         if (cipherBuilder.length() == 0 && cipherTLSv13Builder.length() == 0) {
483             throw new IllegalArgumentException("empty cipher suites");
484         }
485         if (cipherBuilder.length() > 0) {
486             cipherBuilder.setLength(cipherBuilder.length() - 1);
487         }
488         if (cipherTLSv13Builder.length() > 0) {
489             cipherTLSv13Builder.setLength(cipherTLSv13Builder.length() - 1);
490         }
491     }
