16  package io.netty.handler.ssl;
17  
18  import io.netty.channel.Channel;
19  import io.netty.channel.ChannelHandlerContext;
20  import io.netty.channel.ChannelInboundHandlerAdapter;
21  import io.netty.channel.ChannelInitializer;
22  import io.netty.channel.ChannelPipeline;
23  import io.netty.handler.codec.DecoderException;
24  import io.netty.util.internal.ObjectUtil;
25  import io.netty.util.internal.logging.InternalLogger;
26  import io.netty.util.internal.logging.InternalLoggerFactory;
27  
28  import javax.net.ssl.SSLException;
29  
30  /**
31   * Configures a {@link ChannelPipeline} depending on the application-level protocol negotiation result of
32   * {@link SslHandler}.  For example, you could configure your HTTP pipeline depending on the result of ALPN:
33   * <pre>
34   * public class MyInitializer extends {@link ChannelInitializer}&lt;{@link Channel}&gt; {
35   *     private final {@link SslContext} sslCtx;
36   *
37   *     public MyInitializer({@link SslContext} sslCtx) {
38   *         this.sslCtx = sslCtx;
39   *     }
40   *
41   *     protected void initChannel({@link Channel} ch) {
42   *         {@link ChannelPipeline} p = ch.pipeline();
43   *         p.addLast(sslCtx.newHandler(...)); // Adds {@link SslHandler}
44   *         p.addLast(new MyNegotiationHandler());
45   *     }
46   * }
47   *
48   * public class MyNegotiationHandler extends {@link ApplicationProtocolNegotiationHandler} {
49   *     public MyNegotiationHandler() {
50   *         super({@link ApplicationProtocolNames}.HTTP_1_1);
51   *     }
52   *
53   *     protected void configurePipeline({@link ChannelHandlerContext} ctx, String protocol) {
54   *         if ({@link ApplicationProtocolNames}.HTTP_2.equals(protocol) {
55   *             configureHttp2(ctx);
56   *         } else if ({@link ApplicationProtocolNames}.HTTP_1_1.equals(protocol)) {
57   *             configureHttp1(ctx);
58   *         } else {
59   *             throw new IllegalStateException("unknown protocol: " + protocol);
60   *         }
61   *     }
62   * }
63   * </pre>
64   */
65  public abstract class ApplicationProtocolNegotiationHandler extends ChannelInboundHandlerAdapter {
66  
67      private static final InternalLogger logger =
68              InternalLoggerFactory.getInstance(ApplicationProtocolNegotiationHandler.class);
69  
70      private final String fallbackProtocol;
71  
72      /**
73       * Creates a new instance with the specified fallback protocol name.
74       *
75       * @param fallbackProtocol the name of the protocol to use when
76       *                         ALPN/NPN negotiation fails or the client does not support ALPN/NPN
77       */
78      protected ApplicationProtocolNegotiationHandler(String fallbackProtocol) {
79          this.fallbackProtocol = ObjectUtil.checkNotNull(fallbackProtocol, "fallbackProtocol");
80      }
81  
82      @Override
83      public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
84          if (evt instanceof SslHandshakeCompletionEvent) {
85              SslHandshakeCompletionEvent handshakeEvent = (SslHandshakeCompletionEvent) evt;
86              try {
87                  if (handshakeEvent.isSuccess()) {
88                      SslHandler sslHandler = ctx.pipeline().get(SslHandler.class);
89                      if (sslHandler == null) {
90                          throw new IllegalStateException("cannot find an SslHandler in the pipeline (required for "
91                                  + "application-level protocol negotiation)");
92                      }
93                      String protocol = sslHandler.applicationProtocol();
94                      configurePipeline(ctx, protocol != null ? protocol : fallbackProtocol);
95                  } else {
96                      // if the event is not produced because of an successful handshake we will receive the same
97                      // exception in exceptionCaught(...) and handle it there. This will allow us more fine-grained
98                      // control over which exception we propagate down the ChannelPipeline.
99                      //
100                     // See https://github.com/netty/netty/issues/10342
101                 }
102             } catch (Throwable cause) {
103                 exceptionCaught(ctx, cause);
104             } finally {
105                 // Handshake failures are handled in exceptionCaught(...).
106                 if (handshakeEvent.isSuccess()) {
107                     removeSelfIfPresent(ctx);
108                 }
109             }
110         }
111 
112         ctx.fireUserEventTriggered(evt);
113     }
114 
115     private void removeSelfIfPresent(ChannelHandlerContext ctx) {
116         ChannelPipeline pipeline = ctx.pipeline();
117         if (pipeline.context(this) != null) {
118             pipeline.remove(this);
119         }
120     }
121     /**
122      * Invoked on successful initial SSL/TLS handshake. Implement this method to configure your pipeline
123      * for the negotiated application-level protocol.
124      *
125      * @param protocol the name of the negotiated application-level protocol, or
126      *                 the fallback protocol name specified in the constructor call if negotiation failed or the client
127      *                 isn't aware of ALPN/NPN extension
128      */
129     protected abstract void configurePipeline(ChannelHandlerContext ctx, String protocol) throws Exception;
130 
131     /**
132      * Invoked on failed initial SSL/TLS handshake.
133      */
134     protected void handshakeFailure(ChannelHandlerContext ctx, Throwable cause) throws Exception {
135         logger.warn("{} TLS handshake failed:", ctx.channel(), cause);
136         ctx.close();
137     }
138 
139     @Override
140     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
141         Throwable wrapped;
142         if (cause instanceof DecoderException && ((wrapped = cause.getCause()) instanceof SSLException)) {
143             try {
144                 handshakeFailure(ctx, wrapped);
145                 return;
146             } finally {
147                 removeSelfIfPresent(ctx);
148             }
149         }
150         logger.warn("{} Failed to select the application-level protocol:", ctx.channel(), cause);
151         ctx.fireExceptionCaught(cause);
152         ctx.close();
153     }
154 }
