46  final class BouncyCastlePemReader {
47      private static final String BC_PROVIDER = "org.bouncycastle.jce.provider.BouncyCastleProvider";
48      private static final String BC_PEMPARSER = "org.bouncycastle.openssl.PEMParser";
49      private static final InternalLogger logger = InternalLoggerFactory.getInstance(BouncyCastlePemReader.class);
50  
51      private static volatile Throwable unavailabilityCause;
52      private static volatile Provider bcProvider;
53      private static volatile boolean attemptedLoading;
54  
55      public static boolean hasAttemptedLoading() {
56          return attemptedLoading;
57      }
58  
59      public static boolean isAvailable() {
60          if (!hasAttemptedLoading()) {
61              tryLoading();
62          }
63          return unavailabilityCause == null;
64      }
65  
66      /**
67       * @return the cause if unavailable. {@code null} if available.
68       */
69      public static Throwable unavailabilityCause() {
70          return unavailabilityCause;
71      }
72  
73      private static void tryLoading() {
74          AccessController.doPrivileged(new PrivilegedAction<Void>() {
75              @Override
76              public Void run() {
77                  try {
78                      ClassLoader classLoader = getClass().getClassLoader();
79                      // Check for bcprov-jdk15on:
80                      Class<Provider> bcProviderClass =
81                              (Class<Provider>) Class.forName(BC_PROVIDER, true, classLoader);
82                      // Check for bcpkix-jdk15on:
83                      Class.forName(BC_PEMPARSER, true, classLoader);
84                      bcProvider = bcProviderClass.getConstructor().newInstance();
85                      logger.debug("Bouncy Castle provider available");
86                      attemptedLoading = true;
87                  } catch (Throwable e) {
88                      logger.debug("Cannot load Bouncy Castle provider", e);
89                      unavailabilityCause = e;
90                      attemptedLoading = true;
91                  }
92                  return null;
93              }
94          });
95      }
96  
97      /**
98       * Generates a new {@link PrivateKey}.
99       *
100      * @param keyInputStream an input stream for a PKCS#1 or PKCS#8 private key in PEM format.
101      * @param keyPassword the password of the {@code keyFile}.
102      *                    {@code null} if it's not password-protected.
103      * @return generated {@link PrivateKey}.
104      */
105     public static PrivateKey getPrivateKey(InputStream keyInputStream, String keyPassword) {
106         if (!isAvailable()) {
107             if (logger.isDebugEnabled()) {
108                 logger.debug("Bouncy castle provider is unavailable.", unavailabilityCause());
109             }
110             return null;
111         }
112         try {
113             PEMParser parser = newParser(keyInputStream);
114             return getPrivateKey(parser, keyPassword);
115         } catch (Exception e) {
116             logger.debug("Unable to extract private key", e);
117             return null;
118         }
119     }
120 
121     /**
122      * Generates a new {@link PrivateKey}.
123      *
124      * @param keyFile a PKCS#1 or PKCS#8 private key file in PEM format.
125      * @param keyPassword the password of the {@code keyFile}.
126      *                    {@code null} if it's not password-protected.
127      * @return generated {@link PrivateKey}.
128      */
129     public static PrivateKey getPrivateKey(File keyFile, String keyPassword) {
130         if (!isAvailable()) {
131             if (logger.isDebugEnabled()) {
132                 logger.debug("Bouncy castle provider is unavailable.", unavailabilityCause());
133             }
134             return null;
135         }
136         try {
137             PEMParser parser = newParser(keyFile);
138             return getPrivateKey(parser, keyPassword);
139         } catch (Exception e) {
140             logger.debug("Unable to extract private key", e);
141             return null;
142         }
143     }
144 
145     private static JcaPEMKeyConverter newConverter() {
146         return new JcaPEMKeyConverter().setProvider(bcProvider);
147     }
148 
149     private static PrivateKey getPrivateKey(PEMParser pemParser, String keyPassword) throws IOException,
150             PKCSException, OperatorCreationException {
151         try {
152             Object object = pemParser.readObject();
153             if (logger.isDebugEnabled()) {
154                 logger.debug("Parsed PEM object of type {} and assume " +
155                         "key is {}encrypted", object.getClass().getName(), keyPassword == null ? "not " : "");
156             }
157             JcaPEMKeyConverter converter = newConverter();
158             PrivateKey pk = null;
159 
160             if (keyPassword == null) {
161                 // assume private key is not encrypted
162                 if (object instanceof PrivateKeyInfo) {
163                     pk = converter.getPrivateKey((PrivateKeyInfo) object);
164                 } else if (object instanceof PEMKeyPair) {
165                     pk = converter.getKeyPair((PEMKeyPair) object).getPrivate();
166                 } else {
167                     logger.debug("Unable to handle PEM object of type {} as a non encrypted key", object.getClass());
168                 }
169             } else {
170                 // assume private key is encrypted
171                 if (object instanceof PEMEncryptedKeyPair) {
172                     PEMDecryptorProvider decProv = new JcePEMDecryptorProviderBuilder()
173                             .setProvider(bcProvider)
174                             .build(keyPassword.toCharArray());
175                     pk = converter.getKeyPair(((PEMEncryptedKeyPair) object).decryptKeyPair(decProv)).getPrivate();
176                 } else if (object instanceof PKCS8EncryptedPrivateKeyInfo) {
177                     InputDecryptorProvider pkcs8InputDecryptorProvider = new JceOpenSSLPKCS8DecryptorProviderBuilder()
178                             .setProvider(bcProvider)
179                             .build(keyPassword.toCharArray());
180                     pk = converter.getPrivateKey(((PKCS8EncryptedPrivateKeyInfo) object)
181                             .decryptPrivateKeyInfo(pkcs8InputDecryptorProvider));
182                 } else {
183                     logger.debug("Unable to handle PEM object of type {} as a encrypted key", object.getClass());
184                 }
185             }
186 
187             if (pk == null) {
188                 if (logger.isDebugEnabled()) {
189                     logger.debug("No key found");
190                 }
191             }
192 
193             return pk;
194         } finally {
195             if (pemParser != null) {
196                 try {
197                     pemParser.close();
198                 } catch (Exception exception) {
199                     logger.debug("Failed closing pem parser", exception);
200                 }
201             }
202         }
203     }
204 
205     private static PEMParser newParser(File keyFile) throws FileNotFoundException {
206         return new PEMParser(new FileReader(keyFile));
207     }
208 
209     private static PEMParser newParser(InputStream keyInputStream) {
210         return new PEMParser(new InputStreamReader(keyInputStream, CharsetUtil.US_ASCII));
211     }
212 
213     private BouncyCastlePemReader() { }
214 }
