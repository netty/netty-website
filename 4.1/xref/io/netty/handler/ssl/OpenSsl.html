108     static {
109         Throwable cause = null;
110 
111         if (SystemPropertyUtil.getBoolean("io.netty.handler.ssl.noOpenSsl", false)) {
112             cause = new UnsupportedOperationException(
113                     "OpenSSL was explicit disabled with -Dio.netty.handler.ssl.noOpenSsl=true");
114 
115             logger.debug(
116                     "netty-tcnative explicit disabled; " +
117                             OpenSslEngine.class.getSimpleName() + " will be unavailable.", cause);
118         } else {
119             // Test if netty-tcnative is in the classpath first.
120             try {
121                 Class.forName("io.netty.internal.tcnative.SSL", false, OpenSsl.class.getClassLoader());
122             } catch (ClassNotFoundException t) {
123                 cause = t;
124                 logger.debug(
125                         "netty-tcnative not in the classpath; " +
126                                 OpenSslEngine.class.getSimpleName() + " will be unavailable.");
127             }
128 
129             // If in the classpath, try to load the native library and initialize netty-tcnative.
130             if (cause == null) {
131                 try {
132                     // The JNI library was not already loaded. Load it now.
133                     loadTcNative();
134                 } catch (Throwable t) {
135                     cause = t;
136                     logger.debug(
137                             "Failed to load netty-tcnative; " +
138                                     OpenSslEngine.class.getSimpleName() + " will be unavailable, unless the " +
139                                     "application has already loaded the symbols by some other means. " +
140                                     "See https://netty.io/wiki/forked-tomcat-native.html for more information.", t);
141                 }
142 
143                 try {
144                     String engine = SystemPropertyUtil.get("io.netty.handler.ssl.openssl.engine", null);
145                     if (engine == null) {
146                         logger.debug("Initialize netty-tcnative using engine: 'default'");
147                     } else {
148                         logger.debug("Initialize netty-tcnative using engine: '{}'", engine);
149                     }
150                     initializeTcNative(engine);
151 
152                     // The library was initialized successfully. If loading the library failed above,
153                     // reset the cause now since it appears that the library was loaded by some other
154                     // means.
155                     cause = null;
156                 } catch (Throwable t) {
157                     if (cause == null) {
158                         cause = t;
159                     }
160                     logger.debug(
161                             "Failed to initialize netty-tcnative; " +
162                                     OpenSslEngine.class.getSimpleName() + " will be unavailable. " +
163                                     "See https://netty.io/wiki/forked-tomcat-native.html for more information.", t);
164                 }
165             }
166         }
167 
168         UNAVAILABILITY_CAUSE = cause;
169 
170         if (cause == null) {
171             logger.debug("netty-tcnative using native library: {}", SSL.versionString());
172 
173             final List<String> defaultCiphers = new ArrayList<String>();
174             final Set<String> availableOpenSslCipherSuites = new LinkedHashSet<String>(128);
175             boolean supportsKeyManagerFactory = false;
176             boolean useKeyManagerFactory = false;
177             boolean tlsv13Supported = false;
178 
179             IS_BORINGSSL = "BoringSSL".equals(versionString());
180 
181             try {
182                 final long sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_ALL, SSL.SSL_MODE_SERVER);
183                 long certBio = 0;
184                 long keyBio = 0;
185                 long cert = 0;
186                 long key = 0;
187                 try {
188                     try {
189                         StringBuilder tlsv13Ciphers = new StringBuilder();
190 
191                         for (String cipher: TLSV13_CIPHERS) {
192                             String converted = CipherSuiteConverter.toOpenSsl(cipher, IS_BORINGSSL);
193                             if (converted != null) {
194                                 tlsv13Ciphers.append(converted).append(':');
195                             }
196                         }
197                         if (tlsv13Ciphers.length() == 0) {
198                             tlsv13Supported = false;
199                         } else {
200                             tlsv13Ciphers.setLength(tlsv13Ciphers.length() - 1);
201                             SSLContext.setCipherSuite(sslCtx, tlsv13Ciphers.toString() , true);
202                             tlsv13Supported = true;
203                         }
204 
205                     } catch (Exception ignore) {
206                         tlsv13Supported = false;
207                     }
208 
209                     SSLContext.setCipherSuite(sslCtx, "ALL", false);
210 
211                     final long ssl = SSL.newSSL(sslCtx, true);
212                     try {
213                         for (String c: SSL.getCiphers(ssl)) {
214                             // Filter out bad input.
215                             if (c == null || c.isEmpty() || availableOpenSslCipherSuites.contains(c) ||
216                                 // Filter out TLSv1.3 ciphers if not supported.
217                                 !tlsv13Supported && isTLSv13Cipher(c)) {
218                                 continue;
219                             }
220                             availableOpenSslCipherSuites.add(c);
221                         }
222                         if (IS_BORINGSSL) {
223                             // Currently BoringSSL does not include these when calling SSL.getCiphers() even when these
224                             // are supported.
225                             Collections.addAll(availableOpenSslCipherSuites,
226                                                "TLS_AES_128_GCM_SHA256",
227                                                "TLS_AES_256_GCM_SHA384" ,
228                                                "TLS_CHACHA20_POLY1305_SHA256",
229                                                "AEAD-AES128-GCM-SHA256",
230                                                "AEAD-AES256-GCM-SHA384",
231                                                "AEAD-CHACHA20-POLY1305-SHA256");
232                         }
233 
234                         PemEncoded privateKey = PemPrivateKey.valueOf(KEY.getBytes(CharsetUtil.US_ASCII));
235                         try {
236                             // Let's check if we can set a callback, which may not work if the used OpenSSL version
237                             // is to old.
238                             SSLContext.setCertificateCallback(sslCtx, null);
239 
240                             X509Certificate certificate = selfSignedCertificate();
241                             certBio = ReferenceCountedOpenSslContext.toBIO(ByteBufAllocator.DEFAULT, certificate);
242                             cert = SSL.parseX509Chain(certBio);
243 
244                             keyBio = ReferenceCountedOpenSslContext.toBIO(
245                                     UnpooledByteBufAllocator.DEFAULT, privateKey.retain());
246                             key = SSL.parsePrivateKey(keyBio, null);
247 
248                             SSL.setKeyMaterial(ssl, cert, key);
249                             supportsKeyManagerFactory = true;
250                             try {
251                                 boolean propertySet = SystemPropertyUtil.contains(
252                                         "io.netty.handler.ssl.openssl.useKeyManagerFactory");
253                                 if (!IS_BORINGSSL) {
254                                     useKeyManagerFactory = SystemPropertyUtil.getBoolean(
255                                             "io.netty.handler.ssl.openssl.useKeyManagerFactory", true);
256 
257                                     if (propertySet) {
258                                         logger.info("System property " +
259                                                 "'io.netty.handler.ssl.openssl.useKeyManagerFactory'" +
260                                                 " is deprecated and so will be ignored in the future");
261                                     }
262                                 } else {
263                                     useKeyManagerFactory = true;
264                                     if (propertySet) {
265                                         logger.info("System property " +
266                                                 "'io.netty.handler.ssl.openssl.useKeyManagerFactory'" +
267                                                 " is deprecated and will be ignored when using BoringSSL");
268                                     }
269                                 }
270                             } catch (Throwable ignore) {
271                                 logger.debug("Failed to get useKeyManagerFactory system property.");
272                             }
273                         } catch (Error ignore) {
274                             logger.debug("KeyManagerFactory not supported.");
275                         } finally {
276                             privateKey.release();
277                         }
278                     } finally {
279                         SSL.freeSSL(ssl);
280                         if (certBio != 0) {
281                             SSL.freeBIO(certBio);
282                         }
283                         if (keyBio != 0) {
284                             SSL.freeBIO(keyBio);
285                         }
286                         if (cert != 0) {
287                             SSL.freeX509Chain(cert);
288                         }
289                         if (key != 0) {
290                             SSL.freePrivateKey(key);
291                         }
292                     }
293                 } finally {
294                     SSLContext.free(sslCtx);
295                 }
296             } catch (Exception e) {
297                 logger.warn("Failed to get the list of available OpenSSL cipher suites.", e);
298             }
299             AVAILABLE_OPENSSL_CIPHER_SUITES = Collections.unmodifiableSet(availableOpenSslCipherSuites);
300             final Set<String> availableJavaCipherSuites = new LinkedHashSet<String>(
301                     AVAILABLE_OPENSSL_CIPHER_SUITES.size() * 2);
302             for (String cipher: AVAILABLE_OPENSSL_CIPHER_SUITES) {
303                 // Included converted but also openssl cipher name
304                 if (!isTLSv13Cipher(cipher)) {
305                     availableJavaCipherSuites.add(CipherSuiteConverter.toJava(cipher, "TLS"));
306                     availableJavaCipherSuites.add(CipherSuiteConverter.toJava(cipher, "SSL"));
307                 } else {
308                     // TLSv1.3 ciphers have the correct format.
309                     availableJavaCipherSuites.add(cipher);
310                 }
311             }
312 
313             addIfSupported(availableJavaCipherSuites, defaultCiphers, DEFAULT_CIPHER_SUITES);
314             addIfSupported(availableJavaCipherSuites, defaultCiphers, TLSV13_CIPHER_SUITES);
315 
316             useFallbackCiphersIfDefaultIsEmpty(defaultCiphers, availableJavaCipherSuites);
317             DEFAULT_CIPHERS = Collections.unmodifiableList(defaultCiphers);
318 
319             AVAILABLE_JAVA_CIPHER_SUITES = Collections.unmodifiableSet(availableJavaCipherSuites);
320 
321             final Set<String> availableCipherSuites = new LinkedHashSet<String>(
322                     AVAILABLE_OPENSSL_CIPHER_SUITES.size() + AVAILABLE_JAVA_CIPHER_SUITES.size());
323             availableCipherSuites.addAll(AVAILABLE_OPENSSL_CIPHER_SUITES);
324             availableCipherSuites.addAll(AVAILABLE_JAVA_CIPHER_SUITES);
325 
326             AVAILABLE_CIPHER_SUITES = availableCipherSuites;
327             SUPPORTS_KEYMANAGER_FACTORY = supportsKeyManagerFactory;
328             USE_KEYMANAGER_FACTORY = useKeyManagerFactory;
329 
330             Set<String> protocols = new LinkedHashSet<String>(6);
331             // Seems like there is no way to explicitly disable SSLv2Hello in openssl so it is always enabled
332             protocols.add(PROTOCOL_SSL_V2_HELLO);
333             if (doesSupportProtocol(SSL.SSL_PROTOCOL_SSLV2, SSL.SSL_OP_NO_SSLv2)) {
334                 protocols.add(PROTOCOL_SSL_V2);
335             }
336             if (doesSupportProtocol(SSL.SSL_PROTOCOL_SSLV3, SSL.SSL_OP_NO_SSLv3)) {
337                 protocols.add(PROTOCOL_SSL_V3);
338             }
339             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1, SSL.SSL_OP_NO_TLSv1)) {
340                 protocols.add(PROTOCOL_TLS_V1);
341             }
342             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_1, SSL.SSL_OP_NO_TLSv1_1)) {
343                 protocols.add(PROTOCOL_TLS_V1_1);
344             }
345             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_OP_NO_TLSv1_2)) {
346                 protocols.add(PROTOCOL_TLS_V1_2);
347             }
348 
349             // This is only supported by java11 and later.
350             if (tlsv13Supported && doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_3, SSL.SSL_OP_NO_TLSv1_3)) {
351                 protocols.add(PROTOCOL_TLS_V1_3);
352                 TLSV13_SUPPORTED = true;
353             } else {
354                 TLSV13_SUPPORTED = false;
355             }
356 
357             SUPPORTED_PROTOCOLS_SET = Collections.unmodifiableSet(protocols);
358             SUPPORTS_OCSP = doesSupportOcsp();
359 
360             if (logger.isDebugEnabled()) {
361                 logger.debug("Supported protocols (OpenSSL): {} ", SUPPORTED_PROTOCOLS_SET);
362                 logger.debug("Default cipher suites (OpenSSL): {}", DEFAULT_CIPHERS);
363             }
364         } else {
365             DEFAULT_CIPHERS = Collections.emptyList();
366             AVAILABLE_OPENSSL_CIPHER_SUITES = Collections.emptySet();
367             AVAILABLE_JAVA_CIPHER_SUITES = Collections.emptySet();
368             AVAILABLE_CIPHER_SUITES = Collections.emptySet();
369             SUPPORTS_KEYMANAGER_FACTORY = false;
370             USE_KEYMANAGER_FACTORY = false;
371             SUPPORTED_PROTOCOLS_SET = Collections.emptySet();
372             SUPPORTS_OCSP = false;
373             TLSV13_SUPPORTED = false;
374             IS_BORINGSSL = false;
375         }
376     }

381     static X509Certificate selfSignedCertificate() throws CertificateException {
382         return (X509Certificate) SslContext.X509_CERT_FACTORY.generateCertificate(
383                 new ByteArrayInputStream(CERT.getBytes(CharsetUtil.US_ASCII))
384         );

387     private static boolean doesSupportOcsp() {
388         boolean supportsOcsp = false;
389         if (version() >= 0x10002000L) {
390             long sslCtx = -1;
391             try {
392                 sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_MODE_SERVER);
393                 SSLContext.enableOcsp(sslCtx, false);
394                 supportsOcsp = true;
395             } catch (Exception ignore) {
396                 // ignore
397             } finally {
398                 if (sslCtx != -1) {
399                     SSLContext.free(sslCtx);
400                 }
401             }
402         }
403         return supportsOcsp;
404     }
405     private static boolean doesSupportProtocol(int protocol, int opt) {
406         if (opt == 0) {
407             // If the opt is 0 the protocol is not supported. This is for example the case with BoringSSL and SSLv2.
408             return false;
409         }
410         long sslCtx = -1;
411         try {
412             sslCtx = SSLContext.make(protocol, SSL.SSL_MODE_COMBINED);
413             return true;
414         } catch (Exception ignore) {
415             return false;
416         } finally {
417             if (sslCtx != -1) {
418                 SSLContext.free(sslCtx);
419             }
420         }
421     }

554     private static void loadTcNative() throws Exception {
555         String os = PlatformDependent.normalizedOs();
556         String arch = PlatformDependent.normalizedArch();
557 
558         Set<String> libNames = new LinkedHashSet<String>(5);
559         String staticLibName = "netty_tcnative";
560 
561         // First, try loading the platform-specific library. Platform-specific
562         // libraries will be available if using a tcnative uber jar.
563         if ("linux".equalsIgnoreCase(os)) {
564             Set<String> classifiers = PlatformDependent.normalizedLinuxClassifiers();
565             for (String classifier : classifiers) {
566                 libNames.add(staticLibName + "_" + os + '_' + arch + "_" + classifier);
567             }
568             // generic arch-dependent library
569             libNames.add(staticLibName + "_" + os + '_' + arch);
570 
571             // Fedora SSL lib so naming (libssl.so.10 vs libssl.so.1.0.0).
572             // note: should already be included from the classifiers but if not, we use this as an
573             //       additional fallback option here
574             libNames.add(staticLibName + "_" + os + '_' + arch + "_fedora");
575         } else {
576             libNames.add(staticLibName + "_" + os + '_' + arch);
577         }
578         libNames.add(staticLibName + "_" + arch);
579         libNames.add(staticLibName);
580 
581         NativeLibraryLoader.loadFirstAvailable(SSL.class.getClassLoader(),
582             libNames.toArray(new String[0]));
583     }
