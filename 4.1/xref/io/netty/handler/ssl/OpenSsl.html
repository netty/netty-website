328     private static boolean doesSupportOcsp() {
329         boolean supportsOcsp = false;
330         if (version() >= 0x10002000L) {
331             long sslCtx = -1;
332             try {
333                 sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_MODE_SERVER);
334                 SSLContext.enableOcsp(sslCtx, false);
335                 supportsOcsp = true;
336             } catch (Exception ignore) {
337                 // ignore
338             } finally {
339                 if (sslCtx != -1) {
340                     SSLContext.free(sslCtx);
341                 }
342             }
343         }
344         return supportsOcsp;
345     }
346     private static boolean doesSupportProtocol(int protocol, int opt) {
347         if (opt == 0) {
348             // If the opt is 0 the protocol is not supported. This is for example the case with BoringSSL and SSLv2.
349             return false;
350         }
351         long sslCtx = -1;
352         try {
353             sslCtx = SSLContext.make(protocol, SSL.SSL_MODE_COMBINED);
354             return true;
355         } catch (Exception ignore) {
356             return false;
357         } finally {
358             if (sslCtx != -1) {
359                 SSLContext.free(sslCtx);
360             }
361         }

489     private static void loadTcNative() throws Exception {
490         String os = PlatformDependent.normalizedOs();
491         String arch = PlatformDependent.normalizedArch();
492 
493         Set<String> libNames = new LinkedHashSet<String>(4);
494         String staticLibName = "netty_tcnative";
495 
496         // First, try loading the platform-specific library. Platform-specific
497         // libraries will be available if using a tcnative uber jar.
498         libNames.add(staticLibName + "_" + os + '_' + arch);
499         if ("linux".equalsIgnoreCase(os)) {
500             // Fedora SSL lib so naming (libssl.so.10 vs libssl.so.1.0.0)..
501             libNames.add(staticLibName + "_" + os + '_' + arch + "_fedora");
502         }
503         libNames.add(staticLibName + "_" + arch);
504         libNames.add(staticLibName);
505 
506         NativeLibraryLoader.loadFirstAvailable(SSL.class.getClassLoader(),
507             libNames.toArray(new String[0]));
508     }
