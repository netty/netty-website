78  /**
79   * An implementation of {@link SslContext} which works with libraries that support the
80   * <a href="https://www.openssl.org/">OpenSsl</a> C library API.
81   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
82   *
83   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
84   * which depends upon the instance of this class is released. Otherwise if any method of
85   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
86   */
87  public abstract class ReferenceCountedOpenSslContext extends SslContext implements ReferenceCounted {

209     ReferenceCountedOpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
210                                    OpenSslApplicationProtocolNegotiator apn, int mode, Certificate[] keyCertChain,
211                                    ClientAuth clientAuth, String[] protocols, boolean startTls, boolean enableOcsp,
212                                    boolean leakDetection, Map.Entry<SslContextOption<?>, Object>... ctxOptions)
213             throws SSLException {
214         super(startTls);
215 
216         OpenSsl.ensureAvailability();
217 
218         if (enableOcsp && !OpenSsl.isOcspSupported()) {
219             throw new IllegalStateException("OCSP is not supported.");
220         }
221 
222         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
223             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
224         }
225 
226         boolean tlsFalseStart = false;
227         boolean useTasks = USE_TASKS;
228         OpenSslPrivateKeyMethod privateKeyMethod = null;
229         OpenSslAsyncPrivateKeyMethod asyncPrivateKeyMethod = null;
230         OpenSslCertificateCompressionConfig certCompressionConfig = null;
231         Integer maxCertificateList = null;
232 
233         if (ctxOptions != null) {
234             for (Map.Entry<SslContextOption<?>, Object> ctxOpt : ctxOptions) {
235                 SslContextOption<?> option = ctxOpt.getKey();
236 
237                 if (option == OpenSslContextOption.TLS_FALSE_START) {
238                     tlsFalseStart = (Boolean) ctxOpt.getValue();
239                 } else if (option == OpenSslContextOption.USE_TASKS) {
240                     useTasks = (Boolean) ctxOpt.getValue();
241                 } else if (option == OpenSslContextOption.PRIVATE_KEY_METHOD) {
242                     privateKeyMethod = (OpenSslPrivateKeyMethod) ctxOpt.getValue();
243                 } else if (option == OpenSslContextOption.ASYNC_PRIVATE_KEY_METHOD) {
244                     asyncPrivateKeyMethod = (OpenSslAsyncPrivateKeyMethod) ctxOpt.getValue();
245                 } else if (option == OpenSslContextOption.CERTIFICATE_COMPRESSION_ALGORITHMS) {
246                     certCompressionConfig = (OpenSslCertificateCompressionConfig) ctxOpt.getValue();
247                 } else if (option == OpenSslContextOption.MAX_CERTIFICATE_LIST_BYTES) {
248                     maxCertificateList = (Integer) ctxOpt.getValue();
249                 } else {
250                     logger.debug("Skipping unsupported " + SslContextOption.class.getSimpleName()
251                             + ": " + ctxOpt.getKey());
252                 }
253             }
254         }
255         if (privateKeyMethod != null && asyncPrivateKeyMethod != null) {
256             throw new IllegalArgumentException("You can either only use "
257                     + OpenSslAsyncPrivateKeyMethod.class.getSimpleName() + " or "
258                     + OpenSslPrivateKeyMethod.class.getSimpleName());
259         }
260 
261         this.tlsFalseStart = tlsFalseStart;
262 
263         leak = leakDetection ? leakDetector.track(this) : null;
264         this.mode = mode;
265         this.clientAuth = isServer() ? checkNotNull(clientAuth, "clientAuth") : ClientAuth.NONE;
266         this.protocols = protocols == null ? OpenSsl.defaultProtocols(mode == SSL.SSL_MODE_CLIENT) : protocols;
267         this.enableOcsp = enableOcsp;
268 
269         this.keyCertChain = keyCertChain == null ? null : keyCertChain.clone();
270 
271         String[] suites = checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
272                 ciphers, DEFAULT_CIPHERS, availableJavaCipherSuites());
273         // Filter out duplicates.
274         LinkedHashSet<String> suitesSet = new LinkedHashSet<String>(suites.length);
275         Collections.addAll(suitesSet, suites);
276         unmodifiableCiphers = new ArrayList<String>(suitesSet);
277 
278         this.apn = checkNotNull(apn, "apn");
279 
280         // Create a new SSL_CTX and configure it.
281         boolean success = false;
282         try {
283             boolean tlsv13Supported = OpenSsl.isTlsv13Supported();
284 
285             try {
286                 int protocolOpts = SSL.SSL_PROTOCOL_SSLV3 | SSL.SSL_PROTOCOL_TLSV1 |
287                         SSL.SSL_PROTOCOL_TLSV1_1 | SSL.SSL_PROTOCOL_TLSV1_2;
288                 if (tlsv13Supported) {
289                     protocolOpts |= SSL.SSL_PROTOCOL_TLSV1_3;
290                 }
291                 ctx = SSLContext.make(protocolOpts, mode);
292             } catch (Exception e) {
293                 throw new SSLException("failed to create an SSL_CTX", e);
294             }
295 
296             StringBuilder cipherBuilder = new StringBuilder();
297             StringBuilder cipherTLSv13Builder = new StringBuilder();
298 
299             /* List the ciphers that are permitted to negotiate. */
300             try {
301                 if (unmodifiableCiphers.isEmpty()) {
302                     // Set non TLSv1.3 ciphers.
303                     SSLContext.setCipherSuite(ctx, StringUtil.EMPTY_STRING, false);
304                     if (tlsv13Supported) {
305                         // Set TLSv1.3 ciphers.
306                         SSLContext.setCipherSuite(ctx, StringUtil.EMPTY_STRING, true);
307                     }
308                 } else {
309                     CipherSuiteConverter.convertToCipherStrings(
310                             unmodifiableCiphers, cipherBuilder, cipherTLSv13Builder, OpenSsl.isBoringSSL());
311 
312                     // Set non TLSv1.3 ciphers.
313                     SSLContext.setCipherSuite(ctx, cipherBuilder.toString(), false);
314                     if (tlsv13Supported) {
315                         // Set TLSv1.3 ciphers.
316                         SSLContext.setCipherSuite(ctx,
317                                 OpenSsl.checkTls13Ciphers(logger, cipherTLSv13Builder.toString()), true);
318                     }
319                 }
320             } catch (SSLException e) {
321                 throw e;
322             } catch (Exception e) {
323                 throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
324             }
325 
326             int options = SSLContext.getOptions(ctx) |
327                     SSL.SSL_OP_NO_SSLv2 |
328                     SSL.SSL_OP_NO_SSLv3 |
329                     // Disable TLSv1 and TLSv1.1 by default as these are not considered secure anymore
330                     // and the JDK is doing the same:
331                     // https://www.oracle.com/java/technologies/javase/8u291-relnotes.html
332                     SSL.SSL_OP_NO_TLSv1 |
333                     SSL.SSL_OP_NO_TLSv1_1 |
334 
335                     SSL.SSL_OP_CIPHER_SERVER_PREFERENCE |
336 
337                     // We do not support compression at the moment so we should explicitly disable it.
338                     SSL.SSL_OP_NO_COMPRESSION |
339 
340                     // Disable ticket support by default to be more inline with SSLEngineImpl of the JDK.
341                     // This also let SSLSession.getId() work the same way for the JDK implementation and the
342                     // OpenSSLEngine. If tickets are supported SSLSession.getId() will only return an ID on the
343                     // server-side if it could make use of tickets.
344                     SSL.SSL_OP_NO_TICKET;
345 
346             if (cipherBuilder.length() == 0) {
347                 // No ciphers that are compatible with SSLv2 / SSLv3 / TLSv1 / TLSv1.1 / TLSv1.2
348                 options |= SSL.SSL_OP_NO_SSLv2 | SSL.SSL_OP_NO_SSLv3 | SSL.SSL_OP_NO_TLSv1
349                         | SSL.SSL_OP_NO_TLSv1_1 | SSL.SSL_OP_NO_TLSv1_2;
350             }
351 
352             if (!tlsv13Supported) {
353                 // Explicit disable TLSv1.3
354                 // See https://github.com/netty/netty/issues/12968
355                 options |= SSL.SSL_OP_NO_TLSv1_3;
356             }
357 
358             SSLContext.setOptions(ctx, options);
359 
360             // We need to enable SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER as the memory address may change between
361             // calling OpenSSLEngine.wrap(...).
362             // See https://github.com/netty/netty-tcnative/issues/100
363             SSLContext.setMode(ctx, SSLContext.getMode(ctx) | SSL.SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
364 
365             if (DH_KEY_LENGTH != null) {
366                 SSLContext.setTmpDHLength(ctx, DH_KEY_LENGTH);
367             }
368 
369             List<String> nextProtoList = apn.protocols();
370             /* Set next protocols for next protocol negotiation extension, if specified */
371             if (!nextProtoList.isEmpty()) {
372                 String[] appProtocols = nextProtoList.toArray(EmptyArrays.EMPTY_STRINGS);
373                 int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
374 
375                 switch (apn.protocol()) {
376                     case NPN:
377                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
378                         break;
379                     case ALPN:
380                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
381                         break;
382                     case NPN_AND_ALPN:
383                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
384                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
385                         break;
386                     default:
387                         throw new Error();
388                 }
389             }
390 
391             if (enableOcsp) {
392                 SSLContext.enableOcsp(ctx, isClient());
393             }
394 
395             SSLContext.setUseTasks(ctx, useTasks);
396             if (privateKeyMethod != null) {
397                 SSLContext.setPrivateKeyMethod(ctx, new PrivateKeyMethod(engineMap, privateKeyMethod));
398             }
399             if (asyncPrivateKeyMethod != null) {
400                 SSLContext.setPrivateKeyMethod(ctx, new AsyncPrivateKeyMethod(engineMap, asyncPrivateKeyMethod));
401             }
402             if (certCompressionConfig != null) {
403                 for (OpenSslCertificateCompressionConfig.AlgorithmConfig configPair : certCompressionConfig) {
404                     final CertificateCompressionAlgo algo = new CompressionAlgorithm(engineMap, configPair.algorithm());
405                     switch (configPair.mode()) {
406                         case Decompress:
407                             SSLContext.addCertificateCompressionAlgorithm(
408                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_DECOMPRESS, algo);
409                             break;
410                         case Compress:
411                             SSLContext.addCertificateCompressionAlgorithm(
412                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_COMPRESS, algo);
413                             break;
414                         case Both:
415                             SSLContext.addCertificateCompressionAlgorithm(
416                                     ctx, SSL.SSL_CERT_COMPRESSION_DIRECTION_BOTH, algo);
417                             break;
418                         default:
419                             throw new IllegalStateException();
420                     }
421                 }
422             }
423             if (maxCertificateList != null) {
424                 SSLContext.setMaxCertList(ctx, maxCertificateList);
425             }
426             // Set the curves.
427             SSLContext.setCurvesList(ctx, OpenSsl.NAMED_GROUPS);
428             success = true;
429         } finally {
430             if (!success) {
431                 release();
432             }
433         }
434     }

436     private static int opensslSelectorFailureBehavior(ApplicationProtocolConfig.SelectorFailureBehavior behavior) {
437         switch (behavior) {
438             case NO_ADVERTISE:
439                 return SSL.SSL_SELECTOR_FAILURE_NO_ADVERTISE;
440             case CHOOSE_MY_LAST_PROTOCOL:
441                 return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
442             default:
443                 throw new Error();
444         }
445     }

629     //            get access to an OpenSslSessionContext after this method was called to prevent the user from
630     //            producing a segfault.
631     private void destroy() {
632         Lock writerLock = ctxLock.writeLock();
633         writerLock.lock();
634         try {
635             if (ctx != 0) {
636                 if (enableOcsp) {
637                     SSLContext.disableOcsp(ctx);
638                 }

640                 SSLContext.free(ctx);
641                 ctx = 0;
642 
643                 OpenSslSessionContext context = sessionContext();
644                 if (context != null) {
645                     context.destroy();
646                 }
647             }
648         } finally {
649             writerLock.unlock();
650         }

662         for (TrustManager m : managers) {

689      * Translate a {@link ApplicationProtocolConfig} object to a
690      * {@link OpenSslApplicationProtocolNegotiator} object.
691      *
692      * @param config The configuration which defines the translation
693      * @return The results of the translation

695     @SuppressWarnings("deprecation")
696     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
697         if (config == null) {
698             return NONE_PROTOCOL_NEGOTIATOR;
699         }
700 
701         switch (config.protocol()) {
702             case NONE:
703                 return NONE_PROTOCOL_NEGOTIATOR;
704             case ALPN:
705             case NPN:
706             case NPN_AND_ALPN:
707                 switch (config.selectedListenerFailureBehavior()) {
708                     case CHOOSE_MY_LAST_PROTOCOL:
709                     case ACCEPT:
710                         switch (config.selectorFailureBehavior()) {
711                             case CHOOSE_MY_LAST_PROTOCOL:
712                             case NO_ADVERTISE:

714                                         config);
715                             default:

717                                         new StringBuilder("OpenSSL provider does not support ")
718                                                 .append(config.selectorFailureBehavior())
719                                                 .append(" behavior").toString());

721                     default:
722                         throw new UnsupportedOperationException(
723                                 new StringBuilder("OpenSSL provider does not support ")
724                                         .append(config.selectedListenerFailureBehavior())
725                                         .append(" behavior").toString());
726                 }
727             default:
728                 throw new Error();
729         }

776     abstract static class AbstractCertificateVerifier extends CertificateVerifier {
777         private final OpenSslEngineMap engineMap;
778 
779         AbstractCertificateVerifier(OpenSslEngineMap engineMap) {
780             this.engineMap = engineMap;
781         }
782 
783         @Override
784         public final int verify(long ssl, byte[][] chain, String auth) {
785             final ReferenceCountedOpenSslEngine engine = engineMap.get(ssl);
786             if (engine == null) {
787                 // May be null if it was destroyed in the meantime.
788                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
789             }
790             X509Certificate[] peerCerts = certificates(chain);
791             try {
792                 verify(engine, peerCerts, auth);
793                 return CertificateVerifier.X509_V_OK;
794             } catch (Throwable cause) {
795                 logger.debug("verification of certificate failed", cause);
796                 engine.initHandshakeException(cause);
797 
798                 // Try to extract the correct error code that should be used.
799                 if (cause instanceof OpenSslCertificateException) {
800                     // This will never return a negative error code as its validated when constructing the
801                     // OpenSslCertificateException.
802                     return ((OpenSslCertificateException) cause).errorCode();
803                 }
804                 if (cause instanceof CertificateExpiredException) {
805                     return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
806                 }
807                 if (cause instanceof CertificateNotYetValidException) {
808                     return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
809                 }
810                 if (PlatformDependent.javaVersion() >= 7) {
811                     return translateToError(cause);
812                 }
813 
814                 // Could not detect a specific error code to use, so fallback to a default code.
815                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
816             }
817         }
818 
819         @SuppressJava6Requirement(reason = "Usage guarded by java version check")
820         private static int translateToError(Throwable cause) {
821             if (cause instanceof CertificateRevokedException) {
822                 return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
823             }
824 
825             // The X509TrustManagerImpl uses a Validator which wraps a CertPathValidatorException into
826             // an CertificateException. So we need to handle the wrapped CertPathValidatorException to be
827             // able to send the correct alert.
828             Throwable wrapped = cause.getCause();
829             while (wrapped != null) {
830                 if (wrapped instanceof CertPathValidatorException) {
831                     CertPathValidatorException ex = (CertPathValidatorException) wrapped;
832                     CertPathValidatorException.Reason reason = ex.getReason();
833                     if (reason == CertPathValidatorException.BasicReason.EXPIRED) {
834                         return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
835                     }
836                     if (reason == CertPathValidatorException.BasicReason.NOT_YET_VALID) {
837                         return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
838                     }
839                     if (reason == CertPathValidatorException.BasicReason.REVOKED) {
840                         return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
841                     }
842                 }
843                 wrapped = wrapped.getCause();
844             }
845             return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
846         }
847 
848         abstract void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts,
849                              String auth) throws Exception;
850     }

871     static void setKeyMaterial(long ctx, X509Certificate[] keyCertChain, PrivateKey key, String keyPassword)
872             throws SSLException {
873          /* Load the certificate file and private key. */
874         long keyBio = 0;
875         long keyCertChainBio = 0;
876         long keyCertChainBio2 = 0;
877         PemEncoded encoded = null;
878         try {
879             // Only encode one time
880             encoded = PemX509Certificate.toPEM(ByteBufAllocator.DEFAULT, true, keyCertChain);
881             keyCertChainBio = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
882             keyCertChainBio2 = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
883 
884             if (key != null) {
885                 keyBio = toBIO(ByteBufAllocator.DEFAULT, key);
886             }
887 
888             SSLContext.setCertificateBio(
889                     ctx, keyCertChainBio, keyBio,
890                     keyPassword == null ? StringUtil.EMPTY_STRING : keyPassword);
891             // We may have more then one cert in the chain so add all of them now.
892             SSLContext.setCertificateChainBio(ctx, keyCertChainBio2, true);
893         } catch (SSLException e) {
894             throw e;
895         } catch (Exception e) {
896             throw new SSLException("failed to set certificate and key", e);
897         } finally {
898             freeBio(keyBio);
899             freeBio(keyCertChainBio);
900             freeBio(keyCertChainBio2);
901             if (encoded != null) {
902                 encoded.release();
903             }
904         }
905     }

907     static void freeBio(long bio) {
908         if (bio != 0) {
909             SSL.freeBIO(bio);
910         }
911     }

914      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
915      * or {@code 0} if the {@code key} is {@code null}. The BIO contains the content of the {@code key}.
916      */
917     static long toBIO(ByteBufAllocator allocator, PrivateKey key) throws Exception {
918         if (key == null) {
919             return 0;
920         }
921 
922         PemEncoded pem = PemPrivateKey.toPEM(allocator, true, key);
923         try {
924             return toBIO(allocator, pem.retain());
925         } finally {
926             pem.release();
927         }

931      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
932      * or {@code 0} if the {@code certChain} is {@code null}. The BIO contains the content of the {@code certChain}.
933      */
934     static long toBIO(ByteBufAllocator allocator, X509Certificate... certChain) throws Exception {
935         if (certChain == null) {
936             return 0;
937         }
938 
939         checkNonEmpty(certChain, "certChain");
940 
941         PemEncoded pem = PemX509Certificate.toPEM(allocator, true, certChain);
942         try {
943             return toBIO(allocator, pem.retain());
944         } finally {
945             pem.release();
946         }
947     }

949     static long toBIO(ByteBufAllocator allocator, PemEncoded pem) throws Exception {
950         try {
951             // We can turn direct buffers straight into BIOs. No need to
952             // make a yet another copy.
953             ByteBuf content = pem.content();
954 
955             if (content.isDirect()) {
956                 return newBIO(content.retainedSlice());
957             }
958 
959             ByteBuf buffer = allocator.directBuffer(content.readableBytes());
960             try {
961                 buffer.writeBytes(content, content.readerIndex(), content.readableBytes());
962                 return newBIO(buffer.retainedSlice());
963             } finally {
964                 try {
965                     // If the contents of the ByteBuf is sensitive (e.g. a PrivateKey) we
966                     // need to zero out the bytes of the copy before we're releasing it.
967                     if (pem.isSensitive()) {
968                         SslUtils.zeroout(buffer);
969                     }
970                 } finally {
971                     buffer.release();
972                 }
973             }
974         } finally {
975             pem.release();
976         }
977     }
978 
979     private static long newBIO(ByteBuf buffer) throws Exception {
980         try {
981             long bio = SSL.newMemBIO();
982             int readable = buffer.readableBytes();
983             if (SSL.bioWrite(bio, OpenSsl.memoryAddress(buffer) + buffer.readerIndex(), readable) != readable) {
984                 SSL.freeBIO(bio);
985                 throw new IllegalStateException("Could not write data to memory BIO");
986             }
987             return bio;
988         } finally {
989             buffer.release();
990         }
991     }

993     /**
994      * Returns the {@link OpenSslKeyMaterialProvider} that should be used for OpenSSL. Depending on the given
995      * {@link KeyManagerFactory} this may cache the {@link OpenSslKeyMaterial} for better performance if it can
996      * ensure that the same material is always returned for the same alias.
997      */
998     static OpenSslKeyMaterialProvider providerFor(KeyManagerFactory factory, String password) {
999         if (factory instanceof OpenSslX509KeyManagerFactory) {
1000             return ((OpenSslX509KeyManagerFactory) factory).newProvider();
1001         }
1002 
1003         if (factory instanceof OpenSslCachingX509KeyManagerFactory) {
1004             // The user explicit used OpenSslCachingX509KeyManagerFactory which signals us that its fine to cache.
1005             return ((OpenSslCachingX509KeyManagerFactory) factory).newProvider(password);
1006         }
1007         // We can not be sure if the material may change at runtime so we will not cache it.
1008         return new OpenSslKeyMaterialProvider(chooseX509KeyManager(factory.getKeyManagers()), password);
1009     }
