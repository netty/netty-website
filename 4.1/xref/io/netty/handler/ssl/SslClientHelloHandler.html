34  /**
35   * {@link ByteToMessageDecoder} which allows to be notified once a full {@code ClientHello} was received.
36   */
37  public abstract class SslClientHelloHandler<T> extends ByteToMessageDecoder implements ChannelOutboundHandler {
38  
39      private static final InternalLogger logger =
40              InternalLoggerFactory.getInstance(SslClientHelloHandler.class);
41  
42      private boolean handshakeFailed;
43      private boolean suppressRead;
44      private boolean readPending;
45      private ByteBuf handshakeBuffer;
46  
47      @Override
48      protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
49          if (!suppressRead && !handshakeFailed) {
50              try {
51                  int readerIndex = in.readerIndex();
52                  int readableBytes = in.readableBytes();
53                  int handshakeLength = -1;
54  
55                  // Check if we have enough data to determine the record type and length.
56                  while (readableBytes >= SslUtils.SSL_RECORD_HEADER_LENGTH) {
57                      final int contentType = in.getUnsignedByte(readerIndex);
58                      switch (contentType) {
59                          case SslUtils.SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC:
60                              // fall-through
61                          case SslUtils.SSL_CONTENT_TYPE_ALERT:
62                              final int len = SslUtils.getEncryptedPacketLength(in, readerIndex);
63  
64                              // Not an SSL/TLS packet
65                              if (len == SslUtils.NOT_ENCRYPTED) {
66                                  handshakeFailed = true;
67                                  NotSslRecordException e = new NotSslRecordException(
68                                          "not an SSL/TLS record: " + ByteBufUtil.hexDump(in));
69                                  in.skipBytes(in.readableBytes());
70                                  ctx.fireUserEventTriggered(new SniCompletionEvent(e));
71                                  SslUtils.handleHandshakeFailure(ctx, e, true);
72                                  throw e;
73                              }
74                              if (len == SslUtils.NOT_ENOUGH_DATA) {
75                                  // Not enough data
76                                  return;
77                              }
78                              // No ClientHello
79                              select(ctx, null);
80                              return;
81                          case SslUtils.SSL_CONTENT_TYPE_HANDSHAKE:
82                              final int majorVersion = in.getUnsignedByte(readerIndex + 1);
83                              // SSLv3 or TLS
84                              if (majorVersion == 3) {
85                                  int packetLength = in.getUnsignedShort(readerIndex + 3) +
86                                          SslUtils.SSL_RECORD_HEADER_LENGTH;
87  
88                                  if (readableBytes < packetLength) {
89                                      // client hello incomplete; try again to decode once more data is ready.
90                                      return;
91                                  } else if (packetLength == SslUtils.SSL_RECORD_HEADER_LENGTH) {
92                                      select(ctx, null);
93                                      return;
94                                  }
95  
96                                  final int endOffset = readerIndex + packetLength;
97  
98                                  // Let's check if we already parsed the handshake length or not.
99                                  if (handshakeLength == -1) {
100                                     if (readerIndex + 4 > endOffset) {
101                                         // Need more data to read HandshakeType and handshakeLength (4 bytes)
102                                         return;
103                                     }
104 
105                                     final int handshakeType = in.getUnsignedByte(readerIndex +
106                                             SslUtils.SSL_RECORD_HEADER_LENGTH);
107 
108                                     // Check if this is a clientHello(1)
109                                     // See https://tools.ietf.org/html/rfc5246#section-7.4
110                                     if (handshakeType != 1) {
111                                         select(ctx, null);
112                                         return;
113                                     }
114 
115                                     // Read the length of the handshake as it may arrive in fragments
116                                     // See https://tools.ietf.org/html/rfc5246#section-7.4
117                                     handshakeLength = in.getUnsignedMedium(readerIndex +
118                                             SslUtils.SSL_RECORD_HEADER_LENGTH + 1);
119 
120                                     // Consume handshakeType and handshakeLength (this sums up as 4 bytes)
121                                     readerIndex += 4;
122                                     packetLength -= 4;
123 
124                                     if (handshakeLength + 4 + SslUtils.SSL_RECORD_HEADER_LENGTH <= packetLength) {
125                                         // We have everything we need in one packet.
126                                         // Skip the record header
127                                         readerIndex += SslUtils.SSL_RECORD_HEADER_LENGTH;
128                                         select(ctx, in.retainedSlice(readerIndex, handshakeLength));
129                                         return;
130                                     } else {
131                                         if (handshakeBuffer == null) {
132                                             handshakeBuffer = ctx.alloc().buffer(handshakeLength);
133                                         } else {
134                                             // Clear the buffer so we can aggregate into it again.
135                                             handshakeBuffer.clear();
136                                         }
137                                     }
138                                 }
139 
140                                 // Combine the encapsulated data in one buffer but not include the SSL_RECORD_HEADER
141                                 handshakeBuffer.writeBytes(in, readerIndex + SslUtils.SSL_RECORD_HEADER_LENGTH,
142                                         packetLength - SslUtils.SSL_RECORD_HEADER_LENGTH);
143                                 readerIndex += packetLength;
144                                 readableBytes -= packetLength;
145                                 if (handshakeLength <= handshakeBuffer.readableBytes()) {
146                                     ByteBuf clientHello = handshakeBuffer.setIndex(0, handshakeLength);
147                                     handshakeBuffer = null;
148 
149                                     select(ctx, clientHello);
150                                     return;
151                                 }
152                                 break;
153                             }
154                             // fall-through
155                         default:
156                             // not tls, ssl or application data
157                             select(ctx, null);
158                             return;
159                     }
160                 }
161             } catch (NotSslRecordException e) {
162                 // Just rethrow as in this case we also closed the channel and this is consistent with SslHandler.
163                 throw e;
164             } catch (Exception e) {
165                 // unexpected encoding, ignore sni and use default
166                 if (logger.isDebugEnabled()) {
167                     logger.debug("Unexpected client hello packet: " + ByteBufUtil.hexDump(in), e);
168                 }
169                 select(ctx, null);
170             }
171         }
172     }
173 
174     private void releaseHandshakeBuffer() {
175         releaseIfNotNull(handshakeBuffer);
176         handshakeBuffer = null;
177     }
178 
179     private static void releaseIfNotNull(ByteBuf buffer) {
180         if (buffer != null) {
181             buffer.release();
182         }
183     }
184 
185     private void select(final ChannelHandlerContext ctx, ByteBuf clientHello) throws Exception {
186         final Future<T> future;
187         try {
188             future = lookup(ctx, clientHello);
189             if (future.isDone()) {
190                 onLookupComplete(ctx, future);
191             } else {
192                 suppressRead = true;
193                 final ByteBuf finalClientHello = clientHello;
194                 future.addListener(new FutureListener<T>() {
195                     @Override
196                     public void operationComplete(Future<T> future) {
197                         releaseIfNotNull(finalClientHello);
198                         try {
199                             suppressRead = false;
200                             try {
201                                 onLookupComplete(ctx, future);
202                             } catch (DecoderException err) {
203                                 ctx.fireExceptionCaught(err);
204                             } catch (Exception cause) {
205                                 ctx.fireExceptionCaught(new DecoderException(cause));
206                             } catch (Throwable cause) {
207                                 ctx.fireExceptionCaught(cause);
208                             }
209                         } finally {
210                             if (readPending) {
211                                 readPending = false;
212                                 ctx.read();
213                             }
214                         }
215                     }
216                 });
217 
218                 // Ownership was transferred to the FutureListener.
219                 clientHello = null;
220             }
221         } catch (Throwable cause) {
222             PlatformDependent.throwException(cause);
223         } finally {
224             releaseIfNotNull(clientHello);
225         }
226     }
227 
228     @Override
229     protected void handlerRemoved0(ChannelHandlerContext ctx) throws Exception {
230         releaseHandshakeBuffer();
231 
232         super.handlerRemoved0(ctx);
233     }

235     /**
236      * Kicks off a lookup for the given {@code ClientHello} and returns a {@link Future} which in turn will
237      * notify the {@link #onLookupComplete(ChannelHandlerContext, Future)} on completion.
238      *
239      * See https://tools.ietf.org/html/rfc5246#section-7.4.1.2
240      *
241      * <pre>
242      * struct {
243      *    ProtocolVersion client_version;
244      *    Random random;
245      *    SessionID session_id;
246      *    CipherSuite cipher_suites<2..2^16-2>;
247      *    CompressionMethod compression_methods<1..2^8-1>;
248      *    select (extensions_present) {
249      *        case false:
250      *            struct {};
251      *        case true:
252      *            Extension extensions<0..2^16-1>;
253      *    };
254      * } ClientHello;
255      * </pre>
256      *
257      * @see #onLookupComplete(ChannelHandlerContext, Future)
258      */
259     protected abstract Future<T> lookup(ChannelHandlerContext ctx, ByteBuf clientHello) throws Exception;
260 
261     /**
262      * Called upon completion of the {@link #lookup(ChannelHandlerContext, ByteBuf)} {@link Future}.
263      *
264      * @see #lookup(ChannelHandlerContext, ByteBuf)
265      */
266     protected abstract void onLookupComplete(ChannelHandlerContext ctx, Future<T> future) throws Exception;
267 
268     @Override
269     public void read(ChannelHandlerContext ctx) throws Exception {
270         if (suppressRead) {
271             readPending = true;
272         } else {
273             ctx.read();
274         }
275     }
