16  package io.netty.handler.ssl;
17  
18  import io.netty.buffer.ByteBufAllocator;
19  
20  import javax.net.ssl.SSLEngine;
21  
22  /**
23   * Implements a {@link SSLEngine} using
24   * <a href="https://www.openssl.org/docs/crypto/BIO_s_bio.html#EXAMPLE">OpenSSL BIO abstractions</a>.
25   * <p>
26   * This class will use a finalizer to ensure native resources are automatically cleaned up. To avoid finalizers
27   * and manually release the native memory see {@link ReferenceCountedOpenSslEngine}.
28   */
29  public final class OpenSslEngine extends ReferenceCountedOpenSslEngine {
30      OpenSslEngine(OpenSslContext context, ByteBufAllocator alloc, String peerHost, int peerPort,
31                    boolean jdkCompatibilityMode) {
32          super(context, alloc, peerHost, peerPort, jdkCompatibilityMode, false);
33      }
34  
35      @Override
36      @SuppressWarnings("FinalizeDeclaration")
37      protected void finalize() throws Throwable {
38          super.finalize();
39          OpenSsl.releaseIfNeeded(this);
40      }
41  }
