36  import static io.netty.util.internal.ObjectUtil.checkNotNull;

38  /**
39   * {@link OcspServerCertificateValidator} validates incoming server's certificate
40   * using OCSP. Once TLS handshake is completed, {@link SslHandshakeCompletionEvent#SUCCESS} is fired, validator
41   * will perform certificate validation using OCSP over HTTP/1.1 with the server's certificate issuer OCSP responder.
42   */
43  public class OcspServerCertificateValidator extends ChannelInboundHandlerAdapter {
44  
45      private final boolean closeAndThrowIfNotValid;
46      private final boolean validateNonce;
47      private final IoTransport ioTransport;
48      private final DnsNameResolver dnsNameResolver;
49  
50      /**
51       * Create a new {@link OcspServerCertificateValidator} instance without nonce validation
52       * on OCSP response, using default {@link IoTransport#DEFAULT} instance,
53       * default {@link DnsNameResolver} implementation and with {@link #closeAndThrowIfNotValid}
54       * set to {@code true}
55       */
56      public OcspServerCertificateValidator() {
57          this(false);
58      }
59  
60      /**
61       * Create a new {@link OcspServerCertificateValidator} instance with
62       * default {@link IoTransport#DEFAULT} instance and default {@link DnsNameResolver} implementation
63       * and {@link #closeAndThrowIfNotValid} set to {@code true}.
64       *
65       * @param validateNonce Set to {@code true} if we should force nonce validation on
66       *                      OCSP response else set to {@code false}
67       */
68      public OcspServerCertificateValidator(boolean validateNonce) {
69          this(validateNonce, IoTransport.DEFAULT);
70      }
71  
72      /**
73       * Create a new {@link OcspServerCertificateValidator} instance
74       *
75       * @param validateNonce Set to {@code true} if we should force nonce validation on
76       *                      OCSP response else set to {@code false}
77       * @param ioTransport   {@link IoTransport} to use
78       */
79      public OcspServerCertificateValidator(boolean validateNonce, IoTransport ioTransport) {
80          this(validateNonce, ioTransport, createDefaultResolver(ioTransport));
81      }
82  
83      /**
84       * Create a new {@link IoTransport} instance with {@link #closeAndThrowIfNotValid} set to {@code true}
85       *
86       * @param validateNonce   Set to {@code true} if we should force nonce validation on
87       *                        OCSP response else set to {@code false}
88       * @param ioTransport     {@link IoTransport} to use
89       * @param dnsNameResolver {@link DnsNameResolver} implementation to use
90       */
91      public OcspServerCertificateValidator(boolean validateNonce, IoTransport ioTransport,
92                                            DnsNameResolver dnsNameResolver) {
93          this(true, validateNonce, ioTransport, dnsNameResolver);
94      }
95  
96      /**
97       * Create a new {@link IoTransport} instance
98       *
99       * @param closeAndThrowIfNotValid If set to {@code true} then we will close the channel and throw an exception
100      *                                when certificate is not {@link OcspResponse.Status#VALID}.
101      *                                If set to {@code false} then we will simply pass the {@link OcspValidationEvent}
102      *                                to the next handler in pipeline and let it decide what to do.
103      * @param validateNonce           Set to {@code true} if we should force nonce validation on
104      *                                OCSP response else set to {@code false}
105      * @param ioTransport             {@link IoTransport} to use
106      * @param dnsNameResolver         {@link DnsNameResolver} implementation to use
107      */
108     public OcspServerCertificateValidator(boolean closeAndThrowIfNotValid, boolean validateNonce,
109                                           IoTransport ioTransport, DnsNameResolver dnsNameResolver) {
110         this.closeAndThrowIfNotValid = closeAndThrowIfNotValid;
111         this.validateNonce = validateNonce;
112         this.ioTransport = checkNotNull(ioTransport, "IoTransport");
113         this.dnsNameResolver = checkNotNull(dnsNameResolver, "DnsNameResolver");
114     }
115 
116     protected static DnsNameResolver createDefaultResolver(final IoTransport ioTransport) {
117         return new DnsNameResolverBuilder()
118                 .eventLoop(ioTransport.eventLoop())
119                 .channelFactory(ioTransport.datagramChannel())
120                 .socketChannelFactory(ioTransport.socketChannel())
121                 .build();
122     }
123 
124     @Override
125     public void userEventTriggered(final ChannelHandlerContext ctx, final Object evt) throws Exception {
126         ctx.fireUserEventTriggered(evt);
127 
128         if (evt instanceof SslHandshakeCompletionEvent) {
129             SslHandshakeCompletionEvent sslHandshakeCompletionEvent = (SslHandshakeCompletionEvent) evt;
130 
131             // If TLS handshake was successful then only we will perform OCSP certificate validation.
132             // If not, then just forward the event to next handler in pipeline and remove ourselves from pipeline.
133             if (sslHandshakeCompletionEvent.isSuccess()) {
134                 Certificate[] certificates = ctx.pipeline().get(SslHandler.class)
135                         .engine()
136                         .getSession()
137                         .getPeerCertificates();
138 
139                 assert certificates.length >= 2 : "There must an end-entity certificate and issuer certificate";
140 
141                 Promise<BasicOCSPResp> ocspRespPromise = OcspClient.query((X509Certificate) certificates[0],
142                         (X509Certificate) certificates[1], validateNonce, ioTransport, dnsNameResolver);
143 
144                 ocspRespPromise.addListener(new GenericFutureListener<Future<BasicOCSPResp>>() {
145                     @Override
146                     public void operationComplete(Future<BasicOCSPResp> future) throws Exception {
147                         // If Future is success then we have successfully received OCSP response
148                         // from OCSP responder. We will validate it now and process.
149                         if (future.isSuccess()) {
150                             SingleResp response = future.get().getResponses()[0];
151 
152                             Date current = new Date();
153                             if (!(current.after(response.getThisUpdate()) &&
154                                     current.before(response.getNextUpdate()))) {
155                                 ctx.fireExceptionCaught(new IllegalStateException("OCSP Response is out-of-date"));
156                             }
157 
158                             OcspResponse.Status status;
159                             if (response.getCertStatus() == null) {
160                                 // 'null' means certificate is valid
161                                 status = OcspResponse.Status.VALID;
162                             } else if (response.getCertStatus() instanceof RevokedStatus) {
163                                 status = OcspResponse.Status.REVOKED;
164                             } else {
165                                 status = OcspResponse.Status.UNKNOWN;
166                             }
167 
168                             ctx.fireUserEventTriggered(new OcspValidationEvent(
169                                     new OcspResponse(status, response.getThisUpdate(), response.getNextUpdate())));
170 
171                             // If Certificate is not VALID and 'closeAndThrowIfNotValid' is set
172                             // to 'true' then close the channel and throw an exception.
173                             if (status != OcspResponse.Status.VALID && closeAndThrowIfNotValid) {
174                                 ctx.channel().close();
175                                 // Certificate is not valid. Throw
176                                 ctx.fireExceptionCaught(new OCSPException(
177                                         "Certificate not valid. Status: " + status));
178                             }
179                         } else {
180                             ctx.fireExceptionCaught(future.cause());
181                         }
182                     }
183                 });
184             }
185             // Lets remove ourselves from the pipeline because we are done processing validation.
186             ctx.pipeline().remove(this);
187         }
188     }
189 
190     @Override
191     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
192         ctx.channel().close();
193     }
194 }
