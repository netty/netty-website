102     private static String[] defaultProtocols(SSLContext context, SSLEngine engine) {
103         // Choose the sensible default list of protocols that respects JDK flags, eg. jdk.tls.client.protocols
104         final String[] supportedProtocols = context.getDefaultSSLParameters().getProtocols();
105         Set<String> supportedProtocolsSet = new HashSet<String>(supportedProtocols.length);
106         Collections.addAll(supportedProtocolsSet, supportedProtocols);
107         List<String> protocols = new ArrayList<String>();
108         addIfSupported(
109                 supportedProtocolsSet, protocols,
110                 SslUtils.PROTOCOL_TLS_V1_3, SslUtils.PROTOCOL_TLS_V1_2,
111                 SslUtils.PROTOCOL_TLS_V1_1, SslUtils.PROTOCOL_TLS_V1);
112 
113         if (!protocols.isEmpty()) {
114             return protocols.toArray(EmptyArrays.EMPTY_STRINGS);
115         }
116         return engine.getEnabledProtocols();
117     }

119     private static Set<String> supportedCiphers(SSLEngine engine) {
120         // Choose the sensible default list of cipher suites.
121         final String[] supportedCiphers = engine.getSupportedCipherSuites();
122         Set<String> supportedCiphersSet = new LinkedHashSet<String>(supportedCiphers.length);
123         for (int i = 0; i < supportedCiphers.length; ++i) {
124             String supportedCipher = supportedCiphers[i];
125             supportedCiphersSet.add(supportedCipher);
126             // IBM's J9 JVM utilizes a custom naming scheme for ciphers and only returns ciphers with the "SSL_"
127             // prefix instead of the "TLS_" prefix (as defined in the JSSE cipher suite names [1]). According to IBM's
128             // documentation [2] the "SSL_" prefix is "interchangeable" with the "TLS_" prefix.
129             // See the IBM forum discussion [3] and issue on IBM's JVM [4] for more details.
130             //[1] https://docs.oracle.com/javase/8/docs/technotes/guides/security/StandardNames.html#ciphersuites
131             //[2] https://www.ibm.com/support/knowledgecenter/en/SSYKE2_8.0.0/com.ibm.java.security.component.80.doc/
132             // security-component/jsse2Docs/ciphersuites.html
133             //[3] https://www.ibm.com/developerworks/community/forums/html/topic?id=9b5a56a9-fa46-4031-b33b-df91e28d77c2
134             //[4] https://www.ibm.com/developerworks/rfe/execute?use_case=viewRfe&CR_ID=71770
135             if (supportedCipher.startsWith("SSL_")) {
136                 final String tlsPrefixedCipherName = "TLS_" + supportedCipher.substring("SSL_".length());
137                 try {
138                     engine.setEnabledCipherSuites(new String[]{tlsPrefixedCipherName});
139                     supportedCiphersSet.add(tlsPrefixedCipherName);
140                 } catch (IllegalArgumentException ignored) {
141                     // The cipher is not supported ... move on to the next cipher.
142                 }
143             }
144         }
145         return supportedCiphersSet;
146     }

148     private static List<String> defaultCiphers(SSLEngine engine, Set<String> supportedCiphers) {
149         List<String> ciphers = new ArrayList<String>();
150         addIfSupported(supportedCiphers, ciphers, DEFAULT_CIPHER_SUITES);
151         useFallbackCiphersIfDefaultIsEmpty(ciphers, engine.getEnabledCipherSuites());
152         return ciphers;
153     }

155     private static boolean isTlsV13Supported(String[] protocols) {
156         for (String protocol: protocols) {
157             if (SslUtils.PROTOCOL_TLS_V1_3.equals(protocol)) {
158                 return true;
159             }
160         }
161         return false;
162     }

238     @SuppressWarnings("deprecation")
239     JdkSslContext(SSLContext sslContext, boolean isClient, Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
240                   JdkApplicationProtocolNegotiator apn, ClientAuth clientAuth, String[] protocols, boolean startTls) {
241         super(startTls);
242         this.apn = checkNotNull(apn, "apn");
243         this.clientAuth = checkNotNull(clientAuth, "clientAuth");
244         this.sslContext = checkNotNull(sslContext, "sslContext");
245 
246         final List<String> defaultCiphers;
247         final Set<String> supportedCiphers;
248         if (DEFAULT_PROVIDER.equals(sslContext.getProvider())) {
249             this.protocols = protocols == null? DEFAULT_PROTOCOLS : protocols;
250             if (isTlsV13Supported(this.protocols)) {
251                 supportedCiphers = SUPPORTED_CIPHERS;
252                 defaultCiphers = DEFAULT_CIPHERS;
253             } else {
254                 // TLSv1.3 is not supported, ensure we do not include any TLSv1.3 ciphersuite.
255                 supportedCiphers = SUPPORTED_CIPHERS_NON_TLSV13;
256                 defaultCiphers = DEFAULT_CIPHERS_NON_TLSV13;
257             }
258         } else {
259             // This is a different Provider then the one used by the JDK by default so we can not just assume
260             // the same protocols and ciphers are supported. For example even if Java11+ is used Conscrypt will
261             // not support TLSv1.3 and the TLSv1.3 ciphersuites.
262             SSLEngine engine = sslContext.createSSLEngine();
263             try {
264                 if (protocols == null) {
265                     this.protocols = defaultProtocols(sslContext, engine);
266                 } else {
267                     this.protocols = protocols;
268                 }
269                 supportedCiphers = supportedCiphers(engine);
270                 defaultCiphers = defaultCiphers(engine, supportedCiphers);
271                 if (!isTlsV13Supported(this.protocols)) {
272                     // TLSv1.3 is not supported, ensure we do not include any TLSv1.3 ciphersuite.
273                     for (String cipher: SslUtils.DEFAULT_TLSV13_CIPHER_SUITES) {
274                         supportedCiphers.remove(cipher);
275                         defaultCiphers.remove(cipher);
276                     }
277                 }
278             } finally {
279                 ReferenceCountUtil.release(engine);
280             }
281         }
282 
283         cipherSuites = checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
284                 ciphers, defaultCiphers, supportedCiphers);
285 
286         unmodifiableCipherSuites = Collections.unmodifiableList(Arrays.asList(cipherSuites));
287         this.isClient = isClient;
288     }

339     @SuppressWarnings("deprecation")
340     private SSLEngine configureAndWrapEngine(SSLEngine engine, ByteBufAllocator alloc) {
341         engine.setEnabledCipherSuites(cipherSuites);
342         engine.setEnabledProtocols(protocols);
343         engine.setUseClientMode(isClient());
344         if (isServer()) {
345             switch (clientAuth) {
346                 case OPTIONAL:
347                     engine.setWantClientAuth(true);
348                     break;
349                 case REQUIRE:
350                     engine.setNeedClientAuth(true);
351                     break;
352                 case NONE:
353                     break; // exhaustive cases
354                 default:
355                     throw new Error("Unknown auth " + clientAuth);
356             }
357         }
358         JdkApplicationProtocolNegotiator.SslEngineWrapperFactory factory = apn.wrapperFactory();
359         if (factory instanceof JdkApplicationProtocolNegotiator.AllocatorAwareSslEngineWrapperFactory) {
360             return ((JdkApplicationProtocolNegotiator.AllocatorAwareSslEngineWrapperFactory) factory)
361                     .wrapSslEngine(engine, alloc, apn, isServer());
362         }
363         return factory.wrapSslEngine(engine, apn, isServer());
364     }

371     /**
372      * Translate a {@link ApplicationProtocolConfig} object to a {@link JdkApplicationProtocolNegotiator} object.
373      * @param config The configuration which defines the translation
374      * @param isServer {@code true} if a server {@code false} otherwise.
375      * @return The results of the translation
376      */
377     @SuppressWarnings("deprecation")
378     static JdkApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config, boolean isServer) {
379         if (config == null) {
380             return JdkDefaultApplicationProtocolNegotiator.INSTANCE;
381         }
382 
383         switch(config.protocol()) {
384         case NONE:
385             return JdkDefaultApplicationProtocolNegotiator.INSTANCE;
386         case ALPN:
387             if (isServer) {
388                 switch(config.selectorFailureBehavior()) {
389                 case FATAL_ALERT:
390                     return new JdkAlpnApplicationProtocolNegotiator(true, config.supportedProtocols());
391                 case NO_ADVERTISE:
392                     return new JdkAlpnApplicationProtocolNegotiator(false, config.supportedProtocols());
393                 default:
394                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
395                     .append(config.selectorFailureBehavior()).append(" failure behavior").toString());
396                 }
397             } else {
398                 switch(config.selectedListenerFailureBehavior()) {
399                 case ACCEPT:
400                     return new JdkAlpnApplicationProtocolNegotiator(false, config.supportedProtocols());
401                 case FATAL_ALERT:
402                     return new JdkAlpnApplicationProtocolNegotiator(true, config.supportedProtocols());
403                 default:
404                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
405                     .append(config.selectedListenerFailureBehavior()).append(" failure behavior").toString());
406                 }
407             }
408         case NPN:
409             if (isServer) {
410                 switch(config.selectedListenerFailureBehavior()) {
411                 case ACCEPT:
412                     return new JdkNpnApplicationProtocolNegotiator(false, config.supportedProtocols());
413                 case FATAL_ALERT:
414                     return new JdkNpnApplicationProtocolNegotiator(true, config.supportedProtocols());
415                 default:
416                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
417                     .append(config.selectedListenerFailureBehavior()).append(" failure behavior").toString());
418                 }
419             } else {
420                 switch(config.selectorFailureBehavior()) {
421                 case FATAL_ALERT:
422                     return new JdkNpnApplicationProtocolNegotiator(true, config.supportedProtocols());
423                 case NO_ADVERTISE:
424                     return new JdkNpnApplicationProtocolNegotiator(false, config.supportedProtocols());
425                 default:
426                     throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
427                     .append(config.selectorFailureBehavior()).append(" failure behavior").toString());
428                 }
429             }
430         default:
431             throw new UnsupportedOperationException(new StringBuilder("JDK provider does not support ")
432             .append(config.protocol()).append(" protocol").toString());
433         }
434     }
