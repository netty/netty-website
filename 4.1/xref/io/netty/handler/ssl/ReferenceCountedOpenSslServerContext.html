35  /**
36   * A server-side {@link SslContext} which uses OpenSSL's SSL/TLS implementation.
37   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
38   *
39   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
40   * which depends upon the instance of this class is released. Otherwise if any method of
41   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
42   */
43  public final class ReferenceCountedOpenSslServerContext extends ReferenceCountedOpenSslContext {
44      private static final byte[] ID = {'n', 'e', 't', 't', 'y'};
45      private final OpenSslServerSessionContext sessionContext;
46      private final OpenSslKeyMaterialManager keyMaterialManager;
47  
48      ReferenceCountedOpenSslServerContext(
49              X509Certificate[] trustCertCollection, TrustManagerFactory trustManagerFactory,
50              X509Certificate[] keyCertChain, PrivateKey key, String keyPassword, KeyManagerFactory keyManagerFactory,
51              Iterable<String> ciphers, CipherSuiteFilter cipherFilter, ApplicationProtocolConfig apn,
52              long sessionCacheSize, long sessionTimeout, ClientAuth clientAuth) throws SSLException {
53          this(trustCertCollection, trustManagerFactory, keyCertChain, key, keyPassword, keyManagerFactory, ciphers,
54                  cipherFilter, toNegotiator(apn), sessionCacheSize, sessionTimeout, clientAuth);
55      }
56  
57      private ReferenceCountedOpenSslServerContext(
58              X509Certificate[] trustCertCollection, TrustManagerFactory trustManagerFactory,
59              X509Certificate[] keyCertChain, PrivateKey key, String keyPassword, KeyManagerFactory keyManagerFactory,
60              Iterable<String> ciphers, CipherSuiteFilter cipherFilter, OpenSslApplicationProtocolNegotiator apn,
61              long sessionCacheSize, long sessionTimeout, ClientAuth clientAuth) throws SSLException {
62          super(ciphers, cipherFilter, apn, sessionCacheSize, sessionTimeout, SSL.SSL_MODE_SERVER, keyCertChain,
63                clientAuth, true);
64          // Create a new SSL_CTX and configure it.
65          boolean success = false;
66          try {
67              ServerContext context = newSessionContext(this, ctx, engineMap, trustCertCollection, trustManagerFactory,
68                                                        keyCertChain, key, keyPassword, keyManagerFactory);
69              sessionContext = context.sessionContext;
70              keyMaterialManager = context.keyMaterialManager;
71              success = true;
72          } finally {
73              if (!success) {
74                  release();
75              }
76          }
77      }
78  
79      @Override
80      public OpenSslServerSessionContext sessionContext() {
81          return sessionContext;
82      }
83  
84      @Override
85      OpenSslKeyMaterialManager keyMaterialManager() {
86          return keyMaterialManager;
87      }

89      static final class ServerContext {
90          OpenSslServerSessionContext sessionContext;
91          OpenSslKeyMaterialManager keyMaterialManager;
92      }
93  
94      static ServerContext newSessionContext(ReferenceCountedOpenSslContext thiz, long ctx, OpenSslEngineMap engineMap,
95                                             X509Certificate[] trustCertCollection,
96                                             TrustManagerFactory trustManagerFactory,
97                                             X509Certificate[] keyCertChain, PrivateKey key,
98                                             String keyPassword, KeyManagerFactory keyManagerFactory)
99              throws SSLException {
100         ServerContext result = new ServerContext();
101         synchronized (ReferenceCountedOpenSslContext.class) {
102             try {
103                 SSLContext.setVerify(ctx, SSL.SSL_CVERIFY_NONE, VERIFY_DEPTH);
104                 if (!OpenSsl.useKeyManagerFactory()) {
105                     if (keyManagerFactory != null) {
106                         throw new IllegalArgumentException(
107                                 "KeyManagerFactory not supported");
108                     }
109                     checkNotNull(keyCertChain, "keyCertChain");
110 
111                         /* Set certificate verification policy. */
112                     SSLContext.setVerify(ctx, SSL.SSL_CVERIFY_NONE, VERIFY_DEPTH);
113 
114                     setKeyMaterial(ctx, keyCertChain, key, keyPassword);
115                 } else {
116                     // javadocs state that keyManagerFactory has precedent over keyCertChain, and we must have a
117                     // keyManagerFactory for the server so build one if it is not specified.
118                     if (keyManagerFactory == null) {
119                         keyManagerFactory = buildKeyManagerFactory(
120                                 keyCertChain, key, keyPassword, keyManagerFactory);
121                     }
122                     X509KeyManager keyManager = chooseX509KeyManager(keyManagerFactory.getKeyManagers());
123                     result.keyMaterialManager = useExtendedKeyManager(keyManager) ?
124                             new OpenSslExtendedKeyMaterialManager(
125                                     (X509ExtendedKeyManager) keyManager, keyPassword) :
126                             new OpenSslKeyMaterialManager(keyManager, keyPassword);
127                 }
128             } catch (Exception e) {
129                 throw new SSLException("failed to set certificate and key", e);
130             }
131             try {
132                 if (trustCertCollection != null) {
133                     trustManagerFactory = buildTrustManagerFactory(trustCertCollection, trustManagerFactory);
134                 } else if (trustManagerFactory == null) {
135                     // Mimic the way SSLContext.getInstance(KeyManager[], null, null) works
136                     trustManagerFactory = TrustManagerFactory.getInstance(
137                             TrustManagerFactory.getDefaultAlgorithm());
138                     trustManagerFactory.init((KeyStore) null);
139                 }
140 
141                 final X509TrustManager manager = chooseTrustManager(trustManagerFactory.getTrustManagers());
142 
143                 // IMPORTANT: The callbacks set for verification must be static to prevent memory leak as
144                 //            otherwise the context can never be collected. This is because the JNI code holds
145                 //            a global reference to the callbacks.
146                 //
147                 //            See https://github.com/netty/netty/issues/5372
148 
149                 // Use this to prevent an error when running on java < 7
150                 if (useExtendedTrustManager(manager)) {
151                     SSLContext.setCertVerifyCallback(ctx,
152                             new ExtendedTrustManagerVerifyCallback(engineMap, (X509ExtendedTrustManager) manager));
153                 } else {
154                     SSLContext.setCertVerifyCallback(ctx, new TrustManagerVerifyCallback(engineMap, manager));
155                 }
156             } catch (Exception e) {
157                 throw new SSLException("unable to setup trustmanager", e);
158             }
159         }
160 
161         result.sessionContext = new OpenSslServerSessionContext(thiz);
162         result.sessionContext.setSessionIdContext(ID);
163         return result;
164     }
165 
166     private static final class TrustManagerVerifyCallback extends AbstractCertificateVerifier {
167         private final X509TrustManager manager;
168 
169         TrustManagerVerifyCallback(OpenSslEngineMap engineMap, X509TrustManager manager) {
170             super(engineMap);
171             this.manager = manager;
172         }
173 
174         @Override
175         void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts, String auth)
176                 throws Exception {
177             manager.checkClientTrusted(peerCerts, auth);
178         }
179     }
180 
181     private static final class ExtendedTrustManagerVerifyCallback extends AbstractCertificateVerifier {
182         private final X509ExtendedTrustManager manager;
183 
184         ExtendedTrustManagerVerifyCallback(OpenSslEngineMap engineMap, X509ExtendedTrustManager manager) {
185             super(engineMap);
186             this.manager = manager;
187         }
188 
189         @Override
190         void verify(ReferenceCountedOpenSslEngine engine, X509Certificate[] peerCerts, String auth)
191                 throws Exception {
192             manager.checkClientTrusted(peerCerts, auth, engine);
193         }
194     }
