17  package io.netty.resolver;
18  
19  import io.netty.util.concurrent.EventExecutor;
20  import io.netty.util.concurrent.Promise;
21  
22  import java.net.InetAddress;
23  import java.net.InetSocketAddress;
24  import java.net.UnknownHostException;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  /**
29   * A {@link NameResolver} that resolves an {@link InetSocketAddress} using JDK's built-in domain name lookup mechanism.
30   * Note that this resolver performs a blocking name lookup from the caller thread.
31   */
32  public class DefaultNameResolver extends SimpleNameResolver<InetSocketAddress> {
33  
34      public DefaultNameResolver(EventExecutor executor) {
35          super(executor);
36      }
37  
38      @Override
39      protected boolean doIsResolved(InetSocketAddress address) {
40          return !address.isUnresolved();
41      }
42  
43      @Override
44      protected void doResolve(InetSocketAddress unresolvedAddress, Promise<InetSocketAddress> promise) throws Exception {
45          try {
46              // Note that InetSocketAddress.getHostName() will never incur a reverse lookup here,
47              // because an unresolved address always has a host name.
48              promise.setSuccess(new InetSocketAddress(
49                      InetAddress.getByName(unresolvedAddress.getHostName()), unresolvedAddress.getPort()));
50          } catch (UnknownHostException e) {
51              promise.setFailure(e);
52          }
53      }
54  
55      @Override
56      protected void doResolveAll(
57              InetSocketAddress unresolvedAddress, Promise<List<InetSocketAddress>> promise) throws Exception {
58  
59          try {
60              // Note that InetSocketAddress.getHostName() will never incur a reverse lookup here,
61              // because an unresolved address always has a host name.
62              final InetAddress[] resolved = InetAddress.getAllByName(unresolvedAddress.getHostName());
63              final List<InetSocketAddress> result = new ArrayList<InetSocketAddress>(resolved.length);
64              for (InetAddress a: resolved) {
65                  result.add(new InetSocketAddress(a, unresolvedAddress.getPort()));
66              }
67              promise.setSuccess(result);
68          } catch (UnknownHostException e) {
69              promise.setFailure(e);
70          }
71      }
72  }
