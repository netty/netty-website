16  package io.netty.resolver.dns;
17  
18  import io.netty.channel.EventLoop;
19  import io.netty.util.internal.PlatformDependent;
20  import io.netty.util.internal.UnstableApi;
21  
22  import java.net.InetSocketAddress;
23  import java.util.Collections;
24  import java.util.Comparator;
25  import java.util.List;
26  import java.util.concurrent.ConcurrentMap;
27  
28  import static io.netty.util.internal.ObjectUtil.*;
29  
30  /**
31   * Default implementation of {@link AuthoritativeDnsServerCache}, backed by a {@link ConcurrentMap}.
32   */
33  @UnstableApi
34  public class DefaultAuthoritativeDnsServerCache implements AuthoritativeDnsServerCache {
35  
36      private final int minTtl;
37      private final int maxTtl;
38      private final Comparator<InetSocketAddress> comparator;
39      private final Cache<InetSocketAddress> resolveCache = new Cache<InetSocketAddress>() {
40          @Override
41          protected boolean shouldReplaceAll(InetSocketAddress entry) {
42              return false;
43          }
44  
45          @Override
46          protected boolean equals(InetSocketAddress entry, InetSocketAddress otherEntry) {
47              if (PlatformDependent.javaVersion() >= 7) {
48                  return entry.getHostString().equalsIgnoreCase(otherEntry.getHostString());
49              }
50              return entry.getHostName().equalsIgnoreCase(otherEntry.getHostName());
51          }
52  
53          @Override
54          protected void sortEntries(String hostname, List<InetSocketAddress> entries) {
55              if (comparator != null) {
56                  Collections.sort(entries, comparator);
57              }
58          }
59      };
60  
61      /**
62       * Create a cache that respects the TTL returned by the DNS server.
63       */
64      public DefaultAuthoritativeDnsServerCache() {
65          this(0, Cache.MAX_SUPPORTED_TTL_SECS, null);
66      }
67  
68      /**
69       * Create a cache.
70       *
71       * @param minTtl the minimum TTL
72       * @param maxTtl the maximum TTL
73       * @param comparator the {@link Comparator} to order the {@link InetSocketAddress} for a hostname or {@code null}
74       *                   if insertion order should be used.
75       */
76      public DefaultAuthoritativeDnsServerCache(int minTtl, int maxTtl, Comparator<InetSocketAddress> comparator) {
77          this.minTtl = Math.min(Cache.MAX_SUPPORTED_TTL_SECS, checkPositiveOrZero(minTtl, "minTtl"));
78          this.maxTtl = Math.min(Cache.MAX_SUPPORTED_TTL_SECS, checkPositive(maxTtl, "maxTtl"));
79          if (minTtl > maxTtl) {
80              throw new IllegalArgumentException(
81                      "minTtl: " + minTtl + ", maxTtl: " + maxTtl + " (expected: 0 <= minTtl <= maxTtl)");
82          }
83          this.comparator = comparator;
84      }
85  
86      @SuppressWarnings("unchecked")
87      @Override
88      public DnsServerAddressStream get(String hostname) {
89          checkNotNull(hostname, "hostname");
90  
91          List<? extends InetSocketAddress> addresses = resolveCache.get(hostname);
92          if (addresses == null || addresses.isEmpty()) {
93              return null;
94          }
95          return new SequentialDnsServerAddressStream(addresses, 0);
96      }
97  
98      @Override
99      public void cache(String hostname, InetSocketAddress address, long originalTtl, EventLoop loop) {
100         checkNotNull(hostname, "hostname");
101         checkNotNull(address, "address");
102         checkNotNull(loop, "loop");
103 
104         if (PlatformDependent.javaVersion() >= 7 && address.getHostString() == null) {
105             // We only cache addresses that have also a host string as we will need it later when trying to replace
106             // unresolved entries in the cache.
107             return;
108         }
109 
110         resolveCache.cache(hostname, address, Math.max(minTtl, (int) Math.min(maxTtl, originalTtl)), loop);
111     }
112 
113     @Override
114     public void clear() {
115         resolveCache.clear();
116     }
117 
118     @Override
119     public boolean clear(String hostname) {
120         return resolveCache.clear(checkNotNull(hostname, "hostname"));
121     }
122 
123     @Override
124     public String toString() {
125         return "DefaultAuthoritativeDnsServerCache(minTtl=" + minTtl + ", maxTtl=" + maxTtl + ", cached nameservers=" +
126                 resolveCache.size() + ')';
127     }
128 }
