164     /**
165      * Returns {@code true} if any {@link NetworkInterface} supports {@code IPv6}, {@code false} otherwise.
166      */
167     private static boolean anyInterfaceSupportsIpV6() {
168         try {
169             Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
170             while (interfaces.hasMoreElements()) {
171                 NetworkInterface iface = interfaces.nextElement();
172                 Enumeration<InetAddress> addresses = iface.getInetAddresses();
173                 while (addresses.hasMoreElements()) {
174                     InetAddress inetAddress = addresses.nextElement();
175                     if (inetAddress instanceof Inet6Address && !inetAddress.isAnyLocalAddress() &&
176                         !inetAddress.isLoopbackAddress() && !inetAddress.isLinkLocalAddress()) {
177                         return true;
178                     }
179                 }
180             }
181         } catch (SocketException e) {
182             logger.debug("Unable to detect if any interface supports IPv6, assuming IPv4-only", e);
183             // ignore
184         }
185         return false;
186     }

188     @SuppressWarnings("unchecked")
189     private static List<String> getSearchDomainsHack() throws Exception {
190         // Only try if not using Java9 and later
191         // See https://github.com/netty/netty/issues/9500
192         if (PlatformDependent.javaVersion() < 9) {
193             // This code on Java 9+ yields a warning about illegal reflective access that will be denied in
194             // a future release. There doesn't seem to be a better way to get search domains for Windows yet.
195             Class<?> configClass = Class.forName("sun.net.dns.ResolverConfiguration");
196             Method open = configClass.getMethod("open");
197             Method nameservers = configClass.getMethod("searchlist");
198             Object instance = open.invoke(null);
199 
200             return (List<String>) nameservers.invoke(instance);
201         }
202         return Collections.emptyList();
203     }

205     private static final DatagramDnsResponseDecoder DATAGRAM_DECODER = new DatagramDnsResponseDecoder() {
206         @Override
207         protected DnsResponse decodeResponse(ChannelHandlerContext ctx, DatagramPacket packet) throws Exception {
208             DnsResponse response = super.decodeResponse(ctx, packet);
209             if (packet.content().isReadable()) {
210                 // If there is still something to read we did stop parsing because of a truncated message.
211                 // This can happen if we enabled EDNS0 but our MTU is not big enough to handle all the
212                 // data.
213                 response.setTruncated(true);
214 
215                 if (logger.isDebugEnabled()) {
216                     logger.debug(
217                             "{} RECEIVED: UDP truncated packet received, consider adjusting maxPayloadSize for the {}.",
218                             ctx.channel(), StringUtil.simpleClassName(DnsNameResolver.class));
219                 }
220             }
221             return response;
222         }

403     DnsNameResolver(
404             EventLoop eventLoop,
405             ChannelFactory<? extends DatagramChannel> channelFactory,
406             ChannelFactory<? extends SocketChannel> socketChannelFactory,
407             final DnsCache resolveCache,
408             final DnsCnameCache cnameCache,
409             final AuthoritativeDnsServerCache authoritativeDnsServerCache,
410             SocketAddress localAddress,
411             DnsQueryLifecycleObserverFactory dnsQueryLifecycleObserverFactory,
412             long queryTimeoutMillis,
413             ResolvedAddressTypes resolvedAddressTypes,
414             boolean recursionDesired,
415             int maxQueriesPerResolve,
416             boolean traceEnabled,
417             int maxPayloadSize,
418             boolean optResourceEnabled,
419             HostsFileEntriesResolver hostsFileEntriesResolver,
420             DnsServerAddressStreamProvider dnsServerAddressStreamProvider,
421             String[] searchDomains,
422             int ndots,
423             boolean decodeIdn,
424             boolean completeOncePreferredResolved) {
425         super(eventLoop);
426         this.queryTimeoutMillis = queryTimeoutMillis > 0
427             ? queryTimeoutMillis
428             : TimeUnit.SECONDS.toMillis(DEFAULT_OPTIONS.timeout());
429         this.resolvedAddressTypes = resolvedAddressTypes != null ? resolvedAddressTypes : DEFAULT_RESOLVE_ADDRESS_TYPES;
430         this.recursionDesired = recursionDesired;
431         this.maxQueriesPerResolve = maxQueriesPerResolve > 0 ? maxQueriesPerResolve : DEFAULT_OPTIONS.attempts();
432         this.maxPayloadSize = checkPositive(maxPayloadSize, "maxPayloadSize");
433         this.optResourceEnabled = optResourceEnabled;
434         this.hostsFileEntriesResolver = checkNotNull(hostsFileEntriesResolver, "hostsFileEntriesResolver");
435         this.dnsServerAddressStreamProvider =
436                 checkNotNull(dnsServerAddressStreamProvider, "dnsServerAddressStreamProvider");
437         this.resolveCache = checkNotNull(resolveCache, "resolveCache");
438         this.cnameCache = checkNotNull(cnameCache, "cnameCache");
439         this.dnsQueryLifecycleObserverFactory = traceEnabled ?
440                 dnsQueryLifecycleObserverFactory instanceof NoopDnsQueryLifecycleObserverFactory ?
441                         new LoggingDnsQueryLifeCycleObserverFactory() :
442                         new BiDnsQueryLifecycleObserverFactory(new LoggingDnsQueryLifeCycleObserverFactory(),
443                                                                dnsQueryLifecycleObserverFactory) :
444                 checkNotNull(dnsQueryLifecycleObserverFactory, "dnsQueryLifecycleObserverFactory");
445         this.searchDomains = searchDomains != null ? searchDomains.clone() : DEFAULT_SEARCH_DOMAINS;
446         this.ndots = ndots >= 0 ? ndots : DEFAULT_OPTIONS.ndots();
447         this.decodeIdn = decodeIdn;
448         this.completeOncePreferredResolved = completeOncePreferredResolved;
449         this.socketChannelFactory = socketChannelFactory;
450         switch (this.resolvedAddressTypes) {
451             case IPV4_ONLY:
452                 supportsAAAARecords = false;
453                 supportsARecords = true;
454                 resolveRecordTypes = IPV4_ONLY_RESOLVED_RECORD_TYPES;
455                 resolvedInternetProtocolFamilies = IPV4_ONLY_RESOLVED_PROTOCOL_FAMILIES;
456                 break;
457             case IPV4_PREFERRED:
458                 supportsAAAARecords = true;
459                 supportsARecords = true;
460                 resolveRecordTypes = IPV4_PREFERRED_RESOLVED_RECORD_TYPES;
461                 resolvedInternetProtocolFamilies = IPV4_PREFERRED_RESOLVED_PROTOCOL_FAMILIES;
462                 break;
463             case IPV6_ONLY:
464                 supportsAAAARecords = true;
465                 supportsARecords = false;
466                 resolveRecordTypes = IPV6_ONLY_RESOLVED_RECORD_TYPES;
467                 resolvedInternetProtocolFamilies = IPV6_ONLY_RESOLVED_PROTOCOL_FAMILIES;
468                 break;
469             case IPV6_PREFERRED:
470                 supportsAAAARecords = true;
471                 supportsARecords = true;
472                 resolveRecordTypes = IPV6_PREFERRED_RESOLVED_RECORD_TYPES;
473                 resolvedInternetProtocolFamilies = IPV6_PREFERRED_RESOLVED_PROTOCOL_FAMILIES;
474                 break;
475             default:
476                 throw new IllegalArgumentException("Unknown ResolvedAddressTypes " + resolvedAddressTypes);
477         }
478         preferredAddressType = preferredAddressType(this.resolvedAddressTypes);
479         this.authoritativeDnsServerCache = checkNotNull(authoritativeDnsServerCache, "authoritativeDnsServerCache");
480         nameServerComparator = new NameServerComparator(preferredAddressType.addressType());
481 
482         Bootstrap b = new Bootstrap();
483         b.group(executor());
484         b.channelFactory(channelFactory);
485         b.option(ChannelOption.DATAGRAM_CHANNEL_ACTIVE_ON_REGISTRATION, true);
486         final DnsResponseHandler responseHandler = new DnsResponseHandler(executor().<Channel>newPromise());
487         b.handler(new ChannelInitializer<DatagramChannel>() {
488             @Override
489             protected void initChannel(DatagramChannel ch) {
490                 ch.pipeline().addLast(DATAGRAM_ENCODER, DATAGRAM_DECODER, responseHandler);
491             }
492         });
493 
494         channelFuture = responseHandler.channelActivePromise;
495         final ChannelFuture future;
496         if (localAddress == null) {
497             future = b.register();
498         } else {
499             future = b.bind(localAddress);
500         }
501         Throwable cause = future.cause();
502         if (cause != null) {
503             if (cause instanceof RuntimeException) {
504                 throw (RuntimeException) cause;
505             }
506             if (cause instanceof Error) {
507                 throw (Error) cause;
508             }
509             throw new IllegalStateException("Unable to create / register Channel", cause);
510         }
511         ch = future.channel();
512         ch.config().setRecvByteBufAllocator(new FixedRecvByteBufAllocator(maxPayloadSize));
513 
514         ch.closeFuture().addListener(new ChannelFutureListener() {
515             @Override
516             public void operationComplete(ChannelFuture future) {
517                 resolveCache.clear();
518                 cnameCache.clear();
519                 authoritativeDnsServerCache.clear();
520             }
521         });
522     }

524     static InternetProtocolFamily preferredAddressType(ResolvedAddressTypes resolvedAddressTypes) {
525         switch (resolvedAddressTypes) {
526         case IPV4_ONLY:
527         case IPV4_PREFERRED:
528             return InternetProtocolFamily.IPv4;
529         case IPV6_ONLY:
530         case IPV6_PREFERRED:
531             return InternetProtocolFamily.IPv6;
532         default:
533             throw new IllegalArgumentException("Unknown ResolvedAddressTypes " + resolvedAddressTypes);
534         }
535     }

546     /**
547      * Creates a new {@link DnsServerAddressStream} to following a redirected DNS query. By overriding this
548      * it provides the opportunity to sort the name servers before following a redirected DNS query.
549      *
550      * @param hostname the hostname.
551      * @param nameservers The addresses of the DNS servers which are used in the event of a redirect. This may
552      *                    contain resolved and unresolved addresses so the used {@link DnsServerAddressStream} must
553      *                    allow unresolved addresses if you want to include these as well.
554      * @return A {@link DnsServerAddressStream} which will be used to follow the DNS redirect or {@code null} if
555      *         none should be followed.

558             @SuppressWarnings("unused") String hostname, List<InetSocketAddress> nameservers) {
559         DnsServerAddressStream cached = authoritativeDnsServerCache().get(hostname);

561             // If there is no cache hit (which may be the case for example when a NoopAuthoritativeDnsServerCache
562             // is used), we will just directly use the provided nameservers.
563             Collections.sort(nameservers, nameServerComparator);
564             return new SequentialDnsServerAddressStream(nameservers, 0);

836      * If the specified {@link DnsQuestion} is {@code A} or {@code AAAA}, this method looks up the configured

839      *

841      * @param additionals additional records ({@code OPT})

843      *
844      * @return the list of the {@link DnsRecord}s as the result of the resolution

852     private Future<List<DnsRecord>> resolveAll(DnsQuestion question, DnsRecord[] additionals,
853                                                Promise<List<DnsRecord>> promise) {
854         checkNotNull(question, "question");
855         checkNotNull(promise, "promise");
856 
857         // Respect /etc/hosts as well if the record type is A or AAAA.
858         final DnsRecordType type = question.type();
859         final String hostname = question.name();
860 
861         if (type == DnsRecordType.A || type == DnsRecordType.AAAA) {
862             final List<InetAddress> hostsFileEntries = resolveHostsFileEntries(hostname);
863             if (hostsFileEntries != null) {
864                 List<DnsRecord> result = new ArrayList<DnsRecord>();
865                 for (InetAddress hostsFileEntry : hostsFileEntries) {
866                     ByteBuf content = null;
867                     if (hostsFileEntry instanceof Inet4Address) {
868                         if (type == DnsRecordType.A) {
869                             content = Unpooled.wrappedBuffer(hostsFileEntry.getAddress());
870                         }
871                     } else if (hostsFileEntry instanceof Inet6Address) {
872                         if (type == DnsRecordType.AAAA) {
873                             content = Unpooled.wrappedBuffer(hostsFileEntry.getAddress());
874                         }
875                     }
876                     if (content != null) {
877                         // Our current implementation does not support reloading the hosts file,
878                         // so use a fairly large TTL (1 day, i.e. 86400 seconds).
879                         result.add(new DefaultDnsRawRecord(hostname, type, 86400, content));
880                     }
881                 }
882 
883                 if (!result.isEmpty()) {
884                     trySuccess(promise, result);
885                     return promise;
886                 }
887             }
888         }
889 
890         // It was not A/AAAA question or there was no entry in /etc/hosts.
891         final DnsServerAddressStream nameServerAddrs =
892                 dnsServerAddressStreamProvider.nameServerAddressStream(hostname);
893         new DnsRecordResolveContext(this, promise, question, additionals, nameServerAddrs, maxQueriesPerResolve)
894                 .resolve(promise);
895         return promise;
896     }

898     private static DnsRecord[] toArray(Iterable<DnsRecord> additionals, boolean validateType) {
899         checkNotNull(additionals, "additionals");
900         if (additionals instanceof Collection) {
901             Collection<DnsRecord> records = (Collection<DnsRecord>) additionals;
902             for (DnsRecord r: additionals) {
903                 validateAdditional(r, validateType);
904             }
905             return records.toArray(new DnsRecord[records.size()]);
906         }
907 
908         Iterator<DnsRecord> additionalsIt = additionals.iterator();
909         if (!additionalsIt.hasNext()) {
910             return EMPTY_ADDITIONALS;
911         }
912         List<DnsRecord> records = new ArrayList<DnsRecord>();
913         do {
914             DnsRecord r = additionalsIt.next();
915             validateAdditional(r, validateType);
916             records.add(r);
917         } while (additionalsIt.hasNext());
918 
919         return records.toArray(new DnsRecord[records.size()]);
920     }

922     private static void validateAdditional(DnsRecord record, boolean validateType) {
923         checkNotNull(record, "record");
924         if (validateType && record instanceof DnsRawRecord) {
925             throw new IllegalArgumentException("DnsRawRecord implementations not allowed: " + record);
926         }
927     }

933     /**
934      * Hook designed for extensibility so one can pass a different cache on each resolution attempt
935      * instead of using the global one.
936      */
937     protected void doResolve(String inetHost,
938                              DnsRecord[] additionals,
939                              Promise<InetAddress> promise,
940                              DnsCache resolveCache) throws Exception {
941         if (inetHost == null || inetHost.isEmpty()) {
942             // If an empty hostname is used we should use "localhost", just like InetAddress.getByName(...) does.
943             promise.setSuccess(loopbackAddress());
944             return;
945         }
946         final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(inetHost);
947         if (bytes != null) {
948             // The inetHost is actually an ipaddress.
949             promise.setSuccess(InetAddress.getByAddress(bytes));
950             return;
951         }
952 
953         final String hostname = hostname(inetHost);
954 
955         InetAddress hostsFileEntry = resolveHostsFileEntry(hostname);
956         if (hostsFileEntry != null) {
957             promise.setSuccess(hostsFileEntry);
958             return;
959         }
960 
961         if (!doResolveCached(hostname, additionals, promise, resolveCache)) {
962             doResolveUncached(hostname, additionals, promise, resolveCache, true);
963         }
964     }

966     private boolean doResolveCached(String hostname,
967                                     DnsRecord[] additionals,
968                                     Promise<InetAddress> promise,
969                                     DnsCache resolveCache) {
970         final List<? extends DnsCacheEntry> cachedEntries = resolveCache.get(hostname, additionals);
971         if (cachedEntries == null || cachedEntries.isEmpty()) {
972             return false;
973         }
974 
975         Throwable cause = cachedEntries.get(0).cause();
976         if (cause == null) {
977             final int numEntries = cachedEntries.size();
978             // Find the first entry with the preferred address type.
979             for (InternetProtocolFamily f : resolvedInternetProtocolFamilies) {
980                 for (int i = 0; i < numEntries; i++) {
981                     final DnsCacheEntry e = cachedEntries.get(i);
982                     if (f.addressType().isInstance(e.address())) {
983                         trySuccess(promise, e.address());
984                         return true;
985                     }
986                 }
987             }
988             return false;
989         } else {
990             tryFailure(promise, cause);
991             return true;
992         }
993     }

995     static <T> boolean trySuccess(Promise<T> promise, T result) {
996         final boolean notifiedRecords = promise.trySuccess(result);
997         if (!notifiedRecords) {
998             // There is nothing really wrong with not be able to notify the promise as we may have raced here because
999             // of multiple queries that have been executed. Log it with trace level anyway just in case the user
1000             // wants to better understand what happened.
1001             logger.trace("Failed to notify success ({}) to a promise: {}", result, promise);
1002         }
1003         return notifiedRecords;
1004     }
1005 
1006     private static void tryFailure(Promise<?> promise, Throwable cause) {
1007         if (!promise.tryFailure(cause)) {
1008             // There is nothing really wrong with not be able to notify the promise as we may have raced here because
1009             // of multiple queries that have been executed. Log it with trace level anyway just in case the user
1010             // wants to better understand what happened.
1011             logger.trace("Failed to notify failure to a promise: {}", promise, cause);
1012         }
1013     }

1017                                    final Promise<InetAddress> promise,

1044                                 Promise<List<InetAddress>> promise,
1045                                 DnsCache resolveCache) throws Exception {

1052         if (bytes != null) {

1054             promise.setSuccess(Collections.singletonList(InetAddress.getByAddress(bytes)));
1055             return;

1061         if (hostsFileEntries != null) {
1062             promise.setSuccess(hostsFileEntries);
1063             return;
1064         }

1066         if (!doResolveAllCached(hostname, additionals, promise, resolveCache, resolvedInternetProtocolFamilies)) {
1067             doResolveAllUncached(hostname, additionals, promise, promise,
1068                                  resolveCache, completeOncePreferredResolved);

1072     static boolean doResolveAllCached(String hostname,
1073                                       DnsRecord[] additionals,
1074                                       Promise<List<InetAddress>> promise,
1075                                       DnsCache resolveCache,
1076                                       InternetProtocolFamily[] resolvedInternetProtocolFamilies) {
1077         final List<? extends DnsCacheEntry> cachedEntries = resolveCache.get(hostname, additionals);
1078         if (cachedEntries == null || cachedEntries.isEmpty()) {
1079             return false;
1080         }
1081 
1082         Throwable cause = cachedEntries.get(0).cause();
1083         if (cause == null) {
1084             List<InetAddress> result = null;
1085             final int numEntries = cachedEntries.size();
1086             for (InternetProtocolFamily f : resolvedInternetProtocolFamilies) {
1087                 for (int i = 0; i < numEntries; i++) {
1088                     final DnsCacheEntry e = cachedEntries.get(i);
1089                     if (f.addressType().isInstance(e.address())) {
1090                         if (result == null) {
1091                             result = new ArrayList<InetAddress>(numEntries);
1092                         }
1093                         result.add(e.address());
1094                     }
1095                 }
1096             }
1097             if (result != null) {
1098                 trySuccess(promise, result);
1099                 return true;
1100             }
1101             return false;
1102         } else {
1103             tryFailure(promise, cause);
1104             return true;
1105         }
1106     }

1283         @Override
1284         public void channelRead(ChannelHandlerContext ctx, Object msg) {
1285             final DatagramDnsResponse res = (DatagramDnsResponse) msg;
1286             final int queryId = res.id();
1287 
1288             if (logger.isDebugEnabled()) {
1289                 logger.debug("{} RECEIVED: UDP [{}: {}], {}", ch, queryId, res.sender(), res);
1290             }
1291 
1292             final DnsQueryContext qCtx = queryContextManager.get(res.sender(), queryId);
1293             if (qCtx == null) {
1294                 logger.debug("Received a DNS response with an unknown ID: UDP [{}: {}]", ch, queryId);
1295                 res.release();
1296                 return;
1297             }
1298 
1299             // Check if the response was truncated and if we can fallback to TCP to retry.
1300             if (!res.isTruncated() || socketChannelFactory == null) {
1301                 qCtx.finish(res);
1302                 return;
1303             }
1304 
1305             Bootstrap bs = new Bootstrap();
1306             bs.option(ChannelOption.SO_REUSEADDR, true)
1307             .group(executor())
1308             .channelFactory(socketChannelFactory)
1309             .handler(TCP_ENCODER);
1310             bs.connect(res.sender()).addListener(new ChannelFutureListener() {
1311                 @Override
1312                 public void operationComplete(ChannelFuture future) {
1313                     if (!future.isSuccess()) {
1314                         if (logger.isDebugEnabled()) {
1315                             logger.debug("Unable to fallback to TCP [{}]", queryId, future.cause());
1316                         }
1317 
1318                         // TCP fallback failed, just use the truncated response.
1319                         qCtx.finish(res);
1320                         return;
1321                     }
1322                     final Channel channel = future.channel();
1323 
1324                     Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise =
1325                             channel.eventLoop().newPromise();
1326                     final TcpDnsQueryContext tcpCtx = new TcpDnsQueryContext(DnsNameResolver.this, channel,
1327                             (InetSocketAddress) channel.remoteAddress(), qCtx.question(),
1328                             EMPTY_ADDITIONALS, promise);
1329 
1330                     channel.pipeline().addLast(new TcpDnsResponseDecoder());
1331                     channel.pipeline().addLast(new ChannelInboundHandlerAdapter() {
1332                         @Override
1333                         public void channelRead(ChannelHandlerContext ctx, Object msg) {
1334                             Channel channel = ctx.channel();
1335                             DnsResponse response = (DnsResponse) msg;
1336                             int queryId = response.id();
1337 
1338                             if (logger.isDebugEnabled()) {
1339                                 logger.debug("{} RECEIVED: TCP [{}: {}], {}", channel, queryId,
1340                                         channel.remoteAddress(), response);
1341                             }
1342 
1343                             DnsQueryContext foundCtx = queryContextManager.get(res.sender(), queryId);
1344                             if (foundCtx == tcpCtx) {
1345                                 tcpCtx.finish(new AddressedEnvelopeAdapter(
1346                                         (InetSocketAddress) ctx.channel().remoteAddress(),
1347                                         (InetSocketAddress) ctx.channel().localAddress(),
1348                                         response));
1349                             } else {
1350                                 response.release();

1353                                         channel, queryId);
1354                             }
1355                         }

1357                         @Override

1359                             if (tcpCtx.tryFailure("TCP fallback error", cause, false) && logger.isDebugEnabled()) {

1361                                         ctx.channel(), queryId,
1362                                         ctx.channel().remoteAddress(), cause);
1363                             }
1364                         }
1365                     });
1366 
1367                     promise.addListener(
1368                             new FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>>() {
1369                         @Override

1371                                 Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
1372                             channel.close();
1373 
1374                             if (future.isSuccess()) {
1375                                 qCtx.finish(future.getNow());
1376                                 res.release();
1377                             } else {
1378                                 // TCP fallback failed, just use the truncated response.
1379                                 qCtx.finish(res);
1380                             }
1381                         }
1382                     });
1383                     tcpCtx.query(true, future.channel().newPromise());
1384                 }
1385             });
1386         }
