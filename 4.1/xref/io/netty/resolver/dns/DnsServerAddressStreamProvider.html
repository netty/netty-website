16  package io.netty.resolver.dns;
17  
18  /**
19   * Provides an opportunity to override which {@link DnsServerAddressStream} is used to resolve a specific hostname.
20   * <p>
21   * For example this can be used to represent <a href="https://linux.die.net/man/5/resolver">/etc/resolv.conf</a> and
22   * <a href="https://developer.apple.com/legacy/library/documentation/Darwin/Reference/ManPages/man5/resolver.5.html">
23   * /etc/resolver</a>.
24   */
25  public interface DnsServerAddressStreamProvider {
26      /**
27       * Ask this provider for the name servers to query for {@code hostname}.
28       * @param hostname The hostname for which to lookup the DNS server addressed to use.
29       *                 If this is the final {@link DnsServerAddressStreamProvider} to be queried then generally empty
30       *                 string or {@code '.'} correspond to the default {@link DnsServerAddressStream}.
31       * @return The {@link DnsServerAddressStream} which should be used to resolve {@code hostname}.
32       */
33      DnsServerAddressStream nameServerAddressStream(String hostname);
34  }
