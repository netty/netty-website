16  package io.netty.resolver.dns;
17  
18  import static io.netty.resolver.dns.DnsAddressDecoder.decodeAddress;
19  
20  import java.net.InetAddress;
21  import java.net.UnknownHostException;
22  import java.util.Collections;
23  import java.util.List;
24  
25  import io.netty.channel.EventLoop;
26  import io.netty.handler.codec.dns.DnsRecord;
27  import io.netty.handler.codec.dns.DnsRecordType;
28  import io.netty.util.concurrent.Promise;
29  
30  final class DnsAddressResolveContext extends DnsResolveContext<InetAddress> {
31  
32      private final DnsCache resolveCache;
33      private final AuthoritativeDnsServerCache authoritativeDnsServerCache;
34      private final boolean completeEarlyIfPossible;
35  
36      DnsAddressResolveContext(DnsNameResolver parent, String hostname, DnsRecord[] additionals,
37                               DnsServerAddressStream nameServerAddrs, DnsCache resolveCache,
38                               AuthoritativeDnsServerCache authoritativeDnsServerCache,
39                               boolean completeEarlyIfPossible) {
40          super(parent, hostname, DnsRecord.CLASS_IN, parent.resolveRecordTypes(), additionals, nameServerAddrs);
41          this.resolveCache = resolveCache;
42          this.authoritativeDnsServerCache = authoritativeDnsServerCache;
43          this.completeEarlyIfPossible = completeEarlyIfPossible;
44      }
45  
46      @Override
47      DnsResolveContext<InetAddress> newResolverContext(DnsNameResolver parent, String hostname,
48                                                        int dnsClass, DnsRecordType[] expectedTypes,
49                                                        DnsRecord[] additionals,
50                                                        DnsServerAddressStream nameServerAddrs) {
51          return new DnsAddressResolveContext(parent, hostname, additionals, nameServerAddrs, resolveCache,
52                  authoritativeDnsServerCache, completeEarlyIfPossible);
53      }
54  
55      @Override
56      InetAddress convertRecord(DnsRecord record, String hostname, DnsRecord[] additionals, EventLoop eventLoop) {
57          return decodeAddress(record, hostname, parent.isDecodeIdn());
58      }
59  
60      @Override
61      List<InetAddress> filterResults(List<InetAddress> unfiltered) {
62          Collections.sort(unfiltered, PreferredAddressTypeComparator.comparator(parent.preferredAddressType()));
63          return unfiltered;
64      }
65  
66      @Override
67      boolean isCompleteEarly(InetAddress resolved) {
68          return completeEarlyIfPossible && parent.preferredAddressType().addressType() == resolved.getClass();
69      }
70  
71      @Override
72      void cache(String hostname, DnsRecord[] additionals,
73                 DnsRecord result, InetAddress convertedResult) {
74          resolveCache.cache(hostname, additionals, convertedResult, result.timeToLive(), parent.ch.eventLoop());
75      }
76  
77      @Override
78      void cache(String hostname, DnsRecord[] additionals, UnknownHostException cause) {
79          resolveCache.cache(hostname, additionals, cause, parent.ch.eventLoop());
80      }
81  
82      @Override
83      void doSearchDomainQuery(String hostname, Promise<List<InetAddress>> nextPromise) {
84          // Query the cache for the hostname first and only do a query if we could not find it in the cache.
85          if (!DnsNameResolver.doResolveAllCached(
86                  hostname, additionals, nextPromise, resolveCache, parent.resolvedInternetProtocolFamiliesUnsafe())) {
87              super.doSearchDomainQuery(hostname, nextPromise);
88          }
89      }
90  
91      @Override
92      DnsCache resolveCache() {
93          return resolveCache;
94      }
95  
96      @Override
97      AuthoritativeDnsServerCache authoritativeDnsServerCache() {
98          return authoritativeDnsServerCache;
99      }
100 }
