47      private static final InternalLogger logger = InternalLoggerFactory.getInstance(DnsQueryContext.class);

49      private final DnsNameResolver parent;
50      private final Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise;
51      private final int id;
52      private final DnsQuestion question;
53      private final DnsRecord optResource;
54      private final Iterator<InetSocketAddress> nameServerAddresses;

56      private final boolean recursionDesired;
57      private final int maxTries;
58      private int remainingTries;
59      private volatile ScheduledFuture<?> timeoutFuture;

61  
62      DnsQueryContext(DnsNameResolver parent,
63                      Iterable<InetSocketAddress> nameServerAddresses,
64                      DnsQuestion question, Promise<AddressedEnvelope<DnsResponse, InetSocketAddress>> promise) {
65  
66          this.parent = parent;
67          this.promise = promise;
68          this.question = question;
69  
70          id = allocateId();
71          recursionDesired = parent.isRecursionDesired();
72          maxTries = parent.maxTriesPerQuery();
73          remainingTries = maxTries;
74          optResource = new DefaultDnsRawRecord(
75                  StringUtil.EMPTY_STRING, DnsRecordType.OPT, parent.maxPayloadSize(), 0, Unpooled.EMPTY_BUFFER);
76  
77          this.nameServerAddresses = nameServerAddresses.iterator();
78      }
79  
80      private int allocateId() {
81          int id = ThreadLocalRandom.current().nextInt(parent.promises.length());
82          final int maxTries = parent.promises.length() << 1;
83          int tries = 0;
84          for (;;) {
85              if (parent.promises.compareAndSet(id, null, this)) {
86                  return id;
87              }
88  
89              id = id + 1 & 0xFFFF;
90  
91              if (++ tries >= maxTries) {
92                  throw new IllegalStateException("query ID space exhausted: " + question);
93              }
94          }
95      }

109     void query() {
110         final DnsQuestion question = this.question;
111 
112         if (remainingTries <= 0 || !nameServerAddresses.hasNext()) {
113             parent.promises.set(id, null);
114 
115             int tries = maxTries - remainingTries;
116             UnknownHostException cause;
117             if (tries > 1) {
118                 cause = new UnknownHostException(
119                         "failed to resolve " + question + " after " + tries + " attempts:" +
120                         trace);
121             } else {
122                 cause = new UnknownHostException("failed to resolve " + question + ':' + trace);
123             }
124 
125             cache(question, cause);
126             promise.tryFailure(cause);
127             return;
128         }
129 
130         remainingTries --;
131 
132         final InetSocketAddress nameServerAddr = nameServerAddresses.next();
133         final DatagramDnsQuery query = new DatagramDnsQuery(null, nameServerAddr, id);
134         query.setRecursionDesired(recursionDesired);
135         query.setRecord(DnsSection.QUESTION, question);
136         query.setRecord(DnsSection.ADDITIONAL, optResource);
137 
138         if (logger.isDebugEnabled()) {
139             logger.debug("{} WRITE: [{}: {}], {}", parent.ch, id, nameServerAddr, question);
140         }
141 
142         sendQuery(query, nameServerAddr);
143     }
144 
145     private void sendQuery(final DnsQuery query, final InetSocketAddress nameServerAddr) {
146         if (parent.bindFuture.isDone()) {
147             writeQuery(query, nameServerAddr);
148         } else {
149             parent.bindFuture.addListener(new ChannelFutureListener() {
150                 @Override
151                 public void operationComplete(ChannelFuture future) throws Exception {
152                     if (future.isSuccess()) {
153                         writeQuery(query, nameServerAddr);
154                     } else {
155                         promise.tryFailure(future.cause());
156                     }
157                  }
158             });
159         }
160     }
161 
162     private void writeQuery(final DnsQuery query, final InetSocketAddress nameServerAddr) {
163         final ChannelFuture writeFuture = parent.ch.writeAndFlush(query);
164         if (writeFuture.isDone()) {
165             onQueryWriteCompletion(writeFuture, nameServerAddr);
166         } else {
167             writeFuture.addListener(new ChannelFutureListener() {
168                 @Override
169                 public void operationComplete(ChannelFuture future) throws Exception {
170                     onQueryWriteCompletion(writeFuture, nameServerAddr);
171                 }
172             });
173         }
174     }
175 
176     private void onQueryWriteCompletion(ChannelFuture writeFuture, final InetSocketAddress nameServerAddr) {
177         if (!writeFuture.isSuccess()) {
178             retry(nameServerAddr, "failed to send a query: " + writeFuture.cause());
179             return;
180         }
181 
182         // Schedule a query timeout task if necessary.
183         final long queryTimeoutMillis = parent.queryTimeoutMillis();
184         if (queryTimeoutMillis > 0) {
185             timeoutFuture = parent.ch.eventLoop().schedule(new OneTimeTask() {
186                 @Override
187                 public void run() {
188                     if (promise.isDone()) {
189                         // Received a response before the query times out.
190                         return;
191                     }
192 
193                     retry(nameServerAddr, "query timed out after " + queryTimeoutMillis + " milliseconds");
194                 }
195             }, queryTimeoutMillis, TimeUnit.MILLISECONDS);
196         }
197     }
198 
199     void retry(InetSocketAddress nameServerAddr, String message) {
200         if (promise.isCancelled()) {
201             return;
202         }
203 
204         if (trace == null) {
205             trace = new StringBuilder(128);
206         }
207 
208         trace.append(StringUtil.NEWLINE);
209         trace.append("\tfrom ");
210         trace.append(nameServerAddr);
211         trace.append(": ");
212         trace.append(message);
213         query();
214     }
215 
216     private void cache(final DnsQuestion question, Throwable cause) {
217         final int negativeTtl = parent.negativeTtl();
218         if (negativeTtl == 0) {
219             return;
220         }
221 
222         parent.cache(question, new DnsCacheEntry(cause), negativeTtl);
223     }
