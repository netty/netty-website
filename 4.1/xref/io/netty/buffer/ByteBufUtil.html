92      /**
93       * Returns a <a href="http://en.wikipedia.org/wiki/Hex_dump">hex dump</a>
94       * of the specified buffer's sub-region.
95       */
96      public static String hexDump(ByteBuf buffer, int fromIndex, int length) {
97          if (length < 0) {
98              throw new IllegalArgumentException("length: " + length);
99          }
100         if (length == 0) {
101             return "";
102         }
103 
104         int endIndex = fromIndex + length;
105         char[] buf = new char[length << 1];
106 
107         int srcIdx = fromIndex;
108         int dstIdx = 0;
109         for (; srcIdx < endIndex; srcIdx ++, dstIdx += 2) {
110             System.arraycopy(
111                     HEXDUMP_TABLE, buffer.getUnsignedByte(srcIdx) << 1,
112                     buf, dstIdx, 2);
113         }
114 
115         return new String(buf);
116     }

150     /**
151      * Calculates the hash code of the specified buffer.  This method is
152      * useful when implementing a new buffer type.
153      */
154     public static int hashCode(ByteBuf buffer) {
155         final int aLen = buffer.readableBytes();
156         final int intCount = aLen >>> 2;
157         final int byteCount = aLen & 3;
158 
159         int hashCode = 1;
160         int arrayIndex = buffer.readerIndex();
161         if (buffer.order() == ByteOrder.BIG_ENDIAN) {
162             for (int i = intCount; i > 0; i --) {
163                 hashCode = 31 * hashCode + buffer.getInt(arrayIndex);
164                 arrayIndex += 4;
165             }
166         } else {
167             for (int i = intCount; i > 0; i --) {
168                 hashCode = 31 * hashCode + swapInt(buffer.getInt(arrayIndex));
169                 arrayIndex += 4;
170             }
171         }
172 
173         for (int i = byteCount; i > 0; i --) {
174             hashCode = 31 * hashCode + buffer.getByte(arrayIndex ++);
175         }
176 
177         if (hashCode == 0) {
178             hashCode = 1;
179         }
180 
181         return hashCode;
182     }

184     /**
185      * Returns {@code true} if and only if the two specified buffers are
186      * identical to each other for {@code length} bytes starting at {@code aStartIndex}
187      * index for the {@code a} buffer and {@code bStartIndex} index for the {@code b} buffer.
188      * A more compact way to express this is:
189      * <p>
190      * {@code a[aStartIndex : aStartIndex + length] == b[bStartIndex : bStartIndex + length]}
191      */
192     public static boolean equals(ByteBuf a, int aStartIndex, ByteBuf b, int bStartIndex, int length) {
193         if (aStartIndex < 0 || bStartIndex < 0 || length < 0) {
194             throw new IllegalArgumentException("All indexes and lengths must be non-negative");
195         }
196         if (a.writerIndex() - length < aStartIndex || b.writerIndex() - length < bStartIndex) {
197             return false;
198         }
199 
200         final int longCount = length >>> 3;
201         final int byteCount = length & 7;
202 
203         if (a.order() == b.order()) {
204             for (int i = longCount; i > 0; i --) {
205                 if (a.getLong(aStartIndex) != b.getLong(bStartIndex)) {
206                     return false;
207                 }
208                 aStartIndex += 8;
209                 bStartIndex += 8;
210             }
211         } else {
212             for (int i = longCount; i > 0; i --) {
213                 if (a.getLong(aStartIndex) != swapLong(b.getLong(bStartIndex))) {
214                     return false;
215                 }
216                 aStartIndex += 8;
217                 bStartIndex += 8;
218             }
219         }
220 
221         for (int i = byteCount; i > 0; i --) {
222             if (a.getByte(aStartIndex) != b.getByte(bStartIndex)) {
223                 return false;
224             }
225             aStartIndex ++;
226             bStartIndex ++;
227         }
228 
229         return true;
230     }

245     /**
246      * Compares the two specified buffers as described in {@link ByteBuf#compareTo(ByteBuf)}.
247      * This method is useful when implementing a new buffer type.
248      */
249     public static int compare(ByteBuf bufferA, ByteBuf bufferB) {
250         final int aLen = bufferA.readableBytes();
251         final int bLen = bufferB.readableBytes();
252         final int minLength = Math.min(aLen, bLen);
253         final int uintCount = minLength >>> 2;
254         final int byteCount = minLength & 3;
255 
256         int aIndex = bufferA.readerIndex();
257         int bIndex = bufferB.readerIndex();
258 
259         if (bufferA.order() == bufferB.order()) {
260             for (int i = uintCount; i > 0; i --) {
261                 long va = bufferA.getUnsignedInt(aIndex);
262                 long vb = bufferB.getUnsignedInt(bIndex);
263                 if (va > vb) {
264                     return 1;
265                 }
266                 if (va < vb) {
267                     return -1;
268                 }
269                 aIndex += 4;
270                 bIndex += 4;
271             }
272         } else {
273             for (int i = uintCount; i > 0; i --) {
274                 long va = bufferA.getUnsignedInt(aIndex);
275                 long vb = swapInt(bufferB.getInt(bIndex)) & 0xFFFFFFFFL;
276                 if (va > vb) {
277                     return 1;
278                 }
279                 if (va < vb) {
280                     return -1;
281                 }
282                 aIndex += 4;
283                 bIndex += 4;
284             }
285         }
286 
287         for (int i = byteCount; i > 0; i --) {
288             short va = bufferA.getUnsignedByte(aIndex);
289             short vb = bufferB.getUnsignedByte(bIndex);
290             if (va > vb) {
291                 return 1;
292             }
293             if (va < vb) {
294                 return -1;
295             }
296             aIndex ++;
297             bIndex ++;
298         }
299 
300         return aLen - bLen;
301     }

323      * Toggles the endianness of the specified 24-bit medium integer.
324      */
325     public static int swapMedium(int value) {
326         int swapped = value << 16 & 0xff0000 | value & 0xff00 | value >>> 16 & 0xff;
327         if ((swapped & 0x800000) != 0) {
328             swapped |= 0xff000000;
329         }
330         return swapped;

364     private static int firstIndexOf(ByteBuf buffer, int fromIndex, int toIndex, byte value) {
365         fromIndex = Math.max(fromIndex, 0);
366         if (fromIndex >= toIndex || buffer.capacity() == 0) {
367             return -1;
368         }
369 
370         return buffer.forEachByte(fromIndex, toIndex - fromIndex, new ByteProcessor.IndexOfProcessor(value));
371     }
372 
373     private static int lastIndexOf(ByteBuf buffer, int fromIndex, int toIndex, byte value) {
374         fromIndex = Math.min(fromIndex, buffer.capacity());
375         if (fromIndex < 0 || buffer.capacity() == 0) {
376             return -1;
377         }
378 
379         return buffer.forEachByteDesc(toIndex, fromIndex - toIndex, new ByteProcessor.IndexOfProcessor(value));
380     }

382     /**
383      * Encode a {@link CharSequence} in <a href="http://en.wikipedia.org/wiki/UTF-8">UTF-8</a> and write
384      * it to a {@link ByteBuf}.
385      *
386      * This method returns the actual number of bytes written.
387      */
388     public static int writeUtf8(ByteBuf buf, CharSequence seq) {
389         if (buf == null) {
390             throw new NullPointerException("buf");
391         }
392         if (seq == null) {
393             throw new NullPointerException("seq");
394         }
395         // UTF-8 uses max. 3 bytes per char, so calculate the worst case.
396         final int len = seq.length();
397         final int maxSize = len * 3;
398         buf.ensureWritable(maxSize);
399         if (buf instanceof AbstractByteBuf) {
400             // Fast-Path
401             AbstractByteBuf buffer = (AbstractByteBuf) buf;
402             int oldWriterIndex = buffer.writerIndex;
403             int writerIndex = oldWriterIndex;
404 
405             // We can use the _set methods as these not need to do any index checks and reference checks.
406             // This is possible as we called ensureWritable(...) before.
407             for (int i = 0; i < len; i++) {
408                 char c = seq.charAt(i);
409                 if (c < 0x80) {
410                     buffer._setByte(writerIndex++, (byte) c);
411                 } else if (c < 0x800) {
412                     buffer._setByte(writerIndex++, (byte) (0xc0 | (c >> 6)));
413                     buffer._setByte(writerIndex++, (byte) (0x80 | (c & 0x3f)));
414                 } else {
415                     buffer._setByte(writerIndex++, (byte) (0xe0 | (c >> 12)));
416                     buffer._setByte(writerIndex++, (byte) (0x80 | ((c >> 6) & 0x3f)));
417                     buffer._setByte(writerIndex++, (byte) (0x80 | (c & 0x3f)));
418                 }
419             }
420             // update the writerIndex without any extra checks for performance reasons
421             buffer.writerIndex = writerIndex;
422             return writerIndex - oldWriterIndex;
423         } else {
424             // Maybe we could also check if we can unwrap() to access the wrapped buffer which
425             // may be an AbstractByteBuf. But this may be overkill so let us keep it simple for now.
426             byte[] bytes = seq.toString().getBytes(CharsetUtil.UTF_8);
427             buf.writeBytes(bytes);
428             return bytes.length;
429         }
430     }

476     static ByteBuf encodeString0(ByteBufAllocator alloc, boolean enforceHeap, CharBuffer src, Charset charset) {
477         final CharsetEncoder encoder = CharsetUtil.getEncoder(charset);
478         int length = (int) ((double) src.remaining() * encoder.maxBytesPerChar());
479         boolean release = true;
480         final ByteBuf dst;
481         if (enforceHeap) {
482             dst = alloc.heapBuffer(length);
483         } else {
484             dst = alloc.buffer(length);
485         }
486         try {
487             final ByteBuffer dstBuf = dst.internalNioBuffer(0, length);
488             final int pos = dstBuf.position();
489             CoderResult cr = encoder.encode(src, dstBuf, true);
490             if (!cr.isUnderflow()) {
491                 cr.throwException();
492             }
493             cr = encoder.flush(dstBuf);
494             if (!cr.isUnderflow()) {
495                 cr.throwException();
496             }
497             dst.writerIndex(dst.writerIndex() + dstBuf.position() - pos);
498             release = false;
499             return dst;
500         } catch (CharacterCodingException x) {
501             throw new IllegalStateException(x);
502         } finally {
503             if (release) {
504                 dst.release();
505             }
506         }
507     }

509     static String decodeString(ByteBuffer src, Charset charset) {
510         final CharsetDecoder decoder = CharsetUtil.getDecoder(charset);
511         final CharBuffer dst = CharBuffer.allocate(
512                 (int) ((double) src.remaining() * decoder.maxCharsPerByte()));
513         try {
514             CoderResult cr = decoder.decode(src, dst, true);
515             if (!cr.isUnderflow()) {
516                 cr.throwException();
517             }
518             cr = decoder.flush(dst);
519             if (!cr.isUnderflow()) {
520                 cr.throwException();
521             }
522         } catch (CharacterCodingException x) {
523             throw new IllegalStateException(x);
524         }
525         return dst.flip().toString();
526     }

561     /**
562      * Return an array of the underlying storage from {@link buf} into a byte array.
563      * The copy will start at {@code start} and copy {@code length} bytes.
564      * If {@code copy} is true a copy will be made of the memory.
565      * If {@code copy} is false the underlying storage will be shared, if possible.
566      */
567     public static byte[] getBytes(ByteBuf buf, int start, int length, boolean copy) {
568         if (start < 0 || length > checkNotNull(buf, "buf").capacity() - start) {
569             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
570                     + ") <= " + "buf.capacity(" + buf.capacity() + ')');
571         }
572 
573         if (buf.hasArray()) {
574             if (copy || start != 0 || length != buf.capacity()) {
575                 int baseOffset = buf.arrayOffset() + start;
576                 return Arrays.copyOfRange(buf.array(), baseOffset, baseOffset + length);
577             } else {
578                 return buf.array();
579             }
580         }
581 
582         byte[] v = new byte[length];
583         buf.getBytes(start, v);
584         return v;
585     }
