23  /**
24   * Description of algorithm for PageRun/PoolSubpage allocation from PoolChunk
25   *
26   * Notation: The following terms are important to understand the code
27   * > page  - a page is the smallest unit of memory chunk that can be allocated
28   * > chunk - a chunk is a collection of pages
29   * > in this code chunkSize = 2^{maxOrder} * pageSize
30   *
31   * To begin we allocate a byte array of size = chunkSize
32   * Whenever a ByteBuf of given size needs to be created we search for the first position
33   * in the byte array that has enough empty space to accommodate the requested size and
34   * return a (long) handle that encodes this offset information, (this memory segment is then
35   * marked as reserved so it is always used by exactly one ByteBuf and no more)
36   *
37   * For simplicity all sizes are normalized according to PoolArena#normalizeCapacity method
38   * This ensures that when we request for memory segments of size >= pageSize the normalizedCapacity
39   * equals the next nearest power of 2
40   *
41   * To search for the first offset in chunk that has at least requested size available we construct a
42   * complete balanced binary tree and store it in an array (just like heaps) - memoryMap
43   *
44   * The tree looks like this (the size of each node being mentioned in the parenthesis)
45   *
46   * depth=0        1 node (chunkSize)
47   * depth=1        2 nodes (chunkSize/2)
48   * ..
49   * ..
50   * depth=d        2^d nodes (chunkSize/2^d)
51   * ..
52   * depth=maxOrder 2^maxOrder nodes (chunkSize/2^{maxOrder} = pageSize)
53   *
54   * depth=maxOrder is the last level and the leafs consist of pages
55   *
56   * With this tree available searching in chunkArray translates like this:
57   * To allocate a memory segment of size chunkSize/2^k we search for the first node (from left) at height k
58   * which is unused
59   *
60   * Algorithm:
61   * ----------
62   * Encode the tree in memoryMap with the notation
63   *   memoryMap[id] = x => in the subtree rooted at id, the first node that is free to be allocated
64   *   is at depth x (counted from depth=0) i.e., at depths [depth_of_id, x), there is no node that is free
65   *
66   *  As we allocate & free nodes, we update values stored in memoryMap so that the property is maintained
67   *
68   * Initialization -
69   *   In the beginning we construct the memoryMap array by storing the depth of a node at each node
70   *     i.e., memoryMap[id] = depth_of_id
71   *
72   * Observations:
73   * -------------
74   * 1) memoryMap[id] = depth_of_id  => it is free / unallocated
75   * 2) memoryMap[id] > depth_of_id  => at least one of its child nodes is allocated, so we cannot allocate it, but
76   *                                    some of its children can still be allocated based on their availability
77   * 3) memoryMap[id] = maxOrder + 1 => the node is fully allocated & thus none of its children can be allocated, it
78   *                                    is thus marked as unusable
79   *
80   * Algorithm: [allocateNode(d) => we want to find the first node (from left) at height h that can be allocated]
81   * ----------
82   * 1) start at root (i.e., depth = 0 or id = 1)
83   * 2) if memoryMap[1] > d => cannot be allocated from this chunk
84   * 3) if left node value <= h; we can allocate from left subtree so move to left and repeat until found
85   * 4) else try in right subtree
86   *
87   * Algorithm: [allocateRun(size)]
88   * ----------
89   * 1) Compute d = log_2(chunkSize/size)
90   * 2) Return allocateNode(d)
91   *
92   * Algorithm: [allocateSubpage(size)]
93   * ----------
94   * 1) use allocateNode(maxOrder) to find an empty (i.e., unused) leaf (i.e., page)
95   * 2) use this handle to construct the PoolSubpage object or if it already exists just call init(normCapacity)
96   *    note that this PoolSubpage object is added to subpagesPool in the PoolArena when we init() it
97   *
98   * Note:
99   * -----
100  * In the implementation for improving cache coherence,
101  * we store 2 pieces of information depth_of_id and x as two byte values in memoryMap and depthMap respectively
102  *
103  * memoryMap[id]= depth_of_id  is defined above
104  * depthMap[id]= x  indicates that the first node which is free to be allocated is at depth x (from root)
105  */

144     PoolChunk(PoolArena<T> arena, T memory, int pageSize, int maxOrder, int pageShifts, int chunkSize, int offset) {
145         unpooled = false;
146         this.arena = arena;
147         this.memory = memory;
148         this.pageSize = pageSize;
149         this.pageShifts = pageShifts;
150         this.maxOrder = maxOrder;
151         this.chunkSize = chunkSize;
152         this.offset = offset;
153         unusable = (byte) (maxOrder + 1);
154         log2ChunkSize = log2(chunkSize);
155         subpageOverflowMask = ~(pageSize - 1);
156         freeBytes = chunkSize;
157 
158         assert maxOrder < 30 : "maxOrder should be < 30, but is: " + maxOrder;
159         maxSubpageAllocs = 1 << maxOrder;
160 
161         // Generate the memory map.
162         memoryMap = new byte[maxSubpageAllocs << 1];
163         depthMap = new byte[memoryMap.length];
164         int memoryMapIndex = 1;
165         for (int d = 0; d <= maxOrder; ++ d) { // move down the tree one level at a time
166             int depth = 1 << d;
167             for (int p = 0; p < depth; ++ p) {
168                 // in each level traverse left to right and set value to the depth of subtree
169                 memoryMap[memoryMapIndex] = (byte) d;
170                 depthMap[memoryMapIndex] = (byte) d;
171                 memoryMapIndex ++;
172             }
173         }
174 
175         subpages = newSubpageArray(maxSubpageAllocs);
176         cachedNioBuffers = new ArrayDeque<ByteBuffer>(8);
177     }

213     private int usage(int freeBytes) {
214         if (freeBytes == 0) {
215             return 100;
216         }
217 
218         int freePercentage = (int) (freeBytes * 100L / chunkSize);
219         if (freePercentage == 0) {
220             return 99;
221         }
222         return 100 - freePercentage;

225     boolean allocate(PooledByteBuf<T> buf, int reqCapacity, int normCapacity) {
226         final long handle;
227         if ((normCapacity & subpageOverflowMask) != 0) { // >= pageSize
228             handle =  allocateRun(normCapacity);
229         } else {
230             handle = allocateSubpage(normCapacity);
231         }
232 
233         if (handle < 0) {
234             return false;
235         }
236         ByteBuffer nioBuffer = cachedNioBuffers != null ? cachedNioBuffers.pollLast() : null;
237         initBuf(buf, nioBuffer, handle, reqCapacity);
238         return true;
239     }

241     /**
242      * Update method used by allocate
243      * This is triggered only when a successor is allocated and all its predecessors
244      * need to update their state
245      * The minimal depth at which subtree rooted at id has some free space
246      *
247      * @param id id
248      */
249     private void updateParentsAlloc(int id) {
250         while (id > 1) {
251             int parentId = id >>> 1;
252             byte val1 = value(id);
253             byte val2 = value(id ^ 1);
254             byte val = val1 < val2 ? val1 : val2;
255             setValue(parentId, val);
256             id = parentId;
257         }
258     }
259 
260     /**
261      * Update method used by free
262      * This needs to handle the special case when both children are completely free
263      * in which case parent be directly allocated on request of size = child-size * 2
264      *
265      * @param id id
266      */
267     private void updateParentsFree(int id) {
268         int logChild = depth(id) + 1;
269         while (id > 1) {
270             int parentId = id >>> 1;
271             byte val1 = value(id);
272             byte val2 = value(id ^ 1);
273             logChild -= 1; // in first iteration equals log, subsequently reduce 1 from logChild as we traverse up
274 
275             if (val1 == logChild && val2 == logChild) {
276                 setValue(parentId, (byte) (logChild - 1));
277             } else {
278                 byte val = val1 < val2 ? val1 : val2;
279                 setValue(parentId, val);
280             }
281 
282             id = parentId;
283         }
284     }

286     /**
287      * Algorithm to allocate an index in memoryMap when we query for a free node
288      * at depth d
289      *
290      * @param d depth
291      * @return index in memoryMap
292      */
293     private int allocateNode(int d) {
294         int id = 1;
295         int initial = - (1 << d); // has last d bits = 0 and rest all = 1
296         byte val = value(id);
297         if (val > d) { // unusable
298             return -1;
299         }
300         while (val < d || (id & initial) == 0) { // id & initial == 1 << d for all ids at depth d, for < d it is 0
301             id <<= 1;
302             val = value(id);
303             if (val > d) {
304                 id ^= 1;
305                 val = value(id);
306             }
307         }
308         byte value = value(id);
309         assert value == d && (id & initial) == 1 << d : String.format("val = %d, id & initial = %d, d = %d",
310                 value, id & initial, d);
311         setValue(id, unusable); // mark as unusable
312         updateParentsAlloc(id);
313         return id;
314     }

316     /**
317      * Allocate a run of pages (>=1)
318      *
319      * @param normCapacity normalized capacity
320      * @return index in memoryMap
321      */
322     private long allocateRun(int normCapacity) {
323         int d = maxOrder - (log2(normCapacity) - pageShifts);
324         int id = allocateNode(d);
325         if (id < 0) {
326             return id;
327         }
328         freeBytes -= runLength(id);
329         return id;
330     }

332     /**
333      * Create / initialize a new PoolSubpage of normCapacity
334      * Any PoolSubpage created / initialized here is added to subpage pool in the PoolArena that owns this PoolChunk
335      *
336      * @param normCapacity normalized capacity
337      * @return index in memoryMap
338      */
339     private long allocateSubpage(int normCapacity) {
340         // Obtain the head of the PoolSubPage pool that is owned by the PoolArena and synchronize on it.
341         // This is need as we may add it back and so alter the linked-list structure.
342         PoolSubpage<T> head = arena.findSubpagePoolHead(normCapacity);
343         int d = maxOrder; // subpages are only be allocated from pages i.e., leaves
344         synchronized (head) {
345             int id = allocateNode(d);
346             if (id < 0) {
347                 return id;
348             }
349 
350             final PoolSubpage<T>[] subpages = this.subpages;
351             final int pageSize = this.pageSize;
352 
353             freeBytes -= pageSize;
354 
355             int subpageIdx = subpageIdx(id);
356             PoolSubpage<T> subpage = subpages[subpageIdx];
357             if (subpage == null) {
358                 subpage = new PoolSubpage<T>(head, this, id, runOffset(id), pageSize, normCapacity);
359                 subpages[subpageIdx] = subpage;
360             } else {
361                 subpage.init(head, normCapacity);
362             }
363             return subpage.allocate();
364         }
365     }

367     /**
368      * Free a subpage or a run of pages
369      * When a subpage is freed from PoolSubpage, it might be added back to subpage pool of the owning PoolArena
370      * If the subpage pool in PoolArena has at least one other PoolSubpage of given elemSize, we can
371      * completely free the owning Page so it is available for subsequent allocations
372      *
373      * @param handle handle to free
374      */
375     void free(long handle, ByteBuffer nioBuffer) {
376         int memoryMapIdx = memoryMapIdx(handle);
377         int bitmapIdx = bitmapIdx(handle);
378 
379         if (bitmapIdx != 0) { // free a subpage
380             PoolSubpage<T> subpage = subpages[subpageIdx(memoryMapIdx)];
381             assert subpage != null && subpage.doNotDestroy;
382 
383             // Obtain the head of the PoolSubPage pool that is owned by the PoolArena and synchronize on it.
384             // This is need as we may add it back and so alter the linked-list structure.
385             PoolSubpage<T> head = arena.findSubpagePoolHead(subpage.elemSize);
386             synchronized (head) {
387                 if (subpage.free(head, bitmapIdx & 0x3FFFFFFF)) {
388                     return;
389                 }
390             }
391         }
392         freeBytes += runLength(memoryMapIdx);
393         setValue(memoryMapIdx, depth(memoryMapIdx));
394         updateParentsFree(memoryMapIdx);
395 
396         if (nioBuffer != null && cachedNioBuffers != null &&
397                 cachedNioBuffers.size() < PooledByteBufAllocator.DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK) {
398             cachedNioBuffers.offer(nioBuffer);
399         }
400     }

402     void initBuf(PooledByteBuf<T> buf, ByteBuffer nioBuffer, long handle, int reqCapacity) {
403         int memoryMapIdx = memoryMapIdx(handle);
404         int bitmapIdx = bitmapIdx(handle);
405         if (bitmapIdx == 0) {
406             byte val = value(memoryMapIdx);
407             assert val == unusable : String.valueOf(val);
408             buf.init(this, nioBuffer, handle, runOffset(memoryMapIdx) + offset,
409                     reqCapacity, runLength(memoryMapIdx), arena.parent.threadCache());
410         } else {
411             initBufWithSubpage(buf, nioBuffer, handle, bitmapIdx, reqCapacity);
412         }
413     }

415     void initBufWithSubpage(PooledByteBuf<T> buf, ByteBuffer nioBuffer, long handle, int reqCapacity) {
416         initBufWithSubpage(buf, nioBuffer, handle, bitmapIdx(handle), reqCapacity);
417     }

419     private void initBufWithSubpage(PooledByteBuf<T> buf, ByteBuffer nioBuffer,
420                                     long handle, int bitmapIdx, int reqCapacity) {
421         assert bitmapIdx != 0;
422 
423         int memoryMapIdx = memoryMapIdx(handle);
424 
425         PoolSubpage<T> subpage = subpages[subpageIdx(memoryMapIdx)];
426         assert subpage.doNotDestroy;
427         assert reqCapacity <= subpage.elemSize;
428 
429         buf.init(
430             this, nioBuffer, handle,
431             runOffset(memoryMapIdx) + (bitmapIdx & 0x3FFFFFFF) * subpage.elemSize + offset,
432                 reqCapacity, subpage.elemSize, arena.parent.threadCache());
433     }

447     private static int log2(int val) {
448         // compute the (0-based, with lsb = 0) position of highest set bit i.e, log2
449         return INTEGER_SIZE_MINUS_ONE - Integer.numberOfLeadingZeros(val);
450     }
451 
452     private int runLength(int id) {
453         // represents the size in #bytes supported by node 'id' in the tree
454         return 1 << log2ChunkSize - depth(id);
455     }
456 
457     private int runOffset(int id) {
458         // represents the 0-based offset in #bytes from start of the byte-array chunk
459         int shift = id ^ 1 << depth(id);
460         return shift * runLength(id);
461     }
462 
463     private int subpageIdx(int memoryMapIdx) {
464         return memoryMapIdx ^ maxSubpageAllocs; // remove highest set bit, to get offset
465     }

467     private static int memoryMapIdx(long handle) {
468         return (int) handle;

471     private static int bitmapIdx(long handle) {
472         return (int) (handle >>> Integer.SIZE);
