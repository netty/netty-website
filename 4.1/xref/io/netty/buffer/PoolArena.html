76          this.pageSize = pageSize;
77          this.maxOrder = maxOrder;

160     private void allocate(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity) {
161         final int normCapacity = normalizeCapacity(reqCapacity);
162         if (isTinyOrSmall(normCapacity)) { // capacity < pageSize
163             int tableIdx;
164             PoolSubpage<T>[] table;
165             boolean tiny = isTiny(normCapacity);
166             if (tiny) { // < 512
167                 if (cache.allocateTiny(this, buf, reqCapacity, normCapacity)) {
168                     // was able to allocate out of the cache so move on
169                     return;
170                 }
171                 tableIdx = tinyIdx(normCapacity);
172                 table = tinySubpagePools;
173             } else {
174                 if (cache.allocateSmall(this, buf, reqCapacity, normCapacity)) {
175                     // was able to allocate out of the cache so move on
176                     return;
177                 }
178                 tableIdx = smallIdx(normCapacity);
179                 table = smallSubpagePools;
180             }
181 
182             final PoolSubpage<T> head = table[tableIdx];
183 
184             /**
185              * Synchronize on the head. This is needed as {@link PoolSubpage#allocate()} and
186              * {@link PoolSubpage#free(int)} may modify the doubly linked list as well.
187              */
188             synchronized (head) {
189                 final PoolSubpage<T> s = head.next;
190                 if (s != head) {
191                     assert s.doNotDestroy && s.elemSize == normCapacity;
192                     long handle = s.allocate();
193                     assert handle >= 0;
194                     s.chunk.initBufWithSubpage(buf, handle, reqCapacity);
195 
196                     if (tiny) {
197                         ++allocationsTiny;
198                     } else {
199                         ++allocationsSmall;
200                     }
201                     return;
202                 }
203             }
204             allocateNormal(buf, reqCapacity, normCapacity);
205             return;
206         }
207         if (normCapacity <= chunkSize) {
208             if (cache.allocateNormal(this, buf, reqCapacity, normCapacity)) {
209                 // was able to allocate out of the cache so move on
210                 return;
211             }
212             allocateNormal(buf, reqCapacity, normCapacity);
213         } else {
214             // Huge allocations are never served via the cache so just call allocateHuge
215             allocateHuge(buf, reqCapacity);
216         }
217     }

219     private synchronized void allocateNormal(PooledByteBuf<T> buf, int reqCapacity, int normCapacity) {
220             ++allocationsNormal;
221         if (q050.allocate(buf, reqCapacity, normCapacity) || q025.allocate(buf, reqCapacity, normCapacity) ||
222             q000.allocate(buf, reqCapacity, normCapacity) || qInit.allocate(buf, reqCapacity, normCapacity) ||
223             q075.allocate(buf, reqCapacity, normCapacity) || q100.allocate(buf, reqCapacity, normCapacity)) {
224             return;
225         }
226 
227         // Add a new chunk.
228         PoolChunk<T> c = newChunk(pageSize, maxOrder, pageShifts, chunkSize);
229         long handle = c.allocate(normCapacity);
230         assert handle > 0;
231         c.initBuf(buf, handle, reqCapacity);
232         qInit.add(c);
233     }

240     void free(PoolChunk<T> chunk, long handle, int normCapacity, PoolThreadCache cache) {
241         if (chunk.unpooled) {
242             allocationsHuge.decrement();
243             destroyChunk(chunk);
244         } else {
245             SizeClass sizeClass = sizeClass(normCapacity);
246             if (cache != null && cache.add(this, chunk, handle, normCapacity, sizeClass)) {
247                 // cached so not free it.
248                 return;
249             }
250 
251             freeChunk(chunk, handle, sizeClass);
252         }
253     }

255     private SizeClass sizeClass(int normCapacity) {
256         if (!isTinyOrSmall(normCapacity)) {
257             return SizeClass.Normal;
258         }
259         return isTiny(normCapacity) ? SizeClass.Tiny : SizeClass.Small;
260     }

262     void freeChunk(PoolChunk<T> chunk, long handle, SizeClass sizeClass) {
263         final boolean destroyChunk;
264         synchronized (this) {
265             switch (sizeClass) {
266             case Normal:
267                 ++deallocationsNormal;
268                 break;
269             case Small:
270                 ++deallocationsSmall;
271                 break;
272             case Tiny:
273                 ++deallocationsTiny;
274                 break;
275             default:
276                 throw new Error();
277             }
278             destroyChunk = !chunk.parent.free(chunk, handle);
279         }
280         if (destroyChunk) {
281             // destroyChunk not need to be called while holding the synchronized lock.
282             destroyChunk(chunk);
283         }
284     }

286     PoolSubpage<T> findSubpagePoolHead(int elemSize) {
287         int tableIdx;
288         PoolSubpage<T>[] table;
289         if (isTiny(elemSize)) { // < 512
290             tableIdx = elemSize >>> 4;
291             table = tinySubpagePools;
292         } else {
293             tableIdx = 0;
294             elemSize >>>= 10;
295             while (elemSize != 0) {
296                 elemSize >>>= 1;
297                 tableIdx ++;
298             }
299             table = smallSubpagePools;
300         }
301 
302         return table[tableIdx];
303     }

305     int normalizeCapacity(int reqCapacity) {
306         if (reqCapacity < 0) {
307             throw new IllegalArgumentException("capacity: " + reqCapacity + " (expected: 0+)");
308         }
309         if (reqCapacity >= chunkSize) {
310             return reqCapacity;
311         }
312 
313         if (!isTiny(reqCapacity)) { // >= 512
314             // Doubled
315 
316             int normalizedCapacity = reqCapacity;
317             normalizedCapacity --;
318             normalizedCapacity |= normalizedCapacity >>>  1;
319             normalizedCapacity |= normalizedCapacity >>>  2;
320             normalizedCapacity |= normalizedCapacity >>>  4;
321             normalizedCapacity |= normalizedCapacity >>>  8;
322             normalizedCapacity |= normalizedCapacity >>> 16;
323             normalizedCapacity ++;
324 
325             if (normalizedCapacity < 0) {
326                 normalizedCapacity >>>= 1;
327             }
328 
329             return normalizedCapacity;
330         }
331 
332         // Quantum-spaced
333         if ((reqCapacity & 15) == 0) {
334             return reqCapacity;
335         }
336 
337         return (reqCapacity & ~15) + 16;
338     }

340     void reallocate(PooledByteBuf<T> buf, int newCapacity, boolean freeOldMemory) {
341         if (newCapacity < 0 || newCapacity > buf.maxCapacity()) {
342             throw new IllegalArgumentException("newCapacity: " + newCapacity);
343         }
344 
345         int oldCapacity = buf.length;
346         if (oldCapacity == newCapacity) {
347             return;
348         }
349 
350         PoolChunk<T> oldChunk = buf.chunk;
351         long oldHandle = buf.handle;
352         T oldMemory = buf.memory;
353         int oldOffset = buf.offset;
354         int oldMaxLength = buf.maxLength;
355         int readerIndex = buf.readerIndex();
356         int writerIndex = buf.writerIndex();
357 
358         allocate(parent.threadCache(), buf, newCapacity);
359         if (newCapacity > oldCapacity) {
360             memoryCopy(
361                     oldMemory, oldOffset,
362                     buf.memory, buf.offset, oldCapacity);
363         } else if (newCapacity < oldCapacity) {
364             if (readerIndex < newCapacity) {
365                 if (writerIndex > newCapacity) {
366                     writerIndex = newCapacity;
367                 }
368                 memoryCopy(
369                         oldMemory, oldOffset + readerIndex,
370                         buf.memory, buf.offset + readerIndex, writerIndex - readerIndex);
371             } else {
372                 readerIndex = writerIndex = newCapacity;
373             }
374         }
375 
376         buf.setIndex(readerIndex, writerIndex);
377 
378         if (freeOldMemory) {
379             free(oldChunk, oldHandle, oldMaxLength, buf.cache);
380         }
381     }

413     private static List<PoolSubpageMetric> subPageMetricList(PoolSubpage<?>[] pages) {
414         List<PoolSubpageMetric> metrics = new ArrayList<PoolSubpageMetric>();
415         for (int i = 1; i < pages.length; i ++) {
416             PoolSubpage<?> head = pages[i];
417             if (head.next == head) {
418                 continue;
419             }
420             PoolSubpage<?> s = head.next;
421             for (;;) {
422                 metrics.add(s);
423                 s = s.next;
424                 if (s == head) {
425                     break;
426                 }
427             }
428         }
429         return metrics;
430     }

661             if (HAS_UNSAFE) {
662                 return PooledUnsafeDirectByteBuf.newInstance(maxCapacity);
663             } else {
664                 return PooledDirectByteBuf.newInstance(maxCapacity);
665             }
666         }
667 
668         @Override
669         protected void memoryCopy(ByteBuffer src, int srcOffset, ByteBuffer dst, int dstOffset, int length) {
670             if (length == 0) {
671                 return;
672             }
673 
674             if (HAS_UNSAFE) {
675                 PlatformDependent.copyMemory(
676                         PlatformDependent.directBufferAddress(src) + srcOffset,
677                         PlatformDependent.directBufferAddress(dst) + dstOffset, length);
678             } else {
679                 // We must duplicate the NIO buffers because they may be accessed by other Netty buffers.
680                 src = src.duplicate();
681                 dst = dst.duplicate();
682                 src.position(srcOffset).limit(srcOffset + length);
683                 dst.position(dstOffset);
684                 dst.put(src);
685             }
686         }
