29  /**
30   * A hash map implementation of {@link ByteObjectMap} that uses open addressing for keys.
31   * To minimize the memory footprint, this class uses open addressing rather than chaining.
32   * Collisions are resolved using linear probing. Deletions implement compaction, so cost of
33   * remove can approach O(N) for full maps, which makes a small loadFactor recommended.
34   *
35   * @param <V> The value type stored in the map.
36   */
37  public class ByteObjectHashMap<V> implements ByteObjectMap<V> {

79      public ByteObjectHashMap(int initialCapacity, float loadFactor) {
80          if (loadFactor <= 0.0f || loadFactor > 1.0f) {
81              // Cannot exceed 1 because we can never store more than capacity elements;
82              // using a bigger loadFactor would trigger rehashing before the desired load is reached.
83              throw new IllegalArgumentException("loadFactor must be > 0 and <= 1");
84          }
85  
86          this.loadFactor = loadFactor;
87  
88          // Adjust the initial capacity if necessary.
89          int capacity = safeFindNextPositivePowerOfTwo(initialCapacity);
90          mask = capacity - 1;
91  
92          // Allocate the arrays.
93          keys = new byte[capacity];
94          @SuppressWarnings({ "unchecked", "SuspiciousArrayCast" })
95          V[] temp = (V[]) new Object[capacity];
96          values = temp;
97  
98          // Initialize the maximum size value.
99          maxSize = calcMaxSize(capacity);
100     }

117     @Override
118     public V put(byte key, V value) {
119         int startIndex = hashIndex(key);
120         int index = startIndex;
121 
122         for (;;) {
123             if (values[index] == null) {
124                 // Found empty slot, use it.
125                 keys[index] = key;
126                 values[index] = toInternal(value);
127                 growSize();
128                 return null;
129             }
130             if (keys[index] == key) {
131                 // Found existing entry with this key, just replace the value.
132                 V previousValue = values[index];
133                 values[index] = toInternal(value);
134                 return toExternal(previousValue);
135             }
136 
137             // Conflict, keep probing ...
138             if ((index = probeNext(index)) == startIndex) {
139                 // Can only happen if the map was full at MAX_ARRAY_SIZE and couldn't grow.
140                 throw new IllegalStateException("Unable to insert");
141             }
142         }
143     }

145     @Override
146     public void putAll(Map<? extends Byte, ? extends V> sourceMap) {
147         if (sourceMap instanceof ByteObjectHashMap) {
148             // Optimization - iterate through the arrays.
149             @SuppressWarnings("unchecked")
150             ByteObjectHashMap<V> source = (ByteObjectHashMap<V>) sourceMap;
151             for (int i = 0; i < source.values.length; ++i) {
152                 V sourceValue = source.values[i];
153                 if (sourceValue != null) {
154                     put(source.keys[i], sourceValue);
155                 }
156             }
157             return;
158         }
159 
160         // Otherwise, just add each entry.
161         for (Entry<? extends Byte, ? extends V> entry : sourceMap.entrySet()) {
162             put(entry.getKey(), entry.getValue());
163         }
164     }

250     @Override
251     public int hashCode() {
252         // Hashcode is based on all non-zero, valid keys. We have to scan the whole keys
253         // array, which may have different lengths for two maps of same size(), so the
254         // capacity cannot be used as input for hashing but the size can.
255         int hash = size;
256         for (byte key : keys) {
257             // 0 can be a valid key or unused slot, but won't impact the hashcode in either case.
258             // This way we can use a cheap loop without conditionals, or hard-to-unroll operations,
259             // or the devastatingly bad memory locality of visiting value objects.
260             // Also, it's important to use a hash function that does not depend on the ordering
261             // of terms, only their values; since the map is an unordered collection and
262             // entries can end up in different positions in different maps that have the same
263             // elements, but with different history of puts/removes, due to conflicts.
264             hash ^= hashCode(key);
265         }
266         return hash;
267     }

269     @Override

271         if (this == obj) {
272             return true;
273         }
274         if (!(obj instanceof ByteObjectMap)) {
275             return false;
276         }
277         @SuppressWarnings("rawtypes")
278         ByteObjectMap other = (ByteObjectMap) obj;
279         if (size != other.size()) {
280             return false;
281         }
282         for (int i = 0; i < values.length; ++i) {
283             V value = values[i];
284             if (value != null) {
285                 byte key = keys[i];
286                 Object otherValue = other.get(key);
287                 if (value == NULL_VALUE) {
288                     if (otherValue != null) {
289                         return false;
290                     }
291                 } else if (!value.equals(otherValue)) {
292                     return false;
293                 }
294             }
295         }
296         return true;
297     }

333     /**
334      * Locates the index for the given key. This method probes using double hashing.
335      *
336      * @param key the key for an entry in the map.
337      * @return the index where the key was found, or {@code -1} if no entry is found for that key.
338      */
339     private int indexOf(byte key) {
340         int startIndex = hashIndex(key);
341         int index = startIndex;
342 
343         for (;;) {
344             if (values[index] == null) {
345                 // It's available, so no chance that this value exists anywhere in the map.
346                 return -1;
347             }
348             if (key == keys[index]) {
349                 return index;
350             }
351 
352             // Conflict, keep probing ...
353             if ((index = probeNext(index)) == startIndex) {
354                 return -1;
355             }
356         }
357     }

362     private int hashIndex(byte key) {
363         // The array lengths are always a power of two, so we can use a bitmask to stay inside the array bounds.
364         return hashCode(key) & mask;

374     /**
375      * Get the next sequential index after {@code index} and wraps if necessary.
376      */
377     private int probeNext(int index) {
378         // The array lengths are always a power of two, so we can use a bitmask to stay inside the array bounds.
379         return (index + 1) & mask;
380     }

382     /**
383      * Grows the map size after an insertion. If necessary, performs a rehash of the map.
384      */
385     private void growSize() {
386         size++;
387 
388         if (size > maxSize) {
389             if(keys.length == Integer.MAX_VALUE) {
390                 throw new IllegalStateException("Max capacity reached at size=" + size);
391             }
392 
393             // Double the capacity.
394             rehash(keys.length << 1);
395         }
396     }

398     /**
399      * Removes entry at the given index position. Also performs opportunistic, incremental rehashing
400      * if necessary to not break conflict chains.
401      *
402      * @param index the index position of the element to remove.
403      * @return {@code true} if the next item was moved back. {@code false} otherwise.
404      */
405     private boolean removeAt(final int index) {
406         --size;
407         // Clearing the key is not strictly necessary (for GC like in a regular collection),
408         // but recommended for security. The memory location is still fresh in the cache anyway.
409         keys[index] = 0;
410         values[index] = null;
411 
412         // In the interval from index to the next available entry, the arrays may have entries
413         // that are displaced from their base position due to prior conflicts. Iterate these
414         // entries and move them back if possible, optimizing future lookups.
415         // Knuth Section 6.4 Algorithm R, also used by the JDK's IdentityHashMap.
416 
417         boolean movedBack = false;
418         int nextFree = index;
419         for (int i = probeNext(index); values[i] != null; i = probeNext(i)) {
420             int bucket = hashIndex(keys[i]);
421             if (i < bucket && (bucket <= nextFree || nextFree <= i) ||
422                 bucket <= nextFree && nextFree <= i) {
423                 // Move the displaced entry "back" to the first available position.
424                 keys[nextFree] = keys[i];
425                 values[nextFree] = values[i];
426                 movedBack = true;
427                 // Put the first entry after the displaced entry
428                 keys[i] = 0;
429                 values[i] = null;
430                 nextFree = i;
431             }
432         }
433         return movedBack;
434     }

436     /**
437      * Calculates the maximum size allowed before rehashing.
438      */
439     private int calcMaxSize(int capacity) {
440         // Clip the upper bound so that there will always be at least one available slot.
441         int upperBound = capacity - 1;
442         return Math.min(upperBound, (int) (capacity * loadFactor));
443     }

445     /**
446      * Rehashes the map for the given capacity.
447      *
448      * @param newCapacity the new capacity for the map.
449      */
450     private void rehash(int newCapacity) {
451         byte[] oldKeys = keys;
452         V[] oldVals = values;
453 
454         keys = new byte[newCapacity];
455         @SuppressWarnings({ "unchecked", "SuspiciousArrayCast" })
456         V[] temp = (V[]) new Object[newCapacity];
457         values = temp;
458 
459         maxSize = calcMaxSize(newCapacity);
460         mask = newCapacity - 1;
461 
462         // Insert to the new arrays.
463         for (int i = 0; i < oldVals.length; ++i) {
464             V oldVal = oldVals[i];
465             if (oldVal != null) {
466                 // Inlined put(), but much simpler: we don't need to worry about
467                 // duplicated keys, growing/rehashing, or failing to insert.
468                 byte oldKey = oldKeys[i];
469                 int index = hashIndex(oldKey);
470 
471                 for (;;) {
472                     if (values[index] == null) {
473                         keys[index] = oldKey;
474                         values[index] = oldVal;
475                         break;
476                     }
477 
478                     // Conflict, keep probing. Can wrap around, but never reaches startIndex again.
479                     index = probeNext(index);
480                 }
481             }
482         }
483     }

485     @Override
486     public String toString() {
487         if (isEmpty()) {
488             return "{}";
489         }
490         StringBuilder sb = new StringBuilder(4 * size);
491         sb.append('{');
492         boolean first = true;
493         for (int i = 0; i < values.length; ++i) {
494             V value = values[i];
495             if (value != null) {
496                 if (!first) {
497                     sb.append(", ");
498                 }
499                 sb.append(keyToString(keys[i])).append('=').append(value == this ? "(this Map)" :
500                     toExternal(value));
501                 first = false;
502             }
503         }
504         return sb.append('}').toString();
505     }

590     /**
591      * Iterator over primitive entries. Entry key/values are overwritten by each call to {@link #next()}.
592      */
593     private final class PrimitiveIterator implements Iterator<PrimitiveEntry<V>>, PrimitiveEntry<V> {
594         private int prevIndex = -1;
595         private int nextIndex = -1;
596         private int entryIndex = -1;
597 
598         private void scanNext() {
599             for (;;) {
600                 if (++nextIndex == values.length || values[nextIndex] != null) {
601                     break;
602                 }
603             }
604         }
605 
606         @Override
607         public boolean hasNext() {
608             if (nextIndex == -1) {
609                 scanNext();
610             }
611             return nextIndex < keys.length;
612         }
613 
614         @Override
615         public PrimitiveEntry<V> next() {
616             if (!hasNext()) {
617                 throw new NoSuchElementException();
618             }
619 
620             prevIndex = nextIndex;
621             scanNext();
622 
623             // Always return the same Entry object, just change its index each time.
624             entryIndex = prevIndex;
625             return this;
626         }
627 
628         @Override
629         public void remove() {
630             if (prevIndex < 0) {
631                 throw new IllegalStateException("next must be called before each remove.");
632             }
633             if (removeAt(prevIndex)) {
634                 // removeAt may move elements "back" in the array if they have been displaced because their spot in the
635                 // array was occupied when they were inserted. If this occurs then the nextIndex is now invalid and
636                 // should instead point to the prevIndex which now holds an element which was "moved back".
637                 nextIndex = prevIndex;
638             }
639             prevIndex = -1;
640         }
641 
642         // Entry implementation. Since this implementation uses a single Entry, we coalesce that
643         // into the Iterator object (potentially making loop optimization much easier).
644 
645         @Override
646         public byte key() {
647             return keys[entryIndex];
648         }
649 
650         @Override
651         public V value() {
652             return toExternal(values[entryIndex]);
653         }
654 
655         @Override
656         public void setValue(V value) {
657             values[entryIndex] = toInternal(value);
658         }
659     }
