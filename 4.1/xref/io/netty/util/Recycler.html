125         @Override
126         protected void onRemoval(Stack<T> value) {
127             // Let us remove the WeakOrderQueue from the WeakHashMap directly if its safe to remove some overhead
128             if (value.threadRef.get() == Thread.currentThread()) {

155                        int ratio, int maxDelayedQueuesPerThread, int delayedQueueRatio) {

157         delayedQueueInterval = max(0, delayedQueueRatio);
158         if (maxCapacityPerThread <= 0) {

160             this.maxSharedCapacityFactor = 1;
161             this.maxDelayedQueuesPerThread = 0;
162         } else {
163             this.maxCapacityPerThread = maxCapacityPerThread;

165             this.maxDelayedQueuesPerThread = max(0, maxDelayedQueuesPerThread);

169     @SuppressWarnings("unchecked")
170     public final T get() {
171         if (maxCapacityPerThread == 0) {
172             return newObject((Handle<T>) NOOP_HANDLE);
173         }
174         Stack<T> stack = threadLocal.get();
175         DefaultHandle<T> handle = stack.pop();
176         if (handle == null) {
177             handle = stack.newHandle();
178             handle.value = newObject(handle);
179         }
180         return (T) handle.value;
181     }

213     @SuppressWarnings("unchecked")
214     private static final class DefaultHandle<T> implements Handle<T> {
215         private static final AtomicIntegerFieldUpdater<DefaultHandle<?>> LAST_RECYCLED_ID_UPDATER;
216         static {
217             AtomicIntegerFieldUpdater<?> updater = AtomicIntegerFieldUpdater.newUpdater(
218                     DefaultHandle.class, "lastRecycledId");
219             LAST_RECYCLED_ID_UPDATER = (AtomicIntegerFieldUpdater<DefaultHandle<?>>) updater;
220         }
221 
222         volatile int lastRecycledId;
223         int recycleId;
224 
225         boolean hasBeenRecycled;
226 
227         Stack<?> stack;
228         Object value;
229 
230         DefaultHandle(Stack<?> stack) {
231             this.stack = stack;
232         }
233 
234         @Override
235         public void recycle(Object object) {
236             if (object != value) {
237                 throw new IllegalArgumentException("object does not belong to handle");
238             }
239 
240             Stack<?> stack = this.stack;
241             if (lastRecycledId != recycleId || stack == null) {
242                 throw new IllegalStateException("recycled already");
243             }
244 
245             stack.push(this);
246         }

248         public boolean compareAndSetLastRecycledId(int expectLastRecycledId, int updateLastRecycledId) {
249             // Use "weakâ€¦" because we do not need synchronize-with ordering, only atomicity.
250             // Also, spurious failures are fine, since no code should rely on recycling for correctness.
251             return LAST_RECYCLED_ID_UPDATER.weakCompareAndSet(this, expectLastRecycledId, updateLastRecycledId);
252         }

256             new FastThreadLocal<Map<Stack<?>, WeakOrderQueue>>() {

263     // a queue that makes only moderate guarantees about visibility: items are seen in the correct order,
264     // but we aren't absolutely guaranteed to ever see anything at all, thereby keeping the queue cheap to maintain
265     private static final class WeakOrderQueue extends WeakReference<Thread> {
266 
267         static final WeakOrderQueue DUMMY = new WeakOrderQueue();
268 
269         // Let Link extend AtomicInteger for intrinsics. The Link itself will be used as writerIndex.
270         @SuppressWarnings("serial")
271         static final class Link extends AtomicInteger {
272             final DefaultHandle<?>[] elements = new DefaultHandle[LINK_CAPACITY];
273 
274             int readIndex;
275             Link next;
276         }

278         // Its important this does not hold any reference to either Stack or WeakOrderQueue.
279         private static final class Head {
280             private final AtomicInteger availableSharedCapacity;
281 
282             Link link;
283 
284             Head(AtomicInteger availableSharedCapacity) {
285                 this.availableSharedCapacity = availableSharedCapacity;
286             }
287 
288             /**
289              * Reclaim all used space and also unlink the nodes to prevent GC nepotism.
290              */
291             void reclaimAllSpaceAndUnlink() {
292                 Link head = link;
293                 link = null;
294                 int reclaimSpace = 0;
295                 while (head != null) {
296                     reclaimSpace += LINK_CAPACITY;
297                     Link next = head.next;
298                     // Unlink to help GC and guard against GC nepotism.
299                     head.next = null;
300                     head = next;
301                 }
302                 if (reclaimSpace > 0) {
303                     reclaimSpace(reclaimSpace);
304                 }
305             }
306 
307             private void reclaimSpace(int space) {
308                 availableSharedCapacity.addAndGet(space);
309             }
310 
311             void relink(Link link) {
312                 reclaimSpace(LINK_CAPACITY);
313                 this.link = link;
314             }
315 
316             /**
317              * Creates a new {@link} and returns it if we can reserve enough space for it, otherwise it
318              * returns {@code null}.
319              */
320             Link newLink() {
321                 return reserveSpaceForLink(availableSharedCapacity) ? new Link() : null;
322             }
323 
324             static boolean reserveSpaceForLink(AtomicInteger availableSharedCapacity) {
325                 for (;;) {
326                     int available = availableSharedCapacity.get();
327                     if (available < LINK_CAPACITY) {
328                         return false;
329                     }
330                     if (availableSharedCapacity.compareAndSet(available, available - LINK_CAPACITY)) {
331                         return true;
332                     }
333                 }
334             }
335         }
336 
337         // chain of data items
338         private final Head head;
339         private Link tail;
340         // pointer to another queue of delayed items for the same stack
341         private WeakOrderQueue next;
342         private final int id = ID_GENERATOR.getAndIncrement();
343         private final int interval;
344         private int handleRecycleCount;
345 
346         private WeakOrderQueue() {
347             super(null);
348             head = new Head(null);
349             interval = 0;
350         }
351 
352         private WeakOrderQueue(Stack<?> stack, Thread thread) {
353             super(thread);
354             tail = new Link();
355 
356             // Its important that we not store the Stack itself in the WeakOrderQueue as the Stack also is used in
357             // the WeakHashMap as key. So just store the enclosed AtomicInteger which should allow to have the
358             // Stack itself GCed.
359             head = new Head(stack.availableSharedCapacity);
360             head.link = tail;
361             interval = stack.delayedQueueInterval;
362             handleRecycleCount = interval; // Start at interval so the first one will be recycled.
363         }
364 
365         static WeakOrderQueue newQueue(Stack<?> stack, Thread thread) {
366             // We allocated a Link so reserve the space
367             if (!Head.reserveSpaceForLink(stack.availableSharedCapacity)) {
368                 return null;
369             }
370             final WeakOrderQueue queue = new WeakOrderQueue(stack, thread);
371             // Done outside of the constructor to ensure WeakOrderQueue.this does not escape the constructor and so
372             // may be accessed while its still constructed.
373             stack.setHead(queue);
374 
375             return queue;
376         }
377 
378         WeakOrderQueue getNext() {
379             return next;
380         }
381 
382         void setNext(WeakOrderQueue next) {
383             assert next != this;
384             this.next = next;
385         }
386 
387         void reclaimAllSpaceAndUnlink() {
388             head.reclaimAllSpaceAndUnlink();
389             next = null;
390         }

392         void add(DefaultHandle<?> handle) {
393             if (!handle.compareAndSetLastRecycledId(0, id)) {
394                 // Separate threads could be racing to add the handle to each their own WeakOrderQueue.
395                 // We only add the handle to the queue if we win the race and observe that lastRecycledId is zero.
396                 return;
397             }
398 
399             // While we also enforce the recycling ratio when we transfer objects from the WeakOrderQueue to the Stack
400             // we better should enforce it as well early. Missing to do so may let the WeakOrderQueue grow very fast
401             // without control
402             if (!handle.hasBeenRecycled) {
403                 if (handleRecycleCount < interval) {
404                     handleRecycleCount++;
405                     // Drop the item to prevent from recycling too aggressively.
406                     return;
407                 }
408                 handleRecycleCount = 0;
409             }
410 
411             Link tail = this.tail;
412             int writeIndex;
413             if ((writeIndex = tail.get()) == LINK_CAPACITY) {
414                 Link link = head.newLink();
415                 if (link == null) {
416                     // Drop it.
417                     return;
418                 }
419                 // We allocate a Link so reserve the space
420                 this.tail = tail = tail.next = link;
421 
422                 writeIndex = tail.get();
423             }
424             tail.elements[writeIndex] = handle;
425             handle.stack = null;
426             // we lazy set to ensure that setting stack to null appears before we unnull it in the owning thread;
427             // this also means we guarantee visibility of an element in the queue if we see the index updated
428             tail.lazySet(writeIndex + 1);
429         }
430 
431         boolean hasFinalData() {
432             return tail.readIndex != tail.get();
433         }
434 
435         // transfer as many items as we can from this queue to the stack, returning true if any were transferred
436         @SuppressWarnings("rawtypes")
437         boolean transfer(Stack<?> dst) {
438             Link head = this.head.link;
439             if (head == null) {
440                 return false;
441             }
442 
443             if (head.readIndex == LINK_CAPACITY) {
444                 if (head.next == null) {
445                     return false;
446                 }
447                 head = head.next;
448                 this.head.relink(head);
449             }
450 
451             final int srcStart = head.readIndex;
452             int srcEnd = head.get();
453             final int srcSize = srcEnd - srcStart;
454             if (srcSize == 0) {
455                 return false;
456             }
457 
458             final int dstSize = dst.size;
459             final int expectedCapacity = dstSize + srcSize;
460 
461             if (expectedCapacity > dst.elements.length) {
462                 final int actualCapacity = dst.increaseCapacity(expectedCapacity);
463                 srcEnd = min(srcStart + actualCapacity - dstSize, srcEnd);
464             }
465 
466             if (srcStart != srcEnd) {
467                 final DefaultHandle[] srcElems = head.elements;
468                 final DefaultHandle[] dstElems = dst.elements;
469                 int newDstSize = dstSize;
470                 for (int i = srcStart; i < srcEnd; i++) {
471                     DefaultHandle<?> element = srcElems[i];
472                     if (element.recycleId == 0) {
473                         element.recycleId = element.lastRecycledId;
474                     } else if (element.recycleId != element.lastRecycledId) {
475                         throw new IllegalStateException("recycled already");
476                     }
477                     srcElems[i] = null;
478 
479                     if (dst.dropHandle(element)) {
480                         // Drop the object.
481                         continue;
482                     }
483                     element.stack = dst;
484                     dstElems[newDstSize ++] = element;
485                 }
486 
487                 if (srcEnd == LINK_CAPACITY && head.next != null) {
488                     // Add capacity back as the Link is GCed.
489                     this.head.relink(head.next);
490                 }
491 
492                 head.readIndex = srcEnd;
493                 if (dst.size == newDstSize) {
494                     return false;
495                 }
496                 dst.size = newDstSize;
497                 return true;
498             } else {
499                 // The destination stack is full already.
500                 return false;
501             }
502         }
503     }

505     private static final class Stack<T> {
506 
507         // we keep a queue of per-thread queues, which is appended to once only, each time a new thread other
508         // than the stack owner recycles: when we run out of items in our stack we iterate this collection
509         // to scavenge those that can be reused. this permits us to incur minimal thread synchronisation whilst
510         // still recycling all items.
511         final Recycler<T> parent;
512 
513         // We store the Thread in a WeakReference as otherwise we may be the only ones that still hold a strong
514         // Reference to the Thread itself after it died because DefaultHandle will hold a reference to the Stack.
515         //
516         // The biggest issue is if we do not use a WeakReference the Thread may not be able to be collected at all if
517         // the user will store a reference to the DefaultHandle somewhere and never clear this reference (or not clear
518         // it in a timely manner).
519         final WeakReference<Thread> threadRef;
520         final AtomicInteger availableSharedCapacity;
521         private final int maxDelayedQueues;
522 
523         private final int maxCapacity;
524         private final int interval;
525         private final int delayedQueueInterval;
526         DefaultHandle<?>[] elements;
527         int size;
528         private int handleRecycleCount;
529         private WeakOrderQueue cursor, prev;
530         private volatile WeakOrderQueue head;
531 
532         Stack(Recycler<T> parent, Thread thread, int maxCapacity, int maxSharedCapacityFactor,
533               int interval, int maxDelayedQueues, int delayedQueueInterval) {
534             this.parent = parent;
535             threadRef = new WeakReference<Thread>(thread);
536             this.maxCapacity = maxCapacity;
537             availableSharedCapacity = new AtomicInteger(max(maxCapacity / maxSharedCapacityFactor, LINK_CAPACITY));
538             elements = new DefaultHandle[min(INITIAL_CAPACITY, maxCapacity)];
539             this.interval = interval;
540             this.delayedQueueInterval = delayedQueueInterval;
541             handleRecycleCount = interval; // Start at interval so the first one will be recycled.
542             this.maxDelayedQueues = maxDelayedQueues;
543         }
544 
545         // Marked as synchronized to ensure this is serialized.
546         synchronized void setHead(WeakOrderQueue queue) {
547             queue.setNext(head);
548             head = queue;
549         }
550 
551         int increaseCapacity(int expectedCapacity) {
552             int newCapacity = elements.length;
553             int maxCapacity = this.maxCapacity;
554             do {
555                 newCapacity <<= 1;
556             } while (newCapacity < expectedCapacity && newCapacity < maxCapacity);
557 
558             newCapacity = min(newCapacity, maxCapacity);
559             if (newCapacity != elements.length) {
560                 elements = Arrays.copyOf(elements, newCapacity);
561             }
562 
563             return newCapacity;
564         }
565 
566         @SuppressWarnings({ "unchecked", "rawtypes" })
567         DefaultHandle<T> pop() {
568             int size = this.size;
569             if (size == 0) {
570                 if (!scavenge()) {
571                     return null;
572                 }
573                 size = this.size;
574                 if (size <= 0) {
575                     // double check, avoid races
576                     return null;
577                 }
578             }
579             size --;
580             DefaultHandle ret = elements[size];
581             elements[size] = null;
582             // As we already set the element[size] to null we also need to store the updated size before we do
583             // any validation. Otherwise we may see a null value when later try to pop again without a new element
584             // added before.
585             this.size = size;
586 
587             if (ret.lastRecycledId != ret.recycleId) {
588                 throw new IllegalStateException("recycled multiple times");
589             }
590             ret.recycleId = 0;
591             ret.lastRecycledId = 0;
592             return ret;
593         }
594 
595         private boolean scavenge() {
596             // continue an existing scavenge, if any
597             if (scavengeSome()) {
598                 return true;
599             }
600 
601             // reset our scavenge cursor
602             prev = null;
603             cursor = head;
604             return false;
605         }
606 
607         private boolean scavengeSome() {
608             WeakOrderQueue prev;
609             WeakOrderQueue cursor = this.cursor;
610             if (cursor == null) {
611                 prev = null;
612                 cursor = head;
613                 if (cursor == null) {
614                     return false;
615                 }
616             } else {
617                 prev = this.prev;
618             }
619 
620             boolean success = false;
621             do {
622                 if (cursor.transfer(this)) {
623                     success = true;
624                     break;
625                 }
626                 WeakOrderQueue next = cursor.getNext();
627                 if (cursor.get() == null) {
628                     // If the thread associated with the queue is gone, unlink it, after
629                     // performing a volatile read to confirm there is no data left to collect.
630                     // We never unlink the first queue, as we don't want to synchronize on updating the head.
631                     if (cursor.hasFinalData()) {
632                         for (;;) {
633                             if (cursor.transfer(this)) {
634                                 success = true;
635                             } else {
636                                 break;
637                             }
638                         }
639                     }
640 
641                     if (prev != null) {
642                         // Ensure we reclaim all space before dropping the WeakOrderQueue to be GC'ed.
643                         cursor.reclaimAllSpaceAndUnlink();
644                         prev.setNext(next);
645                     }
646                 } else {
647                     prev = cursor;
648                 }
649 
650                 cursor = next;
651 
652             } while (cursor != null && !success);
653 
654             this.prev = prev;
655             this.cursor = cursor;
656             return success;
657         }
658 
659         void push(DefaultHandle<?> item) {
660             Thread currentThread = Thread.currentThread();
661             if (threadRef.get() == currentThread) {
662                 // The current Thread is the thread that belongs to the Stack, we can try to push the object now.
663                 pushNow(item);
664             } else {
665                 // The current Thread is not the one that belongs to the Stack
666                 // (or the Thread that belonged to the Stack was collected already), we need to signal that the push
667                 // happens later.
668                 pushLater(item, currentThread);
669             }
670         }
671 
672         private void pushNow(DefaultHandle<?> item) {
673             if (item.recycleId != 0 || !item.compareAndSetLastRecycledId(0, OWN_THREAD_ID)) {
674                 throw new IllegalStateException("recycled already");
675             }
676             item.recycleId = OWN_THREAD_ID;
677 
678             int size = this.size;
679             if (size >= maxCapacity || dropHandle(item)) {
680                 // Hit the maximum capacity or should drop - drop the possibly youngest object.
681                 return;
682             }
683             if (size == elements.length) {
684                 elements = Arrays.copyOf(elements, min(size << 1, maxCapacity));
685             }
686 
687             elements[size] = item;
688             this.size = size + 1;
689         }
690 
691         private void pushLater(DefaultHandle<?> item, Thread thread) {
692             if (maxDelayedQueues == 0) {
693                 // We don't support recycling across threads and should just drop the item on the floor.
694                 return;
695             }
696 
697             // we don't want to have a ref to the queue as the value in our weak map
698             // so we null it out; to ensure there are no races with restoring it later
699             // we impose a memory ordering here (no-op on x86)
700             Map<Stack<?>, WeakOrderQueue> delayedRecycled = DELAYED_RECYCLED.get();
701             WeakOrderQueue queue = delayedRecycled.get(this);
702             if (queue == null) {
703                 if (delayedRecycled.size() >= maxDelayedQueues) {
704                     // Add a dummy queue so we know we should drop the object
705                     delayedRecycled.put(this, WeakOrderQueue.DUMMY);
706                     return;
707                 }
708                 // Check if we already reached the maximum number of delayed queues and if we can allocate at all.
709                 if ((queue = newWeakOrderQueue(thread)) == null) {
710                     // drop object
711                     return;
712                 }
713                 delayedRecycled.put(this, queue);
714             } else if (queue == WeakOrderQueue.DUMMY) {
715                 // drop object
716                 return;
717             }
718 
719             queue.add(item);
720         }
721 
722         /**
723          * Allocate a new {@link WeakOrderQueue} or return {@code null} if not possible.
724          */
725         private WeakOrderQueue newWeakOrderQueue(Thread thread) {
726             return WeakOrderQueue.newQueue(this, thread);
727         }

729         boolean dropHandle(DefaultHandle<?> handle) {
730             if (!handle.hasBeenRecycled) {
731                 if (handleRecycleCount < interval) {
732                     handleRecycleCount++;
733                     // Drop the object.
734                     return true;
735                 }
736                 handleRecycleCount = 0;
737                 handle.hasBeenRecycled = true;
738             }
739             return false;
740         }

743             return new DefaultHandle<T>(this);
744         }
