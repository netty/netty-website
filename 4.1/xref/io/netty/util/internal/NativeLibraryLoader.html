59      static {
60          String workdir = SystemPropertyUtil.get("io.netty.native.workdir");
61          if (workdir != null) {
62              File f = new File(workdir);
63              f.mkdirs();
64  
65              try {
66                  f = f.getAbsoluteFile();
67              } catch (Exception ignored) {
68                  // Good to have an absolute path, but it's OK.
69              }
70  
71              WORKDIR = f;
72              logger.debug("-Dio.netty.native.workdir: " + WORKDIR);
73          } else {
74              WORKDIR = PlatformDependent.tmpdir();
75              logger.debug("-Dio.netty.native.workdir: " + WORKDIR + " (io.netty.tmpdir)");
76          }
77  
78          DELETE_NATIVE_LIB_AFTER_LOADING = SystemPropertyUtil.getBoolean(
79                  "io.netty.native.deleteLibAfterLoading", true);
80          logger.debug("-Dio.netty.native.deleteLibAfterLoading: {}", DELETE_NATIVE_LIB_AFTER_LOADING);
81  
82          TRY_TO_PATCH_SHADED_ID = SystemPropertyUtil.getBoolean(
83                  "io.netty.native.tryPatchShadedId", true);
84          logger.debug("-Dio.netty.native.tryPatchShadedId: {}", TRY_TO_PATCH_SHADED_ID);
85      }

87      /**
88       * Loads the first available library in the collection with the specified
89       * {@link ClassLoader}.
90       *
91       * @throws IllegalArgumentException
92       *         if none of the given libraries load successfully.
93       */
94      public static void loadFirstAvailable(ClassLoader loader, String... names) {
95          List<Throwable> suppressed = new ArrayList<Throwable>();
96          for (String name : names) {
97              try {
98                  load(name, loader);
99                  logger.debug("Loaded library with name '{}'", name);
100                 return;
101             } catch (Throwable t) {
102                 suppressed.add(t);
103             }
104         }
105 
106         IllegalArgumentException iae =
107                 new IllegalArgumentException("Failed to load any of the given libraries: " + Arrays.toString(names));
108         ThrowableUtil.addSuppressedAndClear(iae, suppressed);
109         throw iae;
110     }

112     /**
113      * The shading prefix added to this class's full name.
114      *
115      * @throws UnsatisfiedLinkError if the shader used something other than a prefix
116      */
117     private static String calculatePackagePrefix() {
118         String maybeShaded = NativeLibraryLoader.class.getName();
119         // Use ! instead of . to avoid shading utilities from modifying the string
120         String expected = "io!netty!util!internal!NativeLibraryLoader".replace('!', '.');
121         if (!maybeShaded.endsWith(expected)) {
122             throw new UnsatisfiedLinkError(String.format(
123                     "Could not find prefix added to %s to get %s. When shading, only adding a "
124                     + "package prefix is supported", expected, maybeShaded));
125         }
126         return maybeShaded.substring(0, maybeShaded.length() - expected.length());
127     }

129     /**
130      * Load the given library with the specified {@link ClassLoader}
131      */
132     public static void load(String originalName, ClassLoader loader) {
133         // Adjust expected name to support shading of native libraries.
134         String packagePrefix = calculatePackagePrefix().replace('.', '_');
135         String name = packagePrefix + originalName;
136         List<Throwable> suppressed = new ArrayList<Throwable>();
137         try {
138             // first try to load from java.library.path
139             loadLibrary(loader, name, false);
140             return;
141         } catch (Throwable ex) {
142             suppressed.add(ex);
143         }
144 
145         String libname = System.mapLibraryName(name);
146         String path = NATIVE_RESOURCE_HOME + libname;
147 
148         InputStream in = null;
149         OutputStream out = null;
150         File tmpFile = null;
151         URL url = getResource(path, loader);
152         try {
153             if (url == null) {
154                 if (PlatformDependent.isOsx()) {
155                     String fileName = path.endsWith(".jnilib") ? NATIVE_RESOURCE_HOME + "lib" + name + ".dynlib" :
156                             NATIVE_RESOURCE_HOME + "lib" + name + ".jnilib";
157                     url = getResource(fileName, loader);
158                     if (url == null) {
159                         FileNotFoundException fnf = new FileNotFoundException(fileName);
160                         ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
161                         throw fnf;
162                     }
163                 } else {
164                     FileNotFoundException fnf = new FileNotFoundException(path);
165                     ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
166                     throw fnf;
167                 }
168             }
169 
170             int index = libname.lastIndexOf('.');
171             String prefix = libname.substring(0, index);
172             String suffix = libname.substring(index);
173 
174             tmpFile = PlatformDependent.createTempFile(prefix, suffix, WORKDIR);
175             in = url.openStream();
176             out = new FileOutputStream(tmpFile);
177 
178             byte[] buffer = new byte[8192];
179             int length;
180             while ((length = in.read(buffer)) > 0) {
181                 out.write(buffer, 0, length);
182             }
183             out.flush();
184 
185             if (shouldShadedLibraryIdBePatched(packagePrefix)) {
186                 // Let's try to patch the id and re-sign it. This is a best-effort and might fail if a
187                 // SecurityManager is setup or the right executables are not installed :/
188                 tryPatchShadedLibraryIdAndSign(tmpFile, originalName);
189             }
190 
191             // Close the output stream before loading the unpacked library,
192             // because otherwise Windows will refuse to load it when it's in use by other process.
193             closeQuietly(out);
194             out = null;
195 
196             loadLibrary(loader, tmpFile.getPath(), true);
197         } catch (UnsatisfiedLinkError e) {
198             try {
199                 if (tmpFile != null && tmpFile.isFile() && tmpFile.canRead() &&
200                     !NoexecVolumeDetector.canExecuteExecutable(tmpFile)) {
201                     // Pass "io.netty.native.workdir" as an argument to allow shading tools to see
202                     // the string. Since this is printed out to users to tell them what to do next,
203                     // we want the value to be correct even when shading.
204                     logger.info("{} exists but cannot be executed even when execute permissions set; " +
205                                 "check volume for \"noexec\" flag; use -D{}=[path] " +
206                                 "to set native working directory separately.",
207                                 tmpFile.getPath(), "io.netty.native.workdir");
208                 }
209             } catch (Throwable t) {
210                 suppressed.add(t);
211                 logger.debug("Error checking if {} is on a file store mounted with noexec", tmpFile, t);
212             }
213             // Re-throw to fail the load
214             ThrowableUtil.addSuppressedAndClear(e, suppressed);
215             throw e;
216         } catch (Exception e) {
217             UnsatisfiedLinkError ule = new UnsatisfiedLinkError("could not load a native library: " + name);
218             ule.initCause(e);
219             ThrowableUtil.addSuppressedAndClear(ule, suppressed);
220             throw ule;
221         } finally {
222             closeQuietly(in);
223             closeQuietly(out);
224             // After we load the library it is safe to delete the file.
225             // We delete the file immediately to free up resources as soon as possible,
226             // and if this fails fallback to deleting on JVM exit.
227             if (tmpFile != null && (!DELETE_NATIVE_LIB_AFTER_LOADING || !tmpFile.delete())) {
228                 tmpFile.deleteOnExit();
229             }
230         }
231     }

233     private static URL getResource(String path, ClassLoader loader) {
234         final Enumeration<URL> urls;
235         try {
236             if (loader == null) {
237                 urls = ClassLoader.getSystemResources(path);
238             } else {
239                 urls = loader.getResources(path);
240             }
241         } catch (IOException iox) {
242             throw new RuntimeException("An error occurred while getting the resources for " + path, iox);
243         }
244 
245         List<URL> urlsList = Collections.list(urls);
246         int size = urlsList.size();
247         switch (size) {
248             case 0:
249                 return null;
250             case 1:
251                 return urlsList.get(0);
252             default:
253                 throw new IllegalStateException("Multiple resources found for '" + path + "': " + urlsList);
254         }
255     }

257     static void tryPatchShadedLibraryIdAndSign(File libraryFile, String originalName) {
258         String newId = new String(generateUniqueId(originalName.length()), CharsetUtil.UTF_8);
259         if (!tryExec("install_name_tool -id " + newId + " " + libraryFile.getAbsolutePath())) {
260             return;
261         }
262 
263         tryExec("codesign -s - " + libraryFile.getAbsolutePath());
264     }

266     private static boolean tryExec(String cmd) {
267         try {
268             int exitValue = Runtime.getRuntime().exec(cmd).waitFor();
269             if (exitValue != 0) {
270                 logger.debug("Execution of '{}' failed: {}", cmd, exitValue);
271                 return false;
272             }
273             logger.debug("Execution of '{}' succeed: {}", cmd, exitValue);
274             return true;
275         } catch (InterruptedException e) {
276             Thread.currentThread().interrupt();
277         } catch (IOException e) {
278             logger.info("Execution of '{}' failed.", cmd, e);
279         } catch (SecurityException e) {
280             logger.error("Execution of '{}' failed.", cmd, e);
281         }
282         return false;
283     }

285     private static boolean shouldShadedLibraryIdBePatched(String packagePrefix) {
286         return TRY_TO_PATCH_SHADED_ID && PlatformDependent.isOsx() && !packagePrefix.isEmpty();
287     }

289     private static byte[] generateUniqueId(int length) {
290         byte[] idBytes = new byte[length];
291         for (int i = 0; i < idBytes.length; i++) {
292             // We should only use bytes as replacement that are in our UNIQUE_ID_BYTES array.
293             idBytes[i] = UNIQUE_ID_BYTES[PlatformDependent.threadLocalRandom()
294                     .nextInt(UNIQUE_ID_BYTES.length)];
295         }
296         return idBytes;
297     }

299     /**
300      * Loading the native library into the specified {@link ClassLoader}.
301      * @param loader - The {@link ClassLoader} where the native library will be loaded into
302      * @param name - The native library path or name
303      * @param absolute - Whether the native library will be loaded by path or by name
304      */
305     private static void loadLibrary(final ClassLoader loader, final String name, final boolean absolute) {
306         Throwable suppressed = null;
307         try {
308             try {
309                 // Make sure the helper belongs to the target ClassLoader.
310                 final Class<?> newHelper = tryToLoadClass(loader, NativeLibraryUtil.class);
311                 loadLibraryByHelper(newHelper, name, absolute);
312                 logger.debug("Successfully loaded the library {}", name);
313                 return;
314             } catch (UnsatisfiedLinkError e) { // Should by pass the UnsatisfiedLinkError here!
315                 suppressed = e;
316             } catch (Exception e) {
317                 suppressed = e;
318             }
319             NativeLibraryUtil.loadLibrary(name, absolute);  // Fallback to local helper class.
320             logger.debug("Successfully loaded the library {}", name);
321         } catch (NoSuchMethodError nsme) {
322             if (suppressed != null) {
323                 ThrowableUtil.addSuppressed(nsme, suppressed);
324             }
325             rethrowWithMoreDetailsIfPossible(name, nsme);
326         } catch (UnsatisfiedLinkError ule) {
327             if (suppressed != null) {
328                 ThrowableUtil.addSuppressed(ule, suppressed);
329             }
330             throw ule;
331         }
332     }

334     @SuppressJava6Requirement(reason = "Guarded by version check")
335     private static void rethrowWithMoreDetailsIfPossible(String name, NoSuchMethodError error) {
336         if (PlatformDependent.javaVersion() >= 7) {
337             throw new LinkageError(
338                     "Possible multiple incompatible native libraries on the classpath for '" + name + "'?", error);
339         }
340         throw error;
341     }

343     private static void loadLibraryByHelper(final Class<?> helper, final String name, final boolean absolute)
344             throws UnsatisfiedLinkError {
345         Object ret = AccessController.doPrivileged(new PrivilegedAction<Object>() {
346             @Override
347             public Object run() {
348                 try {
349                     // Invoke the helper to load the native library, if succeed, then the native
350                     // library belong to the specified ClassLoader.
351                     Method method = helper.getMethod("loadLibrary", String.class, boolean.class);
352                     method.setAccessible(true);
353                     return method.invoke(null, name, absolute);
354                 } catch (Exception e) {
355                     return e;
356                 }
357             }
358         });
359         if (ret instanceof Throwable) {
360             Throwable t = (Throwable) ret;
361             assert !(t instanceof UnsatisfiedLinkError) : t + " should be a wrapper throwable";
362             Throwable cause = t.getCause();
363             if (cause instanceof UnsatisfiedLinkError) {
364                 throw (UnsatisfiedLinkError) cause;
365             }
366             UnsatisfiedLinkError ule = new UnsatisfiedLinkError(t.getMessage());
367             ule.initCause(t);
368             throw ule;
369         }
370     }

372     /**
373      * Try to load the helper {@link Class} into specified {@link ClassLoader}.
374      * @param loader - The {@link ClassLoader} where to load the helper {@link Class}
375      * @param helper - The helper {@link Class}
376      * @return A new helper Class defined in the specified ClassLoader.
377      * @throws ClassNotFoundException Helper class not found or loading failed
378      */
379     private static Class<?> tryToLoadClass(final ClassLoader loader, final Class<?> helper)
380             throws ClassNotFoundException {
381         try {
382             return Class.forName(helper.getName(), false, loader);
383         } catch (ClassNotFoundException e1) {
384             if (loader == null) {
385                 // cannot defineClass inside bootstrap class loader
386                 throw e1;
387             }
388             try {
389                 // The helper class is NOT found in target ClassLoader, we have to define the helper class.
390                 final byte[] classBinary = classToByteArray(helper);
391                 return AccessController.doPrivileged(new PrivilegedAction<Class<?>>() {
392                     @Override
393                     public Class<?> run() {
394                         try {
395                             // Define the helper class in the target ClassLoader,
396                             //  then we can call the helper to load the native library.
397                             Method defineClass = ClassLoader.class.getDeclaredMethod("defineClass", String.class,
398                                     byte[].class, int.class, int.class);
399                             defineClass.setAccessible(true);
400                             return (Class<?>) defineClass.invoke(loader, helper.getName(), classBinary, 0,
401                                     classBinary.length);
402                         } catch (Exception e) {
403                             throw new IllegalStateException("Define class failed!", e);
404                         }
405                     }
406                 });
407             } catch (ClassNotFoundException e2) {
408                 ThrowableUtil.addSuppressed(e2, e1);
409                 throw e2;
410             } catch (RuntimeException e2) {
411                 ThrowableUtil.addSuppressed(e2, e1);
412                 throw e2;
413             } catch (Error e2) {
414                 ThrowableUtil.addSuppressed(e2, e1);
415                 throw e2;
416             }
417         }
418     }

420     /**
421      * Load the helper {@link Class} as a byte array, to be redefined in specified {@link ClassLoader}.
422      * @param clazz - The helper {@link Class} provided by this bundle
423      * @return The binary content of helper {@link Class}.
424      * @throws ClassNotFoundException Helper class not found or loading failed
425      */
426     private static byte[] classToByteArray(Class<?> clazz) throws ClassNotFoundException {
427         String fileName = clazz.getName();
428         int lastDot = fileName.lastIndexOf('.');
429         if (lastDot > 0) {
430             fileName = fileName.substring(lastDot + 1);
431         }
432         URL classUrl = clazz.getResource(fileName + ".class");
433         if (classUrl == null) {
434             throw new ClassNotFoundException(clazz.getName());
435         }
436         byte[] buf = new byte[1024];
437         ByteArrayOutputStream out = new ByteArrayOutputStream(4096);
438         InputStream in = null;
439         try {
440             in = classUrl.openStream();
441             for (int r; (r = in.read(buf)) != -1;) {
442                 out.write(buf, 0, r);
443             }
444             return out.toByteArray();
445         } catch (IOException ex) {
446             throw new ClassNotFoundException(clazz.getName(), ex);
447         } finally {
448             closeQuietly(in);
449             closeQuietly(out);
450         }
451     }

453     private static void closeQuietly(Closeable c) {
454         if (c != null) {
455             try {
456                 c.close();
457             } catch (IOException ignore) {
458                 // ignore
459             }
460         }
461     }

467     private static final class NoexecVolumeDetector {
468 
469         @SuppressJava6Requirement(reason = "Usage guarded by java version check")
470         private static boolean canExecuteExecutable(File file) throws IOException {
471             if (PlatformDependent.javaVersion() < 7) {
472                 // Pre-JDK7, the Java API did not directly support POSIX permissions; instead of implementing a custom
473                 // work-around, assume true, which disables the check.
474                 return true;
475             }
476 
477             // If we can already execute, there is nothing to do.
478             if (file.canExecute()) {
479                 return true;
480             }
481 
482             // On volumes, with noexec set, even files with the executable POSIX permissions will fail to execute.
483             // The File#canExecute() method honors this behavior, probaby via parsing the noexec flag when initializing
484             // the UnixFileStore, though the flag is not exposed via a public API.  To find out if library is being
485             // loaded off a volume with noexec, confirm or add executalbe permissions, then check File#canExecute().
486 
487             // Note: We use FQCN to not break when netty is used in java6
488             Set<java.nio.file.attribute.PosixFilePermission> existingFilePermissions =
489                     java.nio.file.Files.getPosixFilePermissions(file.toPath());
490             Set<java.nio.file.attribute.PosixFilePermission> executePermissions =
491                     EnumSet.of(java.nio.file.attribute.PosixFilePermission.OWNER_EXECUTE,
492                             java.nio.file.attribute.PosixFilePermission.GROUP_EXECUTE,
493                             java.nio.file.attribute.PosixFilePermission.OTHERS_EXECUTE);
494             if (existingFilePermissions.containsAll(executePermissions)) {
495                 return false;
496             }
497 
498             Set<java.nio.file.attribute.PosixFilePermission> newPermissions = EnumSet.copyOf(existingFilePermissions);
499             newPermissions.addAll(executePermissions);
500             java.nio.file.Files.setPosixFilePermissions(file.toPath(), newPermissions);
501             return file.canExecute();
502         }

505             // Utility
506         }
