57      static {
58          String workdir = SystemPropertyUtil.get("io.netty.native.workdir");
59          if (workdir != null) {
60              File f = new File(workdir);
61              f.mkdirs();
62  
63              try {
64                  f = f.getAbsoluteFile();
65              } catch (Exception ignored) {
66                  // Good to have an absolute path, but it's OK.
67              }
68  
69              WORKDIR = f;
70              logger.debug("-Dio.netty.native.workdir: " + WORKDIR);
71          } else {
72              WORKDIR = PlatformDependent.tmpdir();
73              logger.debug("-Dio.netty.native.workdir: " + WORKDIR + " (io.netty.tmpdir)");
74          }
75  
76          DELETE_NATIVE_LIB_AFTER_LOADING = SystemPropertyUtil.getBoolean(
77                  "io.netty.native.deleteLibAfterLoading", true);
78          logger.debug("-Dio.netty.native.deleteLibAfterLoading: {}", DELETE_NATIVE_LIB_AFTER_LOADING);
79  
80          TRY_TO_PATCH_SHADED_ID = SystemPropertyUtil.getBoolean(
81                  "io.netty.native.tryPatchShadedId", true);
82          logger.debug("-Dio.netty.native.tryPatchShadedId: {}", TRY_TO_PATCH_SHADED_ID);
83      }

85      /**
86       * Loads the first available library in the collection with the specified
87       * {@link ClassLoader}.
88       *
89       * @throws IllegalArgumentException
90       *         if none of the given libraries load successfully.
91       */
92      public static void loadFirstAvailable(ClassLoader loader, String... names) {
93          List<Throwable> suppressed = new ArrayList<Throwable>();
94          for (String name : names) {
95              try {
96                  load(name, loader);
97                  return;
98              } catch (Throwable t) {
99                  suppressed.add(t);
100             }
101         }
102 
103         IllegalArgumentException iae =
104                 new IllegalArgumentException("Failed to load any of the given libraries: " + Arrays.toString(names));
105         ThrowableUtil.addSuppressedAndClear(iae, suppressed);
106         throw iae;
107     }

109     /**
110      * The shading prefix added to this class's full name.
111      *
112      * @throws UnsatisfiedLinkError if the shader used something other than a prefix
113      */
114     private static String calculatePackagePrefix() {
115         String maybeShaded = NativeLibraryLoader.class.getName();
116         // Use ! instead of . to avoid shading utilities from modifying the string
117         String expected = "io!netty!util!internal!NativeLibraryLoader".replace('!', '.');
118         if (!maybeShaded.endsWith(expected)) {
119             throw new UnsatisfiedLinkError(String.format(
120                     "Could not find prefix added to %s to get %s. When shading, only adding a "
121                     + "package prefix is supported", expected, maybeShaded));
122         }
123         return maybeShaded.substring(0, maybeShaded.length() - expected.length());
124     }

126     /**
127      * Load the given library with the specified {@link ClassLoader}
128      */
129     public static void load(String originalName, ClassLoader loader) {
130         // Adjust expected name to support shading of native libraries.
131         String packagePrefix = calculatePackagePrefix().replace('.', '_');
132         String name = packagePrefix + originalName;
133         List<Throwable> suppressed = new ArrayList<Throwable>();
134         try {
135             // first try to load from java.library.path
136             loadLibrary(loader, name, false);
137             return;
138         } catch (Throwable ex) {
139             suppressed.add(ex);
140         }
141 
142         String libname = System.mapLibraryName(name);
143         String path = NATIVE_RESOURCE_HOME + libname;
144 
145         InputStream in = null;
146         OutputStream out = null;
147         File tmpFile = null;
148         URL url;
149         if (loader == null) {
150             url = ClassLoader.getSystemResource(path);
151         } else {
152             url = loader.getResource(path);
153         }
154         try {
155             if (url == null) {
156                 if (PlatformDependent.isOsx()) {
157                     String fileName = path.endsWith(".jnilib") ? NATIVE_RESOURCE_HOME + "lib" + name + ".dynlib" :
158                             NATIVE_RESOURCE_HOME + "lib" + name + ".jnilib";
159                     if (loader == null) {
160                         url = ClassLoader.getSystemResource(fileName);
161                     } else {
162                         url = loader.getResource(fileName);
163                     }
164                     if (url == null) {
165                         FileNotFoundException fnf = new FileNotFoundException(fileName);
166                         ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
167                         throw fnf;
168                     }
169                 } else {
170                     FileNotFoundException fnf = new FileNotFoundException(path);
171                     ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
172                     throw fnf;
173                 }
174             }
175 
176             int index = libname.lastIndexOf('.');
177             String prefix = libname.substring(0, index);
178             String suffix = libname.substring(index);
179 
180             tmpFile = PlatformDependent.createTempFile(prefix, suffix, WORKDIR);
181             in = url.openStream();
182             out = new FileOutputStream(tmpFile);
183 
184             if (shouldShadedLibraryIdBePatched(packagePrefix)) {
185                 patchShadedLibraryId(in, out, originalName, name);
186             } else {
187                 byte[] buffer = new byte[8192];
188                 int length;
189                 while ((length = in.read(buffer)) > 0) {
190                     out.write(buffer, 0, length);
191                 }
192             }
193 
194             out.flush();
195 
196             // Close the output stream before loading the unpacked library,
197             // because otherwise Windows will refuse to load it when it's in use by other process.
198             closeQuietly(out);
199             out = null;
200             loadLibrary(loader, tmpFile.getPath(), true);
201         } catch (UnsatisfiedLinkError e) {
202             try {
203                 if (tmpFile != null && tmpFile.isFile() && tmpFile.canRead() &&
204                     !NoexecVolumeDetector.canExecuteExecutable(tmpFile)) {
205                     // Pass "io.netty.native.workdir" as an argument to allow shading tools to see
206                     // the string. Since this is printed out to users to tell them what to do next,
207                     // we want the value to be correct even when shading.
208                     logger.info("{} exists but cannot be executed even when execute permissions set; " +
209                                 "check volume for \"noexec\" flag; use -D{}=[path] " +
210                                 "to set native working directory separately.",
211                                 tmpFile.getPath(), "io.netty.native.workdir");
212                 }
213             } catch (Throwable t) {
214                 suppressed.add(t);
215                 logger.debug("Error checking if {} is on a file store mounted with noexec", tmpFile, t);
216             }
217             // Re-throw to fail the load
218             ThrowableUtil.addSuppressedAndClear(e, suppressed);
219             throw e;
220         } catch (Exception e) {
221             UnsatisfiedLinkError ule = new UnsatisfiedLinkError("could not load a native library: " + name);
222             ule.initCause(e);
223             ThrowableUtil.addSuppressedAndClear(ule, suppressed);
224             throw ule;
225         } finally {
226             closeQuietly(in);
227             closeQuietly(out);
228             // After we load the library it is safe to delete the file.
229             // We delete the file immediately to free up resources as soon as possible,
230             // and if this fails fallback to deleting on JVM exit.
231             if (tmpFile != null && (!DELETE_NATIVE_LIB_AFTER_LOADING || !tmpFile.delete())) {
232                 tmpFile.deleteOnExit();
233             }
234         }
235     }

237     // Package-private for testing.
238     static boolean patchShadedLibraryId(InputStream in, OutputStream out, String originalName, String name)
239             throws IOException {
240         byte[] buffer = new byte[8192];
241         int length;
242         // We read the whole native lib into memory to make it easier to monkey-patch the id.
243         ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream(in.available());
244 
245         while ((length = in.read(buffer)) > 0) {
246             byteArrayOutputStream.write(buffer, 0, length);
247         }
248         byteArrayOutputStream.flush();
249         byte[] bytes = byteArrayOutputStream.toByteArray();
250         byteArrayOutputStream.close();
251 
252         final boolean patched;
253         // Try to patch the library id.
254         if (!patchShadedLibraryId(bytes, originalName, name)) {
255             // We did not find the Id, check if we used a originalName that has the os and arch as suffix.
256             // If this is the case we should also try to patch with the os and arch suffix removed.
257             String os = PlatformDependent.normalizedOs();
258             String arch = PlatformDependent.normalizedArch();
259             String osArch = "_" + os + "_" + arch;
260             if (originalName.endsWith(osArch)) {
261                 patched = patchShadedLibraryId(bytes,
262                         originalName.substring(0, originalName.length() - osArch.length()), name);
263             } else {
264                 patched = false;
265             }
266         } else {
267             patched = true;
268         }
269         out.write(bytes, 0, bytes.length);
270         return patched;
271     }

273     private static boolean shouldShadedLibraryIdBePatched(String packagePrefix) {
274         return TRY_TO_PATCH_SHADED_ID && PlatformDependent.isOsx() && !packagePrefix.isEmpty();
275     }

277     /**
278      * Try to patch shaded library to ensure it uses a unique ID.
279      */
280     private static boolean patchShadedLibraryId(byte[] bytes, String originalName, String name) {
281         // Our native libs always have the name as part of their id so we can search for it and replace it
282         // to make the ID unique if shading is used.
283         byte[] nameBytes = originalName.getBytes(CharsetUtil.UTF_8);
284         int idIdx = -1;
285 
286         // Be aware this is a really raw way of patching a dylib but it does all we need without implementing
287         // a full mach-o parser and writer. Basically we just replace the the original bytes with some
288         // random bytes as part of the ID regeneration. The important thing here is that we need to use the same
289         // length to not corrupt the mach-o header.
290         outerLoop: for (int i = 0; i < bytes.length && bytes.length - i >= nameBytes.length; i++) {
291             int idx = i;
292             for (int j = 0; j < nameBytes.length;) {
293                 if (bytes[idx++] != nameBytes[j++]) {
294                     // Did not match the name, increase the index and try again.
295                     break;
296                 } else if (j == nameBytes.length) {
297                     // We found the index within the id.
298                     idIdx = i;
299                     break outerLoop;
300                 }
301             }
302         }
303 
304         if (idIdx == -1) {
305             logger.debug("Was not able to find the ID of the shaded native library {}, can't adjust it.", name);
306             return false;
307         } else {
308             // We found our ID... now monkey-patch it!
309             for (int i = 0; i < nameBytes.length; i++) {
310                 // We should only use bytes as replacement that are in our UNIQUE_ID_BYTES array.
311                 bytes[idIdx + i] = UNIQUE_ID_BYTES[PlatformDependent.threadLocalRandom()
312                                                                     .nextInt(UNIQUE_ID_BYTES.length)];
313             }
314 
315             if (logger.isDebugEnabled()) {
316                 logger.debug(
317                         "Found the ID of the shaded native library {}. Replacing ID part {} with {}",
318                         name, originalName, new String(bytes, idIdx, nameBytes.length, CharsetUtil.UTF_8));
319             }
320             return true;
321         }
322     }

324     /**
325      * Loading the native library into the specified {@link ClassLoader}.
326      * @param loader - The {@link ClassLoader} where the native library will be loaded into
327      * @param name - The native library path or name
328      * @param absolute - Whether the native library will be loaded by path or by name
329      */
330     private static void loadLibrary(final ClassLoader loader, final String name, final boolean absolute) {
331         Throwable suppressed = null;
332         try {
333             try {
334                 // Make sure the helper is belong to the target ClassLoader.
335                 final Class<?> newHelper = tryToLoadClass(loader, NativeLibraryUtil.class);
336                 loadLibraryByHelper(newHelper, name, absolute);
337                 logger.debug("Successfully loaded the library {}", name);
338                 return;
339             } catch (UnsatisfiedLinkError e) { // Should by pass the UnsatisfiedLinkError here!
340                 suppressed = e;
341             } catch (Exception e) {
342                 suppressed = e;
343             }
344             NativeLibraryUtil.loadLibrary(name, absolute);  // Fallback to local helper class.
345             logger.debug("Successfully loaded the library {}", name);
346         } catch (NoSuchMethodError nsme) {
347             if (suppressed != null) {
348                 ThrowableUtil.addSuppressed(nsme, suppressed);
349             }
350             rethrowWithMoreDetailsIfPossible(name, nsme);
351         } catch (UnsatisfiedLinkError ule) {
352             if (suppressed != null) {
353                 ThrowableUtil.addSuppressed(ule, suppressed);
354             }
355             throw ule;
356         }
357     }

359     @SuppressJava6Requirement(reason = "Guarded by version check")
360     private static void rethrowWithMoreDetailsIfPossible(String name, NoSuchMethodError error) {
361         if (PlatformDependent.javaVersion() >= 7) {
362             throw new LinkageError(
363                     "Possible multiple incompatible native libraries on the classpath for '" + name + "'?", error);
364         }
365         throw error;
366     }

368     private static void loadLibraryByHelper(final Class<?> helper, final String name, final boolean absolute)
369             throws UnsatisfiedLinkError {
370         Object ret = AccessController.doPrivileged(new PrivilegedAction<Object>() {
371             @Override
372             public Object run() {
373                 try {
374                     // Invoke the helper to load the native library, if succeed, then the native
375                     // library belong to the specified ClassLoader.
376                     Method method = helper.getMethod("loadLibrary", String.class, boolean.class);
377                     method.setAccessible(true);
378                     return method.invoke(null, name, absolute);
379                 } catch (Exception e) {
380                     return e;
381                 }
382             }
383         });
384         if (ret instanceof Throwable) {
385             Throwable t = (Throwable) ret;
386             assert !(t instanceof UnsatisfiedLinkError) : t + " should be a wrapper throwable";
387             Throwable cause = t.getCause();
388             if (cause instanceof UnsatisfiedLinkError) {
389                 throw (UnsatisfiedLinkError) cause;
390             }
391             UnsatisfiedLinkError ule = new UnsatisfiedLinkError(t.getMessage());
392             ule.initCause(t);
393             throw ule;
394         }
395     }

397     /**
398      * Try to load the helper {@link Class} into specified {@link ClassLoader}.
399      * @param loader - The {@link ClassLoader} where to load the helper {@link Class}
400      * @param helper - The helper {@link Class}
401      * @return A new helper Class defined in the specified ClassLoader.
402      * @throws ClassNotFoundException Helper class not found or loading failed
403      */
404     private static Class<?> tryToLoadClass(final ClassLoader loader, final Class<?> helper)
405             throws ClassNotFoundException {
406         try {
407             return Class.forName(helper.getName(), false, loader);
408         } catch (ClassNotFoundException e1) {
409             if (loader == null) {
410                 // cannot defineClass inside bootstrap class loader
411                 throw e1;
412             }
413             try {
414                 // The helper class is NOT found in target ClassLoader, we have to define the helper class.
415                 final byte[] classBinary = classToByteArray(helper);
416                 return AccessController.doPrivileged(new PrivilegedAction<Class<?>>() {
417                     @Override
418                     public Class<?> run() {
419                         try {
420                             // Define the helper class in the target ClassLoader,
421                             //  then we can call the helper to load the native library.
422                             Method defineClass = ClassLoader.class.getDeclaredMethod("defineClass", String.class,
423                                     byte[].class, int.class, int.class);
424                             defineClass.setAccessible(true);
425                             return (Class<?>) defineClass.invoke(loader, helper.getName(), classBinary, 0,
426                                     classBinary.length);
427                         } catch (Exception e) {
428                             throw new IllegalStateException("Define class failed!", e);
429                         }
430                     }
431                 });
432             } catch (ClassNotFoundException e2) {
433                 ThrowableUtil.addSuppressed(e2, e1);
434                 throw e2;
435             } catch (RuntimeException e2) {
436                 ThrowableUtil.addSuppressed(e2, e1);
437                 throw e2;
438             } catch (Error e2) {
439                 ThrowableUtil.addSuppressed(e2, e1);
440                 throw e2;
441             }
442         }
443     }

445     /**
446      * Load the helper {@link Class} as a byte array, to be redefined in specified {@link ClassLoader}.
447      * @param clazz - The helper {@link Class} provided by this bundle
448      * @return The binary content of helper {@link Class}.
449      * @throws ClassNotFoundException Helper class not found or loading failed
450      */
451     private static byte[] classToByteArray(Class<?> clazz) throws ClassNotFoundException {
452         String fileName = clazz.getName();
453         int lastDot = fileName.lastIndexOf('.');
454         if (lastDot > 0) {
455             fileName = fileName.substring(lastDot + 1);
456         }
457         URL classUrl = clazz.getResource(fileName + ".class");
458         if (classUrl == null) {
459             throw new ClassNotFoundException(clazz.getName());
460         }
461         byte[] buf = new byte[1024];
462         ByteArrayOutputStream out = new ByteArrayOutputStream(4096);
463         InputStream in = null;
464         try {
465             in = classUrl.openStream();
466             for (int r; (r = in.read(buf)) != -1;) {
467                 out.write(buf, 0, r);
468             }
469             return out.toByteArray();
470         } catch (IOException ex) {
471             throw new ClassNotFoundException(clazz.getName(), ex);
472         } finally {
473             closeQuietly(in);
474             closeQuietly(out);
475         }
476     }

478     private static void closeQuietly(Closeable c) {
479         if (c != null) {
480             try {
481                 c.close();
482             } catch (IOException ignore) {
483                 // ignore
484             }
485         }

492     private static final class NoexecVolumeDetector {
493 
494         @SuppressJava6Requirement(reason = "Usage guarded by java version check")
495         private static boolean canExecuteExecutable(File file) throws IOException {
496             if (PlatformDependent.javaVersion() < 7) {
497                 // Pre-JDK7, the Java API did not directly support POSIX permissions; instead of implementing a custom
498                 // work-around, assume true, which disables the check.
499                 return true;
500             }
501 
502             // If we can already execute, there is nothing to do.
503             if (file.canExecute()) {
504                 return true;
505             }
506 
507             // On volumes, with noexec set, even files with the executable POSIX permissions will fail to execute.
508             // The File#canExecute() method honors this behavior, probaby via parsing the noexec flag when initializing
509             // the UnixFileStore, though the flag is not exposed via a public API.  To find out if library is being
510             // loaded off a volume with noexec, confirm or add executalbe permissions, then check File#canExecute().
511 
512             // Note: We use FQCN to not break when netty is used in java6
513             Set<java.nio.file.attribute.PosixFilePermission> existingFilePermissions =
514                     java.nio.file.Files.getPosixFilePermissions(file.toPath());
515             Set<java.nio.file.attribute.PosixFilePermission> executePermissions =
516                     EnumSet.of(java.nio.file.attribute.PosixFilePermission.OWNER_EXECUTE,
517                             java.nio.file.attribute.PosixFilePermission.GROUP_EXECUTE,
518                             java.nio.file.attribute.PosixFilePermission.OTHERS_EXECUTE);
519             if (existingFilePermissions.containsAll(executePermissions)) {
520                 return false;
521             }
522 
523             Set<java.nio.file.attribute.PosixFilePermission> newPermissions = EnumSet.copyOf(existingFilePermissions);
524             newPermissions.addAll(executePermissions);
525             java.nio.file.Files.setPosixFilePermissions(file.toPath(), newPermissions);
526             return file.canExecute();
527         }
