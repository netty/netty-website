85      /**
86       * Loads the first available library in the collection with the specified
87       * {@link ClassLoader}.
88       *
89       * @throws IllegalArgumentException
90       *         if none of the given libraries load successfully.
91       */
92      public static void loadFirstAvailable(ClassLoader loader, String... names) {
93          List<Throwable> suppressed = new ArrayList<Throwable>();
94          for (String name : names) {
95              try {
96                  load(name, loader);
97                  return;
98              } catch (Throwable t) {
99                  suppressed.add(t);
100                 logger.debug("Unable to load the library '{}', trying next name...", name, t);
101             }
102         }
103         IllegalArgumentException iae =
104                 new IllegalArgumentException("Failed to load any of the given libraries: " + Arrays.toString(names));
105         ThrowableUtil.addSuppressedAndClear(iae, suppressed);
106         throw iae;
107     }

109     /**
110      * The shading prefix added to this class's full name.
111      *
112      * @throws UnsatisfiedLinkError if the shader used something other than a prefix
113      */
114     private static String calculatePackagePrefix() {
115         String maybeShaded = NativeLibraryLoader.class.getName();
116         // Use ! instead of . to avoid shading utilities from modifying the string
117         String expected = "io!netty!util!internal!NativeLibraryLoader".replace('!', '.');
118         if (!maybeShaded.endsWith(expected)) {
119             throw new UnsatisfiedLinkError(String.format(
120                     "Could not find prefix added to %s to get %s. When shading, only adding a "
121                     + "package prefix is supported", expected, maybeShaded));
122         }
123         return maybeShaded.substring(0, maybeShaded.length() - expected.length());
124     }

126     /**
127      * Load the given library with the specified {@link ClassLoader}
128      */
129     public static void load(String originalName, ClassLoader loader) {
130         // Adjust expected name to support shading of native libraries.
131         String packagePrefix = calculatePackagePrefix().replace('.', '_');
132         String name = packagePrefix + originalName;
133         List<Throwable> suppressed = new ArrayList<Throwable>();
134         try {
135             // first try to load from java.library.path
136             loadLibrary(loader, name, false);
137             return;
138         } catch (Throwable ex) {
139             suppressed.add(ex);
140             logger.debug(
141                     "{} cannot be loaded from java.library.path, "
142                     + "now trying export to -Dio.netty.native.workdir: {}", name, WORKDIR, ex);
143         }
144 
145         String libname = System.mapLibraryName(name);
146         String path = NATIVE_RESOURCE_HOME + libname;
147 
148         InputStream in = null;
149         OutputStream out = null;
150         File tmpFile = null;
151         URL url;
152         if (loader == null) {
153             url = ClassLoader.getSystemResource(path);
154         } else {
155             url = loader.getResource(path);
156         }
157         try {
158             if (url == null) {
159                 if (PlatformDependent.isOsx()) {
160                     String fileName = path.endsWith(".jnilib") ? NATIVE_RESOURCE_HOME + "lib" + name + ".dynlib" :
161                             NATIVE_RESOURCE_HOME + "lib" + name + ".jnilib";
162                     if (loader == null) {
163                         url = ClassLoader.getSystemResource(fileName);
164                     } else {
165                         url = loader.getResource(fileName);
166                     }
167                     if (url == null) {
168                         FileNotFoundException fnf = new FileNotFoundException(fileName);
169                         ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
170                         throw fnf;
171                     }
172                 } else {
173                     FileNotFoundException fnf = new FileNotFoundException(path);
174                     ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
175                     throw fnf;
176                 }
177             }
178 
179             int index = libname.lastIndexOf('.');
180             String prefix = libname.substring(0, index);
181             String suffix = libname.substring(index);
182 
183             tmpFile = File.createTempFile(prefix, suffix, WORKDIR);
184             in = url.openStream();
185             out = new FileOutputStream(tmpFile);
186 
187             if (shouldShadedLibraryIdBePatched(packagePrefix)) {
188                 patchShadedLibraryId(in, out, originalName, name);
189             } else {
190                 byte[] buffer = new byte[8192];
191                 int length;
192                 while ((length = in.read(buffer)) > 0) {
193                     out.write(buffer, 0, length);
194                 }
195             }
196 
197             out.flush();
198 
199             // Close the output stream before loading the unpacked library,
200             // because otherwise Windows will refuse to load it when it's in use by other process.
201             closeQuietly(out);
202             out = null;
203             loadLibrary(loader, tmpFile.getPath(), true);
204         } catch (UnsatisfiedLinkError e) {
205             try {
206                 if (tmpFile != null && tmpFile.isFile() && tmpFile.canRead() &&
207                     !NoexecVolumeDetector.canExecuteExecutable(tmpFile)) {
208                     // Pass "io.netty.native.workdir" as an argument to allow shading tools to see
209                     // the string. Since this is printed out to users to tell them what to do next,
210                     // we want the value to be correct even when shading.
211                     logger.info("{} exists but cannot be executed even when execute permissions set; " +
212                                 "check volume for \"noexec\" flag; use -D{}=[path] " +
213                                 "to set native working directory separately.",
214                                 tmpFile.getPath(), "io.netty.native.workdir");
215                 }
216             } catch (Throwable t) {
217                 suppressed.add(t);
218                 logger.debug("Error checking if {} is on a file store mounted with noexec", tmpFile, t);
219             }
220             // Re-throw to fail the load
221             ThrowableUtil.addSuppressedAndClear(e, suppressed);
222             throw e;
223         } catch (Exception e) {
224             UnsatisfiedLinkError ule = new UnsatisfiedLinkError("could not load a native library: " + name);
225             ule.initCause(e);
226             ThrowableUtil.addSuppressedAndClear(ule, suppressed);
227             throw ule;
228         } finally {
229             closeQuietly(in);
230             closeQuietly(out);
231             // After we load the library it is safe to delete the file.
232             // We delete the file immediately to free up resources as soon as possible,
233             // and if this fails fallback to deleting on JVM exit.
234             if (tmpFile != null && (!DELETE_NATIVE_LIB_AFTER_LOADING || !tmpFile.delete())) {
235                 tmpFile.deleteOnExit();
236             }
237         }
238     }

240     // Package-private for testing.
241     static boolean patchShadedLibraryId(InputStream in, OutputStream out, String originalName, String name)
242             throws IOException {
243         byte[] buffer = new byte[8192];
244         int length;
245         // We read the whole native lib into memory to make it easier to monkey-patch the id.
246         ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream(in.available());
247 
248         while ((length = in.read(buffer)) > 0) {
249             byteArrayOutputStream.write(buffer, 0, length);
250         }
251         byteArrayOutputStream.flush();
252         byte[] bytes = byteArrayOutputStream.toByteArray();
253         byteArrayOutputStream.close();
254 
255         final boolean patched;
256         // Try to patch the library id.
257         if (!patchShadedLibraryId(bytes, originalName, name)) {
258             // We did not find the Id, check if we used a originalName that has the os and arch as suffix.
259             // If this is the case we should also try to patch with the os and arch suffix removed.
260             String os = PlatformDependent.normalizedOs();
261             String arch = PlatformDependent.normalizedArch();
262             String osArch = "_" + os + "_" + arch;
263             if (originalName.endsWith(osArch)) {
264                 patched = patchShadedLibraryId(bytes,
265                         originalName.substring(0, originalName.length() - osArch.length()), name);
266             } else {
267                 patched = false;
268             }
269         } else {
270             patched = true;
271         }
272         out.write(bytes, 0, bytes.length);
273         return patched;
274     }

276     private static boolean shouldShadedLibraryIdBePatched(String packagePrefix) {
277         return TRY_TO_PATCH_SHADED_ID && PlatformDependent.isOsx() && !packagePrefix.isEmpty();
278     }

280     /**
281      * Try to patch shaded library to ensure it uses a unique ID.
282      */
283     private static boolean patchShadedLibraryId(byte[] bytes, String originalName, String name) {
284         // Our native libs always have the name as part of their id so we can search for it and replace it
285         // to make the ID unique if shading is used.
286         byte[] nameBytes = originalName.getBytes(CharsetUtil.UTF_8);
287         int idIdx = -1;
288 
289         // Be aware this is a really raw way of patching a dylib but it does all we need without implementing
290         // a full mach-o parser and writer. Basically we just replace the the original bytes with some
291         // random bytes as part of the ID regeneration. The important thing here is that we need to use the same
292         // length to not corrupt the mach-o header.
293         outerLoop: for (int i = 0; i < bytes.length && bytes.length - i >= nameBytes.length; i++) {
294             int idx = i;
295             for (int j = 0; j < nameBytes.length;) {
296                 if (bytes[idx++] != nameBytes[j++]) {
297                     // Did not match the name, increase the index and try again.
298                     break;
299                 } else if (j == nameBytes.length) {
300                     // We found the index within the id.
301                     idIdx = i;
302                     break outerLoop;
303                 }
304             }
305         }
306 
307         if (idIdx == -1) {
308             logger.debug("Was not able to find the ID of the shaded native library {}, can't adjust it.", name);
309             return false;
310         } else {
311             // We found our ID... now monkey-patch it!
312             for (int i = 0; i < nameBytes.length; i++) {
313                 // We should only use bytes as replacement that are in our UNIQUE_ID_BYTES array.
314                 bytes[idIdx + i] = UNIQUE_ID_BYTES[PlatformDependent.threadLocalRandom()
315                                                                     .nextInt(UNIQUE_ID_BYTES.length)];
316             }
317 
318             if (logger.isDebugEnabled()) {
319                 logger.debug(
320                         "Found the ID of the shaded native library {}. Replacing ID part {} with {}",
321                         name, originalName, new String(bytes, idIdx, nameBytes.length, CharsetUtil.UTF_8));
322             }
323             return true;
324         }
325     }

327     /**
328      * Loading the native library into the specified {@link ClassLoader}.
329      * @param loader - The {@link ClassLoader} where the native library will be loaded into
330      * @param name - The native library path or name
331      * @param absolute - Whether the native library will be loaded by path or by name
332      */
333     private static void loadLibrary(final ClassLoader loader, final String name, final boolean absolute) {
334         Throwable suppressed = null;
335         try {
336             try {
337                 // Make sure the helper is belong to the target ClassLoader.
338                 final Class<?> newHelper = tryToLoadClass(loader, NativeLibraryUtil.class);
339                 loadLibraryByHelper(newHelper, name, absolute);
340                 logger.debug("Successfully loaded the library {}", name);
341                 return;
342             } catch (UnsatisfiedLinkError e) { // Should by pass the UnsatisfiedLinkError here!
343                 suppressed = e;
344                 logger.debug("Unable to load the library '{}', trying other loading mechanism.", name, e);
345             } catch (Exception e) {
346                 suppressed = e;
347                 logger.debug("Unable to load the library '{}', trying other loading mechanism.", name, e);
348             }
349             NativeLibraryUtil.loadLibrary(name, absolute);  // Fallback to local helper class.
350             logger.debug("Successfully loaded the library {}", name);
351         } catch (UnsatisfiedLinkError ule) {
352             if (suppressed != null) {
353                 ThrowableUtil.addSuppressed(ule, suppressed);
354             }
355             throw ule;
356         }
357     }

359     private static void loadLibraryByHelper(final Class<?> helper, final String name, final boolean absolute)
360             throws UnsatisfiedLinkError {
361         Object ret = AccessController.doPrivileged(new PrivilegedAction<Object>() {
362             @Override
363             public Object run() {
364                 try {
365                     // Invoke the helper to load the native library, if succeed, then the native
366                     // library belong to the specified ClassLoader.
367                     Method method = helper.getMethod("loadLibrary", String.class, boolean.class);
368                     method.setAccessible(true);
369                     return method.invoke(null, name, absolute);
370                 } catch (Exception e) {
371                     return e;
372                 }
373             }
374         });
375         if (ret instanceof Throwable) {
376             Throwable t = (Throwable) ret;
377             assert !(t instanceof UnsatisfiedLinkError) : t + " should be a wrapper throwable";
378             Throwable cause = t.getCause();
379             if (cause instanceof UnsatisfiedLinkError) {
380                 throw (UnsatisfiedLinkError) cause;
381             }
382             UnsatisfiedLinkError ule = new UnsatisfiedLinkError(t.getMessage());
383             ule.initCause(t);
384             throw ule;
385         }
386     }

388     /**
389      * Try to load the helper {@link Class} into specified {@link ClassLoader}.
390      * @param loader - The {@link ClassLoader} where to load the helper {@link Class}
391      * @param helper - The helper {@link Class}
392      * @return A new helper Class defined in the specified ClassLoader.
393      * @throws ClassNotFoundException Helper class not found or loading failed
394      */
395     private static Class<?> tryToLoadClass(final ClassLoader loader, final Class<?> helper)
396             throws ClassNotFoundException {
397         try {
398             return Class.forName(helper.getName(), false, loader);
399         } catch (ClassNotFoundException e1) {
400             if (loader == null) {
401                 // cannot defineClass inside bootstrap class loader
402                 throw e1;
403             }
404             try {
405                 // The helper class is NOT found in target ClassLoader, we have to define the helper class.
406                 final byte[] classBinary = classToByteArray(helper);
407                 return AccessController.doPrivileged(new PrivilegedAction<Class<?>>() {
408                     @Override
409                     public Class<?> run() {
410                         try {
411                             // Define the helper class in the target ClassLoader,
412                             //  then we can call the helper to load the native library.
413                             Method defineClass = ClassLoader.class.getDeclaredMethod("defineClass", String.class,
414                                     byte[].class, int.class, int.class);
415                             defineClass.setAccessible(true);
416                             return (Class<?>) defineClass.invoke(loader, helper.getName(), classBinary, 0,
417                                     classBinary.length);
418                         } catch (Exception e) {
419                             throw new IllegalStateException("Define class failed!", e);
420                         }
421                     }
422                 });
423             } catch (ClassNotFoundException e2) {
424                 ThrowableUtil.addSuppressed(e2, e1);
425                 throw e2;
426             } catch (RuntimeException e2) {
427                 ThrowableUtil.addSuppressed(e2, e1);
428                 throw e2;
429             } catch (Error e2) {
430                 ThrowableUtil.addSuppressed(e2, e1);
431                 throw e2;
432             }
433         }
434     }

436     /**
437      * Load the helper {@link Class} as a byte array, to be redefined in specified {@link ClassLoader}.
438      * @param clazz - The helper {@link Class} provided by this bundle
439      * @return The binary content of helper {@link Class}.
440      * @throws ClassNotFoundException Helper class not found or loading failed
441      */
442     private static byte[] classToByteArray(Class<?> clazz) throws ClassNotFoundException {
443         String fileName = clazz.getName();
444         int lastDot = fileName.lastIndexOf('.');
445         if (lastDot > 0) {
446             fileName = fileName.substring(lastDot + 1);
447         }
448         URL classUrl = clazz.getResource(fileName + ".class");
449         if (classUrl == null) {
450             throw new ClassNotFoundException(clazz.getName());
451         }
452         byte[] buf = new byte[1024];
453         ByteArrayOutputStream out = new ByteArrayOutputStream(4096);
454         InputStream in = null;
455         try {
456             in = classUrl.openStream();
457             for (int r; (r = in.read(buf)) != -1;) {
458                 out.write(buf, 0, r);
459             }
460             return out.toByteArray();
461         } catch (IOException ex) {
462             throw new ClassNotFoundException(clazz.getName(), ex);
463         } finally {
464             closeQuietly(in);
465             closeQuietly(out);
466         }
467     }

483     private static final class NoexecVolumeDetector {

485         private static boolean canExecuteExecutable(File file) throws IOException {
486             if (PlatformDependent.javaVersion() < 7) {
487                 // Pre-JDK7, the Java API did not directly support POSIX permissions; instead of implementing a custom
488                 // work-around, assume true, which disables the check.
489                 return true;
490             }
491 
492             // If we can already execute, there is nothing to do.
493             if (file.canExecute()) {
494                 return true;
495             }
496 
497             // On volumes, with noexec set, even files with the executable POSIX permissions will fail to execute.
498             // The File#canExecute() method honors this behavior, probaby via parsing the noexec flag when initializing
499             // the UnixFileStore, though the flag is not exposed via a public API.  To find out if library is being
500             // loaded off a volume with noexec, confirm or add executalbe permissions, then check File#canExecute().
501 
502             // Note: We use FQCN to not break when netty is used in java6
503             Set<java.nio.file.attribute.PosixFilePermission> existingFilePermissions =
504                     java.nio.file.Files.getPosixFilePermissions(file.toPath());
505             Set<java.nio.file.attribute.PosixFilePermission> executePermissions =
506                     EnumSet.of(java.nio.file.attribute.PosixFilePermission.OWNER_EXECUTE,
507                             java.nio.file.attribute.PosixFilePermission.GROUP_EXECUTE,
508                             java.nio.file.attribute.PosixFilePermission.OTHERS_EXECUTE);
509             if (existingFilePermissions.containsAll(executePermissions)) {
510                 return false;
511             }
512 
513             Set<java.nio.file.attribute.PosixFilePermission> newPermissions = EnumSet.copyOf(existingFilePermissions);
514             newPermissions.addAll(executePermissions);
515             java.nio.file.Files.setPosixFilePermissions(file.toPath(), newPermissions);
516             return file.canExecute();
517         }
