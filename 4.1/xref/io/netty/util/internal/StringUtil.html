49      static {
50          // Generate the lookup table that converts a byte into a 2-digit hexadecimal integer.
51          int i;
52          for (i = 0; i < 10; i++) {
53              BYTE2HEX_PAD[i] = "0" + i;
54              BYTE2HEX_NOPAD[i] = String.valueOf(i);
55          }
56          for (; i < 16; i++) {
57              char c = (char) ('a' + i - 10);
58              BYTE2HEX_PAD[i] = "0" + c;
59              BYTE2HEX_NOPAD[i] = String.valueOf(c);
60          }
61          for (; i < BYTE2HEX_PAD.length; i++) {
62              String str = Integer.toHexString(i);
63              BYTE2HEX_PAD[i] = str;
64              BYTE2HEX_NOPAD[i] = str;
65          }
66      }

188     /**
189      * Converts the specified byte array into a hexadecimal value and appends it to the specified buffer.
190      */
191     public static <T extends Appendable> T toHexString(T dst, byte[] src, int offset, int length) {
192         assert length >= 0;
193         if (length == 0) {
194             return dst;
195         }
196 
197         final int end = offset + length;
198         final int endMinusOne = end - 1;
199         int i;
200 
201         // Skip preceding zeroes.
202         for (i = offset; i < endMinusOne; i++) {
203             if (src[i] != 0) {
204                 break;
205             }
206         }
207 
208         byteToHexString(dst, src[i++]);
209         int remaining = end - i;
210         toHexStringPadded(dst, src, i, remaining);
211 
212         return dst;
213     }

251     /**
252      * Escapes the specified value, if necessary according to
253      * <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>.
254      *
255      * @param value          The value which will be escaped according to
256      *                       <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>
257      * @param trimWhiteSpace The value will first be trimmed of its optional white-space characters,
258      *                       according to <a href="https://tools.ietf.org/html/rfc7230#section-7">RFC-7230</a>
259      * @return {@link CharSequence} the escaped value if necessary, or the value unchanged
260      */
261     public static CharSequence escapeCsv(CharSequence value, boolean trimWhiteSpace) {
262         int length = checkNotNull(value, "value").length();
263         if (length == 0) {
264             return value;
265         }
266 
267         int start = 0;
268         int last = length - 1;
269         boolean trimmed = false;
270         if (trimWhiteSpace) {
271             start = indexOfFirstNonOwsChar(value, length);
272             if (start == length) {
273                 return EMPTY_STRING;
274             }
275             last = indexOfLastNonOwsChar(value, start, length);
276             trimmed = start > 0 || last < length - 1;
277             if (trimmed) {
278                 length = last - start + 1;
279             }
280         }
281 
282         StringBuilder result = new StringBuilder(length + CSV_NUMBER_ESCAPE_CHARACTERS);
283         boolean quoted = isDoubleQuote(value.charAt(start)) && isDoubleQuote(value.charAt(last)) && length != 1;
284         boolean foundSpecialCharacter = false;
285         boolean escapedDoubleQuote = false;
286         for (int i = start; i <= last; i++) {
287             char current = value.charAt(i);
288             switch (current) {
289                 case DOUBLE_QUOTE:
290                     if (i == start || i == last) {
291                         if (!quoted) {
292                             result.append(DOUBLE_QUOTE);
293                         } else {
294                             continue;
295                         }
296                     } else {
297                         boolean isNextCharDoubleQuote = isDoubleQuote(value.charAt(i + 1));
298                         if (!isDoubleQuote(value.charAt(i - 1)) &&
299                                 (!isNextCharDoubleQuote || i + 1 == last)) {
300                             result.append(DOUBLE_QUOTE);
301                             escapedDoubleQuote = true;
302                         }
303                         break;
304                     }
305                 case LINE_FEED:
306                 case CARRIAGE_RETURN:
307                 case COMMA:
308                     foundSpecialCharacter = true;
309             }
310             result.append(current);
311         }
312 
313         if (escapedDoubleQuote || foundSpecialCharacter && !quoted) {
314             return quote(result);
315         }
316         if (trimmed) {
317             return quoted ? quote(result) : result;
318         }
319         return value;
320     }

322     private static StringBuilder quote(StringBuilder builder) {
323         return builder.insert(0, DOUBLE_QUOTE).append(DOUBLE_QUOTE);
324     }

326     /**
327      * Unescapes the specified escaped CSV field, if necessary according to
328      * <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>.
329      *
330      * @param value The escaped CSV field which will be unescaped according to
331      *              <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>
332      * @return {@link CharSequence} the unescaped value if necessary, or the value unchanged
333      */
334     public static CharSequence unescapeCsv(CharSequence value) {
335         int length = checkNotNull(value, "value").length();
336         if (length == 0) {
337             return value;
338         }
339         int last = length - 1;
340         boolean quoted = isDoubleQuote(value.charAt(0)) && isDoubleQuote(value.charAt(last)) && length != 1;
341         if (!quoted) {
342             validateCsvFormat(value);
343             return value;
344         }
345         StringBuilder unescaped = InternalThreadLocalMap.get().stringBuilder();
346         for (int i = 1; i < last; i++) {
347             char current = value.charAt(i);
348             if (current == DOUBLE_QUOTE) {
349                 if (isDoubleQuote(value.charAt(i + 1)) && (i + 1) != last) {
350                     // Followed by a double-quote but not the last character
351                     // Just skip the next double-quote
352                     i++;
353                 } else {
354                     // Not followed by a double-quote or the following double-quote is the last character
355                     throw newInvalidEscapedCsvFieldException(value, i);
356                 }
357             }
358             unescaped.append(current);
359         }
360         return unescaped.toString();
361     }

363     /**
364      * Unescapes the specified escaped CSV fields according to
365      * <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>.
366      *
367      * @param value A string with multiple CSV escaped fields which will be unescaped according to
368      *              <a href="https://tools.ietf.org/html/rfc4180#section-2">RFC-4180</a>
369      * @return {@link List} the list of unescaped fields
370      */
371     public static List<CharSequence> unescapeCsvFields(CharSequence value) {
372         List<CharSequence> unescaped = new ArrayList<CharSequence>(2);
373         StringBuilder current = InternalThreadLocalMap.get().stringBuilder();
374         boolean quoted = false;
375         int last = value.length() - 1;
376         for (int i = 0; i <= last; i++) {
377             char c = value.charAt(i);
378             if (quoted) {
379                 switch (c) {
380                     case DOUBLE_QUOTE:
381                         if (i == last) {
382                             // Add the last field and return
383                             unescaped.add(current.toString());
384                             return unescaped;
385                         }
386                         char next = value.charAt(++i);
387                         if (next == DOUBLE_QUOTE) {
388                             // 2 double-quotes should be unescaped to one
389                             current.append(DOUBLE_QUOTE);
390                             break;
391                         }
392                         if (next == COMMA) {
393                             // This is the end of a field. Let's start to parse the next field.
394                             quoted = false;
395                             unescaped.add(current.toString());
396                             current.setLength(0);
397                             break;
398                         }
399                         // double-quote followed by other character is invalid
400                         throw newInvalidEscapedCsvFieldException(value, i - 1);
401                     default:
402                         current.append(c);
403                 }
404             } else {
405                 switch (c) {
406                     case COMMA:
407                         // Start to parse the next field
408                         unescaped.add(current.toString());
409                         current.setLength(0);
410                         break;
411                     case DOUBLE_QUOTE:
412                         if (current.length() == 0) {
413                             quoted = true;
414                             break;
415                         }
416                         // double-quote appears without being enclosed with double-quotes
417                     case LINE_FEED:
418                     case CARRIAGE_RETURN:
419                         // special characters appears without being enclosed with double-quotes
420                         throw newInvalidEscapedCsvFieldException(value, i);
421                     default:
422                         current.append(c);
423                 }
424             }
425         }
426         if (quoted) {
427             throw newInvalidEscapedCsvFieldException(value, last);
428         }
429         unescaped.add(current.toString());
430         return unescaped;
431     }

433     /**
434      * Validate if {@code value} is a valid csv field without double-quotes.
435      *
436      * @throws IllegalArgumentException if {@code value} needs to be encoded with double-quotes.
437      */
438     private static void validateCsvFormat(CharSequence value) {
439         int length = value.length();
440         for (int i = 0; i < length; i++) {
441             switch (value.charAt(i)) {
442                 case DOUBLE_QUOTE:
443                 case LINE_FEED:
444                 case CARRIAGE_RETURN:
445                 case COMMA:
446                     // If value contains any special character, it should be enclosed with double-quotes
447                     throw newInvalidEscapedCsvFieldException(value, i);
448                 default:
449             }
450         }
451     }

515     /**
516      * Trim optional white-space characters from the specified value,
517      * according to <a href="https://tools.ietf.org/html/rfc7230#section-7">RFC-7230</a>.
518      *
519      * @param value the value to trim
520      * @return {@link CharSequence} the trimmed value if necessary, or the value unchanged
521      */
522     public static CharSequence trimOws(CharSequence value) {
523         final int length = value.length();
524         if (length == 0) {
525             return value;
526         }
527         int start = indexOfFirstNonOwsChar(value, length);
528         int end = indexOfLastNonOwsChar(value, start, length);
529         return start == 0 && end == length - 1 ? value : value.subSequence(start, end + 1);
530     }

532     /**
533      * @return {@code length} if no OWS is found.
534      */
535     private static int indexOfFirstNonOwsChar(CharSequence value, int length) {
536         int i = 0;
537         while (i < length && isOws(value.charAt(i))) {
538             i++;
539         }
540         return i;
541     }

543     /**
544      * @return {@code start} if no OWS is found.
545      */
546     private static int indexOfLastNonOwsChar(CharSequence value, int start, int length) {
547         int i = length - 1;
548         while (i > start && isOws(value.charAt(i))) {
549             i--;
550         }
551         return i;
552     }
