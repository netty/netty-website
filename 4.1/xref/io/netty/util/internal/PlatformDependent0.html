555             long size = Math.min(length, UNSAFE_COPY_THRESHOLD);
556             UNSAFE.copyMemory(srcAddr, dstAddr, size);
557             length -= size;
558             srcAddr += size;
559             dstAddr += size;

582     static boolean equals(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
583         if (length <= 0) {
584             return true;
585         }
586         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
587         final long baseOffset2 = BYTE_ARRAY_BASE_OFFSET + startPos2;
588         int remainingBytes = length & 7;
589         final long end = baseOffset1 + remainingBytes;
590         for (long i = baseOffset1 - 8 + length, j = baseOffset2 - 8 + length; i >= end; i -= 8, j -= 8) {
591             if (UNSAFE.getLong(bytes1, i) != UNSAFE.getLong(bytes2, j)) {
592                 return false;
593             }
594         }
595 
596         if (remainingBytes >= 4) {
597             remainingBytes -= 4;
598             if (UNSAFE.getInt(bytes1, baseOffset1 + remainingBytes) !=
599                 UNSAFE.getInt(bytes2, baseOffset2 + remainingBytes)) {
600                 return false;
601             }
602         }
603         if (remainingBytes >= 2) {
604             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2) &&
605                    (remainingBytes == 2 || bytes1[startPos1 + 2] == bytes2[startPos2 + 2]);
606         }
607         return bytes1[startPos1] == bytes2[startPos2];
608     }

651     static boolean isZero(byte[] bytes, int startPos, int length) {
652         if (length <= 0) {
653             return true;
654         }
655         final long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
656         int remainingBytes = length & 7;
657         final long end = baseOffset + remainingBytes;
658         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {

660                 return false;
661             }
662         }

664         if (remainingBytes >= 4) {
665             remainingBytes -= 4;

668             }

677     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
678         int hash = HASH_CODE_ASCII_SEED;
679         final long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
680         final int remainingBytes = length & 7;
681         final long end = baseOffset + remainingBytes;
682         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
683             hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
684         }
685         switch(remainingBytes) {
686         case 7:
687             return ((hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset)))
688                           * HASH_CODE_C2 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset + 1)))
689                           * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 3));
690         case 6:
691             return (hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset)))
692                          * HASH_CODE_C2 + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 2));
693         case 5:
694             return (hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset)))
695                          * HASH_CODE_C2 + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 1));
696         case 4:
697             return hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
698         case 3:
699             return (hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset)))
700                          * HASH_CODE_C2 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset + 1));
701         case 2:
702             return hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
703         case 1:
704             return hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
705         default:
706             return hash;
707         }

710     static int hashCodeAsciiCompute(long value, int hash) {
711         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
712         // code the same regardless of character case (upper case or lower case hash is the same).
713         return hash * HASH_CODE_C1 +
714                 // Low order int
715                 hashCodeAsciiSanitize((int) value) * HASH_CODE_C2 +
716                 // High order int
717                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32);
718     }
