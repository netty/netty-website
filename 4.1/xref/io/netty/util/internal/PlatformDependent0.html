578     /**
579      * @param thread The thread to be checked.
580      * @return {@code true} if this {@link Thread} is a virtual thread, {@code false} otherwise.
581      */
582     static boolean isVirtualThread(Thread thread) {
583         // Quick exclusion:
584         if (thread == null || IS_VIRTUAL_THREAD_METHOD == null) {
585             return false;
586         }
587         // Try fast check:
588         if (BASE_VIRTUAL_THREAD_CLASS != null) {
589             return BASE_VIRTUAL_THREAD_CLASS.isInstance(thread);
590         }
591         if (thread instanceof FastThreadLocalThread) {
592             return false;
593         }
594         Class<?> clazz = thread.getClass();
595         // Common cases:
596         if (clazz == Thread.class || clazz.getSuperclass() == Thread.class) {
597             return false;
598         }
599         try {
600             return (Boolean) IS_VIRTUAL_THREAD_METHOD.invoke(thread);
601         } catch (Throwable t) {
602             // Should not happen.
603             if (t instanceof Error) {
604                 throw (Error) t;
605             }
606             throw new Error(t);
607         }
608     }

620         String reason = "io.netty.noUnsafe";
621         String unspecified = "<unspecified>";
622         String unsafeMemoryAccess = SystemPropertyUtil.get("sun.misc.unsafe.memory.access", unspecified);
623         if (!("allow".equals(unsafeMemoryAccess) || unspecified.equals(unsafeMemoryAccess))) {
624             reason = "--sun-misc-unsafe-memory-access=" + unsafeMemoryAccess;
625             noUnsafe = true;
626         }
627 
628         if (noUnsafe) {
629             String msg = "sun.misc.Unsafe: unavailable (" + reason + ')';
630             logger.debug(msg);
631             return new UnsupportedOperationException(msg);
632         }
633 
634         // Legacy properties
635         String unsafePropName;
636         if (SystemPropertyUtil.contains("io.netty.tryUnsafe")) {
637             unsafePropName = "io.netty.tryUnsafe";
638         } else {
639             unsafePropName = "org.jboss.netty.tryUnsafe";
640         }
641 
642         if (!SystemPropertyUtil.getBoolean(unsafePropName, true)) {
643             String msg = "sun.misc.Unsafe: unavailable (" + unsafePropName + ')';
644             logger.debug(msg);
645             return new UnsupportedOperationException(msg);
646         }
647 
648         return null;
649     }

910     static boolean equals(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
911         int remainingBytes = length & 7;
912         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
913         final long diff = startPos2 - startPos1;
914         if (length >= 8) {
915             final long end = baseOffset1 + remainingBytes;
916             for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
917                 if (UNSAFE.getLong(bytes1, i) != UNSAFE.getLong(bytes2, i + diff)) {
918                     return false;
919                 }
920             }
921         }
922         if (remainingBytes >= 4) {
923             remainingBytes -= 4;
924             long pos = baseOffset1 + remainingBytes;
925             if (UNSAFE.getInt(bytes1, pos) != UNSAFE.getInt(bytes2, pos + diff)) {
926                 return false;
927             }
928         }
929         final long baseOffset2 = baseOffset1 + diff;
930         if (remainingBytes >= 2) {
931             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2) &&
932                     (remainingBytes == 2 ||
933                     UNSAFE.getByte(bytes1, baseOffset1 + 2) == UNSAFE.getByte(bytes2, baseOffset2 + 2));
934         }
935         return remainingBytes == 0 ||
936                 UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
937     }

939     static int equalsConstantTime(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
940         long result = 0;
941         long remainingBytes = length & 7;
942         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
943         final long end = baseOffset1 + remainingBytes;
944         final long diff = startPos2 - startPos1;
945         for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
946             result |= UNSAFE.getLong(bytes1, i) ^ UNSAFE.getLong(bytes2, i + diff);
947         }
948         if (remainingBytes >= 4) {
949             result |= UNSAFE.getInt(bytes1, baseOffset1) ^ UNSAFE.getInt(bytes2, baseOffset1 + diff);
950             remainingBytes -= 4;
951         }
952         if (remainingBytes >= 2) {
953             long pos = end - remainingBytes;
954             result |= UNSAFE.getChar(bytes1, pos) ^ UNSAFE.getChar(bytes2, pos + diff);
955             remainingBytes -= 2;
956         }
957         if (remainingBytes == 1) {
958             long pos = end - 1;
959             result |= UNSAFE.getByte(bytes1, pos) ^ UNSAFE.getByte(bytes2, pos + diff);
960         }
961         return ConstantTimeUtils.equalsConstantTime(result, 0);

967         }

974             }

978             remainingBytes -= 4;

990     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
991         int hash = HASH_CODE_ASCII_SEED;
992         long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
993         final int remainingBytes = length & 7;
994         final long end = baseOffset + remainingBytes;
995         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
996             hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
997         }
998         if (remainingBytes == 0) {
999             return hash;
1000         }
1001         int hcConst = HASH_CODE_C1;
1002         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
1003             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
1004             hcConst = HASH_CODE_C2;
1005             baseOffset++;
1006         }
1007         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
1008             hash = hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
1009             hcConst = hcConst == HASH_CODE_C1 ? HASH_CODE_C2 : HASH_CODE_C1;
1010             baseOffset += 2;
1011         }
1012         if (remainingBytes >= 4) { // 4, 5, 6, 7
1013             return hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
1014         }
1015         return hash;
1016     }

1018     static int hashCodeAsciiCompute(long value, int hash) {
1019         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
1020         // code the same regardless of character case (upper case or lower case hash is the same).
1021         return hash * HASH_CODE_C1 +
1022                 // Low order int
1023                 hashCodeAsciiSanitize((int) value) * HASH_CODE_C2 +
1024                 // High order int
1025                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32);
1026     }
