53       * {@code true} if and only if the platform supports unaligned access.
54       *
55       * @see <a href="http://en.wikipedia.org/wiki/Segmentation_fault#Bus_error">Wikipedia on segfault</a>
56       */

59      static {
60          ByteBuffer direct = ByteBuffer.allocateDirect(1);
61          Field addressField;
62          try {
63              addressField = Buffer.class.getDeclaredField("address");
64              addressField.setAccessible(true);
65              if (addressField.getLong(ByteBuffer.allocate(1)) != 0) {
66                  // A heap buffer must have 0 address.
67                  addressField = null;
68              } else {
69                  if (addressField.getLong(direct) == 0) {
70                      // A direct buffer must have non-zero address.
71                      addressField = null;
72                  }
73              }
74          } catch (Throwable t) {
75              // Failed to access the address field.
76              addressField = null;
77          }
78          logger.debug("java.nio.Buffer.address: {}", addressField != null? "available" : "unavailable");
79  
80          Unsafe unsafe;
81          if (addressField != null) {
82              try {
83                  Field unsafeField = Unsafe.class.getDeclaredField("theUnsafe");
84                  unsafeField.setAccessible(true);
85                  unsafe = (Unsafe) unsafeField.get(null);
86                  logger.debug("sun.misc.Unsafe.theUnsafe: {}", unsafe != null ? "available" : "unavailable");
87  
88                  // Ensure the unsafe supports all necessary methods to work around the mistake in the latest OpenJDK.
89                  // https://github.com/netty/netty/issues/1061
90                  // http://www.mail-archive.com/jdk6-dev@openjdk.java.net/msg00698.html
91                  try {
92                      if (unsafe != null) {
93                          unsafe.getClass().getDeclaredMethod(
94                                  "copyMemory", Object.class, long.class, Object.class, long.class, long.class);
95                          logger.debug("sun.misc.Unsafe.copyMemory: available");
96                      }
97                  } catch (NoSuchMethodError t) {
98                      logger.debug("sun.misc.Unsafe.copyMemory: unavailable");
99                      throw t;
100                 } catch (NoSuchMethodException e) {
101                     logger.debug("sun.misc.Unsafe.copyMemory: unavailable");
102                     throw e;
103                 }
104             } catch (Throwable cause) {
105                 // Unsafe.copyMemory(Object, long, Object, long, long) unavailable.
106                 unsafe = null;
107             }
108         } else {
109             // If we cannot access the address of a direct buffer, there's no point of using unsafe.
110             // Let's just pretend unsafe is unavailable for overall simplicity.
111             unsafe = null;
112         }
113 
114         UNSAFE = unsafe;
115 
116         if (unsafe == null) {
117             ADDRESS_FIELD_OFFSET = -1;
118             ARRAY_BASE_OFFSET = -1;
119             UNALIGNED = false;
120         } else {
121             ADDRESS_FIELD_OFFSET = objectFieldOffset(addressField);
122             ARRAY_BASE_OFFSET = UNSAFE.arrayBaseOffset(byte[].class);
123             boolean unaligned;
124             try {
125                 Class<?> bitsClass = Class.forName("java.nio.Bits", false, ClassLoader.getSystemClassLoader());
126                 Method unalignedMethod = bitsClass.getDeclaredMethod("unaligned");
127                 unalignedMethod.setAccessible(true);
128                 unaligned = Boolean.TRUE.equals(unalignedMethod.invoke(null));
129             } catch (Throwable t) {
130                 // We at least know x86 and x64 support unaligned access.
131                 String arch = SystemPropertyUtil.get("os.arch", "");
132                 //noinspection DynamicRegexReplaceableByCompiledPattern
133                 unaligned = arch.matches("^(i[3-6]86|x86(_64)?|x64|amd64)$");
134             }
135 
136             UNALIGNED = unaligned;
137             logger.debug("java.nio.Bits.unaligned: {}", UNALIGNED);
138         }
139     }

302     static void copyMemory(long srcAddr, long dstAddr, long length) {
303         //UNSAFE.copyMemory(srcAddr, dstAddr, length);
304         while (length > 0) {
305             long size = Math.min(length, UNSAFE_COPY_THRESHOLD);
306             UNSAFE.copyMemory(srcAddr, dstAddr, size);
307             length -= size;
308             srcAddr += size;
309             dstAddr += size;
310         }
311     }

324     static boolean equals(byte[] bytes1, int startPos1, int endPos1, byte[] bytes2, int startPos2, int endPos2) {
325         final int len1 = endPos1 - startPos1;
326         final int len2 = endPos2 - startPos2;
327         if (len1 != len2) {
328             return false;
329         }
330         if (len1 == 0) {
331             return true;
332         }
333         final long baseOffset1 = ARRAY_BASE_OFFSET + startPos1;
334         final long baseOffset2 = ARRAY_BASE_OFFSET + startPos2;
335         int remainingBytes = len1 & 7;
336         for (int i = len1 - 8; i >= remainingBytes; i -= 8) {
337             if (UNSAFE.getLong(bytes1, baseOffset1 + i) != UNSAFE.getLong(bytes2, baseOffset2 + i)) {
338                 return false;
339             }
340         }
341         if (remainingBytes >= 4) {
342             remainingBytes -= 4;
343             if (UNSAFE.getInt(bytes1, baseOffset1 + remainingBytes) !=
344                 UNSAFE.getInt(bytes2, baseOffset2 + remainingBytes)) {
345                 return false;
346             }
347         }
348         if (remainingBytes >= 2) {
349             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2) &&
350                    (remainingBytes == 2 || bytes1[startPos1 + 2] == bytes2[startPos2 + 2]);
351         }
352         return bytes1[startPos1] == bytes2[startPos2];
353     }
