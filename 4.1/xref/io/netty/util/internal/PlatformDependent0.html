727     static boolean equals(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
728         int remainingBytes = length & 7;
729         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
730         final long diff = startPos2 - startPos1;
731         if (length >= 8) {
732             final long end = baseOffset1 + remainingBytes;
733             for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
734                 if (UNSAFE.getLong(bytes1, i) != UNSAFE.getLong(bytes2, i + diff)) {
735                     return false;
736                 }
737             }
738         }
739         if (remainingBytes >= 4) {
740             remainingBytes -= 4;
741             long pos = baseOffset1 + remainingBytes;
742             if (UNSAFE.getInt(bytes1, pos) != UNSAFE.getInt(bytes2, pos + diff)) {
743                 return false;
744             }
745         }
746         final long baseOffset2 = baseOffset1 + diff;
747         if (remainingBytes >= 2) {
748             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2) &&
749                     (remainingBytes == 2 ||
750                     UNSAFE.getByte(bytes1, baseOffset1 + 2) == UNSAFE.getByte(bytes2, baseOffset2 + 2));
751         }
752         return remainingBytes == 0 ||
753                 UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
754     }

756     static int equalsConstantTime(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
757         long result = 0;
758         long remainingBytes = length & 7;
759         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
760         final long end = baseOffset1 + remainingBytes;
761         final long diff = startPos2 - startPos1;
762         for (long i = baseOffset1 - 8 + length; i >= end; i -= 8) {
763             result |= UNSAFE.getLong(bytes1, i) ^ UNSAFE.getLong(bytes2, i + diff);
764         }
765         if (remainingBytes >= 4) {
766             result |= UNSAFE.getInt(bytes1, baseOffset1) ^ UNSAFE.getInt(bytes2, baseOffset1 + diff);
767             remainingBytes -= 4;
768         }
769         if (remainingBytes >= 2) {
770             long pos = end - remainingBytes;
771             result |= UNSAFE.getChar(bytes1, pos) ^ UNSAFE.getChar(bytes2, pos + diff);
772             remainingBytes -= 2;
773         }
774         if (remainingBytes == 1) {
775             long pos = end - 1;
776             result |= UNSAFE.getByte(bytes1, pos) ^ UNSAFE.getByte(bytes2, pos + diff);
777         }
778         return ConstantTimeUtils.equalsConstantTime(result, 0);
779     }

781     static boolean isZero(byte[] bytes, int startPos, int length) {
782         if (length <= 0) {
783             return true;
784         }
785         final long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
786         int remainingBytes = length & 7;
787         final long end = baseOffset + remainingBytes;
788         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
789             if (UNSAFE.getLong(bytes, i) != 0) {
790                 return false;
791             }
792         }
793 
794         if (remainingBytes >= 4) {
795             remainingBytes -= 4;
796             if (UNSAFE.getInt(bytes, baseOffset + remainingBytes) != 0) {
797                 return false;
798             }
799         }
800         if (remainingBytes >= 2) {
801             return UNSAFE.getChar(bytes, baseOffset) == 0 &&
802                     (remainingBytes == 2 || bytes[startPos + 2] == 0);
803         }
804         return bytes[startPos] == 0;
805     }

807     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
808         int hash = HASH_CODE_ASCII_SEED;
809         long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
810         final int remainingBytes = length & 7;
811         final long end = baseOffset + remainingBytes;
812         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
813             hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
814         }
815         if (remainingBytes == 0) {
816             return hash;
817         }
818         int hcConst = HASH_CODE_C1;
819         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
820             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
821             hcConst = HASH_CODE_C2;
822             baseOffset++;
823         }
824         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
825             hash = hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
826             hcConst = hcConst == HASH_CODE_C1 ? HASH_CODE_C2 : HASH_CODE_C1;
827             baseOffset += 2;
828         }
829         if (remainingBytes >= 4) { // 4, 5, 6, 7
830             return hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
831         }
832         return hash;
833     }

835     static int hashCodeAsciiCompute(long value, int hash) {
836         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
837         // code the same regardless of character case (upper case or lower case hash is the same).
838         return hash * HASH_CODE_C1 +
839                 // Low order int
840                 hashCodeAsciiSanitize((int) value) * HASH_CODE_C2 +
841                 // High order int
842                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32);
843     }
