14      {
15          // 2 cache lines pad
16          REF_BUFFER_PAD = (PortableJvmInfo.CACHE_LINE_SIZE * 2) >> UnsafeRefArrayAccess.REF_ELEMENT_SHIFT;
17          // Including the buffer pad in the array base offset
18          final int paddingOffset = REF_BUFFER_PAD << UnsafeRefArrayAccess.REF_ELEMENT_SHIFT;
19          REF_ARRAY_BASE = UnsafeRefArrayAccess.REF_ARRAY_BASE + paddingOffset;
20      }
21  
22      @SuppressWarnings("unchecked")
23      public static <E> E[] allocate(int capacity)
24      {
25          return (E[]) new Object[capacity + REF_BUFFER_PAD * 2];
26      }
27  
28      /**
29       * @param index desirable element index
30       * @param mask (length - 1)
31       * @return the offset in bytes within the array for a given index.
32       */
33      protected static long calcElementOffset(long index, long mask)
34      {
35          return REF_ARRAY_BASE + ((index & mask) << UnsafeRefArrayAccess.REF_ELEMENT_SHIFT);
