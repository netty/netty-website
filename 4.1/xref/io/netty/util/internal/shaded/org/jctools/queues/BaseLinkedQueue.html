141 /**
142  * A base data structure for concurrent linked queues. For convenience also pulled in common single consumer
143  * methods since at this time there's no plan to implement MC.
144  */
145 abstract class BaseLinkedQueue<E> extends BaseLinkedQueuePad2<E>
146 {
147 
148     @Override
149     public final Iterator<E> iterator()
150     {
151         throw new UnsupportedOperationException();
152     }
153 
154     @Override
155     public String toString()
156     {
157         return this.getClass().getName();
158     }
159 
160     protected final LinkedQueueNode<E> newNode()
161     {
162         return new LinkedQueueNode<E>();
163     }
164 
165     protected final LinkedQueueNode<E> newNode(E e)
166     {
167         return new LinkedQueueNode<E>(e);
168     }

170     /**
171      * {@inheritDoc} <br>
172      * <p>
173      * IMPLEMENTATION NOTES:<br>
174      * This is an O(n) operation as we run through all the nodes and count them.<br>
175      * The accuracy of the value returned by this method is subject to races with producer/consumer threads. In
176      * particular when racing with the consumer thread this method may under estimate the size.<br>
177      *
178      * @see java.util.Queue#size()
179      */
180     @Override
181     public final int size()
182     {
183         // Read consumer first, this is important because if the producer is node is 'older' than the consumer
184         // the consumer may overtake it (consume past it) invalidating the 'snapshot' notion of size.
185         LinkedQueueNode<E> chaserNode = lvConsumerNode();
186         LinkedQueueNode<E> producerNode = lvProducerNode();
187         int size = 0;
188         // must chase the nodes all the way to the producer node, but there's no need to count beyond expected head.
189         while (chaserNode != producerNode && // don't go passed producer node
190             chaserNode != null && // stop at last node
191             size < Integer.MAX_VALUE) // stop at max int
192         {
193             LinkedQueueNode<E> next;
194             next = chaserNode.lvNext();
195             // check if this node has been consumed, if so return what we have
196             if (next == chaserNode)
197             {
198                 return size;
199             }
200             chaserNode = next;
201             size++;
202         }
203         return size;
204     }
205 
206     /**
207      * {@inheritDoc} <br>
208      * <p>
209      * IMPLEMENTATION NOTES:<br>
210      * Queue is empty when producerNode is the same as consumerNode. An alternative implementation would be to
211      * observe the producerNode.value is null, which also means an empty queue because only the
212      * consumerNode.value is allowed to be null.
213      *
214      * @see MessagePassingQueue#isEmpty()
215      */
216     @Override
217     public boolean isEmpty()
218     {
219         LinkedQueueNode<E> consumerNode = lvConsumerNode();
220         LinkedQueueNode<E> producerNode = lvProducerNode();
221         return consumerNode == producerNode;
222     }
223 
224     protected E getSingleConsumerNodeValue(LinkedQueueNode<E> currConsumerNode, LinkedQueueNode<E> nextNode)
225     {
226         // we have to null out the value because we are going to hang on to the node
227         final E nextValue = nextNode.getAndNullValue();
228 
229         // Fix up the next ref of currConsumerNode to prevent promoted nodes from keeping new ones alive.
230         // We use a reference to self instead of null because null is already a meaningful value (the next of
231         // producer node is null).
232         currConsumerNode.soNext(currConsumerNode);
233         spConsumerNode(nextNode);
234         // currConsumerNode is now no longer referenced and can be collected
235         return nextValue;
236     }
237 
238     /**
239      * {@inheritDoc} <br>
240      * <p>
241      * IMPLEMENTATION NOTES:<br>
242      * Poll is allowed from a SINGLE thread.<br>
243      * Poll is potentially blocking here as the {@link Queue#poll()} does not allow returning {@code null} if the queue is not
244      * empty. This is very different from the original Vyukov guarantees. See {@link #relaxedPoll()} for the original
245      * semantics.<br>
246      * Poll reads {@code consumerNode.next} and:
247      * <ol>
248      * <li>If it is {@code null} AND the queue is empty return {@code null}, <b>if queue is not empty spin wait for
249      * value to become visible</b>.
250      * <li>If it is not {@code null} set it as the consumer node and return it's now evacuated value.
251      * </ol>
252      * This means the consumerNode.value is always {@code null}, which is also the starting point for the queue.
253      * Because {@code null} values are not allowed to be offered this is the only node with it's value set to
254      * {@code null} at any one time.
255      *
256      * @see MessagePassingQueue#poll()
257      * @see java.util.Queue#poll()
258      */
259     @Override
260     public E poll()
261     {
262         final LinkedQueueNode<E> currConsumerNode = lpConsumerNode();
263         LinkedQueueNode<E> nextNode = currConsumerNode.lvNext();
264         if (nextNode != null)
265         {
266             return getSingleConsumerNodeValue(currConsumerNode, nextNode);
267         }
268         else if (currConsumerNode != lvProducerNode())
269         {
270             nextNode = spinWaitForNextNode(currConsumerNode);
271             // got the next node...
272             return getSingleConsumerNodeValue(currConsumerNode, nextNode);
273         }
274         return null;
275     }
276 
277     /**
278      * {@inheritDoc} <br>
279      * <p>
280      * IMPLEMENTATION NOTES:<br>
281      * Peek is allowed from a SINGLE thread.<br>
282      * Peek is potentially blocking here as the {@link Queue#peek()} does not allow returning {@code null} if the queue is not
283      * empty. This is very different from the original Vyukov guarantees. See {@link #relaxedPeek()} for the original
284      * semantics.<br>
285      * Poll reads the next node from the consumerNode and:
286      * <ol>
287      * <li>If it is {@code null} AND the queue is empty return {@code null}, <b>if queue is not empty spin wait for
288      * value to become visible</b>.
289      * <li>If it is not {@code null} return it's value.
290      * </ol>
291      *
292      * @see MessagePassingQueue#peek()
293      * @see java.util.Queue#peek()
294      */
295     @Override
296     public E peek()
297     {
298         final LinkedQueueNode<E> currConsumerNode = lpConsumerNode();
299         LinkedQueueNode<E> nextNode = currConsumerNode.lvNext();
300         if (nextNode != null)
301         {
302             return nextNode.lpValue();
303         }
304         else if (currConsumerNode != lvProducerNode())
305         {
306             nextNode = spinWaitForNextNode(currConsumerNode);
307             // got the next node...
308             return nextNode.lpValue();
309         }
310         return null;
311     }
312 
313     LinkedQueueNode<E> spinWaitForNextNode(LinkedQueueNode<E> currNode)
314     {
315         LinkedQueueNode<E> nextNode;
316         while ((nextNode = currNode.lvNext()) == null)
317         {
318             // spin, we are no longer wait free
319         }
320         return nextNode;
321     }
322 
323     @Override
324     public E relaxedPoll()
325     {
326         final LinkedQueueNode<E> currConsumerNode = lpConsumerNode();
327         final LinkedQueueNode<E> nextNode = currConsumerNode.lvNext();
328         if (nextNode != null)
329         {
330             return getSingleConsumerNodeValue(currConsumerNode, nextNode);
331         }
332         return null;
333     }
334 
335     @Override
336     public E relaxedPeek()
337     {
338         final LinkedQueueNode<E> nextNode = lpConsumerNode().lvNext();
339         if (nextNode != null)
340         {
341             return nextNode.lpValue();
342         }
343         return null;
344     }

346     @Override
347     public boolean relaxedOffer(E e)
348     {
349         return offer(e);
350     }

352     @Override
353     public int drain(Consumer<E> c, int limit)
354     {
355         if (null == c)
356             throw new IllegalArgumentException("c is null");
357         if (limit < 0)
358             throw new IllegalArgumentException("limit is negative: " + limit);
359         if (limit == 0)
360             return 0;
361 
362         LinkedQueueNode<E> chaserNode = this.lpConsumerNode();
363         for (int i = 0; i < limit; i++)
364         {
365             final LinkedQueueNode<E> nextNode = chaserNode.lvNext();
366 
367             if (nextNode == null)
368             {
369                 return i;
370             }
371             // we have to null out the value because we are going to hang on to the node
372             final E nextValue = getSingleConsumerNodeValue(chaserNode, nextNode);
373             chaserNode = nextNode;
374             c.accept(nextValue);
375         }
376         return limit;
377     }

383     }

389     }
