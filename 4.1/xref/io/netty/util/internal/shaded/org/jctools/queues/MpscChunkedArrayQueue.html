24  abstract class MpscChunkedArrayQueueColdProducerFields<E> extends BaseMpscLinkedArrayQueue<E>
25  {
26      protected final long maxQueueCapacity;
27  
28      MpscChunkedArrayQueueColdProducerFields(int initialCapacity, int maxCapacity)
29      {
30          super(initialCapacity);
31          RangeUtil.checkGreaterThanOrEqual(maxCapacity, 4, "maxCapacity");
32          RangeUtil.checkLessThan(roundToPowerOfTwo(initialCapacity), roundToPowerOfTwo(maxCapacity),
33              "initialCapacity");
34          maxQueueCapacity = ((long) Pow2.roundToPowerOfTwo(maxCapacity)) << 1;
35      }
36  }

38  /**
39   * An MPSC array queue which starts at <i>initialCapacity</i> and grows to <i>maxCapacity</i> in linked chunks
40   * of the initial size. The queue grows only when the current chunk is full and elements are not copied on
41   * resize, instead a link to the new chunk is stored in the old chunk for the consumer to follow.
42   */

62      public MpscChunkedArrayQueue(int maxCapacity)

67      /**
68       * @param initialCapacity the queue initial capacity. If chunk size is fixed this will be the chunk size.
69       *                        Must be 2 or more.
70       * @param maxCapacity     the maximum capacity will be rounded up to the closest power of 2 and will be the
71       *                        upper limit of number of elements in this queue. Must be 4 or more and round up to a larger
72       *                        power of 2 than initialCapacity.

74      public MpscChunkedArrayQueue(int initialCapacity, int maxCapacity)

79      @Override
