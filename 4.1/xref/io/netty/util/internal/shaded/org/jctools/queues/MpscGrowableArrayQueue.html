22  /**
23   * An MPSC array queue which starts at <i>initialCapacity</i> and grows to <i>maxCapacity</i> in linked chunks,
24   * doubling theirs size every time until the full blown backing array is used.
25   * The queue grows only when the current chunk is full and elements are not copied on
26   * resize, instead a link to the new chunk is stored in the old chunk for the consumer to follow.
27   */
28  public class MpscGrowableArrayQueue<E> extends MpscChunkedArrayQueue<E>
29  {
30  
31      public MpscGrowableArrayQueue(int maxCapacity)
32      {
33          super(Math.max(2, Pow2.roundToPowerOfTwo(maxCapacity / 8)), maxCapacity);
34      }
35  
36      /**
37       * @param initialCapacity the queue initial capacity. If chunk size is fixed this will be the chunk size.
38       *                        Must be 2 or more.
39       * @param maxCapacity     the maximum capacity will be rounded up to the closest power of 2 and will be the
40       *                        upper limit of number of elements in this queue. Must be 4 or more and round up to a larger
41       *                        power of 2 than initialCapacity.
42       */
43      public MpscGrowableArrayQueue(int initialCapacity, int maxCapacity)
44      {
45          super(initialCapacity, maxCapacity);
46      }
47  
48  
49      @Override
50      protected int getNextBufferSize(E[] buffer)
51      {
52          final long maxSize = maxQueueCapacity / 2;
53          RangeUtil.checkLessThanOrEqual(length(buffer), maxSize, "buffer.length");
54          final int newSize = 2 * (length(buffer) - 1);
55          return newSize + 1;
56      }
57  
58      @Override
59      protected long getCurrentBufferCapacity(long mask)
60      {
61          return (mask + 2 == maxQueueCapacity) ? maxQueueCapacity : mask;
62      }
63  }
