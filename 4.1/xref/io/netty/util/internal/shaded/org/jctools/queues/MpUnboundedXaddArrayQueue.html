214 abstract class MpUnboundedXaddArrayQueue<R extends MpUnboundedXaddChunk<R,E>, E>
215     extends MpUnboundedXaddArrayQueuePad5<R, E>
216     implements MessagePassingQueue<E>, QueueProgressIndicators
217 {
218     // it must be != MpUnboundedXaddChunk.NOT_USED
219     private static final long ROTATION = -2;
220     final int chunkMask;
221     final int chunkShift;
222     final int maxPooledChunks;
223     final SpscArrayQueue<R> freeChunksPool;
224 
225     /**
226      * @param chunkSize The buffer size to be used in each chunk of this queue
227      * @param maxPooledChunks The maximum number of reused chunks kept around to avoid allocation, chunks are pre-allocated
228      */
229     MpUnboundedXaddArrayQueue(int chunkSize, int maxPooledChunks)
230     {
231         if (!UnsafeAccess.SUPPORTS_GET_AND_ADD_LONG)
232         {
233             throw new IllegalStateException("Unsafe::getAndAddLong support (JDK 8+) is required for this queue to work");
234         }
235         if (maxPooledChunks < 0)
236         {
237             throw new IllegalArgumentException("Expecting a positive maxPooledChunks, but got:"+maxPooledChunks);
238         }
239         chunkSize = Pow2.roundToPowerOfTwo(chunkSize);
240 
241         this.chunkMask = chunkSize - 1;
242         this.chunkShift = Integer.numberOfTrailingZeros(chunkSize);
243         freeChunksPool = new SpscArrayQueue<R>(maxPooledChunks);
244 
245         final R first = newChunk(0, null, chunkSize, maxPooledChunks > 0);
246         soProducerChunk(first);
247         soProducerChunkIndex(0);
248         soConsumerChunk(first);
249         for (int i = 1; i < maxPooledChunks; i++)
250         {
251             freeChunksPool.offer(newChunk(NOT_USED, null, chunkSize, true));
252         }
253         this.maxPooledChunks = maxPooledChunks;
254     }
255

257     {
258         return chunkMask + 1;
259     }

280     /**
281      * We're here because currentChunk.index doesn't match the expectedChunkIndex. To resolve we must now chase the linked
282      * chunks to the appropriate chunk. More than one producer may end up racing to add or discover new chunks.
283      *
284      * @param initialChunk the starting point chunk, which does not match the required chunk index
285      * @param requiredChunkIndex the chunk index we need
286      * @return the chunk matching the required index
287      */
288     final R producerChunkForIndex(
289         final R initialChunk,
290         final long requiredChunkIndex)
291     {
292         R currentChunk = initialChunk;
293         long jumpBackward;
294         while (true)
295         {
296             if (currentChunk == null)
297             {
298                 currentChunk = lvProducerChunk();
299             }
300             final long currentChunkIndex = currentChunk.lvIndex();
301             assert currentChunkIndex != NOT_USED;
302             // if the required chunk index is less than the current chunk index then we need to walk the linked list of
303             // chunks back to the required index
304             jumpBackward = currentChunkIndex - requiredChunkIndex;
305             if (jumpBackward >= 0)
306             {
307                 break;
308             }
309             // try validate against the last producer chunk index
310             if (lvProducerChunkIndex() == currentChunkIndex)
311             {
312                 currentChunk = appendNextChunks(currentChunk, currentChunkIndex, -jumpBackward);
313             }
314             else
315             {
316                 currentChunk = null;
317             }
318         }
319         for (long i = 0; i < jumpBackward; i++)
320         {
321             // prev cannot be null, because the consumer cannot null it without consuming the element for which we are
322             // trying to get the chunk.
323             currentChunk = currentChunk.lvPrev();
324             assert currentChunk != null;
325         }
326         assert currentChunk.lvIndex() == requiredChunkIndex;
327         return currentChunk;
328     }
329 
330     private R appendNextChunks(
331         R currentChunk,
332         long currentChunkIndex,
333         long chunksToAppend)
334     {
335         assert currentChunkIndex != NOT_USED;
336         // prevent other concurrent attempts on appendNextChunk
337         if (!casProducerChunkIndex(currentChunkIndex, ROTATION))
338         {
339             return null;
340         }
341         /* LOCKED FOR APPEND */
342         {
343             // it is valid for the currentChunk to be consumed while appending is in flight, but it's not valid for the
344             // current chunk ordering to change otherwise.
345             assert currentChunkIndex == currentChunk.lvIndex();
346 
347             for (long i = 1; i <= chunksToAppend; i++)
348             {
349                 R newChunk = newOrPooledChunk(currentChunk, currentChunkIndex + i);
350                 soProducerChunk(newChunk);
351                 //link the next chunk only when finished
352                 currentChunk.soNext(newChunk);
353                 currentChunk = newChunk;
354             }
355 
356             // release appending
357             soProducerChunkIndex(currentChunkIndex + chunksToAppend);
358         }
359         /* UNLOCKED FOR APPEND */
360         return currentChunk;
361     }
362 
363     private R newOrPooledChunk(R prevChunk, long nextChunkIndex)
364     {
365         R newChunk = freeChunksPool.poll();
366         if (newChunk != null)
367         {
368             // single-writer: prevChunk::index == nextChunkIndex is protecting it
369             assert newChunk.lvIndex() < prevChunk.lvIndex();
370             newChunk.soPrev(prevChunk);
371             // index set is releasing prev, allowing other pending offers to continue
372             newChunk.soIndex(nextChunkIndex);
373         }
374         else
375         {
376             newChunk = newChunk(nextChunkIndex, prevChunk, chunkMask + 1, false);
377         }
378         return newChunk;
379     }

382     /**
383      * Does not null out the first element of `next`, callers must do that
384      */
385     final void moveToNextConsumerChunk(R cChunk, R next)
386     {
387         // avoid GC nepotism
388         cChunk.soNext(null);
389         next.soPrev(null);
390         // no need to cChunk.soIndex(NOT_USED)
391         if (cChunk.isPooled())
392         {
393             final boolean pooled = freeChunksPool.offer(cChunk);
394             assert pooled;
395         }
396         this.soConsumerChunk(next);
397         // MC case:
398         // from now on the code is not single-threaded anymore and
399         // other consumers can move forward consumerIndex
400     }
