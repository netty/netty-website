465     private int offerSlowPath(long mask, long pIndex, long producerLimit) {
466         final long cIndex = lvConsumerIndex();
467         long bufferCapacity = getCurrentBufferCapacity(mask);
468         if (cIndex + bufferCapacity > pIndex) {
469             if (!casProducerLimit(producerLimit, cIndex + bufferCapacity)) {
470                 // retry from top
471                 return RETRY;
472             } else {
473                 // continue to pIndex CAS
474                 return CONTINUE_TO_P_INDEX_CAS;
475             }
476         } else // full and cannot grow
477         if (availableInQueue(pIndex, cIndex) <= 0) {
478             // offer should return false;
479             return QUEUE_FULL;
480         } else // grab index for resize -> set lower bit
481         if (casProducerIndex(pIndex, pIndex + 1)) {
482             // trigger a resize
483             return QUEUE_RESIZE;
484         } else {
485             // failed resize attempt, retry from top
486             return RETRY;
487         }

596     @Override
597     public int fill(Supplier<E> s, int limit) {
598         if (null == s)
599             throw new IllegalArgumentException("supplier is null");
600         if (limit < 0)
601             throw new IllegalArgumentException("limit is negative:" + limit);
602         if (limit == 0)
603             return 0;
604         long mask;
605         AtomicReferenceArray<E> buffer;
606         long pIndex;
607         int claimedSlots;
608         while (true) {
609             long producerLimit = lvProducerLimit();
610             pIndex = lvProducerIndex();
611             // lower bit is indicative of resize, if we see it we spin until it's cleared
612             if ((pIndex & 1) == 1) {
613                 continue;
614             }
615             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1)
616             // NOTE: mask/buffer may get changed by resizing -> only use for array access after successful CAS.
617             // Only by virtue offloading them between the lvProducerIndex and a successful casProducerIndex are they
618             // safe to use.
619             mask = this.producerMask;
620             buffer = this.producerBuffer;
621             // a successful CAS ties the ordering, lv(pIndex) -> [mask/buffer] -> cas(pIndex)
622             // we want 'limit' slots, but will settle for whatever is visible to 'producerLimit'
623             // -> producerLimit >= batchIndex
624             long batchIndex = Math.min(producerLimit, pIndex + 2l * limit);
625             if (pIndex >= producerLimit) {
626                 int result = offerSlowPath(mask, pIndex, producerLimit);
627                 switch(result) {
628                     case CONTINUE_TO_P_INDEX_CAS:
629                     // offer slow path verifies only one slot ahead, we cannot rely on indication here
630                     case RETRY:
631                         continue;
632                     case QUEUE_FULL:
633                         return 0;
634                     case QUEUE_RESIZE:
635                         resize(mask, buffer, pIndex, null, s);
636                         return 1;
637                 }
638             }
