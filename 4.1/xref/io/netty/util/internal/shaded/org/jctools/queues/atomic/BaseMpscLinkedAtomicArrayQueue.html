328         int result;
329         final long cIndex = lvConsumerIndex();
330         long bufferCapacity = getCurrentBufferCapacity(mask);
331         // 0 - goto pIndex CAS
332         result = 0;
333         if (cIndex + bufferCapacity > pIndex) {
334             if (!casProducerLimit(producerLimit, cIndex + bufferCapacity)) {
335                 // retry from top

469     @Override
470     public int fill(Supplier<E> s, int batchSize) {
471         long mask;
472         AtomicReferenceArray<E> buffer;
473         long pIndex;
474         int claimedSlots;
475         while (true) {
476             long producerLimit = lvProducerLimit();
477             pIndex = lvProducerIndex();
478             // lower bit is indicative of resize, if we see it we spin until it's cleared
479             if ((pIndex & 1) == 1) {
480                 continue;
481             }
482             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1)
483             // NOTE: mask/buffer may get changed by resizing -> only use for array access after successful CAS.
484             // Only by virtue ofloading them between the lvProcducerIndex and a successful casProducerIndex are they
485             // safe to use.
486             mask = this.producerMask;
487             buffer = this.producerBuffer;
488             // a successful CAS ties the ordering, lv(pIndex)->[mask/buffer]->cas(pIndex)
489             // we want 'limit' slots, but will settle for whatever is visible to 'producerLimit'
490             long batchIndex = Math.min(producerLimit, pIndex + 2 * batchSize);
491             if (pIndex == producerLimit || producerLimit < batchIndex) {
492                 int result = offerSlowPath(mask, pIndex, producerLimit);
493                 switch(result) {
494                     case 1:
495                         continue;
496                     case 2:
497                         return 0;
498                     case 3:
499                         resize(mask, buffer, pIndex, s.get());
500                         return 1;
501                 }
502             }
503             // claim limit slots at once
504             if (casProducerIndex(pIndex, batchIndex)) {
505                 claimedSlots = (int) ((batchIndex - pIndex) / 2);
506                 break;
507             }
508         }
509         int i = 0;
510         for (i = 0; i < claimedSlots; i++) {
511             final int offset = modifiedCalcElementOffset(pIndex + 2 * i, mask);
512             soElement(buffer, offset, s.get());
513         }
514         return claimedSlots;
515     }

561     private void resize(long oldMask, AtomicReferenceArray<E> oldBuffer, long pIndex, E e) {
562         int newBufferLength = getNextBufferSize(oldBuffer);
563         final AtomicReferenceArray<E> newBuffer = allocate(newBufferLength);
564         producerBuffer = newBuffer;
565         final int newMask = (newBufferLength - 2) << 1;
566         producerMask = newMask;
567         final int offsetInOld = modifiedCalcElementOffset(pIndex, oldMask);
568         final int offsetInNew = modifiedCalcElementOffset(pIndex, newMask);
569         // element in new array
570         soElement(newBuffer, offsetInNew, e);
571         // buffer linked
572         soElement(oldBuffer, nextArrayOffset(oldMask), newBuffer);
573         // ASSERT code
574         final long cIndex = lvConsumerIndex();
575         final long availableInQueue = availableInQueue(pIndex, cIndex);
576         RangeUtil.checkPositive(availableInQueue, "availableInQueue");
577         // Invalidate racing CASs
578         // We never set the limit beyond the bounds of a buffer
579         soProducerLimit(pIndex + Math.min(newMask, availableInQueue));
580         // make resize visible to the other producers
581         soProducerIndex(pIndex + 2);
582         // INDEX visible before ELEMENT, consistent with consumer expectation
583         // make resize visible to consumer
584         soElement(oldBuffer, offsetInOld, JUMP);
585     }
