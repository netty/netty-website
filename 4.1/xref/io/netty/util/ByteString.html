28  /**
29   * The primary use case for this class is to function as an immutable array of bytes. For performance reasons this
30   * class supports sharing memory with external arrays, and also direct access to the underlying memory via
31   * {@link #array()}. Care must be taken when directly accessing the memory as that may invalidate assumptions that
32   * this object is immutable.
33   */
34  public class ByteString {
35      /**
36       * Allows sub classes to take advantage of {@link ByteString} operations which need to generate new
37       * ByteString objects.

40          ByteString newInstance(byte[] value, int start, int length, boolean copy);
41      }

98      /**
99       * Construct a new {@link BinaryString} object from a {@code byte[]} array.
100      * @param copy {@code true} then a copy of the memory will be made. {@code false} the underlying memory
101      * will be shared.
102      */
103     public ByteString(byte[] value, int start, int length, boolean copy) {
104         if (copy) {
105             this.value = Arrays.copyOfRange(value, start, start + length);
106             this.offset = 0;
107             this.length = length;
108         } else {
109             if (start < 0 || start > value.length - length) {
110                 throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" +
111                         length + ") <= " + "value.length(" + value.length + ')');
112             }
113             this.value = value;
114             this.offset = start;
115             this.length = length;
116         }
117     }

119     /**
120      * Create a new object which is equal to {@code value}.
121      * @param value The object to replicate.
122      * @param copy {@code true} mean the underlying storage will be copied.
123      * {@code false} means the underlying storage will be shared.
124      */
125     public ByteString(ByteString value, boolean copy) {
126         checkNotNull(value, "value");
127         this.length = value.length();
128         this.hash = value.hash;
129         if (copy) {
130             this.value = new byte[length];
131             System.arraycopy(value.array(), value.arrayOffset(), this.value, 0, length);
132             this.offset = 0;
133         } else {
134             this.value = value.array();
135             this.offset = value.offset;
136         }
137     }

156 
157     /**
158      * Initialize a {@link ByteString} based upon the underlying storage from {@link value}.
159      * There is a potential to share the underlying array storage if {@link ByteBuffer#hasArray()} is {@code true}.
160      * if {@code copy} is {@code true} a copy will be made of the memory.
161      * if {@code copy} is {@code false} the underlying storage will be shared, if possible.
162      */
163     public ByteString(ByteBuffer value, int start, int length, boolean copy) {
164         if (start < 0 || length > checkNotNull(value, "value").capacity() - start) {
165             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
166                             + ") <= " + "value.capacity(" + value.capacity() + ')');
167         }
168 
169         if (value.hasArray()) {
170             if (copy) {
171                 final int bufferOffset = value.arrayOffset() + start;
172                 this.value = Arrays.copyOfRange(value.array(), bufferOffset, bufferOffset + length);
173                 offset = 0;
174                 this.length = length;
175             } else {
176                 this.value = value.array();
177                 this.offset = start;
178                 this.length = length;
179             }
180         } else {
181             this.value = new byte[length];
182             int oldPos = value.position();
183             value.get(this.value, 0, length);
184             value.position(oldPos);
185             this.offset = 0;
186             this.length = length;
187         }
188     }

197     /**
198      * Create a copy of {@link value} into a {@link ByteString} using the encoding type of {@code charset}.
199      * The copy will start at index {@code start} and copy {@code length} bytes.
200      */
201     public ByteString(char[] value, Charset charset, int start, int length) {
202         if (start < 0 || length > checkNotNull(value, "value").length - start) {
203             throw new IndexOutOfBoundsException("expected: " + "0 <= start(" + start + ") <= start + length(" + length
204                     + ") <= " + "length(" + length + ')');
205         }
206 
207         CharBuffer cbuf = CharBuffer.wrap(value, start, start + length);
208         CharsetEncoder encoder = CharsetUtil.getEncoder(charset);
209         ByteBuffer nativeBuffer = ByteBuffer.allocate((int) (encoder.maxBytesPerChar() * length));
210         encoder.encode(cbuf, nativeBuffer, true);
211         final int bufferOffset = nativeBuffer.arrayOffset();
212         this.value = Arrays.copyOfRange(nativeBuffer.array(), bufferOffset, bufferOffset + nativeBuffer.position());
213         this.offset = 0;
214         this.length =  this.value.length;
215     }

262     /**
263      * Iterates over the specified area of this buffer with the specified {@code processor} in ascending order.
264      * (i.e. {@code index}, {@code (index + 1)},  .. {@code (index + length - 1)}).
265      *
266      * @return {@code -1} if the processor iterated to or beyond the end of the specified area.
267      *         The last-visited index If the {@link ByteProcessor#process(byte)} returned {@code false}.
268      */
269     public final int forEachByte(int index, int length, ByteProcessor visitor) throws Exception {
270         if (index < 0 || length > length() - index) {
271             throw new IndexOutOfBoundsException("expected: " + "0 <= index(" + index + ") <= start + length(" + length
272                     + ") <= " + "length(" + length() + ')');
273         }
274 
275         return forEachByte0(index, length, visitor);
276     }

278     private int forEachByte0(int index, int length, ByteProcessor visitor) throws Exception {
279         final int len = offset + length;
280         for (int i = offset + index; i < len; ++i) {
281             if (!visitor.process(value[i])) {
282                 return i - offset;
283             }
284         }
285         return -1;
286     }

298     /**
299      * Iterates over the specified area of this buffer with the specified {@code processor} in descending order.
300      * (i.e. {@code (index + length - 1)}, {@code (index + length - 2)}, ... {@code index}).
301      *
302      * @return {@code -1} if the processor iterated to or beyond the beginning of the specified area.
303      *         The last-visited index If the {@link ByteProcessor#process(byte)} returned {@code false}.
304      */
305     public final int forEachByteDesc(int index, int length, ByteProcessor visitor) throws Exception {
306         if (index < 0 || length > length() - index) {
307             throw new IndexOutOfBoundsException("expected: " + "0 <= index(" + index + ") <= start + length(" + length
308                     + ") <= " + "length(" + length() + ')');
309         }
310 
311         return forEachByteDesc0(index, length, visitor);
312     }

314     private int forEachByteDesc0(int index, int length, ByteProcessor visitor) throws Exception {
315         final int end = offset + index;
316         for (int i = offset + index + length - 1; i >= end; --i) {
317             if (!visitor.process(value[i])) {
318                 return i - offset;
319             }
320         }
321         return -1;
322     }

324     public final byte byteAt(int index) {
325         // We must do a range check here to enforce the access does not go outside our sub region of the array.
326         // We rely on the array access itself to pick up the array out of bounds conditions
327         if (index < 0 || index >= length) {
328             throw new IndexOutOfBoundsException("index: " + index + " must be in the range [0," + length + ")");
329         }
330         return value[index + offset];
331     }

409     @Override
410     public int hashCode() {
411         int h = hash;
412         if (h == 0) {
413             final int end = offset + length;
414             for (int i = offset; i < end; ++i) {
415                 h = h * HASH_CODE_PRIME + value[i];
416             }
417 
418             hash = h;
419         }
420         return hash;
421     }

457     /**
458      * Either copy or share a subset of underlying sub-sequence of bytes.
459      * @param start the offset of the first character (inclusive).
460      * @param end The index to stop at (exclusive).
461      * @param copy If {@code true} then a copy of the underlying storage will be made.
462      * If {@code false} then the underlying storage will be shared.
463      * @param factory The factory used to generate a new {@link ByteString} object.
464      * @return a new string containing the characters from start to the end of the string.
465      * @throws IndexOutOfBoundsException if {@code start < 0} or {@code start > length()}.
466      */
467     protected ByteString subSequence(int start, int end, boolean copy, ByteStringFactory factory) {
468         if (start < 0 || start > end || end > length()) {
469             throw new IndexOutOfBoundsException("expected: 0 <= start(" + start + ") <= end (" + end + ") <= length("
470                             + length() + ')');
471         }
472 
473         if (start == 0 && end == length()) {
474             return this;
475         }
476 
477         if (end == start) {
478             return EMPTY_STRING;
479         }
480 
481         return factory.newInstance(value, start + offset, end - start, copy);
482     }

496     public final int parseAsciiInt(int start, int end, int radix) {
497         if (radix < Character.MIN_RADIX || radix > Character.MAX_RADIX) {
498             throw new NumberFormatException();
499         }
500 
501         if (start == end) {
502             throw new NumberFormatException();
503         }
504 
505         int i = start;
506         boolean negative = byteAt(i) == '-';
507         if (negative && ++i == end) {
508             throw new NumberFormatException(subSequence(start, end, false).toString());
509         }
510 
511         return parseAsciiInt(i, end, radix, negative);
512     }
513 
514     private int parseAsciiInt(int start, int end, int radix, boolean negative) {
515         int max = Integer.MIN_VALUE / radix;
516         int result = 0;
517         int currOffset = start;
518         while (currOffset < end) {
519             int digit = Character.digit((char) (value[currOffset++ + offset] & 0xFF), radix);
520             if (digit == -1) {
521                 throw new NumberFormatException(subSequence(start, end, false).toString());
522             }
523             if (max > result) {
524                 throw new NumberFormatException(subSequence(start, end, false).toString());
525             }
526             int next = result * radix - digit;
527             if (next > result) {
528                 throw new NumberFormatException(subSequence(start, end, false).toString());
529             }
530             result = next;
531         }
532         if (!negative) {
533             result = -result;
534             if (result < 0) {
535                 throw new NumberFormatException(subSequence(start, end, false).toString());
536             }
537         }
538         return result;
539     }

602     public char parseChar(int start) {
603         if (start + 1 >= length()) {
604             throw new IndexOutOfBoundsException("2 bytes required to convert to character. index " +
605                     start + " would go out of bounds.");
606         }
607         final int startWithOffset = start + offset;
608         return (char) (((value[startWithOffset] & 0xFF) << 8) | (value[startWithOffset + 1] & 0xFF));
609     }
