246     protected static Runnable pollTaskFrom(Queue<Runnable> taskQueue) {
247         for (;;) {
248             Runnable task = taskQueue.poll();
249             if (task != WAKEUP_TASK) {
250                 return task;
251             }
252         }

255     /**
256      * Take the next {@link Runnable} from the task queue and so will block if no task is currently present.
257      * <p>
258      * Be aware that this method will throw an {@link UnsupportedOperationException} if the task queue, which was
259      * created via {@link #newTaskQueue()}, does not implement {@link BlockingQueue}.
260      * </p>
261      *
262      * @return {@code null} if the executor thread has been interrupted or waken up.
263      */
264     protected Runnable takeTask() {
265         assert inEventLoop();
266         if (!(taskQueue instanceof BlockingQueue)) {
267             throw new UnsupportedOperationException();
268         }
269 
270         BlockingQueue<Runnable> taskQueue = (BlockingQueue<Runnable>) this.taskQueue;
271         for (;;) {
272             ScheduledFutureTask<?> scheduledTask = peekScheduledTask();
273             if (scheduledTask == null) {
274                 Runnable task = null;
275                 try {
276                     task = taskQueue.take();
277                     if (task == WAKEUP_TASK) {
278                         task = null;
279                     }
280                 } catch (InterruptedException e) {
281                     // Ignore
282                 }
283                 return task;
284             } else {
285                 long delayNanos = scheduledTask.delayNanos();
286                 Runnable task = null;
287                 if (delayNanos > 0) {
288                     try {
289                         task = taskQueue.poll(delayNanos, TimeUnit.NANOSECONDS);
290                     } catch (InterruptedException e) {
291                         // Waken up.
292                         return null;
293                     }
294                 }
295                 if (task == null) {
296                     // We need to fetch the scheduled tasks now as otherwise there may be a chance that
297                     // scheduled tasks are never executed if there is always one task in the taskQueue.
298                     // This is for example true for the read task of OIO Transport
299                     // See https://github.com/netty/netty/issues/1614
300                     fetchFromScheduledTaskQueue();
301                     task = taskQueue.poll();
302                 }
303 
304                 if (task != null) {
305                     return task;
306                 }
307             }
308         }
309     }

311     private boolean fetchFromScheduledTaskQueue() {
312         if (scheduledTaskQueue == null || scheduledTaskQueue.isEmpty()) {
313             return true;
314         }
315         long nanoTime = AbstractScheduledEventExecutor.nanoTime();
316         for (;;) {
317             Runnable scheduledTask = pollScheduledTask(nanoTime);
318             if (scheduledTask == null) {
319                 return true;
320             }
321             if (!taskQueue.offer(scheduledTask)) {
322                 // No space left in the task queue add it back to the scheduledTaskQueue so we pick it up again.
323                 scheduledTaskQueue.add((ScheduledFutureTask<?>) scheduledTask);
324                 return false;
325             }
326         }
327     }

329     /**
330      * @return {@code true} if at least one scheduled task was executed.
331      */
332     private boolean executeExpiredScheduledTasks() {
333         if (scheduledTaskQueue == null || scheduledTaskQueue.isEmpty()) {
334             return false;
335         }
336         long nanoTime = AbstractScheduledEventExecutor.nanoTime();
337         Runnable scheduledTask = pollScheduledTask(nanoTime);
338         if (scheduledTask == null) {
339             return false;
340         }
341         do {
342             safeExecute(scheduledTask);
343         } while ((scheduledTask = pollScheduledTask(nanoTime)) != null);
344         return true;
345     }

403     /**
404      * Poll all tasks from the task queue and run them via {@link Runnable#run()} method.
405      *
406      * @return {@code true} if and only if at least one task was run
407      */
408     protected boolean runAllTasks() {
409         assert inEventLoop();
410         boolean fetchedAll;
411         boolean ranAtLeastOne = false;
412 
413         do {
414             fetchedAll = fetchFromScheduledTaskQueue();
415             if (runAllTasksFrom(taskQueue)) {
416                 ranAtLeastOne = true;
417             }
418         } while (!fetchedAll); // keep on processing until we fetched all scheduled tasks.
419 
420         if (ranAtLeastOne) {
421             lastExecutionTime = ScheduledFutureTask.nanoTime();
422         }
423         afterRunningAllTasks();
424         return ranAtLeastOne;
425     }

427     /**
428      * Execute all expired scheduled tasks and all current tasks in the executor queue until both queues are empty,
429      * or {@code maxDrainAttempts} has been exceeded.
430      * @param maxDrainAttempts The maximum amount of times this method attempts to drain from queues. This is to prevent
431      *                         continuous task execution and scheduling from preventing the EventExecutor thread to
432      *                         make progress and return to the selector mechanism to process inbound I/O events.
433      * @return {@code true} if at least one task was run.
434      */
435     protected final boolean runScheduledAndExecutorTasks(final int maxDrainAttempts) {
436         assert inEventLoop();
437         boolean ranAtLeastOneTask;
438         int drainAttempt = 0;
439         do {
440             // We must run the taskQueue tasks first, because the scheduled tasks from outside the EventLoop are queued
441             // here because the taskQueue is thread safe and the scheduledTaskQueue is not thread safe.
442             ranAtLeastOneTask = runExistingTasksFrom(taskQueue) | executeExpiredScheduledTasks();
443         } while (ranAtLeastOneTask && ++drainAttempt < maxDrainAttempts);
444 
445         if (drainAttempt > 0) {
446             lastExecutionTime = ScheduledFutureTask.nanoTime();
447         }
448         afterRunningAllTasks();
449 
450         return drainAttempt > 0;
451     }

453     /**
454      * Runs all tasks from the passed {@code taskQueue}.
455      *
456      * @param taskQueue To poll and execute all tasks.
457      *
458      * @return {@code true} if at least one task was executed.
459      */
460     protected final boolean runAllTasksFrom(Queue<Runnable> taskQueue) {
461         Runnable task = pollTaskFrom(taskQueue);
462         if (task == null) {
463             return false;
464         }
465         for (;;) {
466             safeExecute(task);
467             task = pollTaskFrom(taskQueue);
468             if (task == null) {
469                 return true;
470             }
471         }
472     }

474     /**
475      * What ever tasks are present in {@code taskQueue} when this method is invoked will be {@link Runnable#run()}.
476      * @param taskQueue the task queue to drain.
477      * @return {@code true} if at least {@link Runnable#run()} was called.
478      */
479     private boolean runExistingTasksFrom(Queue<Runnable> taskQueue) {
480         Runnable task = pollTaskFrom(taskQueue);
481         if (task == null) {
482             return false;
483         }
484         int remaining = Math.min(maxPendingTasks, taskQueue.size());
485         safeExecute(task);
486         // Use taskQueue.poll() directly rather than pollTaskFrom() since the latter may
487         // silently consume more than one item from the queue (skips over WAKEUP_TASK instances)
488         while (remaining-- > 0 && (task = taskQueue.poll()) != null) {
489             safeExecute(task);
490         }
491         return true;
492     }

494     /**
495      * Poll all tasks from the task queue and run them via {@link Runnable#run()} method.  This method stops running
496      * the tasks in the task queue and returns if it ran longer than {@code timeoutNanos}.
497      */
498     protected boolean runAllTasks(long timeoutNanos) {
499         fetchFromScheduledTaskQueue();
500         Runnable task = pollTask();
501         if (task == null) {
502             afterRunningAllTasks();
503             return false;
504         }
505 
506         final long deadline = ScheduledFutureTask.nanoTime() + timeoutNanos;
507         long runTasks = 0;
508         long lastExecutionTime;
509         for (;;) {
510             safeExecute(task);
511 
512             runTasks ++;
513 
514             // Check timeout every 64 tasks because nanoTime() is relatively expensive.
515             // XXX: Hard-coded value - will make it configurable if it is really a problem.
516             if ((runTasks & 0x3F) == 0) {
517                 lastExecutionTime = ScheduledFutureTask.nanoTime();
518                 if (lastExecutionTime >= deadline) {
519                     break;
520                 }
521             }
522 
523             task = pollTask();
524             if (task == null) {
525                 lastExecutionTime = ScheduledFutureTask.nanoTime();
526                 break;
527             }
528         }
529 
530         afterRunningAllTasks();
531         this.lastExecutionTime = lastExecutionTime;
532         return true;
533     }

589     protected void wakeup(boolean inEventLoop) {
590         if (!inEventLoop) {
591             // Use offer as we actually only need this to unblock the thread and if offer fails we do not care as there
592             // is already something in the queue.
593             taskQueue.offer(WAKEUP_TASK);
594         }

597     @Override
598     final void executeScheduledRunnable(final Runnable runnable, boolean isAddition, long deadlineNanos) {
599         // Don't wakeup if this is a removal task or if beforeScheduledTaskSubmitted returns false
600         if (isAddition && beforeScheduledTaskSubmitted(deadlineNanos)) {
601             super.executeScheduledRunnable(runnable, isAddition, deadlineNanos);
602         } else {
603             super.executeScheduledRunnable(new NonWakeupRunnable() {
604                 @Override
605                 public void run() {
606                     runnable.run();
607                 }
608             }, isAddition, deadlineNanos);
609             // Second hook after scheduling to facilitate race-avoidance
610             if (isAddition && afterScheduledTaskSubmitted(deadlineNanos)) {
611                 wakeup(false);
612             }
613         }
614     }

653     private boolean runShutdownHooks() {
654         boolean ran = false;
655         // Note shutdown hooks can add / remove shutdown hooks.
656         while (!shutdownHooks.isEmpty()) {
657             List<Runnable> copy = new ArrayList<Runnable>(shutdownHooks);
658             shutdownHooks.clear();
659             for (Runnable task: copy) {
660                 try {
661                     task.run();
662                 } catch (Throwable t) {
663                     logger.warn("Shutdown hook raised an exception.", t);
664                 } finally {
665                     ran = true;
666                 }
667             }
668         }
669 
670         if (ran) {
671             lastExecutionTime = ScheduledFutureTask.nanoTime();
672         }
673 
674         return ran;
675     }

677     @Override
678     public Future<?> shutdownGracefully(long quietPeriod, long timeout, TimeUnit unit) {
679         if (quietPeriod < 0) {
680             throw new IllegalArgumentException("quietPeriod: " + quietPeriod + " (expected >= 0)");
681         }
682         if (timeout < quietPeriod) {
683             throw new IllegalArgumentException(
684                     "timeout: " + timeout + " (expected >= quietPeriod (" + quietPeriod + "))");
685         }
686         if (unit == null) {
687             throw new NullPointerException("unit");
688         }
689 
690         if (isShuttingDown()) {
691             return terminationFuture();
692         }
693 
694         boolean inEventLoop = inEventLoop();
695         boolean wakeup;
696         int oldState;
697         for (;;) {
698             if (isShuttingDown()) {
699                 return terminationFuture();
700             }
701             int newState;
702             wakeup = true;
703             oldState = state;
704             if (inEventLoop) {
705                 newState = ST_SHUTTING_DOWN;
706             } else {
707                 switch (oldState) {
708                     case ST_NOT_STARTED:
709                     case ST_STARTED:
710                         newState = ST_SHUTTING_DOWN;
711                         break;
712                     default:
713                         newState = oldState;
714                         wakeup = false;
715                 }
716             }
717             if (STATE_UPDATER.compareAndSet(this, oldState, newState)) {
718                 break;
719             }
720         }
721         gracefulShutdownQuietPeriod = unit.toNanos(quietPeriod);
722         gracefulShutdownTimeout = unit.toNanos(timeout);
723 
724         if (ensureThreadStarted(oldState)) {
725             return terminationFuture;
726         }
727 
728         if (wakeup) {
729             taskQueue.offer(WAKEUP_TASK);
730             if (!addTaskWakesUp) {
731                 wakeup(inEventLoop);
732             }
733         }
734 
735         return terminationFuture();
736     }

806     /**
807      * Confirm that the shutdown if the instance should be done now!
808      */
809     protected boolean confirmShutdown() {
810         if (!isShuttingDown()) {
811             return false;
812         }
813 
814         if (!inEventLoop()) {
815             throw new IllegalStateException("must be invoked from an event loop");
816         }
817 
818         cancelScheduledTasks();
819 
820         if (gracefulShutdownStartTime == 0) {
821             gracefulShutdownStartTime = ScheduledFutureTask.nanoTime();
822         }
823 
824         if (runAllTasks() || runShutdownHooks()) {
825             if (isShutdown()) {
826                 // Executor shut down - no new tasks anymore.
827                 return true;
828             }
829 
830             // There were tasks in the queue. Wait a little bit more until no tasks are queued for the quiet period or
831             // terminate if the quiet period is 0.
832             // See https://github.com/netty/netty/issues/4241
833             if (gracefulShutdownQuietPeriod == 0) {
834                 return true;
835             }
836             taskQueue.offer(WAKEUP_TASK);
837             return false;
838         }
839 
840         final long nanoTime = ScheduledFutureTask.nanoTime();
841 
842         if (isShutdown() || nanoTime - gracefulShutdownStartTime > gracefulShutdownTimeout) {
843             return true;
844         }
845 
846         if (nanoTime - lastExecutionTime <= gracefulShutdownQuietPeriod) {
847             // Check if any tasks were added to the queue every 100ms.
848             // TODO: Change the behavior of takeTask() so that it returns on timeout.
849             taskQueue.offer(WAKEUP_TASK);
850             try {
851                 Thread.sleep(100);
852             } catch (InterruptedException e) {
853                 // Ignore
854             }
855 
856             return false;
857         }
858 
859         // No tasks were added for last quiet period - hopefully safe to shut down.
860         // (Hopefully because we really cannot make a guarantee that there will be no execute() calls by a user.)
861         return true;
862     }

879     @Override
880     public void execute(Runnable task) {
881         if (task == null) {
882             throw new NullPointerException("task");
883         }
884 
885         boolean inEventLoop = inEventLoop();
886         addTask(task);
887         if (!inEventLoop) {
888             startThread();
889             if (isShutdown()) {
890                 boolean reject = false;
891                 try {
892                     if (removeTask(task)) {
893                         reject = true;
894                     }
895                 } catch (UnsupportedOperationException e) {
896                     // The task queue does not support removal so the best thing we can do is to just move on and
897                     // hope we will be able to pick-up the task before its completely terminated.
898                     // In worst case we will log on termination.
899                 }
900                 if (reject) {
901                     reject();
902                 }
903             }
904         }
905 
906         if (!addTaskWakesUp && wakesUpForTask(task)) {
907             wakeup(inEventLoop);
908         }
909     }

1003     private void startThread() {
1004         if (state == ST_NOT_STARTED) {
1005             if (STATE_UPDATER.compareAndSet(this, ST_NOT_STARTED, ST_STARTED)) {
1006                 boolean success = false;
1007                 try {
1008                     doStartThread();
1009                     success = true;
1010                 } finally {
1011                     if (!success) {
1012                         STATE_UPDATER.compareAndSet(this, ST_STARTED, ST_NOT_STARTED);
1013                     }
1014                 }
1015             }
1016         }

1019     private boolean ensureThreadStarted(int oldState) {
1020         if (oldState == ST_NOT_STARTED) {
1021             try {
1022                 doStartThread();
1023             } catch (Throwable cause) {
1024                 STATE_UPDATER.set(this, ST_TERMINATED);
1025                 terminationFuture.tryFailure(cause);
1026 
1027                 if (!(cause instanceof Exception)) {
1028                     // Also rethrow as it may be an OOME for example
1029                     PlatformDependent.throwException(cause);
1030                 }
1031                 return true;
1032             }
1033         }
1034         return false;
1035     }

1037     private void doStartThread() {
1038         assert thread == null;
1039         executor.execute(new Runnable() {
1040             @Override
1041             public void run() {
1042                 thread = Thread.currentThread();
1043                 if (interrupted) {
1044                     thread.interrupt();
1045                 }
1046 
1047                 boolean success = false;
1048                 updateLastExecutionTime();
1049                 try {
1050                     SingleThreadEventExecutor.this.run();
1051                     success = true;
1052                 } catch (Throwable t) {
1053                     logger.warn("Unexpected exception from an event executor: ", t);
1054                 } finally {
1055                     for (;;) {
1056                         int oldState = state;
1057                         if (oldState >= ST_SHUTTING_DOWN || STATE_UPDATER.compareAndSet(
1058                                 SingleThreadEventExecutor.this, oldState, ST_SHUTTING_DOWN)) {
1059                             break;
1060                         }
1061                     }
1062 
1063                     // Check if confirmShutdown() was called at the end of the loop.
1064                     if (success && gracefulShutdownStartTime == 0) {
1065                         if (logger.isErrorEnabled()) {
1066                             logger.error("Buggy " + EventExecutor.class.getSimpleName() + " implementation; " +
1067                                     SingleThreadEventExecutor.class.getSimpleName() + ".confirmShutdown() must " +
1068                                     "be called before run() implementation terminates.");
1069                         }
1070                     }
1071 
1072                     try {
1073                         // Run all remaining tasks and shutdown hooks.
1074                         for (;;) {
1075                             if (confirmShutdown()) {
1076                                 break;
1077                             }
1078                         }
1079                     } finally {
1080                         try {
1081                             cleanup();
1082                         } finally {
1083                             // Lets remove all FastThreadLocals for the Thread as we are about to terminate and notify
1084                             // the future. The user may block on the future and once it unblocks the JVM may terminate
1085                             // and start unloading classes.
1086                             // See https://github.com/netty/netty/issues/6596.
1087                             FastThreadLocal.removeAll();
1088 
1089                             STATE_UPDATER.set(SingleThreadEventExecutor.this, ST_TERMINATED);
1090                             threadLock.countDown();
1091                             if (logger.isWarnEnabled() && !taskQueue.isEmpty()) {
1092                                 logger.warn("An event executor terminated with " +
1093                                         "non-empty task queue (" + taskQueue.size() + ')');
1094                             }
1095                             terminationFuture.setSuccess(null);
1096                         }
1097                     }
1098                 }
1099             }
1100         });
1101     }
