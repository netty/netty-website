427     private void notifyListenersWithStackOverFlowProtection() {
428         EventExecutor executor = executor();
429         if (executor.inEventLoop()) {
430             final InternalThreadLocalMap threadLocals = InternalThreadLocalMap.get();
431             final int stackDepth = threadLocals.futureListenerStackDepth();
432             if (stackDepth < MAX_LISTENER_STACK_DEPTH) {
433                 threadLocals.setFutureListenerStackDepth(stackDepth + 1);
434                 try {
435                     notifyListenersNow();
436                 } finally {
437                     threadLocals.setFutureListenerStackDepth(stackDepth);
438                 }
439                 return;
440             }
441         }

443         safeExecute(executor, new Runnable() {
444             @Override
445             public void run() {
446                 notifyListenersNow();
447             }
448         });

481     private void notifyListenersNow() {
482         Object listeners;
483         synchronized (this) {
484             // Only proceed if there are listeners to notify and we are not already notifying listeners.
485             if (notifyingListeners || this.listeners == null) {
486                 return;
487             }
488             notifyingListeners = true;
489             listeners = this.listeners;
490             this.listeners = null;
491         }
492         for (;;) {
493             if (listeners instanceof DefaultFutureListeners) {
494                 notifyListeners0((DefaultFutureListeners) listeners);
495             } else {
496                 notifyListener0(this, (GenericFutureListener<? extends Future<V>>) listeners);
497             }
498             synchronized (this) {
499                 if (this.listeners == null) {
500                     // Nothing can throw from within this method, so setting notifyingListeners back to false does not
501                     // need to be in a finally block.
502                     notifyingListeners = false;
503                     return;
504                 }
505                 listeners = this.listeners;
506                 this.listeners = null;
507             }
508         }
509     }

589     private boolean await0(long timeoutNanos, boolean interruptable) throws InterruptedException {
590         if (isDone()) {
591             return true;
592         }
593 
594         if (timeoutNanos <= 0) {
595             return isDone();
596         }
597 
598         if (interruptable && Thread.interrupted()) {
599             throw new InterruptedException(toString());
600         }
601 
602         checkDeadLock();
603 
604         long startTime = System.nanoTime();
605         long waitTime = timeoutNanos;
606         boolean interrupted = false;
607         try {
608             for (;;) {
609                 synchronized (this) {
610                     incWaiters();
611                     try {
612                         wait(waitTime / 1000000, (int) (waitTime % 1000000));
613                     } catch (InterruptedException e) {
614                         if (interruptable) {
615                             throw e;
616                         } else {
617                             interrupted = true;
618                         }
619                     } finally {
620                         decWaiters();
621                     }
622                 }
623                 if (isDone()) {
624                     return true;
625                 } else {
626                     waitTime = timeoutNanos - (System.nanoTime() - startTime);
627                     if (waitTime <= 0) {
628                         return isDone();
629                     }
630                 }
631             }
632         } finally {
633             if (interrupted) {
634                 Thread.currentThread().interrupt();
635             }

691      * Returns a {@link GenericProgressiveFutureListener}, an array of {@link GenericProgressiveFutureListener}, or
692      * {@code null}.
693      */
694     private synchronized Object progressiveListeners() {
695         Object listeners = this.listeners;
696         if (listeners == null) {
697             // No listeners added
698             return null;
699         }
700 
701         if (listeners instanceof DefaultFutureListeners) {
702             // Copy DefaultFutureListeners into an array of listeners.
703             DefaultFutureListeners dfl = (DefaultFutureListeners) listeners;
704             int progressiveSize = dfl.progressiveSize();
705             switch (progressiveSize) {
706                 case 0:
707                     return null;
708                 case 1:
709                     for (GenericFutureListener<?> l: dfl.listeners()) {
710                         if (l instanceof GenericProgressiveFutureListener) {
711                             return l;
712                         }
713                     }
714                     return null;
715             }
716 
717             GenericFutureListener<?>[] array = dfl.listeners();
718             GenericProgressiveFutureListener<?>[] copy = new GenericProgressiveFutureListener[progressiveSize];
719             for (int i = 0, j = 0; j < progressiveSize; i ++) {
720                 GenericFutureListener<?> l = array[i];
721                 if (l instanceof GenericProgressiveFutureListener) {
722                     copy[j ++] = (GenericProgressiveFutureListener<?>) l;
723                 }
724             }
725 
726             return copy;
727         } else if (listeners instanceof GenericProgressiveFutureListener) {
728             return listeners;
729         } else {
730             // Only one listener was added and it's not a progressive listener.
731             return null;
732         }
733     }
