104     AbstractChannelHandlerContext(DefaultChannelPipeline pipeline, EventExecutor executor,
105                                   String name, Class<? extends ChannelHandler> handlerClass) {
106         this.name = ObjectUtil.checkNotNull(name, "name");
107         this.pipeline = pipeline;
108         this.executor = executor;
109         this.executionMask = mask(handlerClass);
110         // Its ordered if its driven by the EventLoop or the given Executor is an instanceof OrderedEventExecutor.
111         ordered = executor == null || executor instanceof OrderedEventExecutor;
112     }

277     static void invokeExceptionCaught(final AbstractChannelHandlerContext next, final Throwable cause) {
278         ObjectUtil.checkNotNull(cause, "cause");
279         EventExecutor executor = next.executor();
280         if (executor.inEventLoop()) {
281             next.invokeExceptionCaught(cause);
282         } else {
283             try {
284                 executor.execute(new Runnable() {
285                     @Override
286                     public void run() {
287                         next.invokeExceptionCaught(cause);
288                     }
289                 });
290             } catch (Throwable t) {
291                 if (logger.isWarnEnabled()) {
292                     logger.warn("Failed to submit an exceptionCaught() event.", t);
293                     logger.warn("The exceptionCaught() event that was failed to submit was:", cause);
294                 }
295             }
296         }
297     }
298 
299     private void invokeExceptionCaught(final Throwable cause) {
300         if (invokeHandler()) {
301             try {
302                 handler().exceptionCaught(this, cause);
303             } catch (Throwable error) {
304                 if (logger.isDebugEnabled()) {
305                     logger.debug(
306                         "An exception {}" +
307                         "was thrown by a user handler's exceptionCaught() " +
308                         "method while handling the following exception:",
309                         ThrowableUtil.stackTraceToString(error), cause);
310                 } else if (logger.isWarnEnabled()) {
311                     logger.warn(
312                         "An exception '{}' [enable DEBUG level for full stacktrace] " +
313                         "was thrown by a user handler's exceptionCaught() " +
314                         "method while handling the following exception:", error, cause);
315                 }
316             }
317         } else {
318             fireExceptionCaught(cause);
319         }
320     }

480     @Override
481     public ChannelFuture bind(final SocketAddress localAddress, final ChannelPromise promise) {
482         ObjectUtil.checkNotNull(localAddress, "localAddress");
483         if (isNotValidPromise(promise, false)) {
484             // cancelled
485             return promise;
486         }
487 
488         final AbstractChannelHandlerContext next = findContextOutbound(MASK_BIND);
489         EventExecutor executor = next.executor();
490         if (executor.inEventLoop()) {
491             next.invokeBind(localAddress, promise);
492         } else {
493             safeExecute(executor, new Runnable() {
494                 @Override
495                 public void run() {
496                     next.invokeBind(localAddress, promise);
497                 }
498             }, promise, null, false);
499         }
500         return promise;
501     }

724     public ChannelHandlerContext flush() {
725         final AbstractChannelHandlerContext next = findContextOutbound(MASK_FLUSH);
726         EventExecutor executor = next.executor();
727         if (executor.inEventLoop()) {
728             next.invokeFlush();
729         } else {
730             Tasks tasks = next.invokeTasks;
731             if (tasks == null) {

734             safeExecute(executor, tasks.invokeFlushTask, channel().voidPromise(), null, false);
735         }
736 
737         return this;
738     }

771     private void write(Object msg, boolean flush, ChannelPromise promise) {
772         ObjectUtil.checkNotNull(msg, "msg");
773         try {
774             if (isNotValidPromise(promise, true)) {
775                 ReferenceCountUtil.release(msg);
776                 // cancelled
777                 return;
778             }
779         } catch (RuntimeException e) {
780             ReferenceCountUtil.release(msg);
781             throw e;
782         }
783 
784         final AbstractChannelHandlerContext next = findContextOutbound(flush ?
785                 (MASK_WRITE | MASK_FLUSH) : MASK_WRITE);
786         final Object m = pipeline.touch(msg, next);
787         EventExecutor executor = next.executor();
788         if (executor.inEventLoop()) {
789             if (flush) {
790                 next.invokeWriteAndFlush(m, promise);
791             } else {
792                 next.invokeWrite(m, promise);
793             }
794         } else {
795             final WriteTask task = WriteTask.newInstance(next, m, promise, flush);
796             if (!safeExecute(executor, task, promise, m, !flush)) {
797                 // We failed to submit the WriteTask. We need to cancel it so we decrement the pending bytes
798                 // and put it back in the Recycler for re-use later.
799                 //
800                 // See https://github.com/netty/netty/issues/8343.
801                 task.cancel();
802             }
803         }
804     }

874     private boolean isNotValidPromise(ChannelPromise promise, boolean allowVoidPromise) {
875         ObjectUtil.checkNotNull(promise, "promise");
876 
877         if (promise.isDone()) {
878             // Check if the promise was cancelled and if so signal that the processing of the operation
879             // should not be performed.
880             //
881             // See https://github.com/netty/netty/issues/2349
882             if (promise.isCancelled()) {
883                 return true;
884             }
885             throw new IllegalArgumentException("promise already done: " + promise);
886         }
887 
888         if (promise.channel() != channel()) {
889             throw new IllegalArgumentException(String.format(
890                     "promise.channel does not match: %s (expected: %s)", promise.channel(), channel()));
891         }
892 
893         if (promise.getClass() == DefaultChannelPromise.class) {
894             return false;
895         }
896 
897         if (!allowVoidPromise && promise instanceof VoidChannelPromise) {
898             throw new IllegalArgumentException(
899                     StringUtil.simpleClassName(VoidChannelPromise.class) + " not allowed for this operation");
900         }
901 
902         if (promise instanceof AbstractChannel.CloseFuture) {
903             throw new IllegalArgumentException(
904                     StringUtil.simpleClassName(AbstractChannel.CloseFuture.class) + " not allowed in a pipeline");
905         }
906         return false;
907     }

927     private static boolean skipContext(
928             AbstractChannelHandlerContext ctx, EventExecutor currentExecutor, int mask, int onlyMask) {
929         // Ensure we correctly handle MASK_EXCEPTION_CAUGHT which is not included in the MASK_EXCEPTION_CAUGHT
930         return (ctx.executionMask & (onlyMask | mask)) == 0 ||
931                 // We can only skip if the EventExecutor is the same as otherwise we need to ensure we offload
932                 // everything to preserve ordering.

935                 (ctx.executor() == currentExecutor && (ctx.executionMask & mask) == 0);

947     final boolean setAddComplete() {
948         for (;;) {
949             int oldState = handlerState;
950             if (oldState == REMOVE_COMPLETE) {
951                 return false;
952             }
953             // Ensure we never update when the handlerState is REMOVE_COMPLETE already.
954             // oldState is usually ADD_PENDING but can also be REMOVE_COMPLETE when an EventExecutor is used that is not
955             // exposing ordering guarantees.
956             if (HANDLER_STATE_UPDATER.compareAndSet(this, oldState, ADD_COMPLETE)) {
957                 return true;
958             }
959         }
960     }

962     final void setAddPending() {
963         boolean updated = HANDLER_STATE_UPDATER.compareAndSet(this, INIT, ADD_PENDING);
964         assert updated; // This should always be true as it MUST be called before setAddComplete() or setRemoved().
965     }

967     final void callHandlerAdded() throws Exception {
968         // We must call setAddComplete before calling handlerAdded. Otherwise if the handlerAdded method generates
969         // any pipeline events ctx.handler() will miss them because the state will not allow it.
970         if (setAddComplete()) {
971             handler().handlerAdded(this);
972         }
973     }

975     final void callHandlerRemoved() throws Exception {
976         try {
977             // Only call handlerRemoved(...) if we called handlerAdded(...) before.
978             if (handlerState == ADD_COMPLETE) {
979                 handler().handlerRemoved(this);
980             }
981         } finally {
982             // Mark the handler as removed in any case.
983             setRemoved();
984         }

987     /**
988      * Makes best possible effort to detect if {@link ChannelHandler#handlerAdded(ChannelHandlerContext)} was called
989      * yet. If not return {@code false} and if called or could not detect return {@code true}.
990      *
991      * If this method returns {@code false} we will not invoke the {@link ChannelHandler} but just forward the event.
992      * This is needed as {@link DefaultChannelPipeline} may already put the {@link ChannelHandler} in the linked-list
993      * but not called {@link ChannelHandler#handlerAdded(ChannelHandlerContext)}.
994      */
995     private boolean invokeHandler() {
996         // Store in local variable to reduce volatile reads.
997         int handlerState = this.handlerState;
998         return handlerState == ADD_COMPLETE || (!ordered && handlerState == ADD_PENDING);
999     }

1016     private static boolean safeExecute(EventExecutor executor, Runnable runnable,
1017             ChannelPromise promise, Object msg, boolean lazy) {
1018         try {
1019             if (lazy && executor instanceof AbstractEventExecutor) {
1020                 ((AbstractEventExecutor) executor).lazyExecute(runnable);
1021             } else {
1022                 executor.execute(runnable);
1023             }
1024             return true;
1025         } catch (Throwable cause) {
1026             try {
1027                 promise.setFailure(cause);
1028             } finally {
1029                 if (msg != null) {
1030                     ReferenceCountUtil.release(msg);
1031                 }
1032             }
1033             return false;
1034         }
1035     }

1047     static final class WriteTask implements Runnable {
1048         private static final ObjectPool<WriteTask> RECYCLER = ObjectPool.newPool(new ObjectCreator<WriteTask>() {
1049             @Override
1050             public WriteTask newObject(Handle<WriteTask> handle) {
1051                 return new WriteTask(handle);
1052             }
1053         });

1055         static WriteTask newInstance(AbstractChannelHandlerContext ctx,
1056                 Object msg, ChannelPromise promise, boolean flush) {
1057             WriteTask task = RECYCLER.get();
1058             init(task, ctx, msg, promise, flush);
1059             return task;
1060         }
1061 
1062         private static final boolean ESTIMATE_TASK_SIZE_ON_SUBMIT =
1063                 SystemPropertyUtil.getBoolean("io.netty.transport.estimateSizeOnSubmit", true);
1064 
1065         // Assuming compressed oops, 12 bytes obj header, 4 ref fields and one int field
1066         private static final int WRITE_TASK_OVERHEAD =
1067                 SystemPropertyUtil.getInt("io.netty.transport.writeTaskSizeOverhead", 32);
1068 
1069         private final Handle<WriteTask> handle;
1070         private AbstractChannelHandlerContext ctx;
1071         private Object msg;
1072         private ChannelPromise promise;
1073         private int size; // sign bit controls flush
1074 
1075         @SuppressWarnings("unchecked")
1076         private WriteTask(Handle<? extends WriteTask> handle) {
1077             this.handle = (Handle<WriteTask>) handle;
1078         }
1079 
1080         protected static void init(WriteTask task, AbstractChannelHandlerContext ctx,
1081                                    Object msg, ChannelPromise promise, boolean flush) {
1082             task.ctx = ctx;
1083             task.msg = msg;
1084             task.promise = promise;
1085 
1086             if (ESTIMATE_TASK_SIZE_ON_SUBMIT) {
1087                 task.size = ctx.pipeline.estimatorHandle().size(msg) + WRITE_TASK_OVERHEAD;
1088                 ctx.pipeline.incrementPendingOutboundBytes(task.size);
1089             } else {
1090                 task.size = 0;
1091             }
1092             if (flush) {
1093                 task.size |= Integer.MIN_VALUE;
1094             }
1095         }
1096 
1097         @Override
1098         public void run() {
1099             try {
1100                 decrementPendingOutboundBytes();
1101                 if (size >= 0) {
1102                     ctx.invokeWrite(msg, promise);
1103                 } else {
1104                     ctx.invokeWriteAndFlush(msg, promise);
1105                 }
1106             } finally {
1107                 recycle();
1108             }
1109         }
1110 
1111         void cancel() {
1112             try {
1113                 decrementPendingOutboundBytes();
1114             } finally {
1115                 recycle();
1116             }
1117         }
1118 
1119         private void decrementPendingOutboundBytes() {
1120             if (ESTIMATE_TASK_SIZE_ON_SUBMIT) {
1121                 ctx.pipeline.decrementPendingOutboundBytes(size & Integer.MAX_VALUE);
1122             }
1123         }
1124 
1125         private void recycle() {
1126             // Set to null so the GC can collect them directly
1127             ctx = null;
1128             msg = null;
1129             promise = null;
1130             handle.recycle(this);
1131         }
1132     }
