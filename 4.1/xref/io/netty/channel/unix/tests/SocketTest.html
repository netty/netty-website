16  package io.netty.channel.unix.tests;
17  
18  import io.netty.channel.unix.Socket;
19  import org.junit.After;
20  import org.junit.Before;
21  import org.junit.Test;
22  
23  import java.io.IOException;
24  
25  import static org.junit.Assert.assertEquals;
26  import static org.junit.Assert.assertFalse;
27  import static org.junit.Assert.assertTrue;
28  
29  public abstract class SocketTest<T extends Socket> {
30      protected T socket;
31  
32      protected abstract T newSocket();
33  
34      @Before
35      public void setup() {
36          socket = newSocket();
37      }
38  
39      @After
40      public void tearDown() throws IOException {
41          socket.close();
42      }
43  
44      @Test
45      public void testKeepAlive() throws Exception {
46          assertFalse(socket.isKeepAlive());
47          socket.setKeepAlive(true);
48          assertTrue(socket.isKeepAlive());
49      }
50  
51      @Test
52      public void testTcpNoDelay() throws Exception {
53          assertFalse(socket.isTcpNoDelay());
54          socket.setTcpNoDelay(true);
55          assertTrue(socket.isTcpNoDelay());
56      }
57  
58      @Test
59      public void testReceivedBufferSize() throws Exception {
60          int size = socket.getReceiveBufferSize();
61          int newSize = 65535;
62          assertTrue(size > 0);
63          socket.setReceiveBufferSize(newSize);
64          // Linux usually set it to double what is specified
65          assertTrue(newSize <= socket.getReceiveBufferSize());
66      }
67  
68      @Test
69      public void testSendBufferSize() throws Exception {
70          int size = socket.getSendBufferSize();
71          int newSize = 65535;
72          assertTrue(size > 0);
73          socket.setSendBufferSize(newSize);
74          // Linux usually set it to double what is specified
75          assertTrue(newSize <= socket.getSendBufferSize());
76      }
77  
78      @Test
79      public void testSoLinger() throws Exception {
80          assertEquals(-1, socket.getSoLinger());
81          socket.setSoLinger(10);
82          assertEquals(10, socket.getSoLinger());
83      }
84  
85      @Test
86      public void testDoubleCloseDoesNotThrow() throws IOException {
87          Socket socket = Socket.newSocketStream();
88          socket.close();
89          socket.close();
90      }
91  
92      @Test
93      public void testTrafficClass() throws IOException {
94          // IPTOS_THROUGHPUT
95          final int value = 0x08;
96          socket.setTrafficClass(value);
97          assertEquals(value, socket.getTrafficClass());
98      }
99  }
