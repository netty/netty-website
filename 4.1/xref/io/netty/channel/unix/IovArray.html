25  import static io.netty.channel.unix.Limits.IOV_MAX;
26  import static io.netty.channel.unix.Limits.SSIZE_MAX;
27  import static io.netty.util.internal.ObjectUtil.checkPositive;
28  import static java.lang.Math.min;

30  /**
31   * Represent an array of struct array and so can be passed directly over via JNI without the need to do any more
32   * array copies.
33   *
34   * The buffers are written out directly into direct memory to match the struct iov. See also {@code man writev}.
35   *
36   * <pre>
37   * struct iovec {
38   *   void  *iov_base;
39   *   size_t iov_len;
40   * };
41   * </pre>
42   *
43   * See also
44   * <a href="http://rkennke.wordpress.com/2007/07/30/efficient-jni-programming-iv-wrapping-native-data-objects/"
45   * >Efficient JNI programming IV: Wrapping native data objects</a>.
46   */
47  public final class IovArray implements MessageProcessor {
48  
49      /** The size of an address which should be 8 for 64 bits and 4 for 32 bits. */
50      private static final int ADDRESS_SIZE = Buffer.addressSize();
51  
52      /**
53       * The size of an {@code iovec} struct in bytes. This is calculated as we have 2 entries each of the size of the
54       * address.
55       */
56      private static final int IOV_SIZE = 2 * ADDRESS_SIZE;
57  
58      /**
59       * The needed memory to hold up to {@code IOV_MAX} iov entries, where {@code IOV_MAX} signified
60       * the maximum number of {@code iovec} structs that can be passed to {@code writev(...)}.
61       */
62      private static final int CAPACITY = IOV_MAX * IOV_SIZE;
63  
64      private final ByteBuffer memory;
65      private final long memoryAddress;
66      private int count;
67      private long size;
68      private long maxBytes = SSIZE_MAX;
69  
70      public IovArray() {
71          memory = Buffer.allocateDirectWithNativeOrder(CAPACITY);
72          memoryAddress = Buffer.memoryAddress(memory);
73      }
74  
75      public void clear() {
76          count = 0;
77          size = 0;
78      }
79  
80      /**
81       * Add a {@link ByteBuf} to this {@link IovArray}.
82       * @param buf The {@link ByteBuf} to add.
83       * @return {@code true} if the entire {@link ByteBuf} has been added to this {@link IovArray}. Note in the event
84       * that {@link ByteBuf} is a {@link CompositeByteBuf} {@code false} may be returned even if some of the components
85       * have been added.
86       */
87      public boolean add(ByteBuf buf) {
88          if (count == IOV_MAX) {
89              // No more room!
90              return false;
91          } else if (buf.nioBufferCount() == 1) {
92              final int len = buf.readableBytes();
93              if (len == 0) {
94                  return true;
95              }
96              if (buf.hasMemoryAddress()) {
97                  return add(buf.memoryAddress(), buf.readerIndex(), len);
98              } else {
99                  ByteBuffer nioBuffer = buf.internalNioBuffer(buf.readerIndex(), len);
100                 return add(Buffer.memoryAddress(nioBuffer), nioBuffer.position(), len);
101             }
102         } else {
103             ByteBuffer[] buffers = buf.nioBuffers();
104             for (ByteBuffer nioBuffer : buffers) {
105                 final int len = nioBuffer.remaining();
106                 if (len != 0 &&
107                     (!add(Buffer.memoryAddress(nioBuffer), nioBuffer.position(), len) || count == IOV_MAX)) {
108                     return false;
109                 }
110             }
111             return true;
112         }
113     }
114 
115     private boolean add(long addr, int offset, int len) {
116         assert addr != 0;
117 
118         // If there is at least 1 entry then we enforce the maximum bytes. We want to accept at least one entry so we
119         // will attempt to write some data and make progress.
120         if (maxBytes - len < size && count > 0) {
121             // If the size + len will overflow SSIZE_MAX we stop populate the IovArray. This is done as linux
122             //  not allow to write more bytes then SSIZE_MAX with one writev(...) call and so will
123             // return 'EINVAL', which will raise an IOException.
124             //
125             // See also:
126             // - http://linux.die.net/man/2/writev
127             return false;
128         }
129         final int baseOffset = idx(count);
130         final int lengthOffset = baseOffset + ADDRESS_SIZE;
131 
132         size += len;
133         ++count;
134 
135         if (ADDRESS_SIZE == 8) {
136             // 64bit
137             if (PlatformDependent.hasUnsafe()) {
138                 PlatformDependent.putLong(baseOffset + memoryAddress, addr + offset);
139                 PlatformDependent.putLong(lengthOffset + memoryAddress, len);
140             } else {
141                 memory.putLong(baseOffset, addr + offset);
142                 memory.putLong(lengthOffset, len);
143             }
144         } else {
145             assert ADDRESS_SIZE == 4;
146             if (PlatformDependent.hasUnsafe()) {
147                 PlatformDependent.putInt(baseOffset + memoryAddress, (int) addr + offset);
148                 PlatformDependent.putInt(lengthOffset + memoryAddress, len);
149             } else {
150                 memory.putInt(baseOffset, (int) addr + offset);
151                 memory.putInt(lengthOffset, len);
152             }
153         }
154         return true;
155     }
156 
157     /**
158      * Returns the number if iov entries.
159      */
160     public int count() {
161         return count;
162     }
163 
164     /**
165      * Returns the size in bytes
166      */
167     public long size() {
168         return size;
169     }
170 
171     /**
172      * Set the maximum amount of bytes that can be added to this {@link IovArray} via {@link #add(ByteBuf)}.
173      * <p>
174      * This will not impact the existing state of the {@link IovArray}, and only applies to subsequent calls to
175      * {@link #add(ByteBuf)}.
176      * <p>
177      * In order to ensure some progress is made at least one {@link ByteBuf} will be accepted even if it's size exceeds
178      * this value.
179      * @param maxBytes the maximum amount of bytes that can be added to this {@link IovArray} via {@link #add(ByteBuf)}.
180      */
181     public void maxBytes(long maxBytes) {
182         this.maxBytes = min(SSIZE_MAX, checkPositive(maxBytes, "maxBytes"));
183     }
184 
185     /**
186      * Get the maximum amount of bytes that can be added to this {@link IovArray} via {@link #add(ByteBuf)}.
187      * @return the maximum amount of bytes that can be added to this {@link IovArray} via {@link #add(ByteBuf)}.
188      */
189     public long maxBytes() {
190         return maxBytes;
191     }
192 
193     /**
194      * Returns the {@code memoryAddress} for the given {@code offset}.
195      */
196     public long memoryAddress(int offset) {
197         return memoryAddress + idx(offset);
198     }
199 
200     /**
201      * Release the {@link IovArray}. Once release further using of it may crash the JVM!
202      */
203     public void release() {
204         Buffer.free(memory);
205     }
206 
207     @Override
208     public boolean processMessage(Object msg) throws Exception {
209         return msg instanceof ByteBuf && add((ByteBuf) msg);
210     }
211 
212     private static int idx(int index) {
213         return IOV_SIZE * index;
214     }
