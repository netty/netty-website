118     @Override
119     protected boolean doWriteSingle(ChannelOutboundBuffer in, int writeSpinCount) throws Exception {
120         Object msg = in.current();
121         if (msg instanceof FileDescriptor && Native.sendFd(fd().intValue(), ((FileDescriptor) msg).intValue()) > 0) {
122             // File descriptor was written, so remove it.
123             in.remove();
124             return true;
125         }
126         return super.doWriteSingle(in, writeSpinCount);

145     private final class EpollDomainUnsafe extends EpollStreamUnsafe {
146         @Override
147         void epollInReady() {
148             switch (config().getReadMode()) {
149                 case BYTES:
150                     super.epollInReady();
151                     break;
152                 case FILE_DESCRIPTORS:
153                     epollInReadFd();
154                     break;
155                 default:
156                     throw new Error();
157             }
158         }
159 
160         private void epollInReadFd() {
161             if (fd().isInputShutdown()) {
162                 clearEpollIn0();
163                 return;
164             }
165             final ChannelConfig config = config();
166             final EpollRecvByteAllocatorHandle allocHandle = recvBufAllocHandle();
167             allocHandle.edgeTriggered(isFlagSet(Native.EPOLLET));
168 
169             final ChannelPipeline pipeline = pipeline();
170             allocHandle.reset(config);
171             epollInBefore();
172 
173             try {
174                 readLoop: do {
175                     // lastBytesRead represents the fd. We use lastBytesRead because it must be set so that the
176                     // EpollRecvByteAllocatorHandle knows if it should try to read again or not when autoRead is
177                     // enabled.
178                     allocHandle.lastBytesRead(Native.recvFd(fd().intValue()));
179                     switch(allocHandle.lastBytesRead()) {
180                     case 0:
181                         break readLoop;
182                     case -1:
183                         close(voidPromise());
184                         return;
185                     default:
186                         allocHandle.incMessagesRead(1);
187                         readPending = false;
188                         pipeline.fireChannelRead(new FileDescriptor(allocHandle.lastBytesRead()));
189                         break;
190                     }
191                 } while (allocHandle.continueReading());
192 
193                 allocHandle.readComplete();
194                 pipeline.fireChannelReadComplete();
195             } catch (Throwable t) {
196                 allocHandle.readComplete();
197                 pipeline.fireChannelReadComplete();
198                 pipeline.fireExceptionCaught(t);
199             } finally {
200                 epollInFinally(config);
201             }
202         }
203     }
