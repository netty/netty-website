33  /**
34   * Support <a href="https://linux.die.net//man/2/sendmmsg">sendmmsg(...)</a> on linux with GLIBC 2.14+
35   */
36  final class NativeDatagramPacketArray {
37  
38      // Use UIO_MAX_IOV as this is the maximum number we can write with one sendmmsg(...) call.
39      private final NativeDatagramPacket[] packets = new NativeDatagramPacket[UIO_MAX_IOV];
40  
41      // We share one IovArray for all NativeDatagramPackets to reduce memory overhead. This will allow us to write
42      // up to IOV_MAX iovec across all messages in one sendmmsg(...) call.
43      private final IovArray iovArray = new IovArray();
44  
45      // temporary array to copy the ipv4 part of ipv6-mapped-ipv4 addresses and then create a Inet4Address out of it.
46      private final byte[] ipv4Bytes = new byte[4];
47      private final MyMessageProcessor processor = new MyMessageProcessor();
48  
49      private int count;
50  
51      NativeDatagramPacketArray() {
52          for (int i = 0; i < packets.length; i++) {
53              packets[i] = new NativeDatagramPacket();
54          }
55      }
56  
57      boolean addWritable(ByteBuf buf, int index, int len) {
58          return add0(buf, index, len, 0, null);
59      }
60  
61      private boolean add0(ByteBuf buf, int index, int len, int segmentLen, InetSocketAddress recipient) {
62          if (count == packets.length) {
63              // We already filled up to UIO_MAX_IOV messages. This is the max allowed per
64              // recvmmsg(...) / sendmmsg(...) call, we will try again later.
65              return false;
66          }
67          if (len == 0) {
68              return true;
69          }
70          int offset = iovArray.count();
71          if (offset == Limits.IOV_MAX || !iovArray.add(buf, index, len)) {
72              // Not enough space to hold the whole content, we will try again later.
73              return false;
74          }
75          NativeDatagramPacket p = packets[count];
76          p.init(iovArray.memoryAddress(offset), iovArray.count() - offset, segmentLen, recipient);
77  
78          count++;
79          return true;
80      }
81  
82      void add(ChannelOutboundBuffer buffer, boolean connected, int maxMessagesPerWrite) throws Exception {
83          processor.connected = connected;
84          processor.maxMessagesPerWrite = maxMessagesPerWrite;
85          buffer.forEachFlushedMessage(processor);
86      }
87  
88      /**
89       * Returns the count
90       */
91      int count() {
92          return count;
93      }
94  
95      /**
96       * Returns an array with {@link #count()} {@link NativeDatagramPacket}s filled.
97       */
98      NativeDatagramPacket[] packets() {
99          return packets;
100     }
101 
102     void clear() {
103         this.count = 0;
104         this.iovArray.clear();
105     }
106 
107     void release() {
108         iovArray.release();
109     }
110 
111     private final class MyMessageProcessor implements MessageProcessor {
112         private boolean connected;
113         private int maxMessagesPerWrite;
114 
115         @Override
116         public boolean processMessage(Object msg) {
117             final boolean added;
118             if (msg instanceof DatagramPacket) {
119                 DatagramPacket packet = (DatagramPacket) msg;
120                 ByteBuf buf = packet.content();
121                 int segmentSize = 0;
122                 if (packet instanceof io.netty.channel.unix.SegmentedDatagramPacket) {
123                     int seg = ((io.netty.channel.unix.SegmentedDatagramPacket) packet).segmentSize();
124                     // We only need to tell the kernel that we want to use UDP_SEGMENT if there are multiple
125                     // segments in the packet.
126                     if (buf.readableBytes() > seg) {
127                         segmentSize = seg;
128                     }
129                 }
130                 added = add0(buf, buf.readerIndex(), buf.readableBytes(), segmentSize, packet.recipient());
131             } else if (msg instanceof ByteBuf && connected) {
132                 ByteBuf buf = (ByteBuf) msg;
133                 added = add0(buf, buf.readerIndex(), buf.readableBytes(), 0, null);
134             } else {
135                 added = false;
136             }
137             if (added) {
138                 maxMessagesPerWrite--;
139                 return maxMessagesPerWrite > 0;
140             }
141             return false;
142         }
143     }
144 
145     private static InetSocketAddress newAddress(byte[] addr, int addrLen, int port, int scopeId, byte[] ipv4Bytes)
146             throws UnknownHostException {
147         final InetAddress address;
148         if (addrLen == ipv4Bytes.length) {
149             System.arraycopy(addr, 0, ipv4Bytes, 0, addrLen);
150             address = InetAddress.getByAddress(ipv4Bytes);
151         } else {
152             address = Inet6Address.getByAddress(null, addr, scopeId);
153         }
154         return new InetSocketAddress(address, port);
155     }
156 
157     /**
158      * Used to pass needed data to JNI.
159      */
160     @SuppressWarnings("unused")
161     final class NativeDatagramPacket {
162 
163         // This is the actual struct iovec*
164         private long memoryAddress;
165         private int count;
166 
167         private final byte[] senderAddr = new byte[16];
168         private int senderAddrLen;
169         private int senderScopeId;
170         private int senderPort;
171 
172         private final byte[] recipientAddr = new byte[16];
173         private int recipientAddrLen;
174         private int recipientScopeId;
175         private int recipientPort;
176 
177         private int segmentSize;
178 
179         private void init(long memoryAddress, int count, int segmentSize, InetSocketAddress recipient) {
180             this.memoryAddress = memoryAddress;
181             this.count = count;
182             this.segmentSize = segmentSize;
183 
184             this.senderScopeId = 0;
185             this.senderPort = 0;
186             this.senderAddrLen = 0;
187 
188             if (recipient == null) {
189                 this.recipientScopeId = 0;
190                 this.recipientPort = 0;
191                 this.recipientAddrLen = 0;
192             } else {
193                 InetAddress address = recipient.getAddress();
194                 if (address instanceof Inet6Address) {
195                     System.arraycopy(address.getAddress(), 0, recipientAddr, 0, recipientAddr.length);
196                     recipientScopeId = ((Inet6Address) address).getScopeId();
197                 } else {
198                     copyIpv4MappedIpv6Address(address.getAddress(), recipientAddr);
199                     recipientScopeId = 0;
200                 }
201                 recipientAddrLen = recipientAddr.length;
202                 recipientPort = recipient.getPort();
203             }
204         }
205 
206         DatagramPacket newDatagramPacket(ByteBuf buffer, InetSocketAddress recipient) throws UnknownHostException {
207             InetSocketAddress sender = newAddress(senderAddr, senderAddrLen, senderPort, senderScopeId, ipv4Bytes);
208             if (recipientAddrLen != 0) {
209                 recipient = newAddress(recipientAddr, recipientAddrLen, recipientPort, recipientScopeId, ipv4Bytes);
210             }
211             buffer.writerIndex(count);
212 
213             // UDP_GRO
214             if (segmentSize > 0) {
215                 return new SegmentedDatagramPacket(buffer, segmentSize, recipient, sender);
216             }
217             return new DatagramPacket(buffer, recipient, sender);
218         }
219     }
