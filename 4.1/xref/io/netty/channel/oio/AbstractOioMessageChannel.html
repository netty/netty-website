16  package io.netty.channel.oio;
17  
18  import io.netty.channel.Channel;
19  import io.netty.channel.ChannelConfig;
20  import io.netty.channel.ChannelPipeline;
21  import io.netty.channel.RecvByteBufAllocator;
22  
23  import java.io.IOException;
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  /**
28   * Abstract base class for OIO which reads and writes objects from/to a Socket
29   */
30  public abstract class AbstractOioMessageChannel extends AbstractOioChannel {
31  
32      private final List<Object> readBuf = new ArrayList<Object>();
33  
34      protected AbstractOioMessageChannel(Channel parent) {
35          super(parent);
36      }
37  
38      @Override
39      protected void doRead() {
40          final ChannelConfig config = config();
41          if (!config.isAutoRead() && !isReadPending()) {
42              // ChannelConfig.setAutoRead(false) was called in the meantime
43              return;
44          }
45          // OIO reads are scheduled as a runnable object, the read is not pending as soon as the runnable is run.
46          setReadPending(false);
47  
48          final ChannelPipeline pipeline = pipeline();
49          final RecvByteBufAllocator.Handle allocHandle = unsafe().recvBufAllocHandle();
50          allocHandle.reset(config);
51  
52          boolean closed = false;
53          Throwable exception = null;
54          try {
55              do {
56                  // Perform a read.
57                  int localRead = doReadMessages(readBuf);
58                  if (localRead == 0) {
59                      break;
60                  }
61                  if (localRead < 0) {
62                      closed = true;
63                      break;
64                  }
65  
66                  allocHandle.incMessagesRead(localRead);
67              } while (allocHandle.continueReading());
68          } catch (Throwable t) {
69              exception = t;
70          }
71  
72          int size = readBuf.size();
73          for (int i = 0; i < size; i ++) {
74              pipeline.fireChannelRead(readBuf.get(i));
75          }
76          readBuf.clear();
77          allocHandle.readComplete();
78          pipeline.fireChannelReadComplete();
79  
80          if (exception != null) {
81              if (exception instanceof IOException) {
82                  closed = true;
83              }
84  
85              pipeline.fireExceptionCaught(exception);
86          }
87  
88          if (closed) {
89              if (isOpen()) {
90                  unsafe().close(unsafe().voidPromise());
91              }
92          } else if (allocHandle.lastBytesRead() == 0 && isActive()) {
93              // If the read amount was 0 and the channel is still active we need to trigger a new read()
94              // as otherwise we will never try to read again and the user will never know.
95              // Just call read() is ok here as it will be submitted to the EventLoop as a task and so we are
96              // able to process the rest of the tasks in the queue first.
97              //
98              // See https://github.com/netty/netty/issues/2404
99              read();
100         }
101     }
102 
103     /**
104      * Read messages into the given array and return the amount which was read.
105      */
106     protected abstract int doReadMessages(List<Object> msgs) throws Exception;
107 }
