135     NioEventLoop(NioEventLoopGroup parent, Executor executor, SelectorProvider selectorProvider,
136                  SelectStrategy strategy, RejectedExecutionHandler rejectedExecutionHandler,
137                  EventLoopTaskQueueFactory queueFactory) {
138         super(parent, executor, false, newTaskQueue(queueFactory), newTaskQueue(queueFactory),
139                 rejectedExecutionHandler);
140         this.provider = ObjectUtil.checkNotNull(selectorProvider, "selectorProvider");
141         this.selectStrategy = ObjectUtil.checkNotNull(strategy, "selectStrategy");
142         final SelectorTuple selectorTuple = openSelector();
143         this.selector = selectorTuple.selector;
144         this.unwrappedSelector = selectorTuple.unwrappedSelector;

170     private SelectorTuple openSelector() {
171         final Selector unwrappedSelector;
172         try {
173             unwrappedSelector = provider.openSelector();
174         } catch (IOException e) {
175             throw new ChannelException("failed to open a new selector", e);
176         }
177 
178         if (DISABLE_KEY_SET_OPTIMIZATION) {
179             return new SelectorTuple(unwrappedSelector);
180         }
181 
182         Object maybeSelectorImplClass = AccessController.doPrivileged(new PrivilegedAction<Object>() {
183             @Override
184             public Object run() {
185                 try {
186                     return Class.forName(
187                             "sun.nio.ch.SelectorImpl",
188                             false,
189                             PlatformDependent.getSystemClassLoader());
190                 } catch (Throwable cause) {
191                     return cause;
192                 }
193             }
194         });
195 
196         if (!(maybeSelectorImplClass instanceof Class) ||
197             // ensure the current selector implementation is what we can instrument.
198             !((Class<?>) maybeSelectorImplClass).isAssignableFrom(unwrappedSelector.getClass())) {
199             if (maybeSelectorImplClass instanceof Throwable) {
200                 Throwable t = (Throwable) maybeSelectorImplClass;
201                 logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, t);
202             }
203             return new SelectorTuple(unwrappedSelector);
204         }
205 
206         final Class<?> selectorImplClass = (Class<?>) maybeSelectorImplClass;
207         final SelectedSelectionKeySet selectedKeySet = new SelectedSelectionKeySet();
208 
209         Object maybeException = AccessController.doPrivileged(new PrivilegedAction<Object>() {
210             @Override
211             public Object run() {
212                 try {
213                     Field selectedKeysField = selectorImplClass.getDeclaredField("selectedKeys");
214                     Field publicSelectedKeysField = selectorImplClass.getDeclaredField("publicSelectedKeys");
215 
216                     if (PlatformDependent.javaVersion() >= 9 && PlatformDependent.hasUnsafe()) {
217                         // Let us try to use sun.misc.Unsafe to replace the SelectionKeySet.
218                         // This allows us to also do this in Java9+ without any extra flags.
219                         long selectedKeysFieldOffset = PlatformDependent.objectFieldOffset(selectedKeysField);
220                         long publicSelectedKeysFieldOffset =
221                                 PlatformDependent.objectFieldOffset(publicSelectedKeysField);
222 
223                         if (selectedKeysFieldOffset != -1 && publicSelectedKeysFieldOffset != -1) {
224                             PlatformDependent.putObject(
225                                     unwrappedSelector, selectedKeysFieldOffset, selectedKeySet);
226                             PlatformDependent.putObject(
227                                     unwrappedSelector, publicSelectedKeysFieldOffset, selectedKeySet);
228                             return null;
229                         }
230                         // We could not retrieve the offset, lets try reflection as last-resort.
231                     }
232 
233                     Throwable cause = ReflectionUtil.trySetAccessible(selectedKeysField, true);
234                     if (cause != null) {
235                         return cause;
236                     }
237                     cause = ReflectionUtil.trySetAccessible(publicSelectedKeysField, true);
238                     if (cause != null) {
239                         return cause;
240                     }
241 
242                     selectedKeysField.set(unwrappedSelector, selectedKeySet);
243                     publicSelectedKeysField.set(unwrappedSelector, selectedKeySet);
244                     return null;
245                 } catch (NoSuchFieldException e) {
246                     return e;
247                 } catch (IllegalAccessException e) {
248                     return e;
249                 }
250             }
251         });
252 
253         if (maybeException instanceof Exception) {
254             selectedKeys = null;
255             Exception e = (Exception) maybeException;
256             logger.trace("failed to instrument a special java.util.Set into: {}", unwrappedSelector, e);
257             return new SelectorTuple(unwrappedSelector);
258         }
259         selectedKeys = selectedKeySet;
260         logger.trace("instrumented a special java.util.Set into: {}", unwrappedSelector);
261         return new SelectorTuple(unwrappedSelector,
262                                  new SelectedSelectionKeySetSelector(unwrappedSelector, selectedKeySet));
263     }

283     /**
284      * Registers an arbitrary {@link SelectableChannel}, not necessarily created by Netty, to the {@link Selector}
285      * of this event loop.  Once the specified {@link SelectableChannel} is registered, the specified {@code task} will
286      * be executed by this event loop when the {@link SelectableChannel} is ready.
287      */
288     public void register(final SelectableChannel ch, final int interestOps, final NioTask<?> task) {
289         ObjectUtil.checkNotNull(ch, "ch");
290         if (interestOps == 0) {
291             throw new IllegalArgumentException("interestOps must be non-zero.");
292         }
293         if ((interestOps & ~ch.validOps()) != 0) {
294             throw new IllegalArgumentException(
295                     "invalid interestOps: " + interestOps + "(validOps: " + ch.validOps() + ')');
296         }
297         ObjectUtil.checkNotNull(task, "task");
298 
299         if (isShutdown()) {
300             throw new IllegalStateException("event loop shut down");
301         }
302 
303         if (inEventLoop()) {
304             register0(ch, interestOps, task);
305         } else {
306             try {
307                 // Offload to the EventLoop as otherwise java.nio.channels.spi.AbstractSelectableChannel.register
308                 // may block for a long time while trying to obtain an internal lock that may be hold while selecting.
309                 submit(new Runnable() {
310                     @Override
311                     public void run() {
312                         register0(ch, interestOps, task);
313                     }
314                 }).sync();
315             } catch (InterruptedException ignore) {
316                 // Even if interrupted we did schedule it so just mark the Thread as interrupted.
317                 Thread.currentThread().interrupt();
318             }
319         }
320     }
321 
322     private void register0(SelectableChannel ch, int interestOps, NioTask<?> task) {
323         try {
324             ch.register(unwrappedSelector, interestOps, task);
325         } catch (Exception e) {
326             throw new EventLoopException("failed to register a channel", e);
327         }
328     }

337     /**
338      * Sets the percentage of the desired amount of time spent for I/O in the event loop. Value range from 1-100.
339      * The default value is {@code 50}, which means the event loop will try to spend the same amount of time for I/O
340      * as for non-I/O tasks. The lower the number the more time can be spent on non-I/O tasks. If value set to
341      * {@code 100}, this feature will be disabled and event loop will not attempt to balance I/O and non-I/O tasks.
342      */
343     public void setIoRatio(int ioRatio) {
344         if (ioRatio <= 0 || ioRatio > 100) {
345             throw new IllegalArgumentException("ioRatio: " + ioRatio + " (expected: 0 < ioRatio <= 100)");
346         }
347         this.ioRatio = ioRatio;
348     }

350     /**
351      * Replaces the current {@link Selector} of this event loop with newly created {@link Selector}s to work
352      * around the infamous epoll 100% CPU bug.
353      */
354     public void rebuildSelector() {
355         if (!inEventLoop()) {
356             execute(new Runnable() {
357                 @Override
358                 public void run() {
359                     rebuildSelector0();
360                 }
361             });
362             return;
363         }
364         rebuildSelector0();
365     }

367     @Override
368     public int registeredChannels() {
369         return selector.keys().size() - cancelledKeys;

372     private void rebuildSelector0() {
373         final Selector oldSelector = selector;
374         final SelectorTuple newSelectorTuple;
375 
376         if (oldSelector == null) {
377             return;
378         }
379 
380         try {
381             newSelectorTuple = openSelector();
382         } catch (Exception e) {
383             logger.warn("Failed to create a new Selector.", e);
384             return;
385         }
386 
387         // Register all channels to the new Selector.
388         int nChannels = 0;
389         for (SelectionKey key: oldSelector.keys()) {
390             Object a = key.attachment();
391             try {
392                 if (!key.isValid() || key.channel().keyFor(newSelectorTuple.unwrappedSelector) != null) {
393                     continue;
394                 }
395 
396                 int interestOps = key.interestOps();
397                 key.cancel();
398                 SelectionKey newKey = key.channel().register(newSelectorTuple.unwrappedSelector, interestOps, a);
399                 if (a instanceof AbstractNioChannel) {
400                     // Update SelectionKey
401                     ((AbstractNioChannel) a).selectionKey = newKey;
402                 }
403                 nChannels ++;
404             } catch (Exception e) {
405                 logger.warn("Failed to re-register a Channel to the new Selector.", e);
406                 if (a instanceof AbstractNioChannel) {
407                     AbstractNioChannel ch = (AbstractNioChannel) a;
408                     ch.unsafe().close(ch.unsafe().voidPromise());
409                 } else {
410                     @SuppressWarnings("unchecked")
411                     NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
412                     invokeChannelUnregistered(task, key, e);
413                 }
414             }
415         }
416 
417         selector = newSelectorTuple.selector;
418         unwrappedSelector = newSelectorTuple.unwrappedSelector;
419 
420         try {
421             // time to close the old selector as everything else is registered to the new one
422             oldSelector.close();
423         } catch (Throwable t) {
424             if (logger.isWarnEnabled()) {
425                 logger.warn("Failed to close the old Selector.", t);
426             }
427         }
428 
429         if (logger.isInfoEnabled()) {
430             logger.info("Migrated " + nChannels + " channel(s) to the new Selector.");
431         }
432     }

434     @Override
435     protected void run() {
436         int selectCnt = 0;
437         for (;;) {
438             try {
439                 int strategy;
440                 try {
441                     strategy = selectStrategy.calculateStrategy(selectNowSupplier, hasTasks());
442                     switch (strategy) {
443                     case SelectStrategy.CONTINUE:
444                         continue;
445 
446                     case SelectStrategy.BUSY_WAIT:
447                         // fall-through to SELECT since the busy-wait is not supported with NIO
448 
449                     case SelectStrategy.SELECT:
450                         long curDeadlineNanos = nextScheduledTaskDeadlineNanos();
451                         if (curDeadlineNanos == -1L) {
452                             curDeadlineNanos = NONE; // nothing on the calendar
453                         }
454                         nextWakeupNanos.set(curDeadlineNanos);
455                         try {
456                             if (!hasTasks()) {
457                                 strategy = select(curDeadlineNanos);
458                             }
459                         } finally {
460                             // This update is just to help block unnecessary selector wakeups
461                             // so use of lazySet is ok (no race condition)
462                             nextWakeupNanos.lazySet(AWAKE);
463                         }
464                         // fall through
465                     default:
466                     }
467                 } catch (IOException e) {
468                     // If we receive an IOException here its because the Selector is messed up. Let's rebuild
469                     // the selector and retry. https://github.com/netty/netty/issues/8566
470                     rebuildSelector0();
471                     selectCnt = 0;
472                     handleLoopException(e);
473                     continue;
474                 }
475 
476                 selectCnt++;
477                 cancelledKeys = 0;
478                 needsToSelectAgain = false;
479                 final int ioRatio = this.ioRatio;
480                 boolean ranTasks;
481                 if (ioRatio == 100) {
482                     try {
483                         if (strategy > 0) {
484                             processSelectedKeys();
485                         }
486                     } finally {
487                         // Ensure we always run tasks.
488                         ranTasks = runAllTasks();
489                     }
490                 } else if (strategy > 0) {
491                     final long ioStartTime = System.nanoTime();
492                     try {
493                         processSelectedKeys();
494                     } finally {
495                         // Ensure we always run tasks.
496                         final long ioTime = System.nanoTime() - ioStartTime;
497                         ranTasks = runAllTasks(ioTime * (100 - ioRatio) / ioRatio);
498                     }
499                 } else {
500                     ranTasks = runAllTasks(0); // This will run the minimum number of tasks
501                 }
502 
503                 if (ranTasks || strategy > 0) {
504                     if (selectCnt > MIN_PREMATURE_SELECTOR_RETURNS && logger.isDebugEnabled()) {
505                         logger.debug("Selector.select() returned prematurely {} times in a row for Selector {}.",
506                                 selectCnt - 1, selector);
507                     }
508                     selectCnt = 0;
509                 } else if (unexpectedSelectorWakeup(selectCnt)) { // Unexpected wakeup (unusual case)
510                     selectCnt = 0;
511                 }
512             } catch (CancelledKeyException e) {
513                 // Harmless exception - log anyway
514                 if (logger.isDebugEnabled()) {
515                     logger.debug(CancelledKeyException.class.getSimpleName() + " raised by a Selector {} - JDK bug?",
516                             selector, e);
517                 }
518             } catch (Throwable t) {
519                 handleLoopException(t);
520             }
521             // Always handle shutdown even if the loop processing threw an exception.
522             try {
523                 if (isShuttingDown()) {
524                     closeAll();
525                     if (confirmShutdown()) {
526                         return;
527                     }
528                 }
529             } catch (Throwable t) {
530                 handleLoopException(t);
531             }
532         }
533     }

535     // returns true if selectCnt should be reset
536     private boolean unexpectedSelectorWakeup(int selectCnt) {
537         if (Thread.interrupted()) {
538             // Thread was interrupted so reset selected keys and break so we not run into a busy loop.
539             // As this is most likely a bug in the handler of the user or it's client library we will
540             // also log it.
541             //
542             // See https://github.com/netty/netty/issues/2426
543             if (logger.isDebugEnabled()) {
544                 logger.debug("Selector.select() returned prematurely because " +
545                         "Thread.currentThread().interrupt() was called. Use " +
546                         "NioEventLoop.shutdownGracefully() to shutdown the NioEventLoop.");
547             }
548             return true;
549         }
550         if (SELECTOR_AUTO_REBUILD_THRESHOLD > 0 &&
551                 selectCnt >= SELECTOR_AUTO_REBUILD_THRESHOLD) {
552             // The selector returned prematurely many times in a row.
553             // Rebuild the selector to work around the problem.
554             logger.warn("Selector.select() returned prematurely {} times in a row; rebuilding Selector {}.",
555                     selectCnt, selector);
556             rebuildSelector();
557             return true;
558         }
559         return false;
560     }

562     private static void handleLoopException(Throwable t) {
563         logger.warn("Unexpected exception in the selector loop.", t);
564 
565         // Prevent possible consecutive immediate failures that lead to
566         // excessive CPU consumption.
567         try {
568             Thread.sleep(1000);
569         } catch (InterruptedException e) {
570             // Ignore.
571         }
572     }

575         if (selectedKeys != null) {
576             processSelectedKeysOptimized();
577         } else {
578             processSelectedKeysPlain(selector.selectedKeys());
579         }

591     void cancel(SelectionKey key) {
592         key.cancel();
593         cancelledKeys ++;
594         if (cancelledKeys >= CLEANUP_INTERVAL) {
595             cancelledKeys = 0;
596             needsToSelectAgain = true;
597         }
598     }

600     private void processSelectedKeysPlain(Set<SelectionKey> selectedKeys) {
601         // check if the set is empty and if so just return to not create garbage by
602         // creating a new Iterator every time even if there is nothing to process.
603         // See https://github.com/netty/netty/issues/597
604         if (selectedKeys.isEmpty()) {
605             return;
606         }
607 
608         Iterator<SelectionKey> i = selectedKeys.iterator();
609         for (;;) {
610             final SelectionKey k = i.next();
611             final Object a = k.attachment();
612             i.remove();
613 
614             if (a instanceof AbstractNioChannel) {
615                 processSelectedKey(k, (AbstractNioChannel) a);
616             } else {
617                 @SuppressWarnings("unchecked")
618                 NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
619                 processSelectedKey(k, task);
620             }
621 
622             if (!i.hasNext()) {
623                 break;
624             }
625 
626             if (needsToSelectAgain) {
627                 selectAgain();
628                 selectedKeys = selector.selectedKeys();
629 
630                 // Create the iterator again to avoid ConcurrentModificationException
631                 if (selectedKeys.isEmpty()) {
632                     break;
633                 } else {
634                     i = selectedKeys.iterator();
635                 }
636             }
637         }
638     }

640     private void processSelectedKeysOptimized() {
641         for (int i = 0; i < selectedKeys.size; ++i) {
642             final SelectionKey k = selectedKeys.keys[i];
643             // null out entry in the array to allow to have it GC'ed once the Channel close
644             // See https://github.com/netty/netty/issues/2363
645             selectedKeys.keys[i] = null;
646 
647             final Object a = k.attachment();
648 
649             if (a instanceof AbstractNioChannel) {
650                 processSelectedKey(k, (AbstractNioChannel) a);
651             } else {
652                 @SuppressWarnings("unchecked")
653                 NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
654                 processSelectedKey(k, task);
655             }
656 
657             if (needsToSelectAgain) {
658                 // null out entries in the array to allow to have it GC'ed once the Channel close
659                 // See https://github.com/netty/netty/issues/2363
660                 selectedKeys.reset(i + 1);
661 
662                 selectAgain();
663                 i = -1;
664             }
665         }
666     }

668     private void processSelectedKey(SelectionKey k, AbstractNioChannel ch) {
669         final AbstractNioChannel.NioUnsafe unsafe = ch.unsafe();
670         if (!k.isValid()) {
671             final EventLoop eventLoop;
672             try {
673                 eventLoop = ch.eventLoop();
674             } catch (Throwable ignored) {
675                 // If the channel implementation throws an exception because there is no event loop, we ignore this
676                 // because we are only trying to determine if ch is registered to this event loop and thus has authority
677                 // to close ch.
678                 return;
679             }
680             // Only close ch if ch is still registered to this EventLoop. ch could have deregistered from the event loop
681             // and thus the SelectionKey could be cancelled as part of the deregistration process, but the channel is
682             // still healthy and should not be closed.
683             // See https://github.com/netty/netty/issues/5125
684             if (eventLoop == this) {
685                 // close the channel if the key is not valid anymore
686                 unsafe.close(unsafe.voidPromise());
687             }
688             return;
689         }
690 
691         try {
692             int readyOps = k.readyOps();
693             // We first need to call finishConnect() before try to trigger a read(...) or write(...) as otherwise
694             // the NIO JDK channel implementation may throw a NotYetConnectedException.
695             if ((readyOps & SelectionKey.OP_CONNECT) != 0) {
696                 // remove OP_CONNECT as otherwise Selector.select(..) will always return without blocking
697                 // See https://github.com/netty/netty/issues/924
698                 int ops = k.interestOps();
699                 ops &= ~SelectionKey.OP_CONNECT;
700                 k.interestOps(ops);
701 
702                 unsafe.finishConnect();
703             }
704 
705             // Process OP_WRITE first as we may be able to write some queued buffers and so free memory.
706             if ((readyOps & SelectionKey.OP_WRITE) != 0) {
707                 // Call forceFlush which will also take care of clear the OP_WRITE once there is nothing left to write
708                 ch.unsafe().forceFlush();
709             }
710 
711             // Also check for readOps of 0 to workaround possible JDK bug which may otherwise lead
712             // to a spin loop
713             if ((readyOps & (SelectionKey.OP_READ | SelectionKey.OP_ACCEPT)) != 0 || readyOps == 0) {
714                 unsafe.read();
715             }
716         } catch (CancelledKeyException ignored) {
717             unsafe.close(unsafe.voidPromise());
718         }
719     }

721     private static void processSelectedKey(SelectionKey k, NioTask<SelectableChannel> task) {
722         int state = 0;
723         try {
724             task.channelReady(k.channel(), k);
725             state = 1;
726         } catch (Exception e) {
727             k.cancel();
728             invokeChannelUnregistered(task, k, e);
729             state = 2;
730         } finally {
731             switch (state) {
732             case 0:
733                 k.cancel();
734                 invokeChannelUnregistered(task, k, null);
735                 break;
736             case 1:
737                 if (!k.isValid()) { // Cancelled by channelReady()
738                     invokeChannelUnregistered(task, k, null);
739                 }
740                 break;
741             default:
742                  break;
743             }
744         }
745     }

747     private void closeAll() {
748         selectAgain();
749         Set<SelectionKey> keys = selector.keys();
750         Collection<AbstractNioChannel> channels = new ArrayList<AbstractNioChannel>(keys.size());
751         for (SelectionKey k: keys) {
752             Object a = k.attachment();
753             if (a instanceof AbstractNioChannel) {
754                 channels.add((AbstractNioChannel) a);
755             } else {
756                 k.cancel();
757                 @SuppressWarnings("unchecked")
758                 NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
759                 invokeChannelUnregistered(task, k, null);
760             }
761         }
762 
763         for (AbstractNioChannel ch: channels) {
764             ch.unsafe().close(ch.unsafe().voidPromise());
765         }
766     }

803     private int select(long deadlineNanos) throws IOException {
804         if (deadlineNanos == NONE) {
805             return selector.select();
806         }
807         // Timeout will only be 0 if deadline is within 5 microsecs
808         long timeoutMillis = deadlineToDelayNanos(deadlineNanos + 995000L) / 1000000L;
809         return timeoutMillis <= 0 ? selector.selectNow() : selector.select(timeoutMillis);
