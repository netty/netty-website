162     private void setup(final ChannelHandler... handlers) {
163         ObjectUtil.checkNotNull(handlers, "handlers");
164         ChannelPipeline p = pipeline();
165         p.addLast(new ChannelInitializer<Channel>() {
166             @Override
167             protected void initChannel(Channel ch) throws Exception {
168                 ChannelPipeline pipeline = ch.pipeline();
169                 for (ChannelHandler h: handlers) {
170                     if (h == null) {
171                         break;
172                     }
173                     pipeline.addLast(h);
174                 }
175             }
176         });
177 
178         ChannelFuture future = loop.register(this);
179         assert future.isDone();
180     }

260      * Write messages to the inbound of this {@link Channel}.

262      * @param msgs the messages to be written
263      *
264      * @return {@code true} if the write operation did add something to the inbound buffer
265      */
266     public boolean writeInbound(Object... msgs) {
267         ensureOpen();
268         if (msgs.length == 0) {
269             return isNotEmpty(inboundMessages);
270         }
271 
272         ChannelPipeline p = pipeline();
273         for (Object m: msgs) {

314     private ChannelFuture flushInbound(boolean recordException, ChannelPromise promise) {
315       if (checkOpen(recordException)) {
316           pipeline().fireChannelReadComplete();
317           runPendingTasks();
318       }
319 
320       return checkException(promise);
321     }

323     /**
324      * Write messages to the outbound of this {@link Channel}.
325      *
326      * @param msgs              the messages to be written
327      * @return bufferReadable   returns {@code true} if the write operation did add something to the outbound buffer
328      */
329     public boolean writeOutbound(Object... msgs) {
330         ensureOpen();
331         if (msgs.length == 0) {
332             return isNotEmpty(outboundMessages);
333         }
334 
335         RecyclableArrayList futures = RecyclableArrayList.newInstance(msgs.length);
336         try {
337             for (Object m: msgs) {
338                 if (m == null) {
339                     break;
340                 }
341                 futures.add(write(m));
342             }
343 
344             flushOutbound0();
345 
346             int size = futures.size();
347             for (int i = 0; i < size; i++) {
348                 ChannelFuture future = (ChannelFuture) futures.get(i);
349                 if (future.isDone()) {
350                     recordException(future);
351                 } else {
352                     // The write may be delayed to run later by runPendingTasks()
353                     future.addListener(recordExceptionListener);
354                 }
355             }
356 
357             checkException();
358             return isNotEmpty(outboundMessages);
359         } finally {
360             futures.recycle();
361         }
362     }

400     private void flushOutbound0() {
401         // We need to call runPendingTasks first as a ChannelOutboundHandler may used eventloop.execute(...) to
402         // delay the write on the next eventloop run.
403         runPendingTasks();

427     /**
428      * Mark this {@link Channel} as finished. Any further try to write data to it will fail.
429      *
430      * @param releaseAll if {@code true} all pending message in the inbound and outbound buffer are released.
431      * @return bufferReadable returns {@code true} if any of the used buffers has something left to read
432      */
433     private boolean finish(boolean releaseAll) {
434         close();
435         try {
436             checkException();
437             return isNotEmpty(inboundMessages) || isNotEmpty(outboundMessages);
438         } finally {
439             if (releaseAll) {
440                 releaseAll(inboundMessages);
441                 releaseAll(outboundMessages);
442             }
443         }
444     }

462     private static boolean releaseAll(Queue<Object> queue) {
463         if (isNotEmpty(queue)) {
464             for (;;) {
465                 Object msg = queue.poll();
466                 if (msg == null) {
467                     break;
468                 }
469                 ReferenceCountUtil.release(msg);
470             }
471             return true;
472         }
473         return false;
474     }

476     private void finishPendingTasks(boolean cancel) {
477         runPendingTasks();
478         if (cancel) {
479             // Cancel all scheduled tasks that are left.
480             loop.cancelScheduledTasks();
481         }
482     }

494     @Override
495     public final ChannelFuture close(ChannelPromise promise) {
496         // We need to call runPendingTasks() before calling super.close() as there may be something in the queue
497         // that needs to be run before the actual close takes place.
498         runPendingTasks();
499         ChannelFuture future = super.close(promise);
500 
501         // Now finish everything else and cancel all scheduled tasks that were not ready set.
502         finishPendingTasks(true);
503         return future;
504     }

506     @Override
507     public final ChannelFuture disconnect(ChannelPromise promise) {
508         ChannelFuture future = super.disconnect(promise);
509         finishPendingTasks(!metadata.hasDisconnect());
510         return future;

539     /**
540      * Run all pending scheduled tasks in the {@link EventLoop} for this {@link Channel} and return the

559     private void recordException(Throwable cause) {
560         if (lastException == null) {
561             lastException = cause;
562         } else {
563             logger.warn(
564                     "More than one exception was raised. " +
565                             "Will report only the first one and log others.", cause);
566         }
567     }

569     /**
570      * Checks for the presence of an {@link Exception}.
571      */
572     private ChannelFuture checkException(ChannelPromise promise) {
573       Throwable t = lastException;
574       if (t != null) {
575         lastException = null;
576 
577         if (promise.isVoid()) {
578             PlatformDependent.throwException(t);
579         }
580 
581         return promise.setFailure(t);
582       }
583 
584       return promise.setSuccess();
585     }

594     /**
595      * Returns {@code true} if the {@link Channel} is open and records optionally
596      * an {@link Exception} if it isn't.

598     private boolean checkOpen(boolean recordException) {
599         if (!isOpen()) {

601               recordException(new ClosedChannelException());

604       }

667         for (;;) {
668             Object msg = in.current();

671             }

674             handleOutboundMessage(msg);
675             in.remove();
