209     private void setup(boolean register, final ChannelHandler... handlers) {
210         ObjectUtil.checkNotNull(handlers, "handlers");
211         ChannelPipeline p = pipeline();
212         p.addLast(new ChannelInitializer<Channel>() {
213             @Override
214             protected void initChannel(Channel ch) throws Exception {
215                 ChannelPipeline pipeline = ch.pipeline();
216                 for (ChannelHandler h: handlers) {
217                     if (h == null) {
218                         break;
219                     }
220                     pipeline.addLast(h);
221                 }
222             }
223         });
224         if (register) {
225             ChannelFuture future = loop.register(this);
226             assert future.isDone();
227         }
228     }

230     /**
231      * Register this {@code Channel} on its {@link EventLoop}.
232      */
233     public void register() throws Exception {
234         ChannelFuture future = loop.register(this);
235         assert future.isDone();
236         Throwable cause = future.cause();
237         if (cause != null) {
238             PlatformDependent.throwException(cause);
239         }
240     }

327     /**
328      * Write messages to the inbound of this {@link Channel}.
329      *
330      * @param msgs the messages to be written
331      *
332      * @return {@code true} if the write operation did add something to the inbound buffer
333      */
334     public boolean writeInbound(Object... msgs) {
335         ensureOpen();
336         if (msgs.length == 0) {
337             return isNotEmpty(inboundMessages);
338         }
339 
340         ChannelPipeline p = pipeline();
341         for (Object m: msgs) {
342             p.fireChannelRead(m);
343         }
344 
345         flushInbound(false, voidPromise());
346         return isNotEmpty(inboundMessages);
347     }

382     private ChannelFuture flushInbound(boolean recordException, ChannelPromise promise) {
383       if (checkOpen(recordException)) {
384           pipeline().fireChannelReadComplete();
385           runPendingTasks();
386       }
387 
388       return checkException(promise);
389     }

391     /**
392      * Write messages to the outbound of this {@link Channel}.
393      *
394      * @param msgs              the messages to be written
395      * @return bufferReadable   returns {@code true} if the write operation did add something to the outbound buffer
396      */
397     public boolean writeOutbound(Object... msgs) {
398         ensureOpen();
399         if (msgs.length == 0) {
400             return isNotEmpty(outboundMessages);
401         }
402 
403         RecyclableArrayList futures = RecyclableArrayList.newInstance(msgs.length);
404         try {
405             for (Object m: msgs) {
406                 if (m == null) {
407                     break;
408                 }
409                 futures.add(write(m));
410             }
411 
412             flushOutbound0();
413 
414             int size = futures.size();
415             for (int i = 0; i < size; i++) {
416                 ChannelFuture future = (ChannelFuture) futures.get(i);
417                 if (future.isDone()) {
418                     recordException(future);
419                 } else {
420                     // The write may be delayed to run later by runPendingTasks()
421                     future.addListener(recordExceptionListener);
422                 }
423             }
424 
425             checkException();
426             return isNotEmpty(outboundMessages);
427         } finally {
428             futures.recycle();
429         }
430     }

468     private void flushOutbound0() {
469         // We need to call runPendingTasks first as a ChannelOutboundHandler may used eventloop.execute(...) to
470         // delay the write on the next eventloop run.
471         runPendingTasks();
472 
473         flush();
474     }

486      * Mark this {@link Channel} as finished and release all pending message in the inbound and outbound buffer.

495     /**
496      * Mark this {@link Channel} as finished. Any further try to write data to it will fail.
497      *
498      * @param releaseAll if {@code true} all pending message in the inbound and outbound buffer are released.
499      * @return bufferReadable returns {@code true} if any of the used buffers has something left to read
500      */
501     private boolean finish(boolean releaseAll) {
502         close();
503         try {
504             checkException();
505             return isNotEmpty(inboundMessages) || isNotEmpty(outboundMessages);
506         } finally {
507             if (releaseAll) {
508                 releaseAll(inboundMessages);
509                 releaseAll(outboundMessages);
510             }
511         }
512     }

530     private static boolean releaseAll(Queue<Object> queue) {
531         if (isNotEmpty(queue)) {
532             for (;;) {
533                 Object msg = queue.poll();
534                 if (msg == null) {
535                     break;
536                 }
537                 ReferenceCountUtil.release(msg);
538             }
539             return true;
540         }
541         return false;
542     }

544     private void finishPendingTasks(boolean cancel) {
545         runPendingTasks();
546         if (cancel) {
547             // Cancel all scheduled tasks that are left.
548             loop.cancelScheduledTasks();
549         }
550     }

562     @Override
563     public final ChannelFuture close(ChannelPromise promise) {
564         // We need to call runPendingTasks() before calling super.close() as there may be something in the queue
565         // that needs to be run before the actual close takes place.
566         runPendingTasks();
567         ChannelFuture future = super.close(promise);
568 
569         // Now finish everything else and cancel all scheduled tasks that were not ready set.
570         finishPendingTasks(true);
571         return future;
572     }

574     @Override
575     public final ChannelFuture disconnect(ChannelPromise promise) {
576         ChannelFuture future = super.disconnect(promise);
577         finishPendingTasks(!metadata.hasDisconnect());
578         return future;
579     }

589     /**
590      * Run all tasks (which also includes scheduled tasks) that are pending in the {@link EventLoop}
591      * for this {@link Channel}
592      */
593     public void runPendingTasks() {
594         try {
595             loop.runTasks();
596         } catch (Exception e) {
597             recordException(e);
598         }
599 
600         try {
601             loop.runScheduledTasks();
602         } catch (Exception e) {
603             recordException(e);
604         }
605     }
606 
607     /**
608      * Run all pending scheduled tasks in the {@link EventLoop} for this {@link Channel} and return the
609      * {@code nanoseconds} when the next scheduled task is ready to run. If no other task was scheduled it will return
610      * {@code -1}.
611      */
612     public long runScheduledPendingTasks() {
613         try {
614             return loop.runScheduledTasks();
615         } catch (Exception e) {
616             recordException(e);
617             return loop.nextScheduledTask();
618         }
619     }

621     private void recordException(ChannelFuture future) {
622         if (!future.isSuccess()) {
623             recordException(future.cause());
624         }
625     }
626 
627     private void recordException(Throwable cause) {
628         if (lastException == null) {
629             lastException = cause;
630         } else {
631             logger.warn(
632                     "More than one exception was raised. " +
633                             "Will report only the first one and log others.", cause);
634         }
635     }
636 
637     /**
638      * Checks for the presence of an {@link Exception}.
639      */
640     private ChannelFuture checkException(ChannelPromise promise) {
641       Throwable t = lastException;
642       if (t != null) {
643         lastException = null;
644 
645         if (promise.isVoid()) {
646             PlatformDependent.throwException(t);
647         }
648 
649         return promise.setFailure(t);
650       }
651 
652       return promise.setSuccess();
653     }

662     /**
663      * Returns {@code true} if the {@link Channel} is open and records optionally
664      * an {@link Exception} if it isn't.
665      */
666     private boolean checkOpen(boolean recordException) {
667         if (!isOpen()) {
668           if (recordException) {
669               recordException(new ClosedChannelException());
670           }
671           return false;
672       }
673 
674       return true;
675     }

738     @Override
739     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
740         for (;;) {
741             Object msg = in.current();
742             if (msg == null) {
743                 break;
744             }
745 
746             ReferenceCountUtil.retain(msg);
747             handleOutboundMessage(msg);
748             in.remove();
749         }
750     }
