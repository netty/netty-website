16  package org.jboss.netty.handler.codec.spdy;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.buffer.ChannelBuffers;
20  import org.jboss.netty.channel.ChannelDownstreamHandler;
21  import org.jboss.netty.channel.ChannelEvent;
22  import org.jboss.netty.channel.ChannelHandlerContext;
23  import org.jboss.netty.channel.ChannelStateEvent;
24  import org.jboss.netty.channel.Channels;
25  import org.jboss.netty.channel.MessageEvent;
26  
27  import java.nio.ByteOrder;
28  import java.util.Set;
29  
30  import static org.jboss.netty.handler.codec.spdy.SpdyCodecUtil.*;
31  
32  /**
33   * Encodes a SPDY Frame into a {@link ChannelBuffer}.
34   */
35  public class SpdyFrameEncoder implements ChannelDownstreamHandler {
36  
37      private final int version;
38      private final SpdyHeaderBlockEncoder headerBlockEncoder;
39  
40      /**
41       * Creates a new instance with the specified {@code version} and the
42       * default {@code compressionLevel (6)}, {@code windowBits (15)},
43       * and {@code memLevel (8)}.
44       */
45      public SpdyFrameEncoder(SpdyVersion spdyVersion) {
46          this(spdyVersion, 6, 15, 8);
47      }
48  
49      /**
50       * Creates a new instance with the specified parameters.
51       */
52      public SpdyFrameEncoder(SpdyVersion spdyVersion, int compressionLevel, int windowBits, int memLevel) {
53          this(spdyVersion, SpdyHeaderBlockEncoder.newInstance(
54                  spdyVersion, compressionLevel, windowBits, memLevel));
55      }
56  
57      protected SpdyFrameEncoder(SpdyVersion spdyVersion, SpdyHeaderBlockEncoder headerBlockEncoder) {
58          if (spdyVersion == null) {
59              throw new NullPointerException("spdyVersion");
60          }
61          version = spdyVersion.getVersion();
62          this.headerBlockEncoder = headerBlockEncoder;
63      }
64  
65      public void handleDownstream(
66              final ChannelHandlerContext ctx, ChannelEvent evt) throws Exception {
67          if (evt instanceof ChannelStateEvent) {
68              ChannelStateEvent e = (ChannelStateEvent) evt;
69              switch (e.getState()) {
70              case OPEN:
71              case CONNECTED:
72              case BOUND:
73                  if (Boolean.FALSE.equals(e.getValue()) || e.getValue() == null) {
74                      synchronized (headerBlockEncoder) {
75                          headerBlockEncoder.end();
76                      }
77                  }
78              }
79          }
80  
81          if (!(evt instanceof MessageEvent)) {
82              ctx.sendDownstream(evt);
83              return;
84          }
85  
86          final MessageEvent e = (MessageEvent) evt;
87          Object msg = e.getMessage();
88  
89          if (msg instanceof SpdyDataFrame) {
90  
91              SpdyDataFrame spdyDataFrame = (SpdyDataFrame) msg;
92              ChannelBuffer data = spdyDataFrame.getData();
93              byte flags = spdyDataFrame.isLast() ? SPDY_DATA_FLAG_FIN : 0;
94              ChannelBuffer header = ChannelBuffers.buffer(
95                      ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE);
96              header.writeInt(spdyDataFrame.getStreamId() & 0x7FFFFFFF);
97              header.writeByte(flags);
98              header.writeMedium(data.readableBytes());
99              ChannelBuffer frame = ChannelBuffers.wrappedBuffer(header, data);
100             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
101             return;
102         }
103 
104         if (msg instanceof SpdySynStreamFrame) {
105 
106             synchronized (headerBlockEncoder) {
107                 SpdySynStreamFrame spdySynStreamFrame = (SpdySynStreamFrame) msg;
108                 ChannelBuffer data = headerBlockEncoder.encode(spdySynStreamFrame);
109                 byte flags = spdySynStreamFrame.isLast() ? SPDY_FLAG_FIN : 0;
110                 if (spdySynStreamFrame.isUnidirectional()) {
111                     flags |= SPDY_FLAG_UNIDIRECTIONAL;
112                 }
113                 int headerBlockLength = data.readableBytes();
114                 int length = 10 + headerBlockLength;
115                 ChannelBuffer frame = ChannelBuffers.buffer(
116                         ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 10);
117                 frame.writeShort(version | 0x8000);
118                 frame.writeShort(SPDY_SYN_STREAM_FRAME);
119                 frame.writeByte(flags);
120                 frame.writeMedium(length);
121                 frame.writeInt(spdySynStreamFrame.getStreamId());
122                 frame.writeInt(spdySynStreamFrame.getAssociatedToStreamId());
123                 frame.writeShort((spdySynStreamFrame.getPriority() & 0xFF) << 13);
124                 // Writes of compressed data must occur in order
125                 final ChannelBuffer buffer = ChannelBuffers.wrappedBuffer(frame, data);
126                 Channels.write(ctx, e.getFuture(), buffer, e.getRemoteAddress());
127             }
128             return;
129         }
130 
131         if (msg instanceof SpdySynReplyFrame) {
132 
133             synchronized (headerBlockEncoder) {
134                 SpdySynReplyFrame spdySynReplyFrame = (SpdySynReplyFrame) msg;
135                 ChannelBuffer data = headerBlockEncoder.encode(spdySynReplyFrame);
136                 byte flags = spdySynReplyFrame.isLast() ? SPDY_FLAG_FIN : 0;
137                 int headerBlockLength = data.readableBytes();
138                 int length = 4 + headerBlockLength;
139                 ChannelBuffer frame = ChannelBuffers.buffer(
140                         ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 4);
141                 frame.writeShort(version | 0x8000);
142                 frame.writeShort(SPDY_SYN_REPLY_FRAME);
143                 frame.writeByte(flags);
144                 frame.writeMedium(length);
145                 frame.writeInt(spdySynReplyFrame.getStreamId());
146                 // Writes of compressed data must occur in order
147                 final ChannelBuffer buffer = ChannelBuffers.wrappedBuffer(frame, data);
148                 Channels.write(ctx, e.getFuture(), buffer, e.getRemoteAddress());
149             }
150             return;
151         }
152 
153         if (msg instanceof SpdyRstStreamFrame) {
154 
155             SpdyRstStreamFrame spdyRstStreamFrame = (SpdyRstStreamFrame) msg;
156             ChannelBuffer frame = ChannelBuffers.buffer(
157                     ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 8);
158             frame.writeShort(version | 0x8000);
159             frame.writeShort(SPDY_RST_STREAM_FRAME);
160             frame.writeInt(8);
161             frame.writeInt(spdyRstStreamFrame.getStreamId());
162             frame.writeInt(spdyRstStreamFrame.getStatus().getCode());
163             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
164             return;
165         }
166 
167         if (msg instanceof SpdySettingsFrame) {
168 
169             SpdySettingsFrame spdySettingsFrame = (SpdySettingsFrame) msg;
170             byte flags = spdySettingsFrame.clearPreviouslyPersistedSettings() ?
171                 SPDY_SETTINGS_CLEAR : 0;
172             Set<Integer> IDs = spdySettingsFrame.getIds();
173             int numEntries = IDs.size();
174             int length = 4 + numEntries * 8;
175             ChannelBuffer frame = ChannelBuffers.buffer(
176                     ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + length);
177             frame.writeShort(version | 0x8000);
178             frame.writeShort(SPDY_SETTINGS_FRAME);
179             frame.writeByte(flags);
180             frame.writeMedium(length);
181             frame.writeInt(numEntries);
182             for (Integer id: IDs) {
183                 byte ID_flags = 0;
184                 if (spdySettingsFrame.isPersistValue(id)) {
185                     ID_flags |= SPDY_SETTINGS_PERSIST_VALUE;
186                 }
187                 if (spdySettingsFrame.isPersisted(id)) {
188                     ID_flags |= SPDY_SETTINGS_PERSISTED;
189                 }
190                 frame.writeByte(ID_flags);
191                 frame.writeMedium(id);
192                 frame.writeInt(spdySettingsFrame.getValue(id));
193             }
194             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
195             return;
196         }
197 
198         if (msg instanceof SpdyPingFrame) {
199 
200             SpdyPingFrame spdyPingFrame = (SpdyPingFrame) msg;
201             ChannelBuffer frame = ChannelBuffers.buffer(
202                     ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 4);
203             frame.writeShort(version | 0x8000);
204             frame.writeShort(SPDY_PING_FRAME);
205             frame.writeInt(4);
206             frame.writeInt(spdyPingFrame.getId());
207             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
208             return;
209         }
210 
211         if (msg instanceof SpdyGoAwayFrame) {
212 
213             SpdyGoAwayFrame spdyGoAwayFrame = (SpdyGoAwayFrame) msg;
214             ChannelBuffer frame = ChannelBuffers.buffer(
215                     ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 8);
216             frame.writeShort(version | 0x8000);
217             frame.writeShort(SPDY_GOAWAY_FRAME);
218             frame.writeInt(8);
219             frame.writeInt(spdyGoAwayFrame.getLastGoodStreamId());
220             frame.writeInt(spdyGoAwayFrame.getStatus().getCode());
221             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
222             return;
223         }
224 
225         if (msg instanceof SpdyHeadersFrame) {
226 
227             synchronized (headerBlockEncoder) {
228                 SpdyHeadersFrame spdyHeadersFrame = (SpdyHeadersFrame) msg;
229                 ChannelBuffer data = headerBlockEncoder.encode(spdyHeadersFrame);
230                 byte flags = spdyHeadersFrame.isLast() ? SPDY_FLAG_FIN : 0;
231                 int headerBlockLength = data.readableBytes();
232                 int length = 4 + headerBlockLength;
233                 ChannelBuffer frame = ChannelBuffers.buffer(
234                         ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 4);
235                 frame.writeShort(version | 0x8000);
236                 frame.writeShort(SPDY_HEADERS_FRAME);
237                 frame.writeByte(flags);
238                 frame.writeMedium(length);
239                 frame.writeInt(spdyHeadersFrame.getStreamId());
240                 // Writes of compressed data must occur in order
241                 final ChannelBuffer buffer = ChannelBuffers.wrappedBuffer(frame, data);
242                 Channels.write(ctx, e.getFuture(), buffer, e.getRemoteAddress());
243             }
244             return;
245         }
246 
247         if (msg instanceof SpdyWindowUpdateFrame) {
248             SpdyWindowUpdateFrame spdyWindowUpdateFrame = (SpdyWindowUpdateFrame) msg;
249             ChannelBuffer frame = ChannelBuffers.buffer(
250                     ByteOrder.BIG_ENDIAN, SPDY_HEADER_SIZE + 8);
251             frame.writeShort(version | 0x8000);
252             frame.writeShort(SPDY_WINDOW_UPDATE_FRAME);
253             frame.writeInt(8);
254             frame.writeInt(spdyWindowUpdateFrame.getStreamId());
255             frame.writeInt(spdyWindowUpdateFrame.getDeltaWindowSize());
256             Channels.write(ctx, e.getFuture(), frame, e.getRemoteAddress());
257             return;
258         }
259 
260         // Unknown message type
261         ctx.sendDownstream(evt);
262     }
